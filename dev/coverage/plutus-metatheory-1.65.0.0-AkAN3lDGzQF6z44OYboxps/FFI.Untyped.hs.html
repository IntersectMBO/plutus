   16 -- Untyped (Raw) syntax
   17 
   18 data UTerm
   19   = UVar Integer
   20   | ULambda UTerm
   21   | UApp UTerm UTerm
   22   | UCon (Some (ValueOf DefaultUni))
   23   | UError
   24   | UBuiltin DefaultFun
   25   | UDelay UTerm
   26   | UForce UTerm
   27   | UConstr Integer [UTerm]
   28   | UCase UTerm [UTerm]
   29   deriving stock (Eq, Show, Generic)
   30   deriving anyclass (NFData)
   31 
   32 unIndex :: Index -> Integer
   33 unIndex (Index n) = toInteger n
   34 
   35 convP :: Program NamedDeBruijn DefaultUni DefaultFun a -> UTerm
   36 convP (Program _ _ t) = conv t
   37 
   38 conv :: Term NamedDeBruijn DefaultUni DefaultFun a -> UTerm
   39 conv (Var _ x) = UVar (unIndex (ndbnIndex x) - 1)
   40 conv (LamAbs _ _ t) = ULambda (conv t)
   41 conv (Apply _ t u) = UApp (conv t) (conv u)
   42 conv (Builtin _ b) = UBuiltin b
   43 conv (Constant _ c) = UCon c
   44 conv (Error _) = UError
   45 conv (Delay _ t) = UDelay (conv t)
   46 conv (Force _ t) = UForce (conv t)
   47 conv (Constr _ i es) = UConstr (toInteger i) (toList (fmap conv es))
   48 conv (Case _ arg cs) = UCase (conv arg) (toList (fmap conv cs))
   49 
   50 tmname :: Int -> String
   51 tmname i = 'x' : show i
   52 
   53 uconv :: Int -> UTerm -> Term NamedDeBruijn DefaultUni DefaultFun ()
   54 uconv i (UVar x) =
   55   Var
   56     ()
   57     ( NamedDeBruijn
   58         (T.pack (tmname (i - 1 - fromInteger x)))
   59         -- PLC's debruijn starts counting from 1, while in the metatheory it starts from 0.
   60         (Index (fromInteger x + 1))
   61     )
   62 uconv i (ULambda t) =
   63   LamAbs
   64     ()
   65     (NamedDeBruijn (T.pack (tmname i)) deBruijnInitIndex)
   66     (uconv (i + 1) t)
   67 uconv i (UApp t u) = Apply () (uconv i t) (uconv i u)
   68 uconv _ (UCon c) = Constant () c
   69 uconv _ UError = Error ()
   70 uconv _ (UBuiltin b) = Builtin () b
   71 uconv i (UDelay t) = Delay () (uconv i t)
   72 uconv i (UForce t) = Force () (uconv i t)
   73 uconv i (UConstr j xs) = Constr () (fromInteger j) (fmap (uconv i) xs)
   74 uconv i (UCase t xs) = Case () (uconv i t) (fromList (fmap (uconv i) xs))
