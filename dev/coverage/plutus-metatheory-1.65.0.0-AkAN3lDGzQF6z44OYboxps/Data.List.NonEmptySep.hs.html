   13 -- | A non-empty list of `a`, separated by values of type `sep`
   14 data NonEmptySep sep a
   15   = Cons a sep (NonEmptySep sep a)
   16   | Singleton a
   17   deriving (Show, Generic)

   20 instance Functor (NonEmptySep sep) where
   21   fmap f (Singleton x) = Singleton (f x)
   22   fmap f (Cons x y xs) = Cons (f x) y (fmap f xs)
