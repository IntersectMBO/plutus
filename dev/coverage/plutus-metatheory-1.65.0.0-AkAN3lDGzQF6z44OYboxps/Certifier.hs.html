   45 data CertifierError

   51 data CertifierOutput

   59 prettyCertifierError :: CertifierError -> String
   60 prettyCertifierError (InvalidCertificate certDir report) =
   61   "\n\nInvalid certificate: "
   62     <> certDir
   63     <> "\nThe compilation was not successfully certified. \
   64        \Please open a bug report at https://www.github.com/IntersectMBO/plutus \
   65        \and attach the faulty certificate.\n\
   66        \Certifier report:\n"
   67     <> report
   68     <> "\n"
   69 prettyCertifierError InvalidCompilerOutput =
   70   "\n\nInvalid compiler output: \
   71   \\nThe certifier was not able to process the trace produced by the compiler. \
   72   \Please open a bug report at https://www.github.com/IntersectMBO/plutus \
   73   \containing a minimal program that when compiled reproduces the issue.\n"
   74 prettyCertifierError (ValidationError name) =
   75   "\n\nInvalid certificate name: \
   76   \\nThe certificate name "
   77     <> name
   78     <> " is invalid. \
   79        \Please use only alphanumeric characters, underscores and dashes. \
   80        \The first character must be a letter.\n"

   87 -- | Run the Agda certifier on the simplification trace, if requested
   88 mkCertifier
   89   :: OptimizerTrace UPLC.Name UPLC.DefaultUni UPLC.DefaultFun a
   90   -- ^ The trace produced by the simplification process
   91   -> CertName
   92   -- ^ The name of the certificate to be produced
   93   -> CertifierOutput
   94   -> [ ( Maybe
   95            (CekEvaluationException UPLC.NamedDeBruijn UPLC.DefaultUni UPLC.DefaultFun)
   96        , ExBudget
   97        )
   98      ]
   99   -> Certifier Bool
  100 mkCertifier simplTrace certName certOutput costs = do
  101   let rawAgdaTrace = mkFfiOptimizerTrace simplTrace
  102       costs' :: [EvalResult]
  103       costs' = uncurry toEvalResult <$> reverse costs
  104   case runCertifierMain rawAgdaTrace costs' of
  105     Just (passed, report) -> do
  106       case certOutput of
  107         BasicOutput -> pure ()
  108         ReportOutput file -> liftIO $ T.writeFile file report
  109         ProjectOutput certDir -> do
  110           let cert = mkAgdaCertificateProject $ mkCertificate certName rawAgdaTrace
  111           writeCertificateProject certDir cert
  112           unless passed . throwError $ InvalidCertificate certDir (Text.unpack report)
  113       pure passed
  114     Nothing -> throwError InvalidCompilerOutput

  116 data Certificate = Certificate
  117   { certName :: String
  118   , certTrace :: Trace UTerm
  119   }
  120 
  121 mkCertificate :: String -> Trace UTerm -> Certificate
  122 mkCertificate certName rawTrace =
  123   let trace' = dedup rawTrace
  124    in Certificate
  125         { certName
  126         , certTrace = trace'
  127         }
  128   where
  129     dedup :: Trace UTerm -> Trace UTerm
  130     dedup (Singleton x) = Singleton x
  131     dedup (Cons x sep xs)
  132       -- If subsequent ASTs are equal, drop the pass
  133       | x == NES.head xs = dedup xs
  134       | otherwise = Cons x sep (dedup xs)

  136 {-| Module name with string components, e.g.
  137   "Data" :| ["List", "NonEmpty"] -}
  138 type ModuleName = NonEmpty String

  140 -- | Convert module name to an Agda identifier
  141 toIdent :: ModuleName -> String
  142 toIdent = fold . NE.intersperse "."
  143 
  144 -- | Convert module name to .agda filepath relative to the source directory
  145 toFilePath :: ModuleName -> FilePath
  146 toFilePath m = foldr (</>) "" m ++ ".agda"
  147 
  148 -- | Drops the last component of the module to obtain its directory
  149 moduleDir :: ModuleName -> String
  150 moduleDir m = foldr (</>) "" (NE.init m)

  152 enclose :: String -> String -> String -> String
  153 enclose l r x = l ++ x ++ r

  155 data ImportHow = Open | Closed
  156 
  157 type Import = (ImportHow, ModuleName, [String])
  158 
  159 renderImport :: Import -> String
  160 renderImport (how, name, idents) =
  161   renderHow ++ "import " ++ toIdent name ++ renderUsing
  162   where
  163     renderHow = case how of
  164       Open -> "open "
  165       Closed -> ""
  166     renderUsing
  167       | null idents = ""
  168       | otherwise = " using (" ++ intercalate ", " idents ++ ")"

  170 {-| Module name in agda, for a given ast number and module name. Each list element is a module segment that is to be
  171 separated by "." (agda identifier) or by "/" (files) -}
  172 passModName :: String -> Int -> ModuleName
  173 passModName name n =
  174   ("Pass" <> printf "%03d" n) :| [name]
  175 
  176 astModName :: Int -> ModuleName
  177 astModName n = passModName "AST" n
  178 
  179 proofModName :: Int -> ModuleName
  180 proofModName n = passModName "Proof" n
  181 
  182 proofFile :: Int -> UTerm -> Maybe (OptStage, Hints) -> (ModuleName, String)
  183 proofFile n _ Nothing = (proofModName n, [])
  184 proofFile n _ (Just (pass, _)) =
  185   ( modName
  186   , unlines $
  187       [ "module " <> toIdent modName <> " where"
  188       , ""
  189       ]
  190         ++ map renderImport imports
  191         ++ [ ""
  192            , "related : " ++ proofSig
  193            , "related = " ++ proofExpr
  194            ]
  195   )
  196   where
  197     modName = proofModName n
  198 
  199     preMod = astModName n
  200     postMod = astModName (n + 1)
  201 
  202     preTerm = toIdent preMod ++ ".ast"
  203     postTerm = toIdent postMod ++ ".ast"
  204     hints = toIdent preMod ++ ".hints"
  205 
  206     proofSig :: String
  207     proofSig =
  208       unwords
  209         ["RelationOf", renderAgdaUnparse pass, preTerm, postTerm]
  210 
  211     proofExpr :: String
  212     proofExpr =
  213       unwords
  214         [ "to-witness-T"
  215         , enclose "(" ")" . unwords $
  216             [ "certifyPass"
  217             , renderAgdaUnparse pass
  218             , hints
  219             , preTerm
  220             , postTerm
  221             ]
  222         , "tt"
  223         ]
  224 
  225     imports :: [(ImportHow, ModuleName, [String])]
  226     imports =
  227       [ (Closed, preMod, [])
  228       , (Closed, postMod, [])
  229       , (Open, "Data" :| ["Unit"], ["tt"])
  230       , (Open, "VerifiedCompilation" :| [], [])
  231       , (Open, "VerifiedCompilation" :| "Trace" : [], [])
  232       , (Open, "VerifiedCompilation" :| "Certificate" : [], [])
  233       , (Open, "Utils" :| [], [])
  234       ]
  235 
  236 mkAgdaAstFile :: Int -> UTerm -> Maybe (OptStage, Hints) -> (ModuleName, String)
  237 mkAgdaAstFile n pre mHints =
  238   ( modName
  239   , unlines $
  240       [ "module " <> toIdent modName <> " where"
  241       , ""
  242       , "open import Data.Unit"
  243       , "open import Data.Nat"
  244       , "open import Data.Integer"
  245       , "open import Data.Maybe"
  246       , "open import Data.Bool.Base using (Bool; false; true)"
  247       , "open import Data.List using (List; _∷_; [])"
  248       , ""
  249       , "open import RawU"
  250       , "open import Builtin"
  251       , "open import Utils"
  252       , ""
  253       , "open import Untyped using (_⊢)"
  254       , "open import VerifiedCompilation using (checkScope)"
  255       , "open import VerifiedCompilation.Trace"
  256       , ""
  257       , "raw : Untyped"
  258       , "raw = " <> renderAgdaUnparse pre
  259       , ""
  260       , "ast : 0 ⊢"
  261       , "ast = to-witness-T (checkScope raw) tt"
  262       ]
  263         ++ case mHints of
  264           Just (_, hints) ->
  265             [ ""
  266             , "hints : Hints"
  267             , "hints = " <> renderAgdaUnparse hints
  268             ]
  269           Nothing -> []
  270   )
  271   where
  272     modName = astModName n

  282 mkCertificateFile :: Certificate -> String
  283 mkCertificateFile Certificate {certName, certTrace} =
  284   unlines $
  285     [ "module " <> certName <> " where"
  286     , ""
  287     , "open import VerifiedCompilation.Trace"
  288     , "open import Untyped using (_⊢)"
  289     , "open import Utils hiding (List; _>>=_)"
  290     , "open import VerifiedCompilation"
  291     , "open import Data.Unit"
  292     , ""
  293     ]
  294       ++ map renderImport (astImports ++ proofImports)
  295       ++ [ ""
  296          , "trace : Trace (0 ⊢)"
  297          , "trace ="
  298          ]
  299       ++ map ("  " ++) (printTrace traceExpr)
  300       ++ [ ""
  301          , "certificate : Certificate trace"
  302          , "certificate = "
  303          , renderProof proofExpr
  304          ]
  305   where
  306     astImports :: [Import]
  307     astImports = map (\n -> (Closed, astModName n, [])) [0 .. length certTrace - 1]
  308 
  309     proofImports :: [Import]
  310     proofImports = map (\n -> (Closed, proofModName n, [])) [0 .. length certTrace - 2]
  311 
  312     -- Replace hints and ASTs in the trace by agda variables that point to the right definition
  313     traceExpr :: NonEmptySep (OptStage, AgdaVar) AgdaVar
  314     traceExpr = go 0 certTrace
  315       where
  316         go n (Singleton _) = Singleton (qualify (astModName n) (AgdaVar "ast"))
  317         go n (Cons _ (stage, _) xs) =
  318           Cons
  319             (qualify (astModName n) (AgdaVar "ast"))
  320             (stage, qualify (astModName n) (AgdaVar "hints"))
  321             (go (n + 1) xs)
  322 
  323     -- Non-empty list of variables that point to the proofs per pass
  324     proofExpr :: NonEmpty AgdaVar
  325     proofExpr = go 0 certTrace
  326       where
  327         go _ (Singleton _) = AgdaVar "tt" :| []
  328         go n (Cons _ _ xs) = cons (qualify (proofModName n) (AgdaVar "related")) (go (n + 1) xs)
  329 
  330 -- Print the imported proofs as a big product
  331 renderProof :: NonEmpty AgdaVar -> String
  332 renderProof (AgdaVar x :| xs) =
  333   unlines $
  334     [("  ( " ++ x)]
  335       ++ map (\(AgdaVar p) -> "  , " ++ p) xs
  336       ++ [("  )")]
  337 
  338 -- Ad-hoc pretty printing so it can be printed over multiple lines
  339 printTrace :: NonEmptySep (OptStage, AgdaVar) AgdaVar -> [String]
  340 printTrace (Singleton x) = ["singleton" ++ " " ++ renderAgdaUnparse x]
  341 printTrace (Cons x (stage, y) xs) =
  342   [ renderAgdaUnparse x
  343   , "  ∷[ " ++ renderAgdaUnparse stage ++ " , " ++ renderAgdaUnparse y ++ " ]"
  344   ]
  345     ++ printTrace xs

  347 data AgdaCertificateProject = AgdaCertificateProject
  348   { mainModule :: (FilePath, String)
  349   , astModules :: [(ModuleName, String)]
  350   , proofModNames :: [(ModuleName, String)]
  351   , agdalib :: (FilePath, String)
  352   }

  354 {-| For each term in the trace, create module contents using the given
  355 function, which is also passed the index in the trace. -}
  356 traceToFiles
  357   :: (Int -> UTerm -> Maybe (OptStage, Hints) -> (ModuleName, String))
  358   -> Trace UTerm
  359   -> [(ModuleName, String)]
  360 traceToFiles f t = go 0 t
  361   where
  362     go n (Singleton x) = [f n x Nothing]
  363     go n (Cons x (o, h) xs) = f n x (Just (o, h)) : go (n + 1) xs

  365 mkAgdaLib :: String -> (FilePath, String)
  366 mkAgdaLib name =
  367   let contents =
  368         "name: "
  369           <> name
  370           <> "\ndepend:\
  371              \\n  standard-library-2.3\
  372              \\n  plutus-metatheory\
  373              \\ninclude: src\
  374              \\nflags: --polarity"
  375    in (name <> ".agda-lib", contents)
  376 
  377 mkAgdaCertificateProject
  378   :: Certificate
  379   -> AgdaCertificateProject
  380 mkAgdaCertificateProject cert =
  381   let name = certName cert
  382       mainModule = mkCertificateFile cert
  383       astModules = traceToFiles mkAgdaAstFile (certTrace cert)
  384       proofModNames = traceToFiles proofFile (certTrace cert)
  385       agdalib = mkAgdaLib name
  386    in AgdaCertificateProject
  387         { mainModule = (certName cert <> ".agda", mainModule)
  388         , astModules
  389         , proofModNames
  390         , agdalib
  391         }

  393 writeCertificateProject
  394   :: CertDir
  395   -> AgdaCertificateProject
  396   -> Certifier ()
  397 writeCertificateProject
  398   certDir
  399   AgdaCertificateProject
  400     { mainModule
  401     , astModules
  402     , proofModNames
  403     , agdalib
  404     } =
  405     liftIO $ do
  406       let (mainModulePath, mainModuleContents) = mainModule
  407           (agdalibPath, agdalibContents) = agdalib
  408           certName = takeBaseName mainModulePath
  409       createDirectoryIfMissing True certDir
  410       createDirectoryIfMissing True (certDir </> "src")
  411 
  412       -- directory per AST in trace
  413       for_ (astModules ++ proofModNames) $ \(modName, contents) -> do
  414         createDirectoryIfMissing True (certDir </> "src" </> moduleDir modName)
  415         writeFile (certDir </> "src" </> toFilePath modName) contents
  416 
  417       writeFile (certDir </> "src" </> mainModulePath) mainModuleContents
  418       writeFile (certDir </> agdalibPath) agdalibContents
  419       let readmeTemplate = $(embedStringFile "file-embed/certificate-README.md")
  420       writeFile (certDir </> "README.md") (replace "{{NAME}}" certName readmeTemplate)
