   28 usToHyphen :: String -> String
   29 usToHyphen = map (\c -> if c == '_' then '-' else c)
   30 
   31 -- | A class for types that can be unparsed to Agda code.
   32 class AgdaUnparse a where
   33   agdaUnparse :: a -> Doc ann
   34 
   35 -- | Render an 'AgdaUnparse' value to a 'String'.
   36 renderAgdaUnparse :: AgdaUnparse a => a -> String
   37 renderAgdaUnparse = renderString . layoutPretty (LayoutOptions Unbounded) . agdaUnparse
   38 
   39 instance (AgdaUnparse a, AgdaUnparse b) => AgdaUnparse (NonEmptySep a b) where
   40   agdaUnparse (Singleton x) = parens ("singleton" <+> agdaUnparse x)
   41   agdaUnparse (Cons x y xs) =
   42     parens $
   43       "cons"
   44         <+> agdaUnparse x
   45         <+> agdaUnparse y
   46         <+> agdaUnparse xs
   47 
   48 instance AgdaUnparse AgdaFFI.UTerm where
   49   agdaUnparse =
   50     \case
   51       AgdaFFI.UVar n -> parens ("UVar" <+> agdaUnparse (fromInteger n :: Natural))
   52       AgdaFFI.ULambda term -> parens ("ULambda" <+> agdaUnparse term)
   53       AgdaFFI.UApp t u -> parens ("UApp" <+> agdaUnparse t <+> agdaUnparse u)
   54       AgdaFFI.UCon someValue -> parens ("UCon" <+> agdaUnparse someValue)
   55       AgdaFFI.UError -> "UError"
   56       AgdaFFI.UBuiltin fun -> parens ("UBuiltin" <+> agdaUnparse fun)
   57       AgdaFFI.UDelay term -> parens ("UDelay" <+> agdaUnparse term)
   58       AgdaFFI.UForce term -> parens ("UForce" <+> agdaUnparse term)
   59       AgdaFFI.UConstr i terms ->
   60         parens ("UConstr" <+> agdaUnparse (fromInteger i :: Natural) <+> agdaUnparse terms)
   61       AgdaFFI.UCase term cases -> parens ("UCase" <+> agdaUnparse term <+> agdaUnparse cases)
   62 
   63 instance AgdaUnparse UPLC.DefaultFun where
   64   agdaUnparse = pretty . usToHyphen . lowerInitialChar . show

   66 instance AgdaUnparse CertifiedOptStage where

   81 instance AgdaUnparse Hints.Hints where
   82   agdaUnparse = \case
   83     Hints.NoHints -> "none"
   84     Hints.Inline x -> parens ("inline" <+> agdaUnparse x)
   85 
   86 instance AgdaUnparse Hints.Inline where
   87   agdaUnparse = \case
   88     Hints.InlVar -> "var"
   89     Hints.InlLam t -> parens ("ƛ" <+> agdaUnparse t)
   90     Hints.InlApply f x -> parens (agdaUnparse f <+> "·" <+> agdaUnparse x)
   91     Hints.InlForce t -> parens ("force" <+> agdaUnparse t)
   92     Hints.InlDelay t -> parens ("delay" <+> agdaUnparse t)
   93     Hints.InlCon -> "con"
   94     Hints.InlBuiltin -> "builtin"
   95     Hints.InlError -> "error"
   96     Hints.InlConstr ts -> parens ("constr" <+> agdaUnparse ts)
   97     Hints.InlCase t ts -> parens ("case" <+> agdaUnparse t <+> agdaUnparse ts)
   98     Hints.InlExpand t -> parens ("expand" <+> agdaUnparse t)
   99     Hints.InlDrop t -> parens (agdaUnparse t <+> "·↓")

  101 instance AgdaUnparse Natural where
  102   agdaUnparse = viaShow
  103 
  104 instance AgdaUnparse Integer where
  105   agdaUnparse x
  106     | x < 0 = parens ("ℤ.negsuc" <+> viaShow (abs x - 1))
  107     | otherwise = parens ("ℤ.pos" <+> viaShow x)
  108 
  109 instance AgdaUnparse Bool where
  110   agdaUnparse True = "true"
  111   agdaUnparse False = "false"
  112 
  113 instance AgdaUnparse Char where
  114   agdaUnparse = viaShow
  115 instance AgdaUnparse Text where
  116   agdaUnparse = viaShow . T.unpack
  117 
  118 instance AgdaUnparse ByteString where
  119   agdaUnparse bs = parens ("mkByteString" <+> viaShow bs)
  120 
  121 instance AgdaUnparse () where
  122   agdaUnparse _ = "tt"
  123 
  124 agdaUnfold :: (AgdaUnparse a, Foldable f) => f a -> Doc ann
  125 agdaUnfold l = parens $ foldr (\x xs -> agdaUnparse x <+> "∷" <+> xs) "[]" l
  126 
  127 instance AgdaUnparse a => AgdaUnparse [a] where
  128   agdaUnparse = agdaUnfold
  129 
  130 instance (AgdaUnparse a, AgdaUnparse b) => AgdaUnparse (a, b) where
  131   agdaUnparse (x, y) = parens (agdaUnparse x <+> "," <+> agdaUnparse y)
  132 
  133 instance AgdaUnparse a => AgdaUnparse (Vector a) where
  134   agdaUnparse v = parens ("mkArray" <+> parens (agdaUnfold v))
  135 
  136 instance (AgdaUnparse a, AgdaUnparse b) => AgdaUnparse (Either a b) where
  137   agdaUnparse (Left a) = parens ("inj₁" <+> agdaUnparse a)
  138   agdaUnparse (Right b) = parens ("inj₂" <+> agdaUnparse b)
  139 instance AgdaUnparse Data where
  140   agdaUnparse (Data.Constr i args) =
  141     parens ("ConstrDATA" <+> agdaUnparse i <+> agdaUnparse args)
  142   agdaUnparse (Data.Map assocList) =
  143     parens ("MapDATA" <+> agdaUnparse assocList)
  144   agdaUnparse (Data.List xs) =
  145     parens ("ListDATA" <+> agdaUnparse xs)
  146   agdaUnparse (Data.I i) =
  147     parens ("iDATA" <+> agdaUnparse i)
  148   agdaUnparse (Data.B b) =
  149     parens ("bDATA" <+> agdaUnparse b)

  164 instance AgdaUnparse (UPLC.DefaultUni (PLC.Esc a)) where
  165   agdaUnparse PLC.DefaultUniInteger = "integer"
  166   agdaUnparse PLC.DefaultUniByteString = "bytestring"
  167   agdaUnparse PLC.DefaultUniString = "string"
  168   agdaUnparse PLC.DefaultUniBool = "bool"
  169   agdaUnparse PLC.DefaultUniUnit = "unit"
  170   agdaUnparse PLC.DefaultUniData = "pdata"
  171   agdaUnparse PLC.DefaultUniValue = "value"
  172   agdaUnparse (PLC.DefaultUniList t) =
  173     parens ("list" <+> agdaUnparse t)
  174   agdaUnparse (PLC.DefaultUniPair t1 t2) =
  175     parens ("pair" <+> agdaUnparse t1 <+> agdaUnparse t2)
  176   agdaUnparse PLC.DefaultUniBLS12_381_G1_Element = "bls12-381-g1-element"
  177   agdaUnparse PLC.DefaultUniBLS12_381_G2_Element = "bls12-381-g2-element"
  178   agdaUnparse PLC.DefaultUniBLS12_381_MlResult = "bls12-381-mlresult"
  179   agdaUnparse (PLC.DefaultUniArray t) =
  180     parens ("array" <+> agdaUnparse t)
  181   agdaUnparse (PLC.DefaultUniApply _ _) = error "Application of an unknown type is not supported."
  182 
  183 instance AgdaUnparse (PLC.Some (PLC.ValueOf UPLC.DefaultUni)) where
  184   agdaUnparse (PLC.Some (PLC.ValueOf univ val)) =
  185     parens $
  186       "tagCon"
  187         <+> agdaUnparse univ
  188         <+> PLC.bring (Proxy @AgdaUnparse) univ (agdaUnparse val)
