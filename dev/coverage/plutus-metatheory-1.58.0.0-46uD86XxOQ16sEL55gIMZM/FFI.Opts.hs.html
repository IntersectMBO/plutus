   17 -- the different budget modes of plc-agda
   18 data BudgetMode a
   19   = Silent
   20   | Counting a
   21   | Tallying a
   22   deriving (Functor)
   23 
   24 countingbudget :: Parser (BudgetMode ())
   25 countingbudget =
   26   flag'
   27     (Counting ())
   28     ( long "counting"
   29         <> short 'c'
   30         <> help "Run machine in counting mode and report results"
   31     )
   32 
   33 tallyingbudget :: Parser (BudgetMode ())
   34 tallyingbudget =
   35   flag'
   36     (Tallying ())
   37     ( long "tallying"
   38         <> short 't'
   39         <> help "Run machine in tallying mode and report results"
   40     )
   41 
   42 budgetmode :: Parser (BudgetMode ())
   43 budgetmode =
   44   asum
   45     [ countingbudget
   46     , tallyingbudget
   47     , pure Silent
   48     ]

   50 -- The different evaluation modes of plc-agda
   51 data EvalMode = U | TL | TCK | TCEK deriving stock (Show, Read)
   52 
   53 data EvalOptions a = EvalOpts Input Format EvalMode (BudgetMode a)
   54   deriving (Functor)
   55 
   56 evalMode :: Parser EvalMode
   57 evalMode =
   58   option
   59     auto
   60     ( long "mode"
   61         <> short 'm'
   62         <> metavar "MODE"
   63         <> value TL
   64         <> showDefault
   65         <> help "Evaluation mode (U , TL, TCK, TCEK)"
   66     )
   67 
   68 evalOpts :: Parser (EvalOptions ())
   69 evalOpts = EvalOpts <$> input <*> inputformat <*> evalMode <*> budgetmode
   70 
   71 data TypecheckOptions = TCOpts Input Format
   72 
   73 typecheckOpts :: Parser TypecheckOptions
   74 typecheckOpts = TCOpts <$> input <*> inputformat
   75 
   76 data Command a
   77   = Eval (EvalOptions a)
   78   | Typecheck TypecheckOptions
   79   deriving (Functor)
   80 
   81 commands :: Parser (Command ())
   82 commands =
   83   hsubparser
   84     ( command
   85         "evaluate"
   86         ( info
   87             (Eval <$> evalOpts)
   88             (fullDesc <> progDesc "run a Plutus Core program")
   89         )
   90         <> command
   91           "typecheck"
   92           ( info
   93               (Typecheck <$> typecheckOpts)
   94               (fullDesc <> progDesc "typecheck a Plutus Core program")
   95           )
   96     )

   98 -- A CostModel has all the information to run the Agda machine
   99 -- with cost reporting
  100 type CostModel = (CekMachineCosts, BuiltinCostMap)
  101 
  102 addJSONParameters :: Command a -> Command CostModel
  103 addJSONParameters = fmap (const (defaultCekMachineCostsForTesting, defaultSimpleBuiltinCostModel))

  105 execP :: IO (Command CostModel)
