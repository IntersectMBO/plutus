   10 import Data.Text as T hiding (map)
   11 import GHC.Exts (IsList (..))

   13 -- Untyped (Raw) syntax
   14 
   15 data UTerm
   16   = UVar Integer
   17   | ULambda UTerm
   18   | UApp UTerm UTerm
   19   | UCon (Some (ValueOf DefaultUni))
   20   | UError
   21   | UBuiltin DefaultFun
   22   | UDelay UTerm
   23   | UForce UTerm
   24   | UConstr Integer [UTerm]
   25   | UCase UTerm [UTerm]
   26   deriving (Eq, Show)
   27 
   28 unIndex :: Index -> Integer
   29 unIndex (Index n) = toInteger n
   30 
   31 convP :: Program NamedDeBruijn DefaultUni DefaultFun a -> UTerm
   32 convP (Program _ _ t) = conv t
   33 
   34 conv :: Term NamedDeBruijn DefaultUni DefaultFun a -> UTerm
   35 conv (Var _ x) = UVar (unIndex (ndbnIndex x) - 1)
   36 conv (LamAbs _ _ t) = ULambda (conv t)
   37 conv (Apply _ t u) = UApp (conv t) (conv u)
   38 conv (Builtin _ b) = UBuiltin b
   39 conv (Constant _ c) = UCon c
   40 conv (Error _) = UError
   41 conv (Delay _ t) = UDelay (conv t)
   42 conv (Force _ t) = UForce (conv t)
   43 conv (Constr _ i es) = UConstr (toInteger i) (toList (fmap conv es))
   44 conv (Case _ arg cs) = UCase (conv arg) (toList (fmap conv cs))
   45 
   46 tmname :: Int -> String
   47 tmname i = 'x' : show i
   48 
   49 uconv :: Int -> UTerm -> Term NamedDeBruijn DefaultUni DefaultFun ()
   50 uconv i (UVar x) =
   51   Var
   52     ()
   53     ( NamedDeBruijn
   54         (T.pack (tmname (i - 1 - fromInteger x)))
   55         -- PLC's debruijn starts counting from 1, while in the metatheory it starts from 0.
   56         (Index (fromInteger x + 1))
   57     )
   58 uconv i (ULambda t) =
   59   LamAbs
   60     ()
   61     (NamedDeBruijn (T.pack (tmname i)) deBruijnInitIndex)
   62     (uconv (i + 1) t)
   63 uconv i (UApp t u) = Apply () (uconv i t) (uconv i u)
   64 uconv _ (UCon c) = Constant () c
   65 uconv _ UError = Error ()
   66 uconv _ (UBuiltin b) = Builtin () b
   67 uconv i (UDelay t) = Delay () (uconv i t)
   68 uconv i (UForce t) = Force () (uconv i t)
   69 uconv i (UConstr j xs) = Constr () (fromInteger j) (fmap (uconv i) xs)
   70 uconv i (UCase t xs) = Case () (uconv i t) (fromList (fmap (uconv i) xs))
