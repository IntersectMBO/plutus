   17 type TraceElem a = (SimplifierStage, (Certify.Hints, (a, a)))
   18 type Trace a = [TraceElem a]
   19 
   20 mkFfiSimplifierTrace
   21   :: SimplifierTrace UPLC.Name UPLC.DefaultUni UPLC.DefaultFun a
   22   -> Trace FFI.UTerm
   23 mkFfiSimplifierTrace (SimplifierTrace simplTrace) = reverse $ toFfiAst <$> simplTrace
   24   where
   25     toFfiAst (Simplification before stage hints after) =
   26       case (UPLC.deBruijnTerm before, UPLC.deBruijnTerm after) of
   27         (Right before', Right after') ->
   28           (stage, (hints, (FFI.conv (void before'), FFI.conv (void after'))))
   29         (Left (err :: UPLC.FreeVariableError), _) -> error $ show err
   30         (_, Left (err :: UPLC.FreeVariableError)) -> error $ show err
