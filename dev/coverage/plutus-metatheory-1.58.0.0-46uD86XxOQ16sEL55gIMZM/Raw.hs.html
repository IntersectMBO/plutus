   20 data KIND = Star | Sharp | Arrow KIND KIND
   21   deriving (Show)

   23 data RType
   24   = RTyVar Integer
   25   | RTyFun RType RType
   26   | RTyPi KIND RType
   27   | RTyLambda KIND RType
   28   | RTyApp RType RType
   29   | RTyCon RTyCon
   30   | RTyMu RType RType
   31   | RTySOP [[RType]]
   32   deriving (Show)

   53 data RTerm
   54   = RVar Integer
   55   | RTLambda KIND RTerm
   56   | RTApp RTerm RType
   57   | RLambda RType RTerm
   58   | RApp RTerm RTerm
   59   | RCon (Some (ValueOf DefaultUni))
   60   | RError RType
   61   | RBuiltin DefaultFun
   62   | RWrap RType RType RTerm
   63   | RUnWrap RTerm
   64   | RConstr RType Integer [RTerm]
   65   | RCase RType RTerm [RTerm]
   66   deriving (Show)
   67 
   68 unIndex :: Index -> Integer
   69 unIndex (Index n) = toInteger n

   71 convP :: Program NamedTyDeBruijn NamedDeBruijn DefaultUni DefaultFun a -> RTerm
   72 convP (Program _ _ t) = conv t

   74 convK :: Kind a -> KIND
   75 convK (Type _) = Star
   76 convK (KindArrow _ k k') = Arrow (convK k) (convK k')
   77 
   78 convT :: Type NamedTyDeBruijn DefaultUni a -> RType
   79 convT (TyVar _ (NamedTyDeBruijn x)) = RTyVar (unIndex (ndbnIndex x))
   80 convT (TyFun _ _A _B) = RTyFun (convT _A) (convT _B)
   81 convT (TyForall _ _ _K _A) = RTyPi (convK _K) (convT _A)
   82 convT (TyLam _ _ _K _A) = RTyLambda (convK _K) (convT _A)
   83 convT (TyApp _ _A _B) = RTyApp (convT _A) (convT _B)
   84 convT (TyBuiltin ann (SomeTypeIn (DefaultUniApply f x))) =
   85   RTyApp
   86     (convT (TyBuiltin ann (SomeTypeIn f)))
   87     (convT (TyBuiltin ann (SomeTypeIn x)))
   88 convT (TyBuiltin _ someUni) = convTyCon someUni
   89 convT (TyIFix _ a b) = RTyMu (convT a) (convT b)
   90 convT (TySOP _ xss) = RTySOP (map (map convT) xss)
   91 
   92 convTyCon :: SomeTypeIn DefaultUni -> RType
   93 convTyCon (SomeTypeIn DefaultUniInteger) = RTyCon (RTyConAtom ATyConInt)
   94 convTyCon (SomeTypeIn DefaultUniByteString) = RTyCon (RTyConAtom ATyConBS)
   95 convTyCon (SomeTypeIn DefaultUniString) = RTyCon (RTyConAtom ATyConStr)
   96 convTyCon (SomeTypeIn DefaultUniBool) = RTyCon (RTyConAtom ATyConBool)
   97 convTyCon (SomeTypeIn DefaultUniUnit) = RTyCon (RTyConAtom ATyConUnit)
   98 convTyCon (SomeTypeIn DefaultUniData) = RTyCon (RTyConAtom ATyConData)
   99 convTyCon (SomeTypeIn DefaultUniBLS12_381_G1_Element) = RTyCon (RTyConAtom ATyConBLS12_381_G1_Element)
  100 convTyCon (SomeTypeIn DefaultUniBLS12_381_G2_Element) = RTyCon (RTyConAtom ATyConBLS12_381_G2_Element)
  101 convTyCon (SomeTypeIn DefaultUniBLS12_381_MlResult) = RTyCon (RTyConAtom ATyConBLS12_381_MlResult)
  102 convTyCon (SomeTypeIn DefaultUniProtoList) = RTyCon RTyConList
  103 convTyCon (SomeTypeIn DefaultUniProtoArray) = RTyCon RTyConArray
  104 convTyCon (SomeTypeIn DefaultUniProtoPair) = RTyCon RTyConPair
  105 convTyCon (SomeTypeIn (DefaultUniApply _ _)) = error "unsupported builtin type application"

  107 conv :: Term NamedTyDeBruijn NamedDeBruijn DefaultUni DefaultFun a -> RTerm
  108 conv (Var _ x) = RVar (unIndex (ndbnIndex x))
  109 conv (TyAbs _ _ _K t) = RTLambda (convK _K) (conv t)
  110 conv (TyInst _ t _A) = RTApp (conv t) (convT _A)
  111 conv (LamAbs _ _ _A t) = RLambda (convT _A) (conv t)
  112 conv (Apply _ t u) = RApp (conv t) (conv u)
  113 conv (Builtin _ b) = RBuiltin b
  114 conv (Constant _ c) = RCon c
  115 conv (Unwrap _ t) = RUnWrap (conv t)
  116 conv (IWrap _ ty1 ty2 t) = RWrap (convT ty1) (convT ty2) (conv t)
  117 conv (Error _ _A) = RError (convT _A)
  118 conv (Constr _ _A i cs) = RConstr (convT _A) (toInteger i) (fmap conv cs)
  119 conv (Case _ _A arg cs) = RCase (convT _A) (conv arg) (fmap conv cs)

  121 varTm :: Int -> NamedDeBruijn
  122 varTm i = NamedDeBruijn (T.pack [tmnames !! i]) deBruijnInitIndex
  123 
  124 varTy :: Int -> NamedDeBruijn
  125 varTy i = NamedDeBruijn (T.pack [tynames !! i]) deBruijnInitIndex
  126 
  127 unconvK :: KIND -> Kind ()
  128 unconvK (Arrow k k') = KindArrow () (unconvK k) (unconvK k')
  129 unconvK _ = Type ()
  130 
  131 -- this should take a level and render levels as names
  132 unconvT :: Int -> RType -> Type NamedTyDeBruijn DefaultUni ()
  133 unconvT i (RTyVar x) =
  134   TyVar () (NamedTyDeBruijn (NamedDeBruijn (T.pack [tynames !! (i - fromIntegral x)]) (Index (fromInteger x))))
  135 unconvT i (RTyFun t u) = TyFun () (unconvT i t) (unconvT i u)
  136 unconvT i (RTyPi k t) =
  137   TyForall () (NamedTyDeBruijn (varTy i)) (unconvK k) (unconvT (i + 1) t)
  138 unconvT i (RTyLambda k t) = TyLam () (NamedTyDeBruijn (varTy i)) (unconvK k) (unconvT (i + 1) t)
  139 unconvT i (RTyApp t u) = TyApp () (unconvT i t) (unconvT i u)
  140 unconvT _ (RTyCon c) = TyBuiltin () (unconvTyCon c)
  141 unconvT i (RTyMu t u) = TyIFix () (unconvT i t) (unconvT i u)
  142 unconvT i (RTySOP xss) = TySOP () (map (map (unconvT i)) xss)
  143 
  144 unconvTyCon :: RTyCon -> SomeTypeIn DefaultUni
  145 unconvTyCon (RTyConAtom ATyConInt) = SomeTypeIn DefaultUniInteger
  146 unconvTyCon (RTyConAtom ATyConBS) = SomeTypeIn DefaultUniByteString
  147 unconvTyCon (RTyConAtom ATyConStr) = SomeTypeIn DefaultUniString
  148 unconvTyCon (RTyConAtom ATyConBool) = SomeTypeIn DefaultUniBool
  149 unconvTyCon (RTyConAtom ATyConUnit) = SomeTypeIn DefaultUniUnit
  150 unconvTyCon (RTyConAtom ATyConData) = SomeTypeIn DefaultUniData
  151 unconvTyCon (RTyConAtom ATyConBLS12_381_G1_Element) =
  152   SomeTypeIn DefaultUniBLS12_381_G1_Element
  153 unconvTyCon (RTyConAtom ATyConBLS12_381_G2_Element) =
  154   SomeTypeIn DefaultUniBLS12_381_G2_Element
  155 unconvTyCon (RTyConAtom ATyConBLS12_381_MlResult) =
  156   SomeTypeIn DefaultUniBLS12_381_MlResult
  157 unconvTyCon RTyConList = SomeTypeIn DefaultUniProtoList
  158 unconvTyCon RTyConArray = SomeTypeIn DefaultUniProtoArray
  159 unconvTyCon RTyConPair = SomeTypeIn DefaultUniProtoPair

  166 unconv :: Int -> RTerm -> Term NamedTyDeBruijn NamedDeBruijn DefaultUni DefaultFun ()
  167 unconv i (RVar x) =
  168   Var () (NamedDeBruijn (T.pack [tmnames !! (i - fromInteger x)]) (Index (fromInteger x)))
  169 unconv i (RTLambda k tm) = TyAbs () (NamedTyDeBruijn (varTy i)) (unconvK k) (unconv (i + 1) tm)
  170 unconv i (RTApp t ty) = TyInst () (unconv i t) (unconvT i ty)
  171 unconv i (RLambda ty tm) = LamAbs () (varTm i) (unconvT (i + 1) ty) (unconv (i + 1) tm)
  172 unconv i (RApp t u) = Apply () (unconv i t) (unconv i u)
  173 unconv _ (RCon c) = Constant () c
  174 unconv i (RError ty) = Error () (unconvT i ty)
  175 unconv _ (RBuiltin b) = Builtin () b
  176 unconv i (RWrap tyA tyB t) = IWrap () (unconvT i tyA) (unconvT i tyB) (unconv i t)
  177 unconv i (RUnWrap t) = Unwrap () (unconv i t)
  178 unconv i (RConstr ty j cs) = Constr () (unconvT i ty) (fromInteger j) (fmap (unconv i) cs)
  179 unconv i (RCase ty arg cs) = Case () (unconvT i ty) (unconv i arg) (fmap (unconv i) cs)
