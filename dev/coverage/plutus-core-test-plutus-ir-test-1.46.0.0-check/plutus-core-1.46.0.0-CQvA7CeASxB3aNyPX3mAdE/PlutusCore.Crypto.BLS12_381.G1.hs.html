   37 {- Note [Wrapping the BLS12-381 types in Plutus Core].  In the Haskell bindings
   38 to the `blst` library in cardano-crypto-class, points in G1 and G2 are
   39 represented as ForeignPtrs pointing to C objects, with a phantom type
   40 determining which group is involved. We have to wrap these in a newtype here
   41 because otherwise the builtin machinery spots that they're applications and
   42 can't find the relevant type parameters.  In theory I think we could add a
   43 couple of phantom types to the default universe, but it seemed simpler and safer
   44 to use monomorphic types instead, even though it requires a bit of code
   45 duplication between G1 and G2.  Much of the code here has apparently identical
   46 versions in PlutusCore.Crypto.BLS12_381.G2, but the signatures of the functions
   47 in the two versions are invisibly instantiating the cardano-crypto-class
   48 functions at the appropriate phantom types.
   49 
   50 See also Note [Wrapping the BLS12-381 types in PlutusTx].
   51 -}
   52 newtype Element = Element { unElement :: BlstBindings.Point1 }
   53     deriving newtype (Eq)
   54 instance Show Element where
   55     show  = byteStringAsHex . compress
   56 instance Pretty Element where
   57     pretty = pretty . show
   58 instance PrettyBy ConstConfig Element
   59 {- | We don't support direct flat encoding of G2 elements because of the expense
   60    of on-chain uncompression.  Users should convert between G2 elements and
   61    bytestrings using `compress` and `uncompress`: the bytestrings can be
   62    flat-encoded in the usual way. -}
   63 instance Flat Element where
   64     -- This might happen on the chain, so `fail` rather than `error`.
   65     decode = fail "Flat decoding is not supported for objects of type bls12_381_G1_element: use bls12_381_G1_uncompress on a bytestring instead."
   66     -- This will be a Haskell runtime error, but encoding doesn't happen on chain,
   67     -- so it's not too bad.
   68     encode = error "Flat encoding is not supported for objects of type bls12_381_G1_element: use bls12_381_G1_compress to obtain a bytestring instead."
   69     size _ = id
   70 instance NFData Element where
   71     rnf (Element x) = rwhnf x  -- Just to be on the safe side.
   72 
   73 instance Hashable Element where
   74     hashWithSalt salt = hashWithSalt salt . compress

   76 -- | Add two G1 group elements
   77 add :: Element -> Element -> Element
   78 add = coerce (BlstBindings.blsAddOrDouble @BlstBindings.Curve1)
   79 {-# INLINE add #-}

   81 -- | Negate a G1 group element
   82 neg :: Element -> Element
   83 neg = coerce (BlstBindings.blsNeg @BlstBindings.Curve1)
   84 {-# INLINE neg #-}

   86 -- | Multiplication of group elements by scalars. In the blst library the
   87 -- arguments are the other way round, but scalars acting on the left is more
   88 -- consistent with standard mathematical practice.
   89 scalarMul :: Integer -> Element -> Element
   90 scalarMul = coerce $ flip (BlstBindings.blsMult @BlstBindings.Curve1)
   91 {-# INLINE scalarMul #-}

   93 {- | Compress a G1 element to a bytestring. This serialises a curve point to its
   94  x coordinate only.  The compressed bytestring is 48 bytes long, with three
   95  spare bits used to convey extra information about the point, including
   96  determining which of two possible y coordinates the point has and whether the
   97  point is the point at infinity. See
   98  https://github.com/supranational/blst#serialization-format
   99 -}
  100 compress :: Element -> ByteString
  101 compress = coerce (BlstBindings.blsCompress @BlstBindings.Curve1)
  102 {-# INLINE compress #-}
  103 
  104 {- | Uncompress a bytestring to get a G1 point.  This will fail if any of the
  105    following are true.
  106      * The bytestring is not exactly 48 bytes long.
  107      * The most significant three bits are used incorrectly.
  108      * The bytestring encodes a field element which is not the
  109        x coordinate of a point on the E1 curve.
  110      * The bytestring does represent a point on the E1 curve, but the
  111        point is not in the G1 subgroup.
  112 -}
  113 uncompress :: ByteString -> Either BlstBindings.BLSTError Element
  114 uncompress = coerce (BlstBindings.blsUncompress @BlstBindings.Curve1)
  115 {-# INLINE uncompress #-}

  117 {-  Note [Hashing and Domain Separation Tags].  The hashToGroup functions take a
  118    bytestring and hash it to obtain an element in the relevant group, as
  119    described in
  120 
  121    https://datatracker.ietf.org/doc/html/draft-irtf-cfrg-hash-to-curve.
  122 
  123    In addition to the input bytestring (the "message"), the hashing function
  124    also takes another (possibly empty) bytestring argument, a "Domain Separation
  125    Tag" (DST), which is incorporated in the hashing process; the intention is
  126    that for security purposes different protocols should use different DSTs to
  127    ensure that hashes are unique to the protocol in use: see Section 2.2.5 of
  128    the above specification.  In principle, arbitrary-length DSTs can be used,
  129    but we only allow DSTs of up to 255 bytes, failing if a larger DST is
  130    supplied.  If a larger DST is required, it should be hashed beforehand to
  131    obtain a hash of accpetable size, as described in Section 5.3.3 of the
  132    specification.
  133 
  134    The hashing functions in the blst library allow a third argument as well (an
  135    "augmentation string").  We don't support this functionality directly because
  136    precisely the same effect can be achieved by prepending the augmentation
  137    string to the message.
  138 -}
  139 
  140 -- | Take an arbitrary bytestring and a Domain Separation Tag (DST) and hash
  141 -- them to a get point in G1.
  142 hashToGroup :: ByteString -> ByteString -> Either BLS12_381_Error Element
  143 hashToGroup msg dst =
  144     if Data.ByteString.length dst > 255
  145     then Left HashToCurveDstTooBig
  146     else Right . Element $ BlstBindings.blsHash @BlstBindings.Curve1 msg (Just dst) Nothing
