   62     = BuiltinUnliftingEvaluationError UnliftingEvaluationError
   63     | BuiltinEvaluationFailure
   64     deriving stock (Show, Eq)

   66 -- | The monad that 'makeKnown' runs in.
   67 -- Equivalent to @ExceptT BuiltinError (Writer (DList Text))@, except optimized in two ways:
   68 --
   69 -- 1. everything is strict
   70 -- 2. has the 'BuiltinSuccess' constructor that is used for returning a value with no logs
   71 --    attached, which is the most common case for us, so it helps a lot not to construct and
   72 --    deconstruct a redundant tuple
   73 --
   74 -- Moving from @ExceptT BuiltinError (Writer (DList Text))@ to this data type gave us a speedup of
   75 -- 8% of total evaluation time.
   76 --
   77 -- Logs are represented as a 'DList', because we don't particularly care about the efficiency of
   78 -- logging, since there's no logging on the chain and builtins don't emit much anyway. Otherwise
   79 -- we'd have to use @text-builder@ or @text-builder-linear@ or something of this sort.
   80 data BuiltinResult a
   81     = -- 'BuiltinSuccess' is the first constructor to make it a bit more likely for GHC to
   82       -- branch-predict it (which is something that we want, because most builtins return this
   83       -- constructor). It is however not guaranteed that GHC will predict it, because even though
   84       -- it's likely going to be a recursive case (it certainly is in the CEK machine) and thus the
   85       -- constructor has precedence over 'BuiltinFailure', it doesn't have precedence over
   86       -- 'BuiltinSuccessWithLogs', since that case is equally likely to be recursive.
   87       --
   88       -- Unfortunately, GHC doesn't offer any explicit control over branch-prediction (see this
   89       -- ticket: https://gitlab.haskell.org/ghc/ghc/-/issues/849), so relying on hope is the best we
   90       -- can do here.
   91       BuiltinSuccess a
   92     | BuiltinSuccessWithLogs (DList Text) a
   93     | BuiltinFailure (DList Text) BuiltinError
   94     deriving stock (Show, Foldable)

  107 -- | An 'UnliftingEvaluationError' /is/ an 'EvaluationError', hence for this instance we only
  108 -- require both @structural@ and @operational@ to have '_UnliftingError' prisms, so that we can
  109 -- handle both the cases pointwisely.
  110 instance (AsUnliftingError structural, AsUnliftingError operational) =>
  111         AsUnliftingEvaluationError (EvaluationError structural operational) where
  112     _UnliftingEvaluationError = go . coerced where
  113         go =
  114             prism'
  115                 (bimap
  116                     (review _UnliftingError)
  117                     (review _UnliftingError))
  118                 (bitraverse
  119                     (reoption . matching _UnliftingError)
  120                     (reoption . matching _UnliftingError))
  121     {-# INLINE _UnliftingEvaluationError #-}
  122 
  123 instance AsUnliftingEvaluationError BuiltinError where
  124     _UnliftingEvaluationError = _BuiltinUnliftingEvaluationError . _UnliftingEvaluationError
  125     {-# INLINE _UnliftingEvaluationError #-}

  162 {- Note [INLINE and OPAQUE on error-related definitions]
  163 We mark error-related definitions such as prisms like '_StructuralUnliftingError' and regular
  164 functions like 'throwNotAConstant' with @INLINE@, because this produces significantly less cluttered
  165 GHC Core. Not doing so results in 20+% larger Core for builtins.
  166 
  167 However in a few specific cases we use @OPAQUE@ instead to get tighter Core. @OPAQUE@ is the
  168 same as @NOINLINE@ except the former _actually_ prevents GHC from inlining the definition unlike the
  169 former. See this for details: https://github.com/ghc-proposals/ghc-proposals/blob/5577fd008924de8d89cfa9855fa454512e7dcc75/proposals/0415-opaque-pragma.rst
  170 
  171 It's hard to predict where @OPAQUE@ instead of @INLINE@ will help to make GHC Core tidier, so it's
  172 mostly just looking into the Core and seeing where there's obvious duplication that can be removed.
  173 Such cases tend to be functions returning a value of a concrete error type (as opposed to a type
  174 variable).
  175 -}

  177 -- See Note [Ignoring context in OperationalError].
  178 -- | Construct a prism focusing on the @*EvaluationFailure@ part of @err@ by taking
  179 -- that @*EvaluationFailure@ and
  180 --
  181 -- 1. pretty-printing and embedding it into an 'UnliftingError' for the setter part of the prism
  182 -- 2. returning it directly for the opposite direction (there's no other way to convert an
  183 --    'UnliftingError' to an evaluation failure, since the latter doesn't carry any content)
  184 --
  185 -- This is useful for providing 'AsUnliftingError' instances for types such as 'CkUserError' and
  186 -- 'CekUserError'.
  187 _UnliftingErrorVia :: Pretty err => err -> Prism' err UnliftingError
  188 _UnliftingErrorVia err = iso (MkUnliftingError . display) (const err)
  189 {-# INLINE _UnliftingErrorVia #-}

  209 -- | Add a log line to the logs.
  210 emit :: Text -> BuiltinResult ()
  211 emit txt = BuiltinSuccessWithLogs (pure txt) ()
  212 {-# INLINE emit #-}
  213 
  214 -- | Prepend logs to a 'BuiltinResult' computation.
  215 withLogs :: DList Text -> BuiltinResult a -> BuiltinResult a
  216 withLogs logs1 = \case
  217     BuiltinSuccess x               -> BuiltinSuccessWithLogs logs1 x
  218     BuiltinSuccessWithLogs logs2 x -> BuiltinSuccessWithLogs (logs1 <> logs2) x
  219     BuiltinFailure logs2 err       -> BuiltinFailure (logs1 <> logs2) err
  220 {-# INLINE withLogs #-}
  221 
  222 instance Functor BuiltinResult where
  223     fmap f (BuiltinSuccess x)              = BuiltinSuccess (f x)
  224     fmap f (BuiltinSuccessWithLogs logs x) = BuiltinSuccessWithLogs logs (f x)
  225     fmap _ (BuiltinFailure logs err)       = BuiltinFailure logs err
  226     {-# INLINE fmap #-}
  227 
  228     -- Written out explicitly just in case.
  229     x <$ BuiltinSuccess _              = BuiltinSuccess x
  230     x <$ BuiltinSuccessWithLogs logs _ = BuiltinSuccessWithLogs logs x
  231     _ <$ BuiltinFailure logs err       = BuiltinFailure logs err
  232     {-# INLINE (<$) #-}
  233 
  234 instance Applicative BuiltinResult where
  235     pure = BuiltinSuccess
  236     {-# INLINE pure #-}
  237 
  238     BuiltinSuccess f              <*> a = fmap f a
  239     BuiltinSuccessWithLogs logs f <*> a = withLogs logs $ fmap f a
  240     BuiltinFailure logs err       <*> _ = BuiltinFailure logs err
  241     {-# INLINE (<*>) #-}
  242 
  243     -- Better than the default implementation, because the value in the 'BuiltinSuccess' case
  244     -- doesn't need to be retained.
  245     BuiltinSuccess _              *> b = b
  246     BuiltinSuccessWithLogs logs _ *> b = withLogs logs b
  247     BuiltinFailure logs err       *> _ = BuiltinFailure logs err
  248     {-# INLINE (*>) #-}
  249 
  250 instance Monad BuiltinResult where
  251     BuiltinSuccess x              >>= f = f x
  252     BuiltinSuccessWithLogs logs x >>= f = withLogs logs $ f x
  253     BuiltinFailure logs err       >>= _ = BuiltinFailure logs err
  254     {-# INLINE (>>=) #-}
  255 
  256     (>>) = (*>)
  257     {-# INLINE (>>) #-}

  259 -- | 'throwError' puts every operational unlifting error into the 'BuiltinFailure' logs. This is to
  260 -- compensate for the historical lack of error message content in operational errors (structural
  261 -- ones don't have this problem) in our evaluators (the CK and CEK machines). It would be better to
  262 -- fix the underlying issue and allow operational evaluation errors to carry some form of content,
  263 -- but for now we just fix the symptom in order for the end user to see the error message that they
  264 -- are supposed to see. The fix even makes some sense: what we do here is we emulate logging when
  265 -- the thrown unlifting error is an operational one, i.e. this is similar to what some builtins do
  266 -- manually (like when a crypto builtin fails and puts info about the failure into the logs).
  267 instance MonadError BuiltinError BuiltinResult where
  268     throwError builtinErr = BuiltinFailure operationalLogs builtinErr where
  269         operationalLogs = case builtinErr of
  270             BuiltinUnliftingEvaluationError
  271                 (MkUnliftingEvaluationError
  272                     (OperationalError
  273                         (MkUnliftingError operationalErr))) -> pure operationalErr
  274             _ -> mempty
  275     {-# INLINE throwError #-}
  276 
  277     -- Throwing logs out is lame, but embedding them into the error would be weird, since that
  278     -- would change the error. Not that any of that matters, we only implement this because it's a
  279     -- method of 'MonadError' and we can't not implement it.
  280     --
  281     -- We could make it @MonadError (DList Text, BuiltinError)@, but logs are arbitrary and are not
  282     -- necessarily an inherent part of an error, so preserving them is as questionable as not doing
  283     -- so.
  284     BuiltinFailure _ err `catchError` f = f err
  285     res                  `catchError` _ = res
  286     {-# INLINE catchError #-}
