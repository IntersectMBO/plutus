   47 -- | Errors which can occur during a run of an abstract machine.
   48 data MachineError fun
   49     = NonPolymorphicInstantiationMachineError
   50       -- ^ An attempt to reduce a not immediately reducible type instantiation.
   51     | NonWrapUnwrappedMachineError
   52       -- ^ An attempt to unwrap a not wrapped term.
   53     | NonFunctionalApplicationMachineError
   54       -- ^ An attempt to reduce a not immediately reducible application.
   55     | OpenTermEvaluatedMachineError
   56       -- ^ An attempt to evaluate an open term.
   57     | UnliftingMachineError UnliftingError
   58       -- ^ An attempt to compute a constant application resulted in 'UnliftingError'.
   59     | BuiltinTermArgumentExpectedMachineError
   60       -- ^ A builtin expected a term argument, but something else was received.
   61     | UnexpectedBuiltinTermArgumentMachineError
   62       -- ^ A builtin received a term argument when something else was expected
   63     | NonConstrScrutinizedMachineError
   64       -- ^ An attempt to scrutinize a non-constr.
   65     | MissingCaseBranchMachineError Word64
   66       -- ^ An attempt to go into a non-existent case branch.
   67     | PanicMachineError String
   68       -- ^ A GHC exception was thrown.
   69     deriving stock (Show, Eq, Functor, Generic)
   70     deriving anyclass (NFData)

   80 type EvaluationException structural operational =
   81     ErrorWithCause (EvaluationError structural operational)
   82 
   83 {- Note [Ignoring context in OperationalError]
   84 The 'OperationalError' error has a term argument, but 'splitStructuralOperational' just
   85 discards this and returns 'EvaluationFailure'. This means that, for example, if we use the @plc@
   86 command to execute a program containing a division by zero, @plc@ exits silently without reporting
   87 that anything has gone wrong (but returning a non-zero exit code to the shell via 'exitFailure').
   88 This is because 'OperationalError' is used in cases when a PLC program itself goes wrong
   89 (see the Haddock of 'EvaluationError'). This is used to signal unsuccessful validation and so is
   90 not regarded as a real error; in contrast structural errors are genuine errors and we report their
   91 context if available.
   92 -}
   93 
   94 -- See the Haddock of 'EvaluationError' for what structural and operational errors are.
   95 -- See Note [Ignoring context in OperationalError].
   96 -- | Preserve the contents of an 'StructuralError' as a 'Left' and turn an
   97 -- 'OperationalError' into a @Right EvaluationFailure@ (thus erasing the content of the
   98 -- error in the latter case).
   99 splitStructuralOperational
  100     :: Either (EvaluationException structural operational term) a
  101     -> Either (ErrorWithCause structural term) (EvaluationResult a)
  102 splitStructuralOperational (Right term) = Right $ EvaluationSuccess term
  103 splitStructuralOperational (Left (ErrorWithCause evalErr cause)) = case evalErr of
  104     StructuralError err -> Left $ ErrorWithCause err cause
  105     OperationalError _  -> Right EvaluationFailure
  106 
  107 -- | Throw on a 'StructuralError' and turn an 'OperationalError' into an
  108 -- 'EvaluationFailure' (thus erasing the content of the error in the latter case).
  109 unsafeSplitStructuralOperational
  110     :: (PrettyPlc structural, PrettyPlc term, Typeable structural, Typeable term)
  111     => Either (EvaluationException structural operational term) a
  112     -> EvaluationResult a
  113 unsafeSplitStructuralOperational = unsafeFromEither . splitStructuralOperational

  115 instance (HasPrettyDefaults config ~ 'True, Pretty fun) =>
  116             PrettyBy config (MachineError fun) where
  117     prettyBy _      NonPolymorphicInstantiationMachineError =
  118         "Attempted to instantiate a non-polymorphic term."
  119     prettyBy _      NonWrapUnwrappedMachineError          =
  120         "Cannot unwrap a not wrapped term."
  121     prettyBy _      NonFunctionalApplicationMachineError   =
  122         "Attempted to apply a non-function."
  123     prettyBy _      OpenTermEvaluatedMachineError         =
  124         "Cannot evaluate an open term"
  125     prettyBy _      BuiltinTermArgumentExpectedMachineError =
  126         "A builtin expected a term argument, but something else was received"
  127     prettyBy _      UnexpectedBuiltinTermArgumentMachineError =
  128         "A builtin received a term argument when something else was expected"
  129     prettyBy _      (UnliftingMachineError unliftingError)  =
  130         pretty unliftingError
  131     prettyBy _      NonConstrScrutinizedMachineError =
  132         "A non-constructor value was scrutinized in a case expression"
  133     prettyBy _      (MissingCaseBranchMachineError i) =
  134         "Case expression missing the branch required by the scrutinee tag:" <+> pretty i
  135     prettyBy _      (PanicMachineError err) = vcat
  136         [ "Panic: a GHC exception was thrown, please report this as a bug."
  137         , "The error: " <+> pretty err
  138         ]
