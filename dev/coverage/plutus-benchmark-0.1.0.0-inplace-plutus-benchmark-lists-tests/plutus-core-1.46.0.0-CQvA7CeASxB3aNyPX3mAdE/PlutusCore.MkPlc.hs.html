   71 -- | A final encoding for Term, to allow PLC terms to be used transparently as PIR terms.
   72 class TermLike term tyname name uni fun | term -> tyname name uni fun where
   73     var      :: ann -> name -> term ann
   74     tyAbs    :: ann -> tyname -> Kind ann -> term ann -> term ann
   75     lamAbs   :: ann -> name -> Type tyname uni ann -> term ann -> term ann
   76     apply    :: ann -> term ann -> term ann -> term ann
   77     constant :: ann -> Some (ValueOf uni) -> term ann
   78     builtin  :: ann -> fun -> term ann
   79     tyInst   :: ann -> term ann -> Type tyname uni ann -> term ann
   80     unwrap   :: ann -> term ann -> term ann
   81     iWrap    :: ann -> Type tyname uni ann -> Type tyname uni ann -> term ann -> term ann
   82     error    :: ann -> Type tyname uni ann -> term ann
   83     constr   :: ann -> Type tyname uni ann -> Word64 -> [term ann] -> term ann
   84     kase     :: ann -> Type tyname uni ann -> term ann -> [term ann] -> term ann
   85 
   86     termLet  :: ann -> TermDef term tyname name uni ann -> term ann -> term ann
   87     typeLet  :: ann -> TypeDef tyname uni ann -> term ann -> term ann
   88 
   89     termLet = mkImmediateLamAbs
   90     typeLet = mkImmediateTyAbs

   94 -- | Embed a type (given its explicit type tag) into a PLC type.
   95 mkTyBuiltinOf :: forall k (a :: k) uni tyname ann. ann -> uni (Esc a) -> Type tyname uni ann
   96 mkTyBuiltinOf ann = TyBuiltin ann . SomeTypeIn

   98 -- | Embed a Haskell value (given its explicit type tag) into a PLC term.
   99 mkConstantOf
  100     :: forall a uni fun term tyname name ann. TermLike term tyname name uni fun
  101     => ann -> uni (Esc a) -> a -> term ann
  102 mkConstantOf ann uni = constant ann . someValueOf uni
  103 
  104 -- | Embed a Haskell value (provided its type is in the universe) into a PLC term.
  105 mkConstant
  106     :: forall a uni fun term tyname name ann.
  107        (TermLike term tyname name uni fun, uni `HasTermLevel` a)
  108     => ann -> a -> term ann
  109 mkConstant ann = constant ann . someValue

  111 instance TermLike (Term tyname name uni fun) tyname name uni fun where
  112     var      = Var
  113     tyAbs    = TyAbs
  114     lamAbs   = LamAbs
  115     apply    = Apply
  116     constant = Constant
  117     builtin  = Builtin
  118     tyInst   = TyInst
  119     unwrap   = Unwrap
  120     iWrap    = IWrap
  121     error    = Error
  122     constr   = Constr
  123     kase     = Case
  124 
  125 embedTerm :: TermLike term tyname name uni fun => Term tyname name uni fun ann -> term ann
  126 embedTerm = \case
  127     Var a n           -> var a n
  128     TyAbs a tn k t    -> tyAbs a tn k (embedTerm t)
  129     LamAbs a n ty t   -> lamAbs a n ty (embedTerm t)
  130     Apply a t1 t2     -> apply a (embedTerm t1) (embedTerm t2)
  131     Constant a c      -> constant a c
  132     Builtin a bi      -> builtin a bi
  133     TyInst a t ty     -> tyInst a (embedTerm t) ty
  134     Error a ty        -> error a ty
  135     Unwrap a t        -> unwrap a (embedTerm t)
  136     IWrap a ty1 ty2 t -> iWrap a ty1 ty2 (embedTerm t)
  137     Constr a ty i es  -> constr a ty i (fmap embedTerm es)
  138     Case a ty arg cs  -> kase a ty (embedTerm arg) (fmap embedTerm cs)

  140 -- | Make a 'Var' referencing the given 'VarDecl'.
  141 -- The @ann@ is propagated from the 'VarDecl' to the 'Var'.
  142 mkVar :: TermLike term tyname name uni fun => VarDecl tyname name uni ann -> term ann
  143 mkVar v = var (_varDeclAnn v) (_varDeclName v)

  151 -- | A definition. Pretty much just a pair with more descriptive names.
  152 data Def var val = Def
  153     { defVar :: var
  154     , defVal :: val
  155     } deriving stock (Show, Eq, Ord, Generic)

  157 -- | A term definition as a variable.
  158 type TermDef term tyname name uni ann = Def (VarDecl tyname name uni ann) (term ann)
  159 -- | A type definition as a type variable.
  160 type TypeDef tyname uni ann = Def (TyVarDecl tyname ann) (Type tyname uni ann)

  162 -- | The type of a PLC function.
  163 data FunctionType tyname uni ann = FunctionType
  164     { _functionTypeAnn :: ann                  -- ^ An annotation.
  165     , _functionTypeDom :: Type tyname uni ann  -- ^ The domain of a function.
  166     , _functionTypeCod :: Type tyname uni ann  -- ^ The codomain of the function.
  167     }
  168 
  169 -- Should we parameterize 'VarDecl' by @ty@ rather than @tyname@, so that we can define
  170 -- 'FunctionDef' as 'TermDef FunctionType tyname name uni fun ann'?
  171 -- Perhaps we even should define general 'Decl' and 'Def' that cover all of the cases?
  172 -- | A PLC function.
  173 data FunctionDef term tyname name uni fun ann = FunctionDef
  174     { _functionDefAnn  :: ann                          -- ^ An annotation.
  175     , _functionDefName :: name                         -- ^ The name of a function.
  176     , _functionDefType :: FunctionType tyname uni ann  -- ^ The type of the function.
  177     , _functionDefTerm :: term ann                     -- ^ The definition of the function.
  178     }
  179 
  180 -- | Convert a 'FunctionType' to the corresponding 'Type'.
  181 functionTypeToType :: FunctionType tyname uni ann -> Type tyname uni ann
  182 functionTypeToType (FunctionType ann dom cod) = TyFun ann dom cod
  183 
  184 -- | Get the type of a 'FunctionDef'.
  185 functionDefToType :: FunctionDef term tyname name uni fun ann -> Type tyname uni ann
  186 functionDefToType (FunctionDef _ _ funTy _) = functionTypeToType funTy
  187 
  188 -- | Convert a 'FunctionDef' to a 'VarDecl'. I.e. ignore the actual term.
  189 functionDefVarDecl :: FunctionDef term tyname name uni fun ann -> VarDecl tyname name uni ann
  190 functionDefVarDecl (FunctionDef ann name funTy _) = VarDecl ann name $ functionTypeToType funTy
  191 
  192 -- | Make a 'FunctionDef'. Return 'Nothing' if the provided type is not functional.
  193 mkFunctionDef
  194     :: ann
  195     -> name
  196     -> Type tyname uni ann
  197     -> term ann
  198     -> Maybe (FunctionDef term tyname name uni fun ann)
  199 mkFunctionDef annName name (TyFun annTy dom cod) term =
  200     Just $ FunctionDef annName name (FunctionType annTy dom cod) term
  201 mkFunctionDef _       _    _                     _    = Nothing

  203 -- | Make a "let-binding" for a term as an immediately applied lambda abstraction.
  204 mkImmediateLamAbs
  205     :: TermLike term tyname name uni fun
  206     => ann
  207     -> TermDef term tyname name uni ann
  208     -> term ann -- ^ The body of the let, possibly referencing the name.
  209     -> term ann
  210 mkImmediateLamAbs ann1 (Def (VarDecl ann2 name ty) bind) body =
  211     apply ann1 (lamAbs ann2 name ty body) bind
  212 
  213 -- | Make a "let-binding" for a type as an immediately instantiated type abstraction. Note: the body must be a value.
  214 mkImmediateTyAbs
  215     :: TermLike term tyname name uni fun
  216     => ann
  217     -> TypeDef tyname uni ann
  218     -> term ann -- ^ The body of the let, possibly referencing the name.
  219     -> term ann
  220 mkImmediateTyAbs ann1 (Def (TyVarDecl ann2 name k) bind) body =
  221     tyInst ann1 (tyAbs ann2 name k body) bind

  223 -- | Make an iterated application. Each `apply` node uses the annotation associated with
  224 -- the corresponding argument.
  225 mkIterApp
  226     :: TermLike term tyname name uni fun
  227     => term ann
  228     -> [(ann, term ann)]
  229     -> term ann
  230 mkIterApp = foldl' $ \acc (ann, arg) -> apply ann acc arg
  231 
  232 -- | Make an iterated application with no annotation.
  233 mkIterAppNoAnn
  234     :: TermLike term tyname name uni fun
  235     => term () -- ^ @f@
  236     -> [term ()] -- ^ @[ x0 ... xn ]@
  237     -> term () -- ^ @[f x0 ... xn ]@
  238 mkIterAppNoAnn term = mkIterApp term . fmap ((),)
  239 
  240 -- | An infix synonym for `mkIterAppNoAnn`
  241 (@@) :: TermLike term tyname name uni fun
  242      => term () -- ^ @f@
  243      -> [term ()] -- ^ @[ x0 ... xn ]@
  244      -> term () -- ^ @[f x0 ... xn ]@
  245 (@@) = mkIterAppNoAnn

  247 -- | Make an iterated instantiation. Each `tyInst` node uses the annotation associated with
  248 -- the corresponding argument.
  249 mkIterInst
  250     :: TermLike term tyname name uni fun
  251     => term ann -- ^ @a@
  252     -> [(ann, Type tyname uni ann)] -- ^ @ [ x0 ... xn ] @
  253     -> term ann -- ^ @{ a x0 ... xn }@
  254 mkIterInst = foldl' $ \acc (ann, arg) -> tyInst ann acc arg

  256 -- | Make an iterated instantiation with no annotation.
  257 mkIterInstNoAnn
  258     :: TermLike term tyname name uni fun
  259     => term () -- ^ @a@
  260     -> [Type tyname uni ()] -- ^ @ [ x0 ... xn ] @
  261     -> term () -- ^ @{ a x0 ... xn }@
  262 mkIterInstNoAnn term = mkIterInst term . fmap ((),)

  264 -- | Lambda abstract a list of names.
  265 mkIterLamAbs
  266     :: TermLike term tyname name uni fun
  267     => [VarDecl tyname name uni ann]
  268     -> term ann
  269     -> term ann
  270 mkIterLamAbs args body =
  271     foldr (\(VarDecl ann name ty) acc -> lamAbs ann name ty acc) body args

  276     => [TyVarDecl tyname ann]
  277     -> term ann
  278     -> term ann
  279 mkIterTyAbs args body =
  280     foldr (\(TyVarDecl ann name kind) acc -> tyAbs ann name kind acc) body args

  282 -- | Make an iterated type application. Each `TyApp` node uses the annotation associated with

  286     -> [(ann, Type tyname uni ann)] -- ^ @[ x0 ... xn ]@
  287     -> Type tyname uni ann -- ^ @[ f x0 ... xn ]@
  288 mkIterTyApp = foldl' $ \acc (ann, arg) -> TyApp ann acc arg

  294     -> Type tyname uni () -- ^ @[ f x0 ... xn ]@

  297 -- | Make an iterated function type.
  298 mkIterTyFun
  299     :: ann
  300     -> [Type tyname uni ann]
  301     -> Type tyname uni ann
  302     -> Type tyname uni ann
  303 mkIterTyFun ann tys target = foldr (\ty acc -> TyFun ann ty acc) target tys
  304 
  305 -- | Universally quantify a list of names.
  306 mkIterTyForall
  307     :: [TyVarDecl tyname ann]
  308     -> Type tyname uni ann
  309     -> Type tyname uni ann
  310 mkIterTyForall args body =
  311     foldr (\(TyVarDecl ann name kind) acc -> TyForall ann name kind acc) body args

  314 mkIterTyLam
  315     :: [TyVarDecl tyname ann]
  316     -> Type tyname uni ann
  317     -> Type tyname uni ann
  318 mkIterTyLam args body =

  325     -> Kind ann

  329 {- | A helper to create a single, fresh strict binding; It returns the fresh bound `Var`iable and
  330 a function `Term -> Term`, expecting an "in-Term" to form a let-expression.
  331 -}
  332 mkFreshTermLet :: (MonadQuote m, TermLike t tyname Name uni fun, Monoid a)
  333                => Type tyname uni a -- ^ the type of binding
  334                -> t a -- ^ the term bound to the fresh variable
  335                -> m (t a, t a -> t a) -- ^ the fresh Var and a function that takes an "in" term to construct the Let
  336 mkFreshTermLet aT a = do
  337     -- I wish this was less constrained to Name
  338     genName <- freshName "generated"
  339     pure (var mempty genName, termLet mempty (Def (VarDecl mempty genName aT) a))
  340 
  341 -- | 'apply' the head of the application to the arguments iteratively.
  342 headSpineToTerm :: TermLike term tyname name uni fun => HeadSpine (term ()) -> term ()
  343 headSpineToTerm = foldl1 (apply ())
