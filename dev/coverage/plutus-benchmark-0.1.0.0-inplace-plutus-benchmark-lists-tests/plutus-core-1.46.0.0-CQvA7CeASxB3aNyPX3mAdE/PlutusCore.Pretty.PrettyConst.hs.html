   35 {- Note [Prettyprinting built-in constants]
   36 When we're printing PLC code, the prettyprinter has to render built-in constants. Unfortunately the
   37 instance of "Data.Text.Pretyprint.Doc.Pretty" for 'Text' does the wrong thing if control characters
   38 are involved. For example, the 'Text' "abc\nx\tyz" renders as
   39 
   40 abc
   41 x    yz
   42 
   43 which the PLC parser can't deal with. However, 'show' renders the string as "abc\nx\tyz" (including
   44 the quotes).
   45 
   46 This module provides a 'prettyConst' method which should be used whenever it's necessary to render a
   47 built-in constant: see for example "PlutusCore.Core.Instance.Pretty.Classic". The constraint @uni
   48 `Everywhere` PrettyConst@ occurs in many places in the codebase to make sure that we know how to
   49 print a constant from any type appearing in a universe of built-in types.
   50 
   51 Setting up our own machinery for overloading pretty-printing behavior would be laborious,
   52 but fortunately the @prettyprinter-configurable@ library already provides us with all the tools
   53 for doing that and so we define a dummy config for pretty-printing constants, implement a bunch of
   54 instances and derive pretty-printing behavior for non-polymorphic types (including how lists of
   55 such types are pretty-printed) via 'Show'. However always pretty-printing the spine of, say, a list
   56 via 'Show' while pretty-printing its contents via 'PrettyConst' is not something that can be easily
   57 done with the present-day @prettyprinter-configurable@, so we opt for pretty-printing the spine of
   58 a value of a compound type (list of lists, list of tuples, tuple of lists etc) via 'Pretty'.
   59 In practice this means that we have some additional spaces printed after punctuation symbols
   60 that 'show' alone would have omitted, for example:
   61 
   62 >>> let whateverList = ("abc\nx\tyz∀" :: Text, [((), False), ((), True)])
   63 >>> print $ prettyConst botRenderContext whateverList
   64 ("abc\nx\tyz\8704", [((), False), ((), True)])
   65 >>> putStrLn $ show whateverList
   66 ("abc\nx\tyz\8704",[((),False),((),True)])
   67 
   68 Not a big deal, since our parser isn't whitespace-sensitive.
   69 -}

   71 -- See Note [Prettyprinting built-in constants].
   72 -- | The type of configs used for pretty-printing constants. Has a 'RenderContext' inside, so that
   73 -- we don't add redundant parens to the output.
   74 newtype ConstConfig = ConstConfig
   75     { unConstConfig :: RenderContext
   76     }
   77 type instance HasPrettyDefaults ConstConfig = 'False
   78 
   79 instance HasRenderContext ConstConfig where
   80     renderContext = coerced
   81 
   82 type PrettyConst = PrettyBy ConstConfig
   83 
   84 -- | The set of constraints we need to be able to print built-in types and their values.
   85 type PrettyUni uni = (PrettyParens (SomeTypeIn uni), Closed uni, uni `Everywhere` PrettyConst)
   86 
   87 -- | The set of constraints we need to be able to throw exceptions with things with built-in types
   88 -- and functions in them.
   89 type ThrowableBuiltins uni fun = (PrettyUni uni, Pretty fun, Typeable uni, Typeable fun)
   90 
   91 -- These two can be generalized to any @config@, but that breaks some use cases of 'PrettyAny'
   92 -- then. Perhaps we should split the functionality and have two separate @newtype@ wrappers
   93 -- in @prettyprinter-configurable@ instead of a single 'PrettyAny'.
   94 -- For that we'll also need to ensure that it's alright when @HasPrettyDefaults config ~ 'True@.
   95 instance DefaultPrettyBy ConstConfig (PrettyAny a) => NonDefaultPrettyBy ConstConfig (PrettyAny a)
   96 instance DefaultPrettyBy ConstConfig (PrettyAny a) => PrettyBy ConstConfig (PrettyAny a) where
   97     prettyBy     = defaultPrettyBy
   98     prettyListBy = defaultPrettyListBy
   99 
  100 instance Show a => DefaultPrettyBy ConstConfig (PrettyAny a) where
  101     defaultPrettyBy     _ = pretty . show @a   . coerce
  102     defaultPrettyListBy _ = pretty . show @[a] . coerce
  103 
  104 prettyConst :: PrettyConst a => RenderContext -> a -> Doc ann
  105 prettyConst = prettyBy . ConstConfig
  106 
  107 -- This instance for String quotes control characters (which is what we want)
  108 -- but also Unicode characters (\8704 and so on).
  109 deriving via PrettyAny T.Text  instance NonDefaultPrettyBy ConstConfig T.Text
  110 deriving via PrettyAny ()      instance NonDefaultPrettyBy ConstConfig ()
  111 deriving via PrettyAny Bool    instance NonDefaultPrettyBy ConstConfig Bool
  112 deriving via PrettyAny Integer instance NonDefaultPrettyBy ConstConfig Integer
  113 
  114 -- | For rendering values without parens, i.e. in 'botRenderContext'.
  115 newtype NoParens a = NoParens
  116     { unNoParens :: a
  117     }
  118 
  119 instance PrettyConst a => PrettyBy ConstConfig (NoParens a) where
  120     prettyBy     config = prettyBy     @_ @a (config & renderContext .~ botRenderContext) . coerce
  121     prettyListBy config = prettyListBy @_ @a (config & renderContext .~ botRenderContext) . coerce
  122 
  123 instance PrettyConst a => NonDefaultPrettyBy ConstConfig [a] where
  124     nonDefaultPrettyBy config = defaultPrettyBy @_ @[NoParens a] config . coerce
  125 instance PrettyConst a => NonDefaultPrettyBy ConstConfig (Vector a) where
  126     nonDefaultPrettyBy config = defaultPrettyBy @_ @(Vector (NoParens a)) config . coerce
  127 instance (PrettyConst a, PrettyConst b) => NonDefaultPrettyBy ConstConfig (a, b) where
  128     nonDefaultPrettyBy config = defaultPrettyBy @_ @(NoParens a, NoParens b) config . coerce
  129 
  130 -- Special instance for bytestrings
  131 asBytes :: Word8 -> Doc ann
  132 asBytes x = Text 2 $ T.pack $ addLeadingZero $ showHex x mempty
  133     where addLeadingZero :: String -> String
  134           addLeadingZero
  135               | x < 16    = ('0' :)
  136               | otherwise = id
  137 
  138 toBytes :: BS.ByteString -> Doc ann
  139 toBytes  = foldMap asBytes . BS.unpack
  140 
  141 instance PrettyBy ConstConfig Data where
  142     prettyBy = inContextM $ \d0 -> iterAppDocM $ \_ prettyArg -> case d0 of
  143         Constr i ds ->  ("Constr" <+> prettyArg i) :| [prettyArg ds]
  144         Map ps      ->  "Map" :| [prettyArg ps]
  145         List ds     ->  "List" :| [prettyArg ds]
  146         I i         ->  ("I" <+> prettyArg i) :| []
  147         B b         ->  ("B" <+> prettyArg b) :| []
  148 
  149 instance PrettyBy ConstConfig BS.ByteString where
  150     prettyBy _ b = "#" <> toBytes b
  151 
  152 instance Pretty (SomeTypeIn uni) => Pretty (SomeTypeIn (Kinded uni)) where
  153     pretty (SomeTypeIn (Kinded uni)) = pretty (SomeTypeIn uni)
  154 
  155 -- See Note [Prettyprinting built-in constants].
  156 instance (Closed uni, uni `Everywhere` PrettyConst) => PrettyBy ConstConfig (ValueOf uni a) where
  157     prettyBy config (ValueOf uni x) = bring (Proxy @PrettyConst) uni $ prettyBy config x
  158 
  159 -- See Note [Prettyprinting built-in constants].
  160 instance (Closed uni, uni `Everywhere` PrettyConst) =>
  161         PrettyBy ConstConfig (Some (ValueOf uni)) where
  162     prettyBy config (Some s) = prettyBy config s
  163 
  164 -- See Note [Prettyprinting built-in constants].
  165 instance (Closed uni, uni `Everywhere` PrettyConst) => Pretty (ValueOf uni a) where
  166     pretty = prettyConst juxtRenderContext
  167 
  168 -- See Note [Prettyprinting built-in constants].
  169 instance (Closed uni, uni `Everywhere` PrettyConst) => Pretty (Some (ValueOf uni)) where
  170     pretty = prettyConst juxtRenderContext
