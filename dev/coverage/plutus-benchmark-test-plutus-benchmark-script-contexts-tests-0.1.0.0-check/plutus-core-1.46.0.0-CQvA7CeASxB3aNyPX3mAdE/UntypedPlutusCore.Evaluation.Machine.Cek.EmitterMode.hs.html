   29 -- | No emitter.
   30 noEmitter :: EmitterMode uni fun
   31 noEmitter = EmitterMode $ \_ -> pure $ CekEmitterInfo (\_ -> pure ()) (pure mempty)
   32 
   33 -- | Emits log only.
   34 logEmitter :: EmitterMode uni fun
   35 logEmitter = EmitterMode $ \_ -> do
   36   logsRef <- newSTRef DList.empty
   37   let emitter logs = CekM $ modifySTRef logsRef (`DList.append` logs)
   38   pure $ CekEmitterInfo emitter (DList.toList <$> readSTRef logsRef)
   39 
   40 -- A wrapper around encoding a record. `cassava` insists on including a trailing newline, which is
   41 -- annoying since we're recording the output line-by-line.
   42 encodeRecord :: (CSV.ToRecord a) => a -> T.Text
   43 encodeRecord a = T.stripEnd $ T.decodeUtf8 $ BSL.toStrict $ BS.toLazyByteString $ CSV.encodeRecord a
   44 
   45 -- | Emits log with timestamp.
   46 logWithTimeEmitter :: EmitterMode uni fun
   47 logWithTimeEmitter = EmitterMode $ \_ -> do
   48   logsRef <- newSTRef DList.empty
   49   let emitter logs = CekM $ do
   50         time <- unsafeIOToST getCurrentTime
   51         let secs = let MkFixed s = nominalDiffTimeToSeconds $ utcTimeToPOSIXSeconds time in s
   52         let withTime = logs <&> \str -> encodeRecord (str, secs)
   53         modifySTRef logsRef (`DList.append` withTime)
   54   pure $ CekEmitterInfo emitter (DList.toList <$> readSTRef logsRef)
   55 
   56 instance CSV.ToField ExCPU where

   59 instance CSV.ToField ExMemory where

   62 -- | Emits log with the budget.
   63 logWithBudgetEmitter :: EmitterMode uni fun
   64 logWithBudgetEmitter = EmitterMode $ \getBudget -> do
   65   logsRef <- newSTRef DList.empty
   66   let emitter logs = CekM $ do
   67         ExBudget exCpu exMemory <- getBudget
   68         let withBudget = logs <&> \str -> encodeRecord (str, exCpu, exMemory)
   69         modifySTRef logsRef (`DList.append` withBudget)
   70   pure $ CekEmitterInfo emitter (DList.toList <$> readSTRef logsRef)
