   19 class AnnotateCaseBuiltin uni where
   20     -- | Given a tag for a built-in type and a list of branches, annotate each of the branches with
   21     -- its expected argument types or fail if casing on values of the built-in type isn't supported.
   22     annotateCaseBuiltin
   23         :: UniOf term ~ uni
   24         => SomeTypeIn uni
   25         -> [term]
   26         -> Either Text [(term, [Type TyName uni ann])]
   27 
   28 class CaseBuiltin uni where
   29     -- | Given a constant with its type tag and a vector of branches, choose the appropriate branch
   30     -- or fail if the constant doesn't correspond to any of the branches (or casing on constants of
   31     -- this type isn't supported at all).
   32     caseBuiltin :: UniOf term ~ uni => Some (ValueOf uni) -> Vector term -> Either Text term
   33 
   34 -- See Note [DO NOT newtype-wrap functions].
   35 -- | A @data@ version of 'CaseBuiltin'. we parameterize the evaluator by a 'CaserBuiltin' so that
   36 -- the caller can choose whether to use the 'caseBuiltin' method or the always failing caser (the
   37 -- latter is required for earlier protocol versions when we didn't support casing on builtins).
   38 data CaserBuiltin uni = CaserBuiltin
   39     { unCaserBuiltin
   40         :: !(forall term. UniOf term ~ uni => Some (ValueOf uni) -> Vector term -> Either Text term)
   41     }
   42 
   43 instance NFData (CaserBuiltin uni) where
   44     rnf = rwhnf
   45 
   46 deriving via OnlyCheckWhnfNamed "PlutusCore.Builtin.Case.CaserBuiltin" (CaserBuiltin uni)
   47     instance NoThunks (CaserBuiltin uni)
   48 
   49 instance CaseBuiltin uni => Default (CaserBuiltin uni) where
   50     def = CaserBuiltin caseBuiltin
   51 
   52 unavailableCaserBuiltin :: Int -> CaserBuiltin uni
   53 unavailableCaserBuiltin ver =
   54     CaserBuiltin $ \_ _ -> Left $
   55         "'case' on values of built-in types is not supported in protocol version " <> display ver
