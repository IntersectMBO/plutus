  143 -- | A newtype wrapper around @a@ whose point is to provide a 'Show' instance
  144 -- for anything that has a 'Pretty' instance.
  145 newtype ShowPretty a = ShowPretty
  146     { unShowPretty :: a
  147     } deriving stock (Eq)
  148 
  149 instance Pretty a => Show (ShowPretty a) where
  150     show = display . unShowPretty

  152 instance (Pretty a, Pretty b) => Pretty (Either a b) where
  153     pretty (Left  x) = parens ("Left"  <+> pretty x)
  154     pretty (Right y) = parens ("Right" <+> pretty y)

  164 -- | Coerce the second argument to the result type of the first one. The motivation for this
  165 -- function is that it's often more annoying to explicitly specify a target type for 'coerce' than
  166 -- to construct an explicit coercion function, so this combinator can be used in cases like that.
  167 -- Plus the code reads better, as it becomes clear what and where gets wrapped/unwrapped.
  168 coerceVia :: Coercible a b => (a -> b) -> a -> b
  169 coerceVia _ = coerce
  170 {-# INLINE coerceVia #-}
  171 
  172 -- | Same as @\f -> f . coerce@, but does not create any closures and so is completely free.
  173 coerceArg :: Coercible a b => (a -> s) -> b -> s
  174 coerceArg = coerce
  175 {-# INLINE coerceArg #-}

  177 -- | Same as @\f -> coerce . f@, but does not create any closures and so is completely free.
  178 coerceRes :: Coercible s t => (a -> s) -> a -> t
  179 coerceRes = coerce
  180 {-# INLINE coerceRes #-}

  182 (<<$>>) :: (Functor f1, Functor f2) => (a -> b) -> f1 (f2 a) -> f1 (f2 b)
  183 (<<$>>) = fmap . fmap

  185 (<<*>>) :: (Applicative f1, Applicative f2) => f1 (f2 (a -> b)) -> f1 (f2 a) -> f1 (f2 b)
  186 (<<*>>) = liftA2 (<*>)

  188 -- | Makes an effectful function ignore its result value and return its input value.
  189 through :: Functor f => (a -> f b) -> (a -> f a)
  190 through f x = f x $> x

  192 mtraverse :: (Monad m, Traversable m, Applicative f) => (a -> f (m b)) -> m a -> f (m b)
  193 mtraverse f a = join <$> traverse f a

  195 -- | Fold a monadic function over a 'Foldable'. The monadic version of 'foldMap'.
  196 foldMapM :: (Foldable f, Monad m, Monoid b) => (a -> m b) -> f a -> m b
  197 foldMapM f xs = foldr step return xs mempty where
  198     step x r z = f x >>= \y -> r $! z `mappend` y

  200 -- | This function generalizes 'eitherToMaybe', 'eitherToList',
  201 -- 'listToMaybe' and other such functions.
  202 reoption :: (Foldable f, Alternative g) => f a -> g a
  203 reoption = foldr (const . pure) empty

  205 -- | Basically a @Data.Functor.Representable@ instance for 'Array'.
  206 -- We can't provide an actual instance because of the @Distributive@ superclass: @Array i@ is not
  207 -- @Distributive@ unless we assume that indices in an array range over the entirety of @i@.
  208 tabulateArray :: (Bounded i, Enum i, Ix i) => (i -> a) -> Array i a
  209 tabulateArray f = listArray (minBound, maxBound) $ map f enumerate

  218 -- | @b ? x@ is equal to @pure x@ whenever @b@ holds and is 'empty' otherwise.
  219 (?) :: Alternative f => Bool -> a -> f a
  220 (?) b x = x <$ guard b
  221 
  222 -- | @ensure p x@ is equal to @pure x@ whenever @p x@ holds and is 'empty' otherwise.
  223 ensure :: Alternative f => (a -> Bool) -> a -> f a
  224 ensure p x = p x ? x

  239 -- | Compose two folds to make them run in parallel. The results are concatenated.
  240 (<^>) :: Fold s a -> Fold s a -> Fold s a
  241 (f1 <^> f2) g s = f1 g s *> f2 g s

  243 -- | Zips two lists of the same length together, returning 'Nothing' if they are not
  244 -- the same length.
  245 zipExact :: [a] -> [b] -> Maybe [(a,b)]
  246 zipExact [] []         = Just []
  247 zipExact (a:as) (b:bs) = (:) (a, b) <$> zipExact as bs
  248 zipExact _ _           = Nothing

  250 -- | Similar to Maybe's `fromJust`. Returns the `Right` and errors out with the show instance
  251 -- of the `Left`.
  252 unsafeFromRight :: (Show e) => Either e a -> a
  253 unsafeFromRight (Right a) = a
  254 unsafeFromRight (Left e)  = error $ show e

  256 -- | function recursively applied N times
  257 timesA :: Natural -> (a -> a) -> a -> a

  260 -- | Pair each element of the given list with all the other elements.
  261 --
  262 -- >>> addTheRest "abcd"
  263 -- [('a',"bcd"),('b',"acd"),('c',"abd"),('d',"abc")]
  264 addTheRest :: [a] -> [(a, [a])]
  265 addTheRest []     = []
  266 addTheRest (x:xs) = (x, xs) : map (fmap (x :)) (addTheRest xs)

  268 allSame :: Eq a => [a] -> Bool
  269 allSame []     = True
  270 allSame (x:xs) = all (x ==) xs
  271 
  272 distinct :: Eq a => [a] -> Bool
  273 distinct = not . allSame
