   33 -- | A parsable PLC term.
   34 type PTerm = Term TyName Name DefaultUni DefaultFun SrcSpan
   35 
   36 varTerm :: Parser PTerm
   37 varTerm = withSpan $ \sp ->
   38     Var sp <$> name
   39 
   40 tyAbsTerm :: Parser PTerm
   41 tyAbsTerm = withSpan $ \sp ->
   42     inParens $ TyAbs sp <$> (symbol "abs" *> trailingWhitespace tyName)  <*> kind <*> term
   43 
   44 lamTerm :: Parser PTerm
   45 lamTerm = withSpan $ \sp ->
   46     inParens $ LamAbs sp <$> (symbol "lam" *> trailingWhitespace name) <*> pType <*> term
   47 
   48 appTerm :: Parser PTerm
   49 appTerm = withSpan $ \sp ->
   50     -- TODO: should not use the same `sp` for all arguments.
   51     inBrackets $ mkIterApp <$> term <*> (fmap (sp,) <$> some term)
   52 
   53 conTerm :: Parser PTerm
   54 conTerm = withSpan $ \sp ->
   55     inParens $ Constant sp <$> (symbol "con" *> constant)
   56 
   57 builtinTerm :: Parser PTerm
   58 builtinTerm = withSpan $ \sp ->
   59     inParens $ Builtin sp <$> (symbol "builtin" *> builtinFunction)
   60 
   61 tyInstTerm :: Parser PTerm
   62 tyInstTerm = withSpan $ \sp ->
   63     -- TODO: should not use the same `sp` for all arguments.
   64     inBraces $ mkIterInst <$> term <*> (fmap (sp,) <$> many pType)
   65 
   66 unwrapTerm :: Parser PTerm
   67 unwrapTerm = withSpan $ \sp ->
   68     inParens $ Unwrap sp <$> (symbol "unwrap" *> term)
   69 
   70 iwrapTerm :: Parser PTerm
   71 iwrapTerm = withSpan $ \sp ->
   72     inParens $ IWrap sp <$> (symbol "iwrap" *> pType) <*> pType <*> term
   73 
   74 errorTerm :: Parser PTerm
   75 errorTerm = withSpan $ \sp ->
   76     inParens $ Error sp <$> (symbol "error" *> pType)
   77 
   78 constrTerm :: Parser PTerm
   79 constrTerm = withSpan $ \sp ->
   80     inParens $ do
   81       res <- Constr sp <$> (symbol "constr" *> pType) <*> lexeme Lex.decimal <*> many term
   82       whenVersion (\v -> v < plcVersion110) $ fail "'constr' is not allowed before version 1.1.0"
   83       pure res
   84 
   85 caseTerm :: Parser PTerm
   86 caseTerm = withSpan $ \sp ->
   87     inParens $ do
   88       res <- Case sp <$> (symbol "case" *> pType) <*> term <*> many term
   89       whenVersion (\v -> v < plcVersion110) $ fail "'case' is not allowed before version 1.1.0"
   90       pure res
   91 
   92 -- | Parser for all PLC terms.
   93 term :: Parser PTerm
   94 term = leadingWhitespace go
   95   where
   96     go =
   97         choice $ map try
   98             [ tyAbsTerm
   99             , lamTerm
  100             , appTerm
  101             , conTerm
  102             , builtinTerm
  103             , tyInstTerm
  104             , unwrapTerm
  105             , iwrapTerm
  106             , errorTerm
  107             , varTerm
  108             , constrTerm
  109             , caseTerm
  110             ]

  112 -- | Parse a PLC program. The resulting program will have fresh names. The
  113 -- underlying monad must be capable of handling any parse errors.  This passes
  114 -- "test" to the parser as the name of the input stream; to supply a name
  115 -- explicity, use `parse program <name> <input>`.
  116 parseProgram ::
  117     (AsParserErrorBundle e, MonadError e m, MonadQuote m)
  118     => Text
  119     -> m (Program TyName Name DefaultUni DefaultFun SrcSpan)
  120 parseProgram = parseGen program
  121 
  122 -- | Parser for PLC programs.
  123 program :: Parser (Program TyName Name DefaultUni DefaultFun SrcSpan)
  124 program = leadingWhitespace go
  125   where
  126     go = do
  127         prog <- withSpan $ \sp -> inParens $ do
  128             v <- symbol "program" *> version
  129             withVersion v $ Program sp v <$> term
  130         notFollowedBy anySingle
  131         pure prog
  132 
  133 -- | Parse a PLC term. The resulting program will have fresh names. The underlying monad
  134 -- must be capable of handling any parse errors.
  135 parseTerm :: (AsParserErrorBundle e, MonadError e m, MonadQuote m) =>
  136     Text -> m (Term TyName Name DefaultUni DefaultFun SrcSpan)
  137 parseTerm = parseGen term
  138 
  139 -- | Parse a PLC type. The resulting program will have fresh names. The underlying monad
  140 -- must be capable of handling any parse errors.
  141 parseType :: (AsParserErrorBundle e, MonadError e m, MonadQuote m) =>
  142     Text -> m (Type TyName DefaultUni SrcSpan)
  143 parseType = parseGen pType
