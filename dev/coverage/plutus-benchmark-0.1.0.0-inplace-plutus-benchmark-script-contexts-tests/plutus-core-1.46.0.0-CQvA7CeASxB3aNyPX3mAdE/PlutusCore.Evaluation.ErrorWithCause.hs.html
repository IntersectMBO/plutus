   27 -- | An error and (optionally) what caused it.
   28 data ErrorWithCause err cause = ErrorWithCause
   29     { _ewcError :: !err
   30     , _ewcCause :: !(Maybe cause)
   31     } deriving stock (Eq, Functor, Foldable, Traversable, Generic)
   32       deriving anyclass (NFData)
   33 
   34 instance Bifunctor ErrorWithCause where
   35     bimap f g (ErrorWithCause err cause) = ErrorWithCause (f err) (g <$> cause)
   36     {-# INLINE bimap #-}
   37 
   38 instance AsEvaluationError err structural operational =>
   39         AsEvaluationError (ErrorWithCause err cause) structural operational where
   40     _EvaluationError = iso _ewcError (flip ErrorWithCause Nothing) . _EvaluationError
   41     {-# INLINE _EvaluationError #-}
   42 
   43 instance (Pretty err, Pretty cause) => Pretty (ErrorWithCause err cause) where
   44     pretty (ErrorWithCause e c) = pretty e <+> "caused by:" <+> pretty c
   45 
   46 instance (PrettyBy config cause, PrettyBy config err) =>
   47             PrettyBy config (ErrorWithCause err cause) where
   48     prettyBy config (ErrorWithCause err mayCause) = fold
   49         [ "An error has occurred:"
   50         , hardline
   51         , prettyBy config err
   52         , case mayCause of
   53             Nothing    -> mempty
   54             Just cause -> hardline <> "Caused by:" <+> prettyBy config cause
   55         ]
   56 
   57 instance (PrettyPlc cause, PrettyPlc err) =>
   58             Show (ErrorWithCause err cause) where
   59     show = render . prettyPlcReadable
   60 
   61 deriving anyclass instance (PrettyPlc cause, PrettyPlc err, Typeable cause, Typeable err) =>
   62     Exception (ErrorWithCause err cause)

   64 -- | "Prismatically" throw an error and its (optional) cause.
   65 throwingWithCause
   66     -- Binds @exc@ so it can be used as a convenient parameter with @TypeApplications@.
   67     :: forall exc e t term m x. (exc ~ ErrorWithCause e term, MonadError exc m)
   68     => AReview e t -> t -> term -> m x
   69 throwingWithCause l t cause = reviews l (\e -> throwError . ErrorWithCause e $ Just cause) t
   70 {-# INLINE throwingWithCause #-}
   71 
   72 -- | "Prismatically" throw a contentless error and its (optional) cause. 'throwingWithCause_' is to
   73 -- 'throwingWithCause' as 'throwing_' is to 'throwing'.
   74 throwingWithCause_
   75     -- Binds @exc@ so it can be used as a convenient parameter with @TypeApplications@.
   76     :: forall exc e term m x. (exc ~ ErrorWithCause e term, MonadError exc m)
   77     => AReview e () -> term -> m x
   78 throwingWithCause_ l = throwingWithCause l ()
   79 {-# INLINE throwingWithCause_ #-}
   80 
   81 -- | Attach a @cause@ to a 'BuiltinError' and throw that.
   82 -- Note that an evaluator might require the cause to be computed lazily for best performance on the
   83 -- happy path, hence this function must not force its first argument.
   84 -- TODO: wrap @cause@ in 'Lazy' once we have it.
   85 throwBuiltinErrorWithCause
   86     :: ( MonadError (ErrorWithCause err cause) m
   87        , AsUnliftingEvaluationError err, AsEvaluationFailure err
   88        )
   89     => cause -> BuiltinError -> m void
   90 throwBuiltinErrorWithCause cause = \case
   91     BuiltinUnliftingEvaluationError unlErr ->
   92         throwingWithCause _UnliftingEvaluationError unlErr cause
   93     BuiltinEvaluationFailure ->
   94         throwingWithCause_ _EvaluationFailure cause
   95 {-# INLINE throwBuiltinErrorWithCause #-}
