   36 -- | @List@ as a PLC type.
   37 --
   38 -- > fix \(list :: * -> *) (a :: *) -> all (r :: *). r -> (a -> list a -> r) -> r
   39 listData :: RecursiveType uni fun ()
   40 listData = runQuote $ do
   41     a    <- freshTyName "a"
   42     list <- freshTyName "list"
   43     r    <- freshTyName "r"
   44     let listA = TyApp () (TyVar () list) (TyVar () a)
   45     makeRecursiveType () list [TyVarDecl () a $ Type ()]
   46         . TyForall () r (Type ())
   47         . TyFun () (TyVar () r)
   48         . TyFun () (TyFun () (TyVar () a) . TyFun () listA $ TyVar () r)
   49         $ TyVar () r
   50 
   51 listTy :: Type TyName uni ()
   52 listTy = _recursiveType listData
   53 
   54 -- |  '[]' as a PLC term.
   55 --
   56 -- >  /\(a :: *) -> wrapList [a] /\(r :: *) -> \(z : r) (f : a -> list a -> r) -> z)
   57 nil :: TermLike term TyName Name uni fun => term ()
   58 nil = runQuote $ do
   59     let RecursiveType list wrapList = listData
   60     a <- freshTyName "a"
   61     r <- freshTyName "r"
   62     z <- freshName "z"
   63     f <- freshName "f"
   64     let listA = TyApp () list (TyVar () a)
   65     return
   66         . tyAbs () a (Type ())
   67         . wrapList [TyVar () a]
   68         . tyAbs () r (Type ())
   69         . lamAbs () z (TyVar () r)
   70         . lamAbs () f (TyFun () (TyVar () a) . TyFun () listA $ TyVar () r)
   71         $ var () z
   72 
   73 -- |  '(:)' as a PLC term.
   74 --
   75 -- > /\(a :: *) -> \(x : a) (xs : list a) ->
   76 -- >     wrapList [a] /\(r :: *) -> \(z : r) (f : a -> list a -> r) -> f x xs
   77 cons :: TermLike term TyName Name uni fun => term ()
   78 cons = runQuote $ do
   79     let RecursiveType list wrapList = listData
   80     a  <- freshTyName "a"
   81     x  <- freshName "x"
   82     xs <- freshName "xs"
   83     r  <- freshTyName "r"
   84     z  <- freshName "z"
   85     f  <- freshName "f"
   86     let listA = TyApp () list (TyVar () a)
   87     return
   88         . tyAbs () a (Type ())
   89         . lamAbs () x (TyVar () a)
   90         . lamAbs () xs listA
   91         . wrapList [TyVar () a]
   92         . tyAbs () r (Type ())
   93         . lamAbs () z (TyVar () r)
   94         . lamAbs () f (TyFun () (TyVar () a) . TyFun () listA $ TyVar () r)
   95         $ mkIterAppNoAnn (var () f)
   96           [ var () x
   97           , var () xs
   98           ]

  100 -- |  @foldrList@ as a PLC term.
  101 --
  102 -- > /\(a :: *) (r :: *) -> \(f : a -> r -> r) (z : r) ->
  103 -- >     fix {list a} {r} \(rec : list a -> r) (xs : list a) ->
  104 -- >         unwrap xs {r} z \(x : a) (xs' : list a) -> f x (rec xs')
  105 foldrList :: TermLike term TyName Name uni fun => term ()
  106 foldrList = runQuote $ do
  107     a   <- freshTyName "a"
  108     r   <- freshTyName "r"
  109     f   <- freshName "f"
  110     z   <- freshName "z"
  111     rec <- freshName "rec"
  112     xs  <- freshName "xs"
  113     x   <- freshName "x"
  114     xs' <- freshName "xs'"
  115     let listA = TyApp () listTy (TyVar () a)
  116     return
  117         . tyAbs () a (Type ())
  118         . tyAbs () r (Type ())
  119         . lamAbs () f (TyFun () (TyVar () a) . TyFun () (TyVar () r) $ TyVar () r)
  120         . lamAbs () z (TyVar () r)
  121         . apply () (mkIterInstNoAnn fix [listA, TyVar () r])
  122         . lamAbs () rec (TyFun () listA $ TyVar () r)
  123         . lamAbs () xs listA
  124         . apply () (apply () (tyInst () (unwrap () (var () xs)) $ TyVar () r) $ var () z)
  125         . lamAbs () x (TyVar () a)
  126         . lamAbs () xs' listA
  127         $ mkIterAppNoAnn (var () f)
  128           [ var () x
  129           , apply () (var () rec) $ var () xs'
  130           ]
  131 
  132 -- |  @map@ as a PLC term.
  133 --
  134 -- > /\(a :: *) (b :: *) -> \(f : a -> b) ->
  135 -- >     foldrList {a} {list b} (\(x : a) -> cons {b} (f x)) (nil {b})
  136 map :: TermLike term TyName Name uni fun => term ()
  137 map = runQuote $ do
  138     a <- freshTyName "a"
  139     b <- freshTyName "b"
  140     f <- freshName "f"
  141     x <- freshName "x"
  142     return
  143         . tyAbs () a (Type ())
  144         . tyAbs () b (Type ())
  145         . lamAbs () f (TyFun () (TyVar () a) $ TyVar () b)
  146         . mkIterAppNoAnn (mkIterInstNoAnn foldrList [TyVar () a, TyApp () listTy $ TyVar () b])
  147         $ [   lamAbs () x (TyVar () a)
  148             . apply () (tyInst () cons (TyVar () b))
  149             . apply () (var () f)
  150             $ var () x
  151           , tyInst () nil $ TyVar () b
  152           ]

  154 -- |  'foldl\'' as a PLC term.
  155 --
  156 -- > /\(a :: *) (r :: *) -> \(f : r -> a -> r) ->
  157 -- >     fix {r} {list a -> r} \(rec : r -> list a -> r) (z : r) (xs : list a) ->
  158 -- >         unwrap xs {r} z \(x : a) (xs' : list a) -> rec (f z x) xs'
  159 foldList :: TermLike term TyName Name uni fun => term ()
  160 foldList = runQuote $ do
  161     a   <- freshTyName "a"
  162     r   <- freshTyName "r"
  163     f   <- freshName "f"
  164     rec <- freshName "rec"
  165     z   <- freshName "z"
  166     xs  <- freshName "xs"
  167     x   <- freshName "x"
  168     xs' <- freshName "xs'"
  169     let listA = TyApp () listTy (TyVar () a)
  170     return
  171         . tyAbs () a (Type ())
  172         . tyAbs () r (Type ())
  173         . lamAbs () f (TyFun () (TyVar () r) . TyFun () (TyVar () a) $ TyVar () r)
  174         . apply () (mkIterInstNoAnn fix [TyVar () r, TyFun () listA $ TyVar () r])
  175         . lamAbs () rec (TyFun () (TyVar () r) . TyFun () listA $ TyVar () r)
  176         . lamAbs () z (TyVar () r)
  177         . lamAbs () xs listA
  178         . apply () (apply () (tyInst () (unwrap () (var () xs)) $ TyVar () r) $ var () z)
  179         . lamAbs () x (TyVar () a)
  180         . lamAbs () xs' listA
  181         . mkIterAppNoAnn (var () rec)
  182         $ [ mkIterAppNoAnn (var () f) [var () z, var () x]
  183           , var () xs'
  184           ]

  186 -- |  'reverse' as a PLC term.
  187 --
  188 -- > /\(a :: *) -> \(xs : list a) ->
  189 -- >     foldList {a} {list a} (\(r : list a) (x : a) -> cons {a} x r) (nil {a})
  190 reverse :: TermLike term TyName Name uni fun => term ()
  191 reverse = runQuote $ do
  192     a   <- freshTyName "a"
  193     xs  <- freshName "xs"
  194     x   <- freshName "x"
  195     r   <- freshName "r"
  196     let vA = TyVar () a
  197         listA = TyApp () listTy vA
  198     return
  199         . tyAbs () a (Type ())
  200         . lamAbs () xs listA
  201         $ mkIterAppNoAnn (mkIterInstNoAnn foldList [vA, listA])
  202             [ lamAbs () r listA . lamAbs () x vA $
  203                 mkIterAppNoAnn (tyInst () cons vA) [var () x, var () r]
  204             , tyInst () nil vA
  205             , var () xs
  206             ]

  208 -- | 'enumFromTo' as a PLC term
  209 --
  210 -- > \(n m : integer) ->
  211 -- >     fix {integer} {list (integer)}
  212 -- >         (\(rec : integer -> list (integer)) (n' : integer) ->
  213 -- >             ifThenElse {list (integer)}
  214 -- >                 (lessThanEqualsInteger n' m)
  215 -- >                 (cons {integer} n' (rec (succInteger n')))
  216 -- >                 (nil {integer}))
  217 -- >         n
  218 enumFromTo
  219     :: ( TermLike term TyName Name uni DefaultFun
  220        , uni `HasTypeAndTermLevel` Integer
  221        , uni `HasTypeAndTermLevel` ()
  222        , uni `HasTypeAndTermLevel` Bool
  223        )
  224     => term ()
  225 enumFromTo = runQuote $ do
  226     n   <- freshName "n"
  227     m   <- freshName "m"
  228     rec <- freshName "rec"
  229     n'  <- freshName "n'"
  230     u   <- freshName "u"
  231     let leqInteger  = builtin () LessThanEqualsInteger
  232         int = mkTyBuiltin @_ @Integer ()
  233         listInt = TyApp () listTy int
  234     return
  235         . lamAbs () n int
  236         . lamAbs () m int
  237         . mkIterAppNoAnn (mkIterInstNoAnn fix [int, listInt])
  238         $ [   lamAbs () rec (TyFun () int listInt)
  239             . lamAbs () n' int
  240             . mkIterAppNoAnn (tyInst () ifThenElse listInt)
  241             $ [ mkIterAppNoAnn leqInteger [ var () n' , var () m]
  242               , lamAbs () u unit $ mkIterAppNoAnn (tyInst () cons int)
  243                     [ var () n'
  244                     ,    apply () (var () rec)
  245                        . apply () succInteger
  246                        $ var () n'
  247                     ]
  248               , lamAbs () u unit $ tyInst () nil int
  249               ]
  250           , var () n
  251           ]

  253 -- |  'sum' as a PLC term.
  254 --
  255 -- > foldList {integer} {integer} addInteger 0
  256 sum :: (TermLike term TyName Name uni DefaultFun, uni `HasTypeAndTermLevel` Integer) => term ()
  257 sum = runQuote $ do
  258     let int = mkTyBuiltin @_ @Integer ()
  259         add = builtin () AddInteger
  260     return
  261         . mkIterAppNoAnn (mkIterInstNoAnn foldList [int, int])
  262         $ [ add , mkConstant @Integer () 0]

  268         add = builtin () AddInteger

  270         . mkIterAppNoAnn (mkIterInstNoAnn foldrList [int, int])
