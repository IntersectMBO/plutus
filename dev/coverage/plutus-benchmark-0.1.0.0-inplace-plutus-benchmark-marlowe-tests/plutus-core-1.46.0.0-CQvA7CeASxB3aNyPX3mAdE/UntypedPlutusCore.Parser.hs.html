   37 -- Parsers for UPLC terms
   38 
   39 -- | A parsable UPLC term.
   40 type PTerm = UPLC.Term PLC.Name PLC.DefaultUni PLC.DefaultFun SrcSpan
   41 
   42 conTerm :: Parser PTerm
   43 conTerm = withSpan $ \sp ->
   44     inParens $ UPLC.Constant sp <$> (symbol "con" *> constant)
   45 
   46 builtinTerm :: Parser PTerm
   47 builtinTerm = withSpan $ \sp ->
   48     inParens $ UPLC.Builtin sp <$> (symbol "builtin" *> builtinFunction)
   49 
   50 varTerm :: Parser PTerm
   51 varTerm = withSpan $ \sp ->
   52     UPLC.Var sp <$> name
   53 
   54 lamTerm :: Parser PTerm
   55 lamTerm = withSpan $ \sp ->
   56     inParens $ UPLC.LamAbs sp <$> (symbol "lam" *> trailingWhitespace name) <*> term
   57 
   58 appTerm :: Parser PTerm
   59 appTerm = withSpan $ \sp ->
   60     -- TODO: should not use the same `sp` for all arguments.
   61     inBrackets $ mkIterApp <$> term <*> (fmap (sp,) <$> some term)
   62 
   63 delayTerm :: Parser PTerm
   64 delayTerm = withSpan $ \sp ->
   65     inParens $ UPLC.Delay sp <$> (symbol "delay" *> term)
   66 
   67 forceTerm :: Parser PTerm
   68 forceTerm = withSpan $ \sp ->
   69     inParens $ UPLC.Force sp <$> (symbol "force" *> term)
   70 
   71 errorTerm :: Parser PTerm
   72 errorTerm = withSpan $ \sp ->
   73     inParens $ UPLC.Error sp <$ symbol "error"
   74 
   75 constrTerm :: Parser PTerm
   76 constrTerm = withSpan $ \sp ->
   77     inParens $ do
   78       res <- UPLC.Constr sp <$> (symbol "constr" *> lexeme Lex.decimal) <*> many term
   79       whenVersion (\v -> v < plcVersion110) $ fail "'constr' is not allowed before version 1.1.0"
   80       pure res
   81 
   82 caseTerm :: Parser PTerm
   83 caseTerm = withSpan $ \sp ->
   84     inParens $ do
   85       res <- UPLC.Case sp <$> (symbol "case" *> term) <*> (V.fromList <$> many term)
   86       whenVersion (\v -> v < plcVersion110) $ fail "'case' is not allowed before version 1.1.0"
   87       pure res
   88 
   89 -- | Parser for all UPLC terms.
   90 term :: Parser PTerm
   91 term = leadingWhitespace go
   92   where
   93     go =
   94         choice $ map try [
   95             conTerm
   96             , builtinTerm
   97             , varTerm
   98             , lamTerm
   99             , appTerm
  100             , delayTerm
  101             , forceTerm
  102             , errorTerm
  103             , constrTerm
  104             , caseTerm
  105             ]
  106 
  107 -- | Parser for UPLC programs.
  108 program :: Parser (UPLC.Program PLC.Name PLC.DefaultUni PLC.DefaultFun SrcSpan)
  109 program = leadingWhitespace go
  110   where
  111     go = do
  112         prog <- withSpan $ \sp -> inParens $ do
  113             v <- symbol "program" *> version
  114             withVersion v $ UPLC.Program sp v <$> term
  115         notFollowedBy anySingle
  116         pure prog
  117 
  118 -- | Parse a UPLC term. The resulting program will have fresh names. The underlying monad must be capable
  119 -- of handling any parse errors.
  120 parseTerm :: (AsParserErrorBundle e, MonadError e m, PLC.MonadQuote m) => Text -> m PTerm
  121 parseTerm = parseGen term
  122 
  123 -- | Parse a UPLC program. The resulting program will have fresh names. The
  124 -- underlying monad must be capable of handling any parse errors.  This passes
  125 -- "test" to the parser as the name of the input stream; to supply a name
  126 -- explicity, use `parse program <name> <input>`.`
  127 parseProgram ::
  128     (AsParserErrorBundle e, MonadError e m, PLC.MonadQuote m)
  129     => Text
  130     -> m (UPLC.Program PLC.Name PLC.DefaultUni PLC.DefaultFun SrcSpan)
  131 parseProgram = parseGen program
  132 
  133 -- | Parse and rewrite so that names are globally unique, not just unique within
  134 -- their scope.
  135 parseScoped ::
  136     (AsParserErrorBundle e, PLC.AsUniqueError e SrcSpan, MonadError e m, PLC.MonadQuote m)
  137     => Text
  138     -> m (UPLC.Program PLC.Name PLC.DefaultUni PLC.DefaultFun SrcSpan)
  139 -- don't require there to be no free variables at this point, we might be parsing an open term
  140 parseScoped = through (checkProgram (const True)) <=< rename <=< parseProgram
