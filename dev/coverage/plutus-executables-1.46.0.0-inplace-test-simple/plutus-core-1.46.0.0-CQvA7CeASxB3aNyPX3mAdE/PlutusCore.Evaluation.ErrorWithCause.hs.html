   26 -- | An error and (optionally) what caused it.
   27 data ErrorWithCause err cause = ErrorWithCause
   28     { _ewcError :: !err
   29     , _ewcCause :: !(Maybe cause)
   30     } deriving stock (Eq, Functor, Foldable, Traversable, Generic)
   31       deriving anyclass (NFData)
   32 
   33 instance Bifunctor ErrorWithCause where
   34     bimap f g (ErrorWithCause err cause) = ErrorWithCause (f err) (g <$> cause)
   35     {-# INLINE bimap #-}
   36 
   37 instance AsEvaluationFailure err => AsEvaluationFailure (ErrorWithCause err cause) where
   38     _EvaluationFailure = iso _ewcError (flip ErrorWithCause Nothing) . _EvaluationFailure
   39     {-# INLINE _EvaluationFailure #-}
   40 
   41 instance (Pretty err, Pretty cause) => Pretty (ErrorWithCause err cause) where
   42     pretty (ErrorWithCause e c) = pretty e <+> "caused by:" <+> pretty c
   43 
   44 instance (PrettyBy config cause, PrettyBy config err) =>
   45             PrettyBy config (ErrorWithCause err cause) where
   46     prettyBy config (ErrorWithCause err mayCause) = fold
   47         [ "An error has occurred:"
   48         , hardline
   49         , prettyBy config err
   50         , case mayCause of
   51             Nothing    -> mempty
   52             Just cause -> hardline <> "Caused by:" <+> prettyBy config cause
   53         ]
   54 
   55 instance (PrettyPlc cause, PrettyPlc err) =>
   56             Show (ErrorWithCause err cause) where
   57     show = render . prettyPlcReadable
   58 
   59 deriving anyclass instance (PrettyPlc cause, PrettyPlc err, Typeable cause, Typeable err) =>
   60     Exception (ErrorWithCause err cause)

   62 -- | "Prismatically" throw an error and its (optional) cause.
   63 throwingWithCause
   64     -- Binds @exc@ so it can be used as a convenient parameter with @TypeApplications@.
   65     :: forall exc e t term m x. (exc ~ ErrorWithCause e term, MonadError exc m)
   66     => AReview e t -> t -> Maybe term -> m x
   67 throwingWithCause l t cause = reviews l (\e -> throwError $ ErrorWithCause e cause) t
   68 {-# INLINE throwingWithCause #-}
   69 
   70 -- | "Prismatically" throw a contentless error and its (optional) cause. 'throwingWithCause_' is to
   71 -- 'throwingWithCause' as 'throwing_' is to 'throwing'.
   72 throwingWithCause_
   73     -- Binds @exc@ so it can be used as a convenient parameter with @TypeApplications@.
   74     :: forall exc e term m x. (exc ~ ErrorWithCause e term, MonadError exc m)
   75     => AReview e () -> Maybe term -> m x
   76 throwingWithCause_ l = throwingWithCause l ()
   77 {-# INLINE throwingWithCause_ #-}
   78 
   79 -- | Attach a @cause@ to a 'BuiltinError' and throw that.
   80 -- Note that an evaluator might require the cause to be computed lazily for best performance on the
   81 -- happy path, hence this function must not force its first argument.
   82 -- TODO: wrap @cause@ in 'Lazy' once we have it.
   83 throwBuiltinErrorWithCause
   84     :: ( MonadError (ErrorWithCause err cause) m
   85        , AsUnliftingEvaluationError err, AsEvaluationFailure err
   86        )
   87     => cause -> BuiltinError -> m void
   88 throwBuiltinErrorWithCause cause = \case
   89     BuiltinUnliftingEvaluationError unlErr ->
   90         throwingWithCause _UnliftingEvaluationError unlErr $ Just cause
   91     BuiltinEvaluationFailure ->
   92         throwingWithCause_ _EvaluationFailure $ Just cause
   93 {-# INLINE throwBuiltinErrorWithCause #-}
