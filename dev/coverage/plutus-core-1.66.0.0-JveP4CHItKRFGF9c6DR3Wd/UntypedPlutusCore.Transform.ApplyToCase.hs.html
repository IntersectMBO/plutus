    3 {-|
    4 Transform multi-argument applications into case-constr form.
    5 
    6 This is essentially the reverse of 'CaseReduce'. An application of 3 or more
    7 arguments, e.g., @f x y z@ is rewritten to @case (constr 0 [x, y, z]) [f]@,
    8 which is cheaper, and smaller in AST size, though it is sometimes slightly
    9 bigger in cbor size.
   10 
   11 I cannot think of any case where this pass could enable further optimizations.
   12 It can certainly destroy optimization opportunities, similar to CSE.
   13 So it should only be run once, at the very end of the pipeline. CSE is also
   14 run at the end, but it runs for multiple iterations interleaved with other
   15 optimizations, because CSE can both destroy and enable further optimizations. -}
   16 module UntypedPlutusCore.Transform.ApplyToCase (applyToCase) where
   17 
   18 import Control.Lens (over)
   19 import Data.Vector qualified as V
   20 import PlutusPrelude (getAnn)
   21 import UntypedPlutusCore.Core
   22 import UntypedPlutusCore.Transform.Optimizer
   23   ( OptimizerT
   24   , recordOptimization
   25   , pattern ApplyToCaseStage
   26   )
   27 
   28 minArgs :: Int
   29 minArgs = 3
   30 {-# INLINE minArgs #-}
   31 
   32 applyToCase
   33   :: Monad m
   34   => Term name uni fun a
   35   -> OptimizerT name uni fun a m (Term name uni fun a)
   36 applyToCase term = do
   37   let result = processTerm term
   38   recordOptimization term ApplyToCaseStage result
   39   pure result
   40 
   41 processTerm :: Term name uni fun a -> Term name uni fun a
   42 processTerm t = case splitApplication t of
   43   (fun, args)
   44     | length args >= minArgs ->
   45         let ann = getAnn t
   46          in Case ann (Constr ann 0 (processTerm . snd <$> args)) (V.singleton (processTerm fun))
   47   _ -> over termSubterms processTerm t
