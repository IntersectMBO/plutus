   13 {-| Datatype which represents optimization passes which are also
   14 certified.
   15 
   16 This means that these passes are formalized as part of the certifier,
   17 and adding a new pass constructor to this type means that it is expected
   18 the pass will be also certified in the same PR.
   19 
   20 WARNING: the order of the constructors MUST be the same as the order
   21 of their counterparts in 'VerifiedCompilation.Trace'. -}
   22 data CertifiedOptStage
   23   = FloatDelay
   24   | ForceDelay
   25   | ForceCaseDelay
   26   | Inline
   27   | CSE
   28   | ApplyToCase
   29   | CaseReduce
   30   | LetFloatOut
   31   deriving stock (Show, Generic)
   32   deriving anyclass (NFData)
   33 
   34 {-| Datatype which represents optimization passes which are not yet
   35 certified.
   36 
   37 WARNING: the order of the constructors MUST be the same as the order
   38 of their counterparts in 'VerifiedCompilation.Trace'.
   39 
   40 IMPORTANT: if you add a new pass, or modify an existing pass, without
   41 also modifying the certifier in the same PR, you must add/move its
   42 corresponding constructor to this type. Please also open an issue
   43 at https://github.com/IntersectMBO/plutus/issues. -}
   44 data UncertifiedOptStage
   45   = CaseOfCase
   46   | ConstantFolding
   47   | PolyBuiltin
   48   deriving stock (Show, Generic)
   49   deriving anyclass (NFData)
   50 
   51 type OptStage = Either UncertifiedOptStage CertifiedOptStage
   52 
   53 pattern FloatDelayStage :: OptStage
   54 pattern FloatDelayStage = Right FloatDelay
   55 
   56 pattern ForceDelayStage :: OptStage
   57 pattern ForceDelayStage = Right ForceDelay
   58 
   59 pattern ForceCaseDelayStage :: OptStage
   60 pattern ForceCaseDelayStage = Right ForceCaseDelay
   61 
   62 pattern CaseReduceStage :: OptStage
   63 pattern CaseReduceStage = Right CaseReduce
   64 
   65 pattern InlineStage :: OptStage
   66 pattern InlineStage = Right Inline
   67 
   68 pattern CseStage :: OptStage
   69 pattern CseStage = Right CSE
   70 
   71 pattern ApplyToCaseStage :: OptStage
   72 pattern ApplyToCaseStage = Right ApplyToCase
   73 
   74 pattern CaseOfCaseStage :: OptStage
   75 pattern CaseOfCaseStage = Left CaseOfCase
   76 
   77 pattern LetFloatOutStage :: OptStage
   78 pattern LetFloatOutStage = Right LetFloatOut
   79 
   80 pattern ConstantFoldingStage :: OptStage
   81 pattern ConstantFoldingStage = Left ConstantFolding
   82 
   83 pattern PolyBuiltinStage :: OptStage
   84 pattern PolyBuiltinStage = Left PolyBuiltin
   85

  100 data Optimization name uni fun a
  101   = Optimization
  102   { beforeAST :: Term name uni fun a
  103   , stage :: OptStage
  104   , hints :: Certify.Hints
  105   , afterAST :: Term name uni fun a
  106   }

  108 -- TODO2: we probably don't want this in memory so after MVP
  109 -- we should consider serializing this to disk
  110 newtype OptimizerTrace name uni fun a
  111   = OptimizerTrace
  112   { optimizerTrace
  113       :: [Optimization name uni fun a]
  114   }
  115 
  116 initOptimizerTrace :: OptimizerTrace name uni fun a
  117 initOptimizerTrace = OptimizerTrace []
  118 
  119 allASTs :: OptimizerTrace name uni fun a -> [Term name uni fun a]
  120 allASTs = \case
  121   OptimizerTrace [] -> []
  122   OptimizerTrace xs@(x : _) ->
  123     -- `OptimizerTrace` is in reverse order: the first item is the last pass run.
  124     afterAST x : map beforeAST xs
