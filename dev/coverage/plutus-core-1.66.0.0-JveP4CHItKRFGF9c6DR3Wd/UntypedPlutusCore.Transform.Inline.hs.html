  205 {-| Inline simple bindings. Relies on global uniqueness, and preserves it.
  206 See Note [Inlining and global uniqueness] -}
  207 inline
  208   :: forall name uni fun m a
  209    . ExternalConstraints name uni fun m
  210   => AstSize
  211   -- ^ unconditional threshold
  212   -> AstSize
  213   -- ^ callsite threshold
  214   -> Bool
  215   -- ^ inline constants
  216   -> Bool
  217   -- ^ preserve logging
  218   -> InlineHints name a
  219   -> PLC.BuiltinSemanticsVariant fun
  220   -> Term name uni fun a
  221   -> OptimizerT name uni fun a m (Term name uni fun a)
  222 inline
  223   unconditionalGrowth
  224   callsiteGrowth
  225   inlineConstants
  226   preserveLogging
  227   hints
  228   builtinSemanticsVariant
  229   t = do
  230     decoratedResult <-
  231       liftQuote $
  232         flip evalStateT mempty $
  233           runReaderT
  234             (processTerm (emptyDecoration <$> t))
  235             InlineInfo
  236               { _iiUsages = Usages.termUsages t
  237               , _iiHints = hints
  238               , _iiBuiltinSemanticsVariant = builtinSemanticsVariant
  239               , _iiInlineConstants = inlineConstants
  240               , _iiInlineUnconditionalGrowth = unconditionalGrowth
  241               , _iiInlineCallsiteGrowth = callsiteGrowth
  242               , _iiPreserveLogging = preserveLogging
  243               }
  244     let result = snd <$> decoratedResult
  245     recordOptimizationWithHints
  246       (CertifierHints.Inline (mkHints decoratedResult))
  247       t
  248       InlineStage
  249       result
  250     return result

  254 {-| Extract the list of applications from a term,
  255 a bit like a "multi-beta" reduction.
  256 
  257 Some examples will help (annotations are omitted):
  258 [(\x . t) a] -> Just ([(x, a)], t)
  259 
  260 [[[(\x . (\y . (\z . t))) a] b] c] -> Just ([(x, a), (y, b), (z, c)]) t)
  261 
  262 [[(\x . t) a] b] -> Nothing -}
  263 extractApps
  264   :: Term name uni fun a
  265   -> Maybe ([(UTermDef name uni fun a, a)], Term name uni fun a)
  266 extractApps = go []
  267   where
  268     go argStack (Apply aa f arg) = go ((arg, aa) : argStack) f
  269     go argStack t = matchArgs argStack [] t
  270     matchArgs ((arg, aa) : rest) acc (LamAbs a n body) =
  271       matchArgs rest ((Def (UVarDecl a n) arg, aa) : acc) body
  272     matchArgs [] acc t =
  273       if null acc then Nothing else Just (reverse acc, t)
  274     matchArgs (_ : _) _ _ = Nothing
  275 
  276 -- | The inverse of 'extractApps'.
  277 restoreApps
  278   :: [(Either (Ann a) (UTermDef name uni fun (Ann a)), Ann a)]
  279   -> Term name uni fun (Ann a)
  280   -> Term name uni fun (Ann a)
  281 restoreApps defs t = makeLams [] t (reverse defs)
  282   where
  283     -- `aa` is the annotation on the Apply node, and `al` is the annotation
  284     -- on the LamAbs node.
  285     makeLams args acc ((Left al, aa) : rest) =
  286       -- `Left` means the binding is dropped, so `acc` should be decorated with `al`.
  287       -- This corresponds to Note [Inliner's Certifier Hints], #3.
  288       makeLams ((Nothing, aa) : args) (decorateWith (al ^. decorations) acc) rest
  289     makeLams args acc ((Right (Def (UVarDecl al n) rhs), aa) : rest) =
  290       makeLams ((Just rhs, aa) : args) (LamAbs al n acc) rest
  291     makeLams args acc [] =
  292       makeApps args acc
  293 
  294     makeApps ((Nothing, aa) : args) acc =
  295       -- `Nothing` means the binding is dropped, so we decorate `acc with `aa` plus `DDrop`.
  296       -- This corresponds to Note [Inliner's Certifier Hints], #2.
  297       makeApps args (decorateWith (aa ^. decorations ++ [DDrop]) acc)
  298     makeApps ((Just arg, aa) : args) acc =
  299       makeApps args (Apply aa acc arg)
  300     makeApps [] acc = acc

  302 -- | Run the inliner on a `UntypedPlutusCore.Core.Type.Term`.
  303 processTerm
  304   :: forall name uni fun a
  305    . InliningConstraints name uni fun
  306   => Term name uni fun (Ann a)
  307   -> InlineM name uni fun a (Term name uni fun (Ann a))
  308 processTerm = handleTerm
  309   where
  310     handleTerm
  311       :: Term name uni fun (Ann a)
  312       -> InlineM name uni fun a (Term name uni fun (Ann a))
  313     handleTerm = \case
  314       v@(Var a n) ->
  315         maybe
  316           v
  317           -- See Note [Inliner's Certifier Hints], #1
  318           (decorateWith (a ^. decorations) . decorateWith [DExpand])
  319           <$> substName n
  320       -- See Note [Differences from PIR inliner] 3
  321       (extractApps -> Just (bs, t)) -> do
  322         ebs <- traverse (processSingleBinding t) bs
  323         t' <- processTerm t
  324         pure $ restoreApps ebs t'
  325       t -> inlineSaturatedApp =<< forMOf termSubterms t processTerm
  326 
  327     -- See Note [Renaming strategy]
  328     substName :: name -> InlineM name uni fun a (Maybe (Term name uni fun (Ann a)))
  329     substName name = gets (lookupTerm name) >>= traverse renameTerm
  330 
  331     -- See Note [Inlining approach and 'Secrets of the GHC Inliner']
  332     renameTerm
  333       :: InlineTerm name uni fun (Ann a)
  334       -> InlineM name uni fun a (Term name uni fun (Ann a))
  335     renameTerm = \case
  336       -- Already processed term, just rename and put it in, don't do any
  337       -- further optimization here.
  338       Done t -> liftDupable t
  339 
  340 processSingleBinding
  341   :: InliningConstraints name uni fun
  342   => Term name uni fun (Ann a)
  343   -> (UTermDef name uni fun (Ann a), Ann a)
  344   -> InlineM name uni fun a (Either (Ann a) (UTermDef name uni fun (Ann a)), Ann a)
  345 processSingleBinding body (Def vd@(UVarDecl a n) rhs0, aa) =
  346   (,aa) <$> do
  347     maybeAddSubst body (snd a) n rhs0 >>= \case
  348       Just rhs -> do
  349         let (binders, rhsBody) = UPLC.splitParams rhs
  350         modify' . extendVarInfo n $
  351           VarInfo
  352             { _varBinders = binders
  353             , _varRhs = rhs
  354             , _varRhsBody = Done (dupable rhsBody)
  355             }
  356         pure . Right $ Def vd rhs
  357       Nothing -> pure (Left a)

  359 {-| Check against the heuristics we have for inlining and either inline
  360 the term binding or not. The arguments to this function are the fields of the
  361 'TermBinding' being processed.
  362 Nothing means that we are inlining the term:
  363   * we have extended the substitution, and
  364   * we are removing the binding (hence we return Nothing). -}
  365 maybeAddSubst
  366   :: forall name uni fun a
  367    . InliningConstraints name uni fun
  368   => Term name uni fun (Ann a)
  369   -> a
  370   -> name
  371   -> Term name uni fun (Ann a)
  372   -> InlineM name uni fun a (Maybe (Term name uni fun (Ann a)))
  373 maybeAddSubst body a n rhs0 = do
  374   rhs <- processTerm rhs0
  375 
  376   -- Check whether we've been told specifically to inline this
  377   hints <- view iiHints
  378   case shouldInline hints a n of
  379     AlwaysInline ->
  380       -- if we've been told specifically, then do it right away
  381       extendAndDrop (Done $ dupable rhs)
  382     hint ->
  383       let safeToInline = hint == SafeToInline
  384        in ifM
  385             (shouldUnconditionallyInline safeToInline n rhs body)
  386             (extendAndDrop (Done $ dupable rhs))
  387             (pure $ Just rhs)
  388   where
  389     extendAndDrop
  390       :: forall b
  391        . InlineTerm name uni fun (Ann a)
  392       -> InlineM name uni fun a (Maybe b)
  393     extendAndDrop t = modify' (extendTerm n t) >> pure Nothing
  394 
  395 shouldUnconditionallyInline
  396   :: InliningConstraints name uni fun
  397   => Bool
  398   {-^ Whether we know that the binding is safe to inline.
  399   If so, bypass the purity check. -}
  400   -> name
  401   -> Term name uni fun a
  402   -> Term name uni fun b
  403   -> InlineM name uni fun c Bool
  404 shouldUnconditionallyInline safe n rhs body = do
  405   isTermPure <- checkPurity rhs
  406   inlineConstants <- view iiInlineConstants
  407   preUnconditional isTermPure ||^ postUnconditional inlineConstants isTermPure
  408   where
  409     -- similar to the paper, preUnconditional inlining checks that the binder
  410     -- is 'OnceSafe'.  I.e., it's used at most once AND it neither duplicate code
  411     -- or work. While we don't check for lambda etc like in the paper,
  412     -- 'effectSafe' ensures that it isn't doing any substantial work.
  413     -- We actually also inline 'Dead' binders (i.e., remove dead code) here.
  414     preUnconditional isTermPure =
  415       nameUsedAtMostOnce n &&^ (pure safe ||^ effectSafe body n isTermPure)
  416     -- See Note [Inlining approach and 'Secrets of the GHC Inliner'] and
  417     -- [Inlining and purity]. This is the case where we don't know that the number
  418     -- of occurrences is exactly one, so there's no point checking if the term is
  419     -- immediately evaluated.
  420     postUnconditional inlineConstants isTermPure =
  421       pure (safe || isTermPure) &&^ acceptable inlineConstants rhs
  422 
  423 -- | Check if term is pure. See Note [Inlining and purity]
  424 checkPurity
  425   :: PLC.ToBuiltinMeaning uni fun
  426   => Term name uni fun a
  427   -> InlineM name uni fun b Bool
  428 checkPurity t = do
  429   builtinSemanticsVariant <- view iiBuiltinSemanticsVariant
  430   pure $ isPure builtinSemanticsVariant t

  432 nameUsedAtMostOnce
  433   :: forall name uni fun a
  434    . InliningConstraints name uni fun
  435   => name
  436   -> InlineM name uni fun a Bool
  437 nameUsedAtMostOnce n = do
  438   usgs <- view iiUsages
  439   -- 'inlining' terms used 0 times is a cheap way to remove dead code
  440   -- while we're here
  441   pure $ Usages.getUsageCount n usgs <= 1

  443 isFirstVarBeforeEffects
  444   :: forall name uni fun ann
  445    . InliningConstraints name uni fun
  446   => BuiltinSemanticsVariant fun
  447   -> name
  448   -> Term name uni fun ann
  449   -> Bool
  450 isFirstVarBeforeEffects builtinSemanticsVariant n t =
  451   -- This can in the worst case traverse a lot of the term, which could lead to
  452   -- us doing ~quadratic work as we process the program. However in practice
  453   -- most terms have a relatively short evaluation order before we hit Unknown,
  454   -- so it's not too bad.
  455   go (unEvalOrder (termEvaluationOrder builtinSemanticsVariant t))
  456   where
  457     -- Found the variable we're looking for!
  458     go ((EvalTerm _ _ (Var _ n')) : _) | n == n' = True
  459     -- Found a pure term, ignore it and continue
  460     go ((EvalTerm Pure _ _) : rest) = go rest
  461     -- Found a possibly impure term, our variable is definitely not first
  462     go ((EvalTerm MaybeImpure _ _) : _) = False
  463     -- Don't know, be conservative
  464     go (Unknown : _) = False
  465     go [] = False

  467 {-| Check if the given name is strict in the given term.
  468   This means that at least one occurrence of the name is found outside of the following:
  469   * 'delay' term
  470   * lambda body
  471   * case branch -}
  472 isStrictIn
  473   :: forall name uni fun a
  474    . Eq name
  475   => name
  476   -> Term name uni fun a
  477   -> Bool
  478 isStrictIn name = go
  479   where
  480     go :: Term name uni fun a -> Bool
  481     go = \case
  482       Var _ann name' -> name == name'
  483       LamAbs _ann _paramName _body -> False
  484       Apply _ann t1 t2 -> go t1 || go t2
  485       Force _ann t -> go t
  486       Delay _ann _term -> False
  487       Constant {} -> False
  488       Builtin {} -> False
  489       Error {} -> False
  490       Constr _ann _idx terms -> any go terms
  491       Case _ann scrut _branches -> go scrut
  492 
  493 effectSafe
  494   :: forall name uni fun a b
  495    . InliningConstraints name uni fun
  496   => Term name uni fun a
  497   -> name
  498   -> Bool
  499   -- ^ is it pure? See Note [Inlining and purity]
  500   -> InlineM name uni fun b Bool
  501 effectSafe body n termIsPure = do
  502   preserveLogging <- view iiPreserveLogging
  503   builtinSemantics <- view iiBuiltinSemanticsVariant
  504   return $
  505     termIsPure
  506       || isFirstVarBeforeEffects builtinSemantics n body
  507       || (not preserveLogging && isStrictIn n body)

  509 {-| Should we inline? Should only inline things that won't duplicate work
  510 or code.  See Note [Inlining approach and 'Secrets of the GHC Inliner'] -}
  511 acceptable
  512   :: Bool
  513   -- ^ inline constants
  514   -> Term name uni fun a
  515   -> InlineM name uni fun b Bool
  516 acceptable inlineConstants t = do
  517   unconditionalGrowth <- view iiInlineUnconditionalGrowth
  518   -- See Note [Inlining criteria]
  519   let costOk = costIsAcceptable t
  520       sizeOk = termAstSize t <= 1 + unconditionalGrowth
  521       constantsOk = case t of
  522         Constant {} -> inlineConstants
  523         _ -> True
  524   pure $ costOk && sizeOk && constantsOk
  525 
  526 {-| Is the cost increase (in terms of evaluation work) of inlining a variable
  527 whose RHS is the given term acceptable? -}
  528 costIsAcceptable :: Term name uni fun a -> Bool
  529 costIsAcceptable = \case
  530   Builtin {} -> True
  531   Var {} -> True
  532   Constant {} -> True
  533   Error {} -> True
  534   -- This will mean that we create closures at each use site instead of
  535   -- once, but that's a very low cost which we're okay rounding to 0.
  536   LamAbs {} -> True
  537   Apply {} -> False
  538   -- Inlining constructors of size 1 or 0 seems okay, but does result in doing
  539   -- the work for the elements at each use site.
  540   Constr _ _ es ->
  541     case es of
  542       [] -> True
  543       [e] -> costIsAcceptable e
  544       _ -> False
  545   -- Inlining a case means redoing the match at each use site
  546   Case {} -> False
  547   Force {} -> False
  548   Delay {} -> True

  550 -- | Fully apply and beta reduce.
  551 fullyApplyAndBetaReduce
  552   :: forall name uni fun a
  553    . InliningConstraints name uni fun
  554   => Ann a
  555   -- ^ Annotation on the variable to be replaced
  556   -> VarInfo name uni fun (Ann a)
  557   -> [(Ann a, Term name uni fun (Ann a))]
  558   -> InlineM name uni fun a (Maybe (Term name uni fun (Ann a)))
  559 fullyApplyAndBetaReduce av info args0 = do
  560   rhsBody <-
  561     -- Since we would like to fully apply here, we add decorations to
  562     -- `rhsBody` in the following order:
  563     --
  564     --   * decorations on the outermost Apply node, plus DDrop
  565     --   * decorations on the next Apply node, plus DDrop
  566     --   * ...
  567     --   * decorations on the innermost Apply node, plus DDrop
  568     --   * `av` plus DExpand.
  569     --   * decorations on the first LamAbs
  570     --   * decorations on the second LamAbs
  571     --   * ...
  572     --   * decorations on the last LamAbs
  573     --
  574     -- See Note [Inliner's Certifier Hints].
  575     decorateWith
  576       ( concatMap (\(annApply, _arg) -> annApply ^. decorations ++ [DDrop]) (reverse args0)
  577           ++ av ^. decorations
  578           ++ [DExpand]
  579           ++ concatMap (\(_name, annLam) -> annLam ^. decorations) (info ^. varBinders)
  580       )
  581       <$> liftDupable (let Done rhsBody = info ^. varRhsBody in rhsBody)
  582   let go
  583         :: Term name uni fun (Ann a)
  584         -> [(name, Ann a)]
  585         -> [(Ann a, Term name uni fun (Ann a))]
  586         -> InlineM name uni fun a (Maybe (Term name uni fun (Ann a)))
  587       go acc bs args = case (bs, args) of
  588         ([], _) -> pure . Just $ mkIterApp acc args
  589         ((param, _annLam) : params, (_annArg, arg) : args') -> do
  590           safe <- safeToBetaReduce param arg
  591           if safe
  592             then do
  593               acc' <-
  594                 termSubstNamesM
  595                   ( \n a ->
  596                       if n == param
  597                         then
  598                           Just
  599                             -- See Note [Inliner's Certifier Hints], #1
  600                             . decorateWith (a ^. decorations)
  601                             . decorateWith [DExpand]
  602                             <$> PLC.rename arg
  603                         else pure Nothing
  604                   )
  605                   acc
  606               go acc' params args'
  607             else pure Nothing
  608         _ -> pure Nothing
  609 
  610       -- Is it safe to turn `(\a -> body) arg` into `body [a := arg]`?
  611       -- The criteria is the same as the criteria for unconditionally
  612       -- inlining `a`, since inlining is the same as beta reduction.
  613       safeToBetaReduce
  614         :: name
  615         -> Term name uni fun (Ann a)
  616         -> InlineM name uni fun a Bool
  617       safeToBetaReduce a arg = shouldUnconditionallyInline False a arg rhsBody
  618   go rhsBody (info ^. varBinders) args0

  620 {-| This works in the same way as
  621 'PlutusIR.Transform.Inline.CallSiteInline.inlineSaturatedApp'.
  622 See Note [Inlining and beta reduction of functions]. -}
  623 inlineSaturatedApp
  624   :: forall name uni fun a
  625    . InliningConstraints name uni fun
  626   => Term name uni fun (Ann a)
  627   -> InlineM name uni fun a (Term name uni fun (Ann a))
  628 inlineSaturatedApp t
  629   | (Var av name, args) <- UPLC.splitApplication t =
  630       gets (lookupVarInfo name) >>= \case
  631         Nothing -> pure t
  632         Just varInfo ->
  633           fullyApplyAndBetaReduce av varInfo args >>= \case
  634             Nothing -> pure t
  635             Just fullyApplied -> do
  636               thresh <- view iiInlineCallsiteGrowth
  637               let
  638                 -- Inline only if the size is no bigger than
  639                 -- not inlining plus threshold.
  640                 sizeIsOk =
  641                   termAstSize fullyApplied <= termAstSize t + max 0 thresh
  642                 rhs = varInfo ^. varRhs
  643                 -- Cost is always OK if the RHS is a LamAbs,
  644                 -- but may not be otherwise.
  645                 costIsOk = costIsAcceptable rhs
  646               -- The RHS is always pure if it is a LamAbs,
  647               -- but may not be otherwise.
  648               rhsPure <- checkPurity rhs
  649               pure $
  650                 if sizeIsOk && costIsOk && rhsPure
  651                   then fullyApplied
  652                   else t
  653   | otherwise = pure t

  660 -- | An enclosing Drop or Expand layer
  661 data Decoration = DDrop | DExpand
  662 
  663 type Ann a = ([Decoration], a)
  664 
  665 emptyDecoration :: a -> Ann a
  666 emptyDecoration = (mempty,)
  667 {-# INLINE emptyDecoration #-}
  668 
  669 decorations :: Lens' (Ann a) [Decoration]
  670 decorations = _1
  671 {-# INLINE decorations #-}
  672 
  673 -- | Prepend the given decorations
  674 decorateWith :: [Decoration] -> Term name uni fun (Ann a) -> Term name uni fun (Ann a)
  675 decorateWith ds = modifyAnn (first (ds ++))
  676 {-# INLINE decorateWith #-}
  677 
  678 -- | Fold a decorated term into certifier hints.
  679 mkHints :: Term name uni fun (Ann a) -> CertifierHints.Inline
  680 mkHints = go
  681   where
  682     go t = decorate hints ds
  683       where
  684         decorate :: CertifierHints.Inline -> [Decoration] -> CertifierHints.Inline
  685         decorate = foldr $ \d h -> case d of
  686           DDrop -> CertifierHints.InlDrop h
  687           DExpand -> CertifierHints.InlExpand h
  688 
  689         ds = getAnn t ^. decorations
  690 
  691         hints = case t of
  692           Var {} -> CertifierHints.InlVar
  693           LamAbs _ _ body -> CertifierHints.InlLam (go body)
  694           Apply _ fun arg -> CertifierHints.InlApply (go fun) (go arg)
  695           Force _ body -> CertifierHints.InlForce (go body)
  696           Delay _ body -> CertifierHints.InlDelay (go body)
  697           Constant {} -> CertifierHints.InlCon
  698           Builtin {} -> CertifierHints.InlBuiltin
  699           Error {} -> CertifierHints.InlError
  700           Constr _ _ args -> CertifierHints.InlConstr (go <$> args)
  701           Case _ scrut alts -> CertifierHints.InlCase (go scrut) (go <$> V.toList alts)

  703 {- Note [Inliner's Certifier Hints]
  704 
  705 To certify the inliner, the certifier requires the inliner to emit hints that record
  706 where inlining operations occur. These hints are represented by the `Inline` data type
  707 defined in `UntypedPlutusCore.Transform.Certify.Hints`.
  708 
  709 Given a pre-term and a post-term, the expected hints can be defined by viewing the
  710 inliner as a process that repeatedly performs one of the following two operations:
  711 
  712 1. Substituting a term for a variable
  713 2. Dropping a dead binding
  714 
  715 Certifier hints are then defined constructively according to the following procedure.
  716 In the end, the final hints mirror the post-term structure, except that each node may be
  717 decorated with an arbitrary number of `InlExpand` and `InlDrop` layers,
  718 reflecting the inlining operations that the inliner has performed.
  719 
  720 - Initially the hints mirror the pre-term structure, using constructors of `Inline`
  721   except `InlExpand` and `InlDrop`. Excluding these two constructors,
  722   there is a 1-1 correspondence between hints constructors and AST constructors.
  723 
  724 - Suppose the inliner performs operation 1, substituting term `N` for variable `v`.
  725   Before the substitution, we have
  726 
  727     hints(v) = decorations(InlVar)
  728 
  729   where `decorations` denotes an arbitrary number of enclosing `InlExpand` or `InlDrop`
  730   layers. After the substitution, the hints at this location should become
  731 
  732     decorations(InlExpand hints(N))                          (#1)
  733 
  734   In other words, `InlVar` is replaced by `hints(N)` wrapped in an additional `InlExpand`.
  735 
  736 - Suppose the inliner performs operation 2, turning `(\x‾ₙ y. M) X‾ₙ Y` into `(\x‾ₙ. M) X‾ₙ`,
  737   where `x‾ₙ` and `X‾ₙ` denote sequences `x₁ x₂ ... xₙ` and `X₁ X₂ ... Xₙ` with `n >= 0`.
  738   Before this, we have
  739 
  740     hints((\x‾ₙ y. M) X‾ₙ Y) = decorations₁(InlApply (hints((\x‾ₙ y. M) X‾ₙ)) (hints(Y)))
  741     hints(\y. M) = decorations₂(InlLam hints(M))
  742 
  743   Afterwards, the hints for `(\x‾ₙ. M) X‾ₙ` should become
  744 
  745     decorations₁(InlDrop (hints((\x‾ₙ y. M) X‾ₙ)))           (#2)
  746 
  747   That is, it is decorated with `decorations₁` plus `InlDrop`.
  748   The hints for `M` should become
  749 
  750     decorations₂(hints(M))                                   (#3)
  751 
  752   That is, it is decorated with `decorations₂`. Note that when `n = 0`,
  753   `(\x‾ₙ y. M) X‾ₙ` becomes `M`, and therefore, `hints(M)` should be decorated with
  754   `decorations₁` plus `InlDrop` plus `decorations₂`.
  755 
  756 Refer to the golden tests for examples of hints (look for files ending with
  757 `.golden.certifier-hints`).
  758 
  759 Implementing this is quite fiddly, but is still easier than it sounds from the above
  760 description. Observe that at any point during inlining, the hints structure mirrors
  761 the structure of the current term, except for the decorations. Therefore all we need
  762 is to store the decorations of each node in its `ann`. When inlining is done, we can
  763 construct the hints structure using these decorations.
  764 -}
