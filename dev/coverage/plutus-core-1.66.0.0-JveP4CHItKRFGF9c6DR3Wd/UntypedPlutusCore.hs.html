   27 {-| Applies one program to another. Fails if the versions do not match
   28 and tries to merge annotations. -}
   29 applyProgram
   30   :: (MonadError ApplyProgramError m, Semigroup a)
   31   => Program name uni fun a
   32   -> Program name uni fun a
   33   -> m (Program name uni fun a)
   34 applyProgram (Program a1 v1 t1) (Program a2 v2 t2)
   35   | v1 == v2 =
   36       pure $ Program (a1 <> a2) v1 (applyTerm t1 t2)
   37 applyProgram (Program _a1 v1 _t1) (Program _a2 v2 _t2) =
   38   throwError $ MkApplyProgramError v1 v2

   40 applyTerm
   41   :: Semigroup a
   42   => Term name uni fun a
   43   -> Term name uni fun a
   44   -> Term name uni fun a
   45 applyTerm t1 t2 = Apply (getAnn t1 <> getAnn t2) t1 t2
