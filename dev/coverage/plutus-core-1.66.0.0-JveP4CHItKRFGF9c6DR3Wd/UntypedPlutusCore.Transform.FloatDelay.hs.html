    4 {-| The Float Delay optimization floats `Delay` from arguments into function bodies,
    5 if possible. It turns @(\n -> ...Force n...Force n...) (Delay arg)@ into
    6 @(\n -> ...Force (Delay n)...Force (Delay n)...) arg@.
    7 
    8 The above transformation is performed if:
    9 
   10     * All occurrences of @arg@ are under @Force@.
   11 
   12     * @arg@ is essentially work-free.
   13 
   14     * @arg@ is pure.
   15 
   16 This achieves a similar effect to Plutonomy's "Split Delay" optimization. The difference
   17 is that Split Delay simply splits the @Delay@ argument into multiple arguments, turning the
   18 above example into @(\m -> (\n -> ...Force n...Force n...) (Delay m)) arg@, and then relies
   19 on other optimizations to simplify it further. Specifically, once the inliner inlines
   20 @Delay m@, it will be identical to the result of Float Delay.
   21 
   22 The advantages of Float Delay are:
   23 
   24     * It doesn't rely on the inliner. In this example, Split Delay relies on the inliner to
   25       inline @Delay m@, but there's no guarantee that the inliner will do so, because inlining
   26       it may increase the program size.
   27 
   28       We can potentially modify the inliner such that it is aware of Float Delay and
   29       Force-Delay Cancel, and makes inlining decisions with these other optimizations in mind.
   30       The problem is that, not only does this makes the inlining heuristics much more
   31       complex, but it could easily lead to code duplication. Other optimizations often
   32       need to do some calculation in order to make certain optimization decisions (e.g., in
   33       this case, we want to check whether all occurrences of @arg@ are under @Force@), and
   34       if we rely on the inliner to inline the @Delay@, then the same check would need to be
   35       performed by the inliner.
   36 
   37     * Because Force Delay requires that all occurrences of @arg@ are under @Force@, it
   38       guarantees to not increase the size or the cost of the program. This is not the case
   39       with Split Delay: in this example, if the occurrences of @n@ are not under @Force@,
   40       then Split Delay may increase the size of the program, regardless of whether or not
   41       @Delay m@ is inlined. If @Delay m@ is not inlined, then it will also increase the
   42       cost of the program, due to the additional application.
   43 
   44 The alternative approach that always floats the @Delay@ regardless of whether or not all
   45 occurences of @arg@ are under @Force@ was implemented and tested, and it is strictly worse than
   46 Float Delay on our current test suite (specifically, Split Delay causes one test case
   47 to have a slightly bigger program, and everything else is equal).
   48 
   49 Why is this optimization performed on UPLC, not PIR?
   50 
   51     1. Not only are the types and let-bindings in PIR not useful for this optimization,
   52        they can also get in the way. For example, we cannot transform
   53        @let f = /\a. ...a... in ...{f t1}...{f t2}...@ into
   54        @ket f = ...a... in ...f...f...@.
   55 
   56     2. This optimization mainly interacts with ForceDelayCancel and the inliner, and
   57        both are part of the UPLC simplifier. -}
   58 module UntypedPlutusCore.Transform.FloatDelay (floatDelay) where

   78 floatDelay
   79   :: ( PLC.MonadQuote m
   80      , PLC.Rename (Term name uni fun a)
   81      , PLC.HasUnique name PLC.TermUnique
   82      , ToBuiltinMeaning uni fun
   83      )
   84   => BuiltinSemanticsVariant fun
   85   -> Term name uni fun a
   86   -> OptimizerT name uni fun a m (Term name uni fun a)
   87 floatDelay semvar term = do
   88   result <-
   89     PLC.rename term >>= \t ->
   90       pure . uncurry (flip simplifyBodies) $ simplifyArgs semvar (unforcedVars t) t
   91   recordOptimization term FloatDelayStage result
   92   return result
   93 
   94 {-| First pass. Returns the names of all variables, at least one occurrence
   95 of which is not under `Force`. -}
   96 unforcedVars
   97   :: forall name uni fun a
   98    . PLC.HasUnique name PLC.TermUnique
   99   => Term name uni fun a
  100   -> PLC.UniqueSet PLC.TermUnique
  101 unforcedVars = execWriter . go
  102   where
  103     go :: Term name uni fun a -> Writer (PLC.UniqueSet PLC.TermUnique) ()
  104     go = \case
  105       Var _ n -> tell (USet.singletonName n)
  106       Force _ Var {} -> pure ()
  107       t -> forOf_ termSubterms t go
  108 
  109 {-| Second pass. Removes `Delay` from eligible arguments, and returns
  110 the names of variables whose corresponding arguments are modified. -}
  111 simplifyArgs
  112   :: forall name uni fun a
  113    . ( PLC.HasUnique name PLC.TermUnique
  114      , ToBuiltinMeaning uni fun
  115      )
  116   => BuiltinSemanticsVariant fun
  117   -> PLC.UniqueSet PLC.TermUnique
  118   -- ^ The set of variables returned by `unforcedVars`.
  119   -> Term name uni fun a
  120   -> (Term name uni fun a, PLC.UniqueMap PLC.TermUnique a)
  121 simplifyArgs semvar blacklist = runWriter . go
  122   where
  123     go :: Term name uni fun ann -> Writer (PLC.UniqueMap PLC.TermUnique ann) (Term name uni fun ann)
  124     go = \case
  125       Apply appAnn (LamAbs lamAnn n lamBody) (Delay delayAnn arg)
  126         | isPure semvar arg
  127         , isWorkFree semvar arg
  128         , n `USet.notMemberByName` blacklist -> do
  129             tell (UMap.singletonByName n delayAnn)
  130             (Apply appAnn . LamAbs lamAnn n <$> go lamBody) <*> go arg
  131       t -> forOf termSubterms t go
  132 
  133 -- | Third pass. Turns @Force n@ into @Force (Delay n)@ for all eligibile @n@.
  134 simplifyBodies
  135   :: PLC.HasUnique name PLC.TermUnique
  136   => PLC.UniqueMap PLC.TermUnique a
  137   -> Term name uni fun a
  138   -> Term name uni fun a
  139 simplifyBodies whitelist = transformOf termSubterms $ \case
  140   var@(Var _ n)
  141     | Just ann <- UMap.lookupName n whitelist -> Delay ann var
  142   t -> t
