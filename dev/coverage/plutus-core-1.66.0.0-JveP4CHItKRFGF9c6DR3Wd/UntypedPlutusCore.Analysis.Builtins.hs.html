   31 -- | All non-static information about builtins that the compiler might want.
   32 data BuiltinsInfo (uni :: Type -> Type) fun = BuiltinsInfo
   33   { _biSemanticsVariant :: PLC.BuiltinSemanticsVariant fun
   34   , -- See Note [Unserializable constants]
   35     _biUnserializableConstants :: Some (ValueOf uni) -> Bool
   36   }
   37 
   38 makeLenses ''BuiltinsInfo
   39 
   40 instance Default (BuiltinsInfo DefaultUni DefaultFun) where
   41   def =
   42     BuiltinsInfo
   43       { _biSemanticsVariant = def
   44       , _biUnserializableConstants = defaultUniUnserializableConstants
   45       }
   46 
   47 -- | Get the arity of a builtin function from the 'PLC.BuiltinInfo'.
   48 builtinArityInfo
   49   :: forall uni fun
   50    . ToBuiltinMeaning uni fun
   51   => BuiltinsInfo uni fun
   52   -> fun
   53   -> Arity
   54 builtinArityInfo binfo = builtinArity (Proxy @uni) (binfo ^. biSemanticsVariant)
   55 
   56 constantIsSerializable
   57   :: forall uni fun
   58    . BuiltinsInfo uni fun
   59   -> Some (ValueOf uni)
   60   -> Bool
   61 constantIsSerializable bi v = not $ _biUnserializableConstants bi v
   62 
   63 termIsSerializable :: BuiltinsInfo uni fun -> Term name uni fun a -> Bool
   64 termIsSerializable binfo =
   65   allOf
   66     (termSubtermsDeep . _Constant)
   67     (constantIsSerializable binfo . snd)
   68 
   69 -- See Note [Unserializable constants]
   70 defaultUniUnserializableConstants :: Some (ValueOf DefaultUni) -> Bool
   71 defaultUniUnserializableConstants = \case
   72   Some (ValueOf DefaultUniBLS12_381_G1_Element _) -> True
   73   Some (ValueOf DefaultUniBLS12_381_G2_Element _) -> True
   74   Some (ValueOf DefaultUniBLS12_381_MlResult _) -> True
   75   _ -> False
