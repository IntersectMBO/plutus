    4 {-| Floats immediately-applied lambdas ("let bindings") outwards, as long as doing so
    5 cannot cause any expression to be evaluated more than before.
    6 
    7 This can unlock further optimizations, such as case-constr and force-delay cancellation.
    8 
    9 Specifically, it floats bindings from the following positions: @case@ scrutinee,
   10 @force@ body, and the function of @apply@.
   11 
   12 If we don't care about the order of effects, we can also float bindings from @apply@
   13 arguments and @constr@ arguments, but there's no evidence that doing so is beneficial,
   14 so we don't do it.
   15 
   16 This pass only floats one bindings from each location. Since the pass runs once per
   17 simplifier iteration, this should be sufficient. Floating out multi-lets at once would
   18 complicate things, since we'd need to consider both @(\a b. M) A B@ and
   19 @(\a. (\b -> M) B) A@. -}
   20 module UntypedPlutusCore.Transform.LetFloatOut (letFloatOut) where
   21 
   22 import PlutusCore qualified as PLC
   23 import UntypedPlutusCore.Core
   24 import UntypedPlutusCore.Transform.Optimizer
   25   ( OptimizerT
   26   , recordOptimization
   27   , pattern LetFloatOutStage
   28   )
   29 
   30 import Control.Lens (transformOf)
   31 
   32 letFloatOut
   33   :: ( PLC.MonadQuote m
   34      , PLC.Rename (Term name uni fun a)
   35      )
   36   => Term name uni fun a
   37   -> OptimizerT name uni fun a m (Term name uni fun a)
   38 letFloatOut term = do
   39   result <- transformOf termSubterms processTerm <$> PLC.rename term
   40   recordOptimization term LetFloatOutStage result
   41   pure result
   42 
   43 processTerm :: Term name uni fun a -> Term name uni fun a
   44 processTerm = \case
   45   Case ca (Apply aa (LamAbs la x body) rhs) branches ->
   46     Apply aa (LamAbs la x (Case ca body branches)) rhs
   47   Force fa (Apply aa (LamAbs la x body) rhs) ->
   48     Apply aa (LamAbs la x (Force fa body)) rhs
   49   Apply aa (Apply aa' (LamAbs la x body) rhs) arg ->
   50     Apply aa' (LamAbs la x (Apply aa body arg)) rhs
   51   other -> other
