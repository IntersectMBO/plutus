    5 import Control.DeepSeq
    6 import GHC.Generics

    8 -- | Certifier hints for the inlining pass.
    9 data Inline
   10   = InlVar
   11   | InlLam Inline
   12   | InlApply Inline Inline
   13   | InlForce Inline
   14   | InlDelay Inline
   15   | InlCon
   16   | InlBuiltin
   17   | InlError
   18   | InlConstr [Inline]
   19   | InlCase Inline [Inline]
   20   | InlExpand Inline
   21   | InlDrop Inline
   22   deriving stock (Generic)
   23   deriving anyclass (NFData)
   24 
   25 {-| Hints for the certifier.
   26 
   27 Note that there's a separate and unrelated notion of @InlineHints@. When there's
   28 ambiguity, this should be referred to as "certifier hints". -}
   29 data Hints
   30   = Inline Inline
   31   | NoHints
   32   deriving stock (Generic)
   33   deriving anyclass (NFData)
