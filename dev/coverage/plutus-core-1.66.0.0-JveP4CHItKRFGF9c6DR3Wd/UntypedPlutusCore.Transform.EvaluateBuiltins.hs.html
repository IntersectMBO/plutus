   20 evaluateBuiltinsPass
   21   :: (Monad m, ToBuiltinMeaning uni fun, Typeable name)
   22   => Bool
   23   -- ^ Whether to be conservative and try to retain logging behaviour.
   24   -> BuiltinsInfo uni fun
   25   -> CostingPart uni fun
   26   -> Term name uni fun a
   27   -> OptimizerT name uni fun a m (Term name uni fun a)
   28 evaluateBuiltinsPass preserveLogging binfo costModel term = do
   29   result <- evaluateBuiltins preserveLogging binfo costModel term
   30   recordOptimization term ConstantFoldingStage result
   31   return result

   33 evaluateBuiltins
   34   :: forall m name uni fun a
   35    . ( Monad m
   36      , ToBuiltinMeaning uni fun
   37      , Typeable name
   38      )
   39   => Bool
   40   -- ^ Whether to be conservative and try to retain logging behaviour.
   41   -> BuiltinsInfo uni fun
   42   -> CostingPart uni fun
   43   -> Term name uni fun a
   44   -> OptimizerT name uni fun a m (Term name uni fun a)
   45 evaluateBuiltins preserveLogging binfo costModel =
   46   pure . transformOf termSubterms processTerm
   47   where
   48     eval
   49       :: BuiltinRuntime (Term name uni fun ())
   50       -> AppCtx name uni fun a
   51       -> Maybe (Term name uni fun ())
   52     eval (BuiltinCostedResult _ getFXs) AppCtxEnd =
   53       case getFXs of
   54         BuiltinSuccess y -> Just y
   55         BuiltinSuccessWithLogs _ y -> if preserveLogging then Nothing else Just y
   56         BuiltinFailure {} -> Nothing
   57     eval (BuiltinExpectArgument toRuntime) (AppCtxTerm _ arg ctx) =
   58       eval (toRuntime $ void arg) ctx
   59     eval (BuiltinExpectForce runtime) (AppCtxType _ ctx) =
   60       eval runtime ctx
   61     eval _ _ = Nothing
   62 
   63     processTerm :: Term name uni fun a -> Term name uni fun a
   64     -- See Note [Context splitting in a recursive pass]
   65     processTerm t@(splitAppCtx -> (Builtin x bn, argCtx)) =
   66       let runtime = toBuiltinRuntime costModel (toBuiltinMeaning (binfo ^. biSemanticsVariant) bn)
   67        in case eval runtime argCtx of
   68             -- See Note [Unserializable constants] in PlutusIR.Analysis.Builtins
   69             Just t' | termIsSerializable binfo t' -> x <$ t'
   70             _ -> t
