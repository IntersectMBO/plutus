   39 {- Note [Term constructor ordering and numbers]
   40 Ordering of constructors has a small but real effect on efficiency.
   41 It's slightly more efficient to hit the earlier constructors, so it's better to put the more
   42 common ones first.
   43 
   44 The current ordering is based on their *empirically observed* frequency. We should check this
   45 occasionally.
   46 
   47 Additionally, the first 7 (or 3 on 32-bit systems) constructors will get *pointer tags*, which allows
   48 more efficient access when casing on them. So we ideally want to keep the number of constructors
   49 at 7 or fewer.
   50 
   51 We've got 8 constructors, *but* the last one is Error, which is only going to be seen at most
   52 once per program, so it's not too big a deal if it doesn't get a tag.
   53 
   54 See the GHC Notes "Tagging big families" and "Double switching for big families" in
   55 GHC.StgToCmm.Expr for more details.
   56 -}

   58 {-| The type of Untyped Plutus Core terms. Mirrors the type of Typed Plutus Core terms except
   59 
   60 1. all types are removed
   61 2. 'IWrap' and 'Unwrap' are removed
   62 3. type abstractions are replaced with 'Delay'
   63 4. type instantiations are replaced with 'Force'
   64 
   65 The latter two are due to the fact that we don't have value restriction in Typed Plutus Core
   66 and hence a computation can be stuck expecting only a single type argument for the computation
   67 to become unstuck. Therefore we can't just silently remove type abstractions and instantiations and
   68 need to replace them with something else that also blocks evaluation (in order for the semantics
   69 of an erased program to match with the semantics of the original typed one). 'Delay' and 'Force'
   70 serve exactly this purpose.
   71 -}
   72 -- Making all the fields strict gives us a couple of percent in benchmarks
   73 -- See Note [Term constructor ordering and numbers]
   74 data Term name uni fun ann
   75     = Var !ann !name
   76     | LamAbs !ann !name !(Term name uni fun ann)
   77     | Apply !ann !(Term name uni fun ann) !(Term name uni fun ann)
   78     | Force !ann !(Term name uni fun ann)
   79     | Delay !ann !(Term name uni fun ann)
   80     | Constant !ann !(Some (ValueOf uni))
   81     | Builtin !ann !fun
   82     -- This is the cutoff at which constructors won't get pointer tags
   83     -- See Note [Term constructor ordering and numbers]
   84     | Error !ann
   85     -- TODO: worry about overflow, maybe use an Integer
   86     -- TODO: try spine-strict list or strict list or vector
   87     -- See Note [Constr tag type]
   88     | Constr !ann !Word64 ![Term name uni fun ann]
   89     | Case !ann !(Term name uni fun ann) !(Vector (Term name uni fun ann))
   90     deriving stock (Functor, Generic)
   91 
   92 deriving stock instance (Show name, GShow uni, Everywhere uni Show, Show fun, Show ann, Closed uni)
   93     => Show (Term name uni fun ann)
   94 
   95 deriving anyclass instance (NFData name, NFData fun, NFData ann, Everywhere uni NFData, Closed uni)
   96     => NFData (Term name uni fun ann)

  113 type instance TPLC.UniOf (Term name uni fun ann) = uni

  115 instance TermLike (Term name uni fun) TPLC.TyName name uni fun where
  116     var      = Var
  117     tyAbs    = \ann _ _ -> Delay ann
  118     lamAbs   = \ann name _ -> LamAbs ann name
  119     apply    = Apply
  120     constant = Constant
  121     builtin  = Builtin
  122     tyInst   = \ann term _ -> Force ann term
  123     unwrap   = const id
  124     iWrap    = \_ _ _ -> id
  125     error    = \ann _ -> Error ann
  126     constr   = \ann _ i es -> Constr ann i es
  127     kase     = \ann _ arg cs -> Case ann arg (fromList cs)
  128 
  129 instance TPLC.HasConstant (Term name uni fun ()) where
  130     asConstant (Constant _ val) = pure val
  131     asConstant _                = TPLC.throwNotAConstant
  132 
  133     fromConstant = Constant ()

  145 -- | Return the outermost annotation of a 'Term'.
  146 termAnn :: Term name uni fun ann -> ann
  147 termAnn (Constant ann _) = ann
  148 termAnn (Builtin ann _)  = ann
  149 termAnn (Var ann _)      = ann
  150 termAnn (LamAbs ann _ _) = ann
  151 termAnn (Apply ann _ _)  = ann
  152 termAnn (Delay ann _)    = ann
  153 termAnn (Force ann _)    = ann
  154 termAnn (Error ann)      = ann
  155 termAnn (Constr ann _ _) = ann
  156 termAnn (Case ann _ _)   = ann

  158 bindFunM
  159     :: Monad m
  160     => (ann -> fun -> m (Term name uni fun' ann))
  161     -> Term name uni fun ann
  162     -> m (Term name uni fun' ann)
  163 bindFunM f = go where
  164     go (Constant ann val)     = pure $ Constant ann val
  165     go (Builtin ann fun)      = f ann fun
  166     go (Var ann name)         = pure $ Var ann name
  167     go (LamAbs ann name body) = LamAbs ann name <$> go body
  168     go (Apply ann fun arg)    = Apply ann <$> go fun <*> go arg
  169     go (Delay ann term)       = Delay ann <$> go term
  170     go (Force ann term)       = Force ann <$> go term
  171     go (Error ann)            = pure $ Error ann
  172     go (Constr ann i args)    = Constr ann i <$> traverse go args
  173     go (Case ann arg cs)      = Case ann <$> go arg <*> traverse go cs
  174 
  175 bindFun
  176     :: (ann -> fun -> Term name uni fun' ann)
  177     -> Term name uni fun ann
  178     -> Term name uni fun' ann
  179 bindFun f = runIdentity . bindFunM (coerce f)
  180 
  181 mapFun :: (ann -> fun -> fun') -> Term name uni fun ann -> Term name uni fun' ann
  182 mapFun f = bindFun $ \ann fun -> Builtin ann (f ann fun)
