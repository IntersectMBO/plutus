   44 substTyVarA

  125 -- | Substitute the given closed 'Type' for the given type variable in the given 'Type'. Does not
  126 -- descend under binders that bind the same variable as the one we're substituting for (since from
  127 -- there that variable is no longer free). The resulting 'Term' may and likely will not satisfy
  128 -- global uniqueness.
  129 typeSubstClosedType
  130     :: Eq tyname => tyname -> Type tyname uni a -> Type tyname uni a -> Type tyname uni a
  131 typeSubstClosedType tn0 ty0 = go where
  132     go = \case
  133          TyVar    a tn      -> if tn == tn0 then ty0 else TyVar a tn
  134          TyForall a tn k ty -> TyForall a tn k (goUnder tn ty)
  135          TyLam    a tn k ty -> TyLam    a tn k (goUnder tn ty)
  136          ty                 -> ty & over typeSubtypes go
  137     goUnder tn ty = if tn == tn0 then ty else go ty
  138 
  139 -- | Substitute the given closed 'Type' for the given type variable in the given 'Term'. Does not
  140 -- descend under binders that bind the same variable as the one we're substituting for (since from
  141 -- there that variable is no longer free). The resulting 'Term' may and likely will not satisfy
  142 -- global uniqueness.
  143 termSubstClosedType
  144     :: Eq tyname
  145     => tyname -> Type tyname uni a -> Term tyname name uni fun a -> Term tyname name uni fun a
  146 termSubstClosedType tn0 ty0 = go where
  147     go = \case
  148          TyAbs a tn k body -> TyAbs a tn k (goUnder tn body)
  149          t                 -> t & over termSubtypes goTy & over termSubterms go
  150     goUnder tn term = if tn == tn0 then term else go term
  151     goTy = typeSubstClosedType tn0 ty0
  152 
  153 -- | Substitute the given closed 'Term' for the given term variable in the given 'Term'. Does not
  154 -- descend under binders that bind the same variable as the one we're substituting for (since from
  155 -- there that variable is no longer free). The resulting 'Term' may and likely will not satisfy
  156 -- global uniqueness.
  157 termSubstClosedTerm
  158     :: Eq name
  159     => name
  160     -> Term tyname name uni fun a
  161     -> Term tyname name uni fun a
  162     -> Term tyname name uni fun a
  163 termSubstClosedTerm varFor new = go where
  164     go = \case
  165          Var    a var         -> if var == varFor then new else Var a var
  166          LamAbs a var ty body -> LamAbs a var ty (goUnder var body)
  167          t                    -> t & over termSubterms go
  168     goUnder var term = if var == varFor then term else go term

  172 typeMapNames
  173     :: forall tyname tyname' uni ann
  174     .  (tyname -> tyname')
  175     -> Type tyname uni ann
  176     -> Type tyname' uni ann
  177 typeMapNames f = go
  178     where
  179       go :: Type tyname uni ann -> Type tyname' uni ann
  180       go = \case
  181            TyVar ann tn         -> TyVar ann (f tn)
  182            TyFun ann ty1 ty2    -> TyFun ann (go ty1) (go ty2)
  183            TyIFix ann ty1 ty2   -> TyIFix ann (go ty1) (go ty2)
  184            TyForall ann tn k ty -> TyForall ann (f tn) k (go ty)
  185            TyBuiltin ann s      -> TyBuiltin ann s
  186            TyLam ann tn k ty    -> TyLam ann (f tn) k (go ty)
  187            TyApp ann ty1 ty2    -> TyApp ann (go ty1) (go ty2)
  188            TySOP ann tyls       -> TySOP ann ((fmap . fmap) go tyls)
  189 
  190 -- termMapNames requires two function arguments: one (called f) to modify type names
  191 -- and another (called g) to modify variable names.
  192 termMapNames
  193     :: forall tyname tyname' name name' uni fun ann
  194     .  (tyname -> tyname')
  195     -> (name -> name')
  196     -> Term tyname name uni fun ann
  197     -> Term tyname' name' uni fun ann
  198 termMapNames f g = go
  199     where
  200         go :: Term tyname name uni fun ann -> Term tyname' name' uni fun ann
  201         go = \case
  202             LamAbs ann name ty body -> LamAbs ann (g name) (typeMapNames f ty) (go body)
  203             TyAbs ann tyname k body -> TyAbs ann (f tyname) k (go body)
  204             Var ann name            -> Var ann (g name)
  205             Apply ann t1 t2         -> Apply ann (go t1) (go t2)
  206             Constant ann c          -> Constant ann c
  207             Builtin ann b           -> Builtin ann b
  208             TyInst ann body ty      -> TyInst ann (go body) (typeMapNames f ty)
  209             Unwrap ann body         -> Unwrap ann (go body)
  210             IWrap ann ty1 ty2 body  -> IWrap ann (typeMapNames f ty1) (typeMapNames f ty2) (go body)
  211             Constr ann ty i es      -> Constr ann (typeMapNames f ty) i (fmap go es)
  212             Case ann ty arg cs      -> Case ann (typeMapNames f ty) (go arg) (fmap go cs)
  213             Error ann ty            -> Error ann (typeMapNames f ty)
  214 
  215 programMapNames
  216     :: forall tyname tyname' name name' uni fun ann
  217     .  (tyname -> tyname')
  218     -> (name -> name')
  219     -> Program tyname name uni fun ann
  220     -> Program tyname' name' uni fun ann
  221 programMapNames f g (Program a v term) = Program a v (termMapNames f g term)

  225 -- | Get all the free term variables in a term.
  226 fvTerm :: HasUnique name unique => Traversal' (Term tyname name uni fun ann) name
  227 fvTerm = fvTermCtx mempty
  228 
  229 fvTermCtx
  230     :: HasUnique name unique
  231     => UniqueSet unique -> Traversal' (Term tyname name uni fun ann) name
  232 fvTermCtx bound f = \case
  233     Var a n         -> Var a <$> (if USet.memberByName n bound then pure n else f n)
  234     LamAbs a n ty t -> LamAbs a n ty <$> fvTermCtx (USet.insertByName n bound) f t
  235     t               -> (termSubterms . fvTermCtx bound) f t
  236 
  237 -- | Get all the free type variables in a term.
  238 ftvTerm :: HasUnique tyname unique => Traversal' (Term tyname name uni fun ann) tyname
  239 ftvTerm = ftvTermCtx mempty
  240 
  241 ftvTermCtx
  242     :: HasUnique tyname unique
  243     => UniqueSet unique
  244     -> Traversal' (Term tyname name uni fun ann) tyname
  245 ftvTermCtx bound f = \case
  246     TyAbs a ty k t -> TyAbs a ty k <$> ftvTermCtx (USet.insertByName ty bound) f t
  247     -- sound because the subterms and subtypes are disjoint
  248     t              ->
  249         ((termSubterms . ftvTermCtx bound) `Unsound.adjoin` (termSubtypes . ftvTyCtx bound)) f t
  250 
  251 -- | Get all the free type variables in a type.
  252 ftvTy
  253     :: HasUnique tyname unique =>
  254     Traversal' (Type tyname uni ann) tyname
  255 ftvTy = ftvTyCtx mempty
  256 
  257 ftvTyCtx
  258     :: HasUnique tyname unique
  259     => UniqueSet unique
  260     -> Traversal' (Type tyname uni ann) tyname
  261 ftvTyCtx bound f = \case
  262     TyVar a ty          -> TyVar a <$> (if USet.memberByName ty bound then pure ty else f ty)
  263     TyForall a bnd k ty -> TyForall a bnd k <$> ftvTyCtx (USet.insertByName bnd bound) f ty
  264     TyLam a bnd k ty    -> TyLam a bnd k <$> ftvTyCtx (USet.insertByName bnd bound) f ty
  265     t                   -> (typeSubtypes . ftvTyCtx bound) f t
