   10 module PlutusCore.Evaluation.ErrorWithCause
   11     ( ErrorWithCause (..)
   12     , throwErrorWithCause
   13     ) where
   14 
   15 import PlutusPrelude
   16 
   17 import PlutusCore.Pretty
   18 
   19 import Control.Monad.Except
   20 import Data.Bifunctor
   21 import Prettyprinter
   22 
   23 -- | An error and (optionally) what caused it.
   24 data ErrorWithCause err cause = ErrorWithCause
   25     { _ewcError :: !err
   26     , _ewcCause :: !(Maybe cause)
   27     } deriving stock (Eq, Functor, Foldable, Traversable, Generic)
   28       deriving anyclass (NFData)
   29 
   30 instance Bifunctor ErrorWithCause where
   31     bimap f g (ErrorWithCause err cause) = ErrorWithCause (f err) (g <$> cause)
   32     {-# INLINE bimap #-}
   33 
   34 instance (Pretty err, Pretty cause) => Pretty (ErrorWithCause err cause) where
   35     pretty (ErrorWithCause e c) = pretty e <+> "caused by:" <+> pretty c
   36 
   37 instance (PrettyBy config cause, PrettyBy config err) =>
   38             PrettyBy config (ErrorWithCause err cause) where
   39     prettyBy config (ErrorWithCause err mayCause) = fold
   40         [ "An error has occurred:"
   41         , hardline
   42         , prettyBy config err
   43         , case mayCause of
   44             Nothing    -> mempty
   45             Just cause -> hardline <> "Caused by:" <+> prettyBy config cause
   46         ]
   47 
   48 instance (PrettyPlc cause, PrettyPlc err) =>
   49             Show (ErrorWithCause err cause) where
   50     show = render . prettyPlcReadable
   51 
   52 deriving anyclass instance (PrettyPlc cause, PrettyPlc err, Typeable cause, Typeable err) =>
   53     Exception (ErrorWithCause err cause)
   54 
   55 throwErrorWithCause
   56   :: MonadError (ErrorWithCause e cause) m
   57   => e
   58   -> cause
   59   -> m x
   60 throwErrorWithCause e = throwError . ErrorWithCause e . Just
   61 {-# INLINE throwErrorWithCause #-}
