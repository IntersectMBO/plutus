    7 module PlutusCore.StdLib.Data.Pair
    8     ( pair
    9     , fstPair
   10     , sndPair
   11     , uncurry
   12     ) where
   13 
   14 import Prelude hiding (fst, snd, uncurry)
   15 
   16 import PlutusCore.Core
   17 import PlutusCore.Default
   18 import PlutusCore.MkPlc
   19 import PlutusCore.Name.Unique
   20 import PlutusCore.Quote
   21 
   22 -- | @(,)@ as a built-in PLC type.
   23 pair :: uni `HasTypeLevel` (,) => Type tyname uni ()
   24 pair = mkTyBuiltin @_ @(,) ()
   25 
   26 -- | @fst@ as a PLC term.
   27 --
   28 -- > /\(a :: *) (b :: *) -> \(p : pair a b) -> fst {a} {b} p
   29 fstPair :: TermLike term tyname name DefaultUni DefaultFun => term ()
   30 fstPair = builtin () FstPair
   31 
   32 -- | @snd@ as a PLC term.
   33 --
   34 -- > /\(a :: *) (b :: *) -> \(p : pair a b) -> snd {a} {b} p
   35 sndPair :: TermLike term tyname name DefaultUni DefaultFun => term ()
   36 sndPair = builtin () SndPair
   37 
   38 -- | @uncurry@ as a PLC term.
   39 --
   40 -- > /\(a :: *) (b :: *) (c :: *) -> \(f : a -> b -> c) (p : pair a b) ->
   41 -- >     f (fst {a} {b} p) (snd {a} {b} p)
   42 uncurry :: TermLike term TyName Name DefaultUni DefaultFun => term ()
   43 uncurry = runQuote $ do
   44     a <- freshTyName "a"
   45     b <- freshTyName "b"
   46     c <- freshTyName "c"
   47     f <- freshName "f"
   48     p <- freshName "p"
   49     return
   50         . tyAbs () a (Type ())
   51         . tyAbs () b (Type ())
   52         . tyAbs () c (Type ())
   53         . lamAbs () f (TyFun () (TyVar () a) . TyFun () (TyVar () b) $ TyVar () c)
   54         . lamAbs () p (mkIterTyAppNoAnn pair [TyVar () a, TyVar () b])
   55         $ mkIterAppNoAnn (var () f)
   56             [ apply () (mkIterInstNoAnn fstPair [TyVar () a, TyVar () b]) $ var () p
   57             , apply () (mkIterInstNoAnn sndPair [TyVar () a, TyVar () b]) $ var () p
   58             ]
