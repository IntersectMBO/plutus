   28 -- | @Data@ as a built-in PLC type.
   29 dataTy :: uni `HasTypeLevel` Data => Type tyname uni ()
   30 dataTy = mkTyBuiltin @_ @Data ()
   31 
   32 -- | Pattern matching over 'Data' inside PLC.
   33 --
   34 -- > \(d : data) ->
   35 -- >     /\(r :: *) ->
   36 -- >      \(fConstr : integer -> list data -> r)
   37 -- >       (fMap : list (pair data data) -> r)
   38 -- >       (fList : list data -> r)
   39 -- >       (fI : integer -> r)
   40 -- >       (fB : bytestring -> r) ->
   41 -- >           chooseData
   42 -- >               d
   43 -- >               {unit -> r}
   44 -- >               (\(u : unit) -> uncurry {integer} {list data} {r} fConstr (unConstrB d))
   45 -- >               (\(u : unit) -> fMap (unMapB d))
   46 -- >               (\(u : unit) -> fList (unListB d))
   47 -- >               (\(u : unit) -> fI (unIB d))
   48 -- >               (\(u : unit) -> fB (unBB d))
   49 -- >               unitval
   50 matchData :: TermLike term TyName Name DefaultUni DefaultFun => term ()
   51 matchData = runQuote $ do
   52     r       <- freshTyName "r"
   53     fConstr <- freshName "fConstr"
   54     fMap    <- freshName "fMap"
   55     fList   <- freshName "fList"
   56     fI      <- freshName "fI"
   57     fB      <- freshName "fB"
   58     d       <- freshName "d"
   59     u       <- freshName "u"
   60     let listData = mkTyBuiltin @_ @[Data] ()
   61         listPairData = mkTyBuiltin @_ @[(Data, Data)] ()
   62         bytestring = mkTyBuiltin @_ @ByteString ()
   63     return
   64         . lamAbs () d dataTy
   65         . tyAbs () r (Type ())
   66         . lamAbs () fConstr (TyFun () integer . TyFun () listData $ TyVar () r)
   67         . lamAbs () fMap (TyFun () listPairData $ TyVar () r)
   68         . lamAbs () fList (TyFun () listData $ TyVar () r)
   69         . lamAbs () fI (TyFun () integer $ TyVar () r)
   70         . lamAbs () fB (TyFun () bytestring $ TyVar () r)
   71         $ mkIterAppNoAnn (tyInst () (builtin () ChooseData) . TyFun () unit $ TyVar () r)
   72             [ var () d
   73             ,   lamAbs () u unit
   74               $ mkIterAppNoAnn (mkIterInstNoAnn uncurry [integer, listData, TyVar () r])
   75                   [ var () fConstr
   76                   , apply () (builtin () UnConstrData) $ var () d
   77                   ]
   78             ,   lamAbs () u unit
   79               . apply () (var () fMap)
   80               . apply () (builtin () UnMapData)
   81               $ var () d
   82             ,   lamAbs () u unit
   83               . apply () (var () fList)
   84               . apply () (builtin () UnListData)
   85               $ var () d
   86             ,   lamAbs () u unit
   87               . apply () (var () fI)
   88               . apply () (builtin () UnIData)
   89               $ var () d
   90             ,   lamAbs () u unit
   91               . apply () (var () fB)
   92               . apply () (builtin () UnBData)
   93               $ var () d
   94             , unitval
   95             ]
