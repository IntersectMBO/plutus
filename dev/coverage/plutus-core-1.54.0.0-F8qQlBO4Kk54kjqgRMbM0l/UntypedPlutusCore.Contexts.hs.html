    8 -- | A context for an iterated term/type application.
    9 data AppCtx name uni fun a
   10   =  AppCtxTerm a (Term name uni fun a) (AppCtx name uni fun a)
   11   | AppCtxType a (AppCtx name uni fun a)
   12   | AppCtxEnd
   13 
   14 {-| Takes a term and views it as a head plus an 'AppCtx', e.g.
   15 @
   16   [{ f t } u v]
   17   -->
   18   (f, [{ _ t } u v])
   19   ==
   20   f (AppCtxType t (AppCtxTerm u (AppCtxTerm v AppCtxEnd)))
   21 @
   22 -}
   23 splitAppCtx :: Term nam uni fun a -> (Term nam uni fun a, AppCtx nam uni fun a)
   24 splitAppCtx = go AppCtxEnd
   25  where
   26   go appCtx = \case
   27     Apply ann function argument -> go (AppCtxTerm ann argument appCtx) function
   28     Force ann forcedTerm -> go (AppCtxType ann appCtx) forcedTerm
   29     term -> (term, appCtx)
   30 
   31 -- | Fills in the hole in an 'AppCtx', the inverse of 'splitAppCtx'.
   32 fillAppCtx
   33   :: Term name uni fun ann
   34   -> AppCtx name uni fun ann
   35   -> Term name uni fun ann
   36 fillAppCtx term = \case
   37   AppCtxEnd -> term
   38   AppCtxTerm ann arg ctx -> fillAppCtx (Apply ann term arg) ctx
   39   AppCtxType ann ctx -> fillAppCtx (Force ann term) ctx
