   38 {- Note [Wrapping the BLS12-381 types in Plutus Core].  In the Haskell bindings
   39 to the `blst` library in cardano-crypto-class, points in G1 and G2 are
   40 represented as ForeignPtrs pointing to C objects, with a phantom type
   41 determining which group is involved. We have to wrap these in a newtype here
   42 because otherwise the builtin machinery spots that they're applications and
   43 can't find the relevant type parameters.  In theory I think we could add a
   44 couple of phantom types to the default universe, but it seemed simpler and safer
   45 to use monomorphic types instead, even though it requires a bit of code
   46 duplication between G1 and G2.  Much of the code here has apparently identical
   47 versions in PlutusCore.Crypto.BLS12_381.G2, but the signatures of the functions
   48 in the two versions are invisibly instantiating the cardano-crypto-class
   49 functions at the appropriate phantom types.
   50 
   51 See also Note [Wrapping the BLS12-381 types in PlutusTx].
   52 -}
   53 newtype Element = Element { unElement :: BlstBindings.Point1 }
   54     deriving newtype (Eq)
   55 instance Show Element where
   56     show  = byteStringAsHex . compress
   57 instance Pretty Element where
   58     pretty = pretty . show
   59 instance PrettyBy ConstConfig Element
   60 {- | We don't support direct flat encoding of G2 elements because of the expense
   61    of on-chain uncompression.  Users should convert between G2 elements and
   62    bytestrings using `compress` and `uncompress`: the bytestrings can be
   63    flat-encoded in the usual way. -}
   64 instance Flat Element where
   65     -- This might happen on the chain, so `fail` rather than `error`.
   66     decode = fail "Flat decoding is not supported for objects of type bls12_381_G1_element: use bls12_381_G1_uncompress on a bytestring instead."
   67     -- This will be a Haskell runtime error, but encoding doesn't happen on chain,
   68     -- so it's not too bad.
   69     encode = error "Flat encoding is not supported for objects of type bls12_381_G1_element: use bls12_381_G1_compress to obtain a bytestring instead."
   70     size _ = id
   71 instance NFData Element where
   72     rnf (Element x) = rwhnf x  -- Just to be on the safe side.
   73 
   74 instance Hashable Element where
   75     hashWithSalt salt = hashWithSalt salt . compress
   76 
   77 -- | Add two G1 group elements
   78 add :: Element -> Element -> Element
   79 add = coerce (BlstBindings.blsAddOrDouble @BlstBindings.Curve1)
   80 {-# INLINE add #-}

   82 -- | Negate a G1 group element
   83 neg :: Element -> Element
   84 neg = coerce (BlstBindings.blsNeg @BlstBindings.Curve1)
   85 {-# INLINE neg #-}

   87 -- | Multiplication of group elements by scalars. In the blst library the
   88 -- arguments are the other way round, but scalars acting on the left is more
   89 -- consistent with standard mathematical practice.
   90 scalarMul :: Integer -> Element -> Element
   91 scalarMul = coerce $ flip (BlstBindings.blsMult @BlstBindings.Curve1)
   92 {-# INLINE scalarMul #-}

   94 {- | Compress a G1 element to a bytestring. This serialises a curve point to its
   95  x coordinate only.  The compressed bytestring is 48 bytes long, with three
   96  spare bits used to convey extra information about the point, including
   97  determining which of two possible y coordinates the point has and whether the
   98  point is the point at infinity. See
   99  https://github.com/supranational/blst#serialization-format
  100 -}
  101 compress :: Element -> ByteString
  102 compress = coerce (BlstBindings.blsCompress @BlstBindings.Curve1)
  103 {-# INLINE compress #-}
  104 
  105 {- | Uncompress a bytestring to get a G1 point.  This will fail if any of the
  106    following are true.
  107      * The bytestring is not exactly 48 bytes long.
  108      * The most significant three bits are used incorrectly.
  109      * The bytestring encodes a field element which is not the
  110        x coordinate of a point on the E1 curve.
  111      * The bytestring does represent a point on the E1 curve, but the
  112        point is not in the G1 subgroup.
  113 -}
  114 uncompress :: ByteString -> Either BlstBindings.BLSTError Element
  115 uncompress = coerce (BlstBindings.blsUncompress @BlstBindings.Curve1)
  116 {-# INLINE uncompress #-}

  118 {-  Note [Hashing and Domain Separation Tags].  The hashToGroup functions take a
  119    bytestring and hash it to obtain an element in the relevant group, as
  120    described in
  121 
  122    https://datatracker.ietf.org/doc/html/draft-irtf-cfrg-hash-to-curve.
  123 
  124    In addition to the input bytestring (the "message"), the hashing function
  125    also takes another (possibly empty) bytestring argument, a "Domain Separation
  126    Tag" (DST), which is incorporated in the hashing process; the intention is
  127    that for security purposes different protocols should use different DSTs to
  128    ensure that hashes are unique to the protocol in use: see Section 2.2.5 of
  129    the above specification.  In principle, arbitrary-length DSTs can be used,
  130    but we only allow DSTs of up to 255 bytes, failing if a larger DST is
  131    supplied.  If a larger DST is required, it should be hashed beforehand to
  132    obtain a hash of accpetable size, as described in Section 5.3.3 of the
  133    specification.
  134 
  135    The hashing functions in the blst library allow a third argument as well (an
  136    "augmentation string").  We don't support this functionality directly because
  137    precisely the same effect can be achieved by prepending the augmentation
  138    string to the message.
  139 -}
  140 
  141 -- | Take an arbitrary bytestring and a Domain Separation Tag (DST) and hash
  142 -- them to a get point in G1.
  143 hashToGroup :: ByteString -> ByteString -> Either BLS12_381_Error Element
  144 hashToGroup msg dst =
  145     if Data.ByteString.length dst > 255
  146     then Left HashToCurveDstTooBig
  147     else Right . Element $ BlstBindings.blsHash @BlstBindings.Curve1 msg (Just dst) Nothing
