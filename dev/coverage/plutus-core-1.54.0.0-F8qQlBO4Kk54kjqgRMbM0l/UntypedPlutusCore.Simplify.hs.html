   35 simplifyProgram ::
   36     forall name uni fun m a.
   37     (Compiling m uni fun name a) =>
   38     SimplifyOpts name a ->
   39     BuiltinSemanticsVariant fun ->
   40     Program name uni fun a ->
   41     m (Program name uni fun a)
   42 simplifyProgram opts builtinSemanticsVariant (Program a v t) =
   43   Program a v <$> simplifyTerm opts builtinSemanticsVariant t
   44 
   45 simplifyProgramWithTrace ::
   46     forall name uni fun m a.
   47     (Compiling m uni fun name a) =>
   48     SimplifyOpts name a ->
   49     BuiltinSemanticsVariant fun ->
   50     Program name uni fun a ->
   51     m (Program name uni fun a, SimplifierTrace name uni fun a)
   52 simplifyProgramWithTrace opts builtinSemanticsVariant (Program a v t) = do
   53   (result, trace) <-
   54     runSimplifierT
   55       $ termSimplifier opts builtinSemanticsVariant t
   56   pure (Program a v result, trace)
   57 
   58 simplifyTerm ::
   59     forall name uni fun m a.
   60     (Compiling m uni fun name a) =>
   61     SimplifyOpts name a ->
   62     BuiltinSemanticsVariant fun ->
   63     Term name uni fun a ->
   64     m (Term name uni fun a)
   65 simplifyTerm opts builtinSemanticsVariant term =
   66   evalSimplifierT $ termSimplifier opts builtinSemanticsVariant term
   67 
   68 termSimplifier ::
   69     forall name uni fun m a.
   70     (Compiling m uni fun name a) =>
   71     SimplifyOpts name a ->
   72     BuiltinSemanticsVariant fun ->
   73     Term name uni fun a ->
   74     SimplifierT name uni fun a m (Term name uni fun a)
   75 termSimplifier opts builtinSemanticsVariant =
   76     simplifyNTimes (_soMaxSimplifierIterations opts) >=> cseNTimes cseTimes
   77   where
   78     -- Run the simplifier @n@ times
   79     simplifyNTimes ::
   80       Int ->
   81       Term name uni fun a ->
   82       SimplifierT name uni fun a m (Term name uni fun a)
   83     simplifyNTimes n = List.foldl' (>=>) pure $ map simplifyStep [1..n]
   84 
   85     -- Run CSE @n@ times, interleaved with the simplifier.
   86     -- See Note [CSE]
   87     cseNTimes ::
   88       Int ->
   89       Term name uni fun a ->
   90       SimplifierT name uni fun a m (Term name uni fun a)
   91     cseNTimes n = foldl' (>=>) pure $ concatMap (\i -> [cseStep i, simplifyStep i]) [1..n]
   92 
   93     -- generate simplification step
   94     simplifyStep ::
   95       Int ->
   96       Term name uni fun a ->
   97       SimplifierT name uni fun a m (Term name uni fun a)
   98     simplifyStep _ =
   99         floatDelay
  100         >=> forceCaseDelay
  101         >=> case (eqT @uni @PLC.DefaultUni, eqT @fun @DefaultFun) of
  102             (Just Refl, Just Refl) -> forceDelay builtinSemanticsVariant
  103             _                      -> pure
  104         >=> caseOfCase'
  105         >=> caseReduce
  106         >=> inline
  107               (_soInlineCallsiteGrowth opts)
  108               (_soInlineConstants opts)
  109               (_soPreserveLogging opts)
  110               (_soInlineHints opts)
  111               builtinSemanticsVariant
  112 
  113     caseOfCase' ::
  114       Term name uni fun a ->
  115       SimplifierT name uni fun a m (Term name uni fun a)
  116     caseOfCase' = case eqT @fun @DefaultFun of
  117       Just Refl -> caseOfCase
  118       Nothing   -> pure
  119 
  120     cseStep ::
  121       Int ->
  122       Term name uni fun a ->
  123       SimplifierT name uni fun a m (Term name uni fun a)
  124     cseStep _ =
  125       case (eqT @name @Name, eqT @uni @PLC.DefaultUni) of
  126         (Just Refl, Just Refl) -> cse builtinSemanticsVariant
  127         _                      -> pure
  128 
  129     cseTimes = if _soConservativeOpts opts then 0 else _soMaxCseIterations opts
