   16 -- See Note [Normalization].
   17 -- | Normalize a 'Type'.
   18 normalizeType
   19     :: (HasUnique tyname TypeUnique, MonadNormalizeType uni m)
   20     => Type tyname uni ann -> m (Normalized (Type tyname uni ann))
   21 normalizeType = rename >=> runNormalizeTypeT . normalizeTypeM
   22 
   23 -- | Normalize every 'Type' in a 'Term'.
   24 normalizeTypesIn
   25     :: (HasUnique tyname TypeUnique, HasUnique name TermUnique, MonadNormalizeType uni m)
   26     => Term tyname name uni fun ann -> m (Term tyname name uni fun ann)
   27 normalizeTypesIn = rename >=> runNormalizeTypeT . normalizeTypesInM
   28 
   29 -- | Normalize every 'Type' in a 'Program'.
   30 normalizeTypesInProgram
   31     :: (HasUnique tyname TypeUnique, HasUnique name TermUnique, MonadNormalizeType uni m)
   32     => Program tyname name uni fun ann -> m (Program tyname name uni fun ann)
   33 normalizeTypesInProgram (Program x v t) = Program x v <$> normalizeTypesIn t
