   72 -- | A final encoding for Term, to allow PLC terms to be used transparently as PIR terms.
   73 class TermLike term tyname name uni fun | term -> tyname name uni fun where
   74     var      :: ann -> name -> term ann
   75     tyAbs    :: ann -> tyname -> Kind ann -> term ann -> term ann
   76     lamAbs   :: ann -> name -> Type tyname uni ann -> term ann -> term ann
   77     apply    :: ann -> term ann -> term ann -> term ann
   78     constant :: ann -> Some (ValueOf uni) -> term ann
   79     builtin  :: ann -> fun -> term ann
   80     tyInst   :: ann -> term ann -> Type tyname uni ann -> term ann
   81     unwrap   :: ann -> term ann -> term ann
   82     iWrap    :: ann -> Type tyname uni ann -> Type tyname uni ann -> term ann -> term ann
   83     error    :: ann -> Type tyname uni ann -> term ann
   84     constr   :: ann -> Type tyname uni ann -> Word64 -> [term ann] -> term ann
   85     kase     :: ann -> Type tyname uni ann -> term ann -> [term ann] -> term ann
   86 
   87     termLet  :: ann -> TermDef term tyname name uni ann -> term ann -> term ann
   88     typeLet  :: ann -> TypeDef tyname uni ann -> term ann -> term ann
   89 
   90     termLet = mkImmediateLamAbs
   91     typeLet = mkImmediateTyAbs

   99 -- | Embed a Haskell value (given its explicit type tag) into a PLC term.
  100 mkConstantOf
  101     :: forall a uni fun term tyname name ann. TermLike term tyname name uni fun
  102     => ann -> uni (Esc a) -> a -> term ann
  103 mkConstantOf ann uni = constant ann . someValueOf uni
  104 
  105 -- | Embed a Haskell value (provided its type is in the universe) into a PLC term.
  106 mkConstant
  107     :: forall a uni fun term tyname name ann.
  108        (TermLike term tyname name uni fun, uni `HasTermLevel` a)
  109     => ann -> a -> term ann
  110 mkConstant ann = constant ann . someValue

  112 instance TermLike (Term tyname name uni fun) tyname name uni fun where
  113     var      = Var
  114     tyAbs    = TyAbs
  115     lamAbs   = LamAbs
  116     apply    = Apply
  117     constant = Constant
  118     builtin  = Builtin
  119     tyInst   = TyInst
  120     unwrap   = Unwrap
  121     iWrap    = IWrap
  122     error    = Error
  123     constr   = Constr
  124     kase     = Case
  125 
  126 embedTerm :: TermLike term tyname name uni fun => Term tyname name uni fun ann -> term ann
  127 embedTerm = \case
  128     Var a n           -> var a n
  129     TyAbs a tn k t    -> tyAbs a tn k (embedTerm t)
  130     LamAbs a n ty t   -> lamAbs a n ty (embedTerm t)
  131     Apply a t1 t2     -> apply a (embedTerm t1) (embedTerm t2)
  132     Constant a c      -> constant a c
  133     Builtin a bi      -> builtin a bi
  134     TyInst a t ty     -> tyInst a (embedTerm t) ty
  135     Error a ty        -> error a ty
  136     Unwrap a t        -> unwrap a (embedTerm t)
  137     IWrap a ty1 ty2 t -> iWrap a ty1 ty2 (embedTerm t)
  138     Constr a ty i es  -> constr a ty i (fmap embedTerm es)
  139     Case a ty arg cs  -> kase a ty (embedTerm arg) (fmap embedTerm cs)

  141 -- | Make a 'Var' referencing the given 'VarDecl'.
  142 -- The @ann@ is propagated from the 'VarDecl' to the 'Var'.
  143 mkVar :: TermLike term tyname name uni fun => VarDecl tyname name uni ann -> term ann
  144 mkVar v = var (_varDeclAnn v) (_varDeclName v)

  152 -- | A definition. Pretty much just a pair with more descriptive names.
  153 data Def var val = Def
  154     { defVar :: var
  155     , defVal :: val
  156     } deriving stock (Show, Eq, Ord, Generic)

  158 -- | A term definition as a variable.
  159 type TermDef term tyname name uni ann = Def (VarDecl tyname name uni ann) (term ann)
  160 -- | A type definition as a type variable.
  161 type TypeDef tyname uni ann = Def (TyVarDecl tyname ann) (Type tyname uni ann)

  163 -- | The type of a PLC function.
  164 data FunctionType tyname uni ann = FunctionType
  165     { _functionTypeAnn :: ann                  -- ^ An annotation.
  166     , _functionTypeDom :: Type tyname uni ann  -- ^ The domain of a function.
  167     , _functionTypeCod :: Type tyname uni ann  -- ^ The codomain of the function.
  168     }
  169 
  170 -- Should we parameterize 'VarDecl' by @ty@ rather than @tyname@, so that we can define
  171 -- 'FunctionDef' as 'TermDef FunctionType tyname name uni fun ann'?
  172 -- Perhaps we even should define general 'Decl' and 'Def' that cover all of the cases?
  173 -- | A PLC function.
  174 data FunctionDef term tyname name uni fun ann = FunctionDef
  175     { _functionDefAnn  :: ann                          -- ^ An annotation.
  176     , _functionDefName :: name                         -- ^ The name of a function.
  177     , _functionDefType :: FunctionType tyname uni ann  -- ^ The type of the function.
  178     , _functionDefTerm :: term ann                     -- ^ The definition of the function.
  179     }
  180 
  181 -- | Convert a 'FunctionType' to the corresponding 'Type'.
  182 functionTypeToType :: FunctionType tyname uni ann -> Type tyname uni ann
  183 functionTypeToType (FunctionType ann dom cod) = TyFun ann dom cod
  184 
  185 -- | Get the type of a 'FunctionDef'.
  186 functionDefToType :: FunctionDef term tyname name uni fun ann -> Type tyname uni ann
  187 functionDefToType (FunctionDef _ _ funTy _) = functionTypeToType funTy
  188 
  189 -- | Convert a 'FunctionDef' to a 'VarDecl'. I.e. ignore the actual term.
  190 functionDefVarDecl :: FunctionDef term tyname name uni fun ann -> VarDecl tyname name uni ann
  191 functionDefVarDecl (FunctionDef ann name funTy _) = VarDecl ann name $ functionTypeToType funTy
  192 
  193 -- | Make a 'FunctionDef'. Return 'Nothing' if the provided type is not functional.
  194 mkFunctionDef
  195     :: ann
  196     -> name
  197     -> Type tyname uni ann
  198     -> term ann
  199     -> Maybe (FunctionDef term tyname name uni fun ann)
  200 mkFunctionDef annName name (TyFun annTy dom cod) term =
  201     Just $ FunctionDef annName name (FunctionType annTy dom cod) term
  202 mkFunctionDef _       _    _                     _    = Nothing

  204 -- | Make a "let-binding" for a term as an immediately applied lambda abstraction.
  205 mkImmediateLamAbs
  206     :: TermLike term tyname name uni fun
  207     => ann
  208     -> TermDef term tyname name uni ann
  209     -> term ann -- ^ The body of the let, possibly referencing the name.
  210     -> term ann
  211 mkImmediateLamAbs ann1 (Def (VarDecl ann2 name ty) bind) body =
  212     apply ann1 (lamAbs ann2 name ty body) bind
  213 
  214 -- | Make a "let-binding" for a type as an immediately instantiated type abstraction. Note: the body must be a value.
  215 mkImmediateTyAbs
  216     :: TermLike term tyname name uni fun
  217     => ann
  218     -> TypeDef tyname uni ann
  219     -> term ann -- ^ The body of the let, possibly referencing the name.
  220     -> term ann
  221 mkImmediateTyAbs ann1 (Def (TyVarDecl ann2 name k) bind) body =
  222     tyInst ann1 (tyAbs ann2 name k body) bind

  224 -- | Make an iterated application. Each `apply` node uses the annotation associated with
  225 -- the corresponding argument.
  226 mkIterApp
  227     :: TermLike term tyname name uni fun
  228     => term ann
  229     -> [(ann, term ann)]
  230     -> term ann
  231 mkIterApp = foldl' $ \acc (ann, arg) -> apply ann acc arg
  232 
  233 -- | Make an iterated application with no annotation.
  234 mkIterAppNoAnn
  235     :: TermLike term tyname name uni fun
  236     => term () -- ^ @f@
  237     -> [term ()] -- ^ @[ x0 ... xn ]@
  238     -> term () -- ^ @[f x0 ... xn ]@
  239 mkIterAppNoAnn term = mkIterApp term . fmap ((),)
  240 
  241 -- | An infix synonym for `mkIterAppNoAnn`
  242 (@@) :: TermLike term tyname name uni fun
  243      => term () -- ^ @f@
  244      -> [term ()] -- ^ @[ x0 ... xn ]@
  245      -> term () -- ^ @[f x0 ... xn ]@
  246 (@@) = mkIterAppNoAnn

  248 -- | Make an iterated instantiation. Each `tyInst` node uses the annotation associated with
  249 -- the corresponding argument.
  250 mkIterInst
  251     :: TermLike term tyname name uni fun
  252     => term ann -- ^ @a@
  253     -> [(ann, Type tyname uni ann)] -- ^ @ [ x0 ... xn ] @
  254     -> term ann -- ^ @{ a x0 ... xn }@
  255 mkIterInst = foldl' $ \acc (ann, arg) -> tyInst ann acc arg

  257 -- | Make an iterated instantiation with no annotation.
  258 mkIterInstNoAnn

  260     => term () -- ^ @a@
  261     -> [Type tyname uni ()] -- ^ @ [ x0 ... xn ] @
  262     -> term () -- ^ @{ a x0 ... xn }@
  263 mkIterInstNoAnn term = mkIterInst term . fmap ((),)

  265 -- | Lambda abstract a list of names.
  266 mkIterLamAbs
  267     :: TermLike term tyname name uni fun
  268     => [VarDecl tyname name uni ann]
  269     -> term ann
  270     -> term ann
  271 mkIterLamAbs args body =
  272     foldr (\(VarDecl ann name ty) acc -> lamAbs ann name ty acc) body args

  278     -> term ann
  279     -> term ann
  280 mkIterTyAbs args body =
  281     foldr (\(TyVarDecl ann name kind) acc -> tyAbs ann name kind acc) body args

  289 mkIterTyApp = foldl' $ \acc (ann, arg) -> TyApp ann acc arg

  298 -- | Make an iterated function type.
  299 mkIterTyFun
  300     :: ann
  301     -> [Type tyname uni ann]
  302     -> Type tyname uni ann
  303     -> Type tyname uni ann
  304 mkIterTyFun ann tys target = foldr (\ty acc -> TyFun ann ty acc) target tys

  307 mkIterTyForall
  308     :: [TyVarDecl tyname ann]

  311 mkIterTyForall args body =
  312     foldr (\(TyVarDecl ann name kind) acc -> TyForall ann name kind acc) body args

  330 {- | A helper to create a single, fresh strict binding; It returns the fresh bound `Var`iable and
  331 a function `Term -> Term`, expecting an "in-Term" to form a let-expression.
  332 -}
  333 mkFreshTermLet :: (MonadQuote m, TermLike t tyname Name uni fun, Monoid a)
  334                => Type tyname uni a -- ^ the type of binding
  335                -> t a -- ^ the term bound to the fresh variable
  336                -> m (t a, t a -> t a) -- ^ the fresh Var and a function that takes an "in" term to construct the Let
  337 mkFreshTermLet aT a = do
  338     -- I wish this was less constrained to Name
  339     genName <- freshName "generated"
  340     pure (var mempty genName, termLet mempty (Def (VarDecl mempty genName aT) a))

  342 -- | 'apply' the head of the application to the arguments iteratively.
  343 headSpineToTerm :: TermLike term tyname name uni fun => ann -> MonoHeadSpine (term ann) -> term ann
  344 headSpineToTerm _ (HeadOnly t)       = t
  345 headSpineToTerm ann (HeadSpine t ts) = foldl (apply ann) t ts
  346 
  347 -- | @headSpineToTerm@ but without annotation.
  348 headSpineToTermNoAnn :: TermLike term tyname name uni fun => MonoHeadSpine (term ()) -> term ()
  349 headSpineToTermNoAnn (HeadOnly t)     = t
  350 headSpineToTermNoAnn (HeadSpine t ts) = foldl (apply ()) t ts
