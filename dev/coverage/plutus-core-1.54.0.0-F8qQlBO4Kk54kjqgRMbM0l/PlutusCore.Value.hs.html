   46 type NestedMap = Map ByteString (Map ByteString Integer)

   48 -- | The underlying type of the UPLC built-in type @Value@.
   49 data Value
   50   = Value
   51       !NestedMap
   52       {- ^ Map from (currency symbol, token name) to amount.
   53 
   54       Invariants: no empty inner map, and no zero amount.
   55       -}
   56       !(IntMap Int)
   57       {- ^ Map from size to the number of inner maps that have that size.
   58       This allows efficient retrieval of the size of the largest inner map,
   59       which is useful for costing of operations like `lookupCoin`.
   60 
   61       Invariant: all values are positive.
   62       -}
   63       {-# UNPACK #-} !Int
   64       {-^ Total size, i.e., sum total of inner map sizes. This avoids recomputing
   65       the total size during the costing of operations like `unionValue`.
   66       -}
   67   deriving stock (Eq, Show, Generic)
   68   deriving anyclass (Hashable, Serialise, NFData)
   69 
   70 {-| Unpack a `Value` into a map from (currency symbol, token name) to amount.
   71 
   72 The map is guaranteed to not contain empty inner map or zero amount.
   73 -}
   74 unpack :: Value -> NestedMap
   75 unpack (Value v _ _) = v
   76 {-# INLINE unpack #-}
   77 
   78 {-| Pack a map from (currency symbol, token name) to amount into a `Value`.
   79 
   80 The map will be filtered so that it does not contain empty inner map or zero amount.
   81 -}
   82 pack :: NestedMap -> Value
   83 pack = pack' . normalize
   84 {-# INLINE pack #-}
   85 
   86 -- | Like `pack` but does not normalize.
   87 pack' :: NestedMap -> Value
   88 pack' v = Value v sizes size
   89  where
   90   (sizes, size) = Map.foldl' alg (mempty, 0) v
   91   alg (ss, s) inner =
   92     ( IntMap.alter (maybe (Just 1) (Just . (+ 1))) (Map.size inner) ss
   93     , s + Map.size inner
   94     )
   95 {-# INLINEABLE pack' #-}
   96 
   97 {-| Total size, i.e., the number of distinct `(currency symbol, token name)` pairs
   98 contained in the `Value`.
   99 -}
  100 totalSize :: Value -> Int
  101 totalSize (Value _ _ size) = size
  102 {-# INLINE totalSize #-}
  103 
  104 -- | Size of the largest inner map.
  105 maxInnerSize :: Value -> Int
  106 maxInnerSize (Value _ sizes _) = maybe 0 fst (IntMap.lookupMax sizes)
  107 {-# INLINE maxInnerSize #-}
  108 
  109 empty :: Value
  110 empty = Value mempty mempty 0
  111 {-# INLINE empty #-}

  113 toList :: Value -> [(ByteString, [(ByteString, Integer)])]
  114 toList = Map.toList . Map.map Map.toList . unpack
  115 {-# INLINEABLE toList #-}
  116 
  117 toFlatList :: Value -> [(ByteString, ByteString, Integer)]
  118 toFlatList (toList -> xs) = [(c, t, a) | (c, ys) <- xs, (t, a) <- ys]
  119 {-# INLINEABLE toFlatList #-}

  121 fromList :: [(ByteString, [(ByteString, Integer)])] -> Value
  122 fromList =
  123   pack
  124     . Map.fromListWith (Map.unionWith (+))
  125     . fmap (second (Map.fromListWith (+)))
  126 
  127 normalize :: NestedMap -> NestedMap
  128 normalize = Map.filter (not . Map.null) . Map.map (Map.filter (/= 0))
  129 {-# INLINEABLE normalize #-}

  131 instance Pretty Value where

  136 {-| \(O(\log \max(m, k))\), where \(m\) is the size of the outer map, and \(k\) is
  137 the size of the largest inner map.
  138 -}
  139 insertCoin :: ByteString -> ByteString -> Integer -> Value -> Value
  140 insertCoin currency token amt v@(Value outer sizes size)
  141   | amt == 0 = deleteCoin currency token v
  142   | otherwise =
  143       let (mold, outer') = Map.alterF f currency outer
  144           (sizes', size') = case mold of
  145             Just old -> (updateSizes old (old + 1) sizes, size + 1)
  146             Nothing  -> (sizes, size)
  147        in Value outer' sizes' size'
  148  where
  149   f
  150     :: Maybe (Map ByteString Integer)
  151     -> ( -- Just (old size of inner map) if the total size grows by 1, otherwise Nothing
  152          Maybe Int
  153        , Maybe (Map ByteString Integer)
  154        )
  155   f = \case
  156     Nothing -> (Just 0, Just (Map.singleton token amt))
  157     Just inner ->
  158       let (isJust -> exists, inner') = Map.insertLookupWithKey (\_ _ _ -> amt) token amt inner
  159        in (if exists then Nothing else Just (Map.size inner), Just inner')
  160 {-# INLINEABLE insertCoin #-}
  161 
  162 -- | \(O(\log \max(m, k))\)
  163 deleteCoin :: ByteString -> ByteString -> Value -> Value
  164 deleteCoin currency token (Value outer sizes size) = Value outer' sizes' size'
  165  where
  166   (mold, outer') = Map.alterF f currency outer
  167   (sizes', size') = case mold of
  168     Just old -> (updateSizes old (old - 1) sizes, size - 1)
  169     Nothing  -> (sizes, size)
  170   f
  171     :: Maybe (Map ByteString Integer)
  172     -> ( -- Just (old size of inner map) if the total size shrinks by 1, otherwise Nothing
  173          Maybe Int
  174        , Maybe (Map ByteString Integer)
  175        )
  176   f = \case
  177     Nothing -> (Nothing, Nothing)
  178     Just inner ->
  179       let (amt, inner') = Map.updateLookupWithKey (\_ _ -> Nothing) token inner
  180        in (amt $> Map.size inner, if Map.null inner' then Nothing else Just inner')
  181 
  182 -- | \(O(\log \max(m, k))\)
  183 lookupCoin :: ByteString -> ByteString -> Value -> Integer
  184 lookupCoin currency token (unpack -> outer) = case Map.lookup currency outer of
  185   Nothing    -> 0
  186   Just inner -> Map.findWithDefault 0 token inner
  187 
  188 {-| \(O(n_{2}\log \max(m_{1}, k_{1}))\), where \(n_{2}\) is the total size of the second
  189 `Value`, \(m_{1}\) is the size of the outer map in the first `Value` and \(k_{1}\) is
  190 the size of the largest inner map in the first `Value`.
  191 
  192 @a@ contains @b@ if for each @(currency, token, amount)@ in @b@, if @amount > 0@, then
  193 @lookup currency token a >= amount@, and if @amount < 0@, then
  194 @lookup currency token a == amount@.
  195 -}
  196 valueContains :: Value -> Value -> Bool
  197 valueContains v = Map.foldrWithKey' go True . unpack
  198  where
  199   go c inner = (&&) (Map.foldrWithKey' goInner True inner)
  200    where
  201     goInner t a2 =
  202       (&&)
  203         ( let a1 = lookupCoin c t v
  204            in if a2 > 0
  205                 then a1 >= a2
  206                 else a1 == a2
  207         )
  208 
  209 {-| The precise complexity is complicated, but an upper bound
  210 is \(O(n_{1} \log n_{2}) + O(m)\), where \(n_{1}\) is the total size of the smaller
  211 value, \(n_{2}\) is the total size of the bigger value, and \(m\) is the
  212 combined size of the outer maps.
  213 -}
  214 unionValue :: Value -> Value -> Value
  215 unionValue (unpack -> vA) (unpack -> vB) =
  216   pack' $
  217     M.merge
  218       M.preserveMissing
  219       M.preserveMissing
  220       ( M.zipWithMaybeMatched $ \_ innerA innerB ->
  221           let inner =
  222                 M.merge
  223                   M.preserveMissing
  224                   M.preserveMissing
  225                   ( M.zipWithMaybeMatched $ \_ x y ->
  226                       let z = x + y in if z == 0 then Nothing else Just z
  227                   )
  228                   innerA
  229                   innerB
  230            in if Map.null inner
  231                 then Nothing
  232                 else
  233                   Just inner
  234       )
  235       vA
  236       vB
  237 {-# INLINEABLE unionValue #-}

  239 {-| \(O(n)\). Encodes `Value` as `Data`, in the same way as non-builtin @Value@.
  240 This is the denotation of @ValueData@ in Plutus V1, V2 and V3.
  241 -}
  242 valueData :: Value -> Data
  243 valueData = Map . fmap (bimap B tokensData) . Map.toList . unpack
  244  where
  245   tokensData :: Map ByteString Integer -> Data
  246   tokensData = Map . fmap (bimap B I) . Map.toList
  247 {-# INLINEABLE valueData #-}
  248 
  249 {-| \(O(n \log n)\). Decodes `Data` into `Value`, in the same way as non-builtin @Value@.
  250 This is the denotation of @UnValueData@ in Plutus V1, V2 and V3.
  251 -}
  252 unValueData :: Data -> BuiltinResult Value
  253 unValueData =
  254   fmap pack . \case
  255     Map cs -> fmap (Map.fromListWith (Map.unionWith (+))) (traverse (bitraverse unB unTokens) cs)
  256     _ -> fail "unValueData: non-Map constructor"
  257  where
  258   unB :: Data -> BuiltinResult ByteString
  259   unB = \case
  260     B b -> pure b
  261     _ -> fail "unValueData: non-B constructor"
  262 
  263   unI :: Data -> BuiltinResult Integer
  264   unI = \case
  265     I i -> pure i
  266     _ -> fail "unValueData: non-I constructor"
  267 
  268   unTokens :: Data -> BuiltinResult (Map ByteString Integer)
  269   unTokens = \case
  270     Map ts -> fmap (Map.fromListWith (+)) (traverse (bitraverse unB unI) ts)
  271     _ -> fail "unValueData: non-Map constructor"
  272 {-# INLINEABLE unValueData #-}

  274 -- | Decrement bucket @old@, and increment bucket @new@.
  275 updateSizes :: Int -> Int -> IntMap Int -> IntMap Int
  276 updateSizes old new = dec . inc
  277  where
  278   inc =
  279     if new == 0
  280       then id
  281       else IntMap.alter (maybe (Just 1) (Just . (+ 1))) new
  282   dec =
  283     if old == 0
  284       then id
  285       else IntMap.update (\n -> if n <= 1 then Nothing else Just (n - 1)) old
  286 {-# INLINEABLE updateSizes #-}
