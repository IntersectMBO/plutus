   54 -- Max length (in bytes) for currency symbols and token names in `Value`,
   55 -- both of which cannot exceed 32 bytes. Currency symbols are in fact either
   56 -- empty or 28 bytes, but for simplicity we allow anything between 0 and 32 bytes.
   57 maxKeyLen :: Int
   58 maxKeyLen = 32
   59 {-# INLINE maxKeyLen #-}

   61 -- | A `ByteString` with maximum length of `maxKeyLen` bytes.
   62 newtype K = UnsafeK {unK :: ByteString}
   63   deriving newtype (Eq, Ord, Show, Hashable, NFData)
   64   deriving stock (Generic)

   66 k :: ByteString -> Maybe K
   67 k b = if B.length b <= maxKeyLen then Just (UnsafeK b) else Nothing
   68 {-# INLINEABLE k #-}

   88 -- | The underlying type of the UPLC built-in type @Value@.
   89 data Value
   90   = Value
   91       !NestedMap
   92       {- ^ Map from (currency symbol, token name) to amount.
   93 
   94       Invariants: no empty inner map, and no zero amount.
   95       -}
   96       !(IntMap Int)
   97       {- ^ Map from size to the number of inner maps that have that size.
   98       This allows efficient retrieval of the size of the largest inner map,
   99       which is useful for costing of operations like `lookupCoin`.
  100 
  101       Invariant: all values are positive.
  102       -}
  103       {-# UNPACK #-} !Int
  104       {- ^ Total size, i.e., sum total of inner map sizes. This avoids recomputing
  105       the total size during the costing of operations like `unionValue`.
  106       -}
  107       {-# UNPACK #-} !Int
  108       -- ^ The number of negative amounts it contains.
  109   deriving stock (Eq, Show, Generic)
  110   deriving anyclass (NFData)

  130 {-| Unpack a `Value` into a map from (currency symbol, token name) to amount.
  131 
  132 The map is guaranteed to not contain empty inner map or zero amount.
  133 -}
  134 unpack :: Value -> NestedMap
  135 unpack (Value v _ _ _) = v
  136 {-# INLINE unpack #-}
  137 
  138 {-| Pack a map from (currency symbol, token name) to amount into a `Value`.
  139 
  140 The map will be filtered so that it does not contain empty inner map or zero amount.
  141 -}
  142 pack :: NestedMap -> Value
  143 pack = pack' . normalize
  144 {-# INLINE pack #-}
  145 
  146 -- | Like `pack` but does not normalize.
  147 pack' :: NestedMap -> Value
  148 pack' v = Value v sizes size neg
  149  where
  150   (sizes, size, neg) = Map.foldl' alg (mempty, 0, 0) v
  151   alg (ss, s, n) inner =
  152     ( IntMap.alter (maybe (Just 1) (Just . (+ 1))) (Map.size inner) ss
  153     , s + Map.size inner
  154     , n + Map.size (Map.filter (< 0) inner)
  155     )
  156 {-# INLINEABLE pack' #-}
  157 
  158 {-| Total size, i.e., the number of distinct `(currency symbol, token name)` pairs
  159 contained in the `Value`.

  161 totalSize :: Value -> Int
  162 totalSize (Value _ _ size _) = size

  165 -- | Size of the largest inner map.
  166 maxInnerSize :: Value -> Int
  167 maxInnerSize (Value _ sizes _ _) = maybe 0 fst (IntMap.lookupMax sizes)

  170 negativeAmounts :: Value -> Int

  186 fromList :: [(K, [(K, Integer)])] -> Value
  187 fromList =
  188   pack
  189     . Map.fromListWith (Map.unionWith (+))
  190     . fmap (second (Map.fromListWith (+)))

  193 normalize :: NestedMap -> NestedMap
  194 normalize = Map.filter (not . Map.null) . Map.map (Map.filter (/= 0))

  202 {-| \(O(\log \max(m, k))\), where \(m\) is the size of the outer map, and \(k\) is
  203 the size of the largest inner map.
  204 -}
  205 insertCoin :: ByteString -> ByteString -> Integer -> Value -> BuiltinResult Value
  206 insertCoin currency token amt v@(Value outer sizes size neg)
  207   | amt == 0 = pure $ deleteCoin currency token v
  208   | otherwise = case (k currency, k token) of
  209       (Nothing, _) -> fail $ "insertCoin: invalid currency: " <> show (B.unpack currency)
  210       (_, Nothing) -> fail $ "insertCoin: invalid token: " <> show (B.unpack token)
  211       (Just ck, Just tk) ->
  212         let f
  213               :: Maybe (Map K Integer)
  214               -> ( -- Left (old size of inner map) if the total size grows by 1,
  215                    -- otherwise, Right (old amount)
  216                    Either Int Integer
  217                  , Maybe (Map K Integer)
  218                  )
  219             f = \case
  220               Nothing -> (Left 0, Just (Map.singleton tk amt))
  221               Just inner ->
  222                 let (moldAmt, inner') =
  223                       Map.insertLookupWithKey (\_ _ _ -> amt) tk amt inner
  224                  in (maybe (Left (Map.size inner)) Right moldAmt, Just inner')
  225             (res, outer') = Map.alterF f ck outer
  226             (sizes', size', neg') = case res of
  227               Left oldSize ->
  228                 ( updateSizes oldSize (oldSize + 1) sizes
  229                 , size + 1
  230                 , if amt < 0 then neg + 1 else neg
  231                 )
  232               Right oldAmt ->
  233                 ( sizes
  234                 , size
  235                 , if oldAmt < 0 && amt > 0
  236                     then neg - 1
  237                     else
  238                       if oldAmt > 0 && amt < 0
  239                         then neg + 1
  240                         else neg
  241                 )
  242          in pure $ Value outer' sizes' size' neg'
  243 {-# INLINEABLE insertCoin #-}

  245 -- | \(O(\log \max(m, k))\)
  246 deleteCoin :: ByteString -> ByteString -> Value -> Value
  247 deleteCoin (UnsafeK -> currency) (UnsafeK -> token) (Value outer sizes size neg) =
  248   Value outer' sizes' size' neg'
  249  where
  250   (mold, outer') = Map.alterF f currency outer
  251   (sizes', size', neg') = case mold of
  252     Just (oldSize, oldAmt) ->
  253       ( updateSizes oldSize (oldSize - 1) sizes
  254       , size - 1
  255       , if oldAmt < 0 then neg - 1 else neg
  256       )
  257     Nothing -> (sizes, size, neg)
  258   f
  259     :: Maybe (Map K Integer)
  260     -> ( -- Just (old size of inner map, old amount) if the total size shrinks by 1,
  261          -- otherwise Nothing
  262          Maybe (Int, Integer)
  263        , Maybe (Map K Integer)
  264        )
  265   f = \case
  266     Nothing -> (Nothing, Nothing)
  267     Just inner ->
  268       let (amt, inner') = Map.updateLookupWithKey (\_ _ -> Nothing) token inner
  269        in ((Map.size inner,) <$> amt, if Map.null inner' then Nothing else Just inner')

  271 -- | \(O(\log \max(m, k))\)
  272 lookupCoin :: ByteString -> ByteString -> Value -> Integer
  273 lookupCoin (UnsafeK -> currency) (UnsafeK -> token) (unpack -> outer) =
  274   case Map.lookup currency outer of
  275     Nothing    -> 0
  276     Just inner -> Map.findWithDefault 0 token inner

  278 {-| \(O(n_{2}\log \max(m_{1}, k_{1}))\), where \(n_{2}\) is the total size of the second
  279 `Value`, \(m_{1}\) is the size of the outer map in the first `Value` and \(k_{1}\) is
  280 the size of the largest inner map in the first `Value`.
  281 
  282 @a@ contains @b@ if for each @(currency, token, amount)@ in @b@, if @amount > 0@, then
  283 @lookup currency token a >= amount@, and if @amount < 0@, then
  284 @lookup currency token a == amount@.
  285 -}
  286 valueContains :: Value -> Value -> BuiltinResult Bool
  287 valueContains v1 v2
  288   | negativeAmounts v1 > 0 = fail "valueContains: first value contains negative amounts"
  289   | negativeAmounts v2 > 0 = fail "valueContains: second value contains negative amounts"
  290   | otherwise = BuiltinSuccess . getAll $ Map.foldrWithKey' go mempty (unpack v2)
  291  where
  292   go c inner = (<>) (Map.foldrWithKey' goInner mempty inner)
  293    where
  294     goInner t a2 = (<>) (All $ lookupCoin (unK c) (unK t) v1 >= a2)
  295 {-# INLINEABLE valueContains #-}

  297 {-| \(O(n_{1}) + O(n_{2})\), where \(n_{1}\) and \(n_{2}\) are the total sizes
  298 (i.e., sum of inner map sizes) of the two maps.
  299 -}
  300 unionValue :: Value -> Value -> Value
  301 unionValue (unpack -> vA) (unpack -> vB) =
  302   pack' $
  303     M.merge
  304       M.preserveMissing
  305       M.preserveMissing
  306       ( M.zipWithMaybeMatched $ \_ innerA innerB ->
  307           let inner =
  308                 M.merge
  309                   M.preserveMissing
  310                   M.preserveMissing
  311                   ( M.zipWithMaybeMatched $ \_ x y ->
  312                       let z = x + y in if z == 0 then Nothing else Just z
  313                   )
  314                   innerA
  315                   innerB
  316            in if Map.null inner
  317                 then Nothing
  318                 else
  319                   Just inner
  320       )
  321       vA
  322       vB
  323 {-# INLINEABLE unionValue #-}

  325 {-| \(O(n)\). Encodes `Value` as `Data`, in the same way as non-builtin @Value@.
  326 This is the denotation of @ValueData@ in Plutus V1, V2 and V3.
  327 -}
  328 valueData :: Value -> Data
  329 valueData = Map . fmap (bimap (B . unK) tokensData) . Map.toList . unpack
  330  where
  331   tokensData :: Map K Integer -> Data
  332   tokensData = Map . fmap (bimap (B . unK) I) . Map.toList
  333 {-# INLINEABLE valueData #-}
  334 
  335 {-| \(O(n \log n)\). Decodes `Data` into `Value`, in the same way as non-builtin @Value@.
  336 This is the denotation of @UnValueData@ in Plutus V1, V2 and V3.
  337 -}
  338 unValueData :: Data -> BuiltinResult Value
  339 unValueData =
  340   fmap pack . \case
  341     Map cs -> fmap (Map.fromListWith (Map.unionWith (+))) (traverse (bitraverse unB unTokens) cs)
  342     _ -> fail "unValueData: non-Map constructor"
  343  where
  344   unB :: Data -> BuiltinResult K
  345   unB = \case
  346     B b -> maybe (fail $ "unValueData: invalid key: " <> show (B.unpack b)) pure (k b)
  347     _ -> fail "unValueData: non-B constructor"
  348 
  349   unI :: Data -> BuiltinResult Integer
  350   unI = \case
  351     I i -> pure i
  352     _ -> fail "unValueData: non-I constructor"
  353 
  354   unTokens :: Data -> BuiltinResult (Map K Integer)
  355   unTokens = \case
  356     Map ts -> fmap (Map.fromListWith (+)) (traverse (bitraverse unB unI) ts)
  357     _ -> fail "unValueData: non-Map constructor"
  358 {-# INLINEABLE unValueData #-}

  360 -- | Decrement bucket @old@, and increment bucket @new@.
  361 updateSizes :: Int -> Int -> IntMap Int -> IntMap Int
  362 updateSizes old new = dec . inc
  363  where
  364   inc =
  365     if new == 0
  366       then id
  367       else IntMap.alter (maybe (Just 1) (Just . (+ 1))) new
  368   dec =
  369     if old == 0
  370       then id
  371       else IntMap.update (\n -> if n <= 1 then Nothing else Just (n - 1)) old
  372 {-# INLINEABLE updateSizes #-}
