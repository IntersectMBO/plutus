   18 -- | '()' as a PLC type.
   19 unit :: uni `HasTypeLevel` () => Type tyname uni ()
   20 unit = mkTyBuiltin @_ @() ()
   21 
   22 -- | '()' as a PLC term.
   23 unitval :: (TermLike term tyname name uni fun, uni `HasTermLevel` ()) => term ()
   24 unitval = mkConstant () ()
   25 
   26 -- | 'seq' specified to '()' as a PLC term.
   27 sequ :: (TermLike term tyname Name uni fun, uni `HasTypeAndTermLevel` ()) => term ()
   28 sequ = runQuote $ do
   29     x <- freshName "x"
   30     y <- freshName "y"
   31     return
   32         . lamAbs () x unit
   33         . lamAbs () y unit
   34         $ unitval
