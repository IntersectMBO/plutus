    3 -- | Functions for computing variable usage inside terms.
    4 module UntypedPlutusCore.Analysis.Usages (termUsages, Usages, getUsageCount, allUsed) where
    5 
    6 import UntypedPlutusCore.Core.Type
    7 import UntypedPlutusCore.Subst
    8 
    9 import PlutusCore qualified as PLC
   10 import PlutusCore.Name.Unique qualified as PLC
   11 
   12 import Control.Lens
   13 
   14 import Data.MultiSet qualified as MSet
   15 import Data.MultiSet.Lens
   16 import Data.Set qualified as Set
   17 
   18 type Usages = MSet.MultiSet PLC.Unique
   19 
   20 -- | Get the usage count of @n@.
   21 getUsageCount :: (PLC.HasUnique n unique) => n -> Usages -> Int
   22 getUsageCount n = MSet.occur (n ^. PLC.unique . coerced)
   23 
   24 -- | Get a set of @n@s which are used at least once.
   25 allUsed :: Usages -> Set.Set PLC.Unique
   26 allUsed = MSet.toSet
   27 
   28 termUsages
   29     :: (PLC.HasUnique name PLC.TermUnique)
   30     => Term name uni fun a
   31     -> Usages
   32 termUsages = multiSetOf (vTerm . PLC.theUnique)
