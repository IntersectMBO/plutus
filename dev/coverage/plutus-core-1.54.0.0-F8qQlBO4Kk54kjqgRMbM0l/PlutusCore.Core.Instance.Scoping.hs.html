   19 instance tyname ~ TyName => Reference TyName (Type tyname uni) where
   20     referenceVia reg tyname ty = TyApp NotAName ty $ TyVar (reg tyname) tyname
   21 
   22 instance tyname ~ TyName => Reference TyName (Term tyname name uni fun) where
   23     referenceVia reg tyname term = TyInst NotAName term $ TyVar (reg tyname) tyname
   24 
   25 instance name ~ Name => Reference Name (Term tyname name uni fun) where
   26     referenceVia reg name term = Apply NotAName term $ Var (reg name) name

   28 -- Kinds have no names, hence the simple instance.
   29 instance EstablishScoping Kind where
   30     establishScoping kind = pure $ NotAName <$ kind
   31 
   32 -- Kinds have no names, hence the simple instance.
   33 instance CollectScopeInfo Kind where
   34     collectScopeInfo _ = mempty

   36 instance tyname ~ TyName => EstablishScoping (Type tyname uni) where
   37     establishScoping (TyLam _ nameDup kind ty) = do
   38         name <- freshenTyName nameDup
   39         establishScopingBinder TyLam name kind ty
   40     establishScoping (TyForall _ nameDup kind ty) = do
   41         name <- freshenTyName nameDup
   42         establishScopingBinder TyForall name kind ty
   43     establishScoping (TyIFix _ pat arg) =
   44         TyIFix NotAName <$> establishScoping pat <*> establishScoping arg
   45     establishScoping (TyApp _ fun arg) =
   46         TyApp NotAName <$> establishScoping fun <*> establishScoping arg
   47     establishScoping (TyFun _ dom cod) =
   48         TyFun NotAName <$> establishScoping dom <*> establishScoping cod
   49     establishScoping (TyVar _ nameDup) = do
   50         name <- freshenTyName nameDup
   51         pure $ TyVar (registerFree name) name
   52     establishScoping (TyBuiltin _ someUni) = pure $ TyBuiltin NotAName someUni
   53     establishScoping (TySOP _ tyls) =
   54         TySOP NotAName <$> (traverse . traverse) establishScoping tyls
   55 
   56 firstBound :: Term tyname name uni fun ann -> [name]
   57 firstBound (Apply _ (LamAbs _ name _ body) _) = name : firstBound body
   58 firstBound _                                  = []

   60 instance (tyname ~ TyName, name ~ Name) => EstablishScoping (Term tyname name uni fun) where
   61     establishScoping (LamAbs _ nameDup ty body)  = do
   62         name <- freshenName nameDup
   63         establishScopingBinder LamAbs name ty body
   64     establishScoping (TyAbs _ nameDup kind body) = do
   65         name <- freshenTyName nameDup
   66         establishScopingBinder TyAbs name kind body
   67     establishScoping (IWrap _ pat arg term) =
   68         IWrap NotAName <$> establishScoping pat <*> establishScoping arg <*> establishScoping term
   69     establishScoping (Apply _ fun arg) =
   70         Apply NotAName <$> establishScoping fun <*> establishScoping arg
   71     establishScoping (Unwrap _ term) = Unwrap NotAName <$> establishScoping term
   72     establishScoping (Error _ ty) = Error NotAName <$> establishScoping ty
   73     establishScoping (TyInst _ term ty) =
   74         TyInst NotAName <$> establishScoping term <*> establishScoping ty
   75     establishScoping (Var _ nameDup) = do
   76         name <- freshenName nameDup
   77         pure $ Var (registerFree name) name
   78     establishScoping (Constant _ con) = pure $ Constant NotAName con
   79     establishScoping (Builtin _ bi) = pure $ Builtin NotAName bi
   80     establishScoping (Constr _ ty i es) =
   81       Constr NotAName <$> establishScoping ty <*> pure i <*> traverse establishScoping es
   82     establishScoping (Case _ ty a es) = do
   83         esScoped <- traverse establishScoping es
   84         let esScopedPoked = addTheRest $ map (\e -> (e, firstBound e)) esScoped
   85             branchBounds = map (snd . fst) esScopedPoked
   86             referenceInBranch ((branch, _), others) = referenceOutOfScope (map snd others) branch
   87         tyScoped <- establishScoping ty
   88         aScoped <- establishScoping a
   89         -- For each of the branches reference (as out-of-scope) the variables bound in that branch
   90         -- in all the other ones, as well as outside of the whole case-expression. This is to check
   91         -- that none of the transformations leak variables outside of the branch they're bound in.
   92         pure . referenceOutOfScope branchBounds $
   93             Case NotAName tyScoped aScoped $ map referenceInBranch esScopedPoked
   94 
   95 instance (tyname ~ TyName, name ~ Name) => EstablishScoping (Program tyname name uni fun) where
   96     establishScoping (Program _ ver term) =
   97         Program NotAName ver <$> establishScoping term

   99 instance tyname ~ TyName => CollectScopeInfo (Type tyname uni) where
  100     collectScopeInfo (TyLam ann name kind ty) =
  101         handleSname ann name <> collectScopeInfo kind <> collectScopeInfo ty
  102     collectScopeInfo (TyForall ann name kind ty) =
  103         handleSname ann name <> collectScopeInfo kind <> collectScopeInfo ty
  104     collectScopeInfo (TyIFix _ pat arg) = collectScopeInfo pat <> collectScopeInfo arg
  105     collectScopeInfo (TyApp _ fun arg) = collectScopeInfo fun <> collectScopeInfo arg
  106     collectScopeInfo (TyFun _ dom cod) = collectScopeInfo dom <> collectScopeInfo cod
  107     collectScopeInfo (TyVar ann name) = handleSname ann name
  108     collectScopeInfo (TyBuiltin _ _) = mempty
  109     collectScopeInfo (TySOP _ tyls) = (foldMap . foldMap) collectScopeInfo tyls
  110 
  111 instance (tyname ~ TyName, name ~ Name) => CollectScopeInfo (Term tyname name uni fun) where
  112     collectScopeInfo (LamAbs ann name ty body)  =
  113         handleSname ann name <> collectScopeInfo ty <> collectScopeInfo body
  114     collectScopeInfo (TyAbs ann name kind body) =
  115         handleSname ann name <> collectScopeInfo kind <> collectScopeInfo body
  116     collectScopeInfo (IWrap _ pat arg term)   =
  117         collectScopeInfo pat <> collectScopeInfo arg <> collectScopeInfo term
  118     collectScopeInfo (Apply _ fun arg) = collectScopeInfo fun <> collectScopeInfo arg
  119     collectScopeInfo (Unwrap _ term) = collectScopeInfo term
  120     collectScopeInfo (Error _ ty) = collectScopeInfo ty
  121     collectScopeInfo (TyInst _ term ty) = collectScopeInfo term <> collectScopeInfo ty
  122     collectScopeInfo (Var ann name) = handleSname ann name
  123     collectScopeInfo (Constant _ _) = mempty
  124     collectScopeInfo (Builtin _ _) = mempty
  125     collectScopeInfo (Constr _ ty _ es) =
  126       collectScopeInfo ty <> foldMap collectScopeInfo es
  127     collectScopeInfo (Case _ ty arg cs) =
  128       collectScopeInfo ty <> collectScopeInfo arg <> foldMap collectScopeInfo cs
  129 
  130 instance (tyname ~ TyName, name ~ Name) => CollectScopeInfo (Program tyname name uni fun) where
  131     collectScopeInfo (Program _ _ term) = collectScopeInfo term
