    9 {- | A 'Name' is a datatype used to identify a variable inside the Plutus Core languages.
   10  Name comparisons are a fundamental part of the domain logic, and comparing 'Text' directly
   11  is inefficient. As a solution to this problem, we provide the 'Unique' type which is an
   12  integer associated to the 'Name', unique to each instantiation of the type. We can,
   13  therefore, compare the integers instead, which is obviously much more cost-effective.
   14 
   15  We distinguish between the names of term variables and type variables by defining the
   16  'TyName' wrapper over 'Name'. Since the code we usually write is polymorphic in the
   17  name type, we want to be able to define a class of names which have an associated 'Unique'.
   18  This class is 'HasUnique', see the definition below.
   19 -}
   20

   56 -- | A 'Name' represents variables/names in Plutus Core.
   57 data Name = Name
   58   { _nameText   :: T.Text
   59   -- ^ The identifier name, for use in error messages.
   60   , _nameUnique :: Unique
   61   -- ^ A 'Unique' assigned to the name, allowing for cheap comparisons in the compiler.
   62   }
   63   deriving stock (Show, Generic, Lift)
   64   deriving anyclass (NFData)
   65 
   66 -- | Allowed characters in the starting position of a non-quoted identifier.
   67 isIdentifierStartingChar :: Char -> Bool
   68 isIdentifierStartingChar c = isAscii c && isAlpha c || c == '_'
   69 
   70 -- | Allowed characters in a non-starting position of a non-quoted identifier.
   71 isIdentifierChar :: Char -> Bool
   72 isIdentifierChar c = isIdentifierStartingChar c || isDigit c || c == '\''
   73 
   74 -- | Allowed characters in a quoted identifier.
   75 isQuotedIdentifierChar :: Char -> Bool
   76 isQuotedIdentifierChar c =
   77   (isAlpha c || isDigit c || isPunctuation c || isSymbol c)
   78     && isAscii c
   79     && c /= '`'
   80 
   81 isValidUnquotedName :: Text -> Bool
   82 isValidUnquotedName n = case T.uncons n of
   83   Just (hd, tl) -> isIdentifierStartingChar hd && T.all isIdentifierChar tl
   84   Nothing       -> False
   85 
   86 {- | Quote the name with backticks if it is not a valid unquoted name.
   87 It does not check whether the given name is a valid quoted name.
   88 -}
   89 toPrintedName :: Text -> Text
   90 toPrintedName txt = if isValidUnquotedName txt then txt else "`" <> txt <> "`"

  107 instance (HasPrettyConfigName config) => PrettyBy config Name where
  108   prettyBy config (Name txt (Unique uniq))
  109     | showsUnique = pretty $ toPrintedName txt <> "-" <> render (pretty uniq)
  110     | otherwise = pretty $ toPrintedName txt
  111     where
  112       PrettyConfigName showsUnique = toPrettyConfigName config

  114 instance Eq Name where
  115   (==) = (==) `on` _nameUnique
  116 
  117 instance Ord Name where
  118   (<=) = (<=) `on` _nameUnique
  119 
  120 -- Hashable follows Eq and Ord in only depending on the unique
  121 instance Hashable Name where
  122   hashWithSalt s = hashWithSalt s . _nameUnique

  124 {-| A unique identifier
  125 This is normally a positive integral number, except
  126 in `LetFloatOut.topUnique` where we make use of a negative unique to signify top-level.
  127 -}
  128 newtype Unique = Unique {unUnique :: Int}
  129   deriving stock (Eq, Show, Ord, Lift)
  130   deriving newtype (Enum, NFData, Pretty, Hashable)

  156 -- | Types which have a textual name attached to them.
  157 class HasText a where
  158   theText :: Lens' a Text
  159 
  160 instance HasText Name where
  161   theText = nameText
  162 
  163 instance HasText TyName where
  164   theText = coerced . theText @Name

  166 -- | Types which have a 'Unique' attached to them, mostly names.
  167 class (Coercible unique Unique) => HasUnique a unique | a -> unique where
  168   unique :: Lens' a unique
  169 
  170   -- | The default implementation of 'HasUnique' for newtypes.
  171   default unique ::
  172     (Wrapped a, HasUnique (Unwrapped a) unique', Coercible unique' unique) =>
  173     Lens' a unique
  174   unique = _Wrapped' . unique . coerced
  175 
  176 instance HasUnique Unique Unique where
  177   unique = id
  178 
  179 instance HasUnique Name TermUnique where
  180   unique = nameUnique . coerced
  181 
  182 instance HasUnique TyName TypeUnique

  184 -- | A lens focused on the 'Unique' of a name.
  185 theUnique :: (HasUnique name unique) => Lens' name Unique
  186 theUnique = unique . coerced
