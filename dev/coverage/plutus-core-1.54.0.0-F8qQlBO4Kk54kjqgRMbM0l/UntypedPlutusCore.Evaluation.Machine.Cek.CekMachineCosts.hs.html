   33 -- | Costs for evaluating AST nodes.  Times should be specified in picoseconds, memory sizes in bytes.
   34 
   35 type CekMachineCosts = CekMachineCostsBase Identity

   37 data CekMachineCostsBase f =
   38     CekMachineCostsBase {
   39       cekStartupCost :: f ExBudget  -- General overhead
   40     , cekVarCost     :: f ExBudget
   41     , cekConstCost   :: f ExBudget
   42     , cekLamCost     :: f ExBudget
   43     , cekDelayCost   :: f ExBudget
   44     , cekForceCost   :: f ExBudget
   45     , cekApplyCost   :: f ExBudget
   46     , cekBuiltinCost :: f ExBudget
   47     -- ^ Just the cost of evaluating a Builtin node, not the builtin itself.
   48     -- There's no entry for Error since we'll be exiting anyway; also, what would

   52     }

   72 -- Charge a unit CPU cost for AST nodes: this allows us to count the number of
   73 -- times each node type is evaluated.  For actual prediction/costing we use
   74 -- a different version of CekMachineCosts: see ExBudgetingDefaults.defaultCekMachineCosts.
   75 unitCekMachineCosts :: CekMachineCosts
   76 unitCekMachineCosts =
   77     CekMachineCostsBase
   78     { cekStartupCost = zeroCost
   79     , cekVarCost     = unitCost
   80     , cekConstCost   = unitCost
   81     , cekLamCost     = unitCost
   82     , cekDelayCost   = unitCost
   83     , cekForceCost   = unitCost
   84     , cekApplyCost   = unitCost
   85     , cekBuiltinCost = unitCost
   86     , cekConstrCost  = unitCost
   87     , cekCaseCost    = unitCost
   88     }
   89   where
   90     zeroCost = Identity $ ExBudget 0 0
   91     unitCost = Identity $ ExBudget 1 0
