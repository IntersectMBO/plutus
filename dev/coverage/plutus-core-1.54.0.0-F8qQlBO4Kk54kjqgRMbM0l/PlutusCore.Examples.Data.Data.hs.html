   29 -- | Right-folding over 'Data' inside PLC currently hardcoded to only ever return @Data@ as a
   30 -- result, 'cause we need to be able to map built-in lists and pairs in the definition of the
   31 -- right fold for 'Data' and we can only do that monomorphically
   32 -- (see Note [Representable built-in functions over polymorphic built-in types]),
   33 -- which forces us to always return a 'Data'.
   34 -- Alternatively we could convert built-in lists and pairs to their non-built-in
   35 -- Scott/Church-encoded forms, map polymorphically and convert back at the call site, but we really
   36 -- only use this definition as a test, so it's fine to make it overly specific for the sake of
   37 -- keeping the actual test trivial.
   38 --
   39 -- > metaTypeLet r = data in
   40 -- >     \(fConstr : integer -> list r -> r)
   41 -- >      (fMap : list (pair r r) -> r)
   42 -- >      (fList : list r -> r)
   43 -- >      (fI : integer -> r)
   44 -- >      (fB : bytestring -> r) ->
   45 -- >          fix {data} {r} \(rec : data -> r) (d : data) ->
   46 -- >              matchData
   47 -- >                  d
   48 -- >                  {r}
   49 -- >                  (\(i : integer) (ds : list data) -> fConstr i (omapList {data} rec ds)
   50 -- >                  (\(es : list (pair data data)) ->
   51 -- >                      fMap (omapList {pair data data} (obothPair {data} rec) es))
   52 -- >                  (\(ds : list data) -> fList (omapList {data} rec ds))
   53 -- >                  fI
   54 -- >                  fB
   55 ofoldrData :: MatchOption -> Term TyName Name DefaultUni (Either DefaultFun ExtensionFun) ()
   56 ofoldrData optMatch = runQuote $ do
   57     let r = dataTy
   58     fConstr <- freshName "fConstr"
   59     fMap    <- freshName "fMap"
   60     fList   <- freshName "fList"
   61     fI      <- freshName "fI"
   62     fB      <- freshName "fB"
   63     rec     <- freshName "rec"
   64     d       <- freshName "d"
   65     i       <- freshName "i"
   66     ds      <- freshName "ds"
   67     es      <- freshName "es"
   68     let listData = mkTyBuiltin @_ @[Data] ()
   69         listR = TyApp () list r
   70         opair a = mkIterTyAppNoAnn pair [a, a]
   71         unwrap' ann = apply ann $ mapFun Left matchData
   72     return
   73         . lamAbs () fConstr (TyFun () integer $ TyFun () listR r)
   74         . lamAbs () fMap (TyFun () (TyApp () list $ opair r) r)
   75         . lamAbs () fList (TyFun () listR r)
   76         . lamAbs () fI (TyFun () integer r)
   77         . lamAbs () fB (TyFun () (mkTyBuiltin @_ @ByteString ()) r)
   78         . apply () (mkIterInstNoAnn fix [dataTy, r])
   79         . lamAbs () rec (TyFun () dataTy r)
   80         . lamAbs () d dataTy
   81         $ mkIterAppNoAnn (tyInst () (unwrap' () (var () d)) r)
   82             [   lamAbs () i integer
   83               . lamAbs () ds listData
   84               $ mkIterAppNoAnn (var () fConstr)
   85                   [ var () i
   86                   , mkIterAppNoAnn (tyInst () (omapList optMatch) dataTy)
   87                       [var () rec, var () ds]
   88                   ]
   89             ,   lamAbs () es (TyApp () list $ opair dataTy)
   90               . apply () (var () fMap)
   91               $ mkIterAppNoAnn (tyInst () (omapList optMatch) $ opair dataTy)
   92                   [ apply () (tyInst () obothPair dataTy) $ var () rec
   93                   , var () es
   94                   ]
   95             ,   lamAbs () ds listData
   96               . apply () (var () fList)
   97               $ mkIterAppNoAnn (tyInst () (omapList optMatch) dataTy)
   98                   [ var () rec
   99                   , var () ds
  100                   ]
  101             , var () fI
  102             , var () fB
  103             ]
