   17 caseReduce
   18     :: (Monad m, CaseBuiltin uni)
   19     => Term name uni fun a
   20     -> SimplifierT name uni fun a m (Term name uni fun a)
   21 caseReduce term = do
   22     let result = transformOf termSubterms processTerm term
   23     recordSimplification term CaseReduce result
   24     return result
   25 
   26 processTerm :: CaseBuiltin uni => Term name uni fun a -> Term name uni fun a
   27 processTerm = \case
   28     -- We could've rewritten those patterns as 'Error' in the 'Nothing' cases, but that would turn a
   29     -- structural error into an operational one, which would be unfortunate, so instead we decided
   30     -- not to fully optimize such scripts, since they aren't valid anyway.
   31     Case ann (Constr _ i args) cs | Just c <- (V.!?) cs (fromIntegral i) ->
   32                                     mkIterApp c ((ann,) <$> args)
   33     Case ann (Constant _ con) cs | Right fXs <- caseBuiltin con cs ->
   34                                     headSpineToTerm ann (second (Constant ann) fXs)
   35 
   36     t -> t
