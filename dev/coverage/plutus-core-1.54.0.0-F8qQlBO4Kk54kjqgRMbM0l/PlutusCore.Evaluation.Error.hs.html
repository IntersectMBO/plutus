   27 {- | The type of errors that can occur during evaluation. There are two kinds of errors:
   28 
   29 1. Structural ones -- these are errors that are indicative of the _structure_ of the program being
   30    wrong. For example, a free variable was encountered during evaluation, a non-function was
   31    applied to an argument or 'tailList' was applied to a non-list.
   32 2. Operational ones -- these are errors that are indicative of the _logic_ of the program being
   33    wrong. For example, 'error' was executed, 'tailList' was applied to an empty list or evaluation
   34    ran out of gas.
   35 
   36 On the chain both of these are just regular failures and we don't distinguish between them there:
   37 if a script fails, it fails, it doesn't matter what the reason was. However in the tests it does
   38 matter why the failure occurred: a structural error may indicate that the test was written
   39 incorrectly while an operational error may be entirely expected.
   40 
   41 In other words, structural errors are \"runtime type errors\" and operational errors are regular
   42 runtime errors. Which means that evaluating an (erased) well-typed program should never produce a
   43 structural error, only an operational one. This creates a sort of \"runtime type system\" for UPLC
   44 and it would be great to stick to it and enforce in tests etc, but we currently don't.
   45 -}
   46 data EvaluationError structural operational
   47     = StructuralError !structural
   48     | OperationalError !operational
   49     deriving stock (Show, Eq, Functor, Generic)
   50     deriving anyclass (NFData)
   51 
   52 instance Bifunctor EvaluationError where
   53     bimap f _ (StructuralError err)  = StructuralError $ f err
   54     bimap _ g (OperationalError err) = OperationalError $ g err
   55     {-# INLINE bimap #-}
   56 
   57 instance Bifoldable EvaluationError where
   58     bifoldMap f _ (StructuralError err)  = f err
   59     bifoldMap _ g (OperationalError err) = g err
   60     {-# INLINE bifoldMap #-}
   61 
   62 instance Bitraversable EvaluationError where
   63     bitraverse f _ (StructuralError err)  = StructuralError <$> f err
   64     bitraverse _ g (OperationalError err) = OperationalError <$> g err
   65     {-# INLINE bitraverse #-}
   66 
   67 instance
   68         ( HasPrettyDefaults config ~ 'True
   69         , PrettyBy config structural, Pretty operational
   70         ) => PrettyBy config (EvaluationError structural operational) where
   71     prettyBy config (StructuralError structural)   = prettyBy config structural
   72     prettyBy _      (OperationalError operational) = pretty operational
   73 
   74 instance (Pretty structural, Pretty operational) =>
   75         Pretty (EvaluationError structural operational) where
   76     pretty (StructuralError structural)   = pretty structural
   77     pretty (OperationalError operational) = pretty operational
