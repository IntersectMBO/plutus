   74 data Kind ann
   75     = Type ann
   76     | KindArrow ann (Kind ann) (Kind ann)
   77     deriving stock (Eq, Show, Functor, Generic, Lift)
   78     deriving anyclass (NFData, Hashable)

   80 -- | The kind of a pattern functor (the first 'Type' argument of 'TyIFix') at a given kind (of the
   81 -- second 'Type' argument of 'TyIFix'):
   82 --
   83 -- > toPatFuncKind k = (k -> *) -> k -> *
   84 toPatFuncKind :: Kind () -> Kind ()
   85 toPatFuncKind k = KindArrow () (KindArrow () k (Type ())) (KindArrow () k (Type ()))
   86 
   87 fromPatFuncKind :: Kind () -> Maybe (Kind ())
   88 fromPatFuncKind (KindArrow () (KindArrow () k1 (Type ())) (KindArrow () k2 (Type ())))
   89     | k1 == k2 = Just k1
   90 fromPatFuncKind _ = Nothing
   91 
   92 -- | Extract all @a_i@ from @a_0 -> a_1 -> ... -> r@.
   93 argsFunKind :: Kind ann -> [Kind ann]
   94 argsFunKind Type{}            = []
   95 argsFunKind (KindArrow _ k l) = k : argsFunKind l

   97 -- | A 'Type' assigned to expressions.
   98 type Type :: GHC.Type -> (GHC.Type -> GHC.Type) -> GHC.Type -> GHC.Type
   99 data Type tyname uni ann
  100     = TyVar ann tyname -- ^ Type variable
  101     | TyFun ann (Type tyname uni ann) (Type tyname uni ann) -- ^ Function type
  102     | TyIFix ann (Type tyname uni ann) (Type tyname uni ann)
  103       -- ^ Fix-point type, for constructing self-recursive types
  104     | TyForall ann tyname (Kind ann) (Type tyname uni ann) -- ^ Polymorphic type
  105     | TyBuiltin ann (SomeTypeIn uni) -- ^ Builtin type
  106     | TyLam ann tyname (Kind ann) (Type tyname uni ann) -- ^ Type lambda
  107     | TyApp ann (Type tyname uni ann) (Type tyname uni ann) -- ^ Type application
  108     | TySOP ann [[Type tyname uni ann]] -- ^ Sum-of-products type
  109     deriving stock (Show, Functor, Generic)
  110     deriving anyclass (NFData)
  111 
  112 -- | Get recursively all the domains and codomains of a type.
  113 -- @splitFunTyParts (A->B->C) = [A, B, C]@
  114 -- @splitFunTyParts (X) = [X]@
  115 splitFunTyParts :: Type tyname uni a -> NE.NonEmpty (Type tyname uni a)
  116 splitFunTyParts = \case
  117     TyFun _ t1 t2 -> t1 NE.<| splitFunTyParts t2
  118     t             -> pure t
  119 
  120 -- | Get the argument types of a function type.
  121 -- @funTyArgs (A->B->C) = [A, B]@
  122 funTyArgs :: Type tyname uni a ->  [Type tyname uni a]
  123 funTyArgs = NE.init . splitFunTyParts
  124 
  125 -- | Get the result type of a function.
  126 -- If not a function, then is the same as `id`
  127 -- @funResultType (A->B->C) = C@
  128 -- @funResultType (X) = X@
  129 funTyResultType :: Type tyname uni a ->  Type tyname uni a
  130 funTyResultType = NE.last . splitFunTyParts

  132 data Term tyname name uni fun ann
  133     = Var ann name -- ^ a named variable
  134     | LamAbs ann name (Type tyname uni ann) (Term tyname name uni fun ann) -- ^ lambda abstraction
  135     | Apply ann (Term tyname name uni fun ann) (Term tyname name uni fun ann) -- ^ application
  136     | TyAbs ann tyname (Kind ann) (Term tyname name uni fun ann) -- ^ type abstraction
  137     | TyInst ann (Term tyname name uni fun ann) (Type tyname uni ann) -- ^ instantiation
  138     | IWrap ann (Type tyname uni ann) (Type tyname uni ann) (Term tyname name uni fun ann) -- ^ wrapping
  139     | Unwrap ann (Term tyname name uni fun ann) -- ^ unwrapping
  140     -- See Note [Constr tag type]
  141     | Constr ann (Type tyname uni ann) Word64 [Term tyname name uni fun ann] -- ^ constructor
  142     | Case ann (Type tyname uni ann) (Term tyname name uni fun ann) [Term tyname name uni fun ann] -- ^ case
  143     | Constant ann (Some (ValueOf uni)) -- ^ constants
  144     | Builtin ann fun -- ^ builtin functions
  145     | Error ann (Type tyname uni ann) -- ^ fail with error
  146     deriving stock (Functor, Generic)
  147 
  148 deriving stock instance (Show tyname, Show name, GShow uni, Everywhere uni Show, Show fun, Show ann, Closed uni)
  149     => Show (Term tyname name uni fun ann)
  150 
  151 deriving anyclass instance (NFData tyname, NFData name, NFData fun, NFData ann, Everywhere uni NFData, Closed uni)
  152     => NFData (Term tyname name uni fun ann)

  210 instance HasUnique tyname TypeUnique => HasUnique (TyVarDecl tyname ann) TypeUnique where
  211     unique f (TyVarDecl ann tyname kind) =
  212         unique f tyname <&> \tyname' -> TyVarDecl ann tyname' kind

  214 instance HasUnique name TermUnique => HasUnique (VarDecl tyname name uni ann) TermUnique where
  215     unique f (VarDecl ann name ty) =
  216         unique f name <&> \name' -> VarDecl ann name' ty

  233 typeAnn :: Type tyname uni ann -> ann
  234 typeAnn (TyVar ann _       ) = ann
  235 typeAnn (TyFun ann _ _     ) = ann
  236 typeAnn (TyIFix ann _ _    ) = ann
  237 typeAnn (TyForall ann _ _ _) = ann
  238 typeAnn (TyBuiltin ann _   ) = ann
  239 typeAnn (TyLam ann _ _ _   ) = ann
  240 typeAnn (TyApp ann _ _     ) = ann
  241 typeAnn (TySOP ann _       ) = ann

  243 termAnn :: Term tyname name uni fun ann -> ann
  244 termAnn (Var ann _       ) = ann
  245 termAnn (TyAbs ann _ _ _ ) = ann
  246 termAnn (Apply ann _ _   ) = ann
  247 termAnn (Constant ann _  ) = ann
  248 termAnn (Builtin  ann _  ) = ann
  249 termAnn (TyInst ann _ _  ) = ann
  250 termAnn (Unwrap ann _    ) = ann
  251 termAnn (IWrap ann _ _ _ ) = ann
  252 termAnn (Error ann _     ) = ann
  253 termAnn (LamAbs ann _ _ _) = ann
  254 termAnn (Constr ann _ _ _) = ann
  255 termAnn (Case ann _ _ _  ) = ann
  256 
  257 -- | Map a function over the set of built-in functions.
  258 mapFun :: (fun -> fun') -> Term tyname name uni fun ann -> Term tyname name uni fun' ann
  259 mapFun f = go where
  260     go (LamAbs ann name ty body)  = LamAbs ann name ty (go body)
  261     go (TyAbs ann name kind body) = TyAbs ann name kind (go body)
  262     go (IWrap ann pat arg term)   = IWrap ann pat arg (go term)
  263     go (Apply ann fun arg)        = Apply ann (go fun) (go arg)
  264     go (Unwrap ann term)          = Unwrap ann (go term)
  265     go (Error ann ty)             = Error ann ty
  266     go (TyInst ann term ty)       = TyInst ann (go term) ty
  267     go (Var ann name)             = Var ann name
  268     go (Constant ann con)         = Constant ann con
  269     go (Builtin ann fun)          = Builtin ann (f fun)
  270     go (Constr ann ty i args)     = Constr ann ty i (map go args)
  271     go (Case ann ty arg cs)       = Case ann ty (go arg) (map go cs)

  273 -- | This is a wrapper to mark the place where the binder is introduced (i.e. LamAbs/TyAbs)
  274 -- and not where it is actually used (TyVar/Var..).
  275 -- This marking allows us to skip the (de)serialization of binders at LamAbs/TyAbs positions
  276 -- iff 'name' is DeBruijn-encoded (level or index). See for example the instance of  'UntypedPlutusCore.Core.Instance.Flat'
  277 newtype Binder name = Binder { unBinder :: name }
  278     deriving stock (Eq, Show)
  279     -- using this generates faster code, see discussion at <https://input-output-rnd.slack.com/archives/C48HA39RS/p1687917638566799>
  280     deriving Functor via Identity

  282 {- Note [Constr tag type]
  283 Constructor tags are not dynamically created, they can only come from the program itself.
  284 
  285 So we do not have to worry about e.g. overflow.
  286 
  287 The main constraints that we have are:
  288 1. The use of the tags in the evaluator must be fast.
  289 2. We don't want to allow nonsensical things (e.g. negative tags)
  290 3. We don't want the format to change unexpectedly
  291 
  292 These are in tension. The structure we use for storing case branches is a list, which is
  293 traditionally indexed with an 'Int' (and most of the other alternatives are the same). That
  294 suggests using 'Int' as the tag, but 'Int' allows negative tags. A 'Word' would be more
  295 appropriate (of fixed size to satisfy 3), but then we need to be able to index lists by
  296 'Word's.
  297 
  298 Well, except there's nothing that _stops_ us indexing lists by 'Word's, we just need to
  299 write our own indexing functions. So that's what we have done.
  300 
  301 We use 'Word64' since it's the actual machine word, so no benefit from using a smaller
  302 one (smaller values will be serialized by flat in fewer bytes anyway).
  303 -}
