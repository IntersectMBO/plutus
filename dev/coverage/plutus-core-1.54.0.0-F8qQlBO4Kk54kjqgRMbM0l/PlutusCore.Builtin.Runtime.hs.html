   16 -- | A 'BuiltinRuntime' represents a possibly partial builtin application, including an empty
   17 -- builtin application (i.e. just the builtin with no arguments).
   18 --
   19 -- Applying or type-instantiating a builtin peels off the corresponding constructor from its
   20 -- 'BuiltinRuntime'.
   21 --
   22 -- 'BuiltinCostedResult' contains the cost (an 'ExBudgetStream') and the result (a
   23 -- @BuiltinResult (HeadSpine val)@) of the builtin application. The cost is stored strictly, since
   24 -- the evaluator is going to look at it and the result is stored lazily, since it's not supposed to
   25 -- be forced before accounting for the cost of the application. If the cost exceeds the available
   26 -- budget, the evaluator discards the result of the builtin application without ever forcing it and
   27 -- terminates with evaluation failure. Allowing the user to compute something that they don't have
   28 -- the budget for would be a major bug.
   29 --
   30 -- Evaluators that ignore the entire concept of costing (e.g. the CK machine) may of course force
   31 -- the result of the builtin application unconditionally.
   32 data BuiltinRuntime val
   33     = BuiltinCostedResult ExBudgetStream ~(BuiltinResult val)
   34     | BuiltinExpectArgument (val -> BuiltinRuntime val)
   35     | BuiltinExpectForce (BuiltinRuntime val)
   36 
   37 instance NoThunks (BuiltinRuntime val) where
   38     wNoThunks ctx = \case
   39         -- Unreachable, because we don't allow nullary builtins and the 'BuiltinArrow' case only
   40         -- checks for WHNF without recursing. Hence we can throw if we reach this clause somehow.
   41         -- TODO: remove the CPP when rest of IOE moves to nothunks>=0.2
   42 #if MIN_VERSION_nothunks(0,2,0)
   43         BuiltinCostedResult _ _    -> pure . Just . ThunkInfo $ Left ctx
   44 #else
   45         -- Plutus has moved to nothunks>=0.2, but some other IOE repos are using nothunks<0.2.
   46         -- As a consequence, cardano-constitution:create-json-envelope cannot be build.
   47         -- This is a workaround to make it build (default is buildable:False). See `cabal.project`
   48         BuiltinCostedResult _ _    -> pure . Just $ ThunkInfo ctx
   49 #endif
   50         -- This one doesn't do much. It only checks that the function stored in the 'BuiltinArrow'
   51         -- is in WHNF. The function may contain thunks inside of it. Not sure if it's possible to do
   52         -- better, since the final 'BuiltinCostedResult' contains a thunk for the result of the
   53         -- builtin application anyway.
   54         BuiltinExpectArgument f    -> noThunks ctx f
   55         BuiltinExpectForce runtime -> noThunks ctx runtime
   56 
   57     showTypeOf = const "PlutusCore.Builtin.Runtime.BuiltinRuntime"
   58 
   59 instance NFData (BuiltinRuntime val) where
   60     -- 'BuiltinRuntime' is strict (verified by the 'NoThunks' tests), hence we only need to force
   61     -- this to WHNF to get it forced to NF.
   62     rnf = rwhnf

   64 {- Note [DO NOT newtype-wrap functions]
   65 If you put a functional type into a @newtype@ rather than strict @data@, the GHC optimizer will be
   66 able to see through it and will feel compelled to rearrange lambdas and case-expressions as it sees
   67 fit, which it often does incorrectly due to @-fpedantic-bottoms@ not being enabled by default.
   68 
   69 We've had a very confusing case of GHC turning
   70 
   71     case y of
   72         y' -> \x -> z
   73 
   74 into
   75 
   76     \x -> case y of
   77         y' -> z
   78 
   79 and ruining carefully set up optimizations
   80 (see https://github.com/IntersectMBO/plutus/pull/4914#issuecomment-1396306606).
   81 
   82 Even strict @data@ is slower than @newtype@, but in practice GHC is going to
   83 worker-wrapper-transform such code making it as efficient as the @newtype@ version.
   84 
   85 So you should never newtype-wrap functions, use strict @data@ instead.
   86 -}

   88 -- See Note [DO NOT newtype-wrap functions].
   89 -- | A @data@ wrapper around a function returning the 'BuiltinRuntime' of a built-in function.
   90 --
   91 -- In order for lookups to be efficient the 'BuiltinRuntime's need to be cached, i.e. pulled out
   92 -- of the function statically. See 'makeBuiltinMeaning' for how we achieve that.
   93 data BuiltinsRuntime fun val = BuiltinsRuntime
   94     { unBuiltinsRuntime :: fun -> BuiltinRuntime val
   95     }
   96 
   97 instance (Bounded fun, Enum fun) => NFData (BuiltinsRuntime fun val) where
   98     -- Force every 'BuiltinRuntime' stored in the environment.
   99     rnf (BuiltinsRuntime env) = foldr (\fun res -> env fun `seq` res) () enumerate
  100 
  101 instance (Bounded fun, Enum fun) => NoThunks (BuiltinsRuntime fun val) where
  102     -- Ensure that every 'BuiltinRuntime' doesn't contain thunks after forcing it initially
  103     -- (we can't avoid the initial forcing, because we can't lookup the 'BuiltinRuntime' without
  104     -- forcing it, see https://stackoverflow.com/q/63441862).
  105     wNoThunks ctx (BuiltinsRuntime env) = allNoThunks $ map (wNoThunks ctx . env) enumerate
  106     showTypeOf = const "PlutusCore.Builtin.Runtime.BuiltinsRuntime"
  107 
  108 builtinRuntimeFailure :: BuiltinError -> BuiltinRuntime val
  109 builtinRuntimeFailure = BuiltinCostedResult (ExBudgetLast mempty) . throwError
  110 -- See Note [INLINE and OPAQUE on error-related definitions].
  111 {-# OPAQUE builtinRuntimeFailure #-}

  113 -- | Look up the runtime info of a built-in function during evaluation.
  114 lookupBuiltin :: fun -> BuiltinsRuntime fun val -> BuiltinRuntime val
  115 lookupBuiltin fun (BuiltinsRuntime env) = env fun
