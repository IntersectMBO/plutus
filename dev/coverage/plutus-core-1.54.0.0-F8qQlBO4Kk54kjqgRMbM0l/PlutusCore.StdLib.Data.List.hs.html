   34 -- | Pattern matching on built-in lists. @matchList {a} xs@ on built-in lists is
   35 -- equivalent to @unwrap xs@ on lists defined in PLC itself (hence why we bind @r@ after @xs@).
   36 --
   37 -- Either
   38 --
   39 -- > /\(a :: *) -> \(xs : list a) -> /\(r :: *) -> (z : r) (f : a -> list a -> r) ->
   40 -- >     matchList
   41 -- >         {a}
   42 -- >         {r}
   43 -- >         z
   44 -- >         f
   45 -- >         xs
   46 --
   47 -- or
   48 --
   49 -- > /\(a :: *) -> \(xs : list a) -> /\(r :: *) -> (z : r) (f : a -> list a -> r) ->
   50 -- >     chooseList
   51 -- >         {a}
   52 -- >         {() -> r}
   53 -- >         xs
   54 -- >         (\(u : ()) -> z)
   55 -- >         (\(u : ()) -> f (head {a} xs) (tail {a} xs))
   56 -- >         ()
   57 --
   58 -- depending on the 'MatchOption' argument.
   59 matchList :: TermLike term TyName Name DefaultUni DefaultFun => MatchOption -> term ()
   60 matchList optMatch = runQuote $ do
   61     a <- freshTyName "a"
   62     r <- freshTyName "r"
   63     xs <- freshName "xs"
   64     z <- freshName "z"
   65     f <- freshName "f"
   66     u <- freshName "u"
   67     let listA = TyApp () list $ TyVar () a
   68         funAtXs fun = apply () (tyInst () (builtin () fun) $ TyVar () a) $ var () xs
   69     return
   70         . tyAbs () a (Type ())
   71         . lamAbs () xs listA
   72         . tyAbs () r (Type ())
   73         . lamAbs () z (TyVar () r)
   74         . lamAbs () f (TyFun () (TyVar () a) . TyFun () listA $ TyVar () r)
   75         $ case optMatch of
   76             UseChoose ->
   77                 mkIterAppNoAnn
   78                     (mkIterInstNoAnn (builtin () ChooseList)
   79                         [ TyVar () a
   80                         , TyFun () unit $ TyVar () r
   81                         ])
   82                     [ var () xs
   83                     , lamAbs () u unit $ var () z
   84                     ,   lamAbs () u unit
   85                       $ mkIterAppNoAnn (var () f)
   86                           [funAtXs HeadList, funAtXs TailList]
   87                     , unitval
   88                     ]

   90 -- |  @foldr@ over built-in lists.
   91 --
   92 -- > /\(a :: *) (r :: *) -> \(f : a -> r -> r) (z : r) ->
   93 -- >     fix {list a} {r} \(rec : list a -> r) (xs : list a) ->
   94 -- >         matchList {a} xs {r} z \(x : a) (xs' : list a) -> f x (rec xs')
   95 foldrList :: TermLike term TyName Name DefaultUni DefaultFun => MatchOption -> term ()
   96 foldrList optMatch = runQuote $ do
   97     a   <- freshTyName "a"
   98     r   <- freshTyName "r"
   99     f   <- freshName "f"
  100     z   <- freshName "z"
  101     rec <- freshName "rec"
  102     xs  <- freshName "xs"
  103     x   <- freshName "x"
  104     xs' <- freshName "xs'"
  105     let listA = TyApp () list $ TyVar () a
  106         unwrap' ann = apply ann . tyInst () (matchList optMatch) $ TyVar () a
  107     -- Copypasted verbatim from @foldrList@ over Scott-encoded lists.
  108     return
  109         . tyAbs () a (Type ())
  110         . tyAbs () r (Type ())
  111         . lamAbs () f (TyFun () (TyVar () a) . TyFun () (TyVar () r) $ TyVar () r)
  112         . lamAbs () z (TyVar () r)
  113         . apply () (mkIterInstNoAnn fix [listA, TyVar () r])
  114         . lamAbs () rec (TyFun () listA $ TyVar () r)
  115         . lamAbs () xs listA
  116         . apply () (apply () (tyInst () (unwrap' () (var () xs)) $ TyVar () r) $ var () z)
  117         . lamAbs () x (TyVar () a)
  118         . lamAbs () xs' listA
  119         $ mkIterAppNoAnn (var () f)
  120             [ var () x
  121             , apply () (var () rec) $ var () xs'
  122             ]

  124 -- |  'foldl\'' as a PLC term.
  125 --
  126 -- > /\(a :: *) (r :: *) -> \(f : r -> a -> r) ->
  127 -- >     fix {r} {list a -> r} \(rec : r -> list a -> r) (z : r) (xs : list a) ->
  128 -- >         matchList {a} xs {r} z \(x : a) (xs' : list a) -> rec (f z x) xs'
  129 foldList :: TermLike term TyName Name DefaultUni DefaultFun => MatchOption -> term ()
  130 foldList optMatch = runQuote $ do
  131     a   <- freshTyName "a"
  132     r   <- freshTyName "r"
  133     f   <- freshName "f"
  134     rec <- freshName "rec"
  135     z   <- freshName "z"
  136     xs  <- freshName "xs"
  137     x   <- freshName "x"
  138     xs' <- freshName "xs'"
  139     let listA = TyApp () list $ TyVar () a
  140         unwrap' ann = apply ann . tyInst () (matchList optMatch) $ TyVar () a
  141     return
  142         . tyAbs () a (Type ())
  143         . tyAbs () r (Type ())
  144         . lamAbs () f (TyFun () (TyVar () r) . TyFun () (TyVar () a) $ TyVar () r)
  145         . apply () (mkIterInstNoAnn fix [TyVar () r, TyFun () listA $ TyVar () r])
  146         . lamAbs () rec (TyFun () (TyVar () r) . TyFun () listA $ TyVar () r)
  147         . lamAbs () z (TyVar () r)
  148         . lamAbs () xs listA
  149         . apply () (apply () (tyInst () (unwrap' () (var () xs)) $ TyVar () r) $ var () z)
  150         . lamAbs () x (TyVar () a)
  151         . lamAbs () xs' listA
  152         . mkIterAppNoAnn (var () rec)
  153         $ [ mkIterAppNoAnn (var () f) [var () z, var () x]
  154           , var () xs'
  155           ]

  157 -- > foldList {integer} {integer} addInteger 0
  158 sum :: TermLike term TyName Name DefaultUni DefaultFun => MatchOption -> term ()
  159 sum optMatch = runQuote $ do
  160     let int = mkTyBuiltin @_ @Integer ()
  161         add = builtin () AddInteger
  162     return
  163         . mkIterAppNoAnn (mkIterInstNoAnn (foldList optMatch) [int, int])
  164         $ [ add , mkConstant @Integer () 0]
  165 
  166 -- > foldrList {integer} {integer} 0 addInteger
