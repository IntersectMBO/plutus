   40 {- Note [Term constructor ordering and numbers]
   41 Ordering of constructors has a small but real effect on efficiency.
   42 It's slightly more efficient to hit the earlier constructors, so it's better to put the more
   43 common ones first.
   44 
   45 The current ordering is based on their *empirically observed* frequency. We should check this
   46 occasionally.
   47 
   48 Additionally, the first 7 (or 3 on 32-bit systems) constructors will get *pointer tags*, which allows
   49 more efficient access when casing on them. So we ideally want to keep the number of constructors
   50 at 7 or fewer.
   51 
   52 We've got 8 constructors, *but* the last one is Error, which is only going to be seen at most
   53 once per program, so it's not too big a deal if it doesn't get a tag.
   54 
   55 See the GHC Notes "Tagging big families" and "Double switching for big families" in
   56 GHC.StgToCmm.Expr for more details.
   57 -}

   59 {- Note [Supported case-expressions]
   60 Originally 'Case' was introduced to only work with 'Constr', which together create the
   61 sums-of-products approach to representing data types.
   62 
   63 However in https://github.com/IntersectMBO/plutus/issues/6602 we decided that the best way to
   64 speed up processing values of built-in types is to extend 'Case' such that it supports pattern
   65 matching on those.
   66 
   67 Currently, 'Case' only supports booleans and integers, but we plan to extend it to lists and data.
   68 
   69 See the @CaseBuiltin DefaultUni@ instance for how casing behaves for supported built-in types.
   70 -}

   72 {-| The type of Untyped Plutus Core terms. Mirrors the type of Typed Plutus Core terms except
   73 
   74 1. all types are removed
   75 2. 'IWrap' and 'Unwrap' are removed
   76 3. type abstractions are replaced with 'Delay'
   77 4. type instantiations are replaced with 'Force'
   78 
   79 The latter two are due to the fact that we don't have value restriction in Typed Plutus Core
   80 and hence a computation can be stuck expecting only a single type argument for the computation
   81 to become unstuck. Therefore we can't just silently remove type abstractions and instantiations and
   82 need to replace them with something else that also blocks evaluation (in order for the semantics
   83 of an erased program to match with the semantics of the original typed one). 'Delay' and 'Force'
   84 serve exactly this purpose.
   85 -}
   86 -- Making all the fields strict gives us a couple of percent in benchmarks
   87 -- See Note [Term constructor ordering and numbers]
   88 data Term name uni fun ann
   89     = Var !ann !name
   90     | LamAbs !ann !name !(Term name uni fun ann)
   91     | Apply !ann !(Term name uni fun ann) !(Term name uni fun ann)
   92     | Force !ann !(Term name uni fun ann)
   93     | Delay !ann !(Term name uni fun ann)
   94     | Constant !ann !(Some (ValueOf uni))
   95     | Builtin !ann !fun
   96     -- This is the cutoff at which constructors won't get pointer tags
   97     -- See Note [Term constructor ordering and numbers]
   98     | Error !ann
   99     -- TODO: worry about overflow, maybe use an Integer
  100     -- See Note [Constr tag type]
  101     | Constr !ann !Word64 ![Term name uni fun ann]
  102     -- See Note [Supported case-expressions].
  103     | Case !ann !(Term name uni fun ann) !(Vector (Term name uni fun ann))
  104     deriving stock (Functor, Generic)
  105 
  106 deriving stock instance (Show name, GShow uni, Everywhere uni Show, Show fun, Show ann, Closed uni)
  107     => Show (Term name uni fun ann)
  108 
  109 deriving anyclass instance (NFData name, NFData fun, NFData ann, Everywhere uni NFData, Closed uni)
  110     => NFData (Term name uni fun ann)

  127 type instance TPLC.UniOf (Term name uni fun ann) = uni

  129 instance TermLike (Term name uni fun) TPLC.TyName name uni fun where
  130     var      = Var
  131     tyAbs    = \ann _ _ -> Delay ann
  132     lamAbs   = \ann name _ -> LamAbs ann name
  133     apply    = Apply
  134     constant = Constant
  135     builtin  = Builtin
  136     tyInst   = \ann term _ -> Force ann term
  137     unwrap   = const id
  138     iWrap    = \_ _ _ -> id
  139     error    = \ann _ -> Error ann
  140     constr   = \ann _ i es -> Constr ann i es
  141     kase     = \ann _ arg cs -> Case ann arg (fromList cs)

  143 instance TPLC.HasConstant (Term name uni fun ()) where
  144     asConstant (Constant _ val) = pure val
  145     asConstant _                = throwError TPLC.notAConstant
  146 
  147     fromConstant = Constant ()

  159 -- | Return the outermost annotation of a 'Term'.
  160 termAnn :: Term name uni fun ann -> ann
  161 termAnn (Constant ann _) = ann
  162 termAnn (Builtin ann _)  = ann
  163 termAnn (Var ann _)      = ann
  164 termAnn (LamAbs ann _ _) = ann
  165 termAnn (Apply ann _ _)  = ann
  166 termAnn (Delay ann _)    = ann
  167 termAnn (Force ann _)    = ann
  168 termAnn (Error ann)      = ann
  169 termAnn (Constr ann _ _) = ann
  170 termAnn (Case ann _ _)   = ann

  172 bindFunM
  173     :: Monad m
  174     => (ann -> fun -> m (Term name uni fun' ann))
  175     -> Term name uni fun ann
  176     -> m (Term name uni fun' ann)
  177 bindFunM f = go where
  178     go (Constant ann val)     = pure $ Constant ann val
  179     go (Builtin ann fun)      = f ann fun
  180     go (Var ann name)         = pure $ Var ann name
  181     go (LamAbs ann name body) = LamAbs ann name <$> go body
  182     go (Apply ann fun arg)    = Apply ann <$> go fun <*> go arg
  183     go (Delay ann term)       = Delay ann <$> go term
  184     go (Force ann term)       = Force ann <$> go term
  185     go (Error ann)            = pure $ Error ann
  186     go (Constr ann i args)    = Constr ann i <$> traverse go args
  187     go (Case ann arg cs)      = Case ann <$> go arg <*> traverse go cs
  188 
  189 bindFun
  190     :: (ann -> fun -> Term name uni fun' ann)
  191     -> Term name uni fun ann
  192     -> Term name uni fun' ann
  193 bindFun f = runIdentity . bindFunM (coerce f)
  194 
  195 mapFun :: (ann -> fun -> fun') -> Term name uni fun ann -> Term name uni fun' ann
  196 mapFun f = bindFun $ \ann fun -> Builtin ann (f ann fun)
