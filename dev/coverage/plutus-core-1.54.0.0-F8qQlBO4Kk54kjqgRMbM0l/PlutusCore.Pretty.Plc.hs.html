   39 -- | Whether to pretty-print PLC errors in full or with some information omitted.
   40 data CondensedErrors
   41     = CondensedErrorsYes
   42     | CondensedErrorsNo
   43     deriving stock (Show, Eq)
   44 
   45 -- | Options for pretty-printing PLC entities.
   46 newtype PrettyConfigPlcOptions = PrettyConfigPlcOptions
   47     { _pcpoCondensedErrors :: CondensedErrors
   48     }
   49     deriving stock (Show)
   50 
   51 -- | Strategy for pretty-printing PLC entities.
   52 data PrettyConfigPlcStrategy
   53     = PrettyConfigPlcClassic (PrettyConfigClassic PrettyConfigName)
   54     | PrettyConfigPlcReadable (PrettyConfigReadable PrettyConfigName)
   55     deriving stock (Show)
   56 
   57 -- | Global configuration used for pretty-printing PLC entities.
   58 data PrettyConfigPlc = PrettyConfigPlc
   59     { _pcpOptions  :: PrettyConfigPlcOptions
   60     , _pcpStrategy :: PrettyConfigPlcStrategy
   61     }
   62     deriving stock (Show)
   63 
   64 type instance HasPrettyDefaults PrettyConfigPlc = 'True
   65 
   66 -- | The "pretty-printable PLC entity" constraint.
   67 type PrettyPlc = PrettyBy PrettyConfigPlc
   68 
   69 -- | A constraint that allows to derive @PrettyBy PrettyConfigPlc@ instances, see below.
   70 type DefaultPrettyPlcStrategy a =
   71        ( PrettyClassic a
   72        , PrettyReadable a
   73        )
   74 
   75 instance HasPrettyConfigName PrettyConfigPlcStrategy where
   76     toPrettyConfigName (PrettyConfigPlcClassic configClassic)   = toPrettyConfigName configClassic
   77     toPrettyConfigName (PrettyConfigPlcReadable configReadable) = toPrettyConfigName configReadable
   78 
   79 instance HasPrettyConfigName PrettyConfigPlc where
   80     toPrettyConfigName = toPrettyConfigName . _pcpStrategy
   81 
   82 instance DefaultPrettyPlcStrategy a => PrettyBy PrettyConfigPlcStrategy (PrettyAny a) where
   83     prettyBy (PrettyConfigPlcClassic  configClassic ) = prettyBy configClassic  . unPrettyAny
   84     prettyBy (PrettyConfigPlcReadable configReadable) = prettyBy configReadable . unPrettyAny
   85 
   86 instance DefaultPrettyPlcStrategy a => PrettyBy PrettyConfigPlc (PrettyAny a) where
   87     prettyBy = prettyBy . _pcpStrategy
   88 
   89 -- | The 'PrettyConfigPlcOptions' used by default:
   90 -- print errors in full.
   91 prettyConfigPlcOptions :: PrettyConfigPlcOptions
   92 prettyConfigPlcOptions = PrettyConfigPlcOptions CondensedErrorsNo
   93 
   94 -- | The 'PrettyConfigPlc' used by default:
   95 -- use the classic view and print neither 'Unique's, nor name attachments.
   96 prettyConfigPlcClassic :: PrettyConfigPlcOptions -> PrettyConfigPlc
   97 prettyConfigPlcClassic opts =
   98     PrettyConfigPlc opts $ PrettyConfigPlcClassic prettyConfigClassic
   99 
  100 -- | The 'PrettyConfigPlc' used for debugging:
  101 -- use the classic view and print 'Unique's, but not name attachments.
  102 prettyConfigPlcClassicSimple :: PrettyConfigPlcOptions -> PrettyConfigPlc
  103 prettyConfigPlcClassicSimple opts =
  104     PrettyConfigPlc opts $ PrettyConfigPlcClassic prettyConfigClassicSimple
  105 
  106 -- | The 'PrettyConfigPlc' used by default and for readability:
  107 -- use the refined view and print 'Unique's but not name attachments.
  108 prettyConfigPlcReadable :: PrettyConfigPlcOptions -> PrettyConfigPlc
  109 prettyConfigPlcReadable opts =
  110     PrettyConfigPlc opts . PrettyConfigPlcReadable $
  111         botPrettyConfigReadable prettyConfigName def
  112 
  113 -- | The 'PrettyConfigPlc' used for debugging and readability:
  114 -- use the refined view and print neither 'Unique's nor name attachments.
  115 prettyConfigPlcReadableSimple :: PrettyConfigPlcOptions -> PrettyConfigPlc
  116 prettyConfigPlcReadableSimple opts =
  117     PrettyConfigPlc opts . PrettyConfigPlcReadable $
  118         botPrettyConfigReadable prettyConfigNameSimple def
  119 
  120 -- | Pretty-print a PLC value in the default mode using the classic view.
  121 prettyPlcClassic :: PrettyPlc a => a -> Doc ann
  122 prettyPlcClassic = prettyBy $ prettyConfigPlcClassic prettyConfigPlcOptions
  123 
  124 -- | Pretty-print a PLC value without unique indices using the classic view.
  125 prettyPlcClassicSimple :: PrettyPlc a => a -> Doc ann
  126 prettyPlcClassicSimple = prettyBy $ prettyConfigPlcClassicSimple prettyConfigPlcOptions
  127 
  128 -- | Pretty-print a PLC value in the default mode using the readable view.
  129 prettyPlcReadable :: PrettyPlc a => a -> Doc ann
  130 prettyPlcReadable = prettyBy $ prettyConfigPlcReadable prettyConfigPlcOptions
  131 
  132 -- | Pretty-print a PLC value without unique indices using the readable view.
  133 prettyPlcReadableSimple :: PrettyPlc a => a -> Doc ann
  134 prettyPlcReadableSimple = prettyBy $ prettyConfigPlcReadableSimple prettyConfigPlcOptions
  135 
  136 -- | Pretty-print a PLC value using the condensed way (see 'CondensedErrors')
  137 -- of pretty-printing PLC errors (in case there are any).
  138 prettyPlcCondensedErrorBy
  139     :: PrettyPlc a => (PrettyConfigPlcOptions -> PrettyConfigPlc) -> a -> Doc ann
  140 prettyPlcCondensedErrorBy toConfig = prettyBy . toConfig $ PrettyConfigPlcOptions CondensedErrorsYes
