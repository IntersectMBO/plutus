    1 {-# LANGUAGE LambdaCase #-}
    2 -- | Definition analysis for untyped Plutus Core.
    3 -- This just adapts term-related code from PlutusCore.Analysis.Definitions;
    4 -- we just re-use the typed machinery to do the hard work here.
    5 module UntypedPlutusCore.Analysis.Definitions
    6     ( termDefs
    7     , runTermDefs
    8     ) where
    9 
   10 import UntypedPlutusCore.Core
   11 
   12 import PlutusCore.Analysis.Definitions (ScopeType (TermScope), UniqueInfos, addDef, addUsage)
   13 import PlutusCore.Error (UniqueError)
   14 import PlutusCore.Name.Unique (HasUnique, TermUnique (TermUnique), Unique (Unique))
   15 
   16 import Control.Lens (forMOf_)
   17 import Control.Monad.State (MonadState, execStateT)
   18 import Control.Monad.Writer (MonadWriter, WriterT (runWriterT))
   19 
   20 -- | Given a UPLC term, add all of its term definitions and usages, including its subterms,
   21 -- to a global map.
   22 termDefs
   23     :: (Ord ann,
   24         HasUnique name TermUnique,
   25         MonadState (UniqueInfos ann) m,
   26         MonadWriter [UniqueError ann] m)
   27     => Term name uni fun ann
   28     -> m ()
   29 termDefs tm = do
   30    forMOf_ termSubtermsDeep tm handleTerm
   31 
   32 handleTerm :: (Ord ann,
   33         HasUnique name TermUnique,
   34         MonadState (UniqueInfos ann) m,
   35         MonadWriter [UniqueError ann] m)
   36     => Term name uni fun ann
   37     -> m ()
   38 handleTerm = \case
   39     Var ann n      ->
   40         addUsage n ann TermScope
   41     LamAbs ann n _ ->
   42         addDef n ann TermScope
   43     _               -> pure ()
   44 
   45 runTermDefs
   46     :: (Ord ann,
   47         HasUnique name TermUnique,
   48         Monad m)
   49     => Term name uni fun ann
   50     -> m (UniqueInfos ann, [UniqueError ann])
   51 runTermDefs = runWriterT . flip execStateT mempty . termDefs
