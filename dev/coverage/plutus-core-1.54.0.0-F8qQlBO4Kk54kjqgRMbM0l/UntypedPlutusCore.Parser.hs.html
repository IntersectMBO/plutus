   38 -- Parsers for UPLC terms
   39 
   40 -- | A parsable UPLC term.
   41 type PTerm = UPLC.Term PLC.Name PLC.DefaultUni PLC.DefaultFun SrcSpan
   42 
   43 conTerm :: Parser PTerm
   44 conTerm = withSpan $ \sp ->
   45     inParens $ UPLC.Constant sp <$> (symbol "con" *> constant)
   46 
   47 builtinTerm :: Parser PTerm
   48 builtinTerm = withSpan $ \sp ->
   49     inParens $ UPLC.Builtin sp <$> (symbol "builtin" *> builtinFunction)
   50 
   51 varTerm :: Parser PTerm
   52 varTerm = withSpan $ \sp ->
   53     UPLC.Var sp <$> name
   54 
   55 lamTerm :: Parser PTerm
   56 lamTerm = withSpan $ \sp ->
   57     inParens $ UPLC.LamAbs sp <$> (symbol "lam" *> trailingWhitespace name) <*> term
   58 
   59 appTerm :: Parser PTerm
   60 appTerm = withSpan $ \sp ->
   61     -- TODO: should not use the same `sp` for all arguments.
   62     inBrackets $ mkIterApp <$> term <*> (fmap (sp,) <$> some term)
   63 
   64 delayTerm :: Parser PTerm
   65 delayTerm = withSpan $ \sp ->
   66     inParens $ UPLC.Delay sp <$> (symbol "delay" *> term)
   67 
   68 forceTerm :: Parser PTerm
   69 forceTerm = withSpan $ \sp ->
   70     inParens $ UPLC.Force sp <$> (symbol "force" *> term)
   71 
   72 errorTerm :: Parser PTerm
   73 errorTerm = withSpan $ \sp ->
   74     inParens $ UPLC.Error sp <$ symbol "error"
   75 
   76 constrTerm :: Parser PTerm
   77 constrTerm = withSpan $ \sp ->
   78     inParens $ do
   79       let maxTag = fromIntegral (maxBound :: Word64)
   80       tag :: Integer <- symbol "constr" *> lexeme Lex.decimal
   81       args <- many term
   82       whenVersion (\v -> v < plcVersion110) $ fail "'constr' is not allowed before version 1.1.0"
   83       when (tag > maxTag) $ fail "constr tag too large: must be a legal Word64 value"
   84       pure $ UPLC.Constr sp (fromIntegral tag) args
   85 
   86 caseTerm :: Parser PTerm
   87 caseTerm = withSpan $ \sp ->
   88     inParens $ do
   89       res <- UPLC.Case sp <$> (symbol "case" *> term) <*> (V.fromList <$> many term)
   90       whenVersion (\v -> v < plcVersion110) $ fail "'case' is not allowed before version 1.1.0"
   91       pure res
   92 
   93 -- | Parser for all UPLC terms.
   94 term :: Parser PTerm
   95 term = leadingWhitespace go
   96   where
   97     go =
   98         choice $ map try [
   99             conTerm
  100             , builtinTerm
  101             , varTerm
  102             , lamTerm
  103             , appTerm
  104             , delayTerm
  105             , forceTerm
  106             , errorTerm
  107             , constrTerm
  108             , caseTerm
  109             ]
  110 
  111 -- | Parser for UPLC programs.
  112 program :: Parser (UPLC.Program PLC.Name PLC.DefaultUni PLC.DefaultFun SrcSpan)
  113 program = leadingWhitespace go
  114   where
  115     go = do
  116         prog <- withSpan $ \sp -> inParens $ do
  117             v <- symbol "program" *> version
  118             withVersion v $ UPLC.Program sp v <$> term
  119         notFollowedBy anySingle
  120         pure prog
  121 
  122 -- | Parse a UPLC term. The resulting program will have fresh names. The underlying monad must be capable
  123 -- of handling any parse errors.
  124 parseTerm :: (MonadError PLC.ParserErrorBundle m, PLC.MonadQuote m) => Text -> m PTerm
  125 parseTerm = parseGen term
  126 
  127 -- | Parse a UPLC program. The resulting program will have fresh names. The
  128 -- underlying monad must be capable of handling any parse errors.  This passes
  129 -- "test" to the parser as the name of the input stream; to supply a name
  130 -- explicity, use `parse program <name> <input>`.`
  131 parseProgram ::
  132     (MonadError PLC.ParserErrorBundle m, PLC.MonadQuote m)
  133     => Text
  134     -> m (UPLC.Program PLC.Name PLC.DefaultUni PLC.DefaultFun SrcSpan)
  135 parseProgram = parseGen program
  136 
  137 -- | Parse and rewrite so that names are globally unique, not just unique within
  138 -- their scope.
  139 parseScoped ::
  140     (MonadError (PLC.Error uni fun SrcSpan) m, PLC.MonadQuote m)
  141     => Text
  142     -> m (UPLC.Program PLC.Name PLC.DefaultUni PLC.DefaultFun SrcSpan)
  143 -- don't require there to be no free variables at this point, we might be parsing an open term
  144 parseScoped =
  145   through (modifyError PLC.UniqueCoherencyErrorE . checkProgram (const True))
  146   <=< rename
  147   <=< modifyError PLC.ParseErrorE . parseProgram
