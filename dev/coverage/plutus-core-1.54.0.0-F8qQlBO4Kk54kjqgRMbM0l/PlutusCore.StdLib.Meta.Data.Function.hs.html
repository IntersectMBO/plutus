   13 -- | 'const' as a PLC term.
   14 --
   15 -- > constPartial t = /\(A :: *) -> \(x : A) -> t
   16 constPartial :: TermLike term TyName Name uni fun => term () -> term ()
   17 constPartial t = runQuote $ do
   18     a <- freshTyName "a"
   19     x <- freshName "x"
   20     return
   21         . tyAbs () a (Type ())
   22         . lamAbs () x (TyVar () a)
   23         $ t
   24 
   25 -- | Eta-expand a function at a given type. Note that this has to be a \"meta\" function
   26 -- for it not force the function it receives and instead directly hide it under a lambda.
   27 --
   28 -- > etaExpand ty fun = \(x : ty) -> fun x
   29 etaExpand :: TermLike term tyname Name uni fun => Type tyname uni () -> term () -> term ()
   30 etaExpand ty fun = runQuote $ do
   31     x <- freshName "x"
   32     return
   33         . lamAbs () x ty
   34         . apply () fun
   35         $ var () x
