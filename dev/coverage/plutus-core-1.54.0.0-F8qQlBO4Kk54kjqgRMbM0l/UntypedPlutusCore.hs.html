   25 -- | Applies one program to another. Fails if the versions do not match
   26 -- and tries to merge annotations.
   27 applyProgram
   28     :: (MonadError ApplyProgramError m, Semigroup a)
   29     => Program name uni fun a
   30     -> Program name uni fun a
   31     -> m (Program name uni fun a)
   32 applyProgram (Program a1 v1 t1) (Program a2 v2 t2) | v1 == v2
   33   = pure $ Program (a1 <> a2) v1 (Apply (termAnn t1 <> termAnn t2) t1 t2)
   34 applyProgram (Program _a1 v1 _t1) (Program _a2 v2 _t2) =
   35     throwError $ MkApplyProgramError v1 v2
