   30 {- Note [Example of a scoping check]
   31 Consider the following type:
   32 
   33     \(x_42 :: *) -> x_42
   34 
   35 This Note describes how we can use this type to check that a renamer handles scoping correctly.
   36 Any other type could be used as well (and in property tests we generate random ones), but the type
   37 above is the simplest example, so we're going to use it.
   38 
   39 First, we traverse the type and freshen every single name in it, which gives us
   40 
   41     \(x_0 :: *) -> x_1
   42 
   43 After this procedure all names in the new type are distinct, not just globally unique -- completely
   44 distinct: all variables are free variables with different uniques and all bindings are distinct and
   45 never referenced.
   46 
   47 Now for each binder we insert one in-scope variable and one out-of-scope one by referencing them
   48 in an added constructor (we could use 'TyFun', but we use 'TyApp', 'cause it has an analogue at the
   49 term level -- 'Apply' and we can also reference a type variable in a 'Term' via a similar
   50 constructor -- 'TyInst'). That gives us
   51 
   52     (\(x_0 :: *) -> x_1 x_0) x_0
   53 
   54 (currently we just decorate the binder with those constructors. In future we could employ a fancier
   55 strategy and go under to the leaves of the term being processed etc).
   56 
   57 The next step is to annotate each name with what is supposed to happen to it once the renaming is
   58 performed.
   59 
   60 1. the @x_0@ binding is supposed to be renamed and hence will disappear
   61 2. the @x_1@ variable is free, so it's supposed to stay free
   62 3. the inner @x_0@ variable is in scope and so is supposed to be renamed
   63 4. the outer @x_0@ is out of scope and so is free and is supposed to stay
   64 
   65 In the actual implementation everything that we did above happens in a single definition.
   66 
   67 After this initial scoping setup is performed, we run the provided renamer (which is not supposed
   68 to touch any annotations) and collect all the available information: which names disappeared,
   69 which didn't, which appeared etc, simultaneously checking that the names that were supposed to
   70 disappear indeed disappeared and the names that were supposed to stay indeed stayed.
   71 
   72 Note that to tell that e.g. a binding disappeared it's crucial that the AST node with the appopriate
   73 annotation is itself preserved, only the name changed. If some pass removes the AST node of a
   74 binding that is supposed to be disappear, it won't be accounted for. Which is precisely what we need
   75 for passes like inlining.
   76 
   77 Once all this scoping information is collected, we run 'checkScopeInfo' to check that the
   78 information is coherent. See its docs for the details on what exactly the checked invariants are.
   79 
   80 The advantage of this approach is that we can pinpoint exactly where what is visible and, just
   81 as importantly, what is not.
   82 -}

   84 data ScopedName
   85     = TypeName TyName
   86     | TermName Name
   87     deriving stock (Show, Eq, Ord)
   88 
   89 isSameScope :: ScopedName -> ScopedName -> Bool
   90 isSameScope TypeName{} TypeName{} = True
   91 isSameScope TermName{} TermName{} = True
   92 isSameScope TypeName{} TermName{} = False
   93 isSameScope TermName{} TypeName{} = False

   95 -- | Staying names.
   96 data Stays
   97     = StaysOutOfScopeVariable  -- ^ An out-of-scope variable does not get renamed and hence stays.
   98     | StaysFreeVariable        -- ^ A free variable does not get renamed and hence stays.
   99     deriving stock (Show, Eq)
  100 
  101 -- | Changing names.
  102 data Disappears
  103     = DisappearsBinding   -- ^ A binding gets renamed and hence the name that it binds disappears.
  104     | DisappearsVariable  -- ^ A bound variable gets renamed and hence its name disappears.
  105     deriving stock (Show, Eq)
  106 
  107 -- | A name either stays or disappears.
  108 data NameAction
  109     = Stays Stays
  110     | Disappears Disappears
  111     deriving stock (Show, Eq)
  112 
  113 data NameAnn
  114     = NameAction NameAction ScopedName
  115     | NotAName
  116     deriving stock (Show, Eq)
  117 
  118 instance Pretty NameAnn where
  119     pretty = viaShow

  121 class ToScopedName name where
  122     toScopedName :: name -> ScopedName
  123 
  124 instance ToScopedName TyName where
  125     toScopedName = TypeName
  126 
  127 instance ToScopedName Name where
  128     toScopedName = TermName
  129 
  130 -- Naming: @introduce*@ for bindings and @register*@ for variables.
  131 
  132 -- | Annotation for a binding saying \"supposed to disappear\".
  133 introduceBound :: ToScopedName name => name -> NameAnn
  134 introduceBound = NameAction (Disappears DisappearsBinding) . toScopedName
  135 
  136 -- | Annotation for a bound variable saying \"supposed to disappear\".
  137 registerBound :: ToScopedName name => name -> NameAnn
  138 registerBound = NameAction (Disappears DisappearsVariable) . toScopedName
  139 
  140 -- | Annotation for an out-of-scope variable saying \"supposed to stay out of scope\".
  141 registerOutOfScope :: ToScopedName name => name -> NameAnn
  142 registerOutOfScope = NameAction (Stays StaysOutOfScopeVariable) . toScopedName
  143 
  144 -- | Annotation for a free variable saying \"supposed to stay free\".
  145 registerFree :: ToScopedName name => name -> NameAnn
  146 registerFree = NameAction (Stays StaysFreeVariable) . toScopedName

  148 class Reference n t where
  149     -- | Take a registering function, apply it to the provided name, create a type\/term variable
  150     -- out of the resulting annotation and the original name and reference that variable in the
  151     -- provided type\/term by prepending a constructor to it mentioning the variable.
  152     referenceVia
  153         :: (forall name. ToScopedName name => name -> NameAnn)
  154         -> n
  155         -> t NameAnn
  156         -> t NameAnn
  157 
  158 -- | Reference the provided variable in the provided type\/term as an in-scope one.
  159 referenceBound :: Reference n t => n -> t NameAnn -> t NameAnn
  160 referenceBound = referenceVia registerBound
  161 
  162 -- | Reference the provided variable in the provided type\/term as an out-of-scope one.
  163 referenceOutOfScope :: Reference n t => n -> t NameAnn -> t NameAnn
  164 referenceOutOfScope = referenceVia registerOutOfScope

  170 -- | Each kind of old and new names.
  171 data ScopeEntry
  172     = DisappearedBindings
  173     | DisappearedVariables
  174     | AppearedBindings
  175     | AppearedVariables
  176     | StayedOutOfScopeVariables
  177     | StayedFreeVariables
  178     deriving stock (Show, Eq, Ord)
  179 
  180 -- | A 'ScopeInfo' is a set of 'ScopedName's for each of the 'ScopeEntry'.
  181 -- If a 'ScopeEntry' is not present in the map, the corresponding set of 'ScopeName's is considered
  182 -- to be empty.
  183 newtype ScopeInfo = ScopeInfo (Map ScopeEntry (Set ScopedName))
  184     deriving stock (Show)
  185 
  186 -- Defining manually because of plan to move to @NoFieldSelectors@.
  187 unScopeInfo :: ScopeInfo -> Map ScopeEntry (Set ScopedName)
  188 unScopeInfo = coerce
  189 
  190 -- | Extract the set stored in the provided 'ScopeInfo' at the provided 'ScopeEntry'.
  191 to :: ScopeEntry -> ScopeInfo -> Set ScopedName
  192 to entry = fromMaybe Set.empty . Map.lookup entry . unScopeInfo
  193 
  194 emptyScopeInfo :: ScopeInfo
  195 emptyScopeInfo = ScopeInfo Map.empty
  196 
  197 -- | Check if a set is empty and report an error with the set embedded in it otherwise.
  198 checkEmptyOn
  199     :: (Set ScopedName -> Set ScopedName -> Set ScopedName)
  200     -> (Set ScopedName -> Set ScopedName -> ScopeError)
  201     -> Set ScopedName
  202     -> Set ScopedName
  203     -> Either ScopeError ()
  204 checkEmptyOn f err s1 s2 = unless (Set.null $ f s1 s2) . Left $ err s1 s2
  205 
  206 -- | Merge two 'ScopeInfo's checking that binders in them do not intersect along the way.
  207 mergeScopeInfo :: ScopeInfo -> ScopeInfo -> Either ScopeError ScopeInfo
  208 mergeScopeInfo si1 si2 = do
  209     let disappearedBindings1 = to DisappearedBindings si1
  210         disappearedBindings2 = to DisappearedBindings si2
  211         appearedBindings1    = to AppearedBindings    si1
  212         appearedBindings2    = to AppearedBindings    si2
  213     checkEmptyOn
  214         Set.intersection
  215         DuplicateBindersInTheInput
  216         disappearedBindings1
  217         disappearedBindings2
  218     checkEmptyOn
  219         Set.intersection
  220         DuplicateBindersInTheOutput
  221         appearedBindings1
  222         appearedBindings2
  223     Right $ coerce (Map.unionWith Set.union) si1 si2

  225 -- We might want to use @Validation@ or something instead of 'Either'.
  226 -- @newtype@-ing it for the sake of providing very convenient 'Semigroup' and 'Monoid' instances.
  227 newtype ScopeErrorOrInfo = ScopeErrorOrInfo (Either ScopeError ScopeInfo)
  228 
  229 -- Defining manually because of plan to move to @NoFieldSelectors@.
  230 unScopeErrorOrInfo :: ScopeErrorOrInfo -> Either ScopeError ScopeInfo
  231 unScopeErrorOrInfo = coerce

  233 instance Semigroup ScopeErrorOrInfo where
  234     ScopeErrorOrInfo errOrInfo1 <> ScopeErrorOrInfo errOrInfo2 =
  235         ScopeErrorOrInfo . join $ mergeScopeInfo <$> errOrInfo1 <*> errOrInfo2
  236 
  237 instance Monoid ScopeErrorOrInfo where
  238     mempty = ScopeErrorOrInfo $ Right emptyScopeInfo

  251 class EstablishScoping t where
  252     {-| Traverse a 't' freshening every name (both at the binding and the use sites)
  253     and annotating the freshened names with either 'DisappearsBinding' or 'StaysFreeVariable'
  254     depending on whether the name occurs at the binding or the use site.
  255 
  256     In addition to that every binder should be decorated with one out-of-scope variable (annotated
  257     with 'StaysOutOfScopeVariable') and one in-scope one (annotated with 'DisappearsVariable').
  258 
  259     Note that no original name occurring in 't' should survive this procedure (and hence we don't
  260     care if any of the freshened names clashes with an original one as all original ones are
  261     supposed to be gone).
  262 
  263     How to provide an implementation:
  264 
  265     1. handle bindings with 'freshen*Name' + 'establishScopingBinder' (or similar)
  266     2. handle variables with 'freshen*Name' + 'registerFree'
  267     3. everything else is direct recursion + 'Applicative' stuff
  268     -}
  269     establishScoping :: t ann -> Quote (t NameAnn)
  270 
  271 -- That will retraverse the same type multiple times. Should we have @referenceListVia@ as a
  272 -- primitive instead and derive 'referenceVia' in terms of it for better performance?
  273 -- Should we only pick an arbitrary sublist of the provided list instead of using the whole list
  274 -- for better performance? That requires enhancing 'Reference' with @Hedgehog.Gen@ or something.
  275 instance Reference n t => Reference [n] t where
  276     referenceVia reg = flip . Prelude.foldr $ referenceVia reg
  277 
  278 instance Reference n t => Reference (NonEmpty n) t where
  279     referenceVia reg = referenceVia reg . NonEmpty.toList

  282 -- it would be nice to somehow do that generically by default.
  283 class CollectScopeInfo t where
  284     {-| Collect scoping information after scoping was established and renaming was performed.
  285 
  286     How to provide an implementation:
  287 
  288     1. handle names (both bindings and variables) with 'handleSname'
  289     2. everything else is direct recursion + 'Monoid' stuff
  290     -}
  291     collectScopeInfo :: t NameAnn -> ScopeErrorOrInfo

  299 -- See Note [Example of a scoping check].
  300 type Scoping t = (EstablishScoping t, CollectScopeInfo t)

  302 -- | Take a constructor for a binder, a name bound by it, a sort (kind\/type), a value of that sort
  303 -- (type\/term) and call 'establishScoping' on both the sort and its value and reassemble the
  304 -- original binder with the annotated sort and its value, but also decorate the reassembled binder
  305 -- with one out-of-scope variable and one in-scope one.
  306 establishScopingBinder
  307     :: (Reference name value, ToScopedName name, EstablishScoping sort, EstablishScoping value)
  308     => (NameAnn -> name -> sort NameAnn -> value NameAnn -> value NameAnn)
  309     -> name
  310     -> sort ann
  311     -> value ann
  312     -> Quote (value NameAnn)
  313 establishScopingBinder binder name sort value = do
  314     sortS <- establishScoping sort
  315     referenceOutOfScope name .
  316         binder (introduceBound name) name sortS .
  317             referenceBound name <$>
  318                 establishScoping value

  324 -- | Every kind of error thrown by the scope checking machinery at different stages.
  325 data ScopeError
  326     = UnannotatedName !ScopedName
  327     | NameChangedItsScope
  328         { _oldName :: !ScopedName
  329         , _newName :: !ScopedName
  330         }
  331     | NameUnexpectedlyDisappeared
  332         { _oldName :: !ScopedName
  333         , _newName :: !ScopedName
  334         }
  335     | NameUnexpectedlyStayed !ScopedName
  336     | DuplicateBindersInTheInput
  337         { _duplicateBindersLeft  :: !(Set ScopedName)
  338         , _duplicateBindersRight :: !(Set ScopedName)
  339         }
  340     | DuplicateBindersInTheOutput !(Set ScopedName) !(Set ScopedName)
  341     | DisappearedBindingsDiscordWithBoundVariables
  342         { _disappearedBindings :: !(Set ScopedName)
  343         , _boundVariables      :: !(Set ScopedName)
  344         }
  345     | DisappearedBindingsDiscordWithOutOfScopeVariables
  346         { _disappearedBindings :: !(Set ScopedName)
  347         , _outOfScopeVariables :: !(Set ScopedName)
  348         }
  349     | AppearedBindingsDiscordWithBoundVariables
  350         { _appearedBindings :: !(Set ScopedName)
  351         , _boundVariables   :: !(Set ScopedName)
  352         }
  353     | DisappearedBindingsClashWithFreeVariables
  354         { _disappearedBindings :: !(Set ScopedName)
  355         , _freeVariables       :: !(Set ScopedName)
  356         }
  357     | DisappearedBindingsClashWithAppearedBindings
  358         { _disppearedBindings :: !(Set ScopedName)
  359         , _appearedBindings   :: !(Set ScopedName)
  360         }
  361     | AppearedBindingsClashWithFreeVariabes
  362         { _appearedBindings :: !(Set ScopedName)
  363         , _freeVariables    :: !(Set ScopedName)
  364         }
  365     deriving stock (Show)
  366 
  367 instance Pretty ScopeError where
  368     pretty = viaShow

  370 -- | Override the set at the provided 'ScopeEntry' to contain only the provided 'ScopedName'.
  371 overrideSname :: ScopeEntry -> ScopedName -> ScopeInfo -> ScopeInfo
  372 overrideSname key = coerce . Map.insert key . Set.singleton
  373 
  374 -- | Use a 'Stays' to handle an unchanged old name.
  375 applyStays :: Stays -> ScopedName -> ScopeInfo
  376 applyStays stays sname = overrideSname key sname emptyScopeInfo where
  377     key = case stays of
  378         StaysOutOfScopeVariable -> StayedOutOfScopeVariables
  379         StaysFreeVariable       -> StayedFreeVariables
  380 
  381 -- | Use a 'Disappears' to handle differing old and new names.
  382 applyDisappears :: Disappears -> ScopedName -> ScopedName -> ScopeInfo
  383 applyDisappears disappears snameOld snameNew =
  384     overrideSname keyNew snameNew $ overrideSname keyOld snameOld emptyScopeInfo where
  385         (keyOld, keyNew) = case disappears of
  386             DisappearsBinding  -> (DisappearedBindings, AppearedBindings)
  387             DisappearsVariable -> (DisappearedVariables, AppearedVariables)
  388 
  389 -- | Use a 'NameAction' to handle an old and a new name.
  390 applyNameAction
  391     :: NameAction -> ScopedName -> ScopedName -> Either ScopeError ScopeInfo
  392 applyNameAction (Stays stays) snameOld snameNew
  393     | snameOld == snameNew = Right $ applyStays stays snameOld
  394     | otherwise            = Left $ NameUnexpectedlyDisappeared snameOld snameNew
  395 applyNameAction (Disappears disappears) snameOld snameNew
  396     | snameOld == snameNew = Left $ NameUnexpectedlyStayed snameOld
  397     | otherwise            = Right $ applyDisappears disappears snameOld snameNew
  398 
  399 -- | Use a 'NameAnn' to handle a new name.
  400 handleSname :: ToScopedName name => NameAnn -> name -> ScopeErrorOrInfo
  401 handleSname ann nameNew = ScopeErrorOrInfo $ do
  402     let snameNew = toScopedName nameNew
  403     case ann of
  404         NotAName -> Left $ UnannotatedName snameNew
  405         NameAction action snameOld ->
  406             if snameOld `isSameScope` snameNew
  407                 then applyNameAction action snameOld snameNew
  408                 else Left $ NameChangedItsScope snameOld snameNew
  409 
  410 symmetricDifference :: Ord a => Set a -> Set a -> Set a
  411 symmetricDifference s t = (s `Set.union` t) `Set.difference` (s `Set.intersection` t)
  412 
  413 {-| Check that each kind of 'Set' from 'ScopeInfo' relates to all other ones in a certain way.
  414 We start with these three relations that are based on the assumption that for each binder we add
  415 at least one out-of-scope variable and at least one in-scope one:
  416 
  417 1. disappeared bindings should be the same as stayed out of scope variables
  418      (ensures that old bindings disappear via renaming and not via removal)
  419 2. disappeared bindings should be the same as disappeared variables
  420      (ensures that old names consistently disappear at the binding and use sites)
  421 3. appeared bindings should be the same as appeared variables
  422      (ensures that new names consistently appear at the binding and use sites)
  423 
  424 Once we've ensured all of that, we're left with only three sets and 3C2 equals 3,
  425 so we only need to consider three more relations:
  426 
  427 1. disappeared bindings should not intersect with free variables
  428      (an internal sanity check)
  429 2. appeared bindings should not intersect with disappeared bindings
  430 3. appeared bindings should not intersect with free variables
  431 
  432 The last two ensure that no new name has an old name's unique.
  433 -}
  434 checkScopeInfo :: BindingRemoval -> ScopeInfo -> Either ScopeError ()
  435 checkScopeInfo bindRem scopeInfo = do
  436     let disappearedBindings       = to DisappearedBindings       scopeInfo
  437         disappearedVariables      = to DisappearedVariables      scopeInfo
  438         appearedBindings          = to AppearedBindings          scopeInfo
  439         appearedVariables         = to AppearedVariables         scopeInfo
  440         stayedOutOfScopeVariables = to StayedOutOfScopeVariables scopeInfo
  441         stayedFreeVariables       = to StayedFreeVariables       scopeInfo
  442     unless (bindRem == BindingRemovalOk) $ do
  443         checkEmptyOn
  444             symmetricDifference
  445             DisappearedBindingsDiscordWithOutOfScopeVariables
  446             disappearedBindings
  447             stayedOutOfScopeVariables
  448     checkEmptyOn
  449         symmetricDifference
  450         DisappearedBindingsDiscordWithBoundVariables
  451         disappearedBindings
  452         disappearedVariables
  453     checkEmptyOn
  454         symmetricDifference
  455         AppearedBindingsDiscordWithBoundVariables
  456         appearedBindings
  457         appearedVariables
  458     checkEmptyOn
  459         Set.intersection
  460         DisappearedBindingsClashWithFreeVariables
  461         disappearedBindings
  462         stayedFreeVariables
  463     checkEmptyOn
  464         Set.intersection
  465         DisappearedBindingsClashWithAppearedBindings
  466         appearedBindings
  467         disappearedBindings
  468     checkEmptyOn
  469         Set.intersection
  470         AppearedBindingsClashWithFreeVariabes
  471         appearedBindings
  472         stayedFreeVariables

  496 -- See Note [Example of a scoping check].
  497 -- | Check if a pass respects scoping.
  498 --
  499 -- Returns the thing that the scoping tests run on, the result of the pass and the scope checking
  500 -- outcome, respectively.
  501 checkRespectsScoping
  502     :: Scoping t
  503     => BindingRemoval
  504     -> (t NameAnn -> t NameAnn)  -- ^ For preparation before running the scoping tests.
  505                                  --   Commonly, either @runQuote . rename@ or @id@.
  506     -> (t NameAnn -> t NameAnn)  -- ^ The runner of the pass.
  507     -> t ann
  508     -> Either (ScopeCheckError t) ()
  509 checkRespectsScoping bindRem prep run thing =
  510     first (ScopeCheckError input prepared output) $
  511         unScopeErrorOrInfo (collectScopeInfo output) >>= checkScopeInfo bindRem
  512   where
  513     input    = runQuote $ establishScoping thing
  514     prepared = prep input
  515     output   = run prepared
