   28 -- | Configuration for the classic pretty-printing.
   29 data PrettyConfigClassic configName = PrettyConfigClassic
   30     { _pccConfigName :: configName  -- ^ How to pretty-print names.
   31     , _pccDisplayAnn :: Bool        -- ^ Whether to display annotations.
   32     }
   33     deriving stock (Show)
   34 
   35 type instance HasPrettyDefaults (PrettyConfigClassic _) = 'True
   36 
   37 -- | The "classically pretty-printable" constraint.
   38 type PrettyClassicBy configName = PrettyBy (PrettyConfigClassic configName)
   39 
   40 type PrettyClassic = PrettyClassicBy PrettyConfigName
   41 
   42 instance configName ~ PrettyConfigName => HasPrettyConfigName (PrettyConfigClassic configName) where
   43     toPrettyConfigName = _pccConfigName
   44 
   45 isEmptyDoc :: Doc ann -> Bool
   46 isEmptyDoc Empty = True
   47 isEmptyDoc _     = False
   48 
   49 -- | Add a pretty-printed annotation to a list of 'Doc's if the given config enables pretty-printing
   50 -- of annotations.
   51 consAnnIf :: Pretty ann => PrettyConfigClassic configName -> ann -> [Doc dann] -> [Doc dann]
   52 consAnnIf config ann rest = filter (not . isEmptyDoc) [pretty ann | _pccDisplayAnn config] ++ rest
   53 
   54 prettyConfigClassic :: PrettyConfigClassic PrettyConfigName
   55 prettyConfigClassic = PrettyConfigClassic prettyConfigName False
   56 
   57 prettyConfigClassicSimple :: PrettyConfigClassic PrettyConfigName
   58 prettyConfigClassicSimple = PrettyConfigClassic prettyConfigNameSimple False
   59 
   60 -- | Pretty-print a value in the default mode using the classic view.
   61 prettyClassic :: PrettyClassic a => a -> Doc ann
   62 prettyClassic = prettyBy prettyConfigClassic
   63 
   64 -- | Pretty-print a value in the simple mode using the classic view.
   65 prettyClassicSimple :: PrettyClassic a => a -> Doc ann
   66 prettyClassicSimple = prettyBy prettyConfigClassicSimple
