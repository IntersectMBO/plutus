   72 -- | Check equality of two 'Term's.
   73 eqTermM
   74     :: (GEq uni, Closed uni, uni `Everywhere` Eq, Eq fun, Eq ann, HasUnique name TermUnique)
   75     => Term name uni fun ann -> Term name uni fun ann -> EqRename (Renaming TermUnique)
   76 eqTermM (Constant ann1 con1) (Constant ann2 con2) = do
   77     eqM ann1 ann2
   78     eqM con1 con2
   79 eqTermM (Builtin ann1 bi1) (Builtin ann2 bi2) = do
   80     eqM ann1 ann2
   81     eqM bi1 bi2
   82 eqTermM (Var ann1 name1) (Var ann2 name2) = do
   83     eqM ann1 ann2
   84     eqNameM name1 name2
   85 eqTermM (LamAbs ann1 name1 body1) (LamAbs ann2 name2 body2) = do
   86     eqM ann1 ann2
   87     withTwinBindings name1 name2 $ eqTermM body1 body2
   88 eqTermM (Apply ann1 fun1 arg1) (Apply ann2 fun2 arg2) = do
   89     eqM ann1 ann2
   90     eqTermM fun1 fun2
   91     eqTermM arg1 arg2
   92 eqTermM (Delay ann1 term1) (Delay ann2 term2) = do
   93     eqM ann1 ann2
   94     eqTermM term1 term2
   95 eqTermM (Force ann1 term1) (Force ann2 term2) = do
   96     eqM ann1 ann2
   97     eqTermM term1 term2
   98 eqTermM (Error ann1) (Error ann2) = eqM ann1 ann2
   99 eqTermM (Constr ann1 i1 args1) (Constr ann2 i2 args2) = do
  100     eqM ann1 ann2
  101     eqM i1 i2
  102     case zipExact args1 args2 of
  103         Just ps -> for_ ps $ \(t1, t2) -> eqTermM t1 t2
  104         Nothing -> empty
  105 eqTermM (Case ann1 a1 cs1) (Case ann2 a2 cs2) = do
  106     eqM ann1 ann2
  107     eqTermM a1 a2
  108     case zipExact (toList cs1) (toList cs2) of
  109         Just ps -> for_ ps $ \(t1, t2) -> eqTermM t1 t2
  110         Nothing -> empty
  111 eqTermM Constant{} _ = empty
  112 eqTermM Builtin{}  _ = empty
  113 eqTermM Var{}      _ = empty
  114 eqTermM LamAbs{}   _ = empty
  115 eqTermM Apply{}    _ = empty
  116 eqTermM Delay{}    _ = empty
  117 eqTermM Force{}    _ = empty
  118 eqTermM Error{}    _ = empty
  119 eqTermM Constr{}    _ = empty
  120 eqTermM Case{}    _ = empty
