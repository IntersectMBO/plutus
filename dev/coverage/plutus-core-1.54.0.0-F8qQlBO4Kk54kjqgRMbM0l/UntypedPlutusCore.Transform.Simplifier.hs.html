   26 newtype SimplifierT name uni fun ann m a =
   27   SimplifierT
   28     { getSimplifierT :: StateT (SimplifierTrace name uni fun ann) m a
   29     }
   30   deriving newtype (Functor, Applicative, Monad, MonadTrans)

   32 instance MonadQuote m => MonadQuote (SimplifierT name uni fun ann m)

   34 runSimplifierT
   35   :: SimplifierT name uni fun ann m a
   36   -> m (a, SimplifierTrace name uni fun ann)
   37 runSimplifierT = flip State.runStateT initSimplifierTrace . getSimplifierT
   38 
   39 evalSimplifierT
   40   :: Monad m => SimplifierT name uni fun ann m a -> m a
   41 evalSimplifierT = flip State.evalStateT initSimplifierTrace . getSimplifierT
   42 
   43 execSimplifierT
   44   :: Monad m => SimplifierT name uni fun ann m a -> m (SimplifierTrace name uni fun ann)
   45 execSimplifierT = flip State.execStateT initSimplifierTrace . getSimplifierT

   47 type Simplifier name uni fun ann = SimplifierT name uni fun ann Identity

   49 runSimplifier :: Simplifier name uni fun ann a -> (a, SimplifierTrace name uni fun ann)
   50 runSimplifier = runIdentity . runSimplifierT

   52 evalSimplifier :: Simplifier name uni fun ann a -> a
   53 evalSimplifier = runIdentity . evalSimplifierT

   55 execSimplifier :: Simplifier name uni fun ann a -> SimplifierTrace name uni fun ann
   56 execSimplifier = runIdentity . execSimplifierT

   58 data SimplifierStage
   59   = FloatDelay
   60   | ForceDelay
   61   | ForceCaseDelay
   62   | CaseOfCase
   63   | CaseReduce
   64   | Inline
   65   | CSE
   66 
   67 data Simplification name uni fun a =
   68   Simplification
   69     { beforeAST :: Term name uni fun a
   70     , stage     :: SimplifierStage
   71     , afterAST  :: Term name uni fun a
   72     }
   73 
   74 -- TODO2: we probably don't want this in memory so after MVP
   75 -- we should consider serializing this to disk
   76 newtype SimplifierTrace name uni fun a =
   77   SimplifierTrace
   78     { simplifierTrace
   79       :: [Simplification name uni fun a]
   80     }
   81 
   82 initSimplifierTrace :: SimplifierTrace name uni fun a
   83 initSimplifierTrace = SimplifierTrace []
   84 
   85 recordSimplification
   86   :: Monad m
   87   => Term name uni fun a
   88   -> SimplifierStage
   89   -> Term name uni fun a
   90   -> SimplifierT name uni fun a m ()
   91 recordSimplification beforeAST stage afterAST =
   92   let simplification = Simplification { beforeAST, stage, afterAST }
   93     in
   94       modify $ \st ->
   95         st { simplifierTrace = simplification : simplifierTrace st }
   96   where
   97     modify f = SimplifierT $ State.modify' f
