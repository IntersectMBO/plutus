   38 -- | 'id' as a PLC term.
   39 --
   40 -- > /\(A :: *) -> \(x : A) -> x
   41 idFun :: TermLike term TyName Name uni fun => term ()
   42 idFun = runQuote $ do
   43     a <- freshTyName "a"
   44     x <- freshName "x"
   45     return
   46         . tyAbs () a (Type ())
   47         . lamAbs () x (TyVar () a)
   48         $ var () x

   50 -- | 'const' as a PLC term.
   51 --
   52 -- > /\(A B :: *) -> \(x : A) (y : B) -> x
   53 const :: TermLike term TyName Name uni fun => term ()
   54 const = runQuote $ do
   55     a <- freshTyName "a"
   56     b <- freshTyName "b"
   57     x <- freshName "x"
   58     y <- freshName "y"
   59     return
   60         . tyAbs () a (Type ())
   61         . tyAbs () b (Type ())
   62         . lamAbs () x (TyVar () a)
   63         . lamAbs () y (TyVar () b)
   64         $ var () x

   66 -- | '($)' as a PLC term.
   67 --
   68 -- > /\(A B :: *) -> \(f : A -> B) (x : A) -> f x
   69 applyFun :: TermLike term TyName Name uni fun => term ()
   70 applyFun = runQuote $ do
   71     a <- freshTyName "a"
   72     b <- freshTyName "b"
   73     f <- freshName "f"
   74     x <- freshName "x"
   75     return
   76         . tyAbs () a (Type ())
   77         . tyAbs () b (Type ())
   78         . lamAbs () f (TyFun () (TyVar () a) $ TyVar () b)
   79         . lamAbs () x (TyVar () a)
   80         $ apply () (var () f) (var () x)

   82 {- Note [Recursion combinators]
   83 We create singly recursive and mutually recursive functions using different combinators.
   84 
   85 For singly recursive functions we use the Z combinator (a strict cousin of the Y combinator) that in
   86 UPLC looks like this:
   87 
   88     \f -> (\s -> s s) (\s -> f (\x -> s s x))
   89 
   90 We have benchmarked its Haskell version at
   91   https://github.com/IntersectMBO/plutus/tree/9538fc9829426b2ecb0628d352e2d7af96ec8204/doc/notes/fomega/z-combinator-benchmarks
   92 and observed that in Haskell there's no detectable difference in performance of functions defined
   93 using explicit recursion versus the Z combinator. However Haskell is a compiled language and Plutus
   94 is interpreted, so it's very likely that natively supporting recursion in Plutus instead of
   95 compiling recursive functions to combinators would significantly boost performance.
   96 
   97 We've tried using
   98 
   99    \f -> (\s -> s s) (\s x -> f (s s) x)
  100 
  101 instead of
  102 
  103    \f -> (\s -> s s) (\s -> f (\x -> s s x))
  104 
  105 and while it worked OK at the PLC level, it wasn't a suitable primitive for compilation of recursive
  106 functions, because it would add laziness in unexpected places, see
  107   https://github.com/IntersectMBO/plutus/issues/5961
  108 so we had to change it.
  109 
  110 We use
  111 
  112    \f -> (\s -> s s) (\s x -> f (s s) x)
  113 
  114 instead of the more standard
  115 
  116    \f -> (\s x -> f (s s) x) (\s x -> f (s s) x)
  117 
  118 because in practice @f@ gets inlined and we wouldn't be able to do so if it occurred twice in the
  119 term. Plus the former also allows us to save on the size of the term.
  120 
  121 For mutually recursive functions we use the 'fixBy' combinator, which is, to the best of our
  122 knowledge, our own invention. It was first described at
  123   https://github.com/IntersectMBO/plutus/blob/067e74f0606fddc5e183dd45209b461e293a6224/doc/notes/fomega/mutual-term-level-recursion/FixN.agda
  124 and fully specified in our "Unraveling recursion: compiling an IR with recursion to System F" paper.
  125 -}

  127 -- | @Self@ as a PLC type.
  128 --
  129 -- > fix \(self :: * -> *) (a :: *) -> self a -> a
  130 selfData :: RecursiveType uni fun ()
  131 selfData = runQuote $ do
  132     self <- freshTyName "self"
  133     a    <- freshTyName "a"
  134     makeRecursiveType () self [TyVarDecl () a $ Type ()]
  135         . TyFun () (TyApp () (TyVar () self) (TyVar () a))
  136         $ TyVar () a

  138 -- | @unroll@ as a PLC term.
  139 --
  140 -- > /\(a :: *) -> \(s : self a) -> unwrap s s
  141 unroll :: TermLike term TyName Name uni fun => term ()
  142 unroll = runQuote $ do
  143     let self = _recursiveType selfData
  144     a <- freshTyName "a"
  145     s <- freshName "s"
  146     return
  147         . tyAbs () a (Type ())
  148         . lamAbs () s (TyApp () self $ TyVar () a)
  149         . apply () (unwrap () $ var () s)
  150         $ var () s

  152 -- | 'fix' as a PLC term.
  153 --
  154 -- > /\(a b :: *) -> \(f : (a -> b) -> a -> b) ->
  155 -- >     unroll {a -> b} (iwrap selfF (a -> b) \(s : self (a -> b)) ->
  156 -- >         f (\(x : a) -> unroll {a -> b} s x))
  157 --
  158 -- See @plutus/runQuote $ docs/fomega/z-combinator-benchmarks@ for details.
  159 fix :: TermLike term TyName Name uni fun => term ()
  160 fix = fst fixAndType
  161 
  162 fixAndType :: TermLike term TyName Name uni fun => (term (), Type TyName uni ())
  163 fixAndType = runQuote $ do
  164     let RecursiveType self wrapSelf = selfData
  165     a <- freshTyName "a"
  166     b <- freshTyName "b"
  167     f <- freshName "f"
  168     s <- freshName "s"
  169     x <- freshName "x"
  170     let funAB = TyFun () (TyVar () a) $ TyVar () b
  171         unrollFunAB = tyInst () unroll funAB
  172     let selfFunAB = TyApp () self funAB
  173     let fixTerm =
  174             tyAbs () a (Type ())
  175             . tyAbs () b (Type ())
  176             . lamAbs () f (TyFun () funAB funAB)
  177             . apply () unrollFunAB
  178             . wrapSelf [funAB]
  179             . lamAbs () s selfFunAB
  180             . apply () (var () f)
  181             . lamAbs () x (TyVar () a)
  182             $ mkIterAppNoAnn unrollFunAB
  183                 [ var () s
  184                 , var () x
  185                 ]
  186     let fixType =
  187             TyForall () a (Type ())
  188             . TyForall () b (Type ())
  189             $ TyFun () (TyFun () funAB funAB) funAB
  190     pure (fixTerm, fixType)

  192 -- | A type that looks like a transformation.
  193 --
  194 -- > trans F G Q : F Q -> G Q
  195 trans :: Type tyname uni () -> Type tyname uni () -> Type tyname uni () -> Quote (Type tyname uni ())
  196 trans f g q = pure $ TyFun () (TyApp () f q) (TyApp () g q)
  197 
  198 -- | A type that looks like a natural transformation, sometimes written 'F ~> G'.
  199 --
  200 -- > natTrans F G : forall Q :: * . F Q -> G Q
  201 natTrans :: Type TyName uni () -> Type TyName uni () -> Quote (Type TyName uni ())
  202 natTrans f g = freshTyName "Q" >>= \q -> TyForall () q (Type ()) <$> trans f g (TyVar () q)
  203 
  204 -- | A type that looks like a natural transformation to Id.
  205 --
  206 -- > natTransId F : forall Q :: * . F Q -> Q
  207 natTransId :: Type TyName uni () -> Quote (Type TyName uni ())
  208 natTransId f = do
  209     q <- freshTyName "Q"
  210     pure $ TyForall () q (Type ()) (TyFun () (TyApp () f (TyVar () q)) (TyVar () q))
  211 
  212 -- | The 'fixBy' combinator.
  213 --
  214 -- > fixBy :
  215 -- >     forall (F :: * -> *) .
  216 -- >     ((F ~> Id) -> (F ~> Id)) ->
  217 -- >     ((F ~> F) -> (F ~> Id))
  218 fixBy :: TermLike term TyName Name uni fun => term ()
  219 fixBy = fst fixByAndType
  220 
  221 fixByAndType :: TermLike term TyName Name uni fun => (term (), Type TyName uni ())
  222 fixByAndType = runQuote $ do
  223     f <- freshTyName "F"
  224 
  225     -- by : (F ~> Id) -> (F ~> Id)
  226     by <- freshName "by"
  227     byTy <- do
  228         nt1 <- natTransId (TyVar () f)
  229         nt2 <- natTransId (TyVar () f)
  230         pure $ TyFun () nt1 nt2
  231 
  232     resTy <- do
  233         nt1 <- natTrans (TyVar () f) (TyVar () f)
  234         nt2 <- natTransId (TyVar () f)
  235         pure $ TyFun () nt1 nt2
  236 
  237     -- instantiatedFix = fix {F ~> F} {F ~> Id}
  238     instantiatedFix <- do
  239         nt1 <- natTrans (TyVar () f) (TyVar () f)
  240         nt2 <- natTransId (TyVar () f)
  241         pure $ tyInst () (tyInst () fix nt1) nt2
  242 
  243     -- rec : (F ~> F) -> (F ~> Id)
  244     recc <- freshName "rec"
  245     reccTy <- do
  246       nt <- natTrans (TyVar () f) (TyVar () f)
  247       nt2 <- natTransId (TyVar () f)
  248       pure $ TyFun () nt nt2
  249 
  250     -- h : F ~> F
  251     h <- freshName "h"
  252     hty <- natTrans (TyVar () f) (TyVar () f)
  253 
  254     -- R :: *
  255     -- fr : F R
  256     r <- freshTyName "R"
  257     fr <- freshName "fr"
  258     let frTy = TyApp () (TyVar () f) (TyVar () r)
  259 
  260     -- Q :: *
  261     -- fq : F Q
  262     q <- freshTyName "Q"
  263     fq <- freshName "fq"
  264     let fqTy = TyApp () (TyVar () f) (TyVar () q)
  265 
  266     -- inner = (/\ (Q :: *) -> \ q : F Q -> rec h {Q} (h {Q} q))
  267     let inner =
  268             apply () (var () by) $
  269                 tyAbs () q (Type ()) $
  270                 lamAbs () fq fqTy $
  271                 apply () (tyInst () (apply () (var () recc) (var () h)) (TyVar () q)) $
  272                 apply () (tyInst () (var () h) (TyVar () q)) (var () fq)
  273     let fixByTerm =
  274             tyAbs () f (KindArrow () (Type ()) (Type ())) $
  275             lamAbs () by byTy $
  276             apply () instantiatedFix $
  277             lamAbs () recc reccTy $
  278             lamAbs () h hty $
  279             tyAbs () r (Type ()) $
  280             lamAbs () fr frTy $
  281             apply () (tyInst () inner (TyVar () r)) (var () fr)
  282     let fixByType =
  283             TyForall () f (KindArrow () (Type ()) (Type ())) $
  284             TyFun () byTy resTy
  285     pure (fixByTerm, fixByType)

  288 -- | Make a @n@-ary fixpoint combinator.
  289 --
  290 -- > FixN n :
  291 -- >     forall A1 B1 ... An Bn :: * .
  292 -- >     (forall Q :: * .
  293 -- >         ((A1 -> B1) -> ... -> (An -> Bn) -> Q) ->
  294 -- >         (A1 -> B1) ->
  295 -- >         ... ->
  296 -- >         (An -> Bn) ->
  297 -- >         Q) ->
  298 -- >     (forall R :: * . ((A1 -> B1) -> ... (An -> Bn) -> R) -> R)
  299 fixN :: TermLike term TyName Name uni fun => Integer -> term () -> term ()
  300 fixN n fixByTerm = fst (fixNAndType n fixByTerm)
  301 
  302 fixNAndType :: TermLike term TyName Name uni fun => Integer -> term () -> (term (), Type TyName uni ())
  303 fixNAndType n fixByTerm = runQuote $ do
  304     -- the list of pairs of A and B types
  305     asbs <- replicateM (fromIntegral n) $ do
  306         a <- freshTyName "a"
  307         b <- freshTyName "b"
  308         pure (a, b)
  309 
  310     let abFuns = fmap (\(a, b) -> TyFun () (TyVar () a) (TyVar () b)) asbs
  311     let abTyVars = concatMap (\(a, b) -> [ TyVarDecl () a (Type ()), TyVarDecl () b (Type ())]) asbs
  312 
  313     -- funTysTo X = (A1 -> B1) -> ... -> (An -> Bn) -> X
  314     let funTysTo = mkIterTyFun () abFuns
  315 
  316     -- the type of fixN, as in the header comment
  317     fixNType <- do
  318         q <- freshTyName "Q"
  319         let qvar = TyVar () q
  320         let argTy = TyForall () q (Type ()) (TyFun () (funTysTo qvar) (funTysTo qvar))
  321         r <- freshTyName "R"
  322         let rvar = TyVar () r
  323         let resTy = TyForall () r (Type ()) (TyFun () (funTysTo rvar) rvar)
  324         let fullTy = mkIterTyForall abTyVars $ TyFun () argTy resTy
  325         pure fullTy
  326 
  327     -- instantiatedFix = fixBy { \X :: * -> (A1 -> B1) -> ... -> (An -> Bn) -> X }
  328     instantiatedFix <- do
  329         x <- freshTyName "X"
  330         pure $ tyInst () fixByTerm (TyLam () x (Type ()) (funTysTo (TyVar () x)))
  331 
  332     -- f : forall Q :: * . ((A1 -> B1) -> ... -> (An -> Bn) -> Q) -> (A1 -> B1) -> ... -> (An -> Bn) -> Q)
  333     f <- freshName "f"
  334     fTy <- do
  335         q <- freshTyName "Q"
  336         pure $ TyForall () q (Type ()) $ TyFun () (funTysTo (TyVar () q)) (funTysTo (TyVar () q))
  337 
  338     -- k : forall Q :: * . ((A1 -> B1) -> ... -> (An -> Bn) -> Q) -> Q)
  339     k <- freshName "k"
  340     kTy <- do
  341         q <- freshTyName "Q"
  342         pure $ TyForall () q (Type ()) $ TyFun () (funTysTo (TyVar () q)) (TyVar () q)
  343 
  344     s <- freshTyName "S"
  345 
  346     -- h : (A1 -> B1) -> ... -> (An -> Bn) -> S
  347     h <- freshName "h"
  348     let hTy = funTysTo (TyVar () s)
  349 
  350     -- branch (ai, bi) i = \x : ai -> k { bi } \(f1 : A1 -> B1) ... (fn : An -> Bn) . fi x
  351     let branch (a, b) i = do
  352             -- names and types for the f arguments
  353             fs <- ifor asbs $ \j (a',b') -> do
  354                 f_j <- freshName $ "f_" <> showText j
  355                 pure $ VarDecl () f_j (TyFun () (TyVar () a') (TyVar () b'))
  356 
  357             x <- freshName "x"
  358 
  359             pure $
  360                 lamAbs () x (TyVar () a) $
  361                 apply () (tyInst () (var () k) (TyVar () b)) $
  362                 mkIterLamAbs fs $
  363                 -- this is an ugly but straightforward way of getting the right fi
  364                 apply () (mkVar (fs !! i)) (var () x)
  365 
  366     -- a list of all the branches
  367     branches <- forM (zip asbs [0..]) $ uncurry branch
  368 
  369     -- [A1, B1, ..., An, Bn]
  370     let allAsBs = foldMap (\(a, b) -> [a, b]) asbs
  371     let fixNTerm =
  372           -- abstract out all the As and Bs
  373           mkIterTyAbs (fmap (\tn -> TyVarDecl () tn (Type ())) allAsBs) $
  374           lamAbs () f fTy $
  375           mkIterAppNoAnn instantiatedFix
  376               [ lamAbs () k kTy $
  377                 tyAbs () s (Type ()) $
  378                 lamAbs () h hTy $
  379                 mkIterAppNoAnn (var () h) branches
  380               , var () f
  381               ]
  382     pure (fixNTerm, fixNType)

  384 -- See Note [Recursion combinators].
  385 -- | Get the fixed-point of a single recursive function.
  386 getSingleFixOf
  387     :: (TermLike term TyName Name uni fun)
  388     => ann -> term ann -> FunctionDef term TyName Name uni fun ann -> term ann
  389 getSingleFixOf ann fix1 fun@FunctionDef{_functionDefType=(FunctionType _ dom cod)} =
  390     let instantiatedFix = mkIterInst fix1 [(ann, dom), (ann, cod)]
  391         abstractedBody = mkIterLamAbs [functionDefVarDecl fun] $ _functionDefTerm fun
  392     in apply ann instantiatedFix abstractedBody

  394 -- See Note [Recursion combinators].
  395 -- | Get the fixed-point of a list of mutually recursive functions.
  396 --
  397 -- > MutualFixOf _ fixN [ FunctionDef _ fN1 (FunctionType _ a1 b1) f1
  398 -- >                    , ...
  399 -- >                    , FunctionDef _ fNn (FunctionType _ an bn) fn
  400 -- >                    ] =
  401 -- >     Tuple [(a1 -> b1) ... (an -> bn)] $
  402 -- >         fixN {a1} {b1} ... {an} {bn}
  403 -- >             /\(q :: *) -> \(choose : (a1 -> b1) -> ... -> (an -> bn) -> q) ->
  404 -- >                 \(fN1 : a1 -> b1) ... (fNn : an -> bn) -> choose f1 ... fn
  405 getMutualFixOf
  406     :: (TermLike term TyName Name uni fun)
  407     => ann -> term ann -> [FunctionDef term TyName Name uni fun ann] -> Quote (Tuple term uni ann)
  408 getMutualFixOf ann fixn funs = do
  409     let funTys = map functionDefToType funs
  410 
  411     q <- liftQuote $ freshTyName "Q"
  412     -- TODO: It was 'safeFreshName' previously. Should we perhaps have @freshName = safeFreshName@?
  413     choose <- freshName "choose"
  414     let chooseTy = mkIterTyFun ann funTys (TyVar ann q)
  415 
  416     -- \v1 ... vn -> choose f1 ... fn
  417     let rhss    = map _functionDefTerm funs
  418         chosen  = mkIterApp (var ann choose) ((ann,) <$> rhss)
  419         vsLam   = mkIterLamAbs (map functionDefVarDecl funs) chosen
  420 
  421     -- abstract out Q and choose
  422     let cLam = tyAbs ann q (Type ann) $ lamAbs ann choose chooseTy vsLam
  423 
  424     -- fixN {A1} {B1} ... {An} {Bn}
  425     instantiatedFix <- do
  426         let domCods = foldMap (\(FunctionDef _ _ (FunctionType _ dom cod) _) -> [dom, cod]) funs
  427         pure $ mkIterInst fixn ((ann,) <$> domCods)
  428 
  429     let term = apply ann instantiatedFix cLam
  430 
  431     pure $ Tuple funTys term
