   16 -- | Church-encoded @Nat@ as a PLC type.
   17 --
   18 -- > all (r :: *). r -> (r -> r) -> r
   19 churchNat :: Type TyName uni ()
   20 churchNat = runQuote $ do
   21     r <- freshTyName "r"
   22     return
   23         . TyForall () r (Type ())
   24         . TyFun () (TyVar () r)
   25         . TyFun () (TyFun () (TyVar () r) $ TyVar () r)
   26         $ TyVar () r
   27 
   28 -- | Church-encoded '0' as a PLC term.
   29 --
   30 -- > /\(r :: *) -> \(z : r) (f : r -> r) -> z
   31 churchZero :: TermLike term TyName Name uni fun => term ()
   32 churchZero = runQuote $ do
   33     r <- freshTyName "r"
   34     z <- freshName "z"
   35     f <- freshName "f"
   36     return
   37         . tyAbs  () r (Type ())
   38         . lamAbs () z (TyVar () r)
   39         . lamAbs () f (TyFun () (TyVar () r) $ TyVar () r)
   40         $ var () z
   41 
   42 -- | Church-encoded 'succ' as a PLC term.
   43 --
   44 -- > \(n : nat) -> /\(r :: *) -> \(z : r) (f : r -> r) -> f (n {r} z f)
   45 churchSucc :: TermLike term TyName Name uni fun => term ()
   46 churchSucc = runQuote $ do
   47     n <- freshName "n"
   48     r <- freshTyName "r"
   49     z <- freshName "z"
   50     f <- freshName "f"
   51     return
   52         . lamAbs () n churchNat
   53         . tyAbs  () r (Type ())
   54         . lamAbs () z (TyVar () r)
   55         . lamAbs () f (TyFun () (TyVar () r) $ TyVar () r)
   56         . apply  () (var () f)
   57         $ mkIterAppNoAnn (tyInst () (var () n) $ TyVar () r)
   58           [ var () z
   59           , var () f
   60           ]
