   73 {- Note [PLC types and universes]
   74 We encode built-in types in PLC as tags for Haskell types (the latter are also called meta-types),
   75 see Note [Universes]. A built-in type in PLC is an inhabitant of
   76 
   77     Some (TypeIn uni)
   78 
   79 where @uni@ is some universe, i.e. a collection of tags that have meta-types associated with them.
   80 
   81 A value of a built-in type is a regular Haskell value stored in
   82 
   83     Some (ValueOf uni)
   84 
   85 (together with the tag associated with its type) and such a value is also called a meta-constant.
   86 
   87 The default universe has the following constructor (pattern synonym actually):
   88 
   89     DefaultUniList :: !(DefaultUni a) -> DefaultUni [a]
   90 
   91 But note that this doesn't directly lead to interop between Plutus Core and Haskell, i.e. you can't
   92 have a meta-list whose elements are of a PLC type. You can only have a meta-list constant with
   93 elements of a meta-type (i.e. a type from the universe).
   94 
   95 However it is possible to apply a built-in type to an arbitrary PLC/PIR type, since we can embed
   96 a type of an arbitrary kind into 'Type' and then apply it via 'TyApp'. But we only use it to
   97 get polymorphic built-in functions over polymorphic built-in types, since it's not possible
   98 to juggle values of polymorphic built-in types instantiated with non-built-in types at runtime
   99 (it's not even possible to represent such a value in the AST, even though it's possible to represent
  100 such a 'Type').
  101 
  102 Finally, it is not necessary to allow embedding PLC terms into meta-constants.
  103 We already allow built-in functions with polymorphic types. There might be a way to utilize this
  104 feature and have meta-constructors as built-in functions.
  105 -}

  134 instance GEq DefaultUni where
  135     -- We define 'geq' manually instead of using 'deriveGEq', because the latter creates a single
  136     -- recursive definition and we want two instead. The reason why we want two is because this
  137     -- allows GHC to inline the initial step that appears non-recursive to GHC, because recursion
  138     -- is hidden in the other function that is marked as @OPAQUE@ and is chosen by GHC as a
  139     -- loop-breaker, see https://wiki.haskell.org/Inlining_and_Specialisation#What_is_a_loop-breaker
  140     -- (we're not really sure if this is a reliable solution, but if it stops working, we won't miss
  141     -- very much and we've failed to settle on any other approach).
  142     --
  143     -- This trick gives us a 1% speedup across validation benchmarks (some are up to 4% faster) and
  144     -- a more sensible generated Core where things like @geq DefaulUniBool@ are reduced away.
  145     geq = geqStep where
  146         geqStep :: DefaultUni a1 -> DefaultUni a2 -> Maybe (a1 :~: a2)
  147         geqStep DefaultUniInteger a2 = do
  148             DefaultUniInteger <- Just a2
  149             Just Refl
  150         geqStep DefaultUniByteString a2 = do
  151             DefaultUniByteString <- Just a2
  152             Just Refl
  153         geqStep DefaultUniString a2 = do
  154             DefaultUniString <- Just a2
  155             Just Refl
  156         geqStep DefaultUniUnit a2 = do
  157             DefaultUniUnit <- Just a2
  158             Just Refl
  159         geqStep DefaultUniBool a2 = do
  160             DefaultUniBool <- Just a2
  161             Just Refl
  162         geqStep DefaultUniProtoList a2 = do
  163             DefaultUniProtoList <- Just a2
  164             Just Refl
  165         geqStep DefaultUniProtoArray a2 = do
  166             DefaultUniProtoArray <- Just a2
  167             Just Refl
  168         geqStep DefaultUniProtoPair a2 = do
  169             DefaultUniProtoPair <- Just a2
  170             Just Refl
  171         geqStep (DefaultUniApply f1 x1) a2 = do
  172             DefaultUniApply f2 x2 <- Just a2
  173             Refl <- geqRec f1 f2
  174             Refl <- geqRec x1 x2
  175             Just Refl
  176         geqStep DefaultUniData a2 = do
  177             DefaultUniData <- Just a2
  178             Just Refl
  179         geqStep DefaultUniBLS12_381_G1_Element a2 = do
  180             DefaultUniBLS12_381_G1_Element <- Just a2
  181             Just Refl
  182         geqStep DefaultUniBLS12_381_G2_Element a2 = do
  183             DefaultUniBLS12_381_G2_Element <- Just a2
  184             Just Refl
  185         geqStep DefaultUniBLS12_381_MlResult a2 = do
  186             DefaultUniBLS12_381_MlResult <- Just a2
  187             Just Refl
  188         geqStep DefaultUniValue a2 = do
  189             DefaultUniValue <- Just a2
  190             Just Refl
  191         {-# INLINE geqStep #-}
  192 
  193         geqRec :: DefaultUni a1 -> DefaultUni a2 -> Maybe (a1 :~: a2)
  194         geqRec = geqStep
  195         {-# OPAQUE geqRec #-}

  375 {- Note [Integral types as Integer]
  376 Technically our universe only contains 'Integer', but many of the builtin functions that we would
  377 like to use work over 'Int', 'Word8' etc.
  378 
  379 This is inconvenient and also error-prone: dealing with a function that takes an 'Int' or 'Word8'
  380 means carefully downcasting the 'Integer', running the function, potentially upcasting at the end.
  381 And it's easy to get wrong by e.g. blindly using 'fromInteger' or 'fromIntegral'.
  382 
  383 Moreover, there is a latent risk here: if we *were* to build on a 32-bit architecture, then programs
  384 which use arguments between @maxBound :: Int32@ and @maxBound :: Int64@ would behave differently!
  385 
  386 So, what to do? We adopt the following strategy:
  387 - We allow lifting/unlifting bounded integral types such as 'Word8' or 'Int64' via 'Integer',
  388   including a safe upcast in 'makeKnown' (which we could have on any architecture, but we only add
  389   it for 64-bit for uniformity) and a safe checked downcast in 'readKnown'.
  390 - We allow lifting/unlifting 'Int' via 'Int64' and 'Word' via 'Word64', constraining the conversion
  391   between them to 64-bit architectures where this conversion is safe.
  392 
  393 Doing this effectively bans builds on 32-bit systems, but that's fine, since we don't care about
  394 supporting 32-bit systems anyway, and this way any attempts to build on them will fail fast.
  395 
  396 We used to use newtype- and via-deriving for implementations of relevant instances, but at some
  397 point GHC stopped attaching @INLINE@ annotations to those causing the GHC Core for builtins to have
  398 multiple 'readKnown' and 'makeKnown' calls, so now we don't rely on deriving and implement instances
  399 manually.
  400 -}

  447                     [ Text.pack $ show i

  592 outOfBoundsErr :: Pretty a => a -> Vector.Vector term -> Text
  593 outOfBoundsErr x branches = fold
  594     [ "'case "
  595     , display x
  596     , "' is out of bounds for the given number of branches: "
  597     , display $ Vector.length branches
  598     ]

  600 instance AnnotateCaseBuiltin DefaultUni where
  601     annotateCaseBuiltin ty branches = case ty of
  602         TyBuiltin _ (SomeTypeIn DefaultUniUnit)    ->
  603           case branches of
  604             [x] -> Right $ [(x, [])]
  605             _   -> Left "Casing on unit only allows exactly one branch"
  606         TyBuiltin _ (SomeTypeIn DefaultUniBool)    ->
  607           case branches of
  608             [f]    -> Right $ [(f, [])]
  609             [f, t] -> Right $ [(f, []), (t, [])]
  610             _      -> Left "Casing on bool requires exactly one branch or two branches"
  611         TyBuiltin _ (SomeTypeIn DefaultUniInteger) ->
  612           Right $ map (, []) branches
  613         listTy@(TyApp _ (TyBuiltin _ (SomeTypeIn DefaultUniProtoList)) argTy) ->
  614           case branches of
  615             [cons]      -> Right [(cons, [argTy, listTy])]
  616             [cons, nil] -> Right [(cons, [argTy, listTy]), (nil, [])]
  617             _           -> Left "Casing on list requires exactly one branch or two branches"
  618         (TyApp _ (TyApp _ (TyBuiltin _ (SomeTypeIn DefaultUniProtoPair)) lTyArg) rTyArg) ->
  619           case branches of
  620             [f] -> Right [(f, [lTyArg, rTyArg])]
  621             _   -> Left "Casing on pair requires exactly one branch"
  622         _                 -> Left $ display (() <$ ty) <> " isn't supported in 'case'"
  623 
  624 instance CaseBuiltin DefaultUni where
  625     caseBuiltin someVal@(Some (ValueOf uni x)) branches = case uni of
  626         DefaultUniUnit
  627           | 1 == len   -> Right $ HeadOnly $ branches Vector.! 0
  628           | otherwise -> Left $ outOfBoundsErr someVal branches
  629         DefaultUniBool -> case x of
  630             -- We allow there to be only one branch as long as the scrutinee is 'False'.
  631             -- This is strictly to save size by not having the 'True' branch if it was gonna be
  632             -- 'Error' anyway.
  633             False | len == 1 || len == 2 -> Right $ HeadOnly $ branches Vector.! 0
  634             True  |             len == 2 -> Right $ HeadOnly $ branches Vector.! 1
  635             _                            -> Left  $ outOfBoundsErr someVal branches
  636         DefaultUniInteger
  637             | 0 <= x && x < toInteger len -> Right $ HeadOnly $ branches Vector.! fromInteger x
  638             | otherwise                   -> Left  $ outOfBoundsErr someVal branches
  639         DefaultUniList ty
  640             | len == 1 ->
  641               case x of
  642                 [] -> Left "Expected non-empty list, got empty list for casing list"
  643                 (y : ys) -> Right $ headSpine (branches Vector.! 0) [someValueOf ty y, someValueOf uni ys]
  644             | len == 2 ->
  645               case x of
  646                 []       -> Right $ HeadOnly $ branches Vector.! 1
  647                 (y : ys) -> Right $ headSpine (branches Vector.! 0) [someValueOf ty y, someValueOf uni ys]
  648             | otherwise            -> Left $ outOfBoundsErr someVal branches
  649         DefaultUniPair tyL tyR
  650             | len == 1 ->
  651               case x of
  652                 (l, r) -> Right $ headSpine (branches Vector.! 0) [someValueOf tyL l, someValueOf tyR r]
  653             | otherwise -> Left $ outOfBoundsErr someVal branches
  654         _ -> Left $ display uni <> " isn't supported in 'case'"
  655       where
  656         !len = Vector.length branches
  657     {-# INLINE caseBuiltin #-}

  662 and 'decodeUni' must be amended only in a backwards compatible manner.

  664 See Note [Stable encoding of TPLC]

  728     bring _ DefaultUniByteString r = r

  739         noMoreTypeFunctions f
