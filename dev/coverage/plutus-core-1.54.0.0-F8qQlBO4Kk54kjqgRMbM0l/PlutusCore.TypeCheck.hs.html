   41 -- | The constraint for built-in types\/functions are kind\/type-checkable.
   42 --
   43 -- We keep this separate from 'MonadKindCheck'\/'MonadTypeCheck', because those mainly constrain the
   44 -- monad and 'Typecheckable' constraints only the builtins. In particular useful when the monad gets
   45 -- instantiated and builtins don't. Another reason is that 'Typecheckable' is not required during
   46 -- type checking, since it's only needed for computing 'BuiltinTypes', which is passed as a regular
   47 -- argument to the worker of the type checker.
   48 type Typecheckable uni fun =
   49     (ToKind uni, HasUniApply uni, ToBuiltinMeaning uni fun, AnnotateCaseBuiltin uni)
   50 
   51 -- | The default kind checking config.
   52 defKindCheckConfig :: KindCheckConfig
   53 defKindCheckConfig = KindCheckConfig DetectNameMismatches
   54 
   55 -- | Extract the 'TypeScheme' from a 'BuiltinMeaning' and convert it to the
   56 -- corresponding 'Type' for each built-in function.
   57 builtinMeaningsToTypes
   58     :: (MonadKindCheck (TypeError term uni fun ann) term uni fun ann m, Typecheckable uni fun)
   59     => BuiltinSemanticsVariant fun
   60     -> ann
   61     -> m (BuiltinTypes uni fun)
   62 builtinMeaningsToTypes semvar ann =
   63     runQuoteT . fmap BuiltinTypes . sequence . tabulateArray $ \fun -> do
   64         let ty = typeOfBuiltinFunction semvar fun
   65         _ <- inferKind defKindCheckConfig $ ann <$ ty
   66         dupable <$> normalizeType ty
   67 
   68 -- | Get the default type checking config.
   69 getDefTypeCheckConfig
   70     :: (MonadKindCheck (TypeError term uni fun ann) term uni fun ann m, Typecheckable uni fun)
   71     => ann -> m (TypeCheckConfig uni fun)
   72 getDefTypeCheckConfig ann =
   73     TypeCheckConfig defKindCheckConfig <$> builtinMeaningsToTypes def ann

   75 -- | Infer the kind of a type.
   76 inferKind
   77     :: MonadKindCheck (TypeError term uni fun ann) term uni fun ann m
   78     => KindCheckConfig -> Type TyName uni ann -> m (Kind ())
   79 inferKind config = runTypeCheckM config . inferKindM
   80 
   81 -- | Check a type against a kind.
   82 -- Infers the kind of the type and checks that it's equal to the given kind
   83 -- throwing a 'TypeError' (annotated with the value of the @ann@ argument) otherwise.
   84 checkKind
   85     :: MonadKindCheck (TypeError term uni fun ann) term uni fun ann m
   86     => KindCheckConfig -> ann -> Type TyName uni ann -> Kind () -> m ()
   87 checkKind config ann ty = runTypeCheckM config . checkKindM ann ty
   88 
   89 -- | Infer the type of a term.
   90 inferType
   91     :: MonadTypeCheckPlc uni fun ann m
   92     => TypeCheckConfig uni fun
   93     -> Term TyName Name uni fun ann
   94     -> m (Normalized (Type TyName uni ()))
   95 inferType config = rename >=> runTypeCheckM config . inferTypeM
   96 
   97 -- | Check a term against a type.
   98 -- Infers the type of the term and checks that it's equal to the given type
   99 -- throwing a 'TypeError' (annotated with the value of the @ann@ argument) otherwise.
  100 checkType
  101     :: MonadTypeCheckPlc uni fun ann m
  102     => TypeCheckConfig uni fun
  103     -> ann
  104     -> Term TyName Name uni fun ann
  105     -> Normalized (Type TyName uni ())
  106     -> m ()
  107 checkType config ann term ty = do
  108     termRen <- rename term
  109     runTypeCheckM config $ checkTypeM ann termRen ty
  110 
  111 -- | Infer the type of a program.
  112 inferTypeOfProgram
  113     :: MonadTypeCheckPlc uni fun ann m
  114     => TypeCheckConfig uni fun
  115     -> Program TyName Name uni fun ann
  116     -> m (Normalized (Type TyName uni ()))
  117 inferTypeOfProgram config (Program _ _ term) = inferType config term
  118 
  119 -- | Check a program against a type.
  120 -- Infers the type of the program and checks that it's equal to the given type
  121 -- throwing a 'TypeError' (annotated with the value of the @ann@ argument) otherwise.
  122 checkTypeOfProgram
  123     :: MonadTypeCheckPlc uni fun ann m
  124     => TypeCheckConfig uni fun
  125     -> ann
  126     -> Program TyName Name uni fun ann
  127     -> Normalized (Type TyName uni ())
  128     -> m ()
  129 checkTypeOfProgram config ann (Program _ _ term) = checkType config ann term
