   10 import PlutusCore.Core

   15 -- | '()' as a PLC type.
   16 --
   17 -- > all (A :: *). A -> A
   18 unit :: Type TyName uni ()
   19 unit = runQuote $ do
   20     a <- freshTyName "a"
   21     return
   22         . TyForall () a (Type ())
   23         . TyFun () (TyVar () a)
   24         $ TyVar () a
   25 
   26 -- | '()' as a PLC term.
   27 --
   28 -- > /\(A :: *) -> \(x : A) -> x
   29 unitval :: TermLike term TyName Name uni fun => term ()
   30 unitval = runQuote $ do
   31     a <- freshTyName "a"
   32     x <- freshName "x"
   33     return
   34         . tyAbs  () a (Type ())
   35         . lamAbs () x (TyVar () a)
   36         $ var () x
