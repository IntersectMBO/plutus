   22 -- See Note [Step counter data structure]
   23 -- You might think that since we can store whatever we like in here we might as well
   24 -- use machine words (i.e. 'Word64'), but that is actually slower.
   25 -- | A set of 'Word8' counters that is used in the CEK machine
   26 -- to count steps.
   27 newtype StepCounter (n :: Nat) s = StepCounter (P.MutablePrimArray s Word8)
   28 
   29 -- | Make a new 'StepCounter' with the given number of counters.
   30 newCounter :: (KnownNat n, PrimMonad m) => Proxy n -> m (StepCounter n (PrimState m))
   31 newCounter p = do
   32   let sz = fromIntegral $ natVal p
   33   c <- StepCounter <$> P.newPrimArray sz
   34   resetCounter c
   35   pure c
   36 {-# INLINE newCounter #-}
   37 
   38 -- | Reset all the counters in the given 'StepCounter' to zero.
   39 resetCounter :: forall n m . (KnownNat n, PrimMonad m) => StepCounter n (PrimState m) -> m ()
   40 resetCounter (StepCounter arr) =
   41   let sz = fromIntegral $ natVal (Proxy @n)
   42   in P.setPrimArray arr 0 sz 0
   43 {-# INLINE resetCounter #-}
   44 
   45 -- | Read the value of a counter.
   46 readCounter :: forall m n . PrimMonad m => StepCounter n (PrimState m) -> Int -> m Word8
   47 readCounter =
   48   coerce
   49   @(P.MutablePrimArray (PrimState m) Word8 -> Int -> m Word8)
   50   @(StepCounter n (PrimState m) -> Int -> m Word8)
   51   P.readPrimArray
   52 {-# INLINE readCounter #-}
   53 
   54 -- | Write to a counter.
   55 writeCounter
   56   :: forall m n
   57   . PrimMonad m
   58   => StepCounter n (PrimState m)
   59   -> Int
   60   -> Word8
   61   -> m ()
   62 writeCounter =
   63   coerce
   64   @(P.MutablePrimArray (PrimState m) Word8 -> Int -> Word8 -> m ())
   65   @(StepCounter n (PrimState m) -> Int -> Word8 -> m ())
   66   P.writePrimArray
   67 {-# INLINE writeCounter #-}
   68 
   69 -- | Modify the value of a counter. Returns the modified value.
   70 modifyCounter
   71   :: PrimMonad m
   72   => Int
   73   -> (Word8 -> Word8)
   74   -> StepCounter n (PrimState m)
   75   -> m Word8
   76 modifyCounter i f c = do
   77   v <- readCounter c i
   78   let modified = f v
   79   writeCounter c i modified
   80   pure modified
   81 {-# INLINE modifyCounter #-}

   83 -- | The type of natural numbers in Peano form.
   84 data Peano
   85   = Z
   86   | S Peano
   87 
   88 type NatToPeano :: Nat -> Peano
   89 type family NatToPeano n where
   90     NatToPeano 0 = 'Z
   91     NatToPeano n = 'S (NatToPeano (n - 1))
   92 
   93 type UpwardsM :: (Type -> Type) -> Peano -> Constraint
   94 class Applicative f => UpwardsM f n where
   95   -- | @upwardsM i k@ means @k i *> k (i + 1) *> ... *> k (i + n - 1)@.
   96   -- We use this function in order to statically unroll a loop in 'itraverseCounter_' through
   97   -- instance resolution. This makes @validation@ benchmarks a couple of percent faster.
   98   upwardsM :: Int -> (Int -> f ()) -> f ()
   99 
  100 instance Applicative f => UpwardsM f 'Z where
  101   upwardsM _ _ = pure ()
  102   {-# INLINE upwardsM #-}
  103 
  104 instance UpwardsM f n => UpwardsM f ('S n) where
  105   upwardsM !i k = k i *> upwardsM @f @n (i + 1) k
  106   {-# INLINE upwardsM #-}
  107 
  108 -- | Traverse the counters with an effectful function.
  109 itraverseCounter_
  110   :: forall n m
  111   . (UpwardsM m (NatToPeano n), PrimMonad m)
  112   => (Int -> Word8 -> m ())
  113   -> StepCounter n (PrimState m)
  114   -> m ()
  115 itraverseCounter_ f (StepCounter arr) = do
  116   -- The safety of this operation is a little subtle. The frozen array is only
  117   -- safe to use if the underlying mutable array is not mutated 'afterwards'.
  118   -- In our case it likely _will_ be mutated afterwards... but not until we
  119   -- are done with the frozen version. That ordering is enforced by the fact that
  120   -- the whole thing runs in 'm': future accesses to the mutable array can't
  121   -- happen until this whole function is finished.
  122   arr' <- P.unsafeFreezePrimArray arr
  123   upwardsM @_ @(NatToPeano n) 0 $ \i -> f i $ P.indexPrimArray arr' i
  124 {-# INLINE itraverseCounter_ #-}
  125 
  126 -- | Traverse the counters with an effectful function.
  127 iforCounter_
  128   :: (UpwardsM m (NatToPeano n), PrimMonad m)
  129   => StepCounter n (PrimState m)
  130   -> (Int -> Word8 -> m ())
  131   -> m ()
  132 iforCounter_ = flip itraverseCounter_
  133 {-# INLINE iforCounter_ #-}

  135 {- Note [Step counter data structure]
  136 The step counter data structure has had several iterations.
  137 
  138 Previously we used a "word array", which was a single 'Word64' into which we
  139 packed 8 'Word8's. This worked pretty well: it was pure, and everything reduced
  140 to a bunch of primitive integer operations.
  141 
  142 However, it has a key limitation which is that it can only hold 8 counters.
  143 The obvious attempt to expand it to use a 'Word128' performed badly.
  144 
  145 The 'PrimArray' approach on the other hand was fairly competitive with the
  146 original 'WordArray', and scales fine to more than 8 counters, so we switched
  147 to using that instead.
  148 -}
