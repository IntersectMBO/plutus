   15 -- | Ed25519 signature verification
   16 -- This will fail if the key or the signature are not of the expected length.
   17 -- This version uses the cardano-crypto-class implementation of the verification
   18 -- function (using libsodium).
   19 verifyEd25519Signature
   20     :: BS.ByteString  -- ^ Public Key (32 bytes)
   21     -> BS.ByteString  -- ^ Message    (arbitrary length)
   22     -> BS.ByteString  -- ^ Signature  (64 bytes)
   23     -> BuiltinResult Bool
   24 verifyEd25519Signature pk msg sig =
   25   case DSIGN.rawDeserialiseVerKeyDSIGN @Ed25519DSIGN pk of
   26     Nothing -> failWithMessage loc "Invalid verification key."
   27     Just pk' -> case DSIGN.rawDeserialiseSigDSIGN @Ed25519DSIGN sig of
   28       Nothing -> failWithMessage loc "Invalid signature."
   29       Just sig' ->
   30           pure $
   31                 case DSIGN.verifyDSIGN () pk' msg sig' of
   32                   Left _   -> False
   33                   Right () -> True
   34   where
   35     loc :: Text
   36     loc = "Ed25519 signature verification"
