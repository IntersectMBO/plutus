   63 {-| Evaluate a term using the Steppable CEK machine with logging enabled and keep track of costing.
   64 A wrapper around the internal runCek to debruijn input and undebruijn output.
   65 *THIS FUNCTION IS PARTIAL if the input term contains free variables*
   66 -}
   67 runCek
   68     :: ThrowableBuiltins uni fun
   69     => MachineParameters CekMachineCosts fun (CekValue uni fun ann)
   70     -> ExBudgetMode cost uni fun
   71     -> EmitterMode uni fun
   72     -> Term Name uni fun ann
   73     -> CekReport cost Name uni fun
   74 runCek = Common.runCek S.runCekDeBruijn
   75 
   76 -- | Evaluate a term using the Steppable CEK machine with logging disabled and
   77 -- keep track of costing.
   78 -- *THIS FUNCTION IS PARTIAL if the input term contains free variables*
   79 runCekNoEmit
   80     :: ThrowableBuiltins uni fun
   81     => MachineParameters CekMachineCosts fun (CekValue uni fun ann)
   82     -> ExBudgetMode cost uni fun
   83     -> Term Name uni fun ann
   84     -> (Either (CekEvaluationException Name uni fun) (Term Name uni fun ()), cost)
   85 runCekNoEmit = Common.runCekNoEmit S.runCekDeBruijn
   86 
   87 -- | Evaluate a term using the Steppable CEK machine with logging enabled.
