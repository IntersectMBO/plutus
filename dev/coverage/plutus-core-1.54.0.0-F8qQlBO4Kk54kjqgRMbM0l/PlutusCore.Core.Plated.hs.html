   38 kindSubkinds :: Traversal' (Kind ann) (Kind ann)
   39 kindSubkinds f kind0 = case kind0 of
   40     KindArrow ann dom cod -> KindArrow ann <$> f dom <*> f cod
   41     Type{}                -> pure kind0
   42 
   43 kindSubkindsDeep :: Fold (Kind ann) (Kind ann)
   44 kindSubkindsDeep = cosmosOf kindSubkinds

  100 -- | Get all the direct child 'Type's of the given 'Type'.
  101 typeSubtypes :: Traversal' (Type tyname uni ann) (Type tyname uni ann)
  102 typeSubtypes f ty0 = case ty0 of
  103     TyFun ann ty1 ty2    -> TyFun ann <$> f ty1 <*> f ty2
  104     TyIFix ann pat arg   -> TyIFix ann <$> f pat <*> f arg
  105     TyForall ann tn k ty -> TyForall ann tn k <$> f ty
  106     TyLam ann tn k ty    -> TyLam ann tn k <$> f ty
  107     TyApp ann ty1 ty2    -> TyApp ann <$> f ty1 <*> f ty2
  108     TySOP ann tyls       -> TySOP ann <$> (traverse . traverse) f tyls
  109     TyBuiltin{}          -> pure ty0
  110     TyVar{}              -> pure ty0
  111 {-# INLINE typeSubtypes #-}

  114 typeSubtypesDeep :: Fold (Type tyname uni ann) (Type tyname uni ann)
  115 typeSubtypesDeep = cosmosOf typeSubtypes

  219 -- | Get all the direct child 'Type's of the given 'Term'.
  220 termSubtypes :: Traversal' (Term tyname name uni fun ann) (Type tyname uni ann)
  221 termSubtypes f term0 = case term0 of
  222     LamAbs ann n ty t   -> LamAbs ann n <$> f ty <*> pure t
  223     TyInst ann t ty     -> TyInst ann t <$> f ty
  224     IWrap ann ty1 ty2 t -> IWrap ann <$> f ty1 <*> f ty2 <*> pure t
  225     Error ann ty        -> Error ann <$> f ty
  226     Constr ann ty i es  -> Constr ann <$> f ty <*> pure i <*> pure es
  227     Case ann ty arg cs  -> Case ann <$> f ty <*> pure arg <*> pure cs
  228     TyAbs{}             -> pure term0
  229     Apply{}             -> pure term0
  230     Unwrap{}            -> pure term0
  231     Var{}               -> pure term0
  232     Constant{}          -> pure term0
  233     Builtin{}           -> pure term0
  234 {-# INLINE termSubtypes #-}

  244 -- | Get all the direct child 'Term's of the given 'Term'.
  245 termSubterms :: Traversal' (Term tyname name uni fun ann) (Term tyname name uni fun ann)
  246 termSubterms f term0 = case term0 of
  247     LamAbs ann n ty t   -> LamAbs ann n ty <$> f t
  248     TyInst ann t ty     -> TyInst ann <$> f t <*> pure ty
  249     IWrap ann ty1 ty2 t -> IWrap ann ty1 ty2 <$> f t
  250     TyAbs ann n k t     -> TyAbs ann n k <$> f t
  251     Apply ann t1 t2     -> Apply ann <$> f t1 <*> f t2
  252     Unwrap ann t        -> Unwrap ann <$> f t
  253     Constr ann ty i es  -> Constr ann ty i <$> traverse f es
  254     Case ann ty arg cs  -> Case ann ty <$> f arg <*> traverse f cs
  255     Error{}             -> pure term0
  256     Var{}               -> pure term0
  257     Constant{}          -> pure term0
  258     Builtin{}           -> pure term0
  259 {-# INLINE termSubterms #-}
