   41 -- | Turn a list of Haskell types @args@ into a functional type ending in @res@.
   42 --
   43 -- >>> :set -XDataKinds
   44 -- >>> :kind! FoldArgs [(), Bool] Integer
   45 -- FoldArgs [(), Bool] Integer :: *
   46 -- = () -> Bool -> Integer
   47 type family FoldArgs args res where
   48     FoldArgs '[]           res = res
   49     FoldArgs (arg ': args) res = arg -> FoldArgs args res
   50 
   51 -- | The meaning of a built-in function consists of its type represented as a 'TypeScheme',
   52 -- its Haskell denotation and its uninstantiated runtime denotation.
   53 --
   54 -- The 'TypeScheme' of a built-in function is used for example for
   55 --
   56 -- 1. computing the PLC type of the function to be used during type checking
   57 -- 2. getting arity information
   58 -- 3. generating arbitrary values to apply the function to in tests
   59 --
   60 -- The denotation is lazy, so that we don't need to worry about a builtin being bottom
   61 -- (happens in tests). The production path is not affected by that, since only runtime denotations
   62 -- are used for evaluation.
   63 data BuiltinMeaning val cost =
   64     forall args res. BuiltinMeaning
   65         (TypeScheme val args res)
   66         ~(FoldArgs args res)
   67         (cost -> BuiltinRuntime val)
   68 
   69 -- | Constraints available when defining a built-in function.
   70 type HasMeaningIn uni val = (Typeable val, ExMemoryUsage val, HasConstantIn uni val)

  115 {- Note [Builtin semantics variants]
  116 The purpose of the "builtin semantics variant" feature is to provide multiple,
  117 different denotations (implementations) for the same builtin(s).  An example use
  118 of this feature is for "fixing" the behaviour of `ConsByteString` builtin to
  119 throw an error instead of overflowing its first argument.
  120 
  121 One denotation from each builtin is grouped into a 'BuiltinSemanticsVariant'.
  122 Each Plutus Language version is linked to a specific 'BuiltinSemanticsVariant'
  123 (done by plutus-ledger-api); e.g. plutus-v1 and plutus-v2 are linked to
  124 'DefaultFunSemanticsVariantA' and 'DefaultFunSemanticsVariantB', whereas
  125 plutus-v3 changes the set of denotations to 'DefaultFunSemanticsVariantC' (thus
  126 fixing 'ConsByteString').
  127 
  128 Each 'BuiltinSemanticsVariant' (grouping) can change the denotation of one or
  129 more builtins --- or none, but what's the point in that?
  130 
  131 This 'BuiltinSemanticsVariant' is modelled as a datatype *associated* to the
  132 @fun@. This associated datatype is required to provide an instance of 'Default'
  133 for quality-of-life purpose; the 'def'ault builtin semantics variant is expected
  134 to point to the builtin semantics variant that the plutus-tx/plutus-ir compiler
  135 is currently targeting.
  136 
  137 Note that (old) denotations of a 'BuiltinSemanticsVariant' cannot be removed or
  138 deprecated once published to the chain.
  139 
  140 The way that this feature is implemented buys us more than we currently need:
  141 - allows also a versioned change to a builtin's *type signature*, i.e. type of arguments/result as
  142   well as number of arguments.
  143 - allows also a versioned change to a builtin's cost model parameters
  144 
  145 Besides having no need for this currently, it complicates the codebase since the
  146 typechecker now pointlessly wants to know the builtin semantics before
  147 typechecking. To alleviate this, we use the 'Default.def' builtin semantics
  148 variant during typechecking / lifting. @effectfully: the solution to the problem
  149 would be to establish what kind of backwards compatibility we're willing to
  150 maintain and pull all of that into a separate data type and make it a part of
  151 BuiltinMeaning.
  152 -}

  154 {- Note [Automatic derivation of type schemes]
  155 We use two type classes for automatic derivation of type/runtime schemes and runtime denotations:
  156 'KnownPolytype' and 'KnownMonotype'.
  157 The terminology is due to
  158 https://en.wikipedia.org/wiki/Hindley%E2%80%93Milner_type_system#The_Hindley%E2%80%93Milner_
  159 type_system
  160 
  161 Check out the source of "PlutusCore.Builtin.Runtime" for an explanation of what a runtime
  162 denotation is.
  163 
  164 'KnownPolytype' and 'KnownMonotype' are responsible for deriving polymorphic and monomorphic types,
  165 respectively.
  166 
  167 'KnownPolytype' turns every bound variable into a 'TypeSchemeAll'/'BuiltinExpectForce'. We extract
  168 variables from the type of the Haskell denotation using the 'ToBinds' associated type
  169 family. Variables are collected in the order that they appear in (i.e. just like in Haskell). For
  170 example, processing a type like
  171 
  172       :: ( a ~ Opaque val (TyVarRep ('TyNameRep "a" 0))
  173          , b ~ Opaque val (TyVarRep ('TyNameRep "b" 1))
  174          )
  175       => b -> a -> b
  176 
  177 with 'ToBinds' results in the following list of bindings:
  178 
  179     '[ 'Some ('TyNameRep "b" 1), 'Some ('TyNameRep "a" 0) ]
  180 
  181 'KnownMonotype' turns every argument that the Haskell denotation of a builtin receives into a
  182 'TypeSchemeArrow'/'BuiltinExpectArgument'. We extract the arguments from the type of the Haskell
  183 denotation using the 'GetArgs' type family.
  184 
  185 Higher-kinded type variables are fully supported.
  186 
  187 At this point in the pipeline the type of the denotation of a builtin is assumed to be fully
  188 elaborated (i.e. monomorphic).
  189 -}

  191 -- | Chop a function type to get a list of its argument types.
  192 type GetArgs :: GHC.Type -> [GHC.Type]
  193 type family GetArgs a where
  194     GetArgs (a -> b) = a ': GetArgs b
  195     GetArgs _        = '[]

  197 {- Note [Merging the denotation and the costing function]
  198 The runtime denotation of a builtin computes both the builtin application and its cost
  199 (see the docs of 'BuiltinRuntime' for details). Doing both at the same time has a number of benefits
  200 (see Note [runCostingFun* API]), however in the user-facing API we want to separate the concepts
  201 of the denotation and the costing function. This is because:
  202 
  203 1. the two are fundamentally distinct and we have loads of documentation for each of them
  204    separately, so conflating them in the actual API would be unnecessary coupling
  205 2. right now it's clear which bits of the definition of a builtin constitute evaluation and which
  206    ones constitute costing as the two are different arguments to 'makeBuiltinMeaning'. If evaluation
  207    and costing were intertwined, it would be much harder to review the definition of a builtin
  208 3. ... and it would also be more boilerplate and less clear type signatures
  209 
  210 Hence we want 'makeBuiltinMeaning' to take evaluation and costing bits separately and intertwine
  211 them behind the scenes. Which is straightforward: we only need to pass the two together in the
  212 methods of the 'KnownMonotype' and 'KnownPolytype' classes and zip them argument by argument
  213 into a single 'BuiltinRuntime'.
  214 -}

  216 -- | A class that allows us to derive a monotype for a builtin.
  217 -- We could've computed the runtime denotation from the
  218 -- 'TypeScheme' and the denotation of the builtin, but not statically (due to unfolding not working
  219 -- for recursive functions and 'TypeScheme' being recursive, i.e. requiring the conversion function
  220 -- to be recursive), and so it would cause us to retain a lot of evaluation-irrelevant stuff in the
  221 -- constructors of 'BuiltinRuntime', which has to make evaluation slower (we didn't check) and
  222 -- certainly makes the generated Core much harder to read. Technically speaking, we could get
  223 -- a 'RuntimeScheme' from the 'TypeScheme' and the denotation statically if we changed the
  224 -- definition of 'TypeScheme' and made it a singleton, but then the conversion function would have
  225 -- to become a class anyway and we'd just replicate what we have here, except in a much more
  226 -- complicated way.
  227 class KnownMonotype val args res where
  228     knownMonotype :: TypeScheme val args res
  229 
  230     -- | Convert the denotation of a builtin to its runtime counterpart .
  231     -- The argument is in 'ReadKnownM', because that's what deferred unlifting amounts to:
  232     -- passing the action returning the builtin application around until full saturation, which is
  233     -- when the action actually gets run.
  234     toMonoF
  235         :: ReadKnownM (FoldArgs args res, FoldArgs args ExBudgetStream)
  236         -> BuiltinRuntime val
  237 
  238 -- | Once we've run out of term-level arguments, we return a
  239 -- 'TypeSchemeResult'/'RuntimeSchemeResult'.
  240 instance (Typeable res, KnownTypeAst TyName (UniOf val) res, MakeKnown val res) =>
  241             KnownMonotype val '[] res where
  242     knownMonotype = TypeSchemeResult
  243 
  244     toMonoF =
  245         either
  246             -- Unlifting has failed and we don't care about costing at this point, since we're about
  247             -- to terminate evaluation anyway, hence we put 'mempty' as the cost of the operation.
  248             --
  249             -- Note that putting the cost inside of 'BuiltinResult' is not an option, since forcing
  250             -- the 'BuiltinResult' computation is exactly forcing the builtin application, which we
  251             -- can't do before accounting for the cost of the application, i.e. the cost must be
  252             -- outside of 'BuiltinResult'.
  253             --
  254             -- We could introduce a level of indirection and say that a 'BuiltinCostedResult' is
  255             -- either a budgeting failure or a budgeting success with a cost and a 'BuiltinResult'
  256             -- computation inside, but that would slow things down a bit and the current strategy is
  257             -- reasonable enough.
  258             builtinRuntimeFailure
  259             (\(x, cost) -> BuiltinCostedResult cost $ makeKnown x)
  260     {-# INLINE toMonoF #-}
  261 
  262 {- Note [One-shotting runtime denotations]
  263 In @KnownMonotype val (arg ': args) res@ we 'oneShot' the runtime denotations. Otherwise GHC creates
  264 let-bindings and lifts them out of some of the lambdas in the runtime denotation, which would speed
  265 up partial applications if they were getting reused, but at some point it was verified that we
  266 didn't have any reusage of partial applications: https://github.com/IntersectMBO/plutus/pull/4629
  267 
  268 One-shotting the runtime denotations alone made certain game contracts slower by ~9%. A lot of time
  269 was spent on the investigation, but we still don't know why that was happening. Plus, basically any
  270 other change to the builtins machinery would cause the same kind of slowdown, so we just admitted
  271 defeat and decided it wasn't worth investigating the issue further.
  272 Relevant thread: https://github.com/IntersectMBO/plutus/pull/4620
  273 
  274 The speedup that adding a call to 'oneShot' gives us, if any, is smaller than our noise threshold,
  275 however it also makes those confusing allocations disappear from the generated Core, which is enough
  276 of a reason to add the call.
  277 -}

  279 {- Note [Strict application in runtime denotations]
  280 Runtime denotations contain strict let-bindings. Those are important: without them GHC thinks that
  281 the argument may not be needed in the end and so creates a thunk for it, which is not only
  282 unnecessary allocation, but also prevents things from being unboxed. The argument may indeed not be
  283 needed in the end, but in that case we've got an evaluation failure and we're about to terminate
  284 evaluation anyway, hence we care much more about optimizing the happy path.
  285 -}

  287 -- | Every term-level argument becomes a 'TypeSchemeArrow'/'RuntimeSchemeArrow'.
  288 instance
  289         ( Typeable arg, KnownTypeAst TyName (UniOf val) arg, MakeKnown val arg, ReadKnown val arg
  290         , KnownMonotype val args res
  291         ) => KnownMonotype val (arg ': args) res where
  292     knownMonotype = TypeSchemeArrow knownMonotype
  293 
  294     -- See Note [One-shotting runtime denotations].
  295     -- Grow the builtin application within the received action and recurse on the result.
  296     toMonoF getBoth = BuiltinExpectArgument . oneShot $ \arg ->
  297         -- The lazy application of 'toMonoF' ensures that unlifting of the argument will happen
  298         -- upon full saturation and not before that. This is known as "operationally deferred
  299         -- unlifting" (as opposed to "operationally immediate unlifting") or "call-by-name
  300         -- unlifting" (as opposed to "call-by-value unlifting"). We do it this way to guarantee that
  301         -- the cost of unlifting will be accounted for before unlifting is performed. If we did
  302         -- unlifting eagerly here, this would make the node do work that is not accounted for until
  303         -- full saturation is reached, which may never happen if the partial application is thrown
  304         -- away.
  305         --
  306         -- The disadvantage of this approach is that @addInteger 42@ will always unlift @42@ upon
  307         -- full saturation even if this partial application is saved to a variable. But the way
  308         -- costing calibration benchmarks are set up, we always evaluate a single application, so
  309         -- the cost of unlifting is included in the cost of the builtin regardless of whether
  310         -- there's caching of unlifting or not. Hence the user pays for unlifting anyway and we can
  311         -- prioritize safety over performance here.
  312         --
  313         -- 'oneShot' ensures that GHC doesn't attempt to pull stuff out of the builtin
  314         -- implementation to create thunks. This would give us a "call-by-need" behavior, which may
  315         -- sound enticing as it would give us both caching and operationally deferred unlifting, but
  316         -- this comes at a cost of creating unnecessary thunks in the most common case where there's
  317         -- no benefit from having caching as the builtin application is going to be computed only
  318         -- once. So we choose the "call-by-name" behavior and 'oneShot' is what enables that.
  319         oneShot (toMonoF @val @args @res) $ do
  320             (f, exF) <- getBoth
  321             -- Force the argument that gets passed to the denotation. This seems to help performance
  322             -- a bit (possibly due to its impact on strictness analysis), plus this way we ensure
  323             -- that if computing the argument throws an exception (isn't supposed to happen), we'll
  324             -- catch it in tests.
  325             !x <- readKnown arg
  326             -- See Note [Strict application in runtime denotations].
  327             let !exY = exF x
  328             pure (f x, exY)
  329     {-# INLINE toMonoF #-}

  350 -- Here we unpack an existentially packed @kind@ and constrain it afterwards!
  351 -- So promoted existentials are true sigmas! If we were at the term level, we'd have to pack
  352 -- @kind@ along with the @KnownKind kind@ constraint, otherwise when we unpack the existential,
  353 -- all information is lost and we can't do anything with @kind@.
  354 -- | Every type-level argument becomes a 'TypeSchemeAll'.
  355 instance (KnownSymbol name, KnownNat uniq, KnownKind kind, KnownPolytype binds val args res) =>
  356             KnownPolytype ('Some ('TyNameRep @kind name uniq) ': binds) val args res where
  357     knownPolytype = TypeSchemeAll @name @uniq @kind Proxy $ knownPolytype @binds
  358 
  359     toPolyF = BuiltinExpectForce . toPolyF @binds @val @args @res
  360     {-# INLINE toPolyF #-}

  362 -- | Ensure a built-in function is not nullary and throw a nice error otherwise.
  363 type ThrowOnBothEmpty :: [Some TyNameRep] -> [GHC.Type] -> Bool -> GHC.Type -> GHC.Constraint
  364 type family ThrowOnBothEmpty binds args isBuiltin a where
  365     ThrowOnBothEmpty '[] '[] 'True a =
  366         TypeError (
  367             'Text "A built-in function must take at least one type or term argument" ':$$:
  368             'Text "‘" ':<>: 'ShowType a ':<>: 'Text "’ is a built-in type" ':<>:
  369             'Text " so you can embed any of its values as a constant" ':$$:
  370             'Text "If you still want a built-in function, add a dummy ‘()’ argument"
  371             )
  372     ThrowOnBothEmpty '[] '[] 'False a =
  373         TypeError (
  374             'Text "A built-in function must take at least one type or term argument" ':$$:
  375             'Text "To fix this error add a dummy ‘()’ argument"
  376             )
  377     ThrowOnBothEmpty _ _ _ _ = ()

  379 -- | A function turned into a type class with exactly one fully general instance.
  380 -- We can't package up the constraints of 'makeBuiltinMeaning' (see the instance) into a type or
  381 -- class synonym, because they contain a bunch of variables defined by @~@ or determined via
  382 -- functional dependencies and neither class nor type definitions can handle that
  383 -- (see https://gitlab.haskell.org/ghc/ghc/-/issues/7100). Inlining three lines of constraints
  384 -- whenever we need to call 'makeBuiltinMeaning' over a non-concrete type is a bad option and this
  385 -- abstraction is free anyway, hence its existence.
  386 --
  387 -- The @a@ type variable goes first, because @makeBuiltinMeaning \@A@ is a common pattern.
  388 class MakeBuiltinMeaning a val where
  389     -- See Note [Automatic derivation of type schemes]
  390     -- | Construct the meaning for a built-in function by automatically deriving its
  391     -- 'TypeScheme', given
  392     --
  393     -- 1. the denotation of the builtin
  394     -- 2. an uninstantiated costing function
  395     makeBuiltinMeaning
  396         :: a
  397         -> (cost -> FoldArgs (GetArgs a) ExBudgetStream)
  398         -> BuiltinMeaning val cost
  399 instance
  400         ( uni ~ UniOf val, binds ~ ToBinds uni '[] a, args ~ GetArgs a, a ~ FoldArgs args res
  401         , ThrowOnBothEmpty binds args (IsBuiltin uni a) a
  402         , ElaborateFromTo uni 0 j val a, KnownPolytype binds val args res
  403         ) => MakeBuiltinMeaning a val where
  404     makeBuiltinMeaning f toExF =
  405         BuiltinMeaning (knownPolytype @binds @val @args @res) f $ \cost ->
  406             -- In order to make the 'BuiltinRuntime' of a builtin cacheable we need to tell GHC to
  407             -- create a thunk for it, which we achieve by applying 'lazy' to the 'BuiltinRuntime'
  408             -- here.
  409             --
  410             -- Those thunks however require a lot of care to be properly shared rather than
  411             -- recreated every time a builtin application is evaluated, see 'toBuiltinsRuntime' for
  412             -- how we sort it out.
  413             lazy $ case toExF cost of
  414                 -- See Note [Optimizations of runCostingFun*] for why we use strict @case@.
  415                 !exF -> toPolyF @binds @val @args @res $ pure (f, exF)
  416     {-# INLINE makeBuiltinMeaning #-}

  418 -- | Convert a 'BuiltinMeaning' to a 'BuiltinRuntime' given a cost model.
  419 toBuiltinRuntime :: cost -> BuiltinMeaning val cost -> BuiltinRuntime val
  420 toBuiltinRuntime cost (BuiltinMeaning _ _ denot) = denot cost
  421 {-# INLINE toBuiltinRuntime #-}

  423 -- See Note [Inlining meanings of builtins].
  424 -- | Calculate runtime info for all built-in functions given meanings of builtins (as a constraint),
  425 -- the semantics variant of the set of builtins and a cost model.
  426 toBuiltinsRuntime
  427     :: (cost ~ CostingPart uni fun, ToBuiltinMeaning uni fun, HasMeaningIn uni val)
  428     => BuiltinSemanticsVariant fun
  429     -> cost
  430     -> BuiltinsRuntime fun val
  431 toBuiltinsRuntime semvar cost =
  432     -- A call to 'lazy' is to make sure that the returned 'BuiltinsRuntime' is properly cached in a
  433     -- 'let'-binding. This makes it easier for GHC to optimize the internals of builtins, because
  434     -- without a 'let'-binding GHC would sometimes refuse to cooperate and push 'toBuiltinRuntime'
  435     -- to the inside of the inlined 'toBuiltinMeaning' call, creating lots of 'BuiltinMeaning's
  436     -- instead of 'BuiltinRuntime's with the former hiding the costing optimizations behind a lambda
  437     -- binding the @cost@ variable, which makes the optimizations useless.
  438     -- By using 'lazy' we tell GHC to create a separate thunk, which it can properly optimize,
  439     -- because the other bazillion things don't get in the way. We used to use an explicit
  440     -- 'let'-binding marked with @OPAQUE@, but that turned out to be unreliable, because GHC
  441     -- feels free to turn it into a join point instead of a proper thunk.
  442     lazy . BuiltinsRuntime $ toBuiltinRuntime cost . inline toBuiltinMeaning semvar
  443 {-# INLINE toBuiltinsRuntime #-}
