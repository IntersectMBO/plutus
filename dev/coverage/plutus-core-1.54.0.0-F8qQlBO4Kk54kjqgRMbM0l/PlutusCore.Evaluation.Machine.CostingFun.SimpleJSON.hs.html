   76 {- | This type reflects what is actually in the JSON.  The stuff in
   77    CostingFun.Core and CostingFun.JSON is much more rigid, allowing parsing only
   78    for the model types applicable to the various ModelNArguments types; it also
   79    requires entries for everything in DefaultFun. Using the type defined here
   80    allows us to be more flexible and parse stuff that's not exactly what's
   81    expected in builtinCostModel.json.
   82 -}
   83 data Model
   84     = ConstantCost          Integer
   85     | AddedSizes            LinearFunction
   86     | MultipliedSizes       LinearFunction
   87     | MinSize               LinearFunction
   88     | MaxSize               LinearFunction
   89     | LinearInX             LinearFunction
   90     | LinearInY             LinearFunction
   91     | LinearInZ             LinearFunction
   92     | LiteralInYOrLinearInZ LinearFunction
   93     | LinearInMaxYZ         LinearFunction
   94     | LinearInYAndZ         TwoVariableLinearFunction
   95     | QuadraticInY          OneVariableQuadraticFunction
   96     | QuadraticInZ          OneVariableQuadraticFunction
   97     | QuadraticInXAndY      TwoVariableQuadraticFunction
   98     | SubtractedSizes       LinearFunction Integer
   99     -- ^ Linear model in x-y plus minimum value for the case x-y < 0.
  100     | ConstAboveDiagonal    Integer Model
  101     | ConstBelowDiagonal    Integer Model
  102     | ConstOffDiagonal      Integer Model
  103     | ExpModCost            ExpModCostingFunction
  104       deriving stock (Show, Lift)
  105 
  106 {- The JSON representation consists of a list of pairs of (type, arguments)
  107    values.  The "type" field corresponds to the possible constructors above, the
  108    "arguments" field contains the arguments for that particular constructor.
  109 
  110    The JSON format is a bit inconsistent, which adds some complexity.  For
  111    example, the "arguments" field is sometimes a constant, sometimes an object
  112    representing a linear function, and sometimes an object which contains the
  113    coefficients of a linear function together with some extra data. It would be
  114    nice to rationalise this a bit, but it may be too late to do that.
  115 -}
  116 
  117 instance FromJSON Model where
  118     parseJSON =
  119         withObject "Model" $
  120            \obj -> do
  121              ty   :: Text  <- obj .: "type"
  122              args :: Value <- obj .: "arguments"
  123              {- We always have an "arguments" field which is a Value.  Usually it's
  124                 actually an Object (ie, a map) representing a linear function, but
  125                 sometimes it contains other data, and in those cases we need to
  126                 coerce it to an Object (with objOf) to extract the relevant data.
  127                 We could do that once here and rely on laziness to save us in the
  128                 cases when we don't have an Object, but that looks a bit misleading. -}
  129              case ty of
  130                "constant_cost"               -> ConstantCost          <$> parseJSON args
  131                "added_sizes"                 -> AddedSizes            <$> parseJSON args
  132                "min_size"                    -> MinSize               <$> parseJSON args
  133                "max_size"                    -> MaxSize               <$> parseJSON args
  134                "multiplied_sizes"            -> MultipliedSizes       <$> parseJSON args
  135                "linear_in_x"                 -> LinearInX             <$> parseJSON args
  136                "linear_in_y"                 -> LinearInY             <$> parseJSON args
  137                "linear_in_z"                 -> LinearInZ             <$> parseJSON args
  138                "quadratic_in_y"              -> QuadraticInY          <$> parseJSON args
  139                "quadratic_in_z"              -> QuadraticInZ          <$> parseJSON args
  140                "quadratic_in_x_and_y"        -> QuadraticInXAndY      <$> parseJSON args
  141                "exp_mod_cost"                -> ExpModCost            <$> parseJSON args
  142                "literal_in_y_or_linear_in_z" -> LiteralInYOrLinearInZ <$> parseJSON args
  143                "linear_in_max_yz"            -> LinearInMaxYZ         <$> parseJSON args
  144                "linear_in_y_and_z"           -> LinearInYAndZ         <$> parseJSON args
  145                "subtracted_sizes"            -> SubtractedSizes       <$> parseJSON args <*> objOf args .: "minimum"
  146                "const_above_diagonal"        -> modelWithConstant ConstAboveDiagonal args
  147                "const_below_diagonal"        -> modelWithConstant ConstBelowDiagonal args
  148                "const_off_diagonal"          -> modelWithConstant ConstOffDiagonal   args
  149                {- An adaptor to deal with the old "linear_on_diagonal" tag.  See Note [Backward
  150                   compatibility for costing functions].  We never want to convert back to JSON
  151                   here, so it's OK to forget that we originally got something tagged with
  152                  "linear_on_diagonal". -}
  153                "linear_on_diagonal" ->
  154                  let o = objOf args
  155                  in do
  156                    constant   <- o  .: "constant"
  157                    intercept  <- o .: "intercept"
  158                    slope      <- o .: "slope"
  159                    pure $ ConstOffDiagonal constant (LinearInX $ LinearFunction intercept slope)
  160 
  161                _ -> errorWithoutStackTrace $ "Unknown model type " ++ show ty
  162 
  163                where objOf (Object o) = o
  164                      objOf _          =
  165                       errorWithoutStackTrace "Failed to get Object while parsing \"arguments\""
  166 
  167                      modelWithConstant constr x = constr <$>  o .: "constant" <*> o .: "model"
  168                        where o = objOf x
