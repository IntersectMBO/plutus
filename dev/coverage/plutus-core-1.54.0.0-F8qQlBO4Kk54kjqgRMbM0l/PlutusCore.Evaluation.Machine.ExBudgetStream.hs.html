   15 -- | A lazy stream of 'ExBudget's. Basically @NonEmpty ExBudget@, except the elements are
   16 -- stored strictly.
   17 --
   18 -- The semantics of a stream are those of the 'fold' of its elements. I.e. a stream that is a
   19 -- reordered version of another stream is considered equal to that stream.
   20 --
   21 -- An 'ExBudgetStream' is what one gets by zipping two 'CostStream's (one for CPU, one for memory),
   22 -- which is why the two data types are so similar. The only reason why we don't express both the
   23 -- concepts in terms of a single data type is efficiency, in particular unboxing is crucial for
   24 -- 'CostStream' and we don't care about it in 'ExBudgetStream', because we can't get the spender
   25 -- in the CEK machine to get inlined and so unboxing 'ExBudget' here would only result in boxing it
   26 -- back once it's about to be spent.
   27 data ExBudgetStream
   28     = ExBudgetLast !ExBudget
   29     | ExBudgetCons !ExBudget ExBudgetStream
   30     deriving stock (Show)
   31 
   32 -- See Note [Global local functions].
   33 sumExBudgetStreamGo :: ExBudget -> ExBudgetStream -> ExBudget
   34 sumExBudgetStreamGo !acc (ExBudgetLast budget)         = acc <> budget
   35 sumExBudgetStreamGo !acc (ExBudgetCons budget budgets) = sumExBudgetStreamGo (acc <> budget) budgets
   36 
   37 -- | Add up all the budgets in a 'ExBudgetStream'.
   38 sumExBudgetStream :: ExBudgetStream -> ExBudget
   39 sumExBudgetStream (ExBudgetLast budget0)          = budget0
   40 sumExBudgetStream (ExBudgetCons budget0 budgets0) = sumExBudgetStreamGo budget0 budgets0
   41 {-# INLINE sumExBudgetStream #-}
   42 
   43 -- | Convert a 'CostStream' to an 'ExBudgetStream' by applying a function to each element.
   44 costToExBudgetStream :: (CostingInteger -> ExBudget) -> CostStream -> ExBudgetStream
   45 costToExBudgetStream f = go where
   46     go (CostLast cost)       = ExBudgetLast (f cost)
   47     go (CostCons cost costs) = ExBudgetCons (f cost) $ go costs
   48 {-# INLINE costToExBudgetStream #-}
   49 
   50 -- | Convert a 'CostingInteger' representing a CPU cost and a 'CostingInteger' representing a memory
   51 -- cost to an 'ExBudget'.
   52 toExBudget :: CostingInteger -> CostingInteger -> ExBudget
   53 toExBudget = coerce ExBudget
   54 {-# INLINE toExBudget #-}
   55 
   56 -- See Note [Global local functions].
   57 zipCostStreamGo :: CostStream -> CostStream -> ExBudgetStream
   58 zipCostStreamGo (CostLast cpu) (CostLast mem) =
   59     ExBudgetLast $ toExBudget cpu mem
   60 zipCostStreamGo (CostLast cpu) (CostCons mem mems) =
   61     ExBudgetCons (toExBudget cpu mem) $ costToExBudgetStream (\mem' -> toExBudget 0 mem') mems
   62 zipCostStreamGo (CostCons cpu cpus) (CostLast mem) =
   63     ExBudgetCons (toExBudget cpu mem) $ costToExBudgetStream (\cpu' -> toExBudget cpu' 0) cpus
   64 zipCostStreamGo (CostCons cpu cpus) (CostCons mem mems) =
   65     ExBudgetCons (toExBudget cpu mem) $ zipCostStreamGo cpus mems
   66 
   67 -- | Zip two 'CostStream' together (one with CPU costs and the other one with memory costs,
   68 -- respectively) to get an 'ExBudgetStream'. If one is longer than the other, then it's assumed to
   69 -- contain the required amount of zeros for two streams to have the same length (all those zeros
   70 -- \"appear\" in the tail of the stream).
   71 zipCostStream :: CostStream -> CostStream -> ExBudgetStream
   72 zipCostStream cpus0 mems0 = case (cpus0, mems0) of
   73     -- See Note [Single-element streams].
   74     (CostLast cpu, CostLast mem) -> ExBudgetLast $ toExBudget cpu mem
   75     _                            -> zipCostStreamGo cpus0 mems0
   76 {-# INLINE zipCostStream #-}
