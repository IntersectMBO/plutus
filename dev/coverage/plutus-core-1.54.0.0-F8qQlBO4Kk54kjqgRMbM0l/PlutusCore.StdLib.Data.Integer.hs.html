   21 -- |  @succ :: Integer -> Integer@ as a PLC term.
   22 --
   23 -- > \(i : integer) -> addInteger i 1
   24 succInteger
   25     :: (TermLike term tyname Name uni DefaultFun, uni `HasTypeAndTermLevel` Integer) => term ()
   26 succInteger = runQuote $ do
   27     i  <- freshName "i"
   28     return
   29         . lamAbs () i integer
   30         . mkIterAppNoAnn (builtin () AddInteger)
   31         $ [ var () i
   32           , mkConstant @Integer () 1
   33           ]
