  194 {-| Inline simple bindings. Relies on global uniqueness, and preserves it.
  195 See Note [Inlining and global uniqueness]
  196 -}
  197 inline
  198   :: forall name uni fun m a
  199    . (ExternalConstraints name uni fun m)
  200   => AstSize
  201   -- ^ inline threshold
  202   -> Bool
  203   -- ^ inline constants
  204   -> Bool
  205   -- ^ preserve logging
  206   -> InlineHints name a
  207   -> PLC.BuiltinSemanticsVariant fun
  208   -> Term name uni fun a
  209   -> SimplifierT name uni fun a m (Term name uni fun a)
  210 inline
  211   callsiteGrowth
  212   inlineConstants
  213   preserveLogging
  214   hints
  215   builtinSemanticsVariant
  216   t = do
  217     result <-
  218       liftQuote $
  219         flip evalStateT mempty $
  220           runReaderT
  221             (processTerm t)
  222             InlineInfo
  223               { _iiUsages = Usages.termUsages t
  224               , _iiHints = hints
  225               , _iiBuiltinSemanticsVariant = builtinSemanticsVariant
  226               , _iiInlineConstants = inlineConstants
  227               , _iiInlineCallsiteGrowth = callsiteGrowth
  228               , _iiPreserveLogging = preserveLogging
  229               }
  230     recordSimplification t Inline result
  231     return result

  235 {-| Extract the list of applications from a term,
  236 a bit like a "multi-beta" reduction.
  237 
  238 Some examples will help:
  239 [(\x . t) a] -> Just ([(x, a)], t)
  240 
  241 [[[(\x . (\y . (\z . t))) a] b] c] -> Just ([(x, a), (y, b), (z, c)]) t)
  242 
  243 [[(\x . t) a] b] -> Nothing
  244 -}
  245 extractApps
  246   :: Term name uni fun a
  247   -> Maybe ([UTermDef name uni fun a], Term name uni fun a)
  248 extractApps = go []
  249  where
  250   go argStack (Apply _ f arg) = go (arg : argStack) f
  251   go argStack t               = matchArgs argStack [] t
  252   matchArgs (arg : rest) acc (LamAbs a n body) =
  253     matchArgs rest (Def (UVarDecl a n) arg : acc) body
  254   matchArgs [] acc t =
  255     if null acc then Nothing else Just (reverse acc, t)
  256   matchArgs (_ : _) _ _ = Nothing
  257 
  258 -- | The inverse of 'extractApps'.
  259 restoreApps
  260   :: [UTermDef name uni fun a]
  261   -> Term name uni fun a
  262   -> Term name uni fun a
  263 restoreApps defs t = makeLams [] t (reverse defs)
  264  where
  265   makeLams args acc (Def (UVarDecl a n) rhs : rest) =
  266     makeLams (rhs : args) (LamAbs a n acc) rest
  267   makeLams args acc [] =
  268     makeApps args acc
  269   -- This isn't the best annotation, but it will do
  270   makeApps (arg : args) acc =
  271     makeApps args (Apply (termAnn acc) acc arg)
  272   makeApps [] acc = acc

  274 -- | Run the inliner on a `UntypedPlutusCore.Core.Type.Term`.
  275 processTerm
  276   :: forall name uni fun a
  277    . (InliningConstraints name uni fun)
  278   => Term name uni fun a
  279   -> InlineM name uni fun a (Term name uni fun a)
  280 processTerm = handleTerm
  281  where
  282   handleTerm
  283     :: Term name uni fun a
  284     -> InlineM name uni fun a (Term name uni fun a)
  285   handleTerm = \case
  286     v@(Var _ n) -> fromMaybe v <$> substName n
  287     -- See Note [Differences from PIR inliner] 3
  288     (extractApps -> Just (bs, t)) -> do
  289       bs' <- wither (processSingleBinding t) bs
  290       t' <- processTerm t
  291       pure $ restoreApps bs' t'
  292     t -> inlineSaturatedApp =<< forMOf termSubterms t processTerm
  293 
  294   -- See Note [Renaming strategy]
  295   substName :: name -> InlineM name uni fun a (Maybe (Term name uni fun a))
  296   substName name = gets (lookupTerm name) >>= traverse renameTerm
  297 
  298   -- See Note [Inlining approach and 'Secrets of the GHC Inliner']
  299   renameTerm
  300     :: InlineTerm name uni fun a
  301     -> InlineM name uni fun a (Term name uni fun a)
  302   renameTerm = \case
  303     -- Already processed term, just rename and put it in, don't do any
  304     -- further optimization here.
  305     Done t -> liftDupable t

  307 processSingleBinding
  308   :: (InliningConstraints name uni fun)
  309   => Term name uni fun a
  310   -> UTermDef name uni fun a
  311   -> InlineM name uni fun a (Maybe (UTermDef name uni fun a))
  312 processSingleBinding body (Def vd@(UVarDecl a n) rhs0) = do
  313   maybeAddSubst body a n rhs0 >>= \case
  314     Just rhs -> do
  315       let (binders, rhsBody) = UPLC.splitParams rhs
  316       modify' . extendVarInfo n $
  317         VarInfo
  318           { _varBinders = binders
  319           , _varRhs = rhs
  320           , _varRhsBody = Done (dupable rhsBody)
  321           }
  322       pure . Just $ Def vd rhs
  323     Nothing -> pure Nothing

  325 {-| Check against the heuristics we have for inlining and either inline
  326 the term binding or not. The arguments to this function are the fields of the
  327 'TermBinding' being processed.
  328 Nothing means that we are inlining the term:
  329   * we have extended the substitution, and
  330   * we are removing the binding (hence we return Nothing).
  331 -}
  332 maybeAddSubst
  333   :: forall name uni fun a
  334    . (InliningConstraints name uni fun)
  335   => Term name uni fun a
  336   -> a
  337   -> name
  338   -> Term name uni fun a
  339   -> InlineM name uni fun a (Maybe (Term name uni fun a))
  340 maybeAddSubst body a n rhs0 = do
  341   rhs <- processTerm rhs0
  342 
  343   -- Check whether we've been told specifically to inline this
  344   hints <- view iiHints
  345   case shouldInline hints a n of
  346     AlwaysInline ->
  347       -- if we've been told specifically, then do it right away
  348       extendAndDrop (Done $ dupable rhs)
  349     hint ->
  350       let safeToInline = hint == SafeToInline
  351        in ifM
  352             (shouldUnconditionallyInline safeToInline n rhs body)
  353             (extendAndDrop (Done $ dupable rhs))
  354             (pure $ Just rhs)
  355  where
  356   extendAndDrop
  357     :: forall b
  358      . InlineTerm name uni fun a
  359     -> InlineM name uni fun a (Maybe b)
  360   extendAndDrop t = modify' (extendTerm n t) >> pure Nothing
  361 
  362 shouldUnconditionallyInline
  363   :: (InliningConstraints name uni fun)
  364   => Bool
  365   {- ^ Whether we know that the binding is safe to inline.
  366   If so, bypass the purity check.
  367   -}
  368   -> name
  369   -> Term name uni fun a
  370   -> Term name uni fun a
  371   -> InlineM name uni fun a Bool
  372 shouldUnconditionallyInline safe n rhs body = do
  373   isTermPure <- checkPurity rhs
  374   inlineConstants <- view iiInlineConstants
  375   preUnconditional isTermPure ||^ postUnconditional inlineConstants isTermPure
  376  where
  377   -- similar to the paper, preUnconditional inlining checks that the binder
  378   -- is 'OnceSafe'.  I.e., it's used at most once AND it neither duplicate code
  379   -- or work. While we don't check for lambda etc like in the paper,
  380   -- 'effectSafe' ensures that it isn't doing any substantial work.
  381   -- We actually also inline 'Dead' binders (i.e., remove dead code) here.
  382   preUnconditional isTermPure =
  383     nameUsedAtMostOnce n &&^ (pure safe ||^ effectSafe body n isTermPure)
  384   -- See Note [Inlining approach and 'Secrets of the GHC Inliner'] and
  385   -- [Inlining and purity]. This is the case where we don't know that the number
  386   -- of occurrences is exactly one, so there's no point checking if the term is
  387   -- immediately evaluated.
  388   postUnconditional inlineConstants isTermPure =
  389     pure (safe || isTermPure) &&^ acceptable inlineConstants rhs
  390 
  391 -- | Check if term is pure. See Note [Inlining and purity]
  392 checkPurity
  393   :: (PLC.ToBuiltinMeaning uni fun)
  394   => Term name uni fun a
  395   -> InlineM name uni fun a Bool
  396 checkPurity t = do
  397   builtinSemanticsVariant <- view iiBuiltinSemanticsVariant
  398   pure $ isPure builtinSemanticsVariant t
  399 
  400 nameUsedAtMostOnce
  401   :: forall name uni fun a
  402    . (InliningConstraints name uni fun)
  403   => name
  404   -> InlineM name uni fun a Bool
  405 nameUsedAtMostOnce n = do
  406   usgs <- view iiUsages
  407   -- 'inlining' terms used 0 times is a cheap way to remove dead code
  408   -- while we're here
  409   pure $ Usages.getUsageCount n usgs <= 1

  411 isFirstVarBeforeEffects
  412   :: forall name uni fun ann
  413    . (InliningConstraints name uni fun)
  414   => BuiltinSemanticsVariant fun
  415   -> name
  416   -> Term name uni fun ann
  417   -> Bool
  418 isFirstVarBeforeEffects builtinSemanticsVariant n t =
  419   -- This can in the worst case traverse a lot of the term, which could lead to
  420   -- us doing ~quadratic work as we process the program. However in practice
  421   -- most terms have a relatively short evaluation order before we hit Unknown,
  422   -- so it's not too bad.
  423   go (unEvalOrder (termEvaluationOrder builtinSemanticsVariant t))
  424  where
  425   -- Found the variable we're looking for!
  426   go ((EvalTerm _ _ (Var _ n')) : _) | n == n' = True
  427   -- Found a pure term, ignore it and continue
  428   go ((EvalTerm Pure _ _) : rest) = go rest
  429   -- Found a possibly impure term, our variable is definitely not first
  430   go ((EvalTerm MaybeImpure _ _) : _) = False
  431   -- Don't know, be conservative
  432   go (Unknown : _) = False
  433   go [] = False

  435 {-| Check if the given name is strict in the given term.
  436   This means that at least one occurrence of the name is found outside of the following:
  437   * 'delay' term
  438   * lambda body
  439   * case branch
  440 -}
  441 isStrictIn
  442   :: forall name uni fun a
  443    . (Eq name)
  444   => name
  445   -> Term name uni fun a
  446   -> Bool
  447 isStrictIn name = go
  448  where
  449   go :: Term name uni fun a -> Bool
  450   go = \case
  451     Var _ann name' -> name == name'
  452     LamAbs _ann _paramName _body -> False
  453     Apply _ann t1 t2 -> go t1 || go t2
  454     Force _ann t -> go t
  455     Delay _ann _term -> False
  456     Constant{} -> False
  457     Builtin{} -> False
  458     Error{} -> False
  459     Constr _ann _idx terms -> any go terms
  460     Case _ann scrut _branches -> go scrut
  461 
  462 effectSafe
  463   :: forall name uni fun a
  464    . (InliningConstraints name uni fun)
  465   => Term name uni fun a
  466   -> name
  467   -> Bool
  468   -- ^ is it pure? See Note [Inlining and purity]
  469   -> InlineM name uni fun a Bool
  470 effectSafe body n termIsPure = do
  471   preserveLogging <- view iiPreserveLogging
  472   builtinSemantics <- view iiBuiltinSemanticsVariant
  473   return $
  474     termIsPure
  475       || isFirstVarBeforeEffects builtinSemantics n body
  476       || (not preserveLogging && isStrictIn n body)

  478 {-| Should we inline? Should only inline things that won't duplicate work
  479 or code.  See Note [Inlining approach and 'Secrets of the GHC Inliner']
  480 -}
  481 acceptable
  482   :: Bool
  483   -- ^ inline constants
  484   -> Term name uni fun a
  485   -> InlineM name uni fun a Bool
  486 acceptable inlineConstants t =
  487   -- See Note [Inlining criteria]
  488   pure $ costIsAcceptable t && sizeIsAcceptable inlineConstants t
  489 
  490 {-| Is the cost increase (in terms of evaluation work) of inlining a variable
  491 whose RHS is the given term acceptable?
  492 -}
  493 costIsAcceptable :: Term name uni fun a -> Bool
  494 costIsAcceptable = \case
  495   Builtin{} -> True
  496   Var{} -> True
  497   Constant{} -> True
  498   Error{} -> True
  499   -- This will mean that we create closures at each use site instead of
  500   -- once, but that's a very low cost which we're okay rounding to 0.
  501   LamAbs{} -> True
  502   Apply{} -> False
  503   -- Inlining constructors of size 1 or 0 seems okay, but does result in doing
  504   -- the work for the elements at each use site.
  505   Constr _ _ es ->
  506     case es of
  507       []  -> True
  508       [e] -> costIsAcceptable e
  509       _   -> False
  510   -- Inlining a case means redoing the match at each use site
  511   Case{} -> False
  512   Force{} -> False
  513   Delay{} -> True
  514 
  515 {-| Is the size increase (in the AST) of inlining a variable whose RHS is
  516 the given term acceptable?
  517 -}
  518 sizeIsAcceptable
  519   :: Bool
  520   -- ^ inline constants
  521   -> Term name uni fun a
  522   -> Bool
  523 sizeIsAcceptable inlineConstants = \case
  524   Builtin{} -> True
  525   Var{} -> True
  526   Error{} -> True
  527   -- See Note [Differences from PIR inliner] 4
  528   LamAbs{} -> False
  529   -- Inlining constructors of size 1 or 0 seems okay
  530   Constr _ _ es -> case es of
  531     []  -> True
  532     [e] -> sizeIsAcceptable inlineConstants e
  533     _   -> False
  534   -- Cases are pretty big, due to the case branches
  535   Case{} -> False
  536   -- Inlining constants is deemed acceptable if the 'inlineConstants'
  537   -- flag is turned on, see Note [Inlining constants].
  538   Constant{} -> inlineConstants
  539   Apply{} -> False
  540   Force _ t -> sizeIsAcceptable inlineConstants t
  541   Delay _ t -> sizeIsAcceptable inlineConstants t

  543 -- | Fully apply and beta reduce.
  544 fullyApplyAndBetaReduce
  545   :: forall name uni fun a
  546    . (InliningConstraints name uni fun)
  547   => VarInfo name uni fun a
  548   -> [(a, Term name uni fun a)]
  549   -> InlineM name uni fun a (Maybe (Term name uni fun a))
  550 fullyApplyAndBetaReduce info args0 = do
  551   rhsBody <- liftDupable (let Done rhsBody = info ^. varRhsBody in rhsBody)
  552   let go
  553         :: Term name uni fun a
  554         -> [name]
  555         -> [(a, Term name uni fun a)]
  556         -> InlineM name uni fun a (Maybe (Term name uni fun a))
  557       go acc bs args = case (bs, args) of
  558         ([], _) -> pure . Just $ mkIterApp acc args
  559         (param : params, (_ann, arg) : args') -> do
  560           safe <- safeToBetaReduce param arg
  561           if safe
  562             then do
  563               acc' <-
  564                 termSubstNamesM
  565                   ( \n ->
  566                       if n == param
  567                         then Just <$> PLC.rename arg
  568                         else pure Nothing
  569                   )
  570                   acc
  571               go acc' params args'
  572             else pure Nothing
  573         _ -> pure Nothing
  574 
  575       -- Is it safe to turn `(\a -> body) arg` into `body [a := arg]`?
  576       -- The criteria is the same as the criteria for unconditionally
  577       -- inlining `a`, since inlining is the same as beta reduction.
  578       safeToBetaReduce
  579         :: name
  580         -> Term name uni fun a
  581         -> InlineM name uni fun a Bool
  582       safeToBetaReduce a arg = shouldUnconditionallyInline False a arg rhsBody
  583   go rhsBody (info ^. varBinders) args0

  585 {-| This works in the same way as
  586 'PlutusIR.Transform.Inline.CallSiteInline.inlineSaturatedApp'.
  587 See Note [Inlining and beta reduction of functions].
  588 -}
  589 inlineSaturatedApp
  590   :: forall name uni fun a
  591    . (InliningConstraints name uni fun)
  592   => Term name uni fun a
  593   -> InlineM name uni fun a (Term name uni fun a)
  594 inlineSaturatedApp t
  595   | (Var _ann name, args) <- UPLC.splitApplication t =
  596       gets (lookupVarInfo name) >>= \case
  597         Nothing -> pure t
  598         Just varInfo ->
  599           fullyApplyAndBetaReduce varInfo args >>= \case
  600             Nothing -> pure t
  601             Just fullyApplied -> do
  602               thresh <- view iiInlineCallsiteGrowth
  603               let
  604                 -- Inline only if the size is no bigger than
  605                 -- not inlining plus threshold.
  606                 sizeIsOk =
  607                   termAstSize fullyApplied <= termAstSize t + max 0 thresh
  608                 rhs = varInfo ^. varRhs
  609                 -- Cost is always OK if the RHS is a LamAbs,
  610                 -- but may not be otherwise.
  611                 costIsOk = costIsAcceptable rhs
  612               -- The RHS is always pure if it is a LamAbs,
  613               -- but may not be otherwise.
  614               rhsPure <- checkPurity rhs
  615               pure $
  616                 if sizeIsOk && costIsOk && rhsPure
  617                   then fullyApplied
  618                   else t
  619   | otherwise = pure t
