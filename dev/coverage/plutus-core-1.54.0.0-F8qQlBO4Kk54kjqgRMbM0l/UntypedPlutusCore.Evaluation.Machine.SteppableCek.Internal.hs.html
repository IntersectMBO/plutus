   68 {- Note [Debuggable vs Original versions of CEK]
   69 
   70 The debuggable version of CEK was created from copying over the original CEK/Internal.hs module
   71 and modifying the `computeCek`, `returnCek` functions.
   72 These functions were modified so as to execute a single step (Compute or Return resp.) and immediately
   73 return with the CEK's machine new state (`CekState`), whereas previously these two functions would iteratively run to completion.
   74 
   75 The interface otherwise remains the same. Moreover, the `Original.runCekDeBruijn` and `Debug.runCekDeBruijn` must behave equivalently.
   76 -}

   78 data CekState uni fun ann =
   79     -- loaded a term but not fired the cek yet
   80     Starting (NTerm uni fun ann)
   81     -- the next state is computing
   82     | Computing (Context uni fun ann) (CekValEnv uni fun ann) (NTerm uni fun ann)
   83     -- the next state is returning
   84     | Returning (Context uni fun ann) (CekValue uni fun ann)
   85     -- evaluation finished
   86     | Terminating (DischargeResult uni fun)
   87 
   88 instance Pretty (CekState uni fun ann) where
   89     pretty = \case
   90         Starting{}    -> "Starting"
   91         Computing{}   -> "Computing"
   92         Returning{}   -> "Returning"
   93         Terminating{} -> "Terminating"

   95 -- | Similar to 'Cek.Internal.Context', but augmented with an 'ann'
   96 data Context uni fun ann
   97     = FrameAwaitArg ann !(CekValue uni fun ann) !(Context uni fun ann)                         -- ^ @[V _]@
   98     | FrameAwaitFunTerm ann !(CekValEnv uni fun ann) !(NTerm uni fun ann) !(Context uni fun ann) -- ^ @[_ N]@
   99     | FrameAwaitFunConN ann !(Spine (Some (ValueOf uni))) !(Context uni fun ann)
  100     | FrameAwaitFunValueN ann !(ArgStackNonEmpty uni fun ann) !(Context uni fun ann)
  101     | FrameForce ann !(Context uni fun ann)                                               -- ^ @(force _)@
  102     | FrameConstr ann !(CekValEnv uni fun ann) {-# UNPACK #-} !Word64 ![NTerm uni fun ann] !(ArgStack uni fun ann) !(Context uni fun ann)
  103     | FrameCases ann !(CekValEnv uni fun ann) !(V.Vector (NTerm uni fun ann)) !(Context uni fun ann)
  104     | NoFrame
  105 
  106 deriving stock instance (GShow uni, Everywhere uni Show, Show fun, Show ann, Closed uni)
  107     => Show (Context uni fun ann)

  109 computeCek
  110     :: forall uni fun ann s
  111     . (ThrowableBuiltins uni fun, GivenCekReqs uni fun ann s)
  112     => Context uni fun ann
  113     -> CekValEnv uni fun ann
  114     -> NTerm uni fun ann
  115     -> CekM uni fun s (CekState uni fun ann)
  116 -- s ; ρ ▻ {L A}  ↦ s , {_ A} ; ρ ▻ L
  117 computeCek !ctx !env (Var _ varName) = do
  118     stepAndMaybeSpend BVar
  119     val <- lookupVarName varName env
  120     pure $ Returning ctx val
  121 computeCek !ctx !_ (Constant _ val) = do
  122     stepAndMaybeSpend BConst
  123     pure $ Returning ctx (VCon val)
  124 computeCek !ctx !env (LamAbs _ name body) = do
  125     stepAndMaybeSpend BLamAbs
  126     pure $ Returning ctx (VLamAbs name body env)
  127 computeCek !ctx !env (Delay _ body) = do
  128     stepAndMaybeSpend BDelay
  129     pure $ Returning ctx (VDelay body env)
  130 -- s ; ρ ▻ lam x L  ↦  s ◅ lam x (L , ρ)
  131 computeCek !ctx !env (Force _ body) = do
  132     stepAndMaybeSpend BForce
  133     pure $ Computing (FrameForce (termAnn body) ctx) env body
  134 -- s ; ρ ▻ [L M]  ↦  s , [_ (M,ρ)]  ; ρ ▻ L
  135 computeCek !ctx !env (Apply _ fun arg) = do
  136     stepAndMaybeSpend BApply
  137     pure $ Computing (FrameAwaitFunTerm (termAnn fun) env arg ctx) env fun
  138 -- s ; ρ ▻ abs α L  ↦  s ◅ abs α (L , ρ)
  139 -- s ; ρ ▻ con c  ↦  s ◅ con c
  140 -- s ; ρ ▻ builtin bn  ↦  s ◅ builtin bn arity arity [] [] ρ
  141 computeCek !ctx !_ (Builtin _ bn) = do
  142     stepAndMaybeSpend BBuiltin
  143     let meaning = lookupBuiltin bn ?cekRuntime
  144     -- 'Builtin' is fully discharged.
  145     pure $ Returning ctx (VBuiltin bn (Builtin () bn) meaning)
  146 -- s ; ρ ▻ constr I T0 .. Tn  ↦  s , constr I _ (T1 ... Tn, ρ) ; ρ ▻ T0
  147 computeCek !ctx !env (Constr ann i es) = do
  148     stepAndMaybeSpend BConstr
  149     pure $ case es of
  150         (t : rest) -> Computing (FrameConstr ann env i rest NilStack ctx) env t
  151         []         -> Returning ctx $ VConstr i EmptyStack
  152 -- s ; ρ ▻ case S C0 ... Cn  ↦  s , case _ (C0 ... Cn, ρ) ; ρ ▻ S
  153 computeCek !ctx !env (Case ann scrut cs) = do
  154     stepAndMaybeSpend BCase
  155     pure $ Computing (FrameCases ann env cs ctx) env scrut
  156 -- s ; ρ ▻ error A  ↦  <> A
  157 computeCek !_ !_ (Error _) =
  158     throwErrorWithCause (OperationalError CekEvaluationFailure) (Error ())

  160 returnCek
  161     :: forall uni fun ann s
  162     . (ThrowableBuiltins uni fun, GivenCekReqs uni fun ann s)
  163     => Context uni fun ann
  164     -> CekValue uni fun ann
  165     -> CekM uni fun s (CekState uni fun ann)
  166 --- Instantiate all the free variable of the resulting term in case there are any.
  167 -- . ◅ V           ↦  [] V
  168 returnCek NoFrame val = do
  169     spendAccumulatedBudget
  170     pure $ Terminating (dischargeCekValue val)
  171 -- s , {_ A} ◅ abs α M  ↦  s ; ρ ▻ M [ α / A ]*
  172 returnCek (FrameForce _ ctx) fun = forceEvaluate ctx fun
  173 -- s , [_ (M,ρ)] ◅ V  ↦  s , [V _] ; ρ ▻ M
  174 returnCek (FrameAwaitFunTerm _funAnn argVarEnv arg ctx) fun =
  175     -- MAYBE: perhaps it is worth here to merge the _funAnn with argAnn
  176     pure $ Computing (FrameAwaitArg (termAnn arg) fun ctx) argVarEnv arg
  177 -- s , [(lam x (M,ρ)) _] ◅ V  ↦  s ; ρ [ x  ↦  V ] ▻ M
  178 -- FIXME (https://github.com/IntersectMBO/plutus-private/issues/1878):
  179 -- add rule for VBuiltin once it's in the specification.
  180 returnCek (FrameAwaitArg _ fun ctx) arg =
  181     applyEvaluate ctx fun arg
  182 -- s , [_ V1 .. Vn] ◅ lam x (M,ρ)  ↦  s , [_ V2 .. Vn]; ρ [ x  ↦  V1 ] ▻ M
  183 returnCek (FrameAwaitFunConN ann args ctx) fun =
  184   case args of
  185     SpineLast arg      -> applyEvaluate ctx fun (VCon arg)
  186     SpineCons arg rest -> applyEvaluate (FrameAwaitFunConN ann rest ctx) fun (VCon arg)
  187 -- s , [_ V1 .. Vn] ◅ lam x (M,ρ)  ↦  s , [_ V2 .. Vn]; ρ [ x  ↦  V1 ] ▻ M
  188 returnCek (FrameAwaitFunValueN ann args ctx) fun =
  189     case args of
  190       LastStackNonEmpty arg ->
  191         applyEvaluate ctx fun arg
  192       ConsStackNonEmpty arg rest ->
  193         applyEvaluate (FrameAwaitFunValueN ann rest ctx) fun arg
  194 -- s , constr I V0 ... Vj-1 _ (Tj+1 ... Tn, ρ) ◅ Vj  ↦  s , constr i V0 ... Vj _ (Tj+2... Tn, ρ)  ; ρ ▻ Tj+1
  195 returnCek (FrameConstr ann env i todo done ctx) e = do
  196     case todo of
  197         (next : todo') ->
  198           pure $ Computing (FrameConstr ann env i todo' (ConsStack e done) ctx) env next
  199         []             ->
  200           let go acc NilStack         = acc
  201               go acc (ConsStack x xs) = go (ConsStackNonEmpty x acc) xs
  202           in pure $ Returning ctx $ VConstr i (MultiStack $ go (LastStackNonEmpty e) done)
  203 -- s , case _ (C0 ... CN, ρ) ◅ constr i V1 .. Vm  ↦  s , [_ V1 ... Vm] ; ρ ▻ Ci
  204 returnCek (FrameCases ann env cs ctx) e = case e of
  205     -- If the index is larger than the max bound of an Int, or negative, then it's a bad index
  206     -- As it happens, this will currently never trigger, since i is a Word64, and the largest
  207     -- Word64 value wraps to -1 as an Int64. So you can't wrap around enough to get an
  208     -- "apparently good" value.
  209     (VConstr i _) | fromIntegral @_ @Integer i > fromIntegral @Int @Integer maxBound ->
  210                     throwErrorDischarged (StructuralError $ MissingCaseBranchMachineError i) e
  211     (VConstr i args) -> case (V.!?) cs (fromIntegral i) of
  212         Just t  ->
  213           case args of
  214             EmptyStack      -> computeCek ctx env t
  215             MultiStack rest -> computeCek (FrameAwaitFunValueN ann rest ctx) env t
  216         Nothing -> throwErrorDischarged (StructuralError $ MissingCaseBranchMachineError i) e
  217     VCon val -> case unCaserBuiltin ?cekCaserBuiltin val cs of
  218         Left err  -> throwErrorDischarged (OperationalError $ CekCaseBuiltinError err) e
  219         Right (HeadOnly fX) -> pure $ Computing ctx env fX
  220         Right (HeadSpine f xs) -> pure $ Computing (FrameAwaitFunConN ann xs ctx) env f
  221     _ -> throwErrorDischarged (StructuralError NonConstrScrutinizedMachineError) e
  222 
  223 -- | @force@ a term and proceed.
  224 -- If v is a delay then compute the body of v;
  225 -- if v is a builtin application then check that it's expecting a type argument,
  226 -- and either calculate the builtin application or stick a 'Force' on top of its 'Term'
  227 -- representation depending on whether the application is saturated or not,
  228 -- if v is anything else, fail.
  229 forceEvaluate
  230     :: forall uni fun ann s
  231     . (ThrowableBuiltins uni fun, GivenCekReqs uni fun ann s)
  232     => Context uni fun ann
  233     -> CekValue uni fun ann
  234     -> CekM uni fun s (CekState uni fun ann)
  235 forceEvaluate !ctx (VDelay body env) =
  236     pure $ Computing ctx env body
  237 forceEvaluate !ctx (VBuiltin fun term runtime) = do
  238     -- @term@ is fully discharged, and so @term'@ is, hence we can put it in a 'VBuiltin'.
  239     let term' = Force () term
  240     case runtime of
  241         -- It's only possible to force a builtin application if the builtin expects a type
  242         -- argument next.
  243         BuiltinExpectForce runtime' -> do
  244             -- We allow a type argument to appear last in the type of a built-in function,
  245             -- otherwise we could just assemble a 'VBuiltin' without trying to evaluate the
  246             -- application.
  247             evalBuiltinApp ctx fun term' runtime'
  248         _ ->
  249           throwErrorWithCause (StructuralError BuiltinTermArgumentExpectedMachineError) term'
  250 forceEvaluate !_ val =
  251     throwErrorDischarged (StructuralError NonPolymorphicInstantiationMachineError) val
  252 
  253 -- | Apply a function to an argument and proceed.
  254 -- If the function is a lambda 'lam x ty body' then extend the environment with a binding of @v@
  255 -- to x@ and call 'computeCek' on the body.
  256 -- If the function is a builtin application then check that it's expecting a term argument,
  257 -- and either calculate the builtin application or stick a 'Apply' on top of its 'Term'
  258 -- representation depending on whether the application is saturated or not.
  259 -- If v is anything else, fail.
  260 applyEvaluate
  261     :: forall uni fun ann s
  262     . (ThrowableBuiltins uni fun, GivenCekReqs uni fun ann s)
  263     => Context uni fun ann
  264     -> CekValue uni fun ann   -- lhs of application
  265     -> CekValue uni fun ann   -- rhs of application
  266     -> CekM uni fun s (CekState uni fun ann)
  267 applyEvaluate !ctx (VLamAbs _ body env) arg =
  268     pure $ Computing ctx (Env.cons arg env) body
  269 -- Annotating @f@ and @exF@ with bangs gave us some speed-up, but only until we added a bang to
  270 -- 'VCon'. After that the bangs here were making things a tiny bit slower and so we removed them.
  271 applyEvaluate !ctx (VBuiltin fun term runtime) arg = do
  272     let argTerm = dischargeResultToTerm $ dischargeCekValue arg
  273         -- @term@ and @argTerm@ are fully discharged, and so @term'@ is, hence we can put it
  274         -- in a 'VBuiltin'.
  275         term' = Apply () term argTerm
  276     case runtime of
  277         -- It's only possible to apply a builtin application if the builtin expects a term
  278         -- argument next.
  279         BuiltinExpectArgument f -> evalBuiltinApp ctx fun term' $ f arg
  280         _ ->
  281           throwErrorWithCause (StructuralError UnexpectedBuiltinTermArgumentMachineError) term'
  282 applyEvaluate !_ val _ =
  283     throwErrorDischarged (StructuralError NonFunctionalApplicationMachineError) val

  299 -- | The entering point to the CEK machine's engine.
  300 enterComputeCek
  301     :: forall uni fun ann s
  302     . (ThrowableBuiltins uni fun, GivenCekReqs uni fun ann s)
  303     => Context uni fun ann
  304     -> CekValEnv uni fun ann
  305     -> NTerm uni fun ann
  306     -> CekM uni fun s (DischargeResult uni fun)
  307 enterComputeCek ctx env term = iterToFinalState $ Computing ctx env term
  308  where
  309    iterToFinalState :: CekState uni fun ann -> CekM uni fun s (DischargeResult uni fun)
  310    iterToFinalState = cekTrans
  311                       >=>
  312                       \case
  313                           Terminating t -> pure t
  314                           x             -> iterToFinalState x

  324 -- the type of our state transition function, `s -> m s` , aka `Kleisli m a a`
  325 type Trans m state = state -> m state
  326 type CekTrans uni fun ann s = Trans (CekM uni fun s) (CekState uni fun ann)
  327 
  328 -- | The state transition function of the machine.
  329 cekTrans :: forall uni fun ann s
  330            . (ThrowableBuiltins uni fun, GivenCekReqs uni fun ann s)
  331            => CekTrans uni fun ann s
  332 cekTrans = \case
  333     Starting term          -> pure $ Computing NoFrame Env.empty term
  334     Computing ctx env term -> computeCek ctx env term
  335     Returning ctx val      -> returnCek ctx val
  336     self@Terminating{}     -> pure self -- FINAL STATE, idempotent
  337 
  338 -- | Based on the supplied arguments, initialize the CEK environment and
  339 -- construct a state transition function.
  340 -- Returns the constructed transition function paired with the methods to live access the running budget.
  341 mkCekTrans
  342     :: forall cost uni fun ann m s
  343     . ( ThrowableBuiltins uni fun
  344       , PrimMonad m, s ~ PrimState m) -- the outer monad that initializes the transition function
  345     => MachineParameters CekMachineCosts fun (CekValue uni fun ann)
  346     -> ExBudgetMode cost uni fun
  347     -> EmitterMode uni fun
  348     -> Slippage
  349     -> m (CekTrans uni fun ann s, ExBudgetInfo cost uni fun s)
  350 mkCekTrans
  351         (MachineParameters caser (MachineVariantParameters costs runtime))
  352         (ExBudgetMode getExBudgetInfo)
  353         (EmitterMode getEmitterMode)
  354         slippage = do
  355     exBudgetInfo@ExBudgetInfo{_exBudgetModeSpender, _exBudgetModeGetCumulative} <- liftPrim getExBudgetInfo
  356     CekEmitterInfo{_cekEmitterInfoEmit} <- liftPrim $ getEmitterMode _exBudgetModeGetCumulative
  357     ctr <- newCounter (Proxy @CounterSize)
  358     let ?cekRuntime = runtime
  359         ?cekCaserBuiltin = caser
  360         ?cekEmitter = _cekEmitterInfoEmit
  361         ?cekBudgetSpender = _exBudgetModeSpender
  362         ?cekCosts = costs
  363         ?cekSlippage = slippage
  364         ?cekStepCounter = ctr
  365       in pure (cekTrans, exBudgetInfo)
  366          -- note that we do not call the final budget&emit getters like in `runCekM`,
  367          -- since we do not need it for our usecase.

  376 cekStateContext :: Traversal' (CekState uni fun ann) (Context uni fun ann)
  377 cekStateContext f = \case
  378     Computing k e t -> Computing <$> f k <*> pure e <*> pure t
  379     Returning k v   -> Returning <$> f k <*> pure v
  380     x               -> pure x

  382 cekStateAnn :: CekState uni fun ann -> Maybe ann
  383 cekStateAnn = \case
  384     Computing _ _ t -> pure $ termAnn t
  385     Returning ctx _ -> contextAnn ctx
  386     _               -> empty
  387 
  388 contextAnn :: Context uni fun ann -> Maybe ann
  389 contextAnn = \case
  390     FrameAwaitArg ann _ _       -> pure ann
  391     FrameAwaitFunTerm ann _ _ _ -> pure ann
  392     FrameAwaitFunConN ann _ _  -> pure ann
  393     FrameAwaitFunValueN ann _ _  -> pure ann
  394     FrameForce ann _            -> pure ann
  395     FrameConstr ann _ _ _ _ _   -> pure ann
  396     FrameCases ann _ _ _        -> pure ann
  397     NoFrame                     -> empty
  398 
  399 lenContext :: Context uni fun ann -> Word
  400 lenContext = go 0
  401     where
  402       go :: Word -> Context uni fun ann -> Word
  403       go !n = \case
  404               FrameAwaitArg _ _ k       -> go (n+1) k
  405               FrameAwaitFunTerm _ _ _ k -> go (n+1) k
  406               FrameAwaitFunConN _ _ k  -> go (n+1) k
  407               FrameAwaitFunValueN _ _ k  -> go (n+1) k
  408               FrameForce _ k            -> go (n+1) k
  409               FrameConstr _ _ _ _ _ k   -> go (n+1) k
  410               FrameCases _ _ _ k        -> go (n+1) k
  411               NoFrame                   -> 0

  440 -- | Look up a variable name in the environment.
  441 lookupVarName
  442     :: forall uni fun ann s .
  443        ThrowableBuiltins uni fun
  444     => NamedDeBruijn -> CekValEnv uni fun ann -> CekM uni fun s (CekValue uni fun ann)
  445 lookupVarName varName@(NamedDeBruijn _ varIx) varEnv =
  446     Env.contIndexOne
  447         (throwErrorWithCause (StructuralError OpenTermEvaluatedMachineError) $ Var () varName)
  448         pure
  449         varEnv
  450         (coerce varIx)

  452 -- | Take a possibly partial builtin application and
  453 --
  454 -- - either create a 'CekValue' by evaluating the application if it's saturated (emitting logs, if
  455 --    any, along the way), potentially failing evaluation
  456 -- - or create a partial builtin application otherwise
  457 --
  458 -- and proceed with the returning phase of the CEK machine.
  459 evalBuiltinApp
  460     :: (ThrowableBuiltins uni fun, GivenCekReqs uni fun ann s)
  461     => Context uni fun ann
  462     -> fun
  463     -> NTerm uni fun ()
  464     -> BuiltinRuntime (CekValue uni fun ann)
  465     -> CekM uni fun s (CekState uni fun ann)
  466 evalBuiltinApp ctx fun term runtime = case runtime of
  467     BuiltinCostedResult budgets0 getFXs -> do
  468         let exCat = BBuiltinApp fun
  469             spendBudgets (ExBudgetLast budget) = spendBudget exCat budget
  470             spendBudgets (ExBudgetCons budget budgets) =
  471                 spendBudget exCat budget *> spendBudgets budgets
  472         spendBudgets budgets0
  473         case getFXs of
  474             BuiltinSuccess y ->
  475                 returnCek ctx y
  476             BuiltinSuccessWithLogs logs y -> do
  477                 ?cekEmitter logs
  478                 returnCek ctx y
  479             BuiltinFailure logs err -> do
  480                 ?cekEmitter logs
  481                 throwBuiltinErrorWithCause term err
  482     _ -> returnCek ctx $ VBuiltin fun term runtime
  483 {-# INLINE evalBuiltinApp #-}

  485 spendBudget :: GivenCekSpender uni fun s => ExBudgetCategory fun -> ExBudget -> CekM uni fun s ()
  486 spendBudget = unCekBudgetSpender ?cekBudgetSpender

  488 -- | Spend the budget that has been accumulated for a number of machine steps.
  489 spendAccumulatedBudget :: (GivenCekReqs uni fun ann s) => CekM uni fun s ()
  490 spendAccumulatedBudget = do
  491     let ctr = ?cekStepCounter
  492     iforCounter_ ctr spend
  493     resetCounter ctr
  494   where
