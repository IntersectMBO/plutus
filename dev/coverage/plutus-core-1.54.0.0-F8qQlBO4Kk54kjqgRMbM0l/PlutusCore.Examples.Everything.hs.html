    1 -- | This module exports all available examples via a data type which allows to test
    2 -- various procedures (pretty-printing, type checking, etc) over the entire set of examples
    3 -- in a convenient way: each time a function / data type is added to examples, none of the
    4 -- tests is required to be adapted, instead you just add the new definition to 'examples'
    5 -- defined below and all the tests see it automatically.
    6 
    7 {-# LANGUAGE ScopedTypeVariables #-}
    8 
    9 module PlutusCore.Examples.Everything
   10     ( examples
   11     , builtins
   12     ) where

   33 -- | All examples exported as a single value.
   34 examples :: PlcFolderContents DefaultUni (Either DefaultFun ExtensionFun)
   35 examples =
   36     FolderContents
   37       [ treeFolderContents "Examples"
   38           [ treeFolderContents "Data"
   39               $ plcTermFile "exampleData" exampleData
   40               : [ plcTermFile (name ++ show optMatch) $ f optMatch
   41                 | optMatch <- enumerate
   42                 , (name, f) <- [("ofoldrData", ofoldrData)]
   43                 ]
   44           , treeFolderContents "Function"
   45               [ plcTermFile "unsafeCoerce" unsafeCoerce
   46               ]
   47           , treeFolderContents "InterList"
   48               [ plcTypeFile "InterList"      $ _recursiveType interListData
   49               , plcTermFile "InterNil"       interNil
   50               , plcTermFile "InterCons"      interCons
   51               , plcTermFile "FoldrInterList" foldrInterList
   52               ]
   53           , treeFolderContents "List"
   54               [ plcTermFile (name ++ show optMatch) $ f optMatch
   55               | optMatch <- enumerate
   56               , (name, f) <- [("omapList", omapList)]
   57               ]
   58           , treeFolderContents "Pair"
   59               [ plcTermFile "obothPair" obothPair
   60               ]
   61           , treeFolderContents "TreeForest"
   62               [ plcTypeFile "Tree"       $ _recursiveType treeData
   63               , plcTypeFile "Forest"     $ _recursiveType forestData
   64               , plcTermFile "TreeNode"   treeNode
   65               , plcTermFile "ForestNil"  forestNil
   66               , plcTermFile "ForestCons" forestCons
   67               ]
   68           , treeFolderContents "Vec"
   69               [ plcTypeFile "zeroT"            zeroT
   70               , plcTypeFile "succT"            succT
   71               , plcTypeFile "plusT"            plusT
   72               , plcTypeFile "churchVec"        churchVec
   73               , plcTermFile "churchNil"        churchNil
   74               , plcTermFile "churchCons"       churchCons
   75               , plcTermFile "churchConcat"     churchConcat
   76               , plcTypeFile "scottVec"         scottVec
   77               , plcTermFile "scottNil"         scottNil
   78               , plcTermFile "scottCons"        scottCons
   79               , plcTermFile "scottHead"        scottHead
   80               , plcTermFile "scottSumHeadsOr0" $ mapFun Left scottSumHeadsOr0
   81               ]
   82           , treeFolderContents "Shad"
   83               [ plcTypeFile "shad"   shad
   84               , plcTermFile "mkShad" mkShad
   85               ]
   86           , treeFolderContents "RecUnit"
   87               [ plcTypeFile "recUnit"    recUnit
   88               , plcTermFile "runRecUnit" runRecUnit
   89               ]
   90           ]
   91       ]
   92 
   93 builtins :: PlcFolderContents DefaultUni ExtensionFun
   94 builtins =
   95     FolderContents
   96         [ treeFolderContents "Builtins" $
   97             map (\fun -> plcTermFile (show fun) $ builtin () fun) enumerate
   98         ]
