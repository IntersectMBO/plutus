    1 {- | A type for maps (key-value associations), where the key type can be
    2  identified by 'Unique's. In practice, these types are usually names.
    3  This approach is preferred when it is more efficient to compare the associated
    4  'Unique's instead of the underlying type.
    5 -}

   32 {- | A mapping from 'Unique's to arbitrary values of type @a@.
   33  Since 'Unique' is equivalent to 'Int' (see "PlutusCore.Name.Unique"),
   34  we can use an 'IntMap' representation for this type.
   35 -}
   36 newtype UniqueMap unique a = UniqueMap
   37   { unUniqueMap :: IM.IntMap a
   38   }
   39   deriving stock (Show, Eq)
   40   deriving newtype (Semigroup, Monoid, Functor, Foldable)
   41 
   42 -- | Insert a value @a@ by a @unique@.
   43 insertByUnique ::
   44   (Coercible unique Unique) =>
   45   unique ->
   46   a ->
   47   UniqueMap unique a ->
   48   UniqueMap unique a
   49 insertByUnique uniq = coerce . IM.insert (coerce uniq)
   50 
   51 -- | Insert a value @a@ by the @unique@ of a @name@.
   52 insertByName :: (HasUnique name unique) => name -> a -> UniqueMap unique a -> UniqueMap unique a
   53 insertByName = insertByUnique . view unique
   54 
   55 -- | Create the singleton map of the @unique@ of a @name@ and a value @a@.
   56 singletonByName :: (HasUnique name unique) => name -> a -> UniqueMap unique a
   57 singletonByName n a = insertByName n a mempty
   58 
   59 -- | Insert a named value @a@ by the index of the @unique@ of the @name@.
   60 insertNamed ::
   61   (HasText name, HasUnique name unique) =>
   62   name ->
   63   a ->
   64   UniqueMap unique (Named a) ->
   65   UniqueMap unique (Named a)
   66 insertNamed name = insertByName name . Named (name ^. theText)
   67 
   68 {- | Insert a value by the index of the unique of a name.
   69 Unlike 'insertByUnique' and 'insertByName', this function does not provide any static guarantees,
   70 so you can for example insert by a type-level name in a map from term-level uniques.
   71 -}
   72 insertByNameIndex ::
   73   (HasUnique name unique1, Coercible unique2 Unique) =>
   74   name ->
   75   a ->
   76   UniqueMap unique2 a ->
   77   UniqueMap unique2 a
   78 insertByNameIndex = insertByUnique . coerce . view unique
   79 
   80 -- | Convert a 'Foldable' into a 'UniqueMap' using the given insertion function.
   81 fromFoldable ::
   82   (Foldable f) =>
   83   (i -> a -> UniqueMap unique a -> UniqueMap unique a) ->
   84   f (i, a) ->
   85   UniqueMap unique a
   86 fromFoldable ins = List.foldl' (flip $ uncurry ins) mempty
   87 
   88 -- | Convert a 'Foldable' with uniques into a 'UniqueMap'.
   89 fromUniques :: (Foldable f) => (Coercible Unique unique) => f (unique, a) -> UniqueMap unique a
   90 fromUniques = fromFoldable insertByUnique
   91 
   92 -- | Convert a 'Foldable' with names into a 'UniqueMap'.
   93 fromNames :: (Foldable f) => (HasUnique name unique) => f (name, a) -> UniqueMap unique a
   94 fromNames = fromFoldable insertByName
   95 
   96 -- | Look up a value by a unique.
   97 lookupUnique :: (Coercible unique Unique) => unique -> UniqueMap unique a -> Maybe a
   98 lookupUnique uniq = IM.lookup (coerce uniq) . unUniqueMap
   99 
  100 -- | Look up a value by the unique of a name.
  101 lookupName :: (HasUnique name unique) => name -> UniqueMap unique a -> Maybe a
  102 lookupName = lookupUnique . view unique
  103 
  104 restrictKeys :: UniqueMap unique v -> UniqueSet unique -> UniqueMap unique v
  105 restrictKeys (UniqueMap m) (UniqueSet s) =
  106   UniqueMap $ IM.restrictKeys m s
  107 
  108 {- | Look up a value by the index of the unique of a name.
  109 Unlike 'lookupUnique' and 'lookupName', this function does not provide any static guarantees,
  110 so you can for example look up a type-level name in a map from term-level uniques.
  111 -}
  112 lookupNameIndex ::
  113   (HasUnique name unique1, Coercible unique2 Unique) =>
  114   name ->
  115   UniqueMap unique2 a ->
  116   Maybe a
  117 lookupNameIndex = lookupUnique . coerce . view unique
