   28 {- | This type represents the result of computing a pairing using the Miller
   29    loop.  Values of this type are ephemeral, only created during script
   30    execution.  We do not provide any means of serialising, deserialising,
   31    printing, or parsing MlResult values. -}
   32 newtype MlResult = MlResult { unMlResult :: BlstBindings.PT }
   33     deriving newtype (Eq)
   34 instance Show MlResult where
   35     show _ = "<opaque>"
   36 instance Pretty MlResult where
   37     pretty = pretty . show
   38 instance PrettyBy ConstConfig MlResult where
   39     prettyBy _ = pretty
   40 -- We need a Flat instance to get everything to build properly; however we'll
   41 -- never want MlResult values in serialised scripts, so the decoding and
   42 -- encoding functions just raise errors.
   43 instance Flat MlResult where
   44     -- This might happen on the chain, so `fail` rather than `error`.
   45     decode = fail "Flat decoding is not supported for objects of type bls12_381_mlresult"
   46     -- This will be a Haskell runtime error, but encoding doesn't happen on chain,
   47     -- so it's not too bad.
   48     encode = error "Flat encoding is not supported for objects of type bls12_381_mlresult"
   49     size _ = id
   50 instance NFData MlResult where
   51     rnf _ = ()
   52 
   53 instance Hashable MlResult where
   54     hashWithSalt salt _MlResult = salt
   55 
   56 millerLoop :: G1.Element -> G2.Element -> MlResult
   57 millerLoop = coerce BlstBindings.millerLoop
   58 
   59 mulMlResult :: MlResult -> MlResult -> MlResult
   60 mulMlResult = coerce BlstBindings.ptMult
   61 
   62 finalVerify :: MlResult -> MlResult -> Bool
   63 finalVerify = coerce BlstBindings.ptFinalVerify
   64

   68 -- | Memory usage of an MlResult point (576 bytes)
   69 mlResultMemSizeBytes :: Int

   72 -- | For some of the tests we need a small element of the MlResult type.  We can
   73 -- get the identity element by pairing the zero elements of G1 and G2.
   74 identityMlResult :: MlResult
   75 identityMlResult = millerLoop G1.offchain_zero G2.offchain_zero
