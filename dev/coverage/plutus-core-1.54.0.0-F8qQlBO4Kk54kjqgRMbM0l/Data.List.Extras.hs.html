    7 -- | A variant of 'ix' that takes a 'Word64' instead of an 'Int'.
    8 wix :: Word64 -> Traversal' [a] a
    9 wix k f xs0 = go xs0 k where
   10     go [] _     = pure []
   11     go (a:as) 0 = f a <&> (:as)
   12     go (a:as) i = (a:) <$> (go as $! i - 1)
