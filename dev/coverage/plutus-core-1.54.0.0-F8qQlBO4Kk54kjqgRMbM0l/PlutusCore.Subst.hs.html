   46 substTyVarA
   47     :: Applicative f
   48     => (tyname -> f (Maybe (Type tyname uni ann)))
   49     -> Type tyname uni ann

  127 -- | Substitute the given closed 'Type' for the given type variable in the given 'Type'. Does not
  128 -- descend under binders that bind the same variable as the one we're substituting for (since from
  129 -- there that variable is no longer free). The resulting 'Term' may and likely will not satisfy
  130 -- global uniqueness.
  131 typeSubstClosedType
  132     :: Eq tyname => tyname -> Type tyname uni a -> Type tyname uni a -> Type tyname uni a
  133 typeSubstClosedType tn0 ty0 = go where
  134     go = \case
  135          TyVar    a tn      -> if tn == tn0 then ty0 else TyVar a tn
  136          TyForall a tn k ty -> TyForall a tn k (goUnder tn ty)
  137          TyLam    a tn k ty -> TyLam    a tn k (goUnder tn ty)
  138          ty                 -> ty & over typeSubtypes go
  139     goUnder tn ty = if tn == tn0 then ty else go ty
  140 
  141 -- | Substitute the given closed 'Type' for the given type variable in the given 'Term'. Does not
  142 -- descend under binders that bind the same variable as the one we're substituting for (since from
  143 -- there that variable is no longer free). The resulting 'Term' may and likely will not satisfy
  144 -- global uniqueness.
  145 termSubstClosedType
  146     :: Eq tyname
  147     => tyname -> Type tyname uni a -> Term tyname name uni fun a -> Term tyname name uni fun a
  148 termSubstClosedType tn0 ty0 = go where
  149     go = \case
  150          TyAbs a tn k body -> TyAbs a tn k (goUnder tn body)
  151          t                 -> t & over termSubtypes goTy & over termSubterms go
  152     goUnder tn term = if tn == tn0 then term else go term
  153     goTy = typeSubstClosedType tn0 ty0
  154 
  155 -- | Substitute the given closed 'Term' for the given term variable in the given 'Term'. Does not
  156 -- descend under binders that bind the same variable as the one we're substituting for (since from
  157 -- there that variable is no longer free). The resulting 'Term' may and likely will not satisfy
  158 -- global uniqueness.
  159 termSubstClosedTerm
  160     :: Eq name
  161     => name
  162     -> Term tyname name uni fun a
  163     -> Term tyname name uni fun a
  164     -> Term tyname name uni fun a
  165 termSubstClosedTerm varFor new = go where
  166     go = \case
  167          Var    a var         -> if var == varFor then new else Var a var
  168          LamAbs a var ty body -> LamAbs a var ty (goUnder var body)
  169          t                    -> t & over termSubterms go
  170     goUnder var term = if var == varFor then term else go term

  174 typeMapNames
  175     :: forall tyname tyname' uni ann
  176     .  (tyname -> tyname')
  177     -> Type tyname uni ann
  178     -> Type tyname' uni ann
  179 typeMapNames f = go
  180     where
  181       go :: Type tyname uni ann -> Type tyname' uni ann
  182       go = \case
  183            TyVar ann tn         -> TyVar ann (f tn)
  184            TyFun ann ty1 ty2    -> TyFun ann (go ty1) (go ty2)
  185            TyIFix ann ty1 ty2   -> TyIFix ann (go ty1) (go ty2)
  186            TyForall ann tn k ty -> TyForall ann (f tn) k (go ty)
  187            TyBuiltin ann s      -> TyBuiltin ann s
  188            TyLam ann tn k ty    -> TyLam ann (f tn) k (go ty)
  189            TyApp ann ty1 ty2    -> TyApp ann (go ty1) (go ty2)
  190            TySOP ann tyls       -> TySOP ann ((fmap . fmap) go tyls)
  191 
  192 -- termMapNames requires two function arguments: one (called f) to modify type names
  193 -- and another (called g) to modify variable names.
  194 termMapNames
  195     :: forall tyname tyname' name name' uni fun ann
  196     .  (tyname -> tyname')
  197     -> (name -> name')
  198     -> Term tyname name uni fun ann
  199     -> Term tyname' name' uni fun ann
  200 termMapNames f g = go
  201     where
  202         go :: Term tyname name uni fun ann -> Term tyname' name' uni fun ann
  203         go = \case
  204             LamAbs ann name ty body -> LamAbs ann (g name) (typeMapNames f ty) (go body)
  205             TyAbs ann tyname k body -> TyAbs ann (f tyname) k (go body)
  206             Var ann name            -> Var ann (g name)
  207             Apply ann t1 t2         -> Apply ann (go t1) (go t2)
  208             Constant ann c          -> Constant ann c
  209             Builtin ann b           -> Builtin ann b
  210             TyInst ann body ty      -> TyInst ann (go body) (typeMapNames f ty)
  211             Unwrap ann body         -> Unwrap ann (go body)
  212             IWrap ann ty1 ty2 body  -> IWrap ann (typeMapNames f ty1) (typeMapNames f ty2) (go body)
  213             Constr ann ty i es      -> Constr ann (typeMapNames f ty) i (fmap go es)
  214             Case ann ty arg cs      -> Case ann (typeMapNames f ty) (go arg) (fmap go cs)
  215             Error ann ty            -> Error ann (typeMapNames f ty)
  216 
  217 programMapNames
  218     :: forall tyname tyname' name name' uni fun ann
  219     .  (tyname -> tyname')
  220     -> (name -> name')
  221     -> Program tyname name uni fun ann
  222     -> Program tyname' name' uni fun ann
  223 programMapNames f g (Program a v term) = Program a v (termMapNames f g term)

  227 -- | Get all the free term variables in a term.
  228 fvTerm :: HasUnique name unique => Traversal' (Term tyname name uni fun ann) name
  229 fvTerm = fvTermCtx mempty
  230 
  231 fvTermCtx
  232     :: HasUnique name unique
  233     => UniqueSet unique -> Traversal' (Term tyname name uni fun ann) name
  234 fvTermCtx bound f = \case
  235     Var a n         -> Var a <$> (if USet.memberByName n bound then pure n else f n)
  236     LamAbs a n ty t -> LamAbs a n ty <$> fvTermCtx (USet.insertByName n bound) f t
  237     t               -> (termSubterms . fvTermCtx bound) f t
  238 
  239 -- | Get all the free type variables in a term.
  240 ftvTerm :: HasUnique tyname unique => Traversal' (Term tyname name uni fun ann) tyname
  241 ftvTerm = ftvTermCtx mempty
  242 
  243 ftvTermCtx
  244     :: HasUnique tyname unique
  245     => UniqueSet unique
  246     -> Traversal' (Term tyname name uni fun ann) tyname
  247 ftvTermCtx bound f = \case
  248     TyAbs a ty k t -> TyAbs a ty k <$> ftvTermCtx (USet.insertByName ty bound) f t
  249     -- sound because the subterms and subtypes are disjoint
  250     t              ->
  251         ((termSubterms . ftvTermCtx bound) `Unsound.adjoin` (termSubtypes . ftvTyCtx bound)) f t
  252 
  253 -- | Get all the free type variables in a type.
  254 ftvTy
  255     :: HasUnique tyname unique =>
  256     Traversal' (Type tyname uni ann) tyname
  257 ftvTy = ftvTyCtx mempty
  258 
  259 ftvTyCtx
  260     :: HasUnique tyname unique
  261     => UniqueSet unique
  262     -> Traversal' (Type tyname uni ann) tyname
  263 ftvTyCtx bound f = \case
  264     TyVar a ty          -> TyVar a <$> (if USet.memberByName ty bound then pure ty else f ty)
  265     TyForall a bnd k ty -> TyForall a bnd k <$> ftvTyCtx (USet.insertByName bnd bound) f ty
  266     TyLam a bnd k ty    -> TyLam a bnd k <$> ftvTyCtx (USet.insertByName bnd bound) f ty
  267     t                   -> (typeSubtypes . ftvTyCtx bound) f t
