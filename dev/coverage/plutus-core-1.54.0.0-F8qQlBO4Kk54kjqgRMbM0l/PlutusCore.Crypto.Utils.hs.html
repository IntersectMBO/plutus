   13 failWithMessage :: forall (a :: Type). Text -> Text -> BuiltinResult a

   16   builtinResultFailure

   18 byteStringAsHex :: ByteString -> String
   19 byteStringAsHex bs = "0x" ++ (Prelude.concat $ foldr' (\w s -> (printf "%02x" w):s) [] bs)
