   35 {- Note [MakeKnown in TypeSchemeArrow]
   36 There's a @MakeKnown val arg@ constrained packed in the 'TypeSchemeArrow' constructor. It's not
   37 supposed to be there, but unfortunately, in the @Generators@ tests we use 'TypeScheme' for
   38 generation of arbitrary arguments of builtins and that requires 'makeKnown', which makes us have
   39 the 'MakeKnown' in 'TypeSchemeArrow'.
   40 
   41 The solution is to fix the @Generators@ tests. Explicitly constraining @args@ outside of
   42 'TypeScheme' sounds like a promising strategy. Maybe we could just delete those tests altogether.
   43 
   44 However it's also worth considering untangling 'RuntimeScheme' from 'TypeScheme' and generating the
   45 two in parallel, so that we only need to optimize the former. Then we will be able to afford having
   46 any kind of nonsense in 'TypeScheme'. Another reason for that would be the fact that Core output
   47 has 'typeSchemeToRuntimeScheme' all over the place as it can't be inlined due to being a recursive
   48 function, which we can't turn into an inlinable class method, because the indices of 'TypeScheme'
   49 don't reflect its structure due to the 'TypeSchemeAll' constructor not being reflected at the type
   50 level in any way. It's unlikely that having those 'typeSchemeToRuntimeScheme' has any impact on
   51 performance, because they're only evaluated once during initialization, but it certainly has impact
   52 on readability of the Core output.
   53 -}

   55 -- We have these 'Typeable' constraints here just for the generators tests. It's fine since
   56 -- 'TypeScheme' is not used for evaluation and so we can shove into 'TypeScheme' whatever we want.
   57 -- | The type of type schemes of built-in functions.
   58 -- @args@ is a list of types of arguments, @res@ is the resulting type.
   59 -- E.g. @Text -> Bool -> Integer@ is encoded as @TypeScheme val [Text, Bool] Integer@.
   60 data TypeScheme val (args :: [GHC.Type]) res where
   61     TypeSchemeResult
   62         :: (Typeable res, KnownTypeAst TyName (UniOf val) res, MakeKnown val res)
   63         => TypeScheme val '[] res
   64     TypeSchemeArrow
   65         :: (Typeable arg, KnownTypeAst TyName (UniOf val) arg, MakeKnown val arg, ReadKnown val arg)
   66         => TypeScheme val args res -> TypeScheme val (arg ': args) res
   67     TypeSchemeAll
   68         :: (KnownSymbol text, KnownNat uniq, KnownKind kind)
   69         => Proxy '(text, uniq, kind)
   70         -> TypeScheme val args res
   71         -> TypeScheme val args res
   72 
   73 argProxy :: TypeScheme val (arg ': args) res -> Proxy arg
   74 argProxy _ = Proxy
   75 
   76 -- | Convert a 'TypeScheme' to the corresponding 'Type'.
   77 typeSchemeToType :: TypeScheme val args res -> Type TyName (UniOf val) ()
   78 typeSchemeToType sch@TypeSchemeResult       = toTypeAst sch
   79 typeSchemeToType sch@(TypeSchemeArrow schB) =
   80     TyFun () (toTypeAst $ argProxy sch) $ typeSchemeToType schB
   81 typeSchemeToType (TypeSchemeAll (_ :: Proxy '(text, uniq, kind)) schB) =
   82     let text = Text.pack $ symbolVal @text Proxy
   83         uniq = fromIntegral $ natVal @uniq Proxy
   84         a    = TyName $ Name text $ Unique uniq
   85     in TyForall () a (demoteKind $ knownKind @kind) $ typeSchemeToType schB
   86 
   87 -- The precedence of @->@ is @-1@, which is why this number appears in the implementation of the
   88 -- instance.
   89 instance Show (TypeScheme val args res) where
   90     showsPrec p sch@TypeSchemeResult =
   91         showParen (p > 0)
   92             $ showsPrec (-1) (typeRep sch)
   93     showsPrec p sch@(TypeSchemeArrow schB) =
   94         showParen (p > 0)
   95             $ -- @0@ is to account for associativity, see https://stackoverflow.com/a/43639618
   96               showsPrec 0 (typeRep $ argProxy sch)
   97             . showString " -> "
   98             . showsPrec (-1) schB
   99     showsPrec p (TypeSchemeAll (_ :: Proxy '(text, uniq, kind)) schB) =
  100         showParen (p > 0)
  101             $ showString "forall "
  102             . showString (symbolVal @text Proxy)
  103             . showString ". "
  104             . showsPrec 0 schB
