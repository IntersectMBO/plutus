   45 -- | Errors which can occur during a run of an abstract machine.
   46 data MachineError fun
   47     = NonPolymorphicInstantiationMachineError
   48       -- ^ An attempt to reduce a not immediately reducible type instantiation.
   49     | NonWrapUnwrappedMachineError
   50       -- ^ An attempt to unwrap a not wrapped term.
   51     | NonFunctionalApplicationMachineError
   52       -- ^ An attempt to reduce a not immediately reducible application.
   53     | OpenTermEvaluatedMachineError
   54       -- ^ An attempt to evaluate an open term.
   55     | UnliftingMachineError UnliftingError
   56       -- ^ An attempt to compute a constant application resulted in 'UnliftingError'.
   57     | BuiltinTermArgumentExpectedMachineError
   58       -- ^ A builtin expected a term argument, but something else was received.
   59     | UnexpectedBuiltinTermArgumentMachineError
   60       -- ^ A builtin received a term argument when something else was expected
   61     | NonConstrScrutinizedMachineError
   62       -- ^ An attempt to scrutinize a non-constr.
   63     | MissingCaseBranchMachineError Word64
   64       -- ^ An attempt to go into a non-existent case branch.
   65     | PanicMachineError String
   66       -- ^ A GHC exception was thrown.
   67     deriving stock (Show, Eq, Functor, Generic)
   68     deriving anyclass (NFData)

   70 type EvaluationException structural operational =
   71     ErrorWithCause (EvaluationError structural operational)

   73 {- Note [Ignoring context in OperationalError]
   74 The 'OperationalError' error has a term argument, but 'splitStructuralOperational' just
   75 discards this and returns 'EvaluationFailure'. This means that, for example, if we use the @plc@
   76 command to execute a program containing a division by zero, @plc@ exits silently without reporting
   77 that anything has gone wrong (but returning a non-zero exit code to the shell via 'exitFailure').
   78 This is because 'OperationalError' is used in cases when a PLC program itself goes wrong
   79 (see the Haddock of 'EvaluationError'). This is used to signal unsuccessful validation and so is
   80 not regarded as a real error; in contrast structural errors are genuine errors and we report their
   81 context if available.
   82 -}
   83 
   84 -- See the Haddock of 'EvaluationError' for what structural and operational errors are.
   85 -- See Note [Ignoring context in OperationalError].
   86 -- | Preserve the contents of an 'StructuralError' as a 'Left' and turn an
   87 -- 'OperationalError' into a @Right EvaluationFailure@ (thus erasing the content of the
   88 -- error in the latter case).
   89 splitStructuralOperational
   90     :: Either (EvaluationException structural operational term) a
   91     -> Either (ErrorWithCause structural term) (EvaluationResult a)
   92 splitStructuralOperational (Right term) = Right $ EvaluationSuccess term
   93 splitStructuralOperational (Left (ErrorWithCause evalErr cause)) = case evalErr of
   94     StructuralError err -> Left $ ErrorWithCause err cause
   95     OperationalError _  -> Right EvaluationFailure
   96 
   97 -- | Throw on a 'StructuralError' and turn an 'OperationalError' into an
   98 -- 'EvaluationFailure' (thus erasing the content of the error in the latter case).
   99 unsafeSplitStructuralOperational
  100     :: (PrettyPlc structural, PrettyPlc term, Typeable structural, Typeable term)
  101     => Either (EvaluationException structural operational term) a
  102     -> EvaluationResult a
  103 unsafeSplitStructuralOperational = unsafeFromEither . splitStructuralOperational

  105 instance (HasPrettyDefaults config ~ 'True, Pretty fun) =>
  106             PrettyBy config (MachineError fun) where
  107     prettyBy _      NonPolymorphicInstantiationMachineError =
  108         "Attempted to instantiate a non-polymorphic term."
  109     prettyBy _      NonWrapUnwrappedMachineError          =
  110         "Cannot unwrap a not wrapped term."
  111     prettyBy _      NonFunctionalApplicationMachineError   =
  112         "Attempted to apply a non-function."
  113     prettyBy _      OpenTermEvaluatedMachineError         =
  114         "Cannot evaluate an open term"
  115     prettyBy _      BuiltinTermArgumentExpectedMachineError =
  116         "A builtin expected a term argument, but something else was received"
  117     prettyBy _      UnexpectedBuiltinTermArgumentMachineError =
  118         "A builtin received a term argument when something else was expected"
  119     prettyBy _      (UnliftingMachineError unliftingError)  =
  120         pretty unliftingError
  121     prettyBy _      NonConstrScrutinizedMachineError =
  122         "A non-constructor/non-builtin value was scrutinized in a case expression"
  123     prettyBy _      (MissingCaseBranchMachineError i) =
  124         "Case expression missing the branch required by the scrutinee tag:" <+> pretty i
  125     prettyBy _      (PanicMachineError err) = vcat
  126         [ "Panic: a GHC exception was thrown, please report this as a bug."
  127         , "The error: " <+> pretty err
  128         ]

  130 class BuiltinErrorToEvaluationError structural operational where
  131   builtinErrorToEvaluationError :: BuiltinError -> EvaluationError structural operational
  132 
  133 -- | Attach a @cause@ to a 'BuiltinError' and throw that.
  134 -- Note that an evaluator might require the cause to be computed lazily for best performance on the
  135 -- happy path, hence this function must not force its first argument.
  136 -- TODO: wrap @cause@ in 'Lazy' once we have it.
  137 throwBuiltinErrorWithCause
  138     :: ( MonadError (EvaluationException structural operational cause) m
  139        , BuiltinErrorToEvaluationError structural operational
  140        )
  141     => cause -> BuiltinError -> m void
  142 throwBuiltinErrorWithCause cause e = throwErrorWithCause (builtinErrorToEvaluationError e) cause
  143 {-# INLINE throwBuiltinErrorWithCause #-}
