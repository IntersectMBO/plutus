    4 -- TODO: Extend this to handle the different variants of the cost model
    5 {- | A program to parse a JSON representation of costing functions for Plutus Core
    6    builtins and and produce a simple cost model which can be used from Agda and other
    7    executables -}
    8 module PlutusCore.Evaluation.Machine.SimpleBuiltinCostModel
    9    ( BuiltinCostMap
   10    , BuiltinCostKeyMap
   11    , toSimpleBuiltinCostModel
   12    , defaultSimpleBuiltinCostModel
   13    ) where
   14 
   15 import Data.Aeson.Key as Key (toText)
   16 import Data.Aeson.KeyMap qualified as KeyMap
   17 import Data.Aeson.THReader (readJSONFromFile)
   18 import Data.Bifunctor
   19 import Data.Text (Text, replace)
   20 import PlutusCore.DataFilePaths qualified as DFP
   21 import PlutusCore.Evaluation.Machine.CostingFun.SimpleJSON
   22 
   23 type BuiltinCostMap = [(Text, CpuAndMemoryModel)]
   24 type BuiltinCostKeyMap = KeyMap.KeyMap CpuAndMemoryModel
   25 
   26 -- | The default builtin cost map.
   27  -- TODO: maybe we should take account of the semantic variant here.
   28 defaultBuiltinCostKeyMap :: BuiltinCostKeyMap
   29 defaultBuiltinCostKeyMap =
   30     $$(readJSONFromFile DFP.latestBuiltinCostModelFile)
   31 
   32 -- replace underscores _ by dashes -
   33 builtinName :: Text -> Text
   34 builtinName = replace "_" "-"
   35 
   36 toSimpleBuiltinCostModel :: BuiltinCostKeyMap -> BuiltinCostMap
   37 toSimpleBuiltinCostModel m = map (first (builtinName . toText)) (KeyMap.toList m)
   38 
   39 defaultSimpleBuiltinCostModel :: BuiltinCostMap
   40 defaultSimpleBuiltinCostModel = toSimpleBuiltinCostModel defaultBuiltinCostKeyMap
