   40 -- | Configuration for the readable pretty-printing.
   41 data PrettyConfigReadable configName = PrettyConfigReadable
   42   { _pcrConfigName    :: configName
   43   , _pcrRenderContext :: RenderContext

   73 {- | For rendering things in a readable manner regardless of the pretty-printing function chosen.
   74 I.e. all of 'show', 'pretty', 'prettyClassic' will use 'PrettyReadable' instead of doing what
   75 they normally do. @prettyBy config (AsReadable x)@ requires @config@ to have a 'PrettyConfigName'
   76 and respects it.
   77 
   78 This wrapper can be particularly useful if you want to apply a function having a 'Show' or
   79 'Pretty' or 'PrettyClassic' or 'PrettyPlc' or whatever constraint, but want to get the argument
   80 rendered in a readable manner instead.
   81 -}
   82 newtype AsReadable a = AsReadable
   83   { unAsReadable :: a
   84   }
   85 
   86 instance
   87   (HasPrettyConfigName config, PrettyReadable a) =>
   88   DefaultPrettyBy config (AsReadable a)
   89   where
   90   defaultPrettyBy config (AsReadable x) =
   91     prettyBy (botPrettyConfigReadable (toPrettyConfigName config) def) x
   92 
   93 instance (PrettyReadable a) => Show (AsReadable a) where
   94   show = displayBy $ Sole prettyConfigName

   96 instance (PrettyReadable a) => Pretty (AsReadable a) where
   97   pretty = viaShow

  104 -- | A value of type @a@ to render in parens using the readable pretty-printer.
  105 data Parened a = Parened
  106     { parenOpening :: String
  107     , parenClosing :: String
  108     , parenedValue :: a
  109     }
  110 
  111 instance PrettyReadableBy configName a =>
  112         PrettyBy (PrettyConfigReadable configName) (Parened a) where
  113     prettyBy config (Parened opening closing x) = fold
  114         [ pretty opening
  115         , prettyBy (config & renderContext .~ botRenderContext) x
  116         , pretty closing
  117         ]
  118 
  119 -- | Enclose the given value, so that it's rendered inside of braces with no additional parens
  120 -- regardless of the 'RenderContext'.
  121 inBraces :: a -> Parened a
  122 inBraces = Parened "{" "}"

  140 {- | Lay out an iterated binder. For example, this function lays out iterated lambdas either as
  141 
  142 > \(x : a) (y : b) (z : c) -> body
  143 
  144 or as
  145 
  146 > \(x : a)
  147 >  (y : b)
  148 >  (z : c) ->
  149 >   body
  150 -}
  151 iterBinderPrettyM ::
  152   ( MonadPrettyReadable configName env m
  153   , PrettyReadableBy configName arg
  154   , PrettyReadableBy configName body
  155   ) =>
  156   (Doc ann -> Doc ann) ->
  157   [arg] ->
  158   body ->
  159   m (Doc ann)
  160 iterBinderPrettyM enframe args body =
  161   infixDocM binderFixity $ \prettyBind prettyBody ->
  162     let prettyBinds = align . sep $ map prettyBind args
  163     in enframe prettyBinds <?> prettyBody body
  164 
  165 -- | Lay out an iterated 'TyForall' via 'iterBinderPrettyM'.
  166 iterTyForallPrettyM ::
  167   ( MonadPrettyReadable configName env m
  168   , PrettyReadableBy configName arg
  169   , PrettyReadableBy configName body
  170   ) =>
  171   [arg] ->
  172   body ->
  173   m (Doc ann)
  174 iterTyForallPrettyM = iterBinderPrettyM $ \binds -> "all" <+> binds <> "."
  175 
  176 -- | Lay out an iterated 'LamAbs' via 'iterBinderPrettyM'.
  177 iterLamAbsPrettyM ::
  178   ( MonadPrettyReadable configName env m
  179   , PrettyReadableBy configName arg
  180   , PrettyReadableBy configName body
  181   ) =>
  182   [arg] ->
  183   body ->
  184   m (Doc ann)
  185 iterLamAbsPrettyM = iterBinderPrettyM $ \binds -> "\\" <> binds <+> "->"

  187 -- | Lay out an iterated 'TyAbs' via 'iterBinderPrettyM'.
  188 iterTyAbsPrettyM ::
  189   ( MonadPrettyReadable configName env m
  190   , PrettyReadableBy configName arg
  191   , PrettyReadableBy configName body
  192   ) =>
  193   [arg] ->
  194   body ->
  195   m (Doc ann)

  198 -- | Lay out an iterated @->@.
  199 iterArrowPrettyM ::
  200   (MonadPrettyReadable configName env m, PrettyReadableBy configName a) =>
  201   [a] ->
  202   a ->
  203   m (Doc ann)
  204 iterArrowPrettyM args res =
  205   infixDocM arrowFixity $ \prettyArg prettyRes ->
  206     align . sep $ map (\x -> prettyArg x <+> "->") args ++ [prettyRes res]

  208 -- | The type of a 'PrettyConfigReadable'-based pretty-printer, similar to 'AnyToDoc'.
  209 type ReadableToDoc configName ann = forall a. PrettyReadableBy configName a => a -> Doc ann
  210 
  211 -- | Lay out an iteration application, providing to the caller a function to render the head of the
  212 -- application and a function to render each of the arguments.
  213 iterAppDocM ::
  214    MonadPrettyContext config env m =>
  215    (AnyToDoc config ann -> AnyToDoc config ann -> NonEmpty (Doc ann)) ->
  216    m (Doc ann)
  217 iterAppDocM k =
  218   infixDocM juxtFixity $ \prettyFun prettyArg ->
  219     let fun :| args = k prettyFun prettyArg
  220     in if null args
  221         then fun
  222         else fun <?> vsep args
  223 
  224 {- | Lay out iterated function applications either as
  225 
  226 > foo x y z
  227 
  228 or as
  229 
  230 > foo
  231 >   x
  232 >   y
  233 >   z
  234 -}
  235 iterAppPrettyM ::
  236   ( MonadPrettyContext config env m
  237   , PrettyBy config fun, PrettyBy config term
  238   ) =>
  239   fun ->
  240   [term] ->
  241   m (Doc ann)
  242 iterAppPrettyM fun args =
  243   iterAppDocM $ \prettyFun prettyArg ->
  244     prettyFun fun :| map prettyArg args
  245 
  246 {- | Lay out interleaved function applications either as
  247 
  248 > foo {a} x {b} y z
  249 
  250 or as
  251 
  252 > foo
  253 >   {a}
  254 >   x
  255 >   {b}
  256 >   y
  257 >   z
  258 
  259 'Left's are laid out in braces, 'Right's are laid out without braces.
  260 -}
  261 iterInterAppPrettyM ::
  262   ( MonadPrettyReadable configName env m
  263   , PrettyReadableBy configName fun
  264   , PrettyReadableBy configName ty
  265   , PrettyReadableBy configName term
  266   ) =>
  267   fun ->
  268   [Either ty term] ->
  269   m (Doc ann)
  270 iterInterAppPrettyM fun args =
  271   iterAppDocM $ \prettyFun prettyArg ->
  272     let ppArg (Left ty)    = prettyArg $ inBraces ty
  273         ppArg (Right term) = prettyArg term
  274     in prettyFun fun :| map ppArg args
