   20 class AnnotateCaseBuiltin uni where
   21     -- | Given a tag for a built-in type and a list of branches, annotate each of the branches with
   22     -- its expected argument types or fail if casing on values of the built-in type isn't supported.
   23     -- Note: you don't need to include the resulting type of the whole case matching in the
   24     -- returning list here.
   25     annotateCaseBuiltin
   26         :: UniOf term ~ uni
   27         => Type TyName uni ann
   28         -> [term]
   29         -> Either Text [(term, [Type TyName uni ann])]
   30 
   31 class CaseBuiltin uni where
   32     -- | Given a constant with its type tag and a vector of branches, choose the appropriate branch
   33     -- or fail if the constant doesn't correspond to any of the branches (or casing on constants of
   34     -- this type isn't supported at all).
   35     caseBuiltin
   36         :: UniOf term ~ uni
   37         => Some (ValueOf uni)
   38         -> Vector term
   39         -> Either Text (HeadSpine term (Some (ValueOf uni)))
   40 
   41 -- See Note [DO NOT newtype-wrap functions].
   42 -- | A @data@ version of 'CaseBuiltin'. we parameterize the evaluator by a 'CaserBuiltin' so that
   43 -- the caller can choose whether to use the 'caseBuiltin' method or the always failing caser (the
   44 -- latter is required for earlier protocol versions when we didn't support casing on builtins).
   45 data CaserBuiltin uni = CaserBuiltin
   46     { unCaserBuiltin
   47         :: !(forall term. UniOf term ~ uni => Some (ValueOf uni) -> Vector term -> Either Text (HeadSpine term (Some (ValueOf uni))))
   48     }
   49 
   50 instance NFData (CaserBuiltin uni) where
   51     rnf = rwhnf
   52 
   53 deriving via OnlyCheckWhnfNamed "PlutusCore.Builtin.Case.CaserBuiltin" (CaserBuiltin uni)
   54     instance NoThunks (CaserBuiltin uni)
   55 
   56 instance CaseBuiltin uni => Default (CaserBuiltin uni) where
   57     def = CaserBuiltin caseBuiltin
   58 
   59 unavailableCaserBuiltin :: Int -> CaserBuiltin uni
   60 unavailableCaserBuiltin ver =
   61     CaserBuiltin $ \_ _ -> Left $
   62         "'case' on values of built-in types is not supported in protocol version " <> display ver
