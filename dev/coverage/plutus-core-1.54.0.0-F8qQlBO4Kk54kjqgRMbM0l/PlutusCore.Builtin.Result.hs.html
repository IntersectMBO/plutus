   52 -- | The type of errors that 'readKnown' and 'makeKnown' can return.
   53 data BuiltinError
   54     = BuiltinUnliftingEvaluationError UnliftingEvaluationError
   55     | BuiltinEvaluationFailure
   56     deriving stock (Show, Eq)
   57 
   58 -- | The monad that 'makeKnown' runs in.
   59 -- Equivalent to @ExceptT BuiltinError (Writer (DList Text))@, except optimized in two ways:
   60 --
   61 -- 1. everything is strict
   62 -- 2. has the 'BuiltinSuccess' constructor that is used for returning a value with no logs
   63 --    attached, which is the most common case for us, so it helps a lot not to construct and
   64 --    deconstruct a redundant tuple
   65 --
   66 -- Moving from @ExceptT BuiltinError (Writer (DList Text))@ to this data type gave us a speedup of
   67 -- 8% of total evaluation time.
   68 --
   69 -- Logs are represented as a 'DList', because we don't particularly care about the efficiency of
   70 -- logging, since there's no logging on the chain and builtins don't emit much anyway. Otherwise
   71 -- we'd have to use @text-builder@ or @text-builder-linear@ or something of this sort.
   72 data BuiltinResult a
   73     = -- 'BuiltinSuccess' is the first constructor to make it a bit more likely for GHC to
   74       -- branch-predict it (which is something that we want, because most builtins return this
   75       -- constructor). It is however not guaranteed that GHC will predict it, because even though
   76       -- it's likely going to be a recursive case (it certainly is in the CEK machine) and thus the
   77       -- constructor has precedence over 'BuiltinFailure', it doesn't have precedence over
   78       -- 'BuiltinSuccessWithLogs', since that case is equally likely to be recursive.
   79       --
   80       -- Unfortunately, GHC doesn't offer any explicit control over branch-prediction (see this
   81       -- ticket: https://gitlab.haskell.org/ghc/ghc/-/issues/849), so relying on hope is the best we
   82       -- can do here.
   83       BuiltinSuccess a
   84     | BuiltinSuccessWithLogs (DList Text) a
   85     | BuiltinFailure (DList Text) BuiltinError
   86     deriving stock (Show, Foldable)

  104 {- Note [INLINE and OPAQUE on error-related definitions]
  105 We mark error-related definitions such as prisms like 'structuralUnliftingError' and regular
  106 functions like 'notAConstant' with @INLINE@, because this produces significantly less cluttered
  107 GHC Core. Not doing so results in 20+% larger Core for builtins.
  108 
  109 However in a few specific cases we use @OPAQUE@ instead to get tighter Core. @OPAQUE@ is the
  110 same as @NOINLINE@ except the former _actually_ prevents GHC from inlining the definition unlike the
  111 former. See this for details: https://github.com/ghc-proposals/ghc-proposals/blob/5577fd008924de8d89cfa9855fa454512e7dcc75/proposals/0415-opaque-pragma.rst
  112 
  113 It's hard to predict where @OPAQUE@ instead of @INLINE@ will help to make GHC Core tidier, so it's
  114 mostly just looking into the Core and seeing where there's obvious duplication that can be removed.
  115 Such cases tend to be functions returning a value of a concrete error type (as opposed to a type
  116 variable).
  117 -}

  119 -- See Note [Ignoring context in OperationalError].
  120 -- | Construct a prism focusing on the @*EvaluationFailure@ part of @err@ by taking
  121 -- that @*EvaluationFailure@ and
  122 --
  123 -- 1. pretty-printing and embedding it into an 'UnliftingError' for the setter part of the prism
  124 -- 2. returning it directly for the opposite direction (there's no other way to convert an
  125 --    'UnliftingError' to an evaluation failure, since the latter doesn't carry any content)
  126 --
  127 -- This is useful for providing 'AsUnliftingError' instances for types such as 'CkUserError' and
  128 -- 'CekUserError'.
  129 
  130 operationalUnliftingError :: Text -> BuiltinError
  131 operationalUnliftingError =
  132   BuiltinUnliftingEvaluationError . MkUnliftingEvaluationError . OperationalError . MkUnliftingError
  133 {-# INLINE operationalUnliftingError #-}
  134 
  135 structuralUnliftingError :: Text -> BuiltinError
  136 structuralUnliftingError =
  137   BuiltinUnliftingEvaluationError . MkUnliftingEvaluationError . StructuralError . MkUnliftingError
  138 {-# INLINE structuralUnliftingError #-}

  140 notAConstant :: BuiltinError

  148 -- | Add a log line to the logs.
  149 emit :: Text -> BuiltinResult ()
  150 emit txt = BuiltinSuccessWithLogs (pure txt) ()
  151 {-# INLINE emit #-}
  152 
  153 -- | Prepend logs to a 'BuiltinResult' computation.
  154 withLogs :: DList Text -> BuiltinResult a -> BuiltinResult a
  155 withLogs logs1 = \case
  156     BuiltinSuccess x               -> BuiltinSuccessWithLogs logs1 x
  157     BuiltinSuccessWithLogs logs2 x -> BuiltinSuccessWithLogs (logs1 <> logs2) x
  158     BuiltinFailure logs2 err       -> BuiltinFailure (logs1 <> logs2) err
  159 {-# INLINE withLogs #-}
  160 
  161 instance Functor BuiltinResult where
  162     fmap f (BuiltinSuccess x)              = BuiltinSuccess (f x)
  163     fmap f (BuiltinSuccessWithLogs logs x) = BuiltinSuccessWithLogs logs (f x)
  164     fmap _ (BuiltinFailure logs err)       = BuiltinFailure logs err
  165     {-# INLINE fmap #-}
  166 
  167     -- Written out explicitly just in case.
  168     x <$ BuiltinSuccess _              = BuiltinSuccess x
  169     x <$ BuiltinSuccessWithLogs logs _ = BuiltinSuccessWithLogs logs x
  170     _ <$ BuiltinFailure logs err       = BuiltinFailure logs err
  171     {-# INLINE (<$) #-}
  172 
  173 instance Applicative BuiltinResult where
  174     pure = BuiltinSuccess
  175     {-# INLINE pure #-}
  176 
  177     BuiltinSuccess f              <*> a = fmap f a
  178     BuiltinSuccessWithLogs logs f <*> a = withLogs logs $ fmap f a
  179     BuiltinFailure logs err       <*> _ = BuiltinFailure logs err
  180     {-# INLINE (<*>) #-}
  181 
  182     -- Better than the default implementation, because the value in the 'BuiltinSuccess' case
  183     -- doesn't need to be retained.
  184     BuiltinSuccess _              *> b = b
  185     BuiltinSuccessWithLogs logs _ *> b = withLogs logs b
  186     BuiltinFailure logs err       *> _ = BuiltinFailure logs err
  187     {-# INLINE (*>) #-}
  188 
  189 instance Monad BuiltinResult where
  190     BuiltinSuccess x              >>= f = f x
  191     BuiltinSuccessWithLogs logs x >>= f = withLogs logs $ f x
  192     BuiltinFailure logs err       >>= _ = BuiltinFailure logs err
  193     {-# INLINE (>>=) #-}
  194 
  195     (>>) = (*>)
  196     {-# INLINE (>>) #-}

  198 -- | 'throwError' puts every operational unlifting error into the 'BuiltinFailure' logs. This is to
  199 -- compensate for the historical lack of error message content in operational errors (structural
  200 -- ones don't have this problem) in our evaluators (the CK and CEK machines). It would be better to
  201 -- fix the underlying issue and allow operational evaluation errors to carry some form of content,
  202 -- but for now we just fix the symptom in order for the end user to see the error message that they
  203 -- are supposed to see. The fix even makes some sense: what we do here is we emulate logging when
  204 -- the thrown unlifting error is an operational one, i.e. this is similar to what some builtins do
  205 -- manually (like when a crypto builtin fails and puts info about the failure into the logs).
  206 instance MonadError BuiltinError BuiltinResult where
  207     throwError builtinErr = BuiltinFailure operationalLogs builtinErr where
  208         operationalLogs = case builtinErr of
  209             BuiltinUnliftingEvaluationError
  210                 (MkUnliftingEvaluationError
  211                     (OperationalError
  212                         (MkUnliftingError operationalErr))) -> pure operationalErr
  213             _ -> mempty
  214     {-# INLINE throwError #-}
  215 
  216     -- Throwing logs out is lame, but embedding them into the error would be weird, since that
  217     -- would change the error. Not that any of that matters, we only implement this because it's a
  218     -- method of 'MonadError' and we can't not implement it.
  219     --
  220     -- We could make it @MonadError (DList Text, BuiltinError)@, but logs are arbitrary and are not
  221     -- necessarily an inherent part of an error, so preserving them is as questionable as not doing
  222     -- so.
  223     BuiltinFailure _ err `catchError` f = f err
  224     res                  `catchError` _ = res
  225     {-# INLINE catchError #-}
  226 
  227 builtinResultFailure :: BuiltinResult a
  228 builtinResultFailure = BuiltinFailure mempty BuiltinEvaluationFailure
  229 {-# INLINE builtinResultFailure #-}
