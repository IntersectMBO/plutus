   10 -- | Is the next argument a term or a type?
   11 data Param =
   12     TermParam | TypeParam
   13     deriving stock (Show, Eq)
   14 
   15 instance Pretty Param where
   16   pretty = viaShow
   17 
   18 {-|
   19 The (syntactic) arity of a term. That is, a record of the arguments that the
   20 term expects before it may do some work. Since we have both type and lambda
   21 abstractions, this is not a simple argument count, but rather a list of values
   22 indicating whether the next argument should be a term or a type.
   23 
   24 Note that this is the syntactic arity, i.e. it just corresponds to the number of
   25 syntactic lambda and type abstractions on the outside of the term. It is thus
   26 an under-approximation of how many arguments the term may need.
   27 e.g. consider the term @let id = \x -> x in id@: the variable @id@ has syntactic
   28 arity @[]@, but does in fact need an argument before it does any work.
   29 -}
   30 type Arity = [Param]
   31 
   32 -- | Get the 'Arity' from a 'TypeScheme'.
   33 typeSchemeArity :: TypeScheme val args res -> Arity
   34 typeSchemeArity TypeSchemeResult{}    = []
   35 typeSchemeArity (TypeSchemeArrow sch) = TermParam : typeSchemeArity sch
   36 typeSchemeArity (TypeSchemeAll _ sch) = TypeParam : typeSchemeArity sch
   37 
   38 -- | Get the arity of a builtin function from the 'PLC.BuiltinSemanticsVariant'.
   39 builtinArity
   40     :: forall uni fun
   41     . ToBuiltinMeaning uni fun
   42     => Proxy uni
   43     -> BuiltinSemanticsVariant fun
   44     -> fun
   45     -> Arity
   46 builtinArity _ semvar fun =
   47   case toBuiltinMeaning @uni @fun @(Term TyName Name uni fun ()) semvar fun of
   48     BuiltinMeaning sch _ _ -> typeSchemeArity sch
