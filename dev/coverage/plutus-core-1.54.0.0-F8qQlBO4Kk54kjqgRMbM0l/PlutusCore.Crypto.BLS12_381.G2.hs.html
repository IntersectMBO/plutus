   38 {- | See Note [Wrapping the BLS12-381 types in Plutus Core]. -}
   39 newtype Element = Element { unElement :: BlstBindings.Point2 }
   40     deriving newtype (Eq)
   41 instance Show Element where
   42     show  = byteStringAsHex . compress
   43 instance Pretty Element where
   44     pretty = pretty . show
   45 instance PrettyBy ConstConfig Element
   46 {- | We don't support direct flat encoding of G1 elements because of the expense
   47    of on-chain uncompression.  Users should convert between G1 elements and
   48    bytestrings using `compress` and `uncompress`: the bytestrings can be
   49    flat-encoded in the usual way. -}
   50 instance Flat Element where
   51     -- This might happen on the chain, so `fail` rather than `error`.
   52     decode = fail "Flat decoding is not supported for objects of type bls12_381_G2_element: use bls12_381_G2_uncompress on a bytestring instead."
   53     -- This will be a Haskell runtime error, but encoding doesn't happen on chain,
   54     -- so it's not too bad.
   55     encode = error "Flat encoding is not supported for objects of type bls12_381_G2_element: use bls12_381_G2_compress to obtain a bytestring instead."
   56     size _ = id
   57 instance NFData Element where
   58     rnf (Element x) = rwhnf x  -- Just to be on the safe side.
   59 
   60 instance Hashable Element where
   61     hashWithSalt salt = hashWithSalt salt . compress

   63 -- | Add two G2 group elements
   64 add :: Element -> Element -> Element
   65 add = coerce (BlstBindings.blsAddOrDouble @BlstBindings.Curve2)
   66 {-# INLINE add #-}
   67 
   68 -- | Negate a G2 group element
   69 neg :: Element -> Element
   70 neg = coerce (BlstBindings.blsNeg @BlstBindings.Curve2)
   71 {-# INLINE neg #-}

   73 scalarMul :: Integer -> Element -> Element -- Other way round from library function
   74 scalarMul = coerce $ flip (BlstBindings.blsMult @BlstBindings.Curve2)
   75 {-# INLINE scalarMul #-}

   77 {- | Compress a G2 element to a bytestring. This serialises a curve point to its x
   78  coordinate only, using an extra bit to determine which of two possible y
   79  coordinates the point has. The compressed bytestring is 96 bytes long. See
   80  https://github.com/supranational/blst#serialization-format
   81 -}
   82 compress :: Element -> ByteString
   83 compress = coerce (BlstBindings.blsCompress @BlstBindings.Curve2)
   84 {-# INLINE compress #-}
   85 
   86 {- | Uncompress a bytestring to get a G2 point.  This will fail if any of the
   87    following are true:
   88      * The bytestring is not exactly 96 bytes long
   89      * The most significant three bits are used incorrectly
   90      * The bytestring encodes a field element which is not the
   91        x coordinate of a point on the E2 curve
   92      * The bytestring does represent a point on the E2 curve, but the
   93        point is not in the G2 subgroup
   94 -}
   95 uncompress :: ByteString -> Either BlstBindings.BLSTError Element
   96 uncompress = coerce (BlstBindings.blsUncompress @BlstBindings.Curve2)
   97 {-# INLINE uncompress #-}

   99 -- Take an arbitrary bytestring and a Domain Separation Tag and hash them to a
  100 -- get point in G2.  See Note [Hashing and Domain Separation Tags].
  101 hashToGroup :: ByteString -> ByteString -> Either BLS12_381_Error Element
  102 hashToGroup msg dst =
  103     if Data.ByteString.length dst > 255
  104     then Left HashToCurveDstTooBig
  105     else Right . Element $ BlstBindings.blsHash @BlstBindings.Curve2 msg (Just dst) Nothing

  107 -- | The zero element of G2.  This cannot be flat-serialised and is provided
  108 -- only for off-chain testing.
  109 offchain_zero :: Element
  110 offchain_zero = coerce (BlstBindings.Internal.blsZero @BlstBindings.Curve2)
  111 
  112 -- | The zero element of G2 compressed into a bytestring.  This is provided for
  113 -- convenience in PlutusTx and is not exported as a builtin.
  114 compressed_zero :: ByteString
  115 compressed_zero = compress $ coerce (BlstBindings.Internal.blsZero @BlstBindings.Curve2)
  116 
  117 -- | The standard generator of G2 compressed into a bytestring.  This is
  118 -- provided for convenience in PlutusTx and is not exported as a builtin.
  119 compressed_generator :: ByteString
  120 compressed_generator = compress $ coerce (BlstBindings.Internal.blsGenerator @BlstBindings.Curve2)

  132 -- | Multi-scalar multiplication of G2 points.
  133 multiScalarMul :: [Integer] -> [Element] -> Element
  134 multiScalarMul = coerce (\s p -> BlstBindings.blsMSM @BlstBindings.Curve2 (zip s p))
