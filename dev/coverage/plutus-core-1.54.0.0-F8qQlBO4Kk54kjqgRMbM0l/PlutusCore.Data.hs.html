   33 -- Attempting to make this strict made code slower by 2%,
   34 -- see https://github.com/IntersectMBO/plutus/pull/4622
   35 -- | A generic "data" type.
   36 --
   37 -- The main constructor 'Constr' represents a datatype value in sum-of-products
   38 -- form: @Constr i args@ represents a use of the @i@th constructor along with its arguments.
   39 --
   40 -- The other constructors are various primitives.
   41 data Data =
   42       Constr Integer [Data]
   43     | Map [(Data, Data)]
   44     | List [Data]
   45     | I Integer
   46     | B BS.ByteString
   47     deriving stock (Show, Read, Eq, Ord, Generic, Data.Data.Data)
   48     deriving anyclass (Hashable, NFData, NoThunks)
   49 
   50 instance Pretty Data where
   51     pretty = \case
   52         Constr _ ds -> angles (sep (punctuate comma (fmap pretty ds)))
   53         Map entries ->
   54           braces (sep (punctuate comma (fmap (\(k, v) -> pretty k <> ":" <+> pretty v) entries)))
   55         List ds     -> brackets (sep (punctuate comma (fmap pretty ds)))
   56         I i         -> pretty i
   57         B b         ->
   58             -- Base64 encode the ByteString since it may contain arbitrary bytes
   59             pretty (Text.decodeLatin1 (Base64.encode b))

   61 {- Note [Encoding via Term]
   62 We want to write a custom encoder/decoder for Data (i.e. not use the Generic version), but actually
   63 doing this is a pain. So instead we go via the CBOR 'Term' representation, which lets us process a
   64 more structured representation, which is a lot easier.
   65 -}

   67 instance Serialise Data where
   68     -- See Note [Encoding via Term]
   69     encode = encodeData
   70     decode = decodeData

   72 {- Note [CBOR alternative tags]
   73 We've proposed to add additional tags to the CBOR standard to cover (essentially) sum types.
   74 This is exactly what we need to encode the 'Constr' constructor of 'Data' in an unambiguous way.
   75 
   76 The tags aren't *quite* accepted yet, but they're clearly going to accept so we might as well
   77 start using them.
   78 
   79 The scheme is:
   80 - Alternatives 0-6 -> tags 121-127, followed by the arguments in a list
   81 - Alternatives 7-127 -> tags 1280-1400, followed by the arguments in a list
   82 - Any alternatives, including those that don't fit in the above -> tag 102 followed by a list
   83 containing an unsigned integer for the actual alternative, and then the arguments in a (nested!)
   84 list.
   85 -}
   86 
   87 {- Note [The 64-byte limit]
   88 We impose a 64-byte *on-the-wire* limit on the leaves of a serialized 'Data'. This prevents people
   89 from inserting Mickey Mouse entire.
   90 
   91 The simplest way of doing this is to check during deserialization that we never deserialize
   92 something that uses more than 64-bytes, and this is largely what we do. Then it's the user's problem
   93 to not produce something too big.
   94 
   95 But this is quite inconvenient, so see Note [Evading the 64-byte limit] for how we get around this.
   96 -}
   97 
   98 {- Note [Evading the 64-byte limit]
   99 Implementing Note [The 64-byte limit] naively would be quite annoying:
  100 - Users would be responsible for not creating Data values with leaves that were too big.
  101 - If a script *required* such a thing (e.g. a counter that somehow got above 64 bytes), then the
  102 user is totally stuck: the script demands something they cannot represent.
  103 
  104 This is unpleasant and introduces limits. Probably limits that nobody will hit, but it's nicer to
  105 just not have them.
  106 And it turns out that we can evade the problem with some clever encoding.
  107 
  108 The fundamental trick is that an *indefinite-length* CBOR bytestring is just as obfuscated as a list
  109 of bytestrings, since it consists of a list of definite-length chunks, and each definite-length
  110 chunk must be *tagged* (at least with the size).
  111 So we get a sequence like:
  112 
  113    <list start>
  114    <chunk length metadata>
  115    <chunk>
  116    <chunk length metadata>
  117    ...
  118    <list end>
  119 
  120 The chunk length metadata has a prescribed format, such that it's difficult to manipulate it so that
  121 it matches your "desired" data.
  122 So this effectively breaks up the bytestring in much the same way as a list of <64 byte bytestrings.
  123 
  124 So that solves the problem for bytestrings on the encoding side:
  125 - if they are <=64 bytes, we can just encode them as a normal bytestring
  126 - if they are >64 bytes, we encode them as indefinite-length bytestrings with 64-byte chunks
  127 
  128 On the decoding side, we need to check when we decode that we never decode a definite-length
  129 bytestring of >64 bytes. That covers our two cases:
  130 - Short definite-length bytestrings are fine
  131 - Long indefinite-length bytestrings are just made of short definite-length bytestings.
  132 
  133 Unfortunately this all means that we have to write our own encoders/decoders so we can produce
  134 chunks of the right size and check the sizes when we decode, but that's okay. Users need to do the
  135 same thing: anyone encoding `Data` with their own encoders who doesn't split up big bytestrings in
  136 this way will get failures when we decode them.
  137 
  138 For integers, we have two cases. Small integers (<=64bits) can be encoded normally. Big integers are
  139 already encoded *with a byte string*. The spec allows this to be an indefinite-length bytestring
  140 (although cborg doesn't like it), so we can reuse our trick.
  141 Again, we need to write some manual encoders/decoders.
  142 -}

  144 -- | Turn Data into a CBOR Term.
  145 encodeData :: Data -> Encoding
  146 encodeData = \case
  147     -- See Note [CBOR alternative tags]
  148     Constr i ds | 0 <= i && i < 7   -> CBOR.encodeTag (fromIntegral (121 + i)) <> encode ds
  149     Constr i ds | 7 <= i && i < 128 -> CBOR.encodeTag (fromIntegral (1280 + (i - 7))) <> encode ds
  150     Constr i ds | otherwise         ->
  151         let tagEncoding =
  152               if fromIntegral (minBound @Word64) <= i && i <= fromIntegral (maxBound @Word64)
  153               then CBOR.encodeWord64 (fromIntegral i)
  154               -- This is a "correct"-ish encoding of the tag, but it will *not* deserialise, since
  155               -- we insist on a 'Word64' when we deserialise.
  156               -- So this is really a "soft" failure, without using 'error' or something.
  157               else CBOR.encodeInteger i
  158         in CBOR.encodeTag 102 <> CBOR.encodeListLen 2 <> tagEncoding <> encode ds
  159     Map es ->
  160       CBOR.encodeMapLen (fromIntegral $ length es) <>
  161         mconcat [ encode t <> encode t' | (t, t') <-es ]
  162     List ds                         -> encode ds
  163     I i                             -> encodeInteger i
  164     B b                             -> encodeBs b
  165 
  166 -- Logic for choosing encoding borrowed from Codec.CBOR.Write
  167 -- | Given an integer, create a 'CBOR.Term' that encodes it, following our size restrictions.
  168 encodeInteger :: Integer -> Encoding
  169 -- If it fits in a Word64, then it's less than 64 bytes for sure, and we can just send it off
  170 -- as a normal integer for cborg to deal with
  171 encodeInteger i | i >= 0 , i <= fromIntegral (maxBound :: Word64) = CBOR.encodeInteger i
  172                 | i <  0 , i >= -1 - fromIntegral (maxBound :: Word64) = CBOR.encodeInteger i
  173 -- Otherwise, it would be encoded as a bignum anyway, so we manually do the bignum
  174 -- encoding with a bytestring inside, and since we use bsToTerm, that bytestring will
  175 -- get chunked up if it's too big.
  176 -- See Note [Evading the 64-byte limit]
  177 encodeInteger i | i >= 0 = CBOR.encodeTag 2 <> encodeBs (integerToBytes i)
  178 encodeInteger i | otherwise = CBOR.encodeTag 3 <> encodeBs (integerToBytes (-1 -i))
  179 
  180 -- Taken exactly from Codec.CBOR.Write
  181 integerToBytes :: Integer -> BS.ByteString
  182 integerToBytes n0
  183   | n0 == 0   = BS.pack [0]
  184   | otherwise = BS.pack (reverse (go n0))
  185   where
  186     go n | n == 0    = []
  187          | otherwise = narrow n : go (n `shiftR` 8)
  188 
  189     narrow :: Integer -> Word8
  190     narrow = fromIntegral
  191 
  192 -- | Given an bytestring, create a 'CBOR.Term' that encodes it, following our size restrictions.
  193 encodeBs :: BS.ByteString -> Encoding
  194 encodeBs b | BS.length b <= 64 = CBOR.encodeBytes b
  195 -- It's a bit tricky to get cborg to emit an indefinite-length bytestring with chunks that we
  196 -- control, so we encode it manually
  197 -- See Note [Evading the 64-byte limit]
  198 encodeBs b = CBOR.encodeBytesIndef <> foldMap encode (to64ByteChunks b) <> CBOR.encodeBreak
  199 
  200 -- | Turns a 'BS.ByteString' into a list of <=64 byte chunks.
  201 to64ByteChunks :: BS.ByteString -> [BS.ByteString]
  202 to64ByteChunks b | BS.length b > 64 =
  203                    let (chunk, rest) = BS.splitAt 64 b
  204                    in chunk:to64ByteChunks rest
  205 to64ByteChunks b = [b]

  207 {- Note [Definite and indefinite forms of CBOR]
  208 CBOR is annoying and you can have both definite (with a fixed length) and indefinite lists, maps,
  209 etc.
  210 
  211 So we have to be careful to handle both cases when decoding. When encoding we mostly don't make
  212 the indefinite kinds, but see Note [Evading the 64-byte limit] for some cases where we do.
  213 -}

  215 -- | Turn a CBOR Term into Data if possible.
  216 decodeData :: Decoder s Data
  217 decodeData = CBOR.peekTokenType >>= \case
  218   -- These integers are at most 64 *bits*, so certainly less than 64 *bytes*
  219   CBOR.TypeUInt         -> I <$> CBOR.decodeInteger
  220   CBOR.TypeUInt64       -> I <$> CBOR.decodeInteger
  221   CBOR.TypeNInt         -> I <$> CBOR.decodeInteger
  222   CBOR.TypeNInt64       -> I <$> CBOR.decodeInteger
  223   -- See Note [The 64-byte limit]
  224   CBOR.TypeInteger      -> I <$> decodeBoundedBigInteger
  225 
  226   -- See Note [The 64-byte limit]
  227   CBOR.TypeBytes        -> B <$> decodeBoundedBytes
  228   CBOR.TypeBytesIndef   -> B . BSL.toStrict <$> decodeBoundedBytesIndef
  229 
  230   CBOR.TypeListLen      -> decodeList
  231   CBOR.TypeListLen64    -> decodeList
  232   CBOR.TypeListLenIndef -> decodeList
  233 
  234   CBOR.TypeMapLen       -> decodeMap
  235   CBOR.TypeMapLen64     -> decodeMap
  236   CBOR.TypeMapLenIndef  -> decodeMap
  237 
  238   CBOR.TypeTag          -> decodeConstr
  239   CBOR.TypeTag64        -> decodeConstr
  240 
  241   t                     -> fail ("Unrecognized value of type " ++ show t)
  242 
  243 decodeBoundedBigInteger :: Decoder s Integer
  244 decodeBoundedBigInteger = do
  245     tag <- CBOR.decodeTag
  246     -- Bignums contain a bytestring as the payload
  247     bs <- CBOR.peekTokenType >>= \case
  248         CBOR.TypeBytes      -> decodeBoundedBytes
  249         CBOR.TypeBytesIndef -> BSL.toStrict <$> decodeBoundedBytesIndef
  250         t                   -> fail ("Bignum must contain a byte string, got: " ++ show t)
  251     -- Depending on the tag, the bytestring is either a positive or negative integer
  252     case tag of
  253         2 -> pure $ CBOR.uintegerFromBytes bs
  254         3 -> pure $ CBOR.nintegerFromBytes bs
  255         t -> fail ("Bignum tag must be one of 2 or 3, got: " ++ show t)
  256 
  257 -- Adapted from Codec.CBOR.Read
  258 decodeBoundedBytesIndef :: Decoder s BSL.ByteString
  259 decodeBoundedBytesIndef = CBOR.decodeBytesIndef >> decodeBoundedBytesIndefLen []
  260 
  261 -- Adapted from Codec.CBOR.Read, to call the size-checking bytestring decoder
  262 decodeBoundedBytesIndefLen :: [BS.ByteString] -> Decoder s BSL.ByteString
  263 decodeBoundedBytesIndefLen acc = do
  264     stop <- CBOR.decodeBreakOr
  265     if stop then return $! BSL.fromChunks (reverse acc)
  266             else do !bs <- decodeBoundedBytes
  267                     decodeBoundedBytesIndefLen (bs : acc)
  268 
  269 decodeBoundedBytes :: Decoder s BS.ByteString
  270 decodeBoundedBytes =  do
  271   b <- CBOR.decodeBytes
  272   -- See Note [The 64-byte limit]
  273   unless (BS.length b <= 64) $ fail "ByteString exceeds 64 bytes"
  274   pure b
  275 
  276 decodeList :: Decoder s Data
  277 decodeList = List <$> decodeListOf decodeData
  278 
  279 decodeListOf :: Decoder s x -> Decoder s [x]
  280 decodeListOf decoder = CBOR.decodeListLenOrIndef >>= \case
  281   Nothing -> decodeSequenceLenIndef (flip (:)) [] reverse   decoder
  282   Just n  -> decodeSequenceLenN     (flip (:)) [] reverse n decoder
  283 
  284 decodeMap :: Decoder s Data
  285 decodeMap = CBOR.decodeMapLenOrIndef >>= \case
  286   Nothing -> Map <$> decodeSequenceLenIndef (flip (:)) [] reverse   decodePair
  287   Just n  -> Map <$> decodeSequenceLenN     (flip (:)) [] reverse n decodePair
  288   where
  289   decodePair = (,) <$> decodeData <*> decodeData
  290 
  291 -- See Note [CBOR alternative tags] for the encoding scheme.
  292 decodeConstr :: Decoder s Data
  293 decodeConstr = CBOR.decodeTag64 >>= \case
  294   102 -> decodeConstrExtended
  295   t | 121 <= t && t < 128 ->
  296          Constr (fromIntegral t - 121) <$> decodeListOf decodeData
  297   t | 1280 <= t && t < 1401 ->
  298          Constr ((fromIntegral t - 1280) + 7) <$> decodeListOf decodeData
  299   t -> fail ("Unrecognized tag " ++ show t)
  300   where
  301   decodeConstrExtended = do
  302     len <- CBOR.decodeListLenOrIndef
  303     i <- CBOR.decodeWord64
  304     args <- decodeListOf decodeData
  305     case len of
  306       Nothing -> do
  307         done <- CBOR.decodeBreakOr
  308         unless done $ fail "Expected exactly two elements"
  309       Just n -> unless (n == 2) $ fail "Expected exactly two elements"
  310     pure $ Constr (fromIntegral i) args
