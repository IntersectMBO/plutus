   30 instance Profunctor InContextM where
   31     lmap
   32         :: forall config' config a.
   33            (config' -> config)
   34         -> InContextM config a
   35         -> InContextM config' a
   36     lmap = coerce (withReader :: (config' -> config) -> Reader config a -> Reader config' a)
   37     {-# INLINE lmap #-}
   38 
   39     rmap
   40         :: forall config a b.
   41            (a -> b)
   42         -> InContextM config a
   43         -> InContextM config b
   44     rmap = coerce (mapReader :: (a -> b) -> Reader config a -> Reader config b)
   45     {-# INLINE rmap #-}

   47 -- | For pretty-printing a value with a minimum amount of parens.
   48 type PrettyParens = PrettyBy RenderContext
   49 
   50 -- | An initial 'RenderContext'.
   51 -- An expression printed in this context gets enclosed in parens unless its outermost operator (if
   52 -- any) binds even stronger than function application.
   53 juxtRenderContext :: RenderContext
   54 juxtRenderContext = RenderContext ToTheRight juxtFixity

   56 instance PrettyDefaultBy config [(k, v)] => DefaultPrettyBy config (Map k v) where
   57     defaultPrettyBy config = prettyBy config . Map.toList
   58 deriving via PrettyCommon (Map k v)
   59     instance PrettyDefaultBy config (Map k v) => PrettyBy config (Map k v)

   61 instance PrettyDefaultBy config [a] => DefaultPrettyBy config (Set a) where
   62     defaultPrettyBy config = prettyBy config . Set.toList
   63 deriving via PrettyCommon (Set a)
   64     instance PrettyDefaultBy config (Set a) => PrettyBy config (Set a)
