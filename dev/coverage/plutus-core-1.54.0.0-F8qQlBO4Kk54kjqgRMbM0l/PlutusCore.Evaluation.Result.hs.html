   23 -- | The parameterized type of results various evaluation engines return.
   24 data EvaluationResult a
   25     = EvaluationSuccess !a
   26     | EvaluationFailure
   27     deriving stock (Show, Eq, Generic, Functor, Foldable, Traversable)
   28     deriving anyclass (NFData)
   29 
   30 instance MonadError () EvaluationResult where
   31     throwError () = EvaluationFailure
   32     {-# INLINE throwError #-}
   33 
   34     catchError EvaluationFailure f = f ()
   35     catchError x                 _ = x
   36     {-# INLINE catchError #-}
   37 
   38 instance Applicative EvaluationResult where
   39     pure = EvaluationSuccess
   40     {-# INLINE pure #-}
   41 
   42     EvaluationSuccess f <*> a = fmap f a
   43     EvaluationFailure   <*> _ = EvaluationFailure
   44     {-# INLINE (<*>) #-}
   45 
   46     EvaluationSuccess _ *> b = b
   47     EvaluationFailure   *> _ = EvaluationFailure
   48     {-# INLINE (*>) #-}
   49 
   50 instance Monad EvaluationResult where
   51     EvaluationSuccess x >>= f = f x
   52     EvaluationFailure   >>= _ = EvaluationFailure
   53     {-# INLINE (>>=) #-}
   54 
   55     (>>) = (*>)
   56     {-# INLINE (>>) #-}
   57 
   58 instance Alternative EvaluationResult where
   59     empty = EvaluationFailure
   60     {-# INLINE empty #-}
   61 
   62     a@EvaluationSuccess{} <|> _ = a
   63     EvaluationFailure     <|> b = b
   64     {-# INLINE (<|>) #-}
   65 
   66 instance MonadFail EvaluationResult where
   67     fail _ = EvaluationFailure
   68     {-# INLINE fail #-}
   69 
   70 instance PrettyBy config a => PrettyBy config (EvaluationResult a) where
   71     prettyBy config (EvaluationSuccess x) = prettyBy config x
   72     prettyBy _      EvaluationFailure     = "Failure"
   73 
   74 instance PrettyClassic a => Pretty (EvaluationResult a) where
   75     pretty = prettyClassic
   76 
   77 -- | Check whether an 'EvaluationResult' is an 'EvaluationSuccess'.
   78 isEvaluationSuccess :: EvaluationResult a -> Bool
   79 isEvaluationSuccess = not . null
   80 
   81 -- | Check whether an 'EvaluationResult' is an 'EvaluationFailure'.
   82 isEvaluationFailure :: EvaluationResult a -> Bool
   83 isEvaluationFailure = null
