   25 {- Note [Tree]
   26 Here we encode the following:
   27 
   28     mutual
   29       data Tree (A : Set) : Set where
   30         node : A -> Forest A -> Tree A
   31 
   32       data Forest (A : Set) : Set where
   33         nil  : Forest A
   34         cons : Tree A -> Forest A -> Forest A
   35 
   36     mutual
   37       mapTree : ∀ A B -> (A -> B) -> Tree A -> Tree B
   38       mapTree A B f (node x forest) = node (f x) (mapForest A B f forest)
   39 
   40       mapForest : ∀ A B -> (A -> B) -> Forest A -> Forest B
   41       mapForest A B f  nil               = nil
   42       mapForest A B f (cons rose forest) = cons (mapTree A B f rose) (mapForest A B f forest)
   43 
   44 using this representation:
   45 
   46     {-# OPTIONS_GHC --type-in-type #-}
   47 
   48     {-# NO_POSITIVITY_CHECK #-}
   49     data IFix {A : Set} (F : (A -> Set) -> A -> Set) (x : A) : Set where
   50       wrap : F (IFix F) x -> IFix F x
   51 
   52     Fix₂ : ∀ {A B} -> ((A -> B -> Set) -> A -> B -> Set) -> A -> B -> Set
   53     Fix₂ F x y = IFix (λ B P -> P (F λ x y -> B λ R -> R x y)) (λ R -> R x y)
   54 
   55     treeTag : Set -> Set -> Set
   56     treeTag T F = T
   57 
   58     forestTag : Set -> Set -> Set
   59     forestTag T F = F
   60 
   61     AsTree : (Set -> (Set -> Set -> Set) -> Set) -> Set -> Set
   62     AsTree D A = D A treeTag
   63 
   64     AsForest : (Set -> (Set -> Set -> Set) -> Set) -> Set -> Set
   65     AsForest D A = D A forestTag
   66 
   67     TreeForest : Set -> (Set -> Set -> Set) -> Set
   68     TreeForest =
   69       Fix₂ λ Rec A tag ->
   70       ∀ R ->
   71         tag ((A -> AsForest Rec A -> R) -> R) (R -> (AsTree Rec A -> AsForest Rec A -> R) -> R)
   72 
   73     Tree : Set -> Set
   74     Tree = AsTree TreeForest
   75 
   76     Forest : Set -> Set
   77     Forest = AsForest TreeForest
   78 -}

   94 treeTag :: Type TyName uni ()
   95 treeTag = runQuote $ do
   96     t <- freshTyName "t"
   97     f <- freshTyName "f"
   98     return
   99         . TyLam () t star
  100         . TyLam () f star
  101         $ TyVar () t

  103 forestTag :: Type TyName uni ()
  104 forestTag = runQuote $ do
  105     t <- freshTyName "t"
  106     f <- freshTyName "f"
  107     return
  108         . TyLam () t star
  109         . TyLam () f star
  110         $ TyVar () f

  112 asTree :: Type TyName uni ()
  113 asTree = runQuote $ do
  114     d <- freshTyName "d"
  115     a <- freshTyName "a"
  116     return
  117         . TyLam () d (star ~~> (star ~~> star ~~> star) ~~> star)
  118         . TyLam () a star
  119         $ mkIterTyAppNoAnn (TyVar () d)
  120             [ TyVar () a
  121             , treeTag
  122             ]

  124 asForest :: Type TyName uni ()
  125 asForest = runQuote $ do
  126     d <- freshTyName "d"
  127     a <- freshTyName "a"
  128     return
  129         . TyLam () d (star ~~> (star ~~> star ~~> star) ~~> star)
  130         . TyLam () a star
  131         $ mkIterTyAppNoAnn (TyVar () d)
  132             [ TyVar () a
  133             , forestTag

  136 treeForestData :: RecursiveType uni fun ()
  137 treeForestData = runQuote $ do
  138     treeForest <- freshTyName "treeForest"
  139     a          <- freshTyName "a"
  140     r          <- freshTyName "r"
  141     tag        <- freshTyName "tag"
  142     let vA = TyVar () a
  143         vR = TyVar () r
  144         recSpine = [TyVar () treeForest, vA]
  145     let tree   = mkIterTyAppNoAnn asTree   recSpine
  146         forest = mkIterTyAppNoAnn asForest recSpine
  147         body
  148             = TyForall () r (Type ())
  149             $ mkIterTyAppNoAnn (TyVar () tag)
  150                 [ (vA ~~> forest ~~> vR) ~~> vR
  151                 , vR ~~> (tree ~~> forest ~~> vR) ~~> vR
  152                 ]
  153     makeRecursiveType () treeForest
  154         [TyVarDecl () a star, TyVarDecl () tag $ star ~~> star ~~> star]
  155         body
  156 
  157 treeData :: RecursiveType uni fun ()
  158 treeData = runQuote $ do
  159     let RecursiveType treeForest wrapTreeForest = treeForestData
  160         tree = TyApp () asTree treeForest
  161     return $ RecursiveType tree (\[a] -> wrapTreeForest [a, treeTag])
  162 
  163 forestData :: RecursiveType uni fun ()
  164 forestData = runQuote $ do
  165     let RecursiveType treeForest wrapTreeForest = treeForestData
  166         forest = TyApp () asForest treeForest
  167     return $ RecursiveType forest (\[a] -> wrapTreeForest [a, forestTag])

  169 -- |
  170 --
  171 -- > /\(a :: *) -> \(x : a) (fr : forest a) ->
  172 -- >     wrapTree [a] /\(r :: *) -> \(f : a -> forest a -> r) -> f x fr
  173 treeNode :: HasUniApply uni => Term TyName Name uni fun ()
  174 treeNode = runQuote $ normalizeTypesIn =<< do
  175     let RecursiveType _      wrapTree = treeData
  176         RecursiveType forest _        = forestData
  177     a  <- freshTyName "a"
  178     r  <- freshTyName "r"
  179     x  <- freshName "x"
  180     fr <- freshName "fr"
  181     f  <- freshName "f"
  182     let vA = TyVar () a
  183         vR = TyVar () r
  184     Normalized forestA <- normalizeType $ TyApp () forest vA
  185     return
  186         . TyAbs () a (Type ())
  187         . LamAbs () x vA
  188         . LamAbs () fr forestA
  189         . wrapTree [vA]
  190         . TyAbs () r (Type ())
  191         . LamAbs () f (mkIterTyFun () [vA, forestA] vR)
  192         $ mkIterAppNoAnn (Var () f)
  193             [ Var () x
  194             , Var () fr
  195             ]
