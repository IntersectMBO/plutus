   10 module UntypedPlutusCore.Core.Instance.Pretty.Readable () where

   19 -- | Split an iterated 'LamAbs' (if any) into a list of variables that it binds and its body.
   20 viewLamAbs :: Term name uni fun ann -> Maybe ([name], Term name uni fun ann)
   21 viewLamAbs term0@LamAbs{} = Just $ go term0
   22   where
   23     go (LamAbs _ name body) = first (name :) $ go body
   24     go term                 = ([], term)
   25 viewLamAbs _ = Nothing
   26 
   27 -- | Split an iterated 'Apply' (if any) into the head of the application and the spine.
   28 viewApp ::
   29   Term name uni fun ann ->
   30   Maybe (Term name uni fun ann, [Term name uni fun ann])
   31 viewApp term0 = go term0 [] where
   32     go (Apply _ fun arg) args = go fun $ arg : args
   33     go _                 []   = Nothing
   34     go fun               args = Just (fun, args)
   35 
   36 instance
   37   (PrettyReadableBy configName name, PrettyUni uni, Pretty fun, Show configName) =>
   38   PrettyBy (PrettyConfigReadable configName) (Term name uni fun a)
   39   where
   40   prettyBy = inContextM $ \case
   41     Constant _ val -> lmap (ConstConfig . _pcrRenderContext) $ prettyM val
   42     Builtin _ bi -> unitDocM $ pretty bi
   43     Var _ name -> prettyM name
   44     (viewLamAbs -> Just (args, body)) -> iterLamAbsPrettyM args body
   45     LamAbs{} -> error "Panic: 'LamAbs' is not covered by 'viewLamAbs'"
   46     (viewApp -> Just (fun, args)) -> iterAppPrettyM fun args
   47     Apply{} -> error "Panic: 'Apply' is not covered by 'viewApp'"
   48     Delay _ term -> iterAppDocM $ \_ prettyArg -> "delay" :| [prettyArg term]
   49     Force _ term -> iterAppDocM $ \_ prettyArg -> "force" :| [prettyArg term]
   50     Error _ -> unitDocM "error"
   51     -- Always rendering the tag on the same line for more compact output, it's just a tiny integer
   52     -- anyway.
   53     Constr _ i es -> iterAppDocM $ \_ prettyArg ->
   54       ("constr" <+> prettyArg i) :| [prettyArg es]
   55     Case _ arg cs -> iterAppDocM $ \_ prettyArg -> "case" :| [prettyArg arg, prettyArg (toList cs)]
   56 
   57 instance
   58   (PrettyReadableBy configName (Term name uni fun a)) =>
   59   PrettyBy (PrettyConfigReadable configName) (Program name uni fun a)
   60   where
   61   prettyBy = inContextM $ \(Program _ version term) ->
   62     iterAppDocM $ \_ prettyArg -> "program" :| [pretty version, prettyArg term]
