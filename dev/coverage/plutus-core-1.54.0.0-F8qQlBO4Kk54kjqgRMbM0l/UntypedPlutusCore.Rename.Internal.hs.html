   21 type MonadRename m = (MonadQuote m, MonadReader (Renaming TermUnique) m)
   22 
   23 -- | Rename a 'Term' in the 'RenameM' monad.
   24 renameTermM
   25     :: (MonadRename m, HasUniques (Term name uni fun ann))
   26     => Term name uni fun ann -> m (Term name uni fun ann)
   27 renameTermM (LamAbs ann name body)  =
   28      withFreshenedName name $ \nameFr -> LamAbs ann nameFr <$> renameTermM body
   29 renameTermM (Apply ann fun arg)        = Apply ann <$> renameTermM fun <*> renameTermM arg
   30 renameTermM err@Error{}                = pure err
   31 renameTermM (Var ann name)             = Var ann <$> renameNameM name
   32 renameTermM (Delay ann term)           = Delay ann <$> renameTermM term
   33 renameTermM (Force ann term)           = Force ann <$> renameTermM term
   34 renameTermM (Constr ann i es)          = Constr ann i <$> traverse renameTermM es
   35 renameTermM (Case ann arg cs)          = Case ann <$> renameTermM arg <*> traverse renameTermM cs
   36 renameTermM con@Constant{}             = pure con
   37 renameTermM bi@Builtin{}               = pure bi
   38 
   39 -- | Rename a 'Program' in the 'RenameM' monad.
   40 renameProgramM
   41     :: (MonadRename m, HasUniques (Program name uni fun ann))
   42     => Program name uni fun ann -> m (Program name uni fun ann)
   43 renameProgramM (Program ann ver term) = Program ann ver <$> renameTermM term
