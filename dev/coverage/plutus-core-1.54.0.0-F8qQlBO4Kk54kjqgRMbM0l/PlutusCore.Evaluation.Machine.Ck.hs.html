   60 data CkValue uni fun =
   61     VCon (Some (ValueOf uni))
   62   | VTyAbs TyName (Kind ()) (Term TyName Name uni fun ())
   63   | VLamAbs Name (Type TyName uni ()) (Term TyName Name uni fun ())
   64   | VIWrap (Type TyName uni ()) (Type TyName uni ()) (CkValue uni fun)
   65   | VBuiltin (Term TyName Name uni fun ()) (BuiltinRuntime (CkValue uni fun))
   66   | VConstr (Type TyName uni ()) Word64 [CkValue uni fun]
   67 
   68 deriving stock instance (GShow uni, Everywhere uni Show, Show fun, Closed uni)
   69     => Show (CkValue uni fun)
   70 
   71 ckValueToTerm :: CkValue uni fun -> Term TyName Name uni fun ()
   72 ckValueToTerm = \case
   73     VCon val             -> Constant () val
   74     VTyAbs  tn k body    -> TyAbs  () tn k body
   75     VLamAbs name ty body -> LamAbs () name ty body
   76     VIWrap  ty1 ty2 val  -> IWrap  () ty1 ty2 $ ckValueToTerm val
   77     VBuiltin term _      -> term
   78     VConstr ty i es      -> Constr () ty i (fmap ckValueToTerm es)
   79 
   80 data CkEnv uni fun s = CkEnv
   81     { ckEnvRuntime    :: BuiltinsRuntime fun (CkValue uni fun)
   82     , ckCaserBuiltin  :: CaserBuiltin uni
   83     -- 'Nothing' means no logging. 'DList' is due to the fact that we need efficient append
   84     -- as we store logs as "latest go last".
   85     , ckEnvMayEmitRef :: Maybe (STRef s (DList Text))
   86     }

  101 type CkM uni fun s =
  102     ReaderT (CkEnv uni fun s)
  103         (ExceptT (CkEvaluationException uni fun)

  120 -- The 'DList' is just be consistent with the CEK machine (see Note [DList-based emitting]).
  121 emitCkM :: DList Text -> CkM uni fun s ()
  122 emitCkM logs = do
  123     mayLogsRef <- asks ckEnvMayEmitRef
  124     case mayLogsRef of
  125         Nothing      -> pure ()
  126         Just logsRef -> lift . lift $ modifySTRef logsRef (`DList.append` logs)

  136 data Frame uni fun
  137     = FrameAwaitArg (CkValue uni fun)                       -- ^ @[V _]@
  138     | FrameAwaitFunTerm (Term TyName Name uni fun ())       -- ^ @[_ N]@
  139     | FrameAwaitFunValue (CkValue uni fun)                  -- ^ @[_ V]@
  140     | FrameTyInstArg (Type TyName uni ())                   -- ^ @{_ A}@
  141     | FrameUnwrap                                           -- ^ @(unwrap _)@
  142     | FrameIWrap (Type TyName uni ()) (Type TyName uni ())  -- ^ @(iwrap A B _)@
  143     | FrameConstr (Type TyName uni ()) Word64 [Term TyName Name uni fun ()] [CkValue uni fun]
  144     | FrameCase [Term TyName Name uni fun ()]
  145 
  146 deriving stock instance (GShow uni, Closed uni, uni `Everywhere` Show, Show fun) =>
  147     Show (Frame uni fun)

  155 runCkM
  156     :: BuiltinsRuntime fun (CkValue uni fun)
  157     -> CaserBuiltin uni
  158     -> Bool
  159     -> (forall s. CkM uni fun s a)
  160     -> (Either (CkEvaluationException uni fun) a, [Text])
  161 runCkM runtime caser emitting a = runST $ do
  162     mayLogsRef <- if emitting then Just <$> newSTRef DList.empty else pure Nothing
  163     errOrRes <- runExceptT . runReaderT a $ CkEnv runtime caser mayLogsRef
  164     logs <- case mayLogsRef of
  165         Nothing      -> pure []
  166         Just logsRef -> DList.toList <$> readSTRef logsRef
  167     pure (errOrRes, logs)

  169 -- FIXME: make sure that the specification is up to date and that this matches.
  170 -- Tracked by https://github.com/IntersectMBO/plutus-private/issues/1552.
  171 -- | The computing part of the CK machine. Rules are as follows:
  172 --
  173 -- > s ▷ {M A}      ↦ s , {_ A}        ▷ M
  174 -- > s ▷ [M N]      ↦ s , [_ N]        ▷ M
  175 -- > s ▷ wrap α A M ↦ s , (wrap α S _) ▷ M
  176 -- > s ▷ unwrap M   ↦ s , (unwrap _)   ▷ M
  177 -- > s ▷ abs α K M  ↦ s ◁ abs α K M
  178 -- > s ▷ lam x A M  ↦ s ◁ lam x A M
  179 -- > s ▷ builtin bn ↦ s ◁ builtin (Builtin () bn) (runtimeOf bn)
  180 -- > s ▷ con cn     ↦ s ◁ con cn
  181 -- > s ▻ constr I T0 .. Tn ↦ s , (constr I _ T1 Tn) ▻ T0
  182 -- > s ▻ case S C0 ... Cn ↦ s , (case _ C0 ... Cn) ▻ S
  183 -- > s ▷ error A    ↦ ◆
  184 (|>)
  185     :: Context uni fun -> Term TyName Name uni fun () -> CkM uni fun s (Term TyName Name uni fun ())
  186 stack |> TyInst  _ fun ty        = FrameTyInstArg ty  : stack |> fun
  187 stack |> Apply   _ fun arg       = FrameAwaitFunTerm arg  : stack |> fun
  188 stack |> IWrap   _ pat arg term  = FrameIWrap pat arg : stack |> term
  189 stack |> Unwrap  _ term          = FrameUnwrap        : stack |> term
  190 stack |> TyAbs   _ tn k term     = stack <| VTyAbs tn k term
  191 stack |> LamAbs  _ name ty body  = stack <| VLamAbs name ty body
  192 stack |> Builtin _ bn            = do
  193     runtime <- lookupBuiltin bn . ckEnvRuntime <$> ask
  194     stack <| VBuiltin (Builtin () bn) runtime
  195 stack |> Constant _ val          = stack <| VCon val
  196 stack |> Constr _ ty i es               = case es of
  197     []     -> stack <| VConstr ty i []
  198     t : ts -> FrameConstr ty i ts [] : stack |> t
  199 stack |> Case _ _ arg cs         = FrameCase cs : stack |> arg
  200 _     |> err@Error{}             =
  201     throwErrorWithCause (OperationalError CkEvaluationFailure) $ void err
  202 _     |> var@Var{}               =
  203     throwErrorWithCause (StructuralError OpenTermEvaluatedMachineError) var

  205 -- FIXME: make sure that the specification is up to date and that this matches.
  206 -- Tracked by https://github.com/IntersectMBO/plutus-private/issues/1552.
  207 -- | The returning part of the CK machine. Rules are as follows:
  208 --
  209 -- > s , {_ A}           ◁ abs α K M  ↦ s         ▷ {A/α}M
  210 -- > s , [_ N]           ◁ V          ↦ s , [V _] ▷ N
  211 -- > s , [_ V1 ... Vm]   ◁ (lam x A M) ↦ s , [_ V2 ... Vm] ▷ [V1/x]M
  212 -- > s , [(lam x A M) _] ◁ V          ↦ s         ▷ [V/x]M
  213 -- > s , {_ A}           ◁ F          ↦ s ◁ {F A}  -- Partially instantiated builtin application.
  214 -- > s , [F _]           ◁ V          ↦ s ◁ [F V]  -- Partially saturated builtin application.
  215 -- > s , [F _]           ◁ V          ↦ s ◁ W      -- Fully saturated builtin application, [F V] ~> W.
  216 -- > s , (wrap α S _)    ◁ V          ↦ s ◁ wrap α S V
  217 -- > s , (unwrap _)      ◁ wrap α A V ↦ s ◁ V
  218 -- > s , (constr I V0 ... Vj-1 _ Tj+1 ... Tn) ◅ Vj ↦ s , (constr i V0 ... Vj _ Tj+2... Tn) ▻ Tj+1
  219 -- > s , (case _ C0 ... CN) ◅ (constr i V1 .. Vm) ↦ s , [_ V1 ... Vm] ▻ Ci
  220 (<|)
  221     :: Context uni fun -> CkValue uni fun -> CkM uni fun s (Term TyName Name uni fun ())
  222 []                         <| val     = pure $ ckValueToTerm val
  223 FrameTyInstArg ty  : stack <| fun     = instantiateEvaluate stack ty fun
  224 FrameAwaitFunTerm arg  : stack <| fun = FrameAwaitArg fun : stack |> arg
  225 FrameAwaitArg fun  : stack <| arg     = applyEvaluate stack fun arg
  226 FrameAwaitFunValue arg : stack <| fun = applyEvaluate stack fun arg
  227 FrameIWrap pat arg : stack <| value   = stack <| VIWrap pat arg value
  228 FrameUnwrap        : stack <| wrapped = case wrapped of
  229     VIWrap _ _ term -> stack <| term
  230     _               ->
  231         throwErrorWithCause (StructuralError NonWrapUnwrappedMachineError) $ ckValueToTerm wrapped
  232 FrameConstr ty i todo done : stack <| e =
  233     let done' = e:done
  234     in case todo of
  235         t : ts -> FrameConstr ty i ts done' : stack |> t
  236         []     -> stack <| VConstr ty i (reverse done')
  237 FrameCase cs : stack <| e = case e of
  238     VConstr _ i args -> case cs ^? wix i of
  239         Just t  -> go (reverse args) stack |> t
  240           where
  241             go [] s         = s
  242             go (arg:rest) s = go rest (FrameAwaitFunValue arg : s)
  243         Nothing ->
  244             throwErrorWithCause (StructuralError $ MissingCaseBranchMachineError i) $ ckValueToTerm e
  245     VCon val -> do
  246         caser <- asks ckCaserBuiltin
  247         case unCaserBuiltin caser val $ Vector.fromList cs of
  248             Left err  ->
  249                 throwErrorWithCause (OperationalError $ CkCaseBuiltinError err) $ ckValueToTerm e
  250             Right (HeadOnly fX) -> stack |> fX
  251             Right (HeadSpine f xs) -> transferConstantSpine xs stack |> f
  252     _ -> throwErrorWithCause (StructuralError NonConstrScrutinizedMachineError) $ ckValueToTerm e
  253 
  254 transferConstantSpine :: Spine (Some (ValueOf uni)) -> Context uni fun -> Context uni fun
  255 transferConstantSpine args ctx = foldr ((:) . FrameAwaitFunValue . VCon) ctx args

  257 -- | Take a possibly partial builtin application and
  258 --
  259 -- - either create a 'CkValue' by evaluating the application if it's saturated (emitting logs, if
  260 --    any, along the way), potentially failing evaluation
  261 -- - or create a partial builtin application otherwise
  262 --
  263 -- and proceed with the returning phase of the CK machine.
  264 evalBuiltinApp
  265     :: Context uni fun
  266     -> Term TyName Name uni fun ()
  267     -> BuiltinRuntime (CkValue uni fun)
  268     -> CkM uni fun s (Term TyName Name uni fun ())
  269 evalBuiltinApp stack term runtime = case runtime of
  270     BuiltinCostedResult _ getFXs -> case getFXs of
  271         BuiltinSuccess y              -> stack <| y
  272         BuiltinSuccessWithLogs logs y -> emitCkM logs *> (stack <| y)
  273         BuiltinFailure logs err       -> emitCkM logs *> throwBuiltinErrorWithCause term err
  274     _ -> stack <| VBuiltin term runtime

  276 -- | Instantiate a term with a type and proceed.
  277 -- In case of 'TyAbs' just ignore the type. Otherwise check if the term is builtin application
  278 -- expecting a type argument, in which case either calculate the builtin application or stick a
  279 -- 'TyInst' on top of its 'Term' representation depending on whether the application is saturated or
  280 -- not. In any other case, fail.
  281 instantiateEvaluate
  282     :: Context uni fun
  283     -> Type TyName uni ()
  284     -> CkValue uni fun
  285     -> CkM uni fun s (Term TyName Name uni fun ())
  286 instantiateEvaluate stack ty (VTyAbs tn _k body) =
  287      -- No kind check - too expensive at run time.
  288     stack |> termSubstClosedType tn ty body
  289 instantiateEvaluate stack ty (VBuiltin term runtime) = do
  290     let term' = TyInst () term ty
  291     case runtime of
  292         -- We allow a type argument to appear last in the type of a built-in function,
  293         -- otherwise we could just assemble a 'VBuiltin' without trying to evaluate the
  294         -- application.
  295         BuiltinExpectForce runtime' -> evalBuiltinApp stack term' runtime'
  296         _ -> throwErrorWithCause (StructuralError BuiltinTermArgumentExpectedMachineError) term'
  297 instantiateEvaluate _ _ val =
  298     throwErrorWithCause (StructuralError NonPolymorphicInstantiationMachineError) $ ckValueToTerm val
  299 
  300 -- | Apply a function to an argument and proceed.
  301 -- If the function is a lambda, then perform substitution and proceed.
  302 -- If the function is a builtin application then check that it's expecting a term argument,
  303 -- and either calculate the builtin application or stick a 'Apply' on top of its 'Term'
  304 -- representation depending on whether the application is saturated or not.
  305 applyEvaluate
  306     :: Context uni fun
  307     -> CkValue uni fun
  308     -> CkValue uni fun
  309     -> CkM uni fun s (Term TyName Name uni fun ())
  310 applyEvaluate stack (VLamAbs name _ body) arg =
  311     stack |> termSubstClosedTerm name (ckValueToTerm arg) body
  312 applyEvaluate stack (VBuiltin term runtime) arg = do
  313     let argTerm = ckValueToTerm arg
  314         term' = Apply () term argTerm
  315     case runtime of
  316         -- It's only possible to apply a builtin application if the builtin expects a term
  317         -- argument next.
  318         BuiltinExpectArgument f -> do
  319             evalBuiltinApp stack term' $ f arg
  320         _ ->
  321             throwErrorWithCause (StructuralError UnexpectedBuiltinTermArgumentMachineError) term'
  322 applyEvaluate _ val _ =
  323     throwErrorWithCause (StructuralError NonFunctionalApplicationMachineError) $ ckValueToTerm val

  325 runCk
  326     :: BuiltinsRuntime fun (CkValue uni fun)
  327     -> CaserBuiltin uni
  328     -> Bool
  329     -> Term TyName Name uni fun ()
  330     -> (Either (CkEvaluationException uni fun) (Term TyName Name uni fun ()), [Text])
  331 runCk runtime caser emitting term = runCkM runtime caser emitting $ [] |> term

  333 -- | Evaluate a term using the CK machine with logging enabled.
  334 evaluateCk

  336     -> CaserBuiltin uni
  337     -> Term TyName Name uni fun ()
  338     -> (Either (CkEvaluationException uni fun) (Term TyName Name uni fun ()), [Text])
  339 evaluateCk runtime caser = runCk runtime caser True

  341 -- | Evaluate a term using the CK machine with logging disabled.
  342 evaluateCkNoEmit
  343     :: BuiltinsRuntime fun (CkValue uni fun)
  344     -> CaserBuiltin uni
  345     -> Term TyName Name uni fun ()
  346     -> Either (CkEvaluationException uni fun) (Term TyName Name uni fun ())
