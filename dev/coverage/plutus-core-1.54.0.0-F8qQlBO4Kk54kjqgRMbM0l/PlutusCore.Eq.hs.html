   28 {- Note [Modulo alpha]
   29 The implemented algorithm of checking equality modulo alpha works as follows
   30 (taking types as an example):
   31 
   32 - traverse both the types simultaneously
   33 - if the outermost constructors differ, return 'False'
   34 - otherwise if the constructors are binders, then record that the two possibly distinct bound
   35     variables map to each other
   36 - otherwise if the constructors are variables, look them up in the current scope
   37     * if both are in scope, then those are bound variables, so check that they map to each other
   38         (i.e. are introduced by twin binders)
   39     * if both are not in scope, then those are free variables, so check if their uniques are equal
   40     * if one is in scope and the other one is not, then return 'False'
   41 - otherwise check equality of non-recursive constituents and recurse for recursive occurrences
   42 -}
   43 
   44 {- Note [Scope tracking]
   45 Scopes (term level vs type level) are resolved automatically in a type-driven way much like we do
   46 in the renamer. This allows not to think about scopes when implementing the actual equality checks
   47 and makes it impossible to confuse the scopes and e.g. insert a type-level name into a term-level
   48 scope. Also allows to define a function that records bindings and a function that checks equality
   49 of two names in a general manner for both the scopes.
   50 -}
   51 
   52 {- Note [Side tracking]
   53 The simplest way to track that twin variables map to each other is to have two contexts:
   54 
   55 - left-hand side variables and what they map to
   56 - right-hand side variables and what they map to
   57 
   58 (we refer to a first argument as being on the left-hand side and to a second argument as being on
   59 the right-hand side)
   60 
   61 E.g. when checking equality of these two types:
   62 
   63     all (x_3 :: *) (x_3 :: *). x_3 -> x_3
   64     all (y_4 :: *) (z_5 :: *). y_4 -> z_5
   65 
   66 we first record that @x_3@ maps to @y_4@ and vice versa, then record that @x_3@ maps to @z_5@ and
   67 vice versa. This way when we later check equality of @x_3@ and @y_4@ we know that it doesn't hold,
   68 because even though @y_4@ maps to @x_3@, @x_3@ doesn't map to @y_4@, because that mapping was
   69 overwritten by the @x_3@-to-@z_5@ one.
   70 
   71 For storing the left-to-right and right-to-left mappings separately we use the 'Bilateral' data
   72 type. Given that we track not only sides, but also scopes, we instantiate 'Bilateral' with either
   73 type-level-only renamings (for checking equality of types) or scoped ones (for checking equality
   74 of terms and programs). This amounts to the following generic monad:
   75 
   76     RenameT (Bilateral ren) m
   77 
   78 i.e. regardless of what the underlying renaming is, it has to be bilateral.
   79 
   80 We zoom into the sides of a bilateral renaming using the 'LR' and 'RL' newtype wrappers using the
   81 same 'HasRenaming' machinery that we use for zooming into the scopes of a scoped renaming:
   82 
   83 - the 'LR' wrapper allows to retrieve a lens focusing on the left  renaming
   84 - the 'RL' wrapper allows to retrieve a lens focusing on the right renaming
   85 
   86 I.e. you wrap a name into either 'LR' and 'RL' and depending on that you get focused on either
   87 left-to-right or right-to-left part of a 'Bilateral' renaming.
   88 
   89 So e.g.
   90 
   91     withRenamedName (LR name1) (LR name2)
   92 
   93 reads as "record that @name1@ maps to @name2@ from left to right"
   94 
   95 and
   96 
   97     lookupNameM $ RL name2
   98 
   99 reads as "look the right-to-left mapping of @name2@ up".
  100 
  101 I.e. we first resolve sides using explicit wrappers and then scope resolution happens automatically
  102 on the basis of existing type information (e.g. 'TyName' is a type-level name, hence we need the
  103 type-level renaming).
  104 -}

  106 -- See Note [Side tracking].
  107 -- | From left to right.
  108 newtype LR a = LR
  109     { unLR :: a
  110     } deriving stock (Generic)
  111 
  112 -- See Note [Side tracking].
  113 -- | From right to left.
  114 newtype RL a = RL
  115     { unRL :: a
  116     } deriving stock (Generic)
  117 
  118 -- See Note [Side tracking].
  119 -- | A left @a@ and a right @a@.
  120 data Bilateral a = Bilateral
  121     { _bilateralL :: a
  122     , _bilateralR :: a
  123     }
  124 
  125 makeLenses ''Bilateral
  126 
  127 instance Wrapped (LR a)
  128 instance Wrapped (RL a)
  129 instance HasUnique name unique => HasUnique (LR name) (LR unique)
  130 instance HasUnique name unique => HasUnique (RL name) (RL unique)
  131 
  132 instance Semigroup a => Semigroup (Bilateral a) where
  133     Bilateral l1 r1 <> Bilateral l2 r2 = Bilateral (l1 <> l2) (r1 <> r2)
  134 
  135 instance Monoid a => Monoid (Bilateral a) where
  136     mempty = Bilateral mempty mempty
  137 
  138 -- To rename from left to right is to update the left renaming.
  139 instance HasRenaming ren unique => HasRenaming (Bilateral ren) (LR unique) where
  140     renaming = bilateralL . renaming . coerced @(Renaming unique)
  141 
  142 -- To rename from right to left is to update the right renaming.
  143 instance HasRenaming ren unique => HasRenaming (Bilateral ren) (RL unique) where
  144     renaming = bilateralR . renaming . coerced @(Renaming unique)

  146 -- | The type of a runnable equality check. @Maybe ()@ is isomorphic to 'Bool' and we use it
  147 -- instead of 'Bool', because this unlocks the convenient and readable do-notation and allows for
  148 -- automatic short-circuiting, which would be tedious with @Rename (Bilateral ren) Bool@.
  149 type EqRename ren = RenameT (Bilateral ren) Maybe ()
  150 type ScopedEqRename = EqRename ScopedRenaming
  151 
  152 -- | Run an 'EqRename' computation.
  153 runEqRename :: Monoid ren => EqRename ren -> Bool
  154 runEqRename = isJust . runRenameT

  156 -- See Note [Modulo alpha].
  157 -- | Record that two names map to each other.
  158 withTwinBindings
  159     :: (HasRenaming ren unique, HasUnique name unique, Monad m)
  160     => name -> name -> RenameT (Bilateral ren) m c -> RenameT (Bilateral ren) m c
  161 withTwinBindings name1 name2 k =
  162     withRenamedName (LR name1) (LR name2) $
  163     withRenamedName (RL name2) (RL name1) k

  165 -- See Note [Modulo alpha].
  166 -- | Check equality of two names.
  167 eqNameM
  168     :: (HasRenaming ren unique, HasUnique name unique, Eq unique)
  169     => name -> name -> EqRename ren
  170 eqNameM name1 name2 = do
  171     mayUniq2' <- lookupNameM $ LR name1
  172     mayUniq1' <- lookupNameM $ RL name2
  173     let uniq1 = name1 ^. unique
  174         uniq2 = name2 ^. unique
  175     guard $ case (mayUniq1', mayUniq2') of
  176         (Nothing         , Nothing         ) -> uniq1 == uniq2
  177         (Just (RL uniq1'), Just (LR uniq2')) -> uniq1 == uniq1' && uniq2 == uniq2'
  178         (_               , _               ) -> False
  179 
  180 -- | Check equality of things having an 'Eq' instance.
  181 eqM :: Eq a => a -> a -> EqRename ren
  182 eqM x1 x2 = guard $ x1 == x2
