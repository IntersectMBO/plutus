    1 -- | Machinery defined in this module allows to export mulptiple Plutus Core definitions
    2 -- (types and terms) as a single value which enables convenient testing of various procedures
    3 -- (pretty-printing, type checking, etc): each time a function / data type is added to that value,
    4 -- none of the tests is required to be adapted, instead all the tests see the new definition
    5 -- automatically.
    6 
    7 module PlutusCore.FsTree
    8     ( FsTree (..)
    9     , FolderContents (..)
   10     , PlcEntity (..)
   11     , PlcFsTree
   12     , PlcFolderContents
   13     , treeFolderContents
   14     , plcTypeFile
   15     , plcTermFile
   16     , foldFsTree
   17     , foldPlcFsTree
   18     , foldPlcFolderContents
   19     ) where
   20 
   21 import PlutusCore.Core
   22 import PlutusCore.Name.Unique
   23 
   24 -- We use 'String's for names, because 'FilePath's are 'String's.
   25 -- | An 'FsTree' is either a file or a folder with a list of 'FsTree's inside.
   26 data FsTree a
   27     = FsFolder String (FolderContents a)
   28     | FsFile String a
   29 
   30 -- | The contents of a folder. A wrapper around @[FsTree a]@.
   31 -- Exists because of its 'Semigroup' instance which allows to concatenate two 'FolderContents's
   32 -- without placing them into the same folder immediately, so we can have various PLC "modules"
   33 -- (@stdlib@, @examples@, etc), define compound modules (e.g. @stdlib <> examples@) and run various
   34 -- tests (pretty-printing, type synthesis, etc) against simple and compound modules uniformly.
   35 newtype FolderContents a = FolderContents
   36     { unFolderContents :: [FsTree a]
   37     } deriving newtype (Semigroup, Monoid)
   38 
   39 -- | A 'PlcEntity' is either a 'Type' or a 'Term'.
   40 data PlcEntity uni fun
   41     = PlcType (Type TyName uni ())
   42     | PlcTerm (Term TyName Name uni fun ())
   43 
   44 type PlcFsTree         uni fun = FsTree         (PlcEntity uni fun)
   45 type PlcFolderContents uni fun = FolderContents (PlcEntity uni fun)
   46 
   47 -- | Construct an 'FsTree' out of the name of a folder and a list of 'FsTree's.
   48 treeFolderContents :: String -> [FsTree a] -> FsTree a
   49 treeFolderContents name = FsFolder name . FolderContents
   50 
   51 -- | Construct a single-file 'PlcFsTree' out of a type.
   52 plcTypeFile :: String -> Type TyName uni () -> PlcFsTree uni fun
   53 plcTypeFile name = FsFile name . PlcType
   54 
   55 -- | Construct a single-file 'PlcFsTree' out of a term.
   56 plcTermFile :: String -> Term TyName Name uni fun () -> PlcFsTree uni fun
   57 plcTermFile name = FsFile name . PlcTerm
   58 
   59 -- | Fold a 'FsTree'.
   60 foldFsTree
   61     :: (String -> [b] -> b)  -- ^ What to do on a folder.
   62     -> (String -> a -> b)    -- ^ What to do on a single file in a folder.
   63     -> FsTree a
   64     -> b
   65 foldFsTree onFolder onFile = go where
   66     go (FsFolder name (FolderContents trees)) = onFolder name $ map go trees
   67     go (FsFile name x)                        = onFile name x
   68 
   69 -- | Fold a 'PlcFsTree'.
   70 foldPlcFsTree
   71     :: (String -> [b] -> b)                          -- ^ What to do on a folder.
   72     -> (String -> Type TyName uni ()          -> b)  -- ^ What to do on a type.
   73     -> (String -> Term TyName Name uni fun () -> b)  -- ^ What to do on a term.
   74     -> PlcFsTree uni fun
   75     -> b
   76 foldPlcFsTree onFolder onType onTerm = foldFsTree onFolder onFile where
   77     onFile name (PlcType getTy)   = onType name getTy
   78     onFile name (PlcTerm getTerm) = onTerm name getTerm
   79 
   80 -- | Fold the contents of a PLC folder.
   81 foldPlcFolderContents
   82     :: (String -> [b] -> b)                          -- ^ What to do on a folder.
   83     -> (String -> Type TyName uni ()          -> b)  -- ^ What to do on a type.
   84     -> (String -> Term TyName Name uni fun () -> b)  -- ^ What to do on a term.
   85     -> PlcFolderContents uni fun
   86     -> [b]
   87 foldPlcFolderContents onFolder onType onTerm (FolderContents trees) =
   88     map (foldPlcFsTree onFolder onType onTerm) trees
