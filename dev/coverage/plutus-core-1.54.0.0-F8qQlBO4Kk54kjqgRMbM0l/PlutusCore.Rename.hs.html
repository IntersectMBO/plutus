   25 {- Note [Marking]
   26 We use functions from the @markNonFresh*@ family in order to ensure that bound variables never get
   27 renamed to free ones. This means types/terms/etc are processed twice: once by a @markNonFresh*@
   28 function and once for the actual renaming. We may consider later to do some kind of meta-circular
   29 programming trick in order to perform renaming in a single pass.
   30 -}
   31 
   32 -- | The class of things that can be renamed.
   33 -- I.e. things that are capable of satisfying the global uniqueness condition.
   34 class Rename a where
   35     -- | Rename 'Unique's so that they're globally unique.
   36     -- In case there are any free variables, they must be left untouched and bound variables
   37     -- must not get renamed to free ones.
   38     -- Must always assign new names to bound variables,
   39     -- so that @rename@ can be used for alpha-renaming as well.
   40     rename :: MonadQuote m => a -> m a
   41 
   42 -- | 'rename' a value and wrap the result in 'Renamed', so that it can be passed around and it's
   43 -- visible in the types that the thing inside satisfies global uniqueness.
   44 getRenamed :: (Rename a, MonadQuote m) => a -> m (Renamed a)
   45 getRenamed = fmap Renamed . rename
   46 
   47 -- | Wrap a value in 'Dupable'.
   48 dupable :: a -> Dupable a
   49 dupable = Dupable
   50 
   51 -- | Extract the value stored in a @Dupable a@ and rename it.
   52 liftDupable :: (MonadQuote m, Rename a) => Dupable a -> m a
   53 liftDupable = liftQuote . rename . unDupable

   55 instance HasUniques (Type tyname uni ann) => Rename (Type tyname uni ann) where
   56     -- See Note [Marking].
   57     rename = through markNonFreshType >=> runRenameT @TypeRenaming . renameTypeM
   58 
   59 instance HasUniques (Term tyname name uni fun ann) => Rename (Term tyname name uni fun ann) where
   60     -- See Note [Marking].
   61     rename = through markNonFreshTerm >=> runRenameT . renameTermM
   62 
   63 instance HasUniques (Program tyname name uni fun ann) =>
   64     Rename (Program tyname name uni fun ann) where
   65     -- See Note [Marking].
   66     rename = through markNonFreshProgram >=> runRenameT . renameProgramM
   67 
   68 instance Rename a => Rename (Normalized a) where
   69     rename = traverse rename
