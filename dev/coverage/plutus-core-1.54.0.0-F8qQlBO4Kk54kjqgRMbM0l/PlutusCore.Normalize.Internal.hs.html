   39 {- | Mapping from variables to what they stand for (each row represents a substitution).
   40 Needed for efficiency reasons, otherwise we could just use substitutions.
   41 -}
   42 type TypeVarEnv tyname uni ann = UniqueMap TypeUnique (Dupable (Normalized (Type tyname uni ann)))

   44 -- | The environments that type normalization runs in.
   45 newtype NormalizeTypeEnv tyname uni ann = NormalizeTypeEnv
   46   { _normalizeTypeEnvTypeVarEnv :: TypeVarEnv tyname uni ann
   47   }

   51 {- Note [NormalizeTypeT]
   52 Type normalization requires 'Quote' (because we need to be able to generate fresh names), but we
   53 do not put 'Quote' into 'NormalizeTypeT'. The reason for this is that it makes type signatures of
   54 various runners much nicer and also more generic. For example, we have
   55 
   56     runNormalizeTypeT :: MonadQuote m => NormalizeTypeT m tyname uni ann a -> m a
   57 
   58 If 'NormalizeTypeT' contained 'Quote', it would be
   59 
   60     runNormalizeTypeT :: NormalizeTypeT m tyname uni ann a -> QuoteT m a
   61 
   62 which hardcodes 'QuoteT' to be the outermost transformer.
   63 
   64 Type normalization can run in any @m@ (as long as it's a 'MonadQuote') as witnessed by
   65 the following type signature:
   66 
   67     normalizeTypeM
   68         :: (HasUnique (tyname ann) TypeUnique, MonadQuote m)
   69         => Type tyname uni ann -> NormalizeTypeT m tyname uni ann (Normalized (Type tyname uni ann))
   70 
   71 so it's natural to have runners that do not break this genericity.
   72 -}
   73 
   74 {- Note [Normalization API]
   75 Normalization is split in two parts:
   76 
   77 1. functions returning computations that perform reductions and run in defined in this module
   78    monad transformers (e.g. 'NormalizeTypeT')
   79 2. runners of those computations
   80 
   81 The reason for splitting the API is that this way the type-theoretic notion of normalization is
   82 separated from implementation-specific details. (This used to be more important when we had
   83 to deal with gas, and could maybe be changed now.)
   84 -}

   86 -- See Note [NormalizeTypeT].
   87 -- | The monad transformer that type normalization runs in.
   88 newtype NormalizeTypeT m tyname uni ann a = NormalizeTypeT
   89   { unNormalizeTypeT :: ReaderT (NormalizeTypeEnv tyname uni ann) m a
   90   }
   91   deriving newtype
   92     ( Functor
   93     , Applicative
   94     , Alternative
   95     , Monad
   96     , MonadPlus
   97     , MonadReader (NormalizeTypeEnv tyname uni ann)
   98     , MonadState s
   99     , MonadQuote
  100     )
  101 
  102 -- | The constraints that type normalization requires.
  103 type MonadNormalizeType uni m =
  104   ( MonadQuote m -- Type normalization must preserve global uniqueness.
  105   , HasUniApply uni -- See Note [Normalization of built-in types].
  106   )
  107 
  108 -- | Run a 'NormalizeTypeT' computation.
  109 runNormalizeTypeT :: NormalizeTypeT m tyname uni ann a -> m a
  110 runNormalizeTypeT = flip runReaderT (NormalizeTypeEnv mempty) . unNormalizeTypeT
  111 
  112 -- | Locally extend a 'TypeVarEnv' in a 'NormalizeTypeT' computation.
  113 withExtendedTypeVarEnv ::
  114   (HasUnique tyname TypeUnique, Monad m) =>
  115   tyname ->
  116   Normalized (Type tyname uni ann) ->
  117   NormalizeTypeT m tyname uni ann a ->
  118   NormalizeTypeT m tyname uni ann a
  119 withExtendedTypeVarEnv name =
  120   local . over normalizeTypeEnvTypeVarEnv . insertByName name . dupable
  121 
  122 -- | Look up a @tyname@ in a 'TypeVarEnv'.
  123 lookupTyNameM ::
  124   (HasUnique tyname TypeUnique, Monad m) =>
  125   tyname ->
  126   NormalizeTypeT m tyname uni ann (Maybe (Dupable (Normalized (Type tyname uni ann))))
  127 lookupTyNameM name = asks $ lookupName name . _normalizeTypeEnvTypeVarEnv

  129 {- Note [Normalization]
  130 Normalization works under the assumption that variables are globally unique.
  131 We use environments instead of substitutions as they're more efficient.
  132 
  133 Since all names are unique and there is no need to track scopes, type normalization has only two
  134 interesting cases: function application and a variable usage. In the function application case we
  135 normalize a function and its argument, add the normalized argument to the environment and continue
  136 normalization. In the variable case we look up the variable in the current environment: if it's not
  137 found, we leave the variable untouched. If the variable is found, then what this variable stands for
  138 was previously added to an environment (while handling the function application case), so we pick
  139 this value and rename all bound variables in it to preserve the global uniqueness condition. It is
  140 safe to do so, because picked values cannot contain uninstantiated variables as only normalized
  141 types are added to environments and normalization instantiates all variables presented in an
  142 environment.
  143 
  144 See also Note [Normalization of built-in types].
  145 -}
  146 
  147 {- Note [Normalization of built-in types]
  148 Instantiating a polymorphic built-in type amounts to applying it to some arguments. However,
  149 the notion of "applying" is ambiguous, it can mean one of these two things:
  150 
  151 1. lifting the built-in type to 'Type' and applying that via 'TyApp'
  152 2. applying the built-in type right inside the universe to get a monomorphized type tag
  153    (e.g. the default universe has 'DefaultUniApply' for that purpose)
  154 
  155 We need both of these things. The former allows us to assign types to polymorphic built-in functions
  156 (otherwise applying a built-in type to a type variable would be unrepresentable), the latter is
  157 used at runtime to juggle type tags so that we can avoid @unsafeCoerce@-ing, bring instances in
  158 scope via 'bring' etc -- for all of that we have to have fully monomorphized type tags at runtime.
  159 
  160 So in order for type checking to work we need to normalize polymorphic built-in types. For that
  161 we simply turn intra-universe applications into regular type applications during type normalization.
  162 
  163 We could go the other way around and "reduce" regular type applications into intra-universe ones,
  164 however that would be harder to implement, because collapsing a general 'Type' into a 'SomeTypeIn'
  165 is harder than expanding a 'SomeTypeIn' into a 'Type'. And it would also be impossible to do in the
  166 general case, 'cause you can't collapse an application of a built-in type to, say, a type variable
  167 into an intra-universe application as there are no type variables there. I guess we could "reduce"
  168 application of built-in types in some cases and not reduce them in others and still make the whole
  169 thing work, but that requires substantially more logic and is also a lot harder to get right.
  170 Hence we do the opposite, which is straightforward.
  171 -}

  175 {- | Normalize a built-in type by replacing each application inside the universe with regular
  176 type application.
  177 -}
  178 normalizeUni :: forall k (a :: k) uni tyname. HasUniApply uni => uni (Esc a) -> Type tyname uni ()
  179 normalizeUni uni =
  180   matchUniApply
  181     uni
  182     -- If @uni@ is not an intra-universe application, then we're done.
  183     (mkTyBuiltinOf () uni)
  184     -- If it is, then we turn that application into normal type application and recurse
  185     -- into both the function and its argument.
  186     (\uniF uniA -> TyApp () (normalizeUni uniF) $ normalizeUni uniA)

  190 -- | Normalize a 'Type' in the 'NormalizeTypeT' monad.
  191 normalizeTypeM ::
  192   (HasUnique tyname TypeUnique, MonadNormalizeType uni m) =>
  193   Type tyname uni ann ->
  194   NormalizeTypeT m tyname uni ann (Normalized (Type tyname uni ann))
  195 normalizeTypeM (TyForall ann name kind body) =
  196   TyForall ann name kind <<$>> normalizeTypeM body
  197 normalizeTypeM (TyIFix ann pat arg) =
  198   TyIFix ann <<$>> normalizeTypeM pat <<*>> normalizeTypeM arg
  199 normalizeTypeM (TyFun ann dom cod) =
  200   TyFun ann <<$>> normalizeTypeM dom <<*>> normalizeTypeM cod
  201 normalizeTypeM (TyLam ann name kind body) =
  202   TyLam ann name kind <<$>> normalizeTypeM body
  203 normalizeTypeM (TyApp ann fun arg) = do
  204   vFun <- normalizeTypeM fun
  205   vArg <- normalizeTypeM arg
  206   case unNormalized vFun of
  207     TyLam _ nArg _ body -> substNormalizeTypeM vArg nArg body
  208     _                   -> pure $ TyApp ann <$> vFun <*> vArg
  209 normalizeTypeM var@(TyVar _ name) = do
  210   mayTy <- lookupTyNameM name
  211   case mayTy of
  212     -- A variable is always normalized.
  213     Nothing -> pure $ Normalized var
  214     Just ty -> liftDupable ty
  215 normalizeTypeM (TyBuiltin ann (SomeTypeIn uni)) =
  216   pure . Normalized $ ann <$ normalizeUni uni
  217 normalizeTypeM (TySOP ann tyls) = do
  218   tyls' <- (traverse . traverse) (fmap unNormalized . normalizeTypeM) tyls
  219   pure $ Normalized $ TySOP ann tyls'

  228 -- See Note [Normalizing substitution].
  229 -- | Substitute a type for a variable in a type and normalize in the 'NormalizeTypeT' monad.
  230 substNormalizeTypeM ::
  231   (HasUnique tyname TypeUnique, MonadNormalizeType uni m) =>
  232   -- | @ty@
  233   Normalized (Type tyname uni ann) ->
  234   -- | @name@
  235   tyname ->
  236   -- | @body@
  237   Type tyname uni ann ->
  238   -- | @NORM ([ty / name] body)@
  239   NormalizeTypeT m tyname uni ann (Normalized (Type tyname uni ann))
  240 substNormalizeTypeM ty name = withExtendedTypeVarEnv name ty . normalizeTypeM
  241 
  242 -- | Normalize every 'Type' in a 'Term'.
  243 normalizeTypesInM ::
  244   (HasUnique tyname TypeUnique, MonadNormalizeType uni m) =>
  245   Term tyname name uni fun ann ->
  246   NormalizeTypeT m tyname uni ann (Term tyname name uni fun ann)
  247 normalizeTypesInM = transformMOf termSubterms normalizeChildTypes
  248   where
  249     normalizeChildTypes = termSubtypes (fmap unNormalized . normalizeTypeM)
