   17 -- | Apply a monomorphic function to both components of a pair.
   18 --
   19 -- > /\(a :: *) -> \(f : a -> a) (p : pair a a) ->
   20 -- >     comma {a} {a} (f (fst {a} {a} p)) (f (snd {a} {a} p))
   21 obothPair :: TermLike term TyName Name DefaultUni (Either DefaultFun ExtensionFun) => term ()
   22 obothPair = runQuote $ do
   23     a <- freshTyName "a"
   24     f <- freshName "f"
   25     p <- freshName "p"
   26     let atAA fun = mkIterInstNoAnn (builtin () fun) [TyVar () a, TyVar () a]
   27     return
   28         . tyAbs () a (Type ())
   29         . lamAbs () f (TyFun () (TyVar () a) $ TyVar () a)
   30         . lamAbs () p (mkIterTyAppNoAnn pair [TyVar () a, TyVar () a])
   31         $ mkIterAppNoAnn (atAA $ Right Comma)
   32             [ apply () (var () f) . apply () (atAA $ Left FstPair) $ var () p
   33             , apply () (var () f) . apply () (atAA $ Left SndPair) $ var () p
   34             ]
