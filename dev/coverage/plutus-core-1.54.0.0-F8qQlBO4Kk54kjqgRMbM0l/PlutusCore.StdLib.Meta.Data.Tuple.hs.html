   30 -- | A Plutus Core (Scott-encoded) tuple.
   31 data Tuple term uni ann =
   32     Tuple
   33         { _tupleElementTypes :: [Type TyName uni ann] -- ^ The types of elements of a tuple.
   34         , _tupleTerm         :: term ann              -- ^ A term representation of the tuple.
   35         }
   36 
   37 -- | Get the type of a 'Tuple'.
   38 --
   39 -- > getTupleType _ (Tuple [a1, ... , an] _) = all r. (a1 -> ... -> an -> r) -> r
   40 getTupleType :: MonadQuote m => ann -> Tuple term uni ann -> m (Type TyName uni ann)
   41 getTupleType ann (Tuple elTys _) = liftQuote $ do
   42     r <- freshTyName "r"
   43     let caseTy = mkIterTyFun ann elTys $ TyVar ann r
   44     pure . TyForall ann r (Type ann) . TyFun ann caseTy $ TyVar ann r
   45 
   46 -- | Convert a Haskell spine of 'Term's to a PLC 'Tuple'.
   47 --
   48 -- > getSpineToTuple _ [(a1, x1), ... , (an, xn)] =
   49 -- >     Tuple [a1, ... , an] (/\(r :: *) -> \(f :: a1 -> ... -> an -> r) -> f x1 ... xn)
   50 getSpineToTuple
   51     :: (TermLike term TyName Name uni fun, MonadQuote m)
   52     => ann -> [(Type TyName uni ann, term ann)] -> m (Tuple term uni ann)
   53 getSpineToTuple ann spine = liftQuote $ do
   54     r <- freshTyName "r"
   55     f <- freshName "f"
   56     let (as, xs) = unzip spine
   57         caseTy = mkIterTyFun ann as $ TyVar ann r
   58         y = mkIterApp (var ann f) ((ann,) <$> xs)
   59     pure . Tuple as . tyAbs ann r (Type ann) $ lamAbs ann f caseTy y

   61 -- | Get the type of the ith element of a 'Tuple' along with the element itself.
   62 --
   63 -- > tupleTypeTermAt _ i (Tuple [a0, ... , an] term) =
   64 -- >     (ai, term {ai} (\(x0 : a0) ... (xn : an) -> xi))
   65 tupleTypeTermAt
   66     :: (TermLike term TyName Name uni fun, MonadQuote m)
   67     => ann -> Int -> Tuple term uni ann -> m (Type TyName uni ann, term ann)
   68 tupleTypeTermAt ann ind (Tuple elTys term) = liftQuote $ do
   69     args <- ifor elTys $ \i ty -> do
   70         n <- freshName $ "arg_" <> showText i
   71         pure $ VarDecl ann n ty
   72     let selectedTy  = elTys !! ind
   73         selectedArg = mkVar $ args !! ind
   74         selector    = mkIterLamAbs args selectedArg
   75 
   76     pure
   77         ( selectedTy
   78         , apply ann (tyInst ann term selectedTy) selector
   79         )
   80 
   81 -- | Get the ith element of a 'Tuple'.
   82 tupleTermAt
   83     :: (TermLike term TyName Name uni fun, MonadQuote m)
   84     => ann -> Int -> Tuple term uni ann -> m (term ann)
   85 tupleTermAt ann ind tuple = snd <$> tupleTypeTermAt ann ind tuple
   86 
   87 -- | Get the ith element of a 'Tuple' as a 'TermDef'.
   88 tupleDefAt
   89     :: (TermLike term TyName Name uni fun, MonadQuote m)
   90     => ann
   91     -> Int
   92     -> Name
   93     -> Tuple term uni ann
   94     -> m (TermDef term TyName Name uni ann)
   95 tupleDefAt ann ind name tuple = uncurry (Def . VarDecl ann name) <$> tupleTypeTermAt ann ind tuple

   97 -- | Bind all elements of a 'Tuple' inside a 'Term'.
   98 --
   99 -- > bindTuple _ [x_1, ... , x_n] (Tuple [a1, ... , an] term) body =
  100 -- >     (\(tup : all r. (a_1 -> ... -> a_n -> r) -> r) ->
  101 -- >       let x_1 = _1 tup
  102 -- >           ...
  103 -- >           x_n = _n tup
  104 -- >         in body
  105 -- >     ) term
  106 bindTuple
  107     :: (TermLike term TyName Name uni fun, MonadQuote m)
  108     => ann -> [Name] -> Tuple term uni ann -> term ann -> m (term ann)
  109 bindTuple ann names (Tuple elTys term) body = liftQuote $ do
  110     tup <- freshName "tup"
  111     let tupVar = Tuple elTys $ var ann tup
  112     tupTy <- getTupleType ann tupVar
  113     tupDefs <- itraverse (\i name -> tupleDefAt ann i name tupVar) names
  114     pure $ apply ann (lamAbs ann tup tupTy $ foldr (termLet ann) body tupDefs) term

  116 -- | Given an arity @n@, create the n-ary product type.
  117 --
  118 -- @\(T_1 :: *) .. (T_n :: *) . all (R :: *) . (T_1 -> .. -> T_n -> R) -> R@
  119 prodN :: Int -> Type TyName uni ()
  120 prodN arity = runQuote $ do
  121     tyVars <- for [0..(arity-1)] $ \i -> do
  122         tn <- liftQuote $ freshTyName $ "t_" <> showText i
  123         pure $ TyVarDecl () tn $ Type ()
  124 
  125     resultType <- liftQuote $ freshTyName "r"
  126     let caseType = mkIterTyFun () (fmap (mkTyVar ()) tyVars) (TyVar () resultType)
  127     pure $
  128         -- \T_1 .. T_n
  129         mkIterTyLam tyVars $
  130         -- all R
  131         TyForall () resultType (Type ()) $
  132         -- (T_1 -> .. -> T_n -> r) -> r
  133         TyFun () caseType (TyVar () resultType)
  134 
  135 -- | Given an arity @n@, create the constructor for n-ary products.
  136 --
  137 -- @
  138 --     /\(T_1 :: *) .. (T_n :: *) .
  139 --         \(arg_1 : T_1) .. (arg_n : T_n) .
  140 --             /\(R :: *).
  141 --                 \(case : T_1 -> .. -> T_n -> R) -> case arg_1 .. arg_n
  142 -- @
  143 prodNConstructor :: TermLike term TyName Name uni fun => Int -> term ()
  144 prodNConstructor arity = runQuote $ do
  145     tyVars <- for [0..(arity-1)] $ \i -> do
  146         tn <- liftQuote $ freshTyName $ "t_" <> showText i
  147         pure $ TyVarDecl () tn $ Type ()
  148 
  149     resultType <- liftQuote $ freshTyName "r"
  150 
  151     args <- for [0..(arity -1)] $ \i -> do
  152         n <- liftQuote $ freshName $ "arg_" <> showText i
  153         pure $ VarDecl () n $ mkTyVar () $ tyVars !! i
  154 
  155     caseArg <- liftQuote $ freshName "case"
  156     let caseTy = mkIterTyFun () (fmap (mkTyVar ()) tyVars) (TyVar () resultType)
  157     pure $
  158         -- /\T_1 .. T_n
  159         mkIterTyAbs tyVars $
  160         -- \arg_1 .. arg_n
  161         mkIterLamAbs args $
  162         -- /\R
  163         tyAbs () resultType (Type ()) $
  164         -- \case
  165         lamAbs () caseArg caseTy $
  166         -- case arg_1 .. arg_n
  167         mkIterAppNoAnn (var () caseArg) $ fmap mkVar args
  168 
  169 -- | Given an arity @n@ and an index @i@, create a function for accessing the i'th component of a n-tuple.
  170 --
  171 -- @
  172 --     /\(T_1 :: *) .. (T_n :: *) .
  173 --         \(tuple : all (R :: *) . (T_1 -> .. -> T_n -> R) -> R)) .
  174 --             tuple {T_i} (\(arg_1 : T_1) .. (arg_n : T_n) . arg_i)
  175 -- @
  176 prodNAccessor :: TermLike term TyName Name uni fun => Int -> Int -> term ()
  177 prodNAccessor arity index = runQuote $ do
  178     tyVars <- for [0..(arity-1)] $ \i -> do
  179         tn <- liftQuote $ freshTyName $ "t_" <> showText i
  180         pure $ TyVarDecl () tn $ Type ()
  181 
  182     let tupleTy = mkIterTyAppNoAnn (prodN arity) (fmap (mkTyVar ()) tyVars)
  183         selectedTy = mkTyVar () $ tyVars !! index
  184 
  185     args <- for [0..(arity -1)] $ \i -> do
  186         n <- liftQuote $ freshName $ "arg_" <> showText i
  187         pure $ VarDecl () n $ mkTyVar () $ tyVars !! i
  188     let selectedArg = mkVar $ args !! index
  189 
  190     tupleArg <- liftQuote $ freshName "tuple"
  191     pure $
  192         -- /\T_1 .. T_n
  193         mkIterTyAbs tyVars $
  194         -- \tuple :: (tupleN T_1 .. T_n)
  195         lamAbs () tupleArg tupleTy $
  196         -- tuple {T_i}
  197         apply () (tyInst () (var () tupleArg) selectedTy) $
  198         -- \arg_1 .. arg_n . arg_i
  199         mkIterLamAbs args selectedArg
