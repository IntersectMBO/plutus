   68 {- Note [Why newtype FakeNamedDeBruijn]
   69 We use a newtype wrapper to optimize away the expensive re-traversing of the deserialized Term
   70 for adding fake names everywhere --- the CEK works on names, but the scripts on ledger
   71 don't have names for size reduction.
   72 
   73 Specifically the expensive pipeline:
   74 
   75 ```
   76 decode @(Term DeBruijn)
   77 >>> term-Map-Names(FakeNamedDeBruijn)
   78 >>> cekExecute
   79 ```
   80 
   81 is optimized to to the faster:
   82 
   83 ```
   84 decode @(Term FakeNamedDeBruijn)
   85 >>> coerce @(Term FakeNamedDeBruijn) @(Term NamedDeBruijn)
   86 >>> cekExecute
   87 ```
   88 
   89 To achieve this we make sure:
   90 - to use `coerce` since its 0-cost
   91 - not to GeneralizeNewtypeDeriving the`Flat FakeNamedDeBruijn` instance, but "derive via"
   92 the optimized `Flat DeBruijn` instance. This is ok, because `FND<->D` are
   93 isomorphic.
   94 -}

  201 {- Note [Levels and indices]
  202 The indices ('Index') that we actually store as our de Bruijn indices in the program
  203 are *relative* - that is, they say how many levels above the *current* level to look for
  204 the binder.
  205 
  206 However, when doing conversions it is easier to record the  *absolute* level of a variable,
  207 in our state, since that way we don't have to adjust our mapping when we go under a binder
  208 (whereas for relative indices we would need to increment them all by one, as the current
  209 level has increased).
  210 
  211 However, this means that we *do* need to do an adjustment when we store an index as a level
  212 or extract a level to use it as an index. The adjustment is fairly straightforward:
  213 - An index `i` points to a binder `i` levels above (smaller than) the current level, so the level
  214   of `i` is `current - i`.
  215 - A level `l` which is `i` levels above (smaller than) the current level has an index of `i`, so it
  216   is also calculated as `current - l`.
  217 
  218 We use a newtype to keep these separate, since getting it wrong will lead to annoying bugs.
  219 -}
  220 
  221 -- | An absolute level in the program.
  222 newtype Level = Level Integer deriving newtype (Eq, Ord, Num, Real, Enum, Integral)

  224 {- | During visiting the AST we hold a reader "state" of current level and a current
  225 scoping (levelMapping).
  226 Invariant-A: the current level is positive and greater than all levels in the levelMapping.
  227 Invariant-B: only positive levels are stored in the levelMapping.
  228 -}
  229 data LevelInfo = LevelInfo
  230   { currentLevel :: Level
  231   , levelMapping :: BM.Bimap Unique Level
  232   }
  233 
  234 -- | Declare a name with a unique, recording the mapping to a 'Level'.
  235 declareUnique :: (MonadReader LevelInfo m, HasUnique name unique) => name -> m a -> m a
  236 declareUnique n =
  237   local $ \(LevelInfo current ls) -> LevelInfo current $ BM.insert (n ^. theUnique) current ls
  238 
  239 {- | Declares a new binder by assigning a fresh unique to the *current level*.
  240 Maintains invariant-B of 'LevelInfo' (that only positive levels are stored),
  241 since current level is always positive (invariant-A).
  242 See Note [DeBruijn indices of Binders]
  243 -}
  244 declareBinder :: (MonadReader LevelInfo m, MonadQuote m) => m a -> m a
  245 declareBinder act = do
  246   newU <- freshUnique
  247   local (\(LevelInfo current ls) -> LevelInfo current $ BM.insert newU current ls) act
  248 
  249 {- | Enter a scope, incrementing the current 'Level' by one
  250 Maintains invariant-A (that the current level is positive).
  251 -}
  252 withScope :: (MonadReader LevelInfo m) => m a -> m a
  253 withScope = local $ \(LevelInfo current ls) -> LevelInfo (current + 1) ls

  269 {- | Get the 'Index' corresponding to a given 'Unique'.
  270 Uses supplied handler for free names (uniques).
  271 -}
  272 getIndex :: (MonadReader LevelInfo m) => Unique -> (Unique -> m Index) -> m Index
  273 getIndex u h = do
  274   LevelInfo current ls <- ask
  275   case BM.lookup u ls of
  276     Just foundlvl -> pure $ levelToIx current foundlvl
  277     -- This call should return an index greater than the current level,
  278     -- otherwise it will map unbound variables to bound variables.
  279     Nothing       -> h u
  280   where
  281     -- Compute the relative 'Index' of a absolute 'Level' relative to the current 'Level'.
  282     levelToIx :: Level -> Level -> Index
  283     levelToIx (Level current) (Level foundLvl) =
  284       -- Thanks to invariant-A, we can be sure that 'level >= foundLvl ', since foundLvl
  285       -- is in the levelMapping and thus the computation 'current-foundLvl' is '>=0' and
  286       -- its conversion to Natural will not lead to arithmetic underflow.
  287       fromIntegral $ current - foundLvl
  288 
  289 {- | Get the 'Unique' corresponding to a given 'Index'.
  290 Uses supplied handler for free debruijn indices.
  291 -}
  292 getUnique :: (MonadReader LevelInfo m) => Index -> (Index -> m Unique) -> m Unique
  293 getUnique ix h = do
  294   LevelInfo current ls <- ask
  295   case BM.lookupR (ixToLevel current ix) ls of
  296     -- Because of invariant-B, the levelMapping contains only positive (absolute) levels.
  297     Just u -> pure u
  298     -- This call should return a free/unbound unique,
  299     -- otherwise it will map unbound variables to bound variables.
  300     Nothing ->
  301       -- the lookup failed, meaning the index corresponds to a strictly-negative
  302       -- (absolute) level.
  303       h ix

  355 {- | A different implementation of a handler,  where "free" debruijn indices do not throw an error
  356 but are instead gracefully converted to fresh uniques.
  357 These generated uniques remain free; i.e.  if the original term was open, it will remain open
  358 after applying this handler.
  359 These generated free uniques are consistent across the open term (by using a state cache).
  360 -}
  361 freeIndexAsConsistentLevel
  362   :: (MonadReader LevelInfo m, MonadState (M.Map Level Unique) m, MonadQuote m)
  363   => Index
  364   -> m Unique
  365 freeIndexAsConsistentLevel ix = do
  366   cache <- get
  367   LevelInfo current _ <- ask
  368   -- the absolute level is strictly-negative
  369   let absoluteLevel = ixToLevel current ix
  370   case M.lookup absoluteLevel cache of
  371     Nothing -> do
  372       u <- freshUnique
  373       -- the cache contains only strictly-negative levels
  374       put (M.insert absoluteLevel u cache)
  375       pure u
  376     Just u -> pure u
  377 
  378 -- Compute the absolute 'Level' of a relative 'Index' relative to the current 'Level'.
  379 -- The index `ixAST` may be malformed or point to a free variable because it comes straight
  380 -- from the AST; in such a case, this function may return a negative level.
  381 ixToLevel :: Level -> Index -> Level
  382 ixToLevel (Level current) ixAST = Level $ current - fromIntegral ixAST
