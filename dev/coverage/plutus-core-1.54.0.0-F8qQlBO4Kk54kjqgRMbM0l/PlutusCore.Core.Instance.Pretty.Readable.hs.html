   23 -- | Split an iterated 'KindArrow' (if any) into the list of argument types and the resulting type.
   24 viewKindArrow :: Kind ann -> Maybe ([Kind ann], Kind ann)
   25 viewKindArrow kind0@KindArrow{} = Just $ go kind0 where
   26     go (KindArrow _ dom cod) = first (dom :) $ go cod
   27     go kind                  = ([], kind)
   28 viewKindArrow _ = Nothing
   29 
   30 -- | Split an iterated 'TyFun' (if any) into the list of argument types and the resulting type.
   31 viewTyFun :: Type tyname uni ann -> Maybe ([Type tyname uni ann], Type tyname uni ann)
   32 viewTyFun ty0@TyFun{} = Just $ go ty0 where
   33     go (TyFun _ dom cod) = first (dom :) $ go cod
   34     go ty                = ([], ty)
   35 viewTyFun _ = Nothing

   37 -- | Split an iterated 'TyForall' (if any) into a list of variables that it binds and its body.
   38 viewTyForall :: Type tyname uni ann -> Maybe ([TyVarDecl tyname ann], Type tyname uni ann)
   39 viewTyForall ty0@TyForall{} = Just $ go ty0 where
   40     go (TyForall ann name kind body) = first (TyVarDecl ann name kind :) $ go body
   41     go ty                            = ([], ty)

   51 -- | Split an iterated 'LamAbs' (if any) into a list of variables that it binds and its body.
   52 viewLamAbs
   53     :: Term tyname name uni fun ann
   54     -> Maybe ([VarDecl tyname name uni ann], Term tyname name uni fun ann)
   55 viewLamAbs term0@LamAbs{} = Just $ go term0 where
   56     go (LamAbs ann name ty body) = first (VarDecl ann name ty :) $ go body
   57     go term                      = ([], term)

   60 -- | Split an iterated 'TyAbs' (if any) into a list of variables that it binds and its body.
   61 viewTyAbs
   62     :: Term tyname name uni fun ann -> Maybe ([TyVarDecl tyname ann], Term tyname name uni fun ann)
   63 viewTyAbs term0@TyAbs{} = Just $ go term0 where
   64     go (TyAbs ann name kind body) = first (TyVarDecl ann name kind :) $ go body
   65     go term                       = ([], term)
   66 viewTyAbs _ = Nothing

   68 -- | Split an iterated 'TyApp' (if any) into the head of the application and the spine.
   69 viewTyApp
   70     :: Type tyname uni ann -> Maybe (Type tyname uni ann, [Type tyname uni ann])
   71 viewTyApp ty0 = go ty0 [] where
   72     go (TyApp _ fun arg) args = go fun $ arg : args
   73     go _                 []   = Nothing
   74     go fun               args = Just (fun, args)
   75 
   76 -- | Split an iterated 'Apply'/'TyInst' (if any) into the head of the application and the spine.
   77 viewApp
   78     :: Term tyname name uni fun ann
   79     -> Maybe
   80         ( Term tyname name uni fun ann
   81         , [Either (Type tyname uni ann) (Term tyname name uni fun ann)]
   82         )
   83 viewApp term0 = go term0 [] where
   84     go (Apply _ fun argTerm) args = go fun $ Right argTerm : args
   85     go (TyInst _ fun argTy)  args = go fun $ Left argTy : args
   86     go _                     []   = Nothing
   87     go fun                   args = Just (fun, args)

   89 instance PrettyReadableBy configName tyname =>
   90         PrettyBy (PrettyConfigReadable configName) (TyVarDecl tyname ann) where
   91   prettyBy = inContextM $ \case
   92       TyVarDecl _ x k -> do
   93           showKinds <- view $ prettyConfig . pcrShowKinds
   94           withPrettyAt ToTheRight botFixity $ \prettyBot -> do
   95               case showKinds of
   96                   ShowKindsYes -> encloseM binderFixity $ (prettyBot x <+> "::") <?> prettyBot k
   97                   ShowKindsNonType -> case k of
   98                       Type{} -> pure $ prettyBot x
   99                       _      -> encloseM binderFixity $ (prettyBot x <+> "::") <?> prettyBot k
  100                   ShowKindsNo -> pure $ prettyBot x

  102 instance
  103         ( PrettyReadableBy configName tyname
  104         , PrettyReadableBy configName name
  105         , PrettyUni uni
  106         ) => PrettyBy (PrettyConfigReadable configName) (VarDecl tyname name uni ann) where
  107   prettyBy = inContextM $ \case
  108       VarDecl _ x t -> do
  109           withPrettyAt ToTheRight botFixity $ \prettyBot -> do
  110               encloseM binderFixity $ (prettyBot x <+> ":") <?> prettyBot t

  112 instance PrettyBy (PrettyConfigReadable configName) (Kind a) where
  113     prettyBy = inContextM $ \case
  114         (viewKindArrow -> Just (args, res)) -> iterArrowPrettyM args res
  115         KindArrow {} -> error "Panic: 'KindArrow' is not covered by 'viewKindArrow'"
  116         Type{} -> "*"
  117 
  118 instance (PrettyReadableBy configName tyname, PrettyParens (SomeTypeIn uni)) =>
  119             PrettyBy (PrettyConfigReadable configName) (Type tyname uni a) where
  120     prettyBy = inContextM $ \case
  121         (viewTyApp -> Just (fun, args)) -> iterAppPrettyM fun args
  122         TyApp {} -> error "Panic: 'TyApp' is not covered by 'viewTyApp'"
  123         TyVar _ name -> prettyM name
  124         (viewTyFun -> Just (args, res)) -> iterArrowPrettyM args res
  125         TyFun {} -> error "Panic: 'TyFun' is not covered by 'viewTyFun'"
  126         TyIFix _ pat arg -> iterAppDocM $ \_ prettyArg -> "ifix" :| map prettyArg [pat, arg]
  127         (viewTyForall -> Just (args, body)) -> iterTyForallPrettyM args body
  128         TyForall {} -> error "Panic: 'TyForall' is not covered by 'viewTyForall'"
  129         TyBuiltin _ someUni -> lmap _pcrRenderContext $ prettyM someUni
  130         (viewTyLam -> Just (args, body)) -> iterLamAbsPrettyM args body
  131         TyLam {} -> error "Panic: 'TyLam' is not covered by 'viewTyLam'"
  132         TySOP _ tls -> iterAppDocM $ \_ prettyArg -> "sop" :| fmap prettyArg tls
  133 
  134 instance
  135         ( PrettyReadableBy configName tyname
  136         , PrettyReadableBy configName name
  137         , PrettyUni uni
  138         , Pretty fun
  139         ) => PrettyBy (PrettyConfigReadable configName) (Term tyname name uni fun a) where
  140     prettyBy = inContextM $ \case
  141         Constant _ con -> lmap (ConstConfig . _pcrRenderContext) $ prettyM con
  142         Builtin _ bi -> unitDocM $ pretty bi
  143         (viewApp -> Just (fun, args)) -> iterInterAppPrettyM fun args
  144         Apply {} -> error "Panic: 'Apply' is not covered by 'viewApp'"
  145         TyInst {} -> error "Panic: 'TyInst' is not covered by 'viewApp'"
  146         Var _ name -> prettyM name
  147         (viewTyAbs -> Just (args, body)) -> iterTyAbsPrettyM args body
  148         TyAbs {} -> error "Panic: 'TyAbs' is not covered by 'viewTyAbs'"
  149         (viewLamAbs -> Just (args, body)) -> iterLamAbsPrettyM args body
  150         LamAbs {} -> error "Panic: 'LamAbs' is not covered by 'viewLamAbs'"
  151         Unwrap _ term -> iterAppDocM $ \_ prettyArg -> "unwrap" :| [prettyArg term]
  152         IWrap _ pat arg term ->
  153             iterAppDocM $ \_ prettyArg ->
  154                 "iwrap" :| [prettyArg pat, prettyArg arg, prettyArg term]
  155         Error _ ty -> iterAppDocM $ \_ prettyArg -> "error" :| [prettyArg $ inBraces ty]
  156         Constr _ ty i es ->
  157             iterAppDocM $ \_ prettyArg -> "constr" :| [prettyArg ty, prettyArg i, prettyArg es]
  158         Case _ ty arg cs ->
  159             iterAppDocM $ \_ prettyArg -> "case" :| [prettyArg ty, prettyArg arg, prettyArg cs]

  161 instance PrettyReadableBy configName (Term tyname name uni fun a) =>
  162         PrettyBy (PrettyConfigReadable configName) (Program tyname name uni fun a) where
  163     prettyBy = inContextM $ \(Program _ version term) ->
  164         iterAppDocM $ \_ prettyArg -> "program" :| [pretty version, prettyArg term]
