   11 -- | A separate module for JSON instances, so that we can stick @-O0@ on it and avoid spending
   12 -- a lot of time optimizing loads of Core whose performance doesn't matter.
   13 module PlutusCore.Evaluation.Machine.CostingFun.JSON () where

   20 type ModelJSON prefix = CustomJSON '[FieldLabelModifier (StripPrefix prefix, CamelToSnake)]
   21 
   22 type ModelArgumentJSON prefix =
   23     CustomJSON
   24         -- Without TagSingleConstructors the format can change unexpectedly if
   25         -- you add/remove constructors because you don't get the tags if there's
   26         -- only one constructor but you do if there's more than one.
   27         '[ TagSingleConstructors
   28          , SumTaggedObject "type" "arguments"
   29          , ConstructorTagModifier (StripPrefix prefix, CamelToSnake)]
   30

   33 deriving via ModelJSON "costingFun" (CostingFun model)
