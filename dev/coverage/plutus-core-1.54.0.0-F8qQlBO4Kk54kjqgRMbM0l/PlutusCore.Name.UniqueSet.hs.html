    1 {- | A type for sets of things identified by 'Unique's, usually names.
    2  This approach is preferred when it is more efficient to compare the associated
    3  'Unique's instead of the underlying type.
    4 -}

    6 module PlutusCore.Name.UniqueSet (

   13   fromNames,

   23 import Control.Lens (Getting, view)
   24 import Control.Lens.Getter (views)
   25 import Data.Coerce (Coercible, coerce)
   26 import Data.IntSet qualified as IS
   27 import Data.IntSet.Lens qualified as IS

   29 import PlutusCore.Name.Unique (HasUnique (..), Unique (Unique))

   31 {- | A set containing 'Unique's. Since 'Unique' is equivalent to 'Int'
   32  (see "PlutusCore.Name.Unique"), we can use an 'IntSet' representation for this type.
   33 -}
   34 newtype UniqueSet unique = UniqueSet
   35   { unUniqueSet :: IS.IntSet
   36   }
   37   deriving stock (Show, Eq)
   38   deriving newtype (Semigroup, Monoid)
   39 
   40 -- | Insert a @unique@.
   41 insertByUnique ::
   42   (Coercible unique Unique) =>
   43   unique ->
   44   UniqueSet unique ->
   45   UniqueSet unique
   46 insertByUnique = coerce . IS.insert . coerce
   47 
   48 -- | Insert the @unique@ associated to the @name@.
   49 insertByName :: (HasUnique name unique) => name -> UniqueSet unique -> UniqueSet unique
   50 insertByName = insertByUnique . view unique
   51 
   52 -- | Create the singleton set of the @unique@ associated to the @name@.
   53 singletonName :: (HasUnique name unique) => name -> UniqueSet unique
   54 singletonName n = insertByName n mempty
   55 
   56 -- | Convert a 'Foldable' into a 'UniqueSet' using the given insertion function.
   57 fromFoldable ::
   58   (Foldable f) =>
   59   (i -> UniqueSet unique -> UniqueSet unique) ->
   60   f i ->
   61   UniqueSet unique
   62 fromFoldable ins = List.foldl' (flip ins) mempty
   63 
   64 -- | Convert a 'Foldable' with uniques into a 'UniqueSet'.
   65 fromUniques :: (Foldable f) => (Coercible Unique unique) => f unique -> UniqueSet unique
   66 fromUniques = fromFoldable insertByUnique
   67 
   68 -- | Convert a 'Foldable' with names into a 'UniqueSet'.
   69 fromNames :: (Foldable f) => (HasUnique name unique) => f name -> UniqueSet unique
   70 fromNames = fromFoldable insertByName
   71 
   72 -- | Is the @unique@ a member of the set?
   73 memberByUnique :: (Coercible unique Unique) => unique -> UniqueSet unique -> Bool
   74 memberByUnique uniq = IS.member (coerce uniq) . unUniqueSet
   75 
   76 -- | Is the @name@ associated to the @unique@ a member of the set?
   77 memberByName :: (HasUnique name unique) => name -> UniqueSet unique -> Bool
   78 memberByName = memberByUnique . view unique
   79 
   80 {- | The negation of 'memberByName', useful for converting to operator form,
   81  e.g. @name `notMemberByName` set@.
   82 -}
   83 notMemberByName :: (HasUnique name unique) => name -> UniqueSet unique -> Bool
   84 notMemberByName n = not . memberByName n
   85 
   86 -- | The difference of two 'UniqueSet's.
   87 (\\) :: UniqueSet unique -> UniqueSet unique -> UniqueSet unique
   88 (\\) (UniqueSet s1) (UniqueSet s2) = UniqueSet $ s1 IS.\\ s2
   89 
   90 -- | The union of two 'UniqueSet's.
   91 union :: UniqueSet unique -> UniqueSet unique -> UniqueSet unique
   92 union (UniqueSet s1) (UniqueSet s2) = UniqueSet $ s1 `IS.union` s2
   93 
   94 -- | Build a set of @unique@s from the 'Getting'.
   95 setOfByUnique ::
   96   (Coercible unique Unique) =>
   97   Getting (UniqueSet unique) s unique ->
   98   s ->
   99   UniqueSet unique
  100 setOfByUnique g = UniqueSet <$> IS.setOf (coerce g)
  101 
  102 -- | Build a set of @unique@s associated to the names in the 'Getting'.
  103 setOfByName ::
  104   (HasUnique name unique) =>
  105   Getting (UniqueSet unique) s name ->
  106   s ->
  107   UniqueSet unique
  108 setOfByName l = views l singletonName
