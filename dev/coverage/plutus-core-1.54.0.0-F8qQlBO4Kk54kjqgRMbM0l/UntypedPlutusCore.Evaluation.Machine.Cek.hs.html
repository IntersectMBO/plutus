   71 {-| Evaluate a term using the CEK machine with logging enabled and keep track of costing.
   72 A wrapper around the internal runCek to debruijn input and undebruijn output.
   73 *THIS FUNCTION IS PARTIAL if the input term contains free variables*
   74 -}
   75 runCek
   76     :: ThrowableBuiltins uni fun
   77     => MachineParameters CekMachineCosts fun (CekValue uni fun ann)
   78     -> ExBudgetMode cost uni fun
   79     -> EmitterMode uni fun
   80     -> Term Name uni fun ann
   81     -> CekReport cost Name uni fun
   82 runCek = Common.runCek runCekDeBruijn
   83 
   84 -- | Evaluate a term using the CEK machine with logging disabled and keep track of costing.
   85 -- *THIS FUNCTION IS PARTIAL if the input term contains free variables*
   86 runCekNoEmit
   87     :: ThrowableBuiltins uni fun
   88     => MachineParameters CekMachineCosts fun (CekValue uni fun ann)
   89     -> ExBudgetMode cost uni fun
   90     -> Term Name uni fun ann
   91     -> (Either (CekEvaluationException Name uni fun) (Term Name uni fun ()), cost)
   92 runCekNoEmit = Common.runCekNoEmit runCekDeBruijn
