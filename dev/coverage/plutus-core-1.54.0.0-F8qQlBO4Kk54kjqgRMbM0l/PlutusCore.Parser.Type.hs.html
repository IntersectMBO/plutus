   29 -- | A PLC @Type@ to be parsed. ATM the parser only works
   30 -- for types in the @DefaultUni@ with @DefaultFun@.
   31 type PType = Type TyName DefaultUni SrcSpan
   32 
   33 varType :: Parser PType
   34 varType = withSpan $ \sp ->
   35     TyVar sp <$> tyName
   36 
   37 funType :: Parser PType
   38 funType = withSpan $ \sp ->
   39     inParens $ TyFun sp <$> (symbol "fun" *> pType) <*> pType
   40 
   41 allType :: Parser PType
   42 allType = withSpan $ \sp ->
   43     inParens $ TyForall sp <$> (symbol "all" *> trailingWhitespace tyName) <*> kind <*> pType
   44 
   45 lamType :: Parser PType
   46 lamType = withSpan $ \sp ->
   47     inParens $ TyLam sp <$> (symbol "lam" *> trailingWhitespace tyName) <*> kind <*> pType
   48 
   49 ifixType :: Parser PType
   50 ifixType = withSpan $ \sp ->
   51     inParens $ TyIFix sp <$> (symbol "ifix" *> pType) <*> pType
   52 
   53 builtinType :: Parser PType
   54 builtinType = withSpan $ \sp -> inParens $ do
   55     SomeTypeIn (Kinded uni) <- symbol "con" *> defaultUni
   56     pure $ TyBuiltin sp (SomeTypeIn uni)
   57 
   58 sopType :: Parser PType
   59 sopType = withSpan $ \sp -> inParens $ TySOP sp <$> (symbol "sop" *> many tyList)
   60   where
   61     tyList :: Parser [PType]
   62     tyList = (inBrackets $ many pType) <* whitespace
   63 
   64 appType :: Parser PType
   65 appType = withSpan $ \sp -> inBrackets $ do
   66     fn   <- pType
   67     args <- some pType
   68     -- TODO: should not use the same `sp` for all arguments.
   69     pure $ mkIterTyApp fn ((sp,) <$> args)
   70 
   71 kind :: Parser (Kind SrcSpan)
   72 kind = withSpan $ \sp ->
   73     let typeKind = Type sp <$ symbol "type"
   74         funKind = KindArrow sp <$> (symbol "fun" *> kind) <*> kind
   75      in inParens (typeKind <|> funKind)
   76 
   77 -- | Parser for @PType@.
   78 pType :: Parser PType
   79 pType = choice $ map try
   80     [ funType
   81     , ifixType
   82     , allType
   83     , builtinType
   84     , lamType
   85     , appType
   86     , varType
   87     , sopType
   88     ]
   89 
   90 -- | Parser for built-in type applications.  The textual names here should match
   91 -- the ones in the PrettyBy instance for DefaultUni in PlutusCore.Default.Universe.
   92 defaultUniApplication :: Parser (SomeTypeIn (Kinded DefaultUni))
   93 defaultUniApplication = do
   94     -- Parse the head of the application.
   95     f <- defaultUni
   96     -- Parse the arguments.
   97     as <- many defaultUni
   98     -- Iteratively apply the head to the arguments checking that the kinds match and
   99     -- failing otherwise.
  100     foldM tryUniApply f as

  102 -- | Parser for built-in types (the ones from 'DefaultUni' specifically).
  103 --
  104 -- 'Kinded' is needed for checking that a type function can be applied to its argument.
  105 -- I.e. we do Plutus kind checking of builtin type applications during parsing, which is
  106 -- unfortunate, but there's no way we could construct a 'DefaultUni' otherwise.
  107 --
  108 -- In case of kind error no sensible message is shown, only an overly general one:
  109 --
  110 -- >>> :set -XTypeApplications
  111 -- >>> :set -XOverloadedStrings
  112 -- >>> import PlutusCore.Error
  113 -- >>> import PlutusCore.Quote
  114 -- >>> let runP = putStrLn . either display display . runQuoteT . parseGen @ParserErrorBundle defaultUni
  115 -- >>> runP "(list integer)"
  116 -- (list integer)
  117 -- >>> runP "(bool integer)"
  118 -- test:1:14:
  119 --   |
  120 -- 1 | (bool integer)
  121 --   |              ^
  122 -- expecting "bool", "bytestring", "data", "integer", "list", "pair", "string", "unit", or '('
  123 --
  124 -- This is to be fixed.
  125 --
  126 -- One thing we could do to avoid doing kind checking during parsing is to parse into
  127 --
  128 --     data TextualUni a where
  129 --         TextualUni :: TextualUni (Esc (Tree Text))
  130 --
  131 -- i.e. parse into @Tree Text@ and do the kind checking afterwards, but given that we'll still need
  132 -- to do the kind checking of builtins regardless (even for UPLC), we don't win much by deferring
  133 -- doing it.
  134 defaultUni :: Parser (SomeTypeIn (Kinded DefaultUni))
  135 defaultUni = choice $ map try
  136     [ trailingWhitespace (inParens defaultUniApplication)
  137     , someType @_ @Integer                    <$ symbol "integer"
  138     , someType @_ @ByteString                 <$ symbol "bytestring"
  139     , someType @_ @Text                       <$ symbol "string"
  140     , someType @_ @()                         <$ symbol "unit"
  141     , someType @_ @Bool                       <$ symbol "bool"
  142     , someType @_ @[]                         <$ symbol "list"
  143     , someType @_ @Strict.Vector              <$ symbol "array"
  144     , someType @_ @(,)                        <$ symbol "pair"
  145     , someType @_ @Data                       <$ symbol "data"
  146     , someType @_ @BLS12_381.G1.Element       <$ symbol "bls12_381_G1_element"
  147     , someType @_ @BLS12_381.G2.Element       <$ symbol "bls12_381_G2_element"
  148     , someType @_ @BLS12_381.Pairing.MlResult <$ symbol "bls12_381_mlresult"
  149     , someType @_ @Value                      <$ symbol "value"
  150     ]
  151 
  152 tyName :: Parser TyName
  153 tyName = TyName <$> name
