   19 {- | Newtype to provide 'Serialise' instances for types with a 'Flat' instance
   20   that just encodes the flat-serialized value as a CBOR bytestring
   21 -}
   22 newtype SerialiseViaFlat a = SerialiseViaFlat { unSerialiseViaFlat :: a }
   23 
   24 instance (Flat.Flat a) => Serialise (SerialiseViaFlat a) where
   25   encode = encode . Flat.flat . unSerialiseViaFlat
   26   decode = SerialiseViaFlat <$> decodeViaFlatWith Flat.decode
   27 
   28 decodeViaFlatWith :: Flat.Get a -> CBOR.Decoder s a
   29 decodeViaFlatWith decoder = do
   30   bs <- CBOR.decodeBytes
   31   -- lift any flat's failures to be cborg failures (MonadFail)
   32   fromRightM (fail . show) $ Flat.unflatWith decoder bs
   33 
   34 {- | The errors returned by `cborg` are plain strings (untyped). With this
   35 function we try to map onto datatypes, those cborg error messages that we are
   36 interested in.
   37 
   38 Currently we are only interested in error messages returned by the
   39 `CBOR.decodeBytes` decoder;
   40 see `PlutusLedgerApi.Common.SerialisedScript.scriptCBORDecoder`.
   41 -}
   42 readDeserialiseFailureInfo :: CBOR.DeserialiseFailure -> DeserialiseFailureInfo
   43 readDeserialiseFailureInfo (CBOR.DeserialiseFailure byteOffset reason) =
   44   DeserialiseFailureInfo byteOffset $ interpretReason reason
   45  where
   46   -- Note that this is subject to change if `cborg` dependency changes.
   47   -- Currently: cborg-0.2.10.0
   48   interpretReason :: String -> DeserialiseFailureReason
   49   interpretReason = \case
   50     -- Relevant Sources:
   51     -- <https://github.com/well-typed/cborg/blob/cborg-0.2.10.0/cborg/src/Codec/CBOR/Read.hs#L226>
   52     -- <https://github.com/well-typed/cborg/blob/cborg-0.2.10.0/cborg/src/Codec/CBOR/Read.hs#L1424>
   53     -- <https://github.com/well-typed/cborg/blob/cborg-0.2.10.0/cborg/src/Codec/CBOR/Read.hs#L1441>
   54     "end of input" -> EndOfInput
   55     -- Relevant Sources:
   56     -- <https://github.com/well-typed/cborg/blob/cborg-0.2.10.0/cborg/src/Codec/CBOR/Read.hs#L1051>
   57     "expected bytes" -> ExpectedBytes
   58     msg -> OtherReason msg
   59 
   60 {- | Similar to `CBOR.DeserialiseFailure`, with the difference that plain
   61 string reason messages are turned into the datatype: `DeserialiseFailureReason`.
   62 -}
   63 data DeserialiseFailureInfo = DeserialiseFailureInfo
   64   { dfOffset :: CBOR.ByteOffset
   65   , dfReason :: DeserialiseFailureReason
   66   }
   67   deriving stock (Eq, Show)
   68 
   69 instance Pretty DeserialiseFailureInfo where
   70   pretty (DeserialiseFailureInfo offset reason) =
   71     "CBOR deserialisation failed at the offset"
   72       <+> pretty offset
   73       <+> "for the following reason:"
   74       <+> pretty reason
   75 
   76 -- | The reason of the cbor failure as a datatype, not as a plain string.
   77 data DeserialiseFailureReason
   78   = -- | Not enough input provided
   79     EndOfInput
   80   | -- | The bytes inside the input are malformed.
   81     ExpectedBytes
   82   | -- | This is either a cbor failure that we (plutus) are not aware of,
   83     -- or an underlying flat failure. We use whatever message `cborg` or flat returns.
   84     OtherReason String
   85   deriving stock (Eq, Show)
   86 
   87 instance Pretty DeserialiseFailureReason where
   88   pretty = \case
   89     EndOfInput -> "reached the end of input while more data was expected."
   90     ExpectedBytes -> "the bytes inside the input are malformed."
   91     OtherReason msg -> pretty msg
