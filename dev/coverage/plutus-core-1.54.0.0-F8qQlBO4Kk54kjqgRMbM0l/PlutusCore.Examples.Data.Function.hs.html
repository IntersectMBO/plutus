   14 unsafeCoerce :: Term TyName Name uni fun ()
   15 unsafeCoerce = runQuote $ do
   16     a  <- freshTyName "a"
   17     b  <- freshTyName "b"
   18     return
   19         . TyAbs () a (Type ())
   20         . TyAbs () b (Type ())
   21         . Apply () (mkIterInstNoAnn fix [TyVar () a, TyVar () b])
