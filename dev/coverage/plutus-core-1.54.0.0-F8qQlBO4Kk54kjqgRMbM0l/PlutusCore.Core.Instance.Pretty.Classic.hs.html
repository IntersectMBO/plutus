   25 instance Pretty ann => PrettyBy (PrettyConfigClassic configName) (Kind ann) where
   26     prettyBy config = \case
   27         Type ann ->
   28             parens (sep (consAnnIf config ann
   29                 ["type"]))
   30         KindArrow ann k k' ->
   31             sexp "fun" (consAnnIf config ann
   32                 [prettyBy config k, prettyBy config k'])
   33 
   34 instance (PrettyClassicBy configName tyname, PrettyParens (SomeTypeIn uni), Pretty ann) =>
   35         PrettyBy (PrettyConfigClassic configName) (Type tyname uni ann) where
   36     prettyBy config = \case
   37         TyApp ann t t' ->
   38             brackets' (sep (consAnnIf config ann
   39                 [prettyBy config t, prettyBy config t']))
   40         TyVar ann n ->
   41             sep (consAnnIf config ann
   42                 [prettyBy config n])
   43         TyFun ann t t' ->
   44             sexp "fun" (consAnnIf config ann
   45                 [prettyBy config t, prettyBy config t'])
   46         TyIFix ann pat arg ->
   47             sexp "ifix" (consAnnIf config ann
   48                 [prettyBy config pat, prettyBy config arg])
   49         TyForall ann n k t ->
   50             sexp "all" (consAnnIf config ann
   51                 [prettyBy config n, prettyBy config k, prettyBy config t])
   52         TyBuiltin ann someUni ->
   53             sexp "con" (consAnnIf config ann [prettyBy juxtRenderContext someUni])
   54         TyLam ann n k t ->
   55             sexp "lam" (consAnnIf config ann
   56                 [prettyBy config n, prettyBy config k, prettyBy config t])
   57         TySOP ann tyls ->
   58             sexp "sop" (consAnnIf config ann (fmap prettyTyl tyls))
   59             where
   60               prettyTyl tyl = brackets (sep (fmap (prettyBy config) tyl))
   61 
   62 instance
   63         ( PrettyClassicBy configName tyname
   64         , PrettyClassicBy configName name
   65         , PrettyUni uni
   66         , Pretty fun
   67         , Pretty ann
   68         ) => PrettyBy (PrettyConfigClassic configName) (Term tyname name uni fun ann) where
   69     prettyBy config = \case
   70         Var ann n ->
   71             sep (consAnnIf config ann [prettyBy config n])
   72         TyAbs ann tn k t ->
   73             sexp "abs" (consAnnIf config ann
   74                 [prettyBy config tn, prettyBy config k, prettyBy config t])
   75         LamAbs ann n ty t ->
   76             sexp "lam" (consAnnIf config ann
   77                 [prettyBy config n, prettyBy config ty, prettyBy config t])
   78         Apply ann t1 t2 ->
   79             brackets' (sep (consAnnIf config ann
   80                 [prettyBy config t1, prettyBy config t2]))
   81         Constant ann c ->
   82             sexp "con" (consAnnIf config ann [prettyTypeOf c, pretty c])
   83         Builtin ann bi ->
   84             sexp "builtin" (consAnnIf config ann [pretty bi])
   85         TyInst ann t ty ->
   86             braces' (sep (consAnnIf config ann
   87                 [prettyBy config t, prettyBy config ty]))
   88         Error ann ty ->
   89             sexp "error" (consAnnIf config ann [prettyBy config ty])
   90         IWrap ann ty1 ty2 t ->
   91             sexp "iwrap" (consAnnIf config ann
   92                 [prettyBy config ty1, prettyBy config ty2, prettyBy config t])
   93         Unwrap ann t ->
   94             sexp "unwrap" (consAnnIf config ann [prettyBy config t])
   95         Constr ann ty i es ->
   96             sexp "constr" (consAnnIf config ann ([prettyBy config ty, pretty i ]
   97                                                  ++ (fmap (prettyBy config) es)))
   98         Case ann ty arg cs ->
   99             sexp "case" (consAnnIf config ann ([prettyBy config ty, prettyBy config arg]
  100                                                ++ (fmap (prettyBy config) cs)))
  101       where
  102         prettyTypeOf :: Some (ValueOf uni) -> Doc dann
  103         prettyTypeOf (Some (ValueOf uni _ )) = prettyBy juxtRenderContext $ SomeTypeIn uni
  104 
  105 instance (PrettyClassicBy configName (Term tyname name uni fun ann), Pretty ann) =>
  106         PrettyBy (PrettyConfigClassic configName) (Program tyname name uni fun ann) where
  107     prettyBy config (Program ann version term) =
  108         sexp "program" (consAnnIf config ann [pretty version, prettyBy config term])
