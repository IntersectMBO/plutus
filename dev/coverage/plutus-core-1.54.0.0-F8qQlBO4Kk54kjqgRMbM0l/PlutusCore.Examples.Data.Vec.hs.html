   24 -- > natK = (* -> *) -> * -> *
   25 natK :: Kind ()
   26 natK = KindArrow () (KindArrow () (Type ()) $ Type ()) . KindArrow () (Type ()) $ Type ()

   31 --
   32 -- > getEta n = \(f :: * -> *) (z :: *) -> n f z
   33 getEta :: Type TyName uni () -> Quote (Type TyName uni ())
   34 getEta n = do
   35     f <- freshTyName "f"
   36     z <- freshTyName "z"
   37     return
   38         . TyLam () f (KindArrow () (Type ()) $ Type ())
   39         . TyLam () z (Type ())
   40         $ mkIterTyAppNoAnn n
   41             [ TyVar () f
   42             , TyVar () z
   43             ]

   97 -- |
   98 --
   99 -- > stepFun a r1 r2 = all (p :: natK). a -> r1 p -> r2 (succT p)
  100 getStepFun :: TyName -> Type TyName uni () -> TyName -> Quote (Type TyName uni ())
  101 getStepFun a r1 r2 = do
  102     p <- freshTyName "p"
  103     return
  104         . TyForall () p natK
  105         . TyFun () (TyVar () a)
  106         . TyFun () (TyApp () r1 $ TyVar () p)
  107         . TyApp () (TyVar () r2)
  108         . TyApp () succT
  109         $ TyVar () p

  183 -- |
  184 --
  185 -- > churchConcat =
  186 -- >     /\(a :: *) (n :: natK) (m :: natK) -> \(xs : churchVec a n) (ys : churchVec a m) ->
  187 -- >         /\(r :: natK -> *) -> \(f : all (p :: natK). a -> r p -> r (succT p)) (z : r zeroT) ->
  188 -- >             xs
  189 -- >                 {\(p :: natK) -> r (plusT p m)}
  190 -- >                 (/\(p :: natK) -> f {plusT p m})
  191 -- >                 (ys {r} f z)
  192 churchConcat :: Term TyName Name uni fun ()
  193 churchConcat = runQuote $ do
  194     a <- freshTyName "a"
  195     n <- freshTyName "n"
  196     m <- freshTyName "m"
  197     xs <- freshName "xs"
  198     ys <- freshName "ys"
  199     r <- freshTyName "r"
  200     f <- freshName "f"
  201     z <- freshName "z"
  202     p <- freshTyName "p"
  203     stepFun <- getStepFun a (TyVar () r) r
  204     mEta <- getEta $ TyVar () m
  205     let plusTPM = mkIterTyAppNoAnn plusT [TyVar () p, TyVar () m]
  206     return
  207         . TyAbs () a (Type ())
  208         . TyAbs () n natK
  209         . TyAbs () m natK
  210         . LamAbs () xs (mkIterTyAppNoAnn churchVec [TyVar () a, TyVar () n])
  211         . LamAbs () ys (mkIterTyAppNoAnn churchVec [TyVar () a, mEta])
  212         . TyAbs () r (KindArrow () natK $ Type ())
  213         . LamAbs () f stepFun
  214         . LamAbs () z (TyApp () (TyVar () r) zeroT)
  215         $ mkIterAppNoAnn (TyInst () (Var () xs) . TyLam () p natK $ TyApp () (TyVar () r) plusTPM)
  216             [ TyAbs () p natK $ TyInst () (Var () f) plusTPM
  217             , mkIterAppNoAnn (TyInst () (Var () ys) $ TyVar () r)
  218                 [ Var () f
  219                 , Var () z
  220                 ]
  221             ]

  315 -- > scottHead =
  316 -- >     /\(a :: *) (n :: natK) -> (xs : scottVec a (suc n)) ->
  317 -- >         unwrap
  318 -- >             xs
  319 -- >             {\(p :: natK) -> p (\(z :: *) -> a) unit}
  320 -- >             (/\(p :: natK) (x : a) (xs' : scottVec a p) -> x)
  321 -- >             unitval
  322 scottHead :: uni `HasTypeAndTermLevel` () => Term TyName Name uni fun ()
  323 scottHead = runQuote $ do
  324     a <- freshTyName "a"
  325     n <- freshTyName "n"
  326     p <- freshTyName "p"
  327     z <- freshTyName "z"
  328     x <- freshName "x"
  329     xs <- freshName "xs"
  330     xs' <- freshName "xs'"
  331     return
  332         . TyAbs () a (Type ())
  333         . TyAbs () n natK
  334         . LamAbs () xs (mkIterTyAppNoAnn scottVec [TyVar () a, TyApp () succT $ TyVar () n])
  335         $ mkIterAppNoAnn
  336             (   TyInst () (Unwrap () $ Var () xs)
  337               . TyLam () p natK
  338                 $ mkIterTyAppNoAnn (TyVar () p)
  339                     [ TyLam () z (Type ()) $ TyVar () a
  340                     , unit
  341                     ])
  342             [   TyAbs () p natK
  343               . LamAbs () x (TyVar () a)
  344               . LamAbs () xs' (mkIterTyAppNoAnn scottVec [TyVar () a, TyVar () p])
  345               $ Var () x
  346             , unitval

  350 --
  351 -- > scottSumHeadsOr0 =
  352 -- >     /\(n :: natK) -> (xs ys : scottVec integer n) ->
  353 -- >         unwrap
  354 -- >             xs
  355 -- >             {\(p :: natK) -> (scottVec Integer n -> scottVec integer p) -> integer}
  356 -- >             (/\(p :: natK) (x : integer) (xs' : scottVec integer p)
  357 -- >                 \(coe : scottVec Integer n -> scottVec integer (suc p)) ->
  358 -- >                     x + scottHead {integer} {p} (coe ys))
  359 -- >             (\(coe : scottVec Integer n -> scottVec integer zero) -> 0)
  360 -- >             (/\(xs' :: scottVec Integer n) -> xs')
  361 scottSumHeadsOr0
  362     :: (uni `HasTypeAndTermLevel` Integer, uni `HasTypeAndTermLevel` ())
  363     => Term TyName Name uni DefaultFun ()
  364 scottSumHeadsOr0 = runQuote $ do
  365     n <- freshTyName "n"
  366     p <- freshTyName "p"
  367     x <- freshName "x"
  368     xs <- freshName "xs"
  369     ys <- freshName "ys"
  370     xs' <- freshName "xs'"
  371     coe <- freshName "coe"
  372     let vecInteger l = mkIterTyAppNoAnn scottVec [integer, l]
  373     return
  374         . TyAbs () n natK
  375         . LamAbs () xs (vecInteger $ TyVar () n)
  376         . LamAbs () ys (vecInteger $ TyVar () n)
  377         $ mkIterAppNoAnn
  378             (   TyInst () (Unwrap () $ Var () xs)
  379               . TyLam () p natK
  380               . TyFun () (TyFun () (vecInteger $ TyVar () n) $ vecInteger (TyVar () p))
  381               $ integer
  382             )
  383             [   TyAbs () p natK
  384               . LamAbs () x integer
  385               . LamAbs () xs' (vecInteger $ TyVar () p)
  386               . LamAbs () coe
  387                   (TyFun () (vecInteger $ TyVar () n) $ vecInteger (TyApp () succT $ TyVar () p))
  388               $ mkIterAppNoAnn (builtin () AddInteger)
  389                   [ Var () x
  390                   ,   Apply () (mkIterInstNoAnn scottHead [integer, TyVar () p])
  391                     . Apply () (Var () coe)
  392                     $ Var () ys
  393                   ]
  394             ,   LamAbs () coe (TyFun () (vecInteger $ TyVar () n) $ vecInteger zeroT)
  395               $ mkConstant @Integer () 0
  396             , LamAbs () xs' (vecInteger $ TyVar () n) $ Var () xs'
  397             ]
