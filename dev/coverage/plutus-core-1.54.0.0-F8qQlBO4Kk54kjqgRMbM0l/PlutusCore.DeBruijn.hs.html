   49 {- Note [DeBruijn indices of Binders]
   50 In a debruijnijfied Term AST, the Binders have a debruijn index
   51 at their the specific AST location.
   52 During *undebruijnification* we ignore such binders' indices because they are meaningless,
   53 and instead use the convention that such introduced binders have
   54 a fixed debruijn index '0' at their introduction.
   55 -}

  115 {- Note [De Bruijn conversion and recursion schemes]
  116 These are quite repetitive, but we can't use a catamorphism for this, because
  117 we are not only altering the recursive type, but also the name type parameters.
  118 These are normally constant in a catamorphic application.
  119 -}
  120 
  121 deBruijnTyWithM
  122     :: forall m uni ann. MonadReader LevelInfo m
  123     => (Unique -> m Index)
  124     -> Type TyName uni ann
  125     -> m (Type NamedTyDeBruijn uni ann)
  126 deBruijnTyWithM h = go
  127   where
  128     go :: Type TyName uni ann -> m (Type NamedTyDeBruijn uni ann)
  129     go = \case
  130         -- variable case
  131         TyVar ann n -> TyVar ann <$> tyNameToDeBruijn h n
  132         -- binder cases
  133         TyForall ann tn k ty -> declareUnique tn $ do
  134             tn' <- tyNameToDeBruijn h tn
  135             withScope $ TyForall ann tn' k <$> go ty
  136         TyLam ann tn k ty -> declareUnique tn $ do
  137             tn' <- tyNameToDeBruijn h tn
  138             withScope $ TyLam ann tn' k <$> go ty
  139         -- boring recursive cases
  140         TyFun ann i o -> TyFun ann <$> go i <*> go o
  141         TyApp ann fun arg -> TyApp ann <$> go fun <*> go arg
  142         TyIFix ann pat arg -> TyIFix ann <$> go pat <*> go arg
  143         TySOP ann tyls -> TySOP ann <$> (traverse . traverse) go tyls
  144         -- boring non-recursive cases
  145         TyBuiltin ann someUni -> pure $ TyBuiltin ann someUni
  146 
  147 deBruijnTermWithM
  148     :: forall m uni fun ann. (MonadReader LevelInfo m)
  149     => (Unique -> m Index)
  150     -> Term TyName Name uni fun ann
  151     -> m (Term NamedTyDeBruijn NamedDeBruijn uni fun ann)
  152 deBruijnTermWithM h = go
  153   where
  154     goT :: Type TyName uni ann -> m (Type NamedTyDeBruijn uni ann)
  155     goT = deBruijnTyWithM h
  156 
  157     go :: Term TyName Name uni fun ann -> m (Term NamedTyDeBruijn NamedDeBruijn uni fun ann)
  158     go = \case
  159         -- variable case
  160         Var ann n -> Var ann <$> nameToDeBruijn h n
  161         -- binder cases
  162         TyAbs ann tn k t -> declareUnique tn $ do
  163             tn' <- tyNameToDeBruijn h tn
  164             withScope $ TyAbs ann tn' k <$> go t
  165         LamAbs ann n ty t -> declareUnique n $ do
  166             n' <- nameToDeBruijn h n
  167             withScope $ LamAbs ann n' <$> goT ty <*> go t
  168         -- boring recursive cases
  169         Apply ann t1 t2 -> Apply ann <$> go t1 <*> go t2
  170         TyInst ann t ty -> TyInst ann <$> go t <*> goT ty
  171         Unwrap ann t -> Unwrap ann <$> go t
  172         IWrap ann pat arg t -> IWrap ann <$> goT pat <*> goT arg <*> go t
  173         Error ann ty -> Error ann <$> goT ty
  174         Constr ann ty i es -> Constr ann <$> goT ty <*> pure i <*> traverse go es
  175         Case ann ty arg cs -> Case ann <$> goT ty <*> go arg <*> traverse go cs
  176         -- boring non-recursive cases
  177         Constant ann con -> pure $ Constant ann con
  178         Builtin ann bn -> pure $ Builtin ann bn
  179 
  180 -- | Takes a "handler" function to execute when encountering free variables.
  181 unDeBruijnTyWithM
  182     :: forall m uni ann. (MonadReader LevelInfo m, MonadQuote m)
  183     => (Index -> m Unique)
  184     -> Type NamedTyDeBruijn uni ann
  185     -> m (Type TyName uni ann)
  186 unDeBruijnTyWithM h = go
  187   where
  188     go :: Type NamedTyDeBruijn uni ann -> m (Type TyName uni ann)
  189     go = \case
  190       -- variable case
  191       TyVar ann n -> TyVar ann <$> deBruijnToTyName h n
  192       -- binder cases
  193       TyForall ann tn k ty ->
  194           -- See Note [DeBruijn indices of Binders]
  195           declareBinder $ do
  196             tn' <- deBruijnToTyName h $ set index deBruijnInitIndex tn
  197             withScope $ TyForall ann tn' k <$> go ty
  198       TyLam ann tn k ty ->
  199           -- See Note [DeBruijn indices of Binders]
  200           declareBinder $ do
  201             tn' <- deBruijnToTyName h $ set index deBruijnInitIndex tn
  202             withScope $ TyLam ann tn' k <$> go ty
  203       -- boring recursive cases
  204       TyFun ann i o -> TyFun ann <$> go i <*> go o
  205       TyApp ann fun arg -> TyApp ann <$> go fun <*> go arg
  206       TyIFix ann pat arg -> TyIFix ann <$> go pat <*> go arg
  207       TySOP ann tyls -> TySOP ann <$> (traverse . traverse) go tyls
  208       -- boring non-recursive cases
  209       TyBuiltin ann someUni -> pure $ TyBuiltin ann someUni

  212 unDeBruijnTermWithM

  215     -> Term NamedTyDeBruijn NamedDeBruijn uni fun ann
  216     -> m (Term TyName Name uni fun ann)

  222     go :: Term NamedTyDeBruijn NamedDeBruijn uni fun ann -> m (Term TyName Name uni fun ann)

  232         LamAbs ann n ty t ->
  233             -- See Note [DeBruijn indices of Binders]
  234             declareBinder $ do
  235               n' <- deBruijnToName h $ set index deBruijnInitIndex n
  236               withScope $ LamAbs ann n' <$> goT ty <*> go t
  237         -- boring recursive cases
  238         Apply ann t1 t2 -> Apply ann <$> go t1 <*> go t2
  239         TyInst ann t ty -> TyInst ann <$> go t <*> goT ty
  240         Unwrap ann t -> Unwrap ann <$> go t
  241         IWrap ann pat arg t -> IWrap ann <$> goT pat <*> goT arg <*> go t
  242         Error ann ty -> Error ann <$> goT ty
  243         Constr ann ty i es -> Constr ann <$> goT ty <*> pure i <*> traverse go es
  244         Case ann ty arg cs -> Case ann <$> goT ty <*> go arg <*> traverse go cs
  245         -- boring non-recursive cases

  247         Builtin ann bn -> pure $ Builtin ann bn
