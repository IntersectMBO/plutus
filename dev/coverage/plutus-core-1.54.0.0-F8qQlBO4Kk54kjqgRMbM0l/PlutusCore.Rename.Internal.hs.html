   23 -- | A wrapper for signifying that the value inside of it satisfies global uniqueness.
   24 --
   25 -- It's safe to call 'unRenamed', it's not safe to call 'Renamed', hence the latter is only exported
   26 -- from this internal module and should not be exported from the main API.
   27 --
   28 -- Don't provide any instances allowing the user to create a 'Renamed' (even out of an existing one
   29 -- like with 'Functor').
   30 newtype Renamed a = Renamed
   31     { unRenamed :: a
   32     } deriving stock (Show, Eq)
   33 
   34 -- | @Dupable a@ is isomorphic to @a@, but the only way to extract the @a@ is via 'liftDupable'
   35 -- (defined in the main API module because of a constraint requirement) which renames the stored
   36 -- value along the way. This type is used whenever
   37 --
   38 -- 1. preserving global uniqueness is required
   39 -- 2. some value may be used multiple times
   40 --
   41 -- so we annotate such a value with 'Dupable' and call 'liftDupable' at each usage, which ensures
   42 -- global uniqueness is preserved.
   43 --
   44 -- 'unDupable' is not supposed to be exported. Don't provide any instances allowing the user to
   45 -- access the underlying value.
   46 newtype Dupable a = Dupable
   47     { unDupable :: a
   48     } deriving stock (Show, Eq)

   50 -- | Replace the unique in the name stored in a 'TyVarDecl' by a new unique, save the mapping
   51 -- from the old unique to the new one and supply the updated 'TyVarDecl' to a continuation.
   52 withFreshenedTyVarDecl
   53     :: (HasRenaming ren TypeUnique, HasUniques (Type tyname uni ann), MonadQuote m, MonadReader ren m)
   54     => TyVarDecl tyname ann
   55     -> (TyVarDecl tyname ann -> m c)
   56     -> m c
   57 withFreshenedTyVarDecl (TyVarDecl ann name kind) cont =
   58     withFreshenedName name $ \nameFr -> cont $ TyVarDecl ann nameFr kind
   59 
   60 -- | Replace the unique in the name stored in a 'VarDecl' by a new unique, save the mapping
   61 -- from the old unique to the new one and supply to a continuation the computation that
   62 -- renames the type stored in the updated 'VarDecl'.
   63 -- The reason the continuation receives a computation rather than a pure term is that we may want
   64 -- to bring several term and type variables in scope before renaming the types of term variables.
   65 -- This situation arises when we want to rename a bunch of mutually recursive bindings.
   66 withFreshenedVarDecl
   67     :: (HasUniques (Term tyname name uni fun ann), MonadQuote m, MonadReader ScopedRenaming m)
   68     => VarDecl tyname name uni ann
   69     -> (m (VarDecl tyname name uni ann) -> m c)
   70     -> m c
   71 withFreshenedVarDecl (VarDecl ann name ty) cont =
   72     withFreshenedName name $ \nameFr -> cont $ VarDecl ann nameFr <$> renameTypeM ty
   73 
   74 -- | Rename a 'Type' in the 'RenameM' monad.
   75 renameTypeM
   76     :: (HasRenaming ren TypeUnique, HasUniques (Type tyname uni ann), MonadQuote m, MonadReader ren m)
   77     => Type tyname uni ann -> m (Type tyname uni ann)
   78 renameTypeM (TyLam ann name kind ty)    =
   79     withFreshenedName name $ \nameFr -> TyLam ann nameFr kind <$> renameTypeM ty
   80 renameTypeM (TyForall ann name kind ty) =
   81     withFreshenedName name $ \nameFr -> TyForall ann nameFr kind <$> renameTypeM ty
   82 renameTypeM (TyIFix ann pat arg)        = TyIFix ann <$> renameTypeM pat <*> renameTypeM arg
   83 renameTypeM (TyApp ann fun arg)         = TyApp ann <$> renameTypeM fun <*> renameTypeM arg
   84 renameTypeM (TyFun ann dom cod)         = TyFun ann <$> renameTypeM dom <*> renameTypeM cod
   85 renameTypeM (TyVar ann name)            = TyVar ann <$> renameNameM name
   86 renameTypeM (TySOP ann tyls)            = TySOP ann <$> (traverse . traverse) renameTypeM tyls
   87 renameTypeM ty@TyBuiltin{}              = pure ty
   88 
   89 -- | Rename a 'Term' in the 'RenameM' monad.
   90 renameTermM
   91     :: (HasUniques (Term tyname name uni fun ann), MonadQuote m, MonadReader ScopedRenaming m)
   92     => Term tyname name uni fun ann -> m (Term tyname name uni fun ann)
   93 renameTermM (LamAbs ann name ty body)  =
   94     withFreshenedName name $ \nameFr -> LamAbs ann nameFr <$> renameTypeM ty <*> renameTermM body
   95 renameTermM (TyAbs ann name kind body) =
   96     withFreshenedName name $ \nameFr -> TyAbs ann nameFr kind <$> renameTermM body
   97 renameTermM (IWrap ann pat arg term)   =
   98     IWrap ann <$> renameTypeM pat <*> renameTypeM arg <*> renameTermM term
   99 renameTermM (Apply ann fun arg)        = Apply ann <$> renameTermM fun <*> renameTermM arg
  100 renameTermM (Unwrap ann term)          = Unwrap ann <$> renameTermM term
  101 renameTermM (Error ann ty)             = Error ann <$> renameTypeM ty
  102 renameTermM (TyInst ann term ty)       = TyInst ann <$> renameTermM term <*> renameTypeM ty
  103 renameTermM (Var ann name)             = Var ann <$> renameNameM name
  104 renameTermM (Constr ann ty i es)       = Constr ann <$> renameTypeM ty <*> pure i <*> traverse renameTermM es
  105 renameTermM (Case ann ty arg cs)       = Case ann <$> renameTypeM ty <*> renameTermM arg <*> traverse renameTermM cs
  106 renameTermM con@Constant{}             = pure con
  107 renameTermM bi@Builtin{}               = pure bi
  108 
  109 -- | Rename a 'Program' in the 'RenameM' monad.
  110 renameProgramM
  111     :: (HasUniques (Program tyname name uni fun ann), MonadQuote m, MonadReader ScopedRenaming m)
  112     => Program tyname name uni fun ann -> m (Program tyname name uni fun ann)
  113 renameProgramM (Program ann ver term) = Program ann ver <$> renameTermM term
