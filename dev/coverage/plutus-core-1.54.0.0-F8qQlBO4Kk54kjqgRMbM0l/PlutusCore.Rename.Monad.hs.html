   35 -- | The monad the renamer runs in.
   36 newtype RenameT ren m a = RenameT
   37     { unRenameT :: ReaderT ren m a
   38     } deriving newtype
   39         ( Functor, Applicative, Alternative, Monad
   40         , MonadReader ren
   41         , MonadQuote
   42         )
   43 
   44 -- | A renaming is a mapping from old uniques to new ones.
   45 newtype Renaming unique = Renaming
   46     { unRenaming :: UniqueMap unique unique
   47     } deriving newtype (Semigroup, Monoid)
   48 
   49 -- | A type-level renaming.
   50 -- Needed for instantiating functions running over types in generic @RenameT ren m@ to
   51 -- a particular type of renaming.
   52 type TypeRenaming = Renaming TypeUnique
   53 
   54 -- | A class that specifies which 'Renaming' a @ren@ has inside.
   55 -- A @ren@ can contain several 'Renaming's (like 'Scoped', for example).
   56 class Coercible unique Unique => HasRenaming ren unique where
   57     renaming :: Lens' ren (Renaming unique)
   58 
   59 -- | Scoping-aware mapping from locally unique uniques to globally unique uniques.
   60 data ScopedRenaming = ScopedRenaming
   61     { _scopedRenamingTypes :: Renaming TypeUnique
   62     , _scopedRenamingTerms :: Renaming TermUnique
   63     }
   64 
   65 makeLenses ''ScopedRenaming
   66 
   67 type ScopedRenameT = RenameT ScopedRenaming
   68 
   69 instance Semigroup ScopedRenaming where
   70     ScopedRenaming types1 terms1 <> ScopedRenaming types2 terms2 =
   71         ScopedRenaming (types1 <> types2) (terms1 <> terms2)
   72 
   73 instance Monoid ScopedRenaming where
   74     mempty = ScopedRenaming mempty mempty
   75 
   76 instance (Coercible unique1 Unique, unique1 ~ unique2) =>
   77         HasRenaming (Renaming unique1) unique2 where
   78     renaming = id
   79 
   80 instance HasRenaming ScopedRenaming TypeUnique where
   81     renaming = scopedRenamingTypes . renaming
   82 
   83 instance HasRenaming ScopedRenaming TermUnique where
   84     renaming = scopedRenamingTerms . renaming
   85 
   86 -- | Run a 'RenameT' computation with an empty renaming.
   87 runRenameT :: Monoid ren => RenameT ren m a -> m a
   88 runRenameT (RenameT a) = runReaderT a mempty
   89 
   90 -- | Map the underlying representation of 'Renaming'.
   91 mapRenaming
   92     :: (UniqueMap unique unique -> UniqueMap unique unique)
   93     -> Renaming unique
   94     -> Renaming unique
   95 mapRenaming = coerce
   96 
   97 -- | Save the mapping from the @unique@ of a name to a new @unique@.
   98 insertByNameM
   99     :: (HasUnique name unique, HasRenaming ren unique)
  100     => name -> unique -> ren -> ren
  101 insertByNameM name = over renaming . mapRenaming . insertByName name
  102 
  103 -- | Look up the new unique a name got mapped to.
  104 lookupNameM
  105     :: (HasUnique name unique, HasRenaming ren unique, MonadReader ren m)
  106     => name -> m (Maybe unique)
  107 lookupNameM name = asks $ lookupName name . unRenaming . view renaming
  108 
  109 -- | Rename a name that has a unique inside.
  110 renameNameM
  111     :: (HasRenaming ren unique, HasUnique name unique, MonadReader ren m)
  112     => name -> m name
  113 renameNameM name = do
  114     mayUniqNew <- lookupNameM name
  115     pure $ case mayUniqNew of
  116         Nothing      -> name
  117         Just uniqNew -> name & unique .~ uniqNew
  118 
  119 -- | Replace the unique in a name by a new unique, save the mapping
  120 -- from the old unique to the new one and supply the updated value to a continuation.
  121 withFreshenedName
  122     :: (HasRenaming ren unique, HasUnique name unique, MonadQuote m, MonadReader ren m)
  123     => name -> (name -> m c) -> m c
  124 withFreshenedName nameOld k = do
  125     uniqNew <- coerce <$> freshUnique
  126     local (insertByNameM nameOld uniqNew) $ k (nameOld & unique .~ uniqNew)
  127 
  128 -- | Run a 'RenameT' computation in the environment extended by the mapping from an old name
  129 -- to a new one.
  130 withRenamedName
  131     :: (HasRenaming ren unique, HasUnique name unique, MonadReader ren m)
  132     => name -> name -> m c -> m c
  133 withRenamedName old new = local $ insertByNameM old (new ^. unique)
