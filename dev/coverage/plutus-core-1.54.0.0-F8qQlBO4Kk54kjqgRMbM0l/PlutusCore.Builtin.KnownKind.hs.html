   20 -- | The type of singletonized Haskell kinds representing Plutus kinds.
   21 -- Indexing by a Haskell kind allows us to avoid an 'error' call in the 'ToKind' instance of
   22 -- 'DefaultUni' and not worry about proxies in the type of 'knownKind'.
   23 data SingKind k where
   24     SingType      :: SingKind GHC.Type
   25     SingKindArrow :: SingKind k -> SingKind l -> SingKind (k -> l)
   26 
   27 -- | Feed the 'SingKind' version of the given 'Kind' to the given continuation.
   28 withSingKind :: Kind ann -> (forall k. SingKind k -> r) -> r
   29 withSingKind (Type _) k = k SingType
   30 withSingKind (KindArrow _ dom cod) k =
   31     withSingKind dom $ \domS ->
   32         withSingKind cod $ \codS ->
   33             k $ SingKindArrow domS codS
   34 
   35 -- | For reifying Haskell kinds representing Plutus kinds at the term level.
   36 class KnownKind k where
   37     knownKind :: SingKind k
   38 
   39 -- | Plutus only supports lifted types, hence the equality constraint.
   40 instance rep ~ LiftedRep => KnownKind (TYPE rep) where
   41     knownKind = SingType
   42 
   43 instance (KnownKind dom, KnownKind cod) => KnownKind (dom -> cod) where
   44     knownKind = SingKindArrow (knownKind @dom) (knownKind @cod)
   45 
   46 -- | Satisfy the 'KnownKind' constraint of a continuation using the given 'SingKind'.
   47 bringKnownKind :: SingKind k -> (KnownKind k => r) -> r
   48 bringKnownKind SingType                r = r
   49 bringKnownKind (SingKindArrow dom cod) r = bringKnownKind dom $ bringKnownKind cod r
   50 
   51 withKnownKind :: Kind ann -> (forall k. KnownKind k => Proxy k -> r) -> r
   52 withKnownKind kind k =
   53     withSingKind kind $ \(kindS :: SingKind kind) ->
   54        bringKnownKind kindS $ k $ Proxy @kind
   55 
   56 -- We need this for type checking Plutus, however we get Plutus types/terms/programs by either
   57 -- producing them directly or by parsing/decoding them and in both the cases we have access to the
   58 -- @Typeable@ information for the Haskell kind of a type and so we could just keep it around
   59 -- (instead of throwing it away like we do now) and compute the Plutus kind directly from that.
   60 -- That might be less efficient and probably requires updating the Plutus Tx compiler, so we went
   61 -- with the simplest option for now and it's to have a class. Providing an instance per universe is
   62 -- no big deal.
   63 -- | For computing the Plutus kind of a built-in type. See 'kindOfBuiltinType'.
   64 class ToKind (uni :: GHC.Type -> GHC.Type) where
   65     -- | Reify the kind of a type from the universe at the term level.
   66     toSingKind :: uni (Esc (a :: k)) -> SingKind k
   67 
   68 -- | Convert a reified Haskell kind to a Plutus kind.
   69 demoteKind :: SingKind k -> Kind ()
   70 demoteKind SingType                = Type ()
   71 demoteKind (SingKindArrow dom cod) = KindArrow () (demoteKind dom) (demoteKind cod)
   72 
   73 -- | Compute the kind of a type from a universe.
   74 kindOfBuiltinType :: ToKind uni => uni (Esc a) -> Kind ()
   75 kindOfBuiltinType = demoteKind . toSingKind
