    2 {- Note [Applying force to delays in case branches]
    3 
    4 Often, the following pattern occurs in UPLC terms:
    5 
    6 > force (case scrut [\x1... -> delay term_1, ..., \x1... -> delay term_m])
    7 
    8 It's sound to remove the 'force' and the 'delay's, as long as the original term is "well-formed".
    9 Note that the lambda abstraction may be missing, and we consider that case as well.
   10 
   11 Intuitively, what we mean by "well-formed" is that the term does not evaluate to bottom unless
   12 the user intended it to (i.e. by introducing an 'error' subterm).
   13 
   14 In the the context of the Plinth compiler pipeline, UPLC is always generated from TPLC, by
   15 erasing the types of a TPLC term. So at the beginning of the UPLC phase of the compiler,
   16 the UPLC can be considered "well-formed", since it is guaranteed by the types of the original
   17 TPLC term.
   18 
   19 The other UPLC transformations we have are guaranteed to preserve this property, so we can assume
   20 that any UPLC term coming from the UPLC phase of the compiler is "well-formed".
   21 
   22 What about preserving the correct evaluation strategy? Since we are removing 'delay's, we need to
   23 ensure that the terms under the delays are not evaluated ahead of time. However, this is not a problem
   24 because the 'case' construct is lazy in its branches, so the terms under the 'delay's will not be
   25 evaluated unless the corresponding branch is taken.
   26 
   27 We should, however, formally define what "well-formed" means, and this is left as future work:
   28 FIXME(https://github.com/IntersectMBO/plutus-private/issues/1644).
   29 
   30 -}
   31 module UntypedPlutusCore.Transform.ForceCaseDelay
   32   ( forceCaseDelay,
   33   )
   34 where
   35 
   36 import UntypedPlutusCore.Core
   37 import UntypedPlutusCore.Transform.Simplifier (SimplifierStage (ForceCaseDelay), SimplifierT,
   38                                                recordSimplification)
   39 
   40 import Control.Lens
   41 
   42 forceCaseDelay
   43     :: Monad m
   44     => Term name uni fun a
   45     -> SimplifierT name uni fun a m (Term name uni fun a)
   46 forceCaseDelay term = do
   47   let result = transformOf termSubterms processTerm term
   48   recordSimplification term ForceCaseDelay result
   49   return result
   50 
   51 processTerm :: Term name uni fun a -> Term name uni fun a
   52 processTerm = \case
   53   original@(Force _ (Case cAnn scrut branches)) ->
   54     let mNewBranches = traverse findDelayUnderLambdas branches
   55      in case mNewBranches of
   56           Just newBranches ->
   57             Case cAnn scrut newBranches
   58           Nothing -> original
   59   other -> other
   60   where
   61     findDelayUnderLambdas :: Term name uni fun a -> Maybe (Term name uni fun a)
   62     findDelayUnderLambdas = \case
   63       LamAbs ann var body -> LamAbs ann var <$> findDelayUnderLambdas body
   64       Delay _ term    -> Just term
   65       _               -> Nothing
