    5 {- |
    6 Perform the case-of-case transformation. This pushes
    7 case expressions into the case branches of other case
    8 expressions, which can often yield optimization opportunities.
    9 
   10 Example:
   11 @
   12     case (case s of { C1 a -> x; C2 b -> y; }) of
   13       D1 -> w
   14       D2 -> z
   15 
   16     -->
   17 
   18     case s of
   19       C1 a -> case x of { D1 -> w; D2 -> z; }
   20       C2 b -> case y of { D1 -> w; D2 -> z; }
   21 @
   22 
   23 We also transform
   24 
   25 @
   26     case ((force ifThenElse) b (constr t) (constr f)) alts
   27 @
   28 
   29 into
   30 
   31 @
   32     force (force ifThenElse b (delay (case (constr t) alts)) (delay (case (constr f) alts)))
   33 @
   34 
   35 This is always an improvement.
   36 -}
   37 module UntypedPlutusCore.Transform.CaseOfCase (caseOfCase) where
   38 
   39 import PlutusPrelude
   40 
   41 import PlutusCore qualified as PLC
   42 import PlutusCore.Builtin (CaseBuiltin (..))
   43 import PlutusCore.MkPlc (mkIterApp)
   44 import UntypedPlutusCore.Core
   45 import UntypedPlutusCore.Transform.CaseReduce qualified as CaseReduce
   46 import UntypedPlutusCore.Transform.Simplifier (SimplifierStage (CaseOfCase), SimplifierT,
   47                                                recordSimplification)
   48 
   49 import Control.Lens
   50 import Data.List (nub)
   51 
   52 caseOfCase
   53     :: ( fun ~ PLC.DefaultFun, Monad m, CaseBuiltin uni
   54        , PLC.GEq uni, PLC.Closed uni, uni `PLC.Everywhere` Eq
   55        )
   56     => Term name uni fun a
   57     -> SimplifierT name uni fun a m (Term name uni fun a)
   58 caseOfCase term = do
   59   let result = transformOf termSubterms processTerm term
   60   recordSimplification term CaseOfCase result
   61   return result
   62 
   63 processTerm
   64     :: ( fun ~ PLC.DefaultFun, CaseBuiltin uni
   65        , PLC.GEq uni, PLC.Closed uni, uni `PLC.Everywhere` Eq
   66        )
   67     => Term name uni fun a -> Term name uni fun a
   68 processTerm = \case
   69   Case ann scrut alts
   70     | ( ite@(Force a (Builtin _ PLC.IfThenElse))
   71         , [cond, (trueAnn, true@Constr{}), (falseAnn, false@Constr{})]
   72         ) <-
   73         splitApplication scrut ->
   74         Force a $
   75           mkIterApp
   76             ite
   77             [ cond
   78             -- Here we call a single step of case-reduce in order to immediately clean up the
   79             -- duplication of @alts@. Otherwise optimizing case-of-case-of-case-of... would create
   80             -- exponential blowup of the case branches, which would eventually get deduplicated
   81             -- with case-reduce, but only after that exponential blowup has already slowed the
   82             -- optimizer down unnecessarily.
   83             , (trueAnn, Delay trueAnn . CaseReduce.processTerm $ Case ann true alts)
   84             , (falseAnn, Delay falseAnn . CaseReduce.processTerm $ Case ann false alts)
   85             ]
   86   original@(Case annOuter (Case annInner scrut altsInner) altsOuter) ->
   87     maybe
   88       original
   89       (Case annInner scrut)
   90       (do
   91         constrs <- for altsInner $ \case
   92           c@(Constr _ i _)   -> Just (Left i, c)
   93           c@(Constant _ val) -> Just (Right val, c)
   94           _                  -> Nothing
   95         -- See Note [Case-of-case and duplicating code].
   96         guard $ length (nub . toList $ fmap fst constrs) == length constrs
   97         pure $ constrs <&> \(_, c) -> CaseReduce.processTerm $ Case annOuter c altsOuter)
   98   other -> other
