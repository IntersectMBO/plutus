   18 -- | Convert an object to a hashmap. Compatibility shim for pre-/post-aeson-2.
   19 objToHm :: Object -> HM.HashMap Text.Text Value
   20 #if MIN_VERSION_aeson(2,0,0)
   21 objToHm = toHashMapText
   22 #else
   23 objToHm = id
   24 #endif
   25 
   26 -- | Convert a hashmap to an object. Compatibility shim for pre-/post-aeson-2.
   27 hmToObj :: HM.HashMap Text.Text Value -> Object
   28 #if MIN_VERSION_aeson(2,0,0)
   29 hmToObj = fromHashMapText
   30 #else
   31 hmToObj = id
   32 #endif

   34 -- | Turn a nested object into a "flat" object where the keys represent paths into the original
   35 -- object. The keys in the result will be the keys in the original path, separated by `sep`.
   36 -- The inverse of 'unflattenObject'.
   37 flattenObject :: Text.Text -> Object -> Object
   38 flattenObject sep o = hmToObj $ go Nothing (objToHm o)
   39     where
   40         go :: Maybe Text.Text -> HM.HashMap Text.Text Value -> HM.HashMap Text.Text Value
   41         go mprefix = HM.foldMapWithKey $ \k v ->
   42             let newName = case mprefix of
   43                     Just prefix -> prefix <> sep <> k
   44                     Nothing     -> k
   45             in case v of
   46                 Object o' -> go (Just newName) $ objToHm o'
   47                 leaf      -> HM.singleton newName leaf
   48 
   49 -- | Turn a "flat" object whose keys represent paths into an unflattened object.
   50 -- The keys in the result will be the resulting path, separated by `sep`.
   51 -- The inverse of 'flattenObject'.
   52 unflattenObject :: Text.Text -> Object -> Object
   53 unflattenObject sep o =
   54     HM.foldlWithKey (\acc k v -> mergeObject acc (mkPathObject k v)) mempty (objToHm o)
   55     where
   56         mkPathObject :: Text.Text -> Value -> Object
   57         mkPathObject k value =
   58             let path = Text.splitOn sep k
   59             in hmToObj $ go path value
   60             where
   61                 go :: [Text.Text] -> Value -> HM.HashMap Text.Text Value
   62                 go [] _        = error "empty path"
   63                 go [n] v       = HM.singleton n v
   64                 go (n:n':xs) v = HM.singleton n $ Object $ hmToObj $ go (n':xs) v
   65 
   66 -- | Merge two objects, merging the values where both sides have an entry for a key rather than
   67 -- taking the first.
   68 mergeObject :: Object -> Object -> Object
   69 mergeObject o1 o2 = hmToObj $ HM.unionWith mergeValue (objToHm o1) (objToHm o2)
   70 
   71 -- | Merge two values, merging the objects using 'mergeObject'. Can't merge anything else.
   72 mergeValue :: Value -> Value -> Value
   73 mergeValue (Object o1) (Object o2) = Object $ mergeObject o1 o2
   74 mergeValue _ _                     = error "can't merge"
