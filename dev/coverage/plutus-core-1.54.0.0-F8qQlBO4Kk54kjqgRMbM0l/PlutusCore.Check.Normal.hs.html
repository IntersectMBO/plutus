    6 -- | This module makes sure types are normalized inside programs.
    7 module PlutusCore.Check.Normal
    8     ( checkProgram
    9     , checkTerm
   10     , isNormalType
   11     , NormCheckError (..)
   12     ) where
   13 
   14 import PlutusPrelude
   15 
   16 import PlutusCore.Core
   17 import PlutusCore.Error
   18 import PlutusCore.MkPlc (mkTyBuiltinOf)
   19 
   20 import Control.Monad.Except
   21 import Universe.Core (HasUniApply (matchUniApply), SomeTypeIn (..))
   22 
   23 -- | Ensure that all types in the 'Program' are normalized.
   24 checkProgram
   25     :: (HasUniApply uni, MonadError (NormCheckError tyname name uni fun ann) m)
   26     => Program tyname name uni fun ann -> m ()
   27 checkProgram (Program _ _ t) = checkTerm t
   28 
   29 -- | Ensure that all types in the 'Term' are normalized.
   30 checkTerm
   31     :: (HasUniApply uni, MonadError (NormCheckError tyname name uni fun ann) m)
   32     => Term tyname name uni fun ann -> m ()
   33 checkTerm p = liftEither $ check p
   34 
   35 check
   36     :: HasUniApply uni
   37     => Term tyname name uni fun ann -> Either (NormCheckError tyname name uni fun ann) ()
   38 check (Error _ ty)           = normalType ty
   39 check (TyInst _ t ty)        = check t >> normalType ty
   40 check (IWrap _ pat arg term) = normalType pat >> normalType arg >> check term
   41 check (Unwrap _ t)           = check t
   42 check (LamAbs _ _ ty t)      = normalType ty >> check t
   43 check (Apply _ t1 t2)        = check t1 >> check t2
   44 check (TyAbs _ _ _ t)        = check t
   45 check (Constr _ ty _ es)     = normalType ty >> traverse_ check es
   46 check (Case _ ty arg cs)     = normalType ty >> check arg >> traverse_ check cs
   47 check Var{}                  = pure ()
   48 check Constant{}             = pure ()
   49 check Builtin{}              = pure ()
   50 
   51 isNormalType :: HasUniApply uni => Type tyname uni ann -> Bool
   52 isNormalType = isRight . normalType
   53 
   54 normalType
   55     :: HasUniApply uni
   56     => Type tyname uni ann -> Either (NormCheckError tyname name uni fun ann) ()
   57 normalType (TyFun _ i o)       = normalType i >> normalType o
   58 normalType (TyForall _ _ _ ty) = normalType ty
   59 normalType (TyIFix _ pat arg)  = normalType pat >> normalType arg
   60 normalType (TySOP _ tyls)      = traverse_ (traverse_ normalType) tyls
   61 normalType (TyLam _ _ _ ty)    = normalType ty
   62 normalType ty                  = neutralType ty
   63 
   64 neutralType
   65     :: HasUniApply uni
   66     => Type tyname uni ann -> Either (NormCheckError tyname name uni fun ann) ()
   67 neutralType TyVar{}                 = pure ()
   68 neutralType (TyBuiltin ann someUni) = neutralUni ann someUni
   69 neutralType (TyApp _ ty1 ty2)       = neutralType ty1 >> normalType ty2
   70 neutralType ty                      = Left (BadType (typeAnn ty) ty "neutral type")
   71 
   72 -- See Note [Normalization of built-in types].
   73 neutralUni
   74     :: HasUniApply uni
   75     => ann -> SomeTypeIn uni -> Either (NormCheckError tyname name uni fun ann) ()
   76 neutralUni ann (SomeTypeIn uni) =
   77   matchUniApply
   78     uni
   79     -- If @uni@ is not an intra-universe application, then it's neutral.
   80     (Right ())
   81     -- If it is, then it's not neutral and we throw an error.
   82     (\_ _ -> Left (BadType ann (mkTyBuiltinOf ann uni) "neutral type"))
