   14 {- | An area bracketed by two delimiters. When on multiple lines the delimiters are not
   15 indented but the content is.
   16 -}
   17 section' :: Doc a -> Doc a -> Doc a -> Doc a
   18 -- The subtlety here is that the nest call surrounds the first delimiter and the content, but not
   19 -- the final one. This is because of how nest behaves, where it doesn't indent until it hits
   20 -- the first line break. So we need to include the first delimiter so that the main content gets
   21 -- indented, but not the final delimiter.
   22 section' c1 c2 d = group $ nest 2 (c1 <> d) <> c2
   23 
   24 {- | An area bracketed by two delimiters. When on one line, there are spaces between the delimiters
   25 and the content, when on multiple lines the delimiters are not indented but the content is.
   26 -}
   27 section :: Doc a -> Doc a -> Doc a -> Doc a
   28 section c1 c2 = section' (c1 <> line) (line <> c2)
   29 
   30 {- | This prints a document enclosed by brackets, possibly indenting the output on
   31 a new line if it does not fit.
   32 -}
   33 brackets' :: Doc a -> Doc a
   34 brackets' = section "[" "]"
   35 
   36 {- | This prints a document enclosed by braces, possibly indenting the output on
   37 a new line if it does not fit.
   38 -}
   39 braces' :: Doc a -> Doc a
   40 braces' = section "{" "}"
   41 
   42 {- | This prints a document enclosed by parentheses, aligning the opening and
   43 closing parentheses.
   44 -}
   45 parens' :: Doc a -> Doc a
   46 parens' = section "(" ")"

   48 -- | Print a "sexp", i.e. something like "(keyword arg1 ... argN)".
   49 sexp :: Doc a -> [Doc a] -> Doc a
   50 sexp a es =
   51   -- This is a bit funny, because we want the keyword to "stick" to the opening parenthesis
   52   -- when it's split over multiple lines. So we include it in the "initial" segment. But then
   53   -- we also have to have a space after that rather than no space. So we start with "(keyword"
   54   -- and a line-or-space, but end with a line-or-nothing and ")".
   55   -- However if @es@ is empty, then we don't want to insert a space right before @)@, so in that
   56   -- case we use line-or-nothing as well.
   57   section' ("(" <> a <> if null es then line' else line) (line' <> ")") (sep es)
   58 
   59 {- | Lay out a sequence of documents vertically with forced lines between documents. Useful
   60 for prettyprinting layout-sensitive things like let-bindings.
   61 -}
   62 vcatHard :: [Doc ann] -> Doc ann
   63 vcatHard = concatWith (\x y -> x <> hardline <> y)
   64 
   65 {- | Separate two documents `p` and `q` and increase indentation if `q` has to be put on a new
   66 line. Useful to e.g. pretty-print function application like `fun <?> sep arguments`.
   67 -}
   68 (<?>) :: Doc ann -> Doc ann -> Doc ann
   69 p <?> q = align . nest 2 $ sep [p, q]
