   23 newtype AstSize = AstSize
   24   { unAstSize :: Integer
   25   }
   26   deriving stock (Show)
   27   deriving newtype (Pretty, Eq, Ord, Num)
   28   deriving anyclass (PrettyBy config)

   31 {-| Count the number of AST nodes in a kind.
   32 
   33 >>> kindAstSize $ Type ()
   34 AstSize {unAstSize = 1}
   35 >>> kindAstSize $ KindArrow () (KindArrow () (Type ()) (Type ())) (Type ())
   36 AstSize {unAstSize = 5}
   37 -}
   38 kindAstSize :: Kind a -> AstSize
   39 kindAstSize kind =
   40   fold
   41     [ AstSize 1
   42     , kind ^. kindSubkinds . to kindAstSize
   43     ]
   44 
   45 -- | Count the number of AST nodes in a type.
   46 typeAstSize :: Type tyname uni ann -> AstSize
   47 typeAstSize ty =
   48   fold
   49     [ AstSize 1
   50     , ty ^. typeSubkinds . to kindAstSize
   51     , ty ^. typeSubtypes . to typeAstSize
   52     ]
   53 
   54 tyVarDeclAstSize :: TyVarDecl tyname ann -> AstSize
   55 tyVarDeclAstSize tyVarDecl =
   56   fold
   57     [ AstSize 1
   58     , tyVarDecl ^. tyVarDeclSubkinds . to kindAstSize
   59     ]
   60 
   61 -- | Count the number of AST nodes in a term.
   62 termAstSize :: Term tyname name uni fun ann -> AstSize
   63 termAstSize term =
   64   fold
   65     [ AstSize 1
   66     , term ^. termSubkinds . to kindAstSize
   67     , term ^. termSubtypes . to typeAstSize
   68     , term ^. termSubterms . to termAstSize
   69     ]
   70 
   71 varDeclAstSize :: VarDecl tyname name uni ann -> AstSize
   72 varDeclAstSize varDecl =
   73   fold
   74     [ AstSize 1
   75     , varDecl ^. varDeclSubtypes . to typeAstSize
   76     ]
   77 
   78 -- | Count the number of AST nodes in a program.
   79 programAstSize :: Program tyname name uni fun ann -> AstSize
   80 programAstSize (Program _ _ t) = termAstSize t
   81 
   82 -- | Compute the size of the serializabled form of a value.
   83 serialisedAstSize :: (Flat a) => a -> Integer
   84 serialisedAstSize = fromIntegral . BS.length . flat
