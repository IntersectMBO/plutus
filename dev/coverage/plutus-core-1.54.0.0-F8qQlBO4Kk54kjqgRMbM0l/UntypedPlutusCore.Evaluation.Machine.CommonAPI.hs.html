   72 -- The type of the machine (runner function).
   73 type MachineRunner cost uni fun ann =
   74       MachineParameters CekMachineCosts fun (CekValue uni fun ann)
   75     -> ExBudgetMode cost uni fun
   76     -> EmitterMode uni fun
   77     -> NTerm uni fun ann
   78     -> CekReport cost NamedDeBruijn uni fun
   79 
   80 {- Note [CEK runners naming convention]
   81 A function whose name ends in @NoEmit@ does not perform logging and so does not return any logs.
   82 A function whose name starts with @unsafe@ throws exceptions instead of returning them purely.
   83 A function from the @runCek@ family takes an 'ExBudgetMode' parameter and returns the final
   84 'CekExBudgetState' (and possibly logs). Note that 'runCek' is defined in @...Cek.Internal@ for
   85 reasons explained in Note [Compilation peculiarities].
   86 A function from the @evaluateCek@ family does not return the final 'ExBudgetMode', nor does it
   87 allow one to specify an 'ExBudgetMode'. I.e. such functions are only for fully evaluating programs
   88 (and possibly returning logs). See also haddocks of 'enormousBudget'.
   89 -}

   91 {-| Evaluate a term using a machine with logging enabled and keep track of costing.
   92 A wrapper around the internal runCek to debruijn input and undebruijn output.
   93 *THIS FUNCTION IS PARTIAL if the input term contains free variables*
   94 -}
   95 runCek ::
   96       MachineRunner cost uni fun ann
   97     -> MachineParameters CekMachineCosts fun (CekValue uni fun ann)
   98     -> ExBudgetMode cost uni fun
   99     -> EmitterMode uni fun
  100     -> Term Name uni fun ann
  101     -> CekReport cost Name uni fun
  102 runCek runner params mode emitMode term =
  103     -- translating input
  104     case runExcept @FreeVariableError $ deBruijnTerm term of
  105         Left fvError -> throw fvError
  106         Right dbt -> do
  107             -- Don't use 'let': https://github.com/IntersectMBO/plutus/issues/3876
  108             case runner params mode emitMode dbt of
  109                 -- translating back the output
  110                 CekReport res cost' logs ->
  111                     CekReport (mapTermCekResult gracefulUnDeBruijn res) cost' logs
  112   where
  113     -- *GRACEFULLY* undebruijnifies: a) the error-cause-term (if it exists) or b) the success
  114     -- *value-term.
  115     -- 'Graceful' means that the (a) && (b) undebruijnifications do not throw an error upon a free
  116     -- variable encounter: free debruijn indices will be turned to free, consistent uniques
  117     gracefulUnDeBruijn :: Term NamedDeBruijn uni fun () -> Term Name uni fun ()
  118     gracefulUnDeBruijn t = runQuote
  119                            . flip evalStateT mempty
  120                            $ unDeBruijnTermWith freeIndexAsConsistentLevel t

  122 -- | Evaluate a term using a machine with logging disabled and keep track of costing.
  123 -- *THIS FUNCTION IS PARTIAL if the input term contains free variables*
  124 runCekNoEmit ::
  125       MachineRunner cost uni fun ann
  126     -> MachineParameters CekMachineCosts fun (CekValue uni fun ann)
  127     -> ExBudgetMode cost uni fun
  128     -> Term Name uni fun ann
  129     -> (Either (CekEvaluationException Name uni fun) (Term Name uni fun ()), cost)
  130 runCekNoEmit runner params mode
  131     = -- throw away the logs
  132       (\(CekReport res cost _logs) -> (cekResultToEither res, cost))
  133     . runCek runner params mode noEmitter
  134 
  135 -- | Evaluate a term using a machine with logging enabled.
  136 -- *THIS FUNCTION IS PARTIAL if the input term contains free variables*
  137 evaluateCek
  138     :: ThrowableBuiltins uni fun
  139     => MachineRunner RestrictingSt uni fun ann
  140     -> EmitterMode uni fun
  141     -> MachineParameters CekMachineCosts fun (CekValue uni fun ann)
  142     -> Term Name uni fun ann
  143     -> (Either (CekEvaluationException Name uni fun) (Term Name uni fun ()), [Text])
  144 evaluateCek runner emitMode params
  145     = -- throw away the cost
  146       (\(CekReport res _cost logs) -> (cekResultToEither res, logs))
  147     . runCek runner params restrictingEnormous emitMode
  148 
  149 -- | Evaluate a term using a machine with logging disabled.
  150 -- *THIS FUNCTION IS PARTIAL if the input term contains free variables*
  151 evaluateCekNoEmit
  152     :: ThrowableBuiltins uni fun
  153     => MachineRunner RestrictingSt uni fun ann
  154     -> MachineParameters CekMachineCosts fun (CekValue uni fun ann)
  155     -> Term Name uni fun ann
  156     -> Either (CekEvaluationException Name uni fun) (Term Name uni fun ())
  157 evaluateCekNoEmit runner params = fst . runCekNoEmit runner params restrictingEnormous
