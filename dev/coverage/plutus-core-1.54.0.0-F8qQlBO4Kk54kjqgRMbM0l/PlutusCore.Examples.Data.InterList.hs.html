   20 {- Note [Example: InterList]
   21 We encode the following in this module:
   22 
   23     open import Function
   24 
   25     data InterList (A B : Set) : Set where
   26       InterNil  : InterList A B
   27       InterCons : A -> B -> InterList B A -> InterList A B
   28 
   29     foldrInterList : (A B R : Set) -> (A -> B -> R -> R) -> R -> InterList A B -> R
   30     foldrInterList A0 B0 R f0 z = go A0 B0 f0 where
   31       go : âˆ€ A B -> (A -> B -> R -> R) -> InterList A B -> R
   32       go A B f  InterNil          = z
   33       go A B f (InterCons x y xs) = f x y $ go B A (flip f) xs
   34 -}

   36 -- This definition is used as an example in Note [Spiney API] in "PlutusCore.StdLib.Type",
   37 -- so if you change it here, then also change it there.
   38 -- | @InterList@ as a PLC type.
   39 --
   40 -- > fix \(interlist :: * -> * -> *) (a :: *) (b :: *) ->
   41 -- >     all (r :: *). r -> (a -> b -> interlist b a -> r) -> r
   42 interListData :: RecursiveType uni fun ()
   43 interListData = runQuote $ do
   44     a         <- freshTyName "a"
   45     b         <- freshTyName "b"
   46     interlist <- freshTyName "interlist"
   47     r         <- freshTyName "r"
   48     let interlistBA = mkIterTyAppNoAnn (TyVar () interlist) [TyVar () b, TyVar () a]
   49     makeRecursiveType () interlist [TyVarDecl () a $ Type (), TyVarDecl () b $ Type ()]
   50         . TyForall () r (Type ())
   51         . TyFun () (TyVar () r)
   52         . TyFun () (mkIterTyFun () [TyVar () a, TyVar () b, interlistBA] $ TyVar () r)
   53         $ TyVar () r

   55 interNil :: Term TyName Name uni fun ()
   56 interNil = runQuote $ do
   57     let RecursiveType interlist wrapInterList = interListData
   58     a <- freshTyName "a"
   59     b <- freshTyName "b"
   60     r <- freshTyName "r"
   61     z <- freshName "z"
   62     f <- freshName "f"
   63     let interlistBA = mkIterTyAppNoAnn interlist [TyVar () b, TyVar () a]
   64     return
   65         . TyAbs () a (Type ())
   66         . TyAbs () b (Type ())

   70         . LamAbs () f (mkIterTyFun () [TyVar () a, TyVar () b, interlistBA] $ TyVar () r)

  101 foldrInterList :: uni `HasTypeAndTermLevel` () => Term TyName Name uni fun ()
  102 foldrInterList = runQuote $ do
  103     let interlist = _recursiveType interListData
  104     a0  <- freshTyName "a0"
  105     b0  <- freshTyName "b0"
  106     r   <- freshTyName "r"
  107     f   <- freshName "f"
  108     z   <- freshName "z"
  109     rec <- freshName "rec"
  110     u   <- freshName "u"
  111     a   <- freshTyName "a"
  112     b   <- freshTyName "b"
  113     f'  <- freshName "f'"
  114     xs  <- freshName "xs"
  115     x   <- freshName "x"
  116     y   <- freshName "y"
  117     xs' <- freshName "xs'"
  118     x'  <- freshName "x'"
  119     y'  <- freshName "y'"
  120     let interlistOf a' b' = mkIterTyAppNoAnn interlist [TyVar () a', TyVar () b']
  121         fTy a' b' = mkIterTyFun () [TyVar () a', TyVar () b', TyVar () r] $ TyVar () r
  122         fixTyArg2
  123             = TyForall () a (Type ())
  124             . TyForall () b (Type ())
  125             . TyFun () (fTy a b)
  126             . TyFun () (interlistOf a b)
  127             $ TyVar () r
  128         instedFix = mkIterInstNoAnn fix [unit, fixTyArg2]
  129         unwrappedXs = TyInst () (Unwrap () (Var () xs)) $ TyVar () r
  130         instedRec = mkIterInstNoAnn (Apply () (Var () rec) unitval) [TyVar () b, TyVar () a]
  131     return
  132         . TyAbs () a0 (Type ())
  133         . TyAbs () b0 (Type ())
  134         . TyAbs () r (Type ())
  135         . LamAbs () f (fTy a0 b0)
  136         . LamAbs () z (TyVar () r)
  137         $ mkIterInstNoAnn
  138             ( mkIterAppNoAnn instedFix
  139                 [   LamAbs () rec (TyFun () unit fixTyArg2)
  140                   . LamAbs () u unit
  141                   . TyAbs () a (Type ())
  142                   . TyAbs () b (Type ())
  143                   . LamAbs () f' (fTy a b)
  144                   . LamAbs () xs (interlistOf a b)
  145                   $ mkIterAppNoAnn unwrappedXs
  146                       [ Var () z
  147                       ,    LamAbs () x (TyVar () a)
  148                          . LamAbs () y (TyVar () b)
  149                          . LamAbs () xs' (interlistOf b a)
  150                          $ mkIterAppNoAnn (Var () f')
  151                              [ Var () x
  152                              , Var () y
  153                              , mkIterAppNoAnn instedRec
  154                                  [    LamAbs () y' (TyVar () b)
  155                                     . LamAbs () x' (TyVar () a)
  156                                     $ mkIterAppNoAnn (Var () f')
  157                                         [ Var () x'
  158                                         , Var () y'
  159                                         ]
  160                                  , Var () xs'
  161                                  ]
  162                              ]
  163                       ]
  164                 , unitval
  165                 ]
  166             )
  167             [ TyVar () a0
  168             , TyVar () b0
