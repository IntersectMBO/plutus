   13 -- | Marks all the 'Unique's in a term as used, so they will not be generated in future. Useful if you
   14 -- have a term which was not generated in 'Quote'.
   15 markNonFreshTerm
   16     :: (HasUniques (Term name uni fun ann), MonadQuote m)
   17     => Term name uni fun ann -> m ()
   18 markNonFreshTerm = markNonFreshMax . setOf termUniquesDeep
   19 
   20 -- | Marks all the 'Unique's in a program as used, so they will not be generated in future. Useful if you
   21 -- have a program which was not generated in 'Quote'.
   22 markNonFreshProgram
   23     :: (HasUnique name TermUnique, MonadQuote m)
   24     => Program name uni fun ann
   25     -> m ()
   26 markNonFreshProgram (Program _ _ body) = markNonFreshTerm body
