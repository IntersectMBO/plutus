  146 -- | Applies one program to another. Fails if the versions do not match
  147 -- and tries to merge annotations.
  148 applyProgram
  149     :: (MonadError ApplyProgramError m, Semigroup a)
  150     => Program tyname name uni fun a
  151     -> Program tyname name uni fun a
  152     -> m (Program tyname name uni fun a)
  153 applyProgram (Program a1 v1 t1) (Program a2 v2 t2) | v1 == v2
  154   = pure $ Program (a1 <> a2) v1 (Apply (termAnn t1 <> termAnn t2) t1 t2)
  155 applyProgram (Program _a1 v1 _t1) (Program _a2 v2 _t2) =
  156     throwError $ MkApplyProgramError v1 v2
