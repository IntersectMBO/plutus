   44 -- | Lifts an 'Either' into an error context where we can embed the 'Left' value into the error.
   45 throwingEither :: MonadError e m => AReview e t -> Either t a -> m a
   46 throwingEither r e = case e of
   47     Left t  -> throwing r t
   48     Right v -> pure v

   91 -- | This is needed for nice kind/type checking error messages. In some cases the type checker knows
   92 -- the exact type that an expression has to have for type checking to succeed (see any of
   93 -- 'checkTypeM' functions and its usages), which is what 'ExpectedExact' is suitable for. In other
   94 -- cases the type checker only cares about the shape of the inferred type, e.g. the type checker
   95 -- knows that the type of a function has to be @dom -> cod@ for type checking to succeed, but it
   96 -- doesn't yet care what @dom@ and @cod@ exactly are. Which is what 'ExpectedShape' is useful for as
   97 -- it allows one to specify the shape of an expected type with some existential variables in it when
   98 -- it's impossible to provide an exact type.
   99 data ExpectedShapeOr a
  100     = ExpectedShape
  101         !T.Text
  102         -- ^ The expected shape potentially referencing existential variables.
  103         ![T.Text]
  104         -- ^ The list of existential variables.
  105     | ExpectedExact !a
  106     deriving stock (Show, Eq, Generic, Functor)
  107     deriving anyclass (NFData)
  108 
  109 data TypeError term uni fun ann
  110     = KindMismatch !ann
  111         !(Type TyName uni ())
  112         !(ExpectedShapeOr (Kind ()))
  113         -- ^ The expected type or the shape of a kind.
  114         !(Kind ())
  115         -- ^ The actual kind.
  116     | TypeMismatch !ann
  117         !term
  118         !(ExpectedShapeOr (Type TyName uni ()))
  119         -- ^ The expected type or the shape of a type.
  120         !(Normalized (Type TyName uni ()))
  121         -- ^ The actual type.
  122     | TyNameMismatch !ann !TyName !TyName
  123     | NameMismatch !ann !Name !Name
  124     | FreeTypeVariableE !ann !TyName
  125     | FreeVariableE !ann !Name
  126     | UnknownBuiltinFunctionE !ann !fun
  127     | UnsupportedCaseBuiltin !ann !T.Text
  128     deriving stock (Show, Eq, Generic, Functor)
  129     deriving anyclass (NFData)

  170         "Expected a type of kind star (to later parse a constant), but got:" <+>

  174             "." <> hardline <> "Parsable functions are " <+> pretty lBuiltin
  175     pretty (InvalidBuiltinConstant c s loc) =
  176         "Invalid constant" <+> squotes (pretty c) <+> "of type" <+> squotes (pretty s) <+> "at" <+>
  177             pretty loc

  205 -- | Align a list of existential variables in a pretty way.
  206 --
  207 -- >>> :set -XOverloadedStrings
  208 -- >>> existentialVars []
  209 -- >>> existentialVars ["'a'"]
  210 --  for some 'a'
  211 -- >>> existentialVars ["'a'", "'b'"]
  212 --  for some 'a' and 'b'
  213 -- >>> existentialVars ["'a'", "'b'", "'c'"]
  214 --  for some 'a', 'b' and 'c'
  215 existentialVars :: [Doc ann] -> Doc ann
  216 existentialVars [] = ""
  217 existentialVars (x0:xs0) = " for some " <> go x0 xs0 where
  218     go x []     = x
  219     go x [y]    = x <> " and " <> y
  220     go x (y:xs) = x <> ", " <> go y xs
  221 
  222 instance PrettyBy PrettyConfigPlc a => PrettyBy PrettyConfigPlc (ExpectedShapeOr a) where
  223     prettyBy _ (ExpectedShape shape vars) =
  224         squotes (sexp (pretty shape) []) <> existentialVars (map (squotes . pretty) vars)
  225     prettyBy config (ExpectedExact thing) = squotes (prettyBy config thing)

  227 instance (Pretty term, PrettyUni uni, Pretty fun, Pretty ann) =>
  228         PrettyBy PrettyConfigPlc (TypeError term uni fun ann) where
  229     prettyBy config (KindMismatch ann ty shapeOrK k') =
  230         "Kind mismatch at" <+> pretty ann <>
  231         hardline <>
  232         "Expected a type of kind" <> hardline <> indent 2 (prettyBy config shapeOrK) <>
  233         hardline <>
  234         "But found one of kind" <> hardline <> indent 2 (squotes (prettyBy config k')) <>
  235         hardline <>
  236         "Namely," <> hardline <> indent 2 (squotes (prettyBy config ty))
  237     prettyBy config (TypeMismatch ann t shapeOrTy ty')         =
  238         "Type mismatch at" <+> pretty ann <>
  239         hardline <>
  240         "Expected a term of type" <> hardline <> indent 2 (prettyBy config shapeOrTy) <>
  241         hardline <>
  242         "But found one of type" <> hardline <> indent 2 (squotes (prettyBy config ty')) <>
  243         (if _pcpoCondensedErrors (_pcpOptions config) == CondensedErrorsYes
  244             then mempty
  245             -- TODO: we should use prettyBy here but the problem is
  246             -- that `instance PrettyClassic PIR.Term` whereas `instance PrettyPLC PLC.Term`
  247             else hardline <> "Namely," <> hardline <> indent 2 (squotes (pretty t)))
  248     prettyBy config (FreeTypeVariableE ann name)          =
  249         "Free type variable at " <+> pretty ann <+> ": " <+> prettyBy config name
  250     prettyBy config (FreeVariableE ann name)              =
  251         "Free variable at " <+> pretty ann <+> ": " <+> prettyBy config name
  252     prettyBy _ (UnknownBuiltinFunctionE ann fun) =
  253         "An unknown built-in function at" <+> pretty ann <> ":" <+> pretty fun
  254     prettyBy _ (TyNameMismatch ann name1 name2) = hsep
  255         [ "Type-level name mismatch at"
  256         , pretty ann <> ":"
  257         , pretty $ name1 ^. theText
  258         , "is in scope, but"
  259         , pretty $ name2 ^. theText
  260         , "having the same Unique"
  261         , pretty $ name1 ^. theUnique
  262         , "is attempted to be referenced"
  263         ]
  264     prettyBy _ (NameMismatch ann name1 name2) = hsep
  265         [ "Term-level name mismatch at"
  266         , pretty ann <> ":"
  267         , pretty $ name1 ^. theText
  268         , "is in scope, but"
  269         , pretty $ name2 ^. theText
  270         , "having the same Unique"
  271         , pretty $ name1 ^. theUnique
  272         , "is attempted to be referenced"
  273         ]
  274     prettyBy _ (UnsupportedCaseBuiltin ann err) = hsep
  275         [ "Unsupported 'case' of a value of a built-in type at"
  276         , pretty ann <> ":"
  277         , hardline
  278         , pretty err
  279         ]
