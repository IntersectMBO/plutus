   48 -- | Construct an 'ExBudgetMode' out of a function returning a value of the budgeting state type.
   49 -- The value then gets added to the current state via @(<>)@.
   50 monoidalBudgeting
   51     :: Monoid cost => (ExBudgetCategory fun -> ExBudget -> cost) -> ExBudgetMode cost uni fun
   52 monoidalBudgeting toCost = ExBudgetMode $ do
   53     costRef <- newSTRef mempty
   54     budgetRef <- newSTRef mempty
   55     let spend key budgetToSpend = CekM $ do
   56             modifySTRef' costRef (<> toCost key budgetToSpend)
   57             modifySTRef' budgetRef (<> budgetToSpend)
   58         spender = CekBudgetSpender spend
   59         cumulative = readSTRef budgetRef
   60         final = readSTRef costRef
   61     pure $ ExBudgetInfo spender final cumulative

   63 -- | For calculating the cost of execution by counting up using the 'Monoid' instance of 'ExBudget'.
   64 newtype CountingSt = CountingSt ExBudget
   65     deriving stock (Eq, Show)
   66     deriving newtype (Semigroup, Monoid, PrettyBy config, NFData)
   67 
   68 instance Pretty CountingSt where
   69     pretty (CountingSt budget) = parens $ "required budget:" <+> pretty budget <> line

   71 -- | For calculating the cost of execution.
   72 counting :: ExBudgetMode CountingSt uni fun
   73 counting = monoidalBudgeting $ const CountingSt

   75 -- | For a detailed report on what costs how much + the same overall budget that 'Counting' gives.
   76 -- The (derived) 'Monoid' instance of 'CekExTally' is the main piece of the machinery.
   77 newtype CekExTally fun = CekExTally (MonoidalHashMap (ExBudgetCategory fun) ExBudget)
   78     deriving stock (Eq, Generic, Show)
   79     deriving (Semigroup, Monoid) via (GenericSemigroupMonoid (CekExTally fun))
   80     deriving anyclass (NFData)
   81     deriving (PrettyBy config) via (IgnorePrettyConfig (CekExTally fun))
   82 
   83 instance (Show fun, Ord fun) => Pretty (CekExTally fun) where
   84     pretty (CekExTally m) =
   85         let om = Map.fromList $ HashMap.toList m
   86         in parens $ encloseSep "{" "}" "| " $ fmap group $
   87           Map.elems $ imap (\k v -> (pretty k <+> "causes" <+> group (pretty v))) om
   88 
   89 data TallyingSt fun = TallyingSt (CekExTally fun) ExBudget
   90     deriving stock (Eq, Show, Generic)
   91     deriving (Semigroup, Monoid) via (GenericSemigroupMonoid (TallyingSt fun))
   92     deriving anyclass (NFData)
   93     deriving (PrettyBy config) via (IgnorePrettyConfig (TallyingSt fun))
   94 
   95 instance (Show fun, Ord fun) => Pretty (TallyingSt fun) where
   96     pretty (TallyingSt tally budget) = parens $ fold
   97         [ "{ tally: ", pretty tally, line
   98         , "| budget: ", pretty budget, line
   99         , "}"
  100         ]

  102 -- | For a detailed report on what costs how much + the same overall budget that 'Counting' gives.
  103 tallying :: (Hashable fun) => ExBudgetMode (TallyingSt fun) uni fun
  104 tallying =
  105     monoidalBudgeting $ \key budgetToSpend ->
  106         TallyingSt (CekExTally $ singleton key budgetToSpend) budgetToSpend

  116 -- | For execution, to avoid overruns.
  117 restricting
  118     :: ThrowableBuiltins uni fun
  119     => ExRestrictingBudget -> ExBudgetMode RestrictingSt uni fun
  120 restricting (ExRestrictingBudget initB@(ExBudget cpuInit memInit)) = ExBudgetMode $ do
  121     -- We keep the counters in a PrimArray. This is better than an STRef since it stores its contents unboxed.
  122     --
  123     -- If we don't specify the element type then GHC has difficulty inferring it, but it's
  124     -- annoying to specify the monad, since it refers to the 's' which is not in scope.
  125     ref <- newPrimArray @_ @SatInt 2
  126     let
  127         cpuIx = 0
  128         memIx = 1
  129         readCpu = coerce @_ @ExCPU <$> readPrimArray ref cpuIx
  130         writeCpu cpu = writePrimArray ref cpuIx $ coerce cpu
  131         readMem = coerce @_ @ExMemory <$> readPrimArray ref memIx
  132         writeMem mem = writePrimArray ref memIx $ coerce mem
  133 
  134     writeCpu cpuInit
  135     writeMem memInit
  136     let
  137         spend _ (ExBudget cpuToSpend memToSpend) = do
  138             cpuLeft <- CekM readCpu
  139             memLeft <- CekM readMem
  140             let cpuLeft' = cpuLeft - cpuToSpend
  141             let memLeft' = memLeft - memToSpend
  142             -- Note that even if we throw an out-of-budget error, we still need to record
  143             -- what the final state was.
  144             CekM $ writeCpu cpuLeft'
  145             CekM $ writeMem memLeft'
  146             when (cpuLeft' < 0 || memLeft' < 0) $ do
  147                 let -- You'd think whether the budget is computed strictly or not before throwing
  148                     -- an error isn't important, but GHC refuses to unbox the second argument of
  149                     -- @spend@ without this bang. Bangs on @cpuLeft'@ and @memLeft'@ don't help
  150                     -- either as those are forced by 'writeCpu' and 'writeMem' anyway. Go figure.
  151                     !budgetLeft = ExBudget cpuLeft' memLeft'
  152                 throwError $
  153                   ErrorWithCause
  154                     (OperationalError (CekOutOfExError $ ExRestrictingBudget budgetLeft))
  155                     Nothing
  156         spender = CekBudgetSpender spend
  157         remaining = ExBudget <$> readCpu <*> readMem
  158         cumulative = do
  159             r <- remaining
  160             pure $ initB `minusExBudget` r
  161         final = RestrictingSt . ExRestrictingBudget <$> remaining
  162     pure $ ExBudgetInfo spender final cumulative
