   49 -- | The state contains the "next" 'Unique' that should be used for a name
   50 type FreshState = Unique
   51 
   52 emptyFreshState :: FreshState
   53 emptyFreshState = Unique 0

   55 -- | The "quotation" monad transformer. Within this monad you can do safe construction of PLC
   56 -- terms using quasiquotation, fresh-name generation, and parsing.
   57 newtype QuoteT m a = QuoteT {unQuoteT :: StateT FreshState m a}
   58     deriving newtype
   59         ( Functor
   60         , Applicative
   61         , Monad
   62         , MonadTrans
   63         , MonadFix
   64         , MM.MFunctor
   65         , MonadError e
   66         , MonadReader r
   67         , MonadIO
   68         , MonadWriter w
   69         )

   71 -- Need to write this by hand, deriving wants to derive the one for DefState
   72 instance MonadState s m => MonadState s (QuoteT m) where
   73     get = lift get
   74     put = lift . put
   75     state = lift . state

   77 -- | A monad that allows lifting of quoted expressions.
   78 class Monad m => MonadQuote m where
   79     liftQuote :: Quote a -> m a
   80     -- This means we don't have to implement it when we're writing an instance for a MonadTrans
   81     -- monad. We can't just add an instance declaration for that, because it overlaps with the
   82     -- base instance.
   83     default liftQuote :: (MonadQuote n, MonadTrans t, t n ~ m) => Quote a -> m a
   84     liftQuote = lift . liftQuote

   86 instance (Monad m) => MonadQuote (QuoteT m) where
   87     liftQuote = MM.hoist (pure . runIdentity)

   96 -- | Run a quote from an empty identifier state. Note that the resulting term cannot necessarily
   97 -- be safely combined with other terms - that should happen inside 'QuoteT'.
   98 runQuoteT ::  Monad m => QuoteT m a -> m a
   99 runQuoteT q = evalStateT (unQuoteT q) emptyFreshState

  101 -- | A non-transformer version of 'QuoteT'.
  102 type Quote = QuoteT Identity

  104 -- | See 'runQuoteT'.
  105 runQuote :: Quote a -> a
  106 runQuote = runIdentity . runQuoteT

  108 -- | Get a fresh 'Unique'.
  109 freshUnique :: MonadQuote m => m Unique
  110 freshUnique = liftQuote $ do
  111     nextU <- QuoteT get
  112     QuoteT $ put $ Unique (unUnique nextU + 1)
  113     pure nextU
  114 
  115 -- | Get a fresh 'Name', given the annotation and the 'Text.Text' name.
  116 freshName :: MonadQuote m => Text.Text -> m Name
  117 freshName str = Name str <$> freshUnique
  118 
  119 -- | Make a copy of the given 'Name' that is distinct from the old one.
  120 freshenName :: MonadQuote m => Name -> m Name
  121 freshenName (Name str _) = Name str <$> freshUnique
  122 
  123 -- | Get a fresh 'TyName', given the annotation and the 'Text.Text' name.
  124 freshTyName :: MonadQuote m => Text.Text -> m TyName
  125 freshTyName = fmap TyName <$> freshName
  126 
  127 -- | Make a copy of the given 'TyName' that is distinct from the old one.
  128 freshenTyName :: MonadQuote m => TyName -> m TyName
  129 freshenTyName (TyName name) = TyName <$> freshenName name

  131 -- | Mark all 'Unique's less than the given 'Unique' as used, so they will not be generated
  132 -- in future.
  133 markNonFreshBelow :: MonadQuote m => Unique -> m ()
  134 markNonFreshBelow = liftQuote . QuoteT . modify . max
  135 
  136 -- | Mark a given 'Unique' (and implicitly all 'Unique's less than it) as used, so they will
  137 -- not be generated in future.
  138 markNonFresh :: MonadQuote m => Unique -> m ()
  139 markNonFresh = markNonFreshBelow . succ
  140 
  141 -- | Mark the maximal 'Unique' from a set of 'Unique's (and implicitly all 'Unique's less than it)
  142 -- as used, so they will not be generated in future.
  143 markNonFreshMax :: MonadQuote m => Set Unique -> m ()
  144 markNonFreshMax = markNonFresh . fromMaybe (Unique 0) . Set.lookupMax
