   25 instance (GEq uni, Eq ann) => Eq (Type TyName uni ann) where
   26     ty1 == ty2 = runEqRename @TypeRenaming $ eqTypeM ty1 ty2

   31     term1 == term2 = runEqRename $ eqTermM term1 term2

   33 -- Simple Structural Equality of a `Term NamedDeBruijn`. This implies three things:
   34 -- b) We do not do equality "modulo starting index". E.g. `LamAbs 0 (Var 0) /= LamAbs 1 (Var 1)`.
   35 -- c) We do not do equality ""modulo annotations".
   36 -- Note that we ignore the name part in case of the nameddebruijn
   37 -- If a user wants to ignore annotations he must prior do `void <$> term`, to throw away any annotations.
   38 deriving stock instance
   39    (GEq uni, Closed uni, uni `Everywhere` Eq, Eq fun, Eq ann) =>
   40    Eq (Term NamedTyDeBruijn NamedDeBruijn uni fun ann)

   58 type EqRenameOf ren a = HasUniques a => a -> a -> EqRename ren
   59 
   60 {- Note [No catch-all]
   61 Catch-all clauses like @f _x = <...>@ have the disadvantage that if somebody adds a new constructor
   62 to the definition of the type of @_x@, then GHC will not warn us about @f@ potentially no longer
   63 being defined correctly. This is especially pronounced for equality checking functions, having
   64 
   65     _ == _ = False
   66 
   67 as the last clause of '(==)' makes it much trickier to spot that a new clause needs to be added if
   68 the type of the arguments get extended with a new constructor. For this reason in equality checking
   69 functions we always match on the first argument exhaustively even if it means having a wall of
   70 
   71     C1 == _ = False
   72     C2 == _ = False
   73     <...>
   74     Cn == _ = False
   75 
   76 This way we get a warning from GHC about non-exhaustive pattern matching if the type of the
   77 arguments gets extended with additional constructors.
   78 -}

   84 -- | Check equality of two 'Type's.
   85 eqTypeM :: (HasRenaming ren TypeUnique, GEq uni, Eq ann) => EqRenameOf ren (Type tyname uni ann)
   86 eqTypeM (TyVar ann1 name1) (TyVar ann2 name2) = do
   87     eqM ann1 ann2
   88     eqNameM name1 name2
   89 eqTypeM (TyLam ann1 name1 kind1 ty1) (TyLam ann2 name2 kind2 ty2) = do
   90     eqM ann1 ann2
   91     eqM kind1 kind2
   92     withTwinBindings name1 name2 $ eqTypeM ty1 ty2
   93 eqTypeM (TyForall ann1 name1 kind1 ty1) (TyForall ann2 name2 kind2 ty2) = do
   94     eqM ann1 ann2
   95     eqM kind1 kind2
   96     withTwinBindings name1 name2 $ eqTypeM ty1 ty2
   97 eqTypeM (TyIFix ann1 pat1 arg1) (TyIFix ann2 pat2 arg2) = do
   98     eqM ann1 ann2
   99     eqTypeM pat1 pat2
  100     eqTypeM arg1 arg2
  101 eqTypeM (TyApp ann1 fun1 arg1) (TyApp ann2 fun2 arg2) = do
  102     eqM ann1 ann2
  103     eqTypeM fun1 fun2
  104     eqTypeM arg1 arg2
  105 eqTypeM (TyFun ann1 dom1 cod1) (TyFun ann2 dom2 cod2) = do
  106     eqM ann1 ann2
  107     eqTypeM dom1 dom2
  108     eqTypeM cod1 cod2
  109 eqTypeM (TyBuiltin ann1 someUni1) (TyBuiltin ann2 someUni2) = do
  110     eqM ann1 ann2
  111     eqM someUni1 someUni2
  112 eqTypeM (TySOP ann1 tyls1) (TySOP ann2 tyls2) = do
  113     eqM ann1 ann2
  114     case zipExact tyls1 tyls2 of
  115         Just ps -> for_ ps $ \(ptys1, ptys2) -> case zipExact ptys1 ptys2 of
  116           Just tys -> for_ tys $ \(ty1, ty2) -> eqTypeM ty1 ty2
  117           Nothing  -> empty
  118         Nothing -> empty
  119 eqTypeM TyVar{}     _ = empty
  120 eqTypeM TyLam{}     _ = empty
  121 eqTypeM TyForall{}  _ = empty
  122 eqTypeM TyIFix{}    _ = empty
  123 eqTypeM TyApp{}     _ = empty
  124 eqTypeM TyFun{}     _ = empty
  125 eqTypeM TyBuiltin{} _ = empty
  126 eqTypeM TySOP{} _ = empty

  132 -- | Check equality of two 'Term's.
  133 eqTermM
  134     :: (GEq uni, Closed uni, uni `Everywhere` Eq, Eq fun, Eq ann)
  135     => EqRenameOf ScopedRenaming (Term tyname name uni fun ann)
  136 eqTermM (LamAbs ann1 name1 ty1 body1) (LamAbs ann2 name2 ty2 body2) = do
  137     eqM ann1 ann2
  138     eqTypeM ty1 ty2
  139     withTwinBindings name1 name2 $ eqTermM body1 body2
  140 eqTermM (TyAbs ann1 name1 kind1 body1) (TyAbs ann2 name2 kind2 body2) = do
  141     eqM ann1 ann2
  142     eqM kind1 kind2
  143     withTwinBindings name1 name2 $ eqTermM body1 body2
  144 eqTermM (IWrap ann1 pat1 arg1 term1) (IWrap ann2 pat2 arg2 term2) = do
  145     eqM ann1 ann2
  146     eqTypeM pat1 pat2
  147     eqTypeM arg1 arg2
  148     eqTermM term1 term2
  149 eqTermM (Apply ann1 fun1 arg1) (Apply ann2 fun2 arg2) = do
  150     eqM ann1 ann2
  151     eqTermM fun1 fun2
  152     eqTermM arg1 arg2
  153 eqTermM (Unwrap ann1 term1) (Unwrap ann2 term2) = do
  154     eqM ann1 ann2
  155     eqTermM term1 term2
  156 eqTermM (Error ann1 ty1) (Error ann2 ty2) = do
  157     eqM ann1 ann2
  158     eqTypeM ty1 ty2
  159 eqTermM (TyInst ann1 term1 ty1) (TyInst ann2 term2 ty2) = do
  160     eqM ann1 ann2
  161     eqTermM term1 term2
  162     eqTypeM ty1 ty2
  163 eqTermM (Var ann1 name1) (Var ann2 name2) = do
  164     eqM ann1 ann2
  165     eqNameM name1 name2
  166 eqTermM (Constant ann1 con1) (Constant ann2 con2) = do
  167     eqM ann1 ann2
  168     eqM con1 con2
  169 eqTermM (Builtin ann1 bi1) (Builtin ann2 bi2) = do
  170     eqM ann1 ann2
  171     eqM bi1 bi2
  172 eqTermM (Constr ann1 ty1 i1 args1) (Constr ann2 ty2 i2 args2) = do
  173     eqM ann1 ann2
  174     eqTypeM ty1 ty2
  175     eqM i1 i2
  176     case zipExact args1 args2 of
  177         Just ps -> for_ ps $ \(t1, t2) -> eqTermM t1 t2
  178         Nothing -> empty
  179 eqTermM (Case ann1 ty1 a1 cs1) (Case ann2 ty2 a2 cs2) = do
  180     eqM ann1 ann2
  181     eqTypeM ty1 ty2
  182     eqTermM a1 a2
  183     case zipExact cs1 cs2 of
  184         Just ps -> for_ ps $ \(t1, t2) -> eqTermM t1 t2
  185         Nothing -> empty
  186 eqTermM LamAbs{}   _ = empty
  187 eqTermM TyAbs{}    _ = empty
  188 eqTermM IWrap{}    _ = empty
  189 eqTermM Apply{}    _ = empty
  190 eqTermM Unwrap{}   _ = empty
  191 eqTermM Error{}    _ = empty
  192 eqTermM TyInst{}   _ = empty
  193 eqTermM Var{}      _ = empty
  194 eqTermM Constant{} _ = empty
  195 eqTermM Builtin{}  _ = empty
  196 eqTermM Constr{}  _ = empty
  197 eqTermM Case{}  _ = empty
