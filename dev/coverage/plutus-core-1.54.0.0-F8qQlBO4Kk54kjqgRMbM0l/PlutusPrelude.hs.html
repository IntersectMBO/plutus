  147 -- | A newtype wrapper around @a@ whose point is to provide a 'Show' instance
  148 -- for anything that has a 'Pretty' instance.
  149 newtype ShowPretty a = ShowPretty
  150     { unShowPretty :: a
  151     } deriving stock (Eq)

  153 instance Pretty a => Show (ShowPretty a) where
  154     show = display . unShowPretty

  156 instance (Pretty a, Pretty b) => Pretty (Either a b) where
  157     pretty (Left  x) = parens ("Left"  <+> pretty x)
  158     pretty (Right y) = parens ("Right" <+> pretty y)

  168 -- | Coerce the second argument to the result type of the first one. The motivation for this
  169 -- function is that it's often more annoying to explicitly specify a target type for 'coerce' than
  170 -- to construct an explicit coercion function, so this combinator can be used in cases like that.
  171 -- Plus the code reads better, as it becomes clear what and where gets wrapped/unwrapped.
  172 coerceVia :: Coercible a b => (a -> b) -> a -> b
  173 coerceVia _ = coerce
  174 {-# INLINE coerceVia #-}
  175 
  176 -- | Same as @\f -> f . coerce@, but does not create any closures and so is completely free.
  177 coerceArg :: Coercible a b => (a -> s) -> b -> s
  178 coerceArg = coerce
  179 {-# INLINE coerceArg #-}

  181 -- | Same as @\f -> coerce . f@, but does not create any closures and so is completely free.
  182 coerceRes :: Coercible s t => (a -> s) -> a -> t
  183 coerceRes = coerce

  192 (<<$>>) :: (Functor f1, Functor f2) => (a -> b) -> f1 (f2 a) -> f1 (f2 b)
  193 (<<$>>) = fmap . fmap

  195 (<<*>>) :: (Applicative f1, Applicative f2) => f1 (f2 (a -> b)) -> f1 (f2 a) -> f1 (f2 b)
  196 (<<*>>) = liftA2 (<*>)

  198 -- | Makes an effectful function ignore its result value and return its input value.
  199 through :: Functor f => (a -> f b) -> (a -> f a)
  200 through f x = f x $> x

  202 forJoin :: (Monad m, Traversable m, Applicative f) => m a -> (a -> f (m b)) -> f (m b)
  203 forJoin a f = join <$> for a f

  205 -- | Fold a monadic function over a 'Foldable'. The monadic version of 'foldMap'.
  206 foldMapM :: (Foldable f, Monad m, Monoid b) => (a -> m b) -> f a -> m b
  207 foldMapM f xs = foldr step return xs mempty where
  208     step x r z = f x >>= \y -> r $! z `mappend` y

  210 -- | This function generalizes 'eitherToMaybe', 'eitherToList',
  211 -- 'listToMaybe' and other such functions.
  212 reoption :: (Foldable f, Alternative g) => f a -> g a
  213 reoption = foldr (const . pure) empty

  215 -- | Basically a @Data.Functor.Representable@ instance for 'Array'.
  216 -- We can't provide an actual instance because of the @Distributive@ superclass: @Array i@ is not
  217 -- @Distributive@ unless we assume that indices in an array range over the entirety of @i@.
  218 tabulateArray :: (Bounded i, Enum i, Ix i) => (i -> a) -> Array i a
  219 tabulateArray f = listArray (minBound, maxBound) $ map f enumerate

  228 -- | @b ? x@ is equal to @pure x@ whenever @b@ holds and is 'empty' otherwise.
  229 (?) :: Alternative f => Bool -> a -> f a
  230 (?) b x = x <$ guard b
  231 
  232 -- | @ensure p x@ is equal to @pure x@ whenever @p x@ holds and is 'empty' otherwise.
  233 ensure :: Alternative f => (a -> Bool) -> a -> f a
  234 ensure p x = p x ? x

  252 -- | Compose two folds to make them run in parallel. The results are concatenated.
  253 (<^>) :: Fold s a -> Fold s a -> Fold s a
  254 (f1 <^> f2) g s = f1 g s *> f2 g s

  256 -- | Zips two lists of the same length together, returning 'Nothing' if they are not
  257 -- the same length.
  258 zipExact :: [a] -> [b] -> Maybe [(a,b)]
  259 zipExact [] []         = Just []
  260 zipExact (a:as) (b:bs) = (:) (a, b) <$> zipExact as bs
  261 zipExact _ _           = Nothing

  263 -- | Similar to Maybe's `fromJust`. Returns the `Right` and errors out with the show instance
  264 -- of the `Left`.
  265 unsafeFromRight :: (Show e) => Either e a -> a
  266 unsafeFromRight (Right a) = a
  267 unsafeFromRight (Left e)  = error $ show e

  269 -- | function recursively applied N times
  270 timesA :: Natural -> (a -> a) -> a -> a

  273 -- | Pair each element of the given list with all the other elements.
  274 --
  275 -- >>> addTheRest "abcd"
  276 -- [('a',"bcd"),('b',"acd"),('c',"abd"),('d',"abc")]
  277 addTheRest :: [a] -> [(a, [a])]
  278 addTheRest []     = []
  279 addTheRest (x:xs) = (x, xs) : map (fmap (x :)) (addTheRest xs)

  281 allSame :: Eq a => [a] -> Bool
  282 allSame []     = True
  283 allSame (x:xs) = all (x ==) xs
  284 
  285 distinct :: Eq a => [a] -> Bool
  286 distinct = not . allSame
