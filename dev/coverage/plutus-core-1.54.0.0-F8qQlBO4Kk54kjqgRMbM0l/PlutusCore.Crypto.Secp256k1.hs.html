   18 -- | Verify an ECDSA signature made using the SECP256k1 curve.
   19 --
   20 -- = Note
   21 --
   22 -- There are additional well-formation requirements for the arguments beyond
   23 -- their length:
   24 --
   25 -- * The first byte of the public key must correspond to the sign of the /y/
   26 -- coordinate: this is @0x02@ if /y/ is even, and @0x03@ otherwise.
   27 -- * The remaining bytes of the public key must correspond to the /x/
   28 -- coordinate, as a big-endian integer.
   29 -- * The first 32 bytes of the signature must correspond to the big-endian
   30 -- integer representation of _r_.
   31 -- * The last 32 bytes of the signature must correspond to the big-endian
   32 -- integer representation of _s_.
   33 --
   34 -- While this primitive /accepts/ a hash, any caller should only pass it hashes
   35 -- that they computed themselves: specifically, they should receive the
   36 -- /message/ from a sender and hash it, rather than receiving the /hash/ from
   37 -- said sender. Failure to do so can be
   38 -- [dangerous](https://bitcoin.stackexchange.com/a/81116/35586). Other than
   39 -- length, we make no requirements of what hash gets used.
   40 verifyEcdsaSecp256k1Signature
   41   :: BS.ByteString -- ^ Public key   (33 bytes)
   42   -> BS.ByteString -- ^ Message hash (32 bytes)
   43   -> BS.ByteString -- ^ Signature    (64 bytes)
   44   -> BuiltinResult Bool
   45 verifyEcdsaSecp256k1Signature pk msg sig =
   46   case DSIGN.rawDeserialiseVerKeyDSIGN @EcdsaSecp256k1DSIGN pk of
   47     Nothing -> failWithMessage loc "Invalid verification key."
   48     Just pk' -> case DSIGN.rawDeserialiseSigDSIGN @EcdsaSecp256k1DSIGN sig of
   49       Nothing -> failWithMessage loc "Invalid signature."
   50       Just sig' -> case toMessageHash msg of
   51         Nothing -> failWithMessage loc "Invalid message hash."
   52         Just msg' -> pure $ case DSIGN.verifyDSIGN () pk' msg' sig' of
   53           Left _   -> False
   54           Right () -> True
   55   where
   56     loc :: Text
   57     loc = "ECDSA SECP256k1 signature verification"

   59 -- | Verify a Schnorr signature made using the SECP256k1 curve.
   60 --
   61 -- = Note
   62 --
   63 -- There are additional well-formation requirements for the arguments beyond
   64 -- their length. Throughout, we refer to co-ordinates of the point @R@.
   65 --
   66 -- * The bytes of the public key must correspond to the /x/ coordinate, as a
   67 -- big-endian integer, as specified in BIP-340.
   68 -- * The first 32 bytes of the signature must correspond to the /x/ coordinate,
   69 -- as a big-endian integer, as specified in BIP-340.
   70 -- * The last 32 bytes of the signature must correspond to the bytes of /s/, as
   71 -- a big-endian integer, as specified in BIP-340.
   72 --
   73 -- = See also
   74 --
   75 -- * [BIP-340](https://github.com/bitcoin/bips/blob/master/bip-0340.mediawiki)
   76 verifySchnorrSecp256k1Signature
   77   :: BS.ByteString -- ^ Public key (32 bytes)
   78   -> BS.ByteString -- ^ Message    (arbitrary length)
   79   -> BS.ByteString -- ^ Signature  (64 bytes)
   80   -> BuiltinResult Bool
   81 verifySchnorrSecp256k1Signature pk msg sig =
   82   case DSIGN.rawDeserialiseVerKeyDSIGN @SchnorrSecp256k1DSIGN pk of
   83     Nothing -> failWithMessage loc "Invalid verification key."
   84     Just pk' -> case DSIGN.rawDeserialiseSigDSIGN @SchnorrSecp256k1DSIGN sig of
   85       Nothing -> failWithMessage loc "Invalid signature."
   86       Just sig' -> pure $ case DSIGN.verifyDSIGN () pk' msg sig' of
   87         Left _   -> False
   88         Right () -> True
   89   where
   90     loc :: Text
   91     loc = "Schnorr SECP256k1 signature verification"
