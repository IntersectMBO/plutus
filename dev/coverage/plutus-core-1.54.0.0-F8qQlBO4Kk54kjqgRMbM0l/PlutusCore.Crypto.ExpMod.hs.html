    7 import PlutusCore.Builtin

    9 import GHC.Natural
   10 import GHC.Num.Integer
   11 
   12 -- | Modular exponentiation.  This uses GHC.Num.integerPowMod#, which gives the
   13 -- wrong answer in some cases.  TODO: we'll be able to remove some of the guards
   14 -- when/if integerPowMod# gets fixed.
   15 expMod :: Integer -> Integer -> Natural -> BuiltinResult Natural
   16 expMod b e m
   17   | m <= 0 = fail "expMod: invalid modulus"
   18   -- ^ We can't have m<0 when m is a Natural, but we may as well be paranoid.
   19   | m == 1 = pure 0
   20   -- ^ Just in case: GHC.Num.Integer.integerRecip# gets this wrong.  Note that 0
   21   -- is invertible modulo 1, with inverse 0.
   22   | b == 0 && e < 0 = failNonInvertible 0 m
   23   -- ^ integerPowMod# incorrectly returns 0 in this case.
   24   | otherwise =
   25       case integerPowMod# b e m of
   26         (# n | #)  -> pure n
   27         (# | () #) -> failNonInvertible b m
   28   where failNonInvertible :: Integer -> Natural -> BuiltinResult Natural
   29         failNonInvertible b1 m1 =
   30           fail ("expMod: " ++ (show b1) ++ " is not invertible modulo " ++ (show m1))
   31 {-# INLINE expMod #-}
