   55 {- | Run an action conditionally based on a predicate on the version.
   56 If we don't know the version then the predicate is assumed to be
   57 false, i.e. we act if we _know_ the predicate is satisfied.
   58 -}
   59 whenVersion :: (Version -> Bool) -> Parser () -> Parser ()
   60 whenVersion p act = do
   61   mv <- getVersion
   62   case mv of
   63     Nothing -> pure ()
   64     Just v  -> when (p v) act

   66 parse
   67   :: (MonadError ParserErrorBundle m, MonadQuote m)
   68   => Parser a
   69   -> String
   70   -> Text
   71   -> m a
   72 parse p file str = do
   73   let res = fmap toErrorB (runReaderT (evalStateT (runParserT p file str) initial) Nothing)
   74   liftEither =<< liftQuote res

   76 toErrorB :: Either (ParseErrorBundle Text ParserError) a -> Either ParserErrorBundle a
   77 toErrorB (Left err) = Left $ ParseErrorB err
   78 toErrorB (Right a)  = Right a

   94 -- This is samething from @Text.Megaparsec.Stream@.
   95 reachOffsetNoLine' ::
   96   forall s.
   97   (Stream s) =>
   98   -- | How to split input stream at given offset
   99   (Int -> s -> (Tokens s, s)) ->
  100   -- | How to fold over input stream
  101   (forall b. (b -> Token s -> b) -> b -> Tokens s -> b) ->
  102   -- | Newline token and tab token
  103   (Token s, Token s) ->
  104   -- | Offset to reach
  105   -- | Increment in column position for a token
  106   (Token s -> Pos) ->
  107   Int ->
  108   -- | Initial 'PosState' to use
  109   PosState s ->
  110   -- | Updated 'PosState'
  111   PosState s
  112 reachOffsetNoLine'
  113   splitAt'
  114   foldl''
  115   (newlineTok, tabTok)
  116   columnIncrement
  117   o
  118   PosState {..} =
  119     ( PosState
  120         { pstateInput = post,
  121           pstateOffset = max pstateOffset o,
  122           pstateSourcePos = spos,
  123           pstateTabWidth = pstateTabWidth,
  124           pstateLinePrefix = pstateLinePrefix
  125         }
  126     )
  127     where
  128       spos = foldl'' go pstateSourcePos pre
  129       (pre, post) = splitAt' (o - pstateOffset) pstateInput
  130       go (SourcePos n l c) ch =
  131         let c' = unPos c
  132             w = unPos pstateTabWidth
  133          in if
  134               | ch == newlineTok ->
  135                   SourcePos n (l <> pos1) pos1
  136               | ch == tabTok ->
  137                   SourcePos n l (mkPos $ c' + w - ((c' - 1) `rem` w))
  138               | otherwise ->
  139                   SourcePos n l (c <> columnIncrement ch)
  140 {-# INLINE reachOffsetNoLine' #-}
  141 
  142 getSourcePos' :: MonadParsec e Text m => m SourcePos
  143 getSourcePos' = do
  144   st <- getParserState
  145   let
  146     pst =
  147       reachOffsetNoLine'
  148         Text.splitAt
  149         Text.foldl'
  150         ('\n', '\t')
  151         (const pos1)
  152         (stateOffset st)
  153         (statePosState st)
  154   setParserState st {statePosState = pst}
  155   return (pstateSourcePos pst)

  157 {- | Returns a parser for @a@ by calling the supplied function on the starting
  158 and ending positions of @a@.
  159 
  160 The supplied function should usually return a parser that does /not/ consume trailing
  161 whitespaces. Otherwise, the end position will be the first character after the
  162 trailing whitespaces.
  163 -}
  164 withSpan' :: (SrcSpan -> Parser a) -> Parser a
  165 withSpan' f = mdo
  166   start <- getSourcePos'
  167   res <- f sp
  168   end <- getSourcePos'
  169   let sp = toSrcSpan start end
  170   pure res
  171 
  172 {- | Like `withSpan'`, but the result parser consumes whitespaces.
  173 
  174 @withSpan = (<* whitespace) . withSpan'@
  175 -}
  176 withSpan :: (SrcSpan -> Parser a) -> Parser a
  177 withSpan = (<* whitespace) . withSpan'

  194 toSrcSpan :: SourcePos -> SourcePos -> SrcSpan
  195 toSrcSpan start end =
  196   SrcSpan
  197     { srcSpanFile = sourceName start
  198     , srcSpanSLine = unPos (sourceLine start)
  199     , srcSpanSCol = unPos (sourceColumn start)
  200     , srcSpanELine = unPos (sourceLine end)
  201     , srcSpanECol = unPos (sourceColumn end)
  202     }

  212 -- | Parses a `Name`. Does not consume leading or trailing whitespaces.
  213 name :: Parser Name
  214 name = try $ parseUnquoted <|> parseQuoted
  215   where
  216     parseUnquoted :: Parser Name
  217     parseUnquoted = do
  218       _ <- lookAhead (satisfy isIdentifierStartingChar)
  219       str <- takeWhileP (Just "identifier-unquoted") isIdentifierChar
  220       Name str <$> uniqueSuffix str
  221 
  222     parseQuoted :: Parser Name
  223     parseQuoted = do
  224       _ <- char '`'
  225       _ <- lookAhead (satisfy isQuotedIdentifierChar)
  226       str <- takeWhileP (Just "identifier-quoted") isQuotedIdentifierChar
  227       _ <- char '`'
  228       Name str <$> uniqueSuffix str
  229 
  230     -- Tries to parse a `Unique` value.
  231     -- If it fails then looks up the `Unique` value for the given name.
  232     -- If lookup fails too then generates a fresh `Unique` value.
  233     uniqueSuffix :: Text -> Parser Unique
  234     uniqueSuffix nameStr = try (Unique <$> (char '-' *> Lex.decimal)) <|> uniqueForName nameStr
  235 
  236     -- Return the unique identifier of a name.
  237     -- If it's not in the current parser state, map the name to a fresh id and add it to the state.
  238     uniqueForName :: (MonadState ParserState m, MonadQuote m) => Text -> m Unique
  239     uniqueForName nameStr = do
  240       parserState <- get
  241       case M.lookup nameStr (identifiers parserState) of
  242         Just u -> pure u
  243         Nothing -> do
  244           fresh <- freshUnique
  245           put $ ParserState $ M.insert nameStr fresh $ identifiers parserState
  246           pure fresh
