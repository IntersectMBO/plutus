   46 {- Note [Global uniqueness in the type checker]
   47 WARNING: type inference/checking works under the assumption that the global uniqueness condition
   48 is satisfied. The invariant is not checked, enforced or automatically fulfilled. So you must ensure
   49 that the global uniqueness condition is satisfied before calling 'inferTypeM' or 'checkTypeM'.
   50 
   51 The invariant is preserved. In future we will enforce the invariant.
   52 -}

   54 {- Note [Typing rules]
   55 We write type rules in the bidirectional style.
   56 
   57 [infer| G !- x : a] -- means that the inferred type of 'x' in the context 'G' is 'a'.
   58 'a' is not necessary a varible, e.g. [infer| G !- fun : dom -> cod] is fine too.
   59 It reads as follows: "infer the type of 'fun' in the context 'G', check that it's functional and
   60 bind the 'dom' variable to the domain and the 'cod' variable to the codomain of this type".
   61 
   62 Analogously, [infer| G !- t :: k] means that the inferred kind of 't' in the context 'G' is 'k'.
   63 The [infer| G !- x : a] judgement appears in conclusions in the clauses of the 'inferTypeM'
   64 function.
   65 
   66 [check| G !- x : a] -- check that the type of 'x' in the context 'G' is 'a'.
   67 Since Plutus Core is a fully elaborated language, this amounts to inferring the type of 'x' and
   68 checking that it's equal to 'a'.
   69 
   70 Analogously, [check| G !- t :: k] means "check that the kind of 't' in the context 'G' is 'k'".
   71 The [check| G !- x : a] judgement appears in the conclusion in the sole clause of
   72 the 'checkTypeM' function.
   73 
   74 The equality check is denoted as "a ~ b".
   75 
   76 We use unified contexts in rules, i.e. a context can carry type variables as well as term variables.
   77 
   78 The "NORM a" notation reads as "normalize 'a'".
   79 
   80 The "a ~> b" notations reads as "normalize 'a' to 'b'".
   81 
   82 Functions that can fail start with either @infer@ or @check@ prefixes,
   83 functions that cannot fail look like this:
   84 
   85     kindOfBuiltinType
   86     typeOfBuiltinFunction
   87 -}

   93 -- | Mapping from 'Builtin's to their 'Normalized' 'Type's.
   94 newtype BuiltinTypes uni fun = BuiltinTypes
   95     { unBuiltinTypes :: Array fun (Dupable (Normalized (Type TyName uni ())))
   96     }

   98 type TyVarKinds = UniqueMap TypeUnique (Named (Kind ()))
   99 type VarTypes uni = UniqueMap TermUnique (Named (Dupable (Normalized (Type TyName uni ()))))

  101 -- | Decides what to do upon encountering a variable whose name doesn't match the name of the
  102 -- variable with the same unique that is currently in the scope. Consider for example this type:
  103 --
  104 --     \(a_0 :: *) (b_0 :: *) -> a_0
  105 --
  106 -- here @b_0@ shadows @a_0@ and so any variable having the @0@th unique within the body of the
  107 -- lambda references @b_0@, but we have @a_0@ there and so there's a name mismatch. Technically,
  108 -- it's not a type error to have a name mismatch, because uniques are the single source of truth,
  109 -- however a name mismatch is deeply suspicious and is likely to be caused by a bug somewhere.
  110 --
  111 -- We perform the same check for term-level variables too.
  112 data HandleNameMismatches
  113     = DetectNameMismatches  -- ^ Throw upon encountering such a name.
  114     | IgnoreNameMismatches  -- ^ Ignore it.
  115     deriving stock (Show, Eq)

  117 {- Note [Alignment of kind and type checker configs]
  118 Kind checking is performed as a part of type checking meaning we always need a kind checking config
  119 whenever a type checking one is required. There are three ways we can align the two sorts of config:
  120 
  121 1. store them separately and require the caller to provide both
  122 2. put the type checking config into the kind checking config
  123 3. put the kind checking config into the type checking config
  124 
  125 1 is straightforward, but makes the caller provide one config for kind checking and two configs for
  126 type checking, which is inconsistent boilerplate and so it's not really a good option.
  127 
  128 2 is better: it makes the caller provide only one config: the type checking config stored in the
  129 kind checking config. However that makes the type signatures unwieldy:
  130 
  131     KindCheckConfig (TypeCheckConfig uni fun) => <...>
  132 
  133 plus it's kinda weird to put the type checking config inside the kind checking one.
  134 
  135 3 is what we choose. It makes the caller only worry about the type checking config when doing type
  136 checking, hence no syntactical or semantical overhead, plus it makes the type signatures symmetric:
  137 
  138     (MonadKindCheck err term uni fun ann m, HasKindCheckConfig cfg) => <...>
  139 
  140 vs
  141 
  142     (MonadTypeCheck err term uni fun ann m, HasTypeCheckConfig cfg uni fun) => <...>
  143 
  144 This approach does require a bit of boilerplate at the definition site (see 'HasTypeCheckConfig'),
  145 but it's not much and it doesn't proliferate into user space unlike with the other approaches.
  146 -}

  148 -- | Configuration of the kind checker.
  149 newtype KindCheckConfig = KindCheckConfig
  150     { _kccHandleNameMismatches :: HandleNameMismatches
  151     }
  152 makeClassy ''KindCheckConfig

  154 -- | Configuration of the type checker.
  155 data TypeCheckConfig uni fun = TypeCheckConfig
  156     { _tccKindCheckConfig :: KindCheckConfig
  157     , _tccBuiltinTypes    :: BuiltinTypes uni fun
  158     }
  159 
  160 -- | We want 'HasKindCheckConfig' to be a superclass of 'HasTypeCheckConfig' for being able to
  161 -- seamlessly call the kind checker from the type checker, hence we're rolling out our own
  162 -- 'makeClassy' here just to add the constraint.
  163 class HasKindCheckConfig cfg => HasTypeCheckConfig cfg uni fun | cfg -> uni fun where
  164     typeCheckConfig :: Lens' cfg (TypeCheckConfig uni fun)
  165 
  166     tccKindCheckConfig :: Lens' cfg KindCheckConfig
  167     tccKindCheckConfig =
  168         typeCheckConfig . lens _tccKindCheckConfig (\s b -> s { _tccKindCheckConfig = b })
  169 
  170     tccBuiltinTypes :: Lens' cfg (BuiltinTypes uni fun)
  171     tccBuiltinTypes =
  172         typeCheckConfig . lens _tccBuiltinTypes (\s b -> s { _tccBuiltinTypes = b })
  173 
  174 instance HasKindCheckConfig (TypeCheckConfig uni fun) where
  175     kindCheckConfig = tccKindCheckConfig
  176 
  177 instance HasTypeCheckConfig (TypeCheckConfig uni fun) uni fun where
  178     typeCheckConfig = id
  179 
  180 -- | The environment that the type checker runs in.
  181 data TypeCheckEnv uni fun cfg = TypeCheckEnv
  182     { _tceTypeCheckConfig :: cfg
  183     , _tceTyVarKinds      :: TyVarKinds
  184     , _tceVarTypes        :: VarTypes uni
  185     }
  186 makeLenses ''TypeCheckEnv

  188 -- | The type checking monad that the type checker runs in.
  189 -- In contains a 'TypeCheckEnv' and allows to throw 'TypeError's.
  190 type TypeCheckT uni fun cfg m = ReaderT (TypeCheckEnv uni fun cfg) m
  191 
  192 -- | The constraints that are required for kind checking.
  193 type MonadKindCheck err term uni fun ann m =
  194     ( MonadError err m                  -- Kind/type checking can fail
  195     , ToKind uni                        -- For getting the kind of a built-in type.
  196     )
  197 
  198 -- | The general constraints that are required for type checking a Plutus AST.
  199 type MonadTypeCheck err term uni fun ann m =
  200     ( MonadKindCheck err term uni fun ann m  -- Kind checking is run during type checking (this
  201                                              -- includes the constraint for throwing errors).
  202     , Norm.MonadNormalizeType uni m          -- Type lambdas open up type computation.
  203     , AnnotateCaseBuiltin uni
  204     , GEq uni                                -- For checking equality of built-in types.
  205     , Ix fun                                 -- For indexing into the precomputed array of
  206                                              -- types of built-in functions.
  207     )
  208 
  209 -- | The PLC type error type.
  210 type TypeErrorPlc uni fun ann = TypeError (Term TyName Name uni fun ()) uni fun ann
  211 
  212 -- | The constraints that are required for type checking Plutus Core.
  213 type MonadTypeCheckPlc uni fun ann m =
  214     MonadTypeCheck
  215       (TypeErrorPlc uni fun ann)
  216       (Term TyName Name uni fun ()) uni fun ann m

  222 -- | Run a 'TypeCheckM' computation by supplying a 'TypeCheckConfig' to it.
  223 --
  224 -- Used for both type and kind checking, because we need to do kind checking during type checking
  225 -- and so it makes sense to keep a single monad. However type checking requires a 'TypeCheckConfig',
  226 -- while kind checking doesn't, hence we keep the kind checker fully polymorphic over the type of
  227 -- config, so that the kinder checker can be run with an empty config (such as @()@) and access to
  228 -- a 'TypeCheckConfig' is not needed.
  229 runTypeCheckM :: cfg -> TypeCheckT uni fun cfg m a -> m a
  230 runTypeCheckM config a = runReaderT a $ TypeCheckEnv config mempty mempty
  231 
  232 -- | Extend the context of a 'TypeCheckM' computation with a kinded variable.
  233 withTyVar :: TyName -> Kind () -> TypeCheckT uni fun cfg m a -> TypeCheckT uni fun cfg m a
  234 withTyVar name = local . over tceTyVarKinds . insertNamed name
  235 
  236 -- | Look up the type of a built-in function.
  237 lookupBuiltinM
  238     :: (MonadTypeCheck (TypeError term uni fun ann) term uni fun ann m, HasTypeCheckConfig cfg uni fun)
  239     => ann -> fun -> TypeCheckT uni fun cfg m (Normalized (Type TyName uni ()))
  240 lookupBuiltinM ann fun = do
  241     BuiltinTypes arr <- view $ tceTypeCheckConfig . tccBuiltinTypes
  242     -- Believe it or not, but 'Data.Array' doesn't seem to expose any way of indexing into an array
  243     -- safely.
  244     case preview (ix fun) arr of
  245         Nothing -> throwError $ UnknownBuiltinFunctionE ann fun
  246         Just ty -> liftDupable ty
  247 
  248 -- | Extend the context of a 'TypeCheckM' computation with a typed variable.
  249 withVar
  250     :: Name
  251     -> Normalized (Type TyName uni ())
  252     -> TypeCheckT uni fun cfg m a
  253     -> TypeCheckT uni fun cfg m a
  254 withVar name = local . over tceVarTypes . insertNamed name . dupable
  255 
  256 -- | Look up a type variable in the current context.
  257 lookupTyVarM
  258     :: (MonadKindCheck (TypeError term uni fun ann) term uni fun ann m, HasKindCheckConfig cfg)
  259     => ann -> TyName -> TypeCheckT uni fun cfg m (Kind ())
  260 lookupTyVarM ann name = do
  261     env <- ask
  262     let handleNameMismatches = env ^. tceTypeCheckConfig . kccHandleNameMismatches
  263     case lookupName name $ _tceTyVarKinds env of
  264         Nothing                    -> throwError $ FreeTypeVariableE ann name
  265         Just (Named nameOrig kind) ->
  266             if handleNameMismatches == IgnoreNameMismatches || view theText name == nameOrig
  267                 then pure kind
  268                 else throwError $
  269                         TyNameMismatch ann (TyName . Name nameOrig $ name ^. theUnique) name
  270 
  271 -- | Look up a term variable in the current context.
  272 lookupVarM
  273     :: (MonadTypeCheck (TypeError term uni fun ann) term uni fun ann m, HasTypeCheckConfig cfg uni fun)
  274     => ann -> Name -> TypeCheckT uni fun cfg m (Normalized (Type TyName uni ()))
  275 lookupVarM ann name = do
  276     env <- ask
  277     let handleNameMismatches =
  278             env ^. tceTypeCheckConfig . tccKindCheckConfig . kccHandleNameMismatches
  279     case lookupName name $ _tceVarTypes env of
  280         Nothing                  -> throwError $ FreeVariableE ann name
  281         Just (Named nameOrig ty) ->
  282             if handleNameMismatches == IgnoreNameMismatches || view theText name == nameOrig
  283                 then liftDupable ty
  284                 else throwError $
  285                         NameMismatch ann (Name nameOrig $ name ^. theUnique) name

  311 -- | Infer the kind of a type.
  312 inferKindM
  313     :: (MonadKindCheck (TypeError term uni fun ann) term uni fun ann m, HasKindCheckConfig cfg)
  314     => Type TyName uni ann -> TypeCheckT uni fun cfg m (Kind ())
  315 
  316 -- b :: k
  317 -- ------------------------
  318 -- [infer| G !- con b :: k]
  319 inferKindM (TyBuiltin _ (SomeTypeIn uni)) =
  320     pure $ kindOfBuiltinType uni
  321 
  322 -- [infer| G !- v :: k]
  323 -- ------------------------
  324 -- [infer| G !- var v :: k]
  325 inferKindM (TyVar ann v)           =
  326     lookupTyVarM ann v
  327 
  328 -- [infer| G , n :: dom !- body :: cod]
  329 -- -------------------------------------------------
  330 -- [infer| G !- (\(n :: dom) -> body) :: dom -> cod]
  331 inferKindM (TyLam _ n dom body)    = do
  332     let dom_ = void dom
  333     withTyVar n dom_ $ KindArrow () dom_ <$> inferKindM body
  334 
  335 -- [infer| G !- fun :: dom -> cod]    [check| G !- arg :: dom]
  336 -- -----------------------------------------------------------
  337 -- [infer| G !- fun arg :: cod]
  338 inferKindM (TyApp ann fun arg)     = do
  339     funKind <- inferKindM fun
  340     case funKind of
  341         KindArrow _ dom cod -> do
  342             checkKindM ann arg dom
  343             pure cod
  344         _ -> do
  345             let expectedKindArrow = ExpectedShape "fun k l" ["k", "l"]
  346             throwError $ KindMismatch ann (void fun) expectedKindArrow funKind
  347 
  348 -- [check| G !- a :: *]    [check| G !- b :: *]
  349 -- --------------------------------------------
  350 -- [infer| G !- a -> b :: *]
  351 inferKindM (TyFun ann dom cod)     = do
  352     checkKindM ann dom $ Type ()
  353     checkKindM ann cod $ Type ()
  354     pure $ Type ()
  355 
  356 -- [check| G , n :: k !- body :: *]
  357 -- ---------------------------------------
  358 -- [infer| G !- (all (n :: k). body) :: *]
  359 inferKindM (TyForall ann n k body) = do
  360     withTyVar n (void k) $ checkKindM ann body (Type ())
  361     pure $ Type ()
  362 
  363 
  364 -- [infer| G !- arg :: k]    [check| G !- pat :: (k -> *) -> k -> *]
  365 -- -----------------------------------------------------------------
  366 -- [infer| G !- ifix pat arg :: *]
  367 inferKindM (TyIFix ann pat arg)    = do
  368     k <- inferKindM arg
  369     checkKindM ann pat $ toPatFuncKind k
  370     pure $ Type ()
  371 
  372 -- s_0 = [p_0_0 ... p_0_m]   [check| G !- p_0_0 :: *] ... [check| G !- p_0_m :: *]
  373 -- ...
  374 -- s_n = [p_n_0 ... p_n_m]   [check| G !- p_n_0 :: *] ... [check| G !- p_n_m :: *]
  375 -- -------------------------------------------------------------------------------
  376 -- [infer| G !- sop s_0 ... s_n :: *]
  377 inferKindM (TySOP ann tyls)        = do
  378     for_ tyls $ \tyl -> for_ tyl $ \ty -> checkKindM ann ty (Type ())
  379     pure $ Type ()
  380 
  381 -- | Check a 'Type' against a 'Kind'.
  382 checkKindM
  383     :: (MonadKindCheck (TypeError term uni fun ann) term uni fun ann m, HasKindCheckConfig cfg)
  384     => ann -> Type TyName uni ann -> Kind () -> TypeCheckT uni fun cfg m ()
  385 
  386 -- [infer| G !- ty : tyK]    tyK ~ k
  387 -- ---------------------------------
  388 -- [check| G !- ty : k]
  389 checkKindM ann ty k = do
  390     tyK <- inferKindM ty
  391     when (tyK /= k) $ throwError (KindMismatch ann (void ty) (ExpectedExact k) tyK)
  392

  394 -- ## Type checking ##
  395 -- ###################

  397 -- | @unfoldIFixOf pat arg k = NORM (vPat (\(a :: k) -> ifix vPat a) arg)@
  398 unfoldIFixOf
  399     :: MonadNormalizeType uni m
  400     => Normalized (Type TyName uni ())  -- ^ @vPat@
  401     -> Normalized (Type TyName uni ())  -- ^ @vArg@
  402     -> Kind ()                          -- ^ @k@
  403     -> TypeCheckT uni fun cfg m (Normalized (Type TyName uni ()))
  404 unfoldIFixOf pat arg k = do
  405     let vPat = unNormalized pat
  406         vArg = unNormalized arg
  407     a <- liftQuote $ freshTyName "a"
  408     -- We need to rename @vPat@, otherwise it would be used twice below, which would break global
  409     -- uniqueness. Alternatively, we could use 'normalizeType' instead of 'normalizeTypeM' as the
  410     -- former performs renaming before doing normalization, but renaming the entire type implicitly
  411     -- would be less efficient than renaming a subpart of the type explicitly.
  412     --
  413     -- Note however that breaking global uniqueness here most likely would not result in buggy
  414     -- behavior, see https://github.com/IntersectMBO/plutus/pull/2219#issuecomment-672815272
  415     -- But breaking global uniqueness is a bad idea regardless.
  416     vPat' <- rename vPat
  417     normalizeTypeM $
  418         mkIterTyAppNoAnn vPat'
  419             [ TyLam () a k . TyIFix () vPat $ TyVar () a
  420             , vArg
  421             ]

  423 -- See Note [Global uniqueness in the type checker].
  424 -- See Note [Typing rules].
  425 -- | Synthesize the type of a term, returning a normalized type.
  426 inferTypeM
  427     :: (MonadTypeCheckPlc uni fun ann m, HasTypeCheckConfig cfg uni fun)
  428     => Term TyName Name uni fun ann -> TypeCheckT uni fun cfg m (Normalized (Type TyName uni ()))
  429 
  430 -- c : vTy
  431 -- -------------------------
  432 -- [infer| G !- con c : vTy]
  433 inferTypeM (Constant _ (Some (ValueOf uni _))) =
  434     -- See Note [Normalization of built-in types].
  435     normalizeTypeM $ mkTyBuiltinOf () uni
  436 
  437 -- [infer| G !- bi : vTy]
  438 -- ------------------------------
  439 -- [infer| G !- builtin bi : vTy]
  440 inferTypeM (Builtin ann fun) =
  441     lookupBuiltinM ann fun
  442 
  443 -- [infer| G !- v : ty]    ty ~> vTy
  444 -- ---------------------------------
  445 -- [infer| G !- var v : vTy]
  446 inferTypeM (Var ann name) =
  447     lookupVarM ann name
  448 
  449 -- [check| G !- dom :: *]    dom ~> vDom    [infer| G , n : dom !- body : vCod]
  450 -- ----------------------------------------------------------------------------
  451 -- [infer| G !- lam n dom body : vDom -> vCod]
  452 inferTypeM (LamAbs ann n dom body) = do
  453     checkKindM ann dom $ Type ()
  454     vDom <- normalizeTypeM $ void dom
  455     TyFun () <<$>> pure vDom <<*>> withVar n vDom (inferTypeM body)
  456 
  457 -- [infer| G , n :: nK !- body : vBodyTy]
  458 -- ---------------------------------------------------
  459 -- [infer| G !- abs n nK body : all (n :: nK) vBodyTy]
  460 inferTypeM (TyAbs _ n nK body) = do
  461     let nK_ = void nK
  462     TyForall () n nK_ <<$>> withTyVar n nK_ (inferTypeM body)
  463 
  464 -- [infer| G !- fun : vDom -> vCod]    [check| G !- arg : vDom]
  465 -- ------------------------------------------------------------
  466 -- [infer| G !- fun arg : vCod]
  467 inferTypeM (Apply ann fun arg) = do
  468     vFunTy <- inferTypeM fun
  469     case unNormalized vFunTy of
  470         TyFun _ vDom vCod -> do
  471             -- Subparts of a normalized type, so normalized.
  472             checkTypeM ann arg $ Normalized vDom
  473             pure $ Normalized vCod
  474         _ -> do
  475             let expectedTyFun = ExpectedShape "fun k l" ["k", "l"]
  476             throwError (TypeMismatch ann (void fun) expectedTyFun vFunTy)
  477 
  478 -- [infer| G !- body : all (n :: nK) vCod]    [check| G !- ty :: tyK]    ty ~> vTy
  479 -- -------------------------------------------------------------------------------
  480 -- [infer| G !- body {ty} : NORM ([vTy / n] vCod)]
  481 inferTypeM (TyInst ann body ty) = do
  482     vBodyTy <- inferTypeM body
  483     case unNormalized vBodyTy of
  484         TyForall _ n nK vCod -> do
  485             checkKindM ann ty nK
  486             vTy <- normalizeTypeM $ void ty
  487             substNormalizeTypeM vTy n vCod
  488         _ -> do
  489             let expectedTyForall = ExpectedShape "all a kind body" ["a", "kind", "body"]
  490             throwError (TypeMismatch ann (void body) expectedTyForall vBodyTy)
  491 
  492 -- [infer| G !- arg :: k]    [check| G !- pat :: (k -> *) -> k -> *]    pat ~> vPat    arg ~> vArg
  493 -- [check| G !- term : NORM (vPat (\(a :: k) -> ifix vPat a) vArg)]
  494 -- -----------------------------------------------------------------------------------------------
  495 -- [infer| G !- iwrap pat arg term : ifix vPat vArg]
  496 inferTypeM (IWrap ann pat arg term) = do
  497     k <- inferKindM arg
  498     checkKindM ann pat $ toPatFuncKind k
  499     vPat <- normalizeTypeM $ void pat
  500     vArg <- normalizeTypeM $ void arg
  501     checkTypeM ann term =<< unfoldIFixOf vPat vArg k
  502     pure $ TyIFix () <$> vPat <*> vArg
  503 
  504 -- [infer| G !- term : ifix vPat vArg]    [infer| G !- vArg :: k]
  505 -- -----------------------------------------------------------------------
  506 -- [infer| G !- unwrap term : NORM (vPat (\(a :: k) -> ifix vPat a) vArg)]
  507 inferTypeM (Unwrap ann term) = do
  508     vTermTy <- inferTypeM term
  509     case unNormalized vTermTy of
  510         TyIFix _ vPat vArg -> do
  511             k <- inferKindM $ ann <$ vArg
  512             -- Subparts of a normalized type, so normalized.
  513             unfoldIFixOf (Normalized vPat) (Normalized vArg) k
  514         _                  -> do
  515             let expectedTyIFix = ExpectedShape "ifix pat arg" ["pat", "arg"]
  516             throwError (TypeMismatch ann (void term) expectedTyIFix vTermTy)
  517 
  518 -- [check| G !- ty :: *]    ty ~> vTy
  519 -- ----------------------------------
  520 -- [infer| G !- error ty : vTy]
  521 inferTypeM (Error ann ty) = do
  522     checkKindM ann ty $ Type ()
  523     normalizeTypeM $ void ty
  524 
  525 -- resTy ~> vResTy     vResTy = sop s_0 ... s_i ... s_n
  526 -- s_i = [p_i_0 ... p_i_m]   [check| G !- t_0 : p_i_0] ... [check| G !- t_m : p_i_m]
  527 -- ---------------------------------------------------------------------------------
  528 -- [infer| G !- constr resTy i t_0 ... t_m : vResTy]
  529 inferTypeM t@(Constr ann resTy i args) = do
  530     vResTy <- normalizeTypeM $ void resTy
  531 
  532     -- We don't know exactly what to expect, we only know what the i-th sum should look like, so we
  533     -- assert that we should have some types in the sum up to there, and then the known product type.
  534     let -- 'toInteger' is necessary, because @i@ is a @Word64@ and therefore @i - 1@ would be
  535         -- @maxBound :: Word64@ for @i = 0@ if we didn't have 'toInteger'.
  536         prodPrefix  = map (\j -> "prod_" <> Text.pack (show j)) [0 .. toInteger i - 1]
  537         fields      = map (\k -> "field_" <> Text.pack (show k)) [0 .. length args - 1]
  538         prod_i      = "[" <> Text.intercalate " " fields <> "]"
  539         shape       = "sop " <> foldMap (<> " ") prodPrefix <> prod_i <> " ..."
  540         vars        = prodPrefix ++ fields
  541         expectedSop = ExpectedShape shape vars
  542     case unNormalized vResTy of
  543         TySOP _ vSTys -> case vSTys ^? wix i of
  544             Just pTys -> case zipExact args pTys of
  545                 -- pTy is a sub-part of a normalized type, so normalized
  546                 Just ps -> for_ ps $ \(arg, pTy) -> checkTypeM ann arg (Normalized pTy)
  547                 -- the number of args does not match the number of types in the i'th SOP
  548                 -- alternative
  549                 Nothing -> throwError (TypeMismatch ann (void t) expectedSop vResTy)
  550             -- result type does not contain an i'th sum alternative
  551             Nothing -> throwError (TypeMismatch ann (void t) expectedSop vResTy)
  552         -- result type is not a SOP type
  553         _ -> throwError (TypeMismatch ann (void t) expectedSop vResTy)
  554 
  555     pure vResTy
  556 
  557 -- resTy ~> vResTy   [infer| G !- scrut : sop s_0 ... s_n]
  558 -- s_0 = [p_0_0 ... p_0_m]   [check| G !- c_0 : p_0_0 -> ... -> p_0_m -> vResTy]
  559 -- ...
  560 -- s_n = [p_n_0 ... p_n_m]   [check| G !- c_n : p_n_0 -> ... -> p_n_m -> vResTy]
  561 -- -----------------------------------------------------------------------------
  562 -- [infer| G !- case resTy scrut c_0 ... c_n : vResTy]
  563 inferTypeM (Case ann resTy scrut branches) = do
  564     vResTy <- normalizeTypeM $ void resTy
  565     vScrutTy <- inferTypeM scrut
  566 
  567     -- We don't know exactly what to expect, we only know that it should
  568     -- be a SOP with the right number of sum alternatives when type of scrutinee is SOP
  569     let prods = map (\j -> "prod_" <> Text.pack (show j)) [0 .. length branches - 1]
  570         expectedSop = ExpectedShape (Text.intercalate " " $ "sop" : prods) prods
  571     case unNormalized vScrutTy of
  572         TySOP _ sTys -> case zipExact branches sTys of
  573             Just branchesAndArgTypes -> for_ branchesAndArgTypes $ \(c, argTypes) ->
  574                 -- made of sub-parts of a normalized type, so normalized
  575                 checkTypeM ann c (Normalized $ mkIterTyFun () argTypes (unNormalized vResTy))
  576             -- scrutinee does not have a SOP type with the right number of alternatives
  577             -- for the number of branches
  578             Nothing -> throwError (TypeMismatch ann (void scrut) expectedSop vScrutTy)
  579         vTy -> case annotateCaseBuiltin vTy branches of
  580             Right branchesAndArgTypes -> for_ branchesAndArgTypes $ \(c, argTypes) -> do
  581                 vArgTypes <- traverse (fmap unNormalized . normalizeTypeM) argTypes
  582                 -- made of sub-parts of a normalized type, so normalized
  583                 checkTypeM ann c (Normalized $ mkIterTyFun () vArgTypes (unNormalized vResTy))
  584             Left err -> throwError $ UnsupportedCaseBuiltin ann err
  585 
  586     -- If we got through all that, then every case type is correct, including that
  587     -- they all result in vResTy, so we can safely conclude that that is the type of the
  588     -- whole expression.
  589     pure vResTy
  590 
  591 -- See Note [Global uniqueness in the type checker].
  592 -- See Note [Typing rules].
  593 -- | Check a 'Term' against a 'NormalizedType'.
  594 checkTypeM
  595     :: (MonadTypeCheckPlc uni fun ann m, HasTypeCheckConfig cfg uni fun)
  596     => ann
  597     -> Term TyName Name uni fun ann
  598     -> Normalized (Type TyName uni ())
  599     -> TypeCheckT uni fun cfg m ()
  600 
  601 -- [infer| G !- term : vTermTy]    vTermTy ~ vTy
  602 -- ---------------------------------------------
  603 -- [check| G !- term : vTy]
  604 checkTypeM ann term vTy = do
  605     vTermTy <- inferTypeM term
  606     when (vTermTy /= vTy) $ do
  607         let expectedVTy = ExpectedExact $ unNormalized vTy
  608         throwError $ TypeMismatch ann (void term) expectedVTy vTermTy
