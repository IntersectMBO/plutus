   20 -- | Compile a PLC term to UPLC, and optimize it.
   21 compileTerm
   22   :: (Compiling m uni fun name a
   23   , MonadReader (CompilationOpts name fun a) m
   24   )
   25   => Term tyname name uni fun a
   26   -> m (UPLC.Term name uni fun a)
   27 compileTerm t = do
   28   simplOpts <- view coSimplifyOpts
   29   builtinSemanticsVariant <- view coBuiltinSemanticsVariant
   30   let erased = eraseTerm t
   31   renamed <- rename erased
   32   UPLC.simplifyTerm simplOpts builtinSemanticsVariant renamed
   33 
   34 -- | Compile a PLC program to UPLC, and optimize it.
   35 compileProgram
   36   :: (Compiling m uni fun name a
   37   , MonadReader (CompilationOpts name fun a) m
   38   )
   39   => Program tyname name uni fun a
   40   -> m (UPLC.Program name uni fun a)
   41 compileProgram (Program a v t) = UPLC.Program a v <$> compileTerm t
   42 
   43 -- | Compile a PLC program to UPLC, and optimize it. This includes
   44 -- the compilation trace in the result.
   45 compileProgramWithTrace
   46   :: (Compiling m uni fun name a
   47   , MonadReader (CompilationOpts name fun a) m
   48   )
   49   => Program tyname name uni fun a
   50   -> m (UPLC.Program name uni fun a, UPLC.SimplifierTrace name uni fun a)
   51 compileProgramWithTrace (Program a v t) = do
   52   simplOpts <- view coSimplifyOpts
   53   builtinSemanticsVariant <- view coBuiltinSemanticsVariant
   54   let erased = eraseTerm t
   55   renamedProgram <- UPLC.Program a v <$> rename erased
   56   UPLC.simplifyProgramWithTrace
   57     simplOpts
   58     builtinSemanticsVariant
   59     renamedProgram
