   29 {- Note [Unique usage errors]
   30 The definitions analysis can find a number of problems with usage of uniques, however
   31 clients may not always want to throw an error on all of them, hence we simply return
   32 them all and allow the client to chose if they want to throw some of them.
   33 -}

   35 -- | Information about a unique, a pair of a definition if we have one and a set of uses.
   36 type UniqueInfo ann = (Maybe (ScopedLoc ann), Set.Set (ScopedLoc ann))
   37 type UniqueInfos ann = UniqueMap Unique (UniqueInfo ann)
   38 
   39 data ScopedLoc ann = ScopedLoc ScopeType ann deriving stock (Eq, Ord)
   40 
   41 -- | Tag for distinguishing between whether we are talking about the term scope
   42 -- for variables or the type scope for variables.
   43 data ScopeType = TermScope | TypeScope deriving stock (Eq, Ord)
   44 
   45 lookupDef
   46     :: (Ord ann,
   47         HasUnique name unique,
   48         MonadState (UniqueInfos ann) m)
   49     => name
   50     -> m (UniqueInfo ann)
   51 lookupDef n = do
   52     previousDef <- gets $ lookupNameIndex n
   53     case previousDef of
   54         Just d -> pure d
   55         Nothing -> do
   56             let empty = (Nothing, mempty)
   57             modify $ insertByNameIndex n empty
   58             pure empty
   59 
   60 addDef
   61     :: (Ord ann,
   62         HasUnique n unique,
   63         MonadState (UniqueInfos ann) m,
   64         MonadWriter [UniqueError ann] m)
   65     => n -- ^ The variable
   66     -> ann -- ^ The annotation of the variable
   67     -> ScopeType -- ^ The scope type
   68     -> m ()
   69 addDef n newDef tpe = do
   70     let def = ScopedLoc tpe newDef
   71 
   72     d@(_, uses) <- lookupDef n
   73     checkUndefined n def d
   74     modify $ insertByNameIndex n (Just def, uses)
   75 
   76 -- | Check that a variable is currently undefined.
   77 checkUndefined
   78     :: (HasUnique n u,
   79         MonadState (UniqueInfos ann) m,
   80         MonadWriter [UniqueError ann] m)
   81     => n -- ^ The variable
   82     -> ScopedLoc ann -- ^ The new definition
   83     -> UniqueInfo ann -- ^ The existing info
   84     -> m ()
   85 checkUndefined n (ScopedLoc _ newDef) info = case info of
   86     (Just (ScopedLoc _ prevDef), _) -> tell [MultiplyDefined (n ^. theUnique) prevDef newDef]
   87     _                               -> pure ()
   88 
   89 addUsage
   90     :: (Ord ann,
   91         HasUnique n unique,
   92         MonadState (UniqueInfos ann) m,
   93         MonadWriter [UniqueError ann] m)
   94     => n -- ^ The variable
   95     -> ann -- ^ The annotation of the variable
   96     -> ScopeType -- ^ The scope type
   97     -> m ()
   98 addUsage n newUse tpe = do
   99     let use = ScopedLoc tpe newUse
  100 
  101     d@(def, uses) <- lookupDef n
  102     checkCoherency n use d
  103     checkDefined n use d
  104     modify $ insertByNameIndex n (def, Set.insert use uses)
  105 
  106 checkDefined
  107     :: (HasUnique n u,
  108         MonadWriter [UniqueError ann] m)
  109     => n -- ^ The variable
  110     -> ScopedLoc ann -- ^ The new definition
  111     -> UniqueInfo ann -- ^ The existing info
  112     -> m ()
  113 checkDefined n (ScopedLoc _ loc) (def, _) = case def of
  114     Nothing -> tell [FreeVariable (n ^. theUnique) loc]
  115     Just _  -> pure ()
  116 
  117 checkCoherency
  118     :: (HasUnique n u,
  119         MonadWriter [UniqueError ann] m)
  120     => n -- ^ The variable
  121     -> ScopedLoc ann -- ^ The new definition
  122     -> UniqueInfo ann -- ^ The existing info
  123     -> m ()
  124 checkCoherency n (ScopedLoc tpe loc) (def, uses) = do
  125     for_ def checkLoc
  126     for_ (Set.toList uses) checkLoc
  127 
  128     where
  129         checkLoc (ScopedLoc tpe' loc') = when (tpe' /= tpe) $
  130             tell [IncoherentUsage (n ^. theUnique) loc' loc]

  132 -- | Given a PLC term, add all of its term and type definitions and usages, including its subterms
  133 -- and subtypes, to a global map.
  134 termDefs
  135     :: (Ord ann,
  136         HasUnique name TermUnique,
  137         HasUnique tyname TypeUnique,
  138         MonadState (UniqueInfos ann) m,
  139         MonadWriter [UniqueError ann] m)
  140     => Term tyname name uni fun ann
  141     -> m ()
  142 termDefs tm = do
  143    forMOf_ termSubtermsDeep tm handleTerm
  144    forMOf_ termSubtypesDeep tm handleType
  145 
  146 handleTerm :: (Ord ann,
  147         HasUnique name TermUnique,
  148         HasUnique tyname TypeUnique,
  149         MonadState (UniqueInfos ann) m,
  150         MonadWriter [UniqueError ann] m)
  151     => Term tyname name uni fun ann
  152     -> m ()
  153 handleTerm = \case
  154     Var ann n         ->
  155         addUsage n ann TermScope
  156     LamAbs ann n _ _ -> do
  157         addDef n ann TermScope
  158     TyAbs ann tn _ _  -> do
  159         addDef tn ann TypeScope
  160     _                  -> pure ()
  161 
  162 -- | Given a type, add its type definition/usage, including its subtypes, to a global map.
  163 handleType
  164     :: (Ord ann,
  165         HasUnique tyname TypeUnique,
  166         MonadState (UniqueInfos ann) m,
  167         MonadWriter [UniqueError ann] m)
  168     => Type tyname uni ann
  169     -> m ()
  170 handleType = \case
  171     TyVar ann n         ->
  172         addUsage n ann TypeScope
  173     TyForall ann tn _ _ ->
  174         addDef tn ann TypeScope
  175     TyLam ann tn _ _    ->
  176         addDef tn ann TypeScope
  177     _                    -> pure ()
  178 
  179 runTermDefs
  180     :: (Ord ann,
  181         HasUnique name TermUnique,
  182         HasUnique tyname TypeUnique,
  183         Monad m)
  184     => Term tyname name uni fun ann
  185     -> m (UniqueInfos ann, [UniqueError ann])
  186 runTermDefs = runWriterT . flip execStateT mempty . termDefs
