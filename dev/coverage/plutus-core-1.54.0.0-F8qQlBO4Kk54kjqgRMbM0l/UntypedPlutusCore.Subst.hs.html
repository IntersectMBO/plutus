   25 -- | Applicatively replace a variable using the given function.
   26 substVarA
   27     :: Applicative f
   28     => (name -> f (Maybe (Term name uni fun ann)))
   29     -> Term name uni fun ann
   30     -> f (Term name uni fun ann)
   31 substVarA nameF t@(Var _ name) = fromMaybe t <$> nameF name
   32 substVarA _     t              = pure t

   56 -- | Completely replace the names with a new name type.
   57 termMapNames
   58     :: forall name name' uni fun ann
   59     . (name -> name')
   60     -> Term name uni fun ann
   61     -> Term name' uni fun ann
   62 termMapNames f = go
   63     where
   64         -- This is all a bit clunky because of the type-changing, I'm not sure of a nicer way to do it
   65         go :: Term name uni fun ann -> Term name' uni fun ann
   66         go = \case
   67             LamAbs ann name body -> LamAbs ann (f name) (go body)
   68             Var ann name         -> Var ann (f name)
   69 
   70             Apply ann t1 t2      -> Apply ann (go t1) (go t2)
   71             Delay ann t          -> Delay ann (go t)
   72             Force ann t          -> Force ann (go t)
   73             Constr ann i es      -> Constr ann i (fmap go es)
   74             Case ann arg cs      -> Case ann (go arg) (fmap go cs)
   75 
   76             Constant ann c       -> Constant ann c
   77             Builtin ann b        -> Builtin ann b
   78             Error ann            -> Error ann
   79 
   80 programMapNames
   81     :: forall name name' uni fun ann
   82     . (name -> name')
   83     -> Program name uni fun ann
   84     -> Program name' uni fun ann
   85 programMapNames f (Program a v term) = Program a v (termMapNames f term)
