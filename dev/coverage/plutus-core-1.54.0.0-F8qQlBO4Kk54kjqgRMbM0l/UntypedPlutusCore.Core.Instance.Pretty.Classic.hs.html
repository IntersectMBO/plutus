    1 -- | A "classic" (i.e. as seen in the specification) way to pretty-print Untyped Plutus Core terms.
    2 
    3 {-# OPTIONS_GHC -fno-warn-orphans #-}
    4 
    5 {-# LANGUAGE LambdaCase            #-}
    6 {-# LANGUAGE MultiParamTypeClasses #-}
    7 {-# LANGUAGE OverloadedStrings     #-}
    8 {-# LANGUAGE TypeOperators         #-}
    9 {-# LANGUAGE UndecidableInstances  #-}
   10 
   11 module UntypedPlutusCore.Core.Instance.Pretty.Classic () where
   12 
   13 import PlutusPrelude
   14 
   15 import UntypedPlutusCore.Core.Type
   16 
   17 import PlutusCore.Pretty.Classic
   18 import PlutusCore.Pretty.PrettyConst
   19 
   20 import Prettyprinter
   21 import Prettyprinter.Custom
   22 import Universe (Some (..), SomeTypeIn (SomeTypeIn), ValueOf (..))
   23 
   24 instance (PrettyClassicBy configName name, PrettyUni uni, Pretty fun, Pretty ann) =>
   25         PrettyBy (PrettyConfigClassic configName) (Term name uni fun ann) where
   26     prettyBy config = \case
   27         Var ann n ->
   28             sep (consAnnIf config ann [prettyBy config n])
   29         LamAbs ann n t ->
   30             sexp "lam" (consAnnIf config ann
   31                 [prettyBy config n, prettyBy config t])
   32         Apply ann t1 t2 ->
   33             brackets' (sep (consAnnIf config ann
   34                 [prettyBy config t1, prettyBy config t2]))
   35         Constant ann c ->
   36             sexp "con" (consAnnIf config ann [prettyTypeOf c, pretty c])
   37         Builtin ann bi ->
   38             sexp "builtin" (consAnnIf config ann
   39                 [pretty bi])
   40         Error ann ->
   41             sexp "error" (consAnnIf config ann [])
   42         Delay ann term ->
   43             sexp "delay" (consAnnIf config ann
   44                 [prettyBy config term])
   45         Force ann term ->
   46             sexp "force" (consAnnIf config ann
   47                 [prettyBy config term])
   48         Constr ann i es ->
   49             sexp "constr" (consAnnIf config ann (pretty i : fmap (prettyBy config) es))
   50         Case ann arg cs ->
   51             sexp "case" (consAnnIf config ann
   52                 (prettyBy config arg : fmap (prettyBy config) (toList cs)))
   53       where
   54         prettyTypeOf :: Some (ValueOf uni) -> Doc dann
   55         prettyTypeOf (Some (ValueOf uni _ )) = prettyBy juxtRenderContext $ SomeTypeIn uni
   56 
   57 instance (PrettyClassicBy configName (Term name uni fun ann), Pretty ann) =>
   58         PrettyBy (PrettyConfigClassic configName) (Program name uni fun ann) where
   59     prettyBy config (Program ann version term) =
   60         sexp "program" (consAnnIf config ann [pretty version, prettyBy config term])
