   48 {- Note [Input length limitation for IntegerToByteString].
   49 We make `integerToByteString` and `replicateByte` fail if they're called with arguments which would
   50 cause the length of the result to exceed about 8K bytes because the execution time becomes difficult
   51 to predict accurately beyond this point (benchmarks on a number of different machines show that the
   52 CPU time increases smoothly for inputs up to about 8K then increases sharply, becoming chaotic after
   53 about 14K).  This restriction may be removed once a more efficient implementation becomes available.-}
   54 {- NB: if we do relax the length restriction then we will need two variants of integerToByteString in
   55    Plutus Core so that we can continue to support the current behaviour for old scripts.-}
   56 maximumOutputLength :: Integer
   57 maximumOutputLength = 8192

   59 -- | Wrapper for 'unsafeIntegerToByteString' to make it more convenient to define as a builtin.
   60 integerToByteString :: Bool -> Integer -> Integer -> BuiltinResult ByteString
   61 integerToByteString endiannessArg lengthArg input
   62   -- Check that the length is non-negative.
   63   | lengthArg < 0 = do
   64       emit "integerToByteString: negative length argument"
   65       emit $ "Length requested: " <> (pack . show $ input)
   66       builtinResultFailure
   67   -- Check that the requested length does not exceed the limit.  *NB*: if we remove the limit we'll
   68   -- still have to make sure that the length fits into an Int.
   69   | lengthArg > maximumOutputLength = do
   70       emit . pack $ "integerToByteString: requested length is too long (maximum is "
   71                ++ show maximumOutputLength
   72                ++ " bytes)"
   73       emit $ "Length requested: " <> (pack . show $ lengthArg)
   74       builtinResultFailure
   75   -- If the requested length is zero (ie, an explicit output size is not
   76   -- specified) we still have to make sure that the output won't exceed the size
   77   -- limit.  If the requested length is nonzero and less than the limit,
   78   -- integerToByteString checks that the input fits.
   79   | lengthArg == 0 -- integerLog2 n is one less than the number of significant bits in n
   80        && fromIntegral (integerLog2 input) >= 8 * maximumOutputLength =
   81     let bytesRequiredFor n = integerLog2 n `div` 8 + 1
   82         -- ^ This gives 1 instead of 0 for n=0, but we'll never get that.
   83     in do
   84       emit . pack $ "integerToByteString: input too long (maximum is 2^"
   85                ++ show (8 * maximumOutputLength)
   86                ++ "-1)"
   87       emit $ "Length required: " <> (pack . show $ bytesRequiredFor input)
   88       builtinResultFailure
   89   | otherwise = let endianness = endiannessArgToByteOrder endiannessArg in
   90     -- We use fromIntegral here, despite advice to the contrary in general when defining builtin
   91     -- denotations. This is because, if we've made it this far, we know that overflow or truncation
   92     -- are impossible: we've checked that whatever we got given fits inside a (non-negative) Int.
   93     case unsafeIntegerToByteString endianness (fromIntegral lengthArg) input of
   94       Left err -> case err of
   95         NegativeInput -> do
   96           emit "integerToByteString: cannot convert negative Integer"
   97           -- This does work proportional to the size of input. However, we're in a failing case
   98           -- anyway, and the user's paid for work proportional to this size in any case.
   99           emit $ "Input: " <> (pack . show $ input)
  100           builtinResultFailure
  101         NotEnoughDigits -> do
  102           emit "integerToByteString: cannot represent Integer in given number of bytes"
  103           -- This does work proportional to the size of input. However, we're in a failing case
  104           -- anyway, and the user's paid for work proportional to this size in any case.
  105           emit $ "Input: " <> (pack . show $ input)
  106           emit $ "Bytes requested: " <> (pack . show $ lengthArg)
  107           builtinResultFailure
  108       Right result -> pure result

  123 -- For performance and clarity, the endianness argument uses
  124 -- 'ByteOrder', and the length argument is an 'Int'.
  125 -- This may not actually be unsafe, but it shouldn't be used outside this module.
  126 unsafeIntegerToByteString :: ByteOrder -> Int -> Integer -> Either IntegerToByteStringError ByteString
  127 unsafeIntegerToByteString requestedByteOrder requestedLength input
  128   | input < 0 = Left NegativeInput
  129   | input == 0 = Right . BS.replicate requestedLength $ 0x00
  130   -- We use manual specialization to ensure as few branches in loop bodies as
  131   -- we can. See Note [Manual specialization] for details.
  132   | requestedLength == 0 = Right . Builder.builderBytes $ case requestedByteOrder of
  133       LittleEndian -> goLENoLimit mempty input
  134       BigEndian    -> goBENoLimit mempty input
  135   | otherwise = do
  136       let result = case requestedByteOrder of
  137                     LittleEndian -> goLELimit mempty input
  138                     BigEndian    -> goBELimit mempty input
  139       case result of
  140         Nothing -> Left NotEnoughDigits
  141         Just b  -> Right . Builder.builderBytes $ b
  142   where
  143     goLELimit :: Builder -> Integer -> Maybe Builder
  144     goLELimit acc remaining
  145       | remaining == 0 = pure $ padLE acc
  146       | otherwise = do
  147           -- builderLength is constant time, so we don't track the length ourselves
  148           guard (Builder.builderLength acc < requestedLength)
  149           -- This allows extracting eight digits at once. See Note [Loop sectioning] for details on
  150           -- why we do this. We also duplicate this code in several places: see Note [Manual
  151           -- specialization] for why.
  152           --
  153           -- The code is basically equivalent to remaining `quotRem` 2^64, but more efficient. This
  154           -- is for two reasons: firstly, GHC does not optimize divisions into shifts for Integer
  155           -- (even if the divisor is constant), and secondly, the pair generated by `quotRem` costs
  156           -- us as much as 15% peformance, and GHC seems unable to eliminate it. Thus, we have to do
  157           -- it like this instead.
  158           let newRemaining = remaining `unsafeShiftR` 64
  159           -- Given that remaining must be non-negative, fromInteger here effectively truncates to a
  160           -- Word64, by retaining only the least-significant 8 bytes.
  161           let digitGroup :: Word64 = fromInteger remaining
  162           case newRemaining of
  163             0 -> finishLELimit acc digitGroup
  164             _ -> goLELimit (acc <> Builder.storable digitGroup) newRemaining
  165     finishLELimit :: Builder -> Word64 -> Maybe Builder
  166     finishLELimit acc remaining
  167       | remaining == 0 = pure $ padLE acc
  168       | otherwise = do
  169           guard (Builder.builderLength acc < requestedLength)
  170           -- This is equivalent to 'remaining `quotRem` 256' followed by a conversion of the
  171           -- remainder, but faster. This is similar to the larger example above, and we do it for
  172           -- the same reasons.
  173           let newRemaining = remaining `unsafeShiftR` 8
  174           let digit :: Word8 = fromIntegral remaining
  175           finishLELimit (acc <> Builder.word8 digit) newRemaining
  176     -- By separating the case where we don't need to concern ourselves with a
  177     -- user-specified limit, we can avoid branching needlessly, or doing a
  178     -- complex expression check on every loop. See Note [Manual specialization]
  179     -- for why this matters.
  180     goLENoLimit :: Builder -> Integer -> Builder
  181     goLENoLimit acc remaining
  182       | remaining == 0 = acc
  183       | otherwise = let newRemaining = remaining `unsafeShiftR` 64
  184                         digitGroup :: Word64 = fromInteger remaining
  185                       in case newRemaining of
  186                         0 -> finishLENoLimit acc digitGroup
  187                         _ -> goLENoLimit (acc <> Builder.storable digitGroup) newRemaining
  188     finishLENoLimit :: Builder -> Word64 -> Builder
  189     finishLENoLimit acc remaining
  190       | remaining == 0 = acc
  191       | otherwise =
  192           let newRemaining = remaining `unsafeShiftR` 8
  193               digit :: Word8 = fromIntegral remaining
  194            in finishLENoLimit (acc <> Builder.word8 digit) newRemaining
  195     padLE :: Builder -> Builder
  196     padLE acc = let paddingLength = requestedLength - Builder.builderLength acc
  197       in acc <> Builder.bytes (BS.replicate paddingLength 0x0)
  198     -- We manually specialize the big-endian case: see Note [Manual specialization] for why.
  199     goBELimit :: Builder -> Integer -> Maybe Builder
  200     goBELimit acc remaining
  201       | remaining == 0 = pure $ padBE acc
  202       | otherwise = do
  203           guard (Builder.builderLength acc < requestedLength)
  204           let newRemaining = remaining `unsafeShiftR` 64
  205           let digitGroup :: Word64 = fromInteger remaining
  206           case newRemaining of
  207             0 -> finishBELimit acc digitGroup
  208             _ -> goBELimit (Builder.word64BE digitGroup <> acc) newRemaining
  209     finishBELimit :: Builder -> Word64 -> Maybe Builder
  210     finishBELimit acc remaining
  211       | remaining == 0 = pure $ padBE acc
  212       | otherwise = do
  213           guard (Builder.builderLength acc < requestedLength)
  214           let newRemaining = remaining `unsafeShiftR` 8
  215           let digit = fromIntegral remaining
  216           finishBELimit (Builder.word8 digit <> acc) newRemaining
  217     goBENoLimit :: Builder -> Integer -> Builder
  218     goBENoLimit acc remaining
  219       | remaining == 0 = acc
  220       | otherwise = let newRemaining = remaining `unsafeShiftR` 64
  221                         digitGroup = fromInteger remaining
  222                       in case newRemaining of
  223                         0 -> finishBENoLimit acc digitGroup
  224                         _ -> goBENoLimit (Builder.word64BE digitGroup <> acc) newRemaining
  225     finishBENoLimit :: Builder -> Word64 -> Builder
  226     finishBENoLimit acc remaining
  227       | remaining == 0 = acc
  228       | otherwise = let newRemaining = remaining `unsafeShiftR` 8
  229                         digit = fromIntegral remaining
  230                       in finishBENoLimit (Builder.word8 digit <> acc) newRemaining
  231     padBE :: Builder -> Builder
  232     padBE acc = let paddingLength = requestedLength - Builder.builderLength acc in
  233       Builder.bytes (BS.replicate paddingLength 0x0) <> acc

  245 -- For clarity, the stated endianness argument uses 'ByteOrder'.
  246 -- This function may not actually be unsafe, but it shouldn't be used outside this module.
  247 unsafeByteStringToInteger :: ByteOrder -> ByteString -> Integer
  248   -- We use manual specialization to ensure as few branches in loop bodies as we can. See Note
  249   -- [Manual specialization] for details.
  250 unsafeByteStringToInteger statedByteOrder input = case statedByteOrder of
  251     -- Since padding bytes in the most-significant-last representation go at
  252     -- the end of the input, we can skip decoding them, as they won't affect
  253     -- the result in any way.
  254     LittleEndian -> case BS.findIndexEnd (/= 0x00) input of
  255       -- If there are no nonzero bytes, it must be zero.
  256       Nothing  -> 0
  257       Just end -> goLE 0 end 0
  258     -- Since padding bytes in the most-significant-first representation go at
  259     -- the beginning of the input, we can skip decoding them, as they won't
  260     -- affect the result in any way.
  261     BigEndian -> case BS.findIndex (/= 0x00) input of
  262       Nothing  -> 0
  263       Just end -> goBE 0 end 0 (BS.length input - 1)
  264   where
  265     -- Like with toByteString, we use loop sectioning to decode eight digits at once. See Note [Loop
  266     -- sectioning] for why we do this.
  267     goLE :: Integer -> Int -> Int -> Integer
  268     goLE acc limit ix
  269       | ix <= (limit - 7) =
  270           let digitGroup = read64LE ix
  271               -- Same as ix * 8, but faster. GHC might already do this optimization, but we may as
  272               -- well be sure.
  273               shift = ix `unsafeShiftL` 3
  274               newIx = ix + 8
  275               -- We use unsafeShiftL to move a group of eight digits into the right position in
  276               -- the result, then combine with the accumulator. This is equivalent to a
  277               -- multiplication by 2^64*k, but significantly faster, as GHC doesn't optimize
  278               -- such multiplications into shifts for Integers.
  279               newAcc = acc + fromIntegral digitGroup `unsafeShiftL` shift
  280             in goLE newAcc limit newIx
  281       | otherwise = finishLE acc limit ix
  282     finishLE :: Integer -> Int -> Int -> Integer
  283     finishLE acc limit ix
  284       | ix > limit = acc
  285       | otherwise =
  286           let digit = BS.index input ix
  287               shift = ix `unsafeShiftL` 3
  288               newIx = ix + 1
  289               -- Similarly to before, we use unsafeShiftL to move a single digit into the right
  290               -- position in the result.
  291               newAcc = acc + fromIntegral digit `unsafeShiftL` shift
  292             in finishLE newAcc limit newIx
  293     -- Technically, ByteString does not allow reading of anything bigger than a single byte.
  294     -- However, because ByteStrings are counted arrays, caching already brings in adjacent bytes,
  295     -- which makes fetching them quite cheap. Additionally, GHC appears to optimize this into a
  296     -- block read of 64 bits at once, which saves memory movement. See Note [Superscalarity and
  297     -- caching] for details of why this matters.
  298     read64LE :: Int -> Word64
  299     read64LE startIx =
  300       fromIntegral (BS.index input startIx)
  301         .|. (fromIntegral (BS.index input (startIx + 1)) `unsafeShiftL` 8)
  302         .|. (fromIntegral (BS.index input (startIx + 2)) `unsafeShiftL` 16)
  303         .|. (fromIntegral (BS.index input (startIx + 3)) `unsafeShiftL` 24)
  304         .|. (fromIntegral (BS.index input (startIx + 4)) `unsafeShiftL` 32)
  305         .|. (fromIntegral (BS.index input (startIx + 5)) `unsafeShiftL` 40)
  306         .|. (fromIntegral (BS.index input (startIx + 6)) `unsafeShiftL` 48)
  307         .|. (fromIntegral (BS.index input (startIx + 7)) `unsafeShiftL` 56)
  308     -- We manually specialize the big-endian cases: see Note [Manual specialization] for why.
  309     --
  310     -- In the big-endian case, shifts and indexes change in different ways: indexes start _high_
  311     -- and _reduce_, but shifts start _low_ and rise. This is different to the little-endian case,
  312     -- where both start low and rise. Thus, we track the index and shift separately in the
  313     -- big-endian case: it makes the adjustments easier, and doesn't really change anything, as if
  314     -- we wanted to compute the shift, we'd have to pass an offset argument anyway.
  315     goBE :: Integer -> Int -> Int -> Int -> Integer
  316     goBE acc limit shift ix
  317       | ix >= (limit + 7) =
  318           let digitGroup = read64BE ix
  319               newShift = shift + 64
  320               newIx = ix - 8
  321               newAcc = acc + fromIntegral digitGroup `unsafeShiftL` shift
  322            in goBE newAcc limit newShift newIx
  323       | otherwise = finishBE acc limit shift ix
  324     finishBE :: Integer -> Int -> Int -> Int -> Integer
  325     finishBE acc limit shift ix
  326       | ix < limit = acc
  327       | otherwise =
  328           let digit = BS.index input ix
  329               newShift = shift + 8
  330               newIx = ix - 1
  331               newAcc = acc + fromIntegral digit `unsafeShiftL` shift
  332            in finishBE newAcc limit newShift newIx
  333     read64BE :: Int -> Word64
  334     read64BE endIx =
  335       fromIntegral (BS.index input endIx)
  336         .|. (fromIntegral (BS.index input (endIx - 1)) `unsafeShiftL` 8)
  337         .|. (fromIntegral (BS.index input (endIx - 2)) `unsafeShiftL` 16)
  338         .|. (fromIntegral (BS.index input (endIx - 3)) `unsafeShiftL` 24)
  339         .|. (fromIntegral (BS.index input (endIx - 4)) `unsafeShiftL` 32)
  340         .|. (fromIntegral (BS.index input (endIx - 5)) `unsafeShiftL` 40)
  341         .|. (fromIntegral (BS.index input (endIx - 6)) `unsafeShiftL` 48)
  342         .|. (fromIntegral (BS.index input (endIx - 7)) `unsafeShiftL` 56)

  344 {- Note [Binary bitwise operation implementation and manual specialization]
  345 
  346    All of the 'binary' bitwise operations (namely `andByteString`,
  347    `orByteString` and `xorByteString`) operate similarly:
  348 
  349    1. Decide which of their two `ByteString` arguments determines the length
  350       of the result. For padding semantics, this is the _longer_ argument,
  351       whereas for truncation semantics, it's the _shorter_ one. If both
  352       `ByteString` arguments have identical length, it doesn't matter which we
  353       choose.
  354    2. Copy the choice made in step 1 into a fresh mutable buffer.
  355    3. Traverse over each byte of the argument _not_ chosen in step 1, and
  356       combine each of those bytes with the byte at the corresponding index of
  357       the fresh mutable buffer from step 2 (`.&.` for `andByteString`,
  358       `.|.` for `orByteString`, `xor` for `xorByteString`).
  359 
  360   We also make use of loop sectioning to optimize this operation: see Note
  361   [Loop sectioning] explaining why we do this. Fundamentally, this doesn't
  362   change the logic of the operation, but means that step 3 is split into
  363   two smaller sub-steps: we first word 8 bytes at a time, then one byte at a
  364   time to finish up if necessary. Other than the choice of 'combining
  365   operation', the structure of the computation is the same, which suggests that
  366   we want a helper function with a signature like
  367 
  368   helper1 ::
  369     (Word64 -> Word64 -> Word64) ->
  370     (Word8 -> Word8 -> Word8) ->
  371     ByteString ->
  372     ByteString ->
  373     Int ->
  374     ByteString
  375 
  376   or possibly (to avoid duplicate argument passing) like
  377 
  378   helper2 ::
  379     (forall (a :: Type) . Bits a => a -> a -> a) ->
  380     ByteString ->
  381     ByteString ->
  382     Int ->
  383     ByteString
  384 
  385   This would allow us to share all this logic, and have each of the 'top-level'
  386   operations just dispatch to either of the helpers with the appropriate
  387   function argument(s). Instead, we chose to write a manual copy of this logic
  388   for each of the 'top-level' operations, substituting only the 'combining
  389   operation'.
  390 
  391   We made this choice as any design based on either `helper1` or `helper2` is
  392   significantly slower (at least 50% worse, and the penalty _percentage_ grows
  393   with argument size). While `helper2` is significantly more penalizing than
  394   `helper1`, even `helper1` reaches an almost threefold slowdown at the higher
  395   input sizes we are interested in relative the manual version we use here.
  396   Due to the 'low-level' nature of Plutus Core primops, we consider these costs
  397   unacceptable relative the (small) benefits to code clarity and maintainability
  398   any solution using either style of helper would provide.
  399 
  400   The reason for `helper2` under-performing is unsurprising: any argument whose
  401   type is rank-2 polymorphic with a dictionary constraint essentially acts as
  402   a 'program template', which gets interpreted at runtime given some dictionary
  403   for a `Bits` instance. GHC can do practically nothing to optimize this, as
  404   there is no way to tell, for any given argument, _which_ definitions of an
  405   instance would be required here, even if the set of operations we use is
  406   finite, since any instance can make use of the full power of Haskell, which
  407   essentially lands us in Rice's Theorem territory. For `helper1`, the reasons
  408   are similar: it _must_ be able to work regardless of what functions (assuming
  409   appropriate types) it is given, which means in general, GHC is forced to
  410   compile mother-may-I-style code involving pointer chasing those arguments at
  411   runtime. This explains why the 'blowup' becomes worse with argument length.
  412 
  413   While in theory inlining could help with at least the `helper1` case (
  414   `helper2` is beyond that technique), it doesn't seem like GHC is able to
  415   figure this out, even with `INLINE` is placed on `helper1`.
  416   -}

  418 -- | Bitwise logical AND, as per [CIP-122](https://github.com/cardano-foundation/CIPs/tree/master/CIP-0122).
  419 andByteString :: Bool -> ByteString -> ByteString -> ByteString
  420 andByteString shouldPad bs1 bs2 =
  421   let (shorter, longer) = if BS.length bs1 < BS.length bs2 then (bs1, bs2) else (bs2, bs1)
  422       (toCopy, toTraverse) = if shouldPad then (longer, shorter) else (shorter, longer)
  423    in go toCopy toTraverse (BS.length shorter)
  424   where
  425     go :: ByteString -> ByteString -> Int -> ByteString
  426     go toCopy toTraverse traverseLen =
  427       unsafeDupablePerformIO . BS.useAsCStringLen toCopy $ \(copyPtr, copyLen) ->
  428         BS.useAsCString toTraverse $ \traversePtr -> do
  429           BSI.create copyLen $ \dstPtr -> do
  430             copyBytes dstPtr (castPtr copyPtr) copyLen
  431             let (bigStrides, littleStrides) = traverseLen `quotRem` 8
  432             let offset = bigStrides * 8
  433             let bigDstPtr :: Ptr Word64 = castPtr dstPtr
  434             let bigTraversePtr :: Ptr Word64 = castPtr traversePtr
  435             for_ [0 .. bigStrides - 1] $ \i -> do
  436               w64_1 <- peekElemOff bigDstPtr i
  437               w64_2 <- peekElemOff bigTraversePtr i
  438               pokeElemOff bigDstPtr i $ w64_1 Bits..&. w64_2
  439             let smallDstPtr :: Ptr Word8 = plusPtr dstPtr offset
  440             let smallTraversePtr :: Ptr Word8 = plusPtr traversePtr offset
  441             for_ [0 .. littleStrides - 1] $ \i -> do
  442               w8_1 <- peekElemOff smallDstPtr i
  443               w8_2 <- peekElemOff smallTraversePtr i
  444               pokeElemOff smallDstPtr i $ w8_1 Bits..&. w8_2

  481    in go toCopy toTraverse (BS.length shorter)

  485       unsafeDupablePerformIO . BS.useAsCStringLen toCopy $ \(copyPtr, copyLen) ->

  487           BSI.create copyLen $ \dstPtr -> do
  488             copyBytes dstPtr (castPtr copyPtr) copyLen
  489             let (bigStrides, littleStrides) = traverseLen `quotRem` 8

  494               w64_1 <- peekElemOff bigDstPtr i
  495               w64_2 <- peekElemOff bigTraversePtr i
  496               pokeElemOff bigDstPtr i $ Bits.xor w64_1 w64_2

  498             let smallTraversePtr :: Ptr Word8 = plusPtr traversePtr offset

  500               w8_1 <- peekElemOff smallDstPtr i

  544 -- | Bulk bit write, as per [CIP-122](https://github.com/cardano-foundation/CIPs/tree/master/CIP-0122)
  545 writeBits :: ByteString -> [Integer] -> Bool -> BuiltinResult ByteString
  546 writeBits bs ixs bit = case unsafeDupablePerformIO . try $ go of
  547   Left (WriteBitsException i) -> do
  548     emit "writeBits: index out of bounds"
  549     emit $ "Index: " <> (pack . show $ i)
  550     builtinResultFailure
  551   Right result -> pure result

  557           BSI.create len $
  558             \dstPtr ->
  559               let go2 (i:is) = setOrClearAtIx dstPtr i *> go2 is
  560                   go2 _      = pure ()

  562                 copyBytes dstPtr (castPtr srcPtr) len

  565     len = BS.length bs

  570       | i < 0 = throwIO $ WriteBitsException i
  571       | i >= bitLen = throwIO $ WriteBitsException i

  576           let toWrite = if bit
  577                         then Bits.setBit w8 . fromIntegral $ littleIx
  578                         else Bits.clearBit w8 . fromIntegral $ littleIx
  579           pokeByteOff ptr flipIx toWrite

  600 --
  601 -- * Empty 'ByteString's and zero moves don't do anything
  602 -- * Bit moves whose absolute value is larger than the bit length produce all-zeroes
  603 --
  604 -- This also ensures we don't accidentally hit integer overflow issues.
  605 shiftByteString :: ByteString -> Integer -> ByteString
  606 shiftByteString bs bitMove
  607   | BS.null bs = bs

  609   | otherwise = let len = BS.length bs
  610                     bitLen = fromIntegral $ 8 * len
  611                   in if abs bitMove >= bitLen
  612                      then BS.replicate len 0x00
  613                      -- fromIntegral is safe to use here, as the only way this
  614                      -- could overflow (or underflow) an Int is if we had a
  615                      -- ByteString onchain that was over 30 petabytes in size.
  616                      else unsafeShiftByteString bs (fromIntegral bitMove)

  618 -- | Wrapper for calling 'unsafeRotateByteString' safely. Specifically, we avoid various edge cases:
  619 --
  620 -- * Empty 'ByteString's and zero moves don't do anything
  621 -- * Bit moves whose absolute value is larger than the bit length are reduced modulo the length
  622 --
  623 -- Furthermore, we can convert all rotations into positive rotations, by noting that a rotation by @b@
  624 -- is the same as a rotation by @b `mod` bitLen@, where @bitLen@ is the length of the 'ByteString'
  625 -- argument in bits. This value is always non-negative, and if we get 0, we have nothing to do. This
  626 -- reduction also helps us avoid integer overflow issues.
  627 rotateByteString :: ByteString -> Integer -> ByteString
  628 rotateByteString bs bitMove
  629   | BS.null bs = bs
  630   | otherwise = let bitLen = fromIntegral $ 8 * BS.length bs
  631                     -- This is guaranteed non-negative
  632                     reducedBitMove = bitMove `mod` bitLen
  633                   in if reducedBitMove == 0
  634                      then bs
  635                      -- fromIntegral is safe to use here since for a bytestring to have a
  636                      -- size that doesn't fit into an `Int` it would have to have a size
  637                      -- exceeding something like 37 petabytes.
  638                      else unsafeRotateByteString bs (fromIntegral reducedBitMove)

  640 {- Note [Shift and rotation implementation]
  641 
  642 Both shifts and rotations work similarly: they effectively impose a 'write
  643 offset' to bits in the data argument, then write those bits to the result with
  644 this offset applied. The difference between them is in what should be done if
  645 the resulting offset index would fall out of bounds: shifts just discard the
  646 data (and fill whatever remains with zeroes), while rotations 'wrap around'
  647 modularly. This operation is bit parallel by definition, thus theoretically
  648 making it amenable to the techniques described in Note [Loop sectioning].
  649 
  650 However, the naive way of doing this runs into a problem: the byte ordering
  651 on Tier 1 platforms inside `Word64` means that consecutive bit indexes
  652 according to CIP-122 don't remain that way. We could avoid this by using a
  653 byte flip followed by an adjustment in the opposite direction, then a byte flip
  654 back again. However, this is a costly operation, and would also be extremely
  655 fiddly across stride boundaries, making both performance and implementation
  656 clarity suffer. Instead, we use a different observation, namely that both
  657 shifts and rotations on the same input are monoidally homomorphic into
  658 natural number addition (assuming the same 'direction' for shifts). Using
  659 this, combined with Euclidean division, we can decompose any shift or
  660 rotation by `i` into two consecutive shifts and rotations:
  661 
  662 1. A 'large' shift or rotation, by `div i 8`; and
  663 2. A 'small' shift or rotation, by `mod i 8`.
  664 
  665 While on paper, this seems much less efficient (as our stride is smaller),
  666 we also observe that the 'large' shift moves around whole bytes, while also
  667 keeping consecutive bytes consecutive, assuming their bit indices remain
  668 in-bounds. This means that we can implement step 1 both simply and efficiently:
  669 
  670 * For shifts, we perform a partial copy of all the bytes whose bits remain
  671   in-bounds, followed by clearing of whatever remains.
  672 * For rotations, we perform two partial copies: first of all the bytes whose
  673   bits remain in-bounds, followed by whatever remains, at the 'opposite end'.
  674 
  675 These can make use of the bulk copying and clearing operations provided by the
  676 GHC runtime. Not only are these shorter and more readable, they are also _far_
  677 more efficient than anything we could do, as they rely on optimized C called
  678 via the runtime (meaning no FFI penalty). From our experiments, both with
  679 these operations, and others from CIP-122, we note that the cost of these is
  680 essentially constant up to about the size of 1-2 cache lines (64-128 bytes):
  681 since we anticipate smaller inputs are far more likely, this actually runs
  682 _faster_ than our proposed sectioning approach, while being easier to read
  683 and write.
  684 
  685 It is arguable that our approach forces 'double writing', as Step 2 has to
  686 possibly overwrite our work in Step 1. However, by avoiding the need to
  687 perform byte flips, as well as benefitting from the huge speedups gained
  688 from our split approach, this cost is essentially negligible, especially
  689 given that we can operate mutably throughout. We also have an additional
  690 benefit: if the requested rotation or shift happens to be an exact multiple
  691 of 8, we can be _much_ faster, as Step 2 becomes unnecessary in that case.
  692 -}

  694 -- | Shifts, as per [CIP-123](https://github.com/cardano-foundation/CIPs/tree/master/CIP-0123).
  695 -- This may not actually be unsafe, but it shouldn't be used outside this module.
  696 unsafeShiftByteString :: ByteString -> Int -> ByteString
  697 unsafeShiftByteString bs bitMove = unsafeDupablePerformIO . BS.useAsCString bs $ \srcPtr ->
  698       BSI.create len $ \dstPtr -> do
  699         -- To simplify our calculations, we work only with absolute values,
  700         -- letting different functions control for direction, instead of
  701         -- trying to unify the scheme for both positive and negative shifts.
  702         let magnitude = abs bitMove
  703         -- Instead of worrying about partial clearing, we just zero the entire
  704         -- block of memory, as the cost is essentially negligible and saves us
  705         -- a bunch of offset arithmetic.
  706         fillBytes dstPtr 0x00 len
  707         unless (magnitude >= bitLen) $ do
  708           let (bigShift, smallShift) = magnitude `quotRem` 8
  709           case signum bitMove of
  710             (-1) -> negativeShift (castPtr srcPtr) dstPtr bigShift smallShift
  711             _    -> positiveShift (castPtr srcPtr) dstPtr bigShift smallShift
  712   where
  713     len :: Int
  714     !len = BS.length bs
  715     bitLen :: Int
  716     !bitLen = len * 8
  717     negativeShift :: Ptr Word8 -> Ptr Word8 -> Int -> Int -> IO ()
  718     negativeShift srcPtr dstPtr bigShift smallShift = do
  719       let copyDstPtr = plusPtr dstPtr bigShift
  720       let copyLen = len - bigShift
  721       -- Since we already zeroed everything, we only do the partial copy.
  722       copyBytes copyDstPtr srcPtr copyLen
  723       when (smallShift > 0) $ do
  724         -- When working with the small shift, we have to shift bits across
  725         -- byte boundaries. Thus, we have to make sure that:
  726         --
  727         -- 1. We 'save' our first byte from being processed.
  728         -- 2. We can 'select' the bits that would be shifted over the
  729         --    boundary and apply them.
  730         let !invSmallShift = 8 - smallShift
  731         let !mask = 0xFF `Bits.unsafeShiftR` invSmallShift
  732         for_ [len - 1, len - 2 .. len - copyLen] $ \byteIx -> do
  733           -- To handle shifts across byte boundaries, we have to 'read
  734           -- backwards', mask off the relevant part, then recombine.
  735           !(currentByte :: Word8) <- peekByteOff dstPtr byteIx
  736           !(prevByte :: Word8) <- peekByteOff dstPtr (byteIx - 1)
  737           let !prevOverflowBits = prevByte Bits..&. mask
  738           let !newCurrentByte =
  739                 (currentByte `Bits.unsafeShiftR` smallShift)
  740                   Bits..|. (prevOverflowBits `Bits.unsafeShiftL` invSmallShift)
  741           pokeByteOff dstPtr byteIx newCurrentByte
  742         !(firstByte :: Word8) <- peekByteOff dstPtr (len - copyLen - 1)
  743         pokeByteOff dstPtr (len - copyLen - 1) (firstByte `Bits.unsafeShiftR` smallShift)
  744     -- This works similarly to `negativeShift` above, but in the opposite direction.
  745     positiveShift :: Ptr Word8 -> Ptr Word8 -> Int -> Int -> IO ()
  746     positiveShift srcPtr dstPtr bigShift smallShift = do
  747       let copySrcPtr = plusPtr srcPtr bigShift
  748       let copyLen = len - bigShift
  749       copyBytes dstPtr copySrcPtr copyLen
  750       when (smallShift > 0) $ do
  751         let !invSmallShift = 8 - smallShift
  752         let !mask = 0xFF `Bits.unsafeShiftL` invSmallShift
  753         for_ [0, 1 .. copyLen - 2] $ \byteIx -> do
  754           !(currentByte :: Word8) <- peekByteOff dstPtr byteIx
  755           !(nextByte :: Word8) <- peekByteOff dstPtr (byteIx + 1)
  756           let !nextOverflowBits = nextByte Bits..&. mask
  757           let !newCurrentByte =
  758                 (currentByte `Bits.unsafeShiftL` smallShift)
  759                   Bits..|. (nextOverflowBits `Bits.unsafeShiftR` invSmallShift)
  760           pokeByteOff dstPtr byteIx newCurrentByte
  761         !(lastByte :: Word8) <- peekByteOff dstPtr (copyLen - 1)
  762         pokeByteOff dstPtr (copyLen - 1) (lastByte `Bits.unsafeShiftL` smallShift)
  763 
  764 -- | Rotations, as per [CIP-123](https://github.com/cardano-foundation/CIPs/tree/master/CIP-0123).
  765 -- This is defintely unsafe: calling it with bitMove = minBound::Int can cause a
  766 -- segmentation fault.  It must not be used outside this module.
  767 unsafeRotateByteString :: ByteString -> Int -> ByteString
  768 unsafeRotateByteString bs bitMove = unsafeDupablePerformIO . BS.useAsCString bs $ \srcPtr ->
  769   BSI.create len $ \dstPtr -> do
  770     -- The move is guaranteed positive and reduced already. Thus, unlike for
  771     -- shifts, we don't need two variants for different directions.
  772     let (bigRotation, smallRotation) = bitMove `quotRem` 8
  773     go (castPtr srcPtr) dstPtr bigRotation smallRotation
  774   where
  775     len :: Int
  776     !len = BS.length bs
  777     go :: Ptr Word8 -> Ptr Word8 -> Int -> Int -> IO ()
  778     go srcPtr dstPtr bigRotate smallRotate = do
  779       -- Two partial copies are needed here, unlike with shifts, because
  780       -- there's no point zeroing our data, since it'll all be overwritten
  781       -- with stuff from the input anyway.
  782       let copyStartSrcPtr = plusPtr srcPtr bigRotate
  783       let copyStartLen = len - bigRotate
  784       copyBytes dstPtr copyStartSrcPtr copyStartLen
  785       let copyEndDstPtr = plusPtr dstPtr copyStartLen
  786       copyBytes copyEndDstPtr srcPtr bigRotate
  787       when (smallRotate > 0) $ do
  788         -- This works similarly to shifts
  789         let !invSmallRotate = 8 - smallRotate
  790         let !mask = 0xFF `Bits.unsafeShiftL` invSmallRotate
  791         !(cloneFirstByte :: Word8) <- peekByteOff dstPtr 0
  792         for_ [0, 1 .. len - 2] $ \byteIx -> do
  793           !(currentByte :: Word8) <- peekByteOff dstPtr byteIx
  794           !(nextByte :: Word8) <- peekByteOff dstPtr (byteIx + 1)
  795           let !nextOverflowBits = nextByte Bits..&. mask
  796           let !newCurrentByte =
  797                 (currentByte `Bits.unsafeShiftL` smallRotate)
  798                   Bits..|. (nextOverflowBits `Bits.unsafeShiftR` invSmallRotate)
  799           pokeByteOff dstPtr byteIx newCurrentByte
  800         !(lastByte :: Word8) <- peekByteOff dstPtr (len - 1)
  801         let !firstOverflowBits = cloneFirstByte Bits..&. mask
  802         let !newLastByte =
  803               (lastByte `Bits.unsafeShiftL` smallRotate)
  804                 Bits..|. (firstOverflowBits `Bits.unsafeShiftR` invSmallRotate)
  805         pokeByteOff dstPtr (len - 1) newLastByte

  807 -- | Counting the number of set bits, as per [CIP-123](https://github.com/cardano-foundation/CIPs/tree/master/CIP-0123).
  808 countSetBits :: ByteString -> Int
  809 countSetBits bs = unsafeDupablePerformIO . BS.useAsCString bs $ \srcPtr -> do
  810   -- See Note [Loop sectioning] for details of why we
  811   -- define this function the way it is. We make use of the fact that `popCount`
  812   -- is bit-parallel, and has a constant-time implementation for `Word64` and `Word8`.
  813   let bigSrcPtr :: Ptr Word64 = castPtr srcPtr
  814   let smallSrcPtr :: Ptr Word8 = plusPtr srcPtr offset
  815   goBig bigSrcPtr smallSrcPtr 0 0
  816   where
  817     len :: Int
  818     !len = BS.length bs
  819     -- We do this as two separate bindings, for similar reasons as for
  820     -- `integerToByteString`: we take a surprising hit to performance when
  821     -- using a pair, even though eliminating it should be possible here.
  822     bigStrides :: Int
  823     !bigStrides = len `quot` 8
  824     smallStrides :: Int
  825     !smallStrides = len `rem` 8
  826     offset :: Int
  827     !offset = bigStrides * 8
  828     goBig :: Ptr Word64 -> Ptr Word8 -> Int -> Int -> IO Int
  829     goBig !bigSrcPtr !smallSrcPtr !acc !bigIx
  830       | bigIx == bigStrides = goSmall smallSrcPtr acc 0
  831       | otherwise = do
  832           !w64 <- peekElemOff bigSrcPtr bigIx
  833           goBig bigSrcPtr smallSrcPtr (acc + Bits.popCount w64) (bigIx + 1)
  834     goSmall :: Ptr Word8 -> Int -> Int -> IO Int
  835     goSmall !smallSrcPtr !acc !smallIx
  836       | smallIx == smallStrides = pure acc
  837       | otherwise = do
  838           !w8 <- peekElemOff smallSrcPtr smallIx
  839           goSmall smallSrcPtr (acc + Bits.popCount w8) (smallIx + 1)

  841 -- | Finding the first set bit's index, as per [CIP-123](https://github.com/cardano-foundation/CIPs/tree/master/CIP-0123).
  842 findFirstSetBit :: ByteString -> Int
  843 findFirstSetBit bs = unsafeDupablePerformIO . BS.useAsCString bs $ \srcPtr -> do
  844   let bigSrcPtr :: Ptr Word64 = castPtr srcPtr
  845   goBig bigSrcPtr 0 (len - 8)
  846   where
  847     -- We implement this operation in a somewhat unusual way, to try and
  848     -- benefit from bit paralellism, thus allowing loop sectioning as well:
  849     -- see Note [Loop sectioning] as to why we choose to
  850     -- do this.
  851     --
  852     -- Finding the first set bit is not (inherently) bit parallel, as there is
  853     -- a clear 'horizontal dependency'. Thus, we instead 'localize' this
  854     -- 'horizontal dependency' by noting that the following operations _are_
  855     -- bit-parallel:
  856     --
  857     -- 1. Checking if all bits are zero
  858     -- 2. Keeping an additive accumulator
  859     --
  860     -- Essentially, we begin by taking large steps through our data, checking
  861     -- whether we only have zeroes. This can be done in strides of 64 bits at a
  862     -- time, and every time we find that many zeroes, we keep track. After we
  863     -- encounter a nonzero `Word64`, we 'step down' to `Word8`-sized steps,
  864     -- continuing to count zero blocks the same way. Once we encounter a
  865     -- non-zero `Word8`, we can resort to the specialized operation for
  866     -- counting trailing zeroes from `Data.Bits`, and 'top up' our accumulated
  867     -- count to produce the index we want. If we ever 'walk off the end', we
  868     -- know that there's no way we could find any set bits and return -1.
  869     --
  870     -- This is complicated slightly by us having to walk the input backwards
  871     -- instead of forwards, but due to the requirements of the CIP-122 bit
  872     -- indexing scheme, we don't really have a choice here. This doesn't
  873     -- affect the description above however: it just complicates the indexing
  874     -- maths required.
  875     goBig :: Ptr Word64 -> Int -> Int -> IO Int
  876     goBig !bigSrcPtr !acc !byteIx
  877         -- We can do at least one large step. This works because we read
  878         -- backwards, which means that `byteIx` is the _last_ position we read
  879       | byteIx >= 0 = do
  880           !(w64 :: Word64) <- peekByteOff bigSrcPtr byteIx
  881           -- In theory, we could use the same technique here as we do in
  882           -- `goSmall`, namely count speculatively and then compare to 64.
  883           -- However this is not possible for us, as the native byte ordering
  884           -- on Tier 1 platforms does not keep consecutive bits _across_ bytes
  885           -- consecutive, which would make this result unreliable. While we
  886           -- _could_ do a byte order flip before counting (from the opposite
  887           -- end) to avoid this, the cost of this operation is much larger
  888           -- than a comparison to zero, and would only benefit us _once_,
  889           -- instead of once-per-stride. Thus, we instead use the approach
  890           -- here.
  891           if w64 == 0x0
  892             then goBig bigSrcPtr (acc + 64) (byteIx - 8)
  893             else goSmall (castPtr bigSrcPtr) acc (byteIx + 7)
  894         -- We've 'walked off the end' and not found anything, so everything
  895         -- must be zeroes
  896       | byteIx <= (-8) = pure (-1)
  897         -- We can end up here in one of two ways:
  898         --
  899         -- 1. Our input `ByteString` is 7 bytes long or smaller; or
  900         -- 2. We have done all the large steps we can, and have between 1
  901         --    and 7 bytes to go.
  902         --
  903         -- In either case, we forward the accumulator (which will be 0 in
  904         -- case 1) to small stepping. Combining these cases allows us to
  905         -- avoid separate tests for these conditions.
  906       | otherwise = goSmall (castPtr bigSrcPtr) acc (8 + byteIx - 1)
  907     goSmall :: Ptr Word8 -> Int -> Int -> IO Int
  908     goSmall !smallSrcPtr !acc !byteIx
  909       | byteIx < 0 = pure (-1)
  910       | otherwise = do
  911           !(w8 :: Word8) <- peekByteOff smallSrcPtr byteIx
  912           -- Instead of redundantly first checking for a zero byte,
  913           -- then counting, we speculatively count, relying on the behaviour of
  914           -- `countTrailingZeros` that, on a zero byte, we get 8.
  915           let !counted = Bits.countTrailingZeros w8
  916           let !newAcc = acc + counted
  917           if counted == 8
  918             then goSmall smallSrcPtr newAcc (byteIx - 1)
  919             else pure newAcc
  920     len :: Int
  921     !len = BS.length bs

  925 {- Note [writeBits and exceptions]
  926 
  927    As `writeBits` allows us to pass a changelist argument of any length, we
  928    potentially could have an out-of-bounds index anywhere in the list. As we
  929    have to fail on such cases (and report them appropriately), we end up needing
  930    _both_ IO (to do mutable things) as well as a way to signal errors. We can
  931    do this in two ways:
  932 
  933    1. Pre-scan the changelist for any out-of-bounds indexes, fail if we see any,
  934       then apply the necessary changes if no out-of-bounds indexes are found.
  935    2. Speculatively allocate the new `ByteString`, then do the changes in the
  936       changelist argument one at a time, failing as soon as we see an out-of-bounds
  937       index.
  938 
  939   Option 1 would require traversing the changelist argument twice, which is
  940   undesirable, which means that option 2 is the more efficient choice. The
  941   natural choice for option 2 would be something similar to `ExceptT Int IO`
  942   (with the `Int` being an out-of-bounds index). However, we aren't able to do
  943   this, as ultimately, `ByteString`s are implemented as `ForeignPtr`s, forcing
  944   us to use the following function to interact with them, directly or not:
  945 
  946   withForeignPtr :: forall (a :: Type) . ForeignPtr a -> (Ptr a -> IO b) -> IO b
  947 
  948   Notably, the function argument produces a result of `IO b`, whereas we would
  949   need `MonadIO m => m b` instead. This means that our _only_ choice is to
  950   use the exception mechanism, either directly or via some wrappers like
  951   `MonadUnliftIO`. While this is unusual, and arguably against the spirit of
  952   the use of `IO` relative `ByteString` construction, we don't have any other
  953   choice. We decided to use the exception mechanism directly, as while
  954   `MonadUnliftIO` is a bit cleaner, it ultimately ends up doing the same thing
  955   anyway, and this method at least makes it clear what we're doing.
  956 
  957   This doesn't pose any problems from the point of view of Plutus Core, as this
  958   exception cannot 'leak'; we handle it entirely within `writeBits`, and no
  959   other Plutus Core code can ever see it.
  960 -}
  961 newtype WriteBitsException = WriteBitsException Integer
  962   deriving stock (Eq, Show)

  966 {- Note [Manual specialization]
  967 For both integerToByteString and byteStringToInteger, we have to perform very
  968 similar operations, but with small variations:
  969 
  970 - Most-significant-first versus most-significant-last (for both)
  971 - Whether we have a size limit or not (for integerToByteString)
  972 
  973 Additionally, loop sectioning (see Note [Loop sectioning]) requires us to have
  974 separate 'big-stride' and 'small-stride' operations to ensure universality of
  975 input handling. Lastly, we have several subroutines (digit extraction, for
  976 instance) that may vary in similar ways. In such a case, generalization by
  977 means of abstraction seems like a good idea, as the operations (and
  978 subroutines) vary little.
  979 
  980 At the same time, to determine which variation of any given function (or
  981 subroutine) we need, we only have to scrutinize the relevant argument(s) once:
  982 these specifics (such as byte order) don't change during the course of the
  983 operation. Thus, we want to make sure that these checks in the code are _also_
  984 performed only once, ideally at the beginning.
  985 
  986 However, if we write such operations naively as so:
  987 
  988 > subroutine byteOrder arg1 arg2 = case byteOrder of
  989 >       LittleEndian -> ...
  990 >       BigEndian -> ...
  991 
  992 the byteOrder argument will be scrutinized on each call of subroutine. This is
  993 correct in general (as there is no guarantee that the argument will be stable).
  994 Strangely, however, even in a case like this one:
  995 
  996 > mainRoutine byteOrder arg1 arg2 = ...
  997 >    where
  998 >       subroutine arg3 = case byteOrder of
  999 >           LittleEndian -> ...
 1000 >           BigEndian -> ...
 1001 
 1002 GHC _still_ re-scrutinizes byteOrder in every call of subroutine! This penalty
 1003 can be somewhat lessened using a form similar to this:
 1004 
 1005 > mainRoutine byteOrder arg1 arg2 = ...
 1006 >     where
 1007 >        !subroutine = case byteOrder of
 1008 >             LittleEndian -> \arg3 -> ...
 1009 >             BigEndian -> \arg3 -> ...
 1010 
 1011 but this is _still_ between 20 and 30% worse than doing something like this:
 1012 
 1013 > mainRoutine byteOrder arg1 arg2 = case byteOrder of
 1014 >     LittleEndian -> [code calling subroutineLE where needed]
 1015 >     BigEndian -> [code calling subroutineBE where needed]
 1016 >     where
 1017 >         subroutineLE arg3 = ...
 1018 >         subroutineBE arg3 = ...
 1019 
 1020 This form _ensures_ we scrutinize (and branch) only the number of times we have
 1021 to, and in a predictable place. Since these are backends for Plutus Core primops,
 1022 and performance is thus critical, we choose to use this manually-specialized form
 1023 for each combination of relevant arguments. While this is repetitive, and thus
 1024 also somewhat error-prone, the performance penalty for not doing this is
 1025 unacceptable.
 1026 -}

 1028 {- Note [Loop sectioning]
 1029 
 1030 Several operations in this module (including binary logical operations,
 1031 `integerToByteString` and `byteStringToInteger`) effectively function as loops
 1032 over fixed-width binary chunks: these can be bytes (for logical operations),
 1033 digits (for conversions) or something else. These chunks have to be read,
 1034 written or both, and may also require processing using fixed-width,
 1035 constant-time operations over those chunks from the Haskell side, in some
 1036 cases effectively 'translating' these fixed-size operations into variable-width
 1037 equivalents over `ByteString`s. In all cases, this involves trafficking data
 1038 between memory and machine registers (as `ByteString`s and `Integer`s are both
 1039 wrappers around counted arrays), as well as the overheads of looping
 1040 (involving comparison and branches). This trafficking is necessary not only to
 1041 move the memory around, but also to process it, as on modern architectures,
 1042 data must first be moved into a register in order to do anything with it.
 1043 
 1044 However, on all architectures of interest (essentially, 64-bit Tier 1),
 1045 general-purpose registers (GPRs henceforth) are 64 bits (or 8 bytes).
 1046 Furthermore, the primary cost of moving data between memory and registers is
 1047 having to overcome the 'memory wall': the exact amount of data being moved
 1048 doesn't affect this very much. In addition to this, when we operate on single
 1049 bytes, the remaining 56 bits of the GPR holding that data are essentially
 1050 'wasted'. In the situation we have (namely, operating over arrays, whose data
 1051 is adjacent in memory), we thus get two sources of inefficiency:
 1052 
 1053 * Despite paying the cost for a memory transfer, we transfer only one-eighth
 1054   the data we could; and
 1055 * Despite transferring data from memory to registers, we utilize the register
 1056   at only one-eighth capacity.
 1057 
 1058 This essentially means we perform _eight times_ more rotations of the loop,
 1059 and memory moves, than we need to!
 1060 
 1061 To avoid this, we use a technique known as _loop sectioning_.
 1062 Effectively, this transforms our homogenous loop (that always operates one byte at
 1063 a time) into a heterogenous loop: first, we operate on a larger section (called
 1064 a _stride_) until we can no longer do this, and then we finish up using byte
 1065 at a time processing. Essentially, when given an input like this:
 1066 
 1067 [ b1, b2, b3, b4, b5, b6, b7, b8, b9, b10 ]
 1068 
 1069 the homogeous byte-at-a-time approach would process it like so:
 1070 
 1071   _   _   _   _   _   _   _   _   _   _
 1072 [ b1, b2, b3, b4, b5, b6, b7, b8, b9, b10 ]
 1073 
 1074 for a total of 10 memory transfers and 10 loop spins, whereas a loop-sectioned
 1075 approach with a stride of 8 would instead process like so:
 1076 
 1077   ______________________________  _   _
 1078 [ b1, b2, b3, b4, b5, b6, b7, b8, b9, b10 ]
 1079 
 1080 Giving us only _three_ memory transfers and _three_ loop spins instead. This
 1081 effectively reduces our work by a factor of 8. In our cases, this is almost
 1082 free, as there is no data processing to be done: all we need to do is copy
 1083 from one place to another, essentially.
 1084 
 1085 This technique only benefits us because counted arrays are cache-friendly: see
 1086 Note [Superscalarity and caching] for a longer explanation of this and why it
 1087 matters.
 1088

 1090

 1092     https://gitlab.haskell.org/ghc/ghc/-/wikis/platforms#tier-1-platforms
 1093 - Memory wall:
 1094     https://link.springer.com/referenceworkentry/10.1007/978-0-387-09766-4_234
 1095 - Loop sectioning in more detail:
 1096     http://physics.ujep.cz/~zmoravec/prga/main_for/mergedProjects/optaps_for/common/optaps_vec_mine.htm

 1099 {- Note [Superscalarity and caching]
 1100 On modern architectures, in order to process data, it must first be moved from
 1101 memory into a register. This operation has some cost (known as the 'memory wall'),
 1102 which is largely independent of how much data gets moved (assuming the register
 1103 can hold it): moving one byte, or a whole register's worth, costs about the same.
 1104 To reduce this cost, CPU manufacturers have introduced _cache hierarchies_,
 1105 which are designed to limit the cost of the wall, as long as the data access
 1106 matches the cache's optimal usage pattern. Thus, while an idealized view of
 1107 the memory hierachy is this:
 1108 
 1109 Registers
 1110 ---------
 1111 Memory
 1112 
 1113 in reality, the view is more like this:
 1114 
 1115 Registers
 1116 ---------
 1117 L1 cache
 1118 ---------
 1119 L2 cache
 1120 ---------
 1121 L3 cache (on some platforms)
 1122 ---------
 1123 Memory
 1124 
 1125 Each 'higher' cache in the hierarchy is smaller, but faster, and when a memory
 1126 fetch is requested in code, in addition to moving the requested data to a
 1127 register, that data (plus some more) is moved into caches as well. The amount
 1128 of data moved into cache (a _cache line_) is typically eight machine words on
 1129 modern architectures (and definitely is the case for all Tier 1 GHC platforms):
 1130 for the cases concerning Plutus, that is 64 bytes. Therefore, if data we need
 1131 soon after a fetch is _physically_ nearby, it won't need to be fetched from
 1132 memory: instead, it would come from a cache, which is faster (by a considerable
 1133 margin).
 1134 
 1135 To see how this can matter, consider the following ByteString:
 1136 
 1137 [ b1, b2, b3, b4, b5, b6, b7, b8, b9, b10, b11 ]
 1138 
 1139 The ByteString (being a counted array) has all of its data physically adjacent
 1140 to each other. Suppose we wanted to fetch the byte at index 1 (second position).
 1141 The naive view of what happens is like this:
 1142 
 1143 Registers: [b2] [ ] [ ] .... [ ]
 1144 Memory: [ b1, b2, b3, b4, b5, b6, b7, b8, b9, b10, b11 ]
 1145 
 1146 Thus, it would appear that, if we wanted a different position's value, we would
 1147 need to fetch from memory again. However, what _actually_ happens is more like this:
 1148 
 1149 Registers: [b2] [ ] [ ] .... [ ]
 1150 L1 cache: [ b2, b3, b4, b5, b6, b7, b8, b9, b10, b11 ]
 1151 Memory: [ b1, b2, b3, b4, b5, b6, b7, b8, b9, b10, b11 ]
 1152 
 1153 We note that b2, as well as its adjacent elements, were _all_ pulled into the L1
 1154 cache. This can only work because all these elements are physically adjacent in
 1155 memory. The improvement in performance from this cache use is _very_ non-trivial:
 1156 an L1 cache is about 200 times faster than a memory access, and an L2 cache about
 1157 20 times faster.
 1158 
 1159 To take further advantage of this, modern CPUs (and all Tier 1 GHC platforms have
 1160 this capability) are _superscalar_. To explain what this means, let's consider the
 1161 naive view of how CPUs execute instructions: namely, it is one-at-a-time, and
 1162 synchronous. While CPUs must give the _appearance_ that they behave this way, in
 1163 practice, CPU execution is very much asynchronous: due to the proliferation of ALUs
 1164 on a single chip, having twice as many processing units is much cheaper than having
 1165 processing units run twice as fast. Thus, if there are no data dependencies
 1166 between instructions, CPUs can (and do!) execute them simultaneously, stalling to
 1167 await results if a data dependency is detected. This can be done automatically
 1168 using Tomasulo's algorithm, which ensures no conflicts with maximum throughput.
 1169 
 1170 Superscalarity interacts well with the cache hierarchy, as it makes data more
 1171 easily available for processing, provided there is enough 'work to do', and no
 1172 data dependencies. In our situation, most of what we do is data _movement_ from
 1173 one memory location to another, which by its very nature lacks any data
 1174 dependencies.

 1180 - Tomasulo's algorithm: https://en.wikipedia.org/wiki/Tomasulo%27s_algorithm
