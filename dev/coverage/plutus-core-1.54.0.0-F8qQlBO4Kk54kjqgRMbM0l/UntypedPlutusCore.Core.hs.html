   15 -- | Strips off lambda binders.
   16 splitParams :: Term name uni fun a -> ([name], Term name uni fun a)
   17 splitParams = \case
   18   LamAbs _ n t -> first (n :) (splitParams t)

   21 -- | Strip off arguments
   22 splitApplication :: Term name uni fun a -> (Term name uni fun a, [(a, Term name uni fun a)])
   23 splitApplication = go []
   24   where
   25     go acc = \case
   26       Apply ann fun arg -> go ((ann, arg) : acc) fun
   27       t                 -> (t, acc)
