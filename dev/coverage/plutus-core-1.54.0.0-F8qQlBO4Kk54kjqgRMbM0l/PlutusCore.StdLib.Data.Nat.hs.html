   28 -- | @Nat@ as a PLC type.
   29 --
   30 -- > fix \(nat :: *) -> all r. r -> (nat -> r) -> r
   31 natData :: RecursiveType uni fun ()
   32 natData = runQuote $ do
   33     nat <- freshTyName "nat"
   34     r   <- freshTyName "r"
   35     makeRecursiveType () nat []
   36         . TyForall () r (Type ())
   37         . TyFun () (TyVar () r)
   38         . TyFun () (TyFun () (TyVar () nat) $ TyVar () r)
   39         $ TyVar () r
   40 
   41 natTy :: Type TyName uni ()
   42 natTy = _recursiveType natData
   43 
   44 -- |  '0' as a PLC term.
   45 --
   46 -- > wrapNat [] /\(r :: *) -> \(z : r) (f : nat -> r) -> z
   47 zero :: TermLike term TyName Name uni fun => term ()
   48 zero = runQuote $ do
   49     let RecursiveType nat wrapNat = natData
   50     r <- freshTyName "r"
   51     z <- freshName "z"
   52     f <- freshName "f"
   53     return
   54         . wrapNat []
   55         . tyAbs () r (Type ())
   56         . lamAbs () z (TyVar () r)
   57         . lamAbs () f (TyFun () nat $ TyVar () r)
   58         $ var () z
   59 
   60 -- |  'succ' as a PLC term.
   61 --
   62 -- > \(n : nat) -> wrapNat [] /\(r :: *) -> \(z : r) (f : nat -> r) -> f n
   63 succ :: TermLike term TyName Name uni fun => term ()
   64 succ = runQuote $ do
   65     let RecursiveType nat wrapNat = natData
   66     n <- freshName "n"
   67     r <- freshTyName "r"
   68     z <- freshName "z"
   69     f <- freshName "f"
   70     return
   71         . lamAbs () n nat
   72         . wrapNat []
   73         . tyAbs () r (Type ())
   74         . lamAbs () z (TyVar () r)
   75         . lamAbs () f (TyFun () nat $ TyVar () r)
   76         . apply () (var () f)
   77         $ var () n

   79 -- |  @foldrNat@ as a PLC term.
   80 --
   81 -- > /\(r :: *) -> \(f : r -> r) (z : r) ->
   82 -- >     fix {nat} {r} \(rec : nat -> r) (n : nat) ->
   83 -- >         unwrap n {r} z \(n' : nat) -> f (rec n')
   84 foldrNat :: TermLike term TyName Name uni fun => term ()
   85 foldrNat = runQuote $ do
   86     let nat = _recursiveType natData
   87     r   <- freshTyName "r"
   88     f   <- freshName "f"
   89     z   <- freshName "z"
   90     rec <- freshName "rec"
   91     n   <- freshName "n"
   92     n'  <- freshName "n'"
   93     return
   94         . tyAbs () r (Type ())
   95         . lamAbs () f (TyFun () (TyVar () r) (TyVar () r))
   96         . lamAbs () z (TyVar () r)
   97         . apply () (mkIterInstNoAnn fix [nat, TyVar () r])
   98         . lamAbs () rec (TyFun () nat $ TyVar () r)
   99         . lamAbs () n nat
  100         . apply () (apply () (tyInst () (unwrap () (var () n)) $ TyVar () r) $ var () z)
  101         . lamAbs () n' nat
  102         . apply () (var () f)
  103         . apply () (var () rec)
  104         $ var () n'
  105 
  106 -- |  @foldNat@ as a PLC term.
  107 --
  108 -- > /\(r :: *) -> \(f : r -> r) ->
  109 -- >     fix {r} {nat -> r} \(rec : r -> nat -> r) (z : r) (n : nat) ->
  110 -- >         unwrap n {r} z (\(n' : nat) -> rec (f z) n')
  111 foldNat :: TermLike term TyName Name uni fun => term ()
  112 foldNat = runQuote $ do
  113     let nat = _recursiveType natData
  114     r   <- freshTyName "r"
  115     f   <- freshName "f"
  116     rec <- freshName "rec"
  117     z   <- freshName "z"
  118     n   <- freshName "n"
  119     n'  <- freshName "n'"
  120     return
  121         . tyAbs () r (Type ())
  122         . lamAbs () f (TyFun () (TyVar () r) (TyVar () r))
  123         . apply () (mkIterInstNoAnn fix [TyVar () r, TyFun () nat $ TyVar () r])
  124         . lamAbs () rec (TyFun () (TyVar () r) . TyFun () nat $ TyVar () r)
  125         . lamAbs () z (TyVar () r)
  126         . lamAbs () n nat
  127         . apply () (apply () (tyInst () (unwrap () (var () n)) $ TyVar () r) $ var () z)
  128         . lamAbs () n' nat
  129         . mkIterAppNoAnn (var () rec)
  130         $ [ apply () (var () f) $ var () z
  131           , var () n'
  132           ]

  134 -- | Convert a @nat@ to an @integer@.
  135 --
  136 -- > foldNat {integer} (addInteger 1) 1
  137 natToInteger
  138     :: (TermLike term TyName Name uni DefaultFun, uni `HasTypeAndTermLevel` Integer) => term ()
  139 natToInteger = runQuote $ do
  140     return $
  141         mkIterAppNoAnn (tyInst () foldNat $ mkTyBuiltin @_ @Integer ())
  142           [ apply () (builtin () AddInteger) (mkConstant @Integer () 1)
  143           , mkConstant @Integer () 0
  144           ]
