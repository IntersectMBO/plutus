   42 -- | Convert a 'Term' with 'Name's into a 'Term' with 'DeBruijn's.
   43 -- Will throw an error if a free variable is encountered.
   44 deBruijnTerm
   45     :: (MonadError FreeVariableError m)
   46     => Term Name uni fun ann -> m (Term NamedDeBruijn uni fun ann)
   47 deBruijnTerm = deBruijnTermWith freeUniqueThrow
   48 
   49 -- | Convert a 'Term' with 'DeBruijn's into a 'Term' with 'Name's.
   50 -- Will throw an error if a free variable is encountered.
   51 unDeBruijnTerm
   52     :: (MonadQuote m, MonadError FreeVariableError m)
   53     => Term NamedDeBruijn uni fun ann -> m (Term Name uni fun ann)
   54 unDeBruijnTerm = unDeBruijnTermWith freeIndexThrow

   56 -- | Takes a "handler" function to execute when encountering free variables.
   57 deBruijnTermWith
   58     :: Monad m
   59     => (Unique -> ReaderT LevelInfo m Index)
   60     -> Term Name uni fun ann
   61     -> m (Term NamedDeBruijn uni fun ann)
   62 deBruijnTermWith = (runDeBruijnT .) . deBruijnTermWithM

   64 -- | Takes a "handler" function to execute when encountering free variables.
   65 unDeBruijnTermWith
   66     :: MonadQuote m
   67     => (Index -> ReaderT LevelInfo m Unique)

   69     -> m (Term Name uni fun ann)

   72 deBruijnTermWithM
   73     :: MonadReader LevelInfo m
   74     => (Unique -> m Index)
   75     -> Term Name uni fun ann
   76     -> m (Term NamedDeBruijn uni fun ann)
   77 deBruijnTermWithM h = go
   78  where
   79    go = \case
   80        -- variable case
   81        Var ann n -> Var ann <$> nameToDeBruijn h n
   82        -- binder cases
   83        LamAbs ann n t -> declareUnique n $ do
   84            n' <- nameToDeBruijn h n
   85            withScope $ LamAbs ann n' <$> go t
   86        -- boring recursive cases
   87        Apply ann t1 t2 -> Apply ann <$> go t1 <*> go t2
   88        Delay ann t -> Delay ann <$> go t
   89        Force ann t -> Force ann <$> go t
   90        Constr ann i es -> Constr ann i <$> traverse go es
   91        Case ann arg cs -> Case ann <$> go arg <*> traverse go cs
   92        -- boring non-recursive cases
   93        Constant ann con -> pure $ Constant ann con
   94        Builtin ann bn -> pure $ Builtin ann bn
   95        Error ann -> pure $ Error ann
   96 
   97 -- | Takes a "handler" function to execute when encountering free variables.
   98 unDeBruijnTermWithM
   99     :: (MonadReader LevelInfo m, MonadQuote m)
  100     => (Index -> m Unique)
  101     -> Term NamedDeBruijn uni fun ann
  102     -> m (Term Name uni fun ann)
  103 unDeBruijnTermWithM h = go
  104   where
  105     go = \case
  106         -- variable case
  107         Var ann n -> Var ann <$> deBruijnToName h n
  108         -- binder cases
  109         LamAbs ann n t ->
  110             -- See Note [DeBruijn indices of Binders]
  111             declareBinder $ do
  112               n' <- deBruijnToName h $ set index deBruijnInitIndex n
  113               withScope $ LamAbs ann n' <$> go t
  114         -- boring recursive cases
  115         Apply ann t1 t2 -> Apply ann <$> go t1 <*> go t2
  116         Delay ann t -> Delay ann <$> go t
  117         Force ann t -> Force ann <$> go t
  118         Constr ann i es -> Constr ann i <$> traverse go es
  119         Case ann arg cs -> Case ann <$> go arg <*> traverse go cs
  120         -- boring non-recursive cases
  121         Constant ann con -> pure $ Constant ann con
  122         Builtin ann bn -> pure $ Builtin ann bn
  123         Error ann -> pure $ Error ann
