   33 {- Note [Stepping the driver]
   34 
   35 The client instructs the driver to do a single step of the Cek (`Step` Cmd) or multiple steps until
   36 a condition is met (`Continue`,`Next`,`Finish`). The driver upon receiving a Command (via `inputF`)
   37 fires a series of `stepF` actions to fulfill the command. A `stepF` action is supposed to move the
   38 underlying CEK machine a *single* step forward. Since the driver is written as a free monad, the
   39 interpretation of the `stepF` action is left open.
   40 
   41 When the driver calls `stepF state`, it yields its coroutine with the current driver/cek state
   42 (before the stepping interpretation). The interpreter of `stepF` receives the currentState and
   43 calculates a newstate from it (e.g. **actually stepping the cek machine**). The interpreter then
   44 resumes the driver coroutine with the new state.
   45 
   46 The sensible interpretation to this is the CEK's state transition function (`cekTrans oldState
   47 ===> newState`), but other exotic options may be: logging the before/after states, or even
   48 "mutating" the running program's variables in the environment before/after the state transition
   49 (similar to C debuggers). The interpreter can even supply a different state transition function
   50 (e.g. `id`), but we see little benefit in it at the moment. Currently all interpreters
   51 (brick,repl,testing) just call`cekTrans`.
   52 -}

   55 -- | Leave abstract the types of annotation and breakpoints.
   56 -- The only thing the driver requires is an inclusion relation of breakpoints into the Annotation
   57 class Breakpointable ann bps | ann -> bps where
   58     -- MAYBE: we cannot know which breakpoint fired, return instead `Maybe Breakpoint`?
   59     hasBreakpoints :: ann -> bps -> Bool
   60 
   61 -- | The commands that the driver may receive from the client (tui,cli,test,etc)
   62 data Cmd bps
   63   = Step -- ^ Instruct the driver to a *SINGLE* step.
   64          -- Note: No need to pass breakpoints here because the stepping granularity is *minimal*.
   65   | Continue bps -- ^ Instruct to multi-step until end-of-program or until breakpoint reached
   66   | Next bps -- ^ Instruct to multi-step over the function call at point or until breakpoint reached
   67   | Finish bps -- ^ Instruct to multi-step to end of current function or until breakpoint reached
   68   deriving stock (Show, Read)
   69 
   70 -- | The drivers's suspension functor
   71 data DebugF uni fun ann bps a
   72    -- | Await for the client (e.g. TUI) to tell what to do next (Cmd).
   73   = InputF (Cmd bps -> a)
   74     -- | The debug driver wants to log something
   75   | DriverLogF Text a
   76     -- | An enumeratee of Driver State (generator+iteratee):
   77     -- Yield a state before doing a step, then await for a state to resume after the step.
   78     -- See Note [Stepping the driver].
   79   | StepF
   80       (CekState uni fun ann) -- ^ yield with the current driver's state before running a step
   81       (CekState uni fun ann -> a) -- ^ resume back with a state after the step interpretation
   82     -- | A generator of CekState to yield to client (e.g. TUI)
   83   | UpdateClientF (CekState uni fun ann) a
   84   deriving stock Functor
   85 
   86 -- | The monad that the driver operates in
   87 type Driving m uni fun ann bps =
   88     ( MonadReader (CekState uni fun ann) m -- the state of the debugger
   89     , MonadFree (DebugF uni fun ann bps) m -- the effects of the driver
   90     , Breakpointable ann bps
   91     )
   92 
   93 -- | Entrypoint of the driver
   94 runDriverT :: forall uni fun ann bps m.
   95             (Breakpointable ann bps, MonadFree (DebugF uni fun ann bps) m)
   96           => NTerm uni fun ann -> m ()
   97 runDriverT = void . runReaderT driver . initState
   98     where
   99       initState :: NTerm uni fun ann -> CekState uni fun ann
  100       initState = Starting
  101 
  102 -- | The driver action. The driver repeatedly:
  103 ---
  104 -- 1) waits for a `Cmd`
  105 -- 2) runs one or more CEK steps
  106 -- 3) informs the client for CEK updates&logs
  107 ---
  108 -- The driver computation exits when it reaches a CEK `Terminating` state.
  109 driver :: Driving m uni fun ann bps => m ()
  110 driver = inputF >>= \case
  111     -- Condition immediately satisfied
  112     Step -> multiStepUntil $ const True
  113 
  114     Continue bs -> multiStepUntil $ maybe False (`hasBreakpoints` bs) . cekStateAnn
  115 
  116     Next bs -> do
  117         startState <- ask
  118         multiStepUntil $ \curState ->
  119             maybe False (`hasBreakpoints` bs) (cekStateAnn curState)  ||
  120             -- has activation record length been restored?
  121             let leCtx = onCtxLen (<)
  122             in curState `leCtx` startState
  123 
  124     Finish bs -> do
  125         startState <- ask
  126         multiStepUntil $ \curState ->
  127             maybe False (`hasBreakpoints` bs) (cekStateAnn curState) ||
  128             -- has activation record length become smaller?
  129             let ltCtx = onCtxLen (<=)
  130             in curState `ltCtx` startState
  131   where
  132     -- | Comparison on states' contexts
  133     onCtxLen :: (state ~ CekState uni fun ann
  134                , ctxLen ~ Maybe Word -- `Maybe` because ctx can be missing if terminating
  135                )
  136              => (ctxLen -> ctxLen -> a)
  137              -> state -> state -> a
  138     onCtxLen = (`on` preview (cekStateContext . to lenContext))
  139 
  140 -- | Do one or more cek steps until Terminating state is reached or condition on 'CekState' is met.
  141 multiStepUntil :: Driving m uni fun ann bp
  142                => (CekState uni fun ann -> Bool) -> m ()
  143 multiStepUntil cond = do
  144     driverLogF "Driver is going to do a single step"
  145     newState <- stepF =<< ask
  146     case newState of
  147         Terminating{} ->
  148             -- don't recurse to driver, but EXIT the driver
  149             updateClientF newState
  150         _ -> -- update state
  151             local (const newState) $
  152                 if cond newState
  153                 then do
  154                     updateClientF newState
  155                     driver -- tail recurse
  156                 else
  157                     multiStepUntil cond -- tail recurse
  158 
  159 -- * boilerplate "suspension actions"
  160 -- Being in 'Driving' monad here is too constraining, but it does not matter.
  161 inputF :: Driving m uni fun ann bps => m (Cmd bps)
  162 inputF = liftF $ InputF id
  163 driverLogF :: Driving m uni fun ann bps => Text -> m ()
  164 driverLogF text = liftF $ DriverLogF text ()
  165 updateClientF :: Driving m uni fun ann bps => CekState uni fun ann -> m ()
  166 updateClientF dState = liftF $ UpdateClientF dState ()
  167 stepF :: Driving m uni fun ann bps => CekState uni fun ann -> m (CekState uni fun ann)
  168 stepF yieldState = liftF $ StepF yieldState id
