    5 module UntypedPlutusCore.Check.Scope (checkScope) where

    7 import Control.Lens hiding (index)
    8 import UntypedPlutusCore.Core.Type as UPLC
    9 import UntypedPlutusCore.DeBruijn as UPLC
   10 
   11 import Control.Monad (unless)
   12 import Control.Monad.Except (MonadError, throwError)
   13 
   14 {- | A pass to check that the input term:
   15 1) does not contain free variables and
   16 2) that all binders are set to debruijn index 0.
   17 
   18 Feeding the result of the debruijnification to this function is expected to pass.
   19 
   20 On the other hand, because of (2), this pass is
   21 stricter than the undebruijnification's (indirect)
   22 scope-checking, see Note [DeBruijn indices of Binders].
   23 
   24 Inlining this function makes a big difference,
   25 since it will usually be called in a context where all the type variables are known.
   26 That then means that GHC can optimize go locally in a completely monomorphic setting, which helps a lot.
   27 -}
   28 checkScope :: forall m name uni fun a.
   29              (HasIndex name, MonadError FreeVariableError m)
   30            => UPLC.Term name uni fun a
   31            -> m ()
   32 checkScope = go 0
   33   where
   34     -- the current level as a reader value
   35     go :: Word -> UPLC.Term name uni fun a -> m ()
   36     go !lvl = \case
   37         Var _ n       -> do
   38             let i = n ^. index
   39             -- var index must be larger than 0
   40             -- var index must be LEQ to the current level
   41             unless (i > 0 && fromIntegral i <= lvl) $
   42                 throwError $ FreeIndex i
   43         LamAbs _ binder t  -> do
   44             let bIx = binder^.index
   45             -- binder index must be equal to 0
   46             unless (bIx == 0) $
   47                 throwError $ FreeIndex bIx
   48             go (lvl+1) t
   49         Apply _ t1 t2 -> go lvl t1 >> go lvl t2
   50         Force _ t     -> go lvl t
   51         Delay _ t     -> go lvl t
   52         _             -> pure ()
   53 {-# INLINE checkScope #-}
