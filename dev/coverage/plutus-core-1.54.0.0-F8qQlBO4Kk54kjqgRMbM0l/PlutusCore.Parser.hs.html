   36 -- | A parsable PLC term.
   37 type PTerm = Term TyName Name DefaultUni DefaultFun SrcSpan
   38 
   39 varTerm :: Parser PTerm
   40 varTerm = withSpan $ \sp ->
   41     Var sp <$> name
   42 
   43 tyAbsTerm :: Parser PTerm
   44 tyAbsTerm = withSpan $ \sp ->
   45     inParens $ TyAbs sp <$> (symbol "abs" *> trailingWhitespace tyName)  <*> kind <*> term
   46 
   47 lamTerm :: Parser PTerm
   48 lamTerm = withSpan $ \sp ->
   49     inParens $ LamAbs sp <$> (symbol "lam" *> trailingWhitespace name) <*> pType <*> term
   50 
   51 appTerm :: Parser PTerm
   52 appTerm = withSpan $ \sp ->
   53     -- TODO: should not use the same `sp` for all arguments.
   54     inBrackets $ mkIterApp <$> term <*> (fmap (sp,) <$> some term)
   55 
   56 conTerm :: Parser PTerm
   57 conTerm = withSpan $ \sp ->
   58     inParens $ Constant sp <$> (symbol "con" *> constant)
   59 
   60 builtinTerm :: Parser PTerm
   61 builtinTerm = withSpan $ \sp ->
   62     inParens $ Builtin sp <$> (symbol "builtin" *> builtinFunction)
   63 
   64 tyInstTerm :: Parser PTerm
   65 tyInstTerm = withSpan $ \sp ->
   66     -- TODO: should not use the same `sp` for all arguments.
   67     inBraces $ mkIterInst <$> term <*> (fmap (sp,) <$> many pType)
   68 
   69 unwrapTerm :: Parser PTerm
   70 unwrapTerm = withSpan $ \sp ->
   71     inParens $ Unwrap sp <$> (symbol "unwrap" *> term)
   72 
   73 iwrapTerm :: Parser PTerm
   74 iwrapTerm = withSpan $ \sp ->
   75     inParens $ IWrap sp <$> (symbol "iwrap" *> pType) <*> pType <*> term
   76 
   77 errorTerm :: Parser PTerm
   78 errorTerm = withSpan $ \sp ->
   79     inParens $ Error sp <$> (symbol "error" *> pType)
   80 
   81 constrTerm :: Parser PTerm
   82 constrTerm = withSpan $ \sp ->
   83     inParens $ do
   84       let maxTag = fromIntegral (maxBound :: Word64)
   85       ty <- symbol "constr" *> pType
   86       tag :: Integer <- lexeme Lex.decimal
   87       args <- many term
   88       whenVersion (\v -> v < plcVersion110) $ fail "'constr' is not allowed before version 1.1.0"
   89       when (tag > maxTag) $ fail "constr tag too large: must be a legal Word64 value"
   90       pure $ Constr sp ty (fromIntegral tag) args
   91 
   92 caseTerm :: Parser PTerm
   93 caseTerm = withSpan $ \sp ->
   94     inParens $ do
   95       res <- Case sp <$> (symbol "case" *> pType) <*> term <*> many term
   96       whenVersion (\v -> v < plcVersion110) $ fail "'case' is not allowed before version 1.1.0"
   97       pure res
   98 
   99 -- | Parser for all PLC terms.
  100 term :: Parser PTerm
  101 term = leadingWhitespace go
  102   where
  103     go =
  104         choice $ map try
  105             [ tyAbsTerm
  106             , lamTerm
  107             , appTerm
  108             , conTerm
  109             , builtinTerm
  110             , tyInstTerm
  111             , unwrapTerm
  112             , iwrapTerm
  113             , errorTerm
  114             , varTerm
  115             , constrTerm
  116             , caseTerm
  117             ]

  119 -- | Parse a PLC program. The resulting program will have fresh names. The
  120 -- underlying monad must be capable of handling any parse errors.  This passes
  121 -- "test" to the parser as the name of the input stream; to supply a name
  122 -- explicity, use `parse program <name> <input>`.
  123 parseProgram ::
  124     (MonadError ParserErrorBundle m, MonadQuote m)
  125     => Text
  126     -> m (Program TyName Name DefaultUni DefaultFun SrcSpan)
  127 parseProgram = parseGen program
  128 
  129 -- | Parser for PLC programs.
  130 program :: Parser (Program TyName Name DefaultUni DefaultFun SrcSpan)
  131 program = leadingWhitespace go
  132   where
  133     go = do
  134         prog <- withSpan $ \sp -> inParens $ do
  135             v <- symbol "program" *> version
  136             withVersion v $ Program sp v <$> term
  137         notFollowedBy anySingle
  138         pure prog
  139 
  140 -- | Parse a PLC term. The resulting program will have fresh names. The underlying monad
  141 -- must be capable of handling any parse errors.
  142 parseTerm :: (MonadError ParserErrorBundle m, MonadQuote m) =>
  143     Text -> m (Term TyName Name DefaultUni DefaultFun SrcSpan)
  144 parseTerm = parseGen term

  146 -- | Parse a PLC type. The resulting program will have fresh names. The underlying monad
  147 -- must be capable of handling any parse errors.
  148 parseType :: (MonadError ParserErrorBundle m, MonadQuote m) =>

  150 parseType = parseGen pType
