   70 {- Note [Why newtype FakeNamedDeBruijn]
   71 We use a newtype wrapper to optimize away the expensive re-traversing of the deserialized Term
   72 for adding fake names everywhere --- the CEK works on names, but the scripts on ledger
   73 don't have names for size reduction.
   74 
   75 Specifically the expensive pipeline:
   76 
   77 ```
   78 decode @(Term DeBruijn)
   79 >>> term-Map-Names(FakeNamedDeBruijn)
   80 >>> cekExecute
   81 ```
   82 
   83 is optimized to to the faster:
   84 
   85 ```
   86 decode @(Term FakeNamedDeBruijn)
   87 >>> coerce @(Term FakeNamedDeBruijn) @(Term NamedDeBruijn)
   88 >>> cekExecute
   89 ```
   90 
   91 To achieve this we make sure:
   92 - to use `coerce` since its 0-cost
   93 - not to GeneralizeNewtypeDeriving the`Flat FakeNamedDeBruijn` instance, but "derive via"
   94 the optimized `Flat DeBruijn` instance. This is ok, because `FND<->D` are
   95 isomorphic.
   96 -}

  203 {- Note [Levels and indices]
  204 The indices ('Index') that we actually store as our de Bruijn indices in the program
  205 are *relative* - that is, they say how many levels above the *current* level to look for
  206 the binder.
  207 
  208 However, when doing conversions it is easier to record the  *absolute* level of a variable,
  209 in our state, since that way we don't have to adjust our mapping when we go under a binder
  210 (whereas for relative indices we would need to increment them all by one, as the current
  211 level has increased).
  212 
  213 However, this means that we *do* need to do an adjustment when we store an index as a level
  214 or extract a level to use it as an index. The adjustment is fairly straightforward:
  215 - An index `i` points to a binder `i` levels above (smaller than) the current level, so the level
  216   of `i` is `current - i`.
  217 - A level `l` which is `i` levels above (smaller than) the current level has an index of `i`, so it
  218   is also calculated as `current - l`.
  219 
  220 We use a newtype to keep these separate, since getting it wrong will lead to annoying bugs.
  221 -}
  222 
  223 -- | An absolute level in the program.
  224 newtype Level = Level Integer deriving newtype (Eq, Ord, Num, Real, Enum, Integral)

  226 {- | During visiting the AST we hold a reader "state" of current level and a current
  227 scoping (levelMapping).
  228 Invariant-A: the current level is positive and greater than all levels in the levelMapping.
  229 Invariant-B: only positive levels are stored in the levelMapping.
  230 -}
  231 data LevelInfo = LevelInfo
  232   { currentLevel :: Level
  233   , levelMapping :: BM.Bimap Unique Level
  234   }
  235 
  236 -- | Declare a name with a unique, recording the mapping to a 'Level'.
  237 declareUnique :: (MonadReader LevelInfo m, HasUnique name unique) => name -> m a -> m a
  238 declareUnique n =
  239   local $ \(LevelInfo current ls) -> LevelInfo current $ BM.insert (n ^. theUnique) current ls
  240 
  241 {- | Declares a new binder by assigning a fresh unique to the *current level*.
  242 Maintains invariant-B of 'LevelInfo' (that only positive levels are stored),
  243 since current level is always positive (invariant-A).
  244 See Note [DeBruijn indices of Binders]
  245 -}
  246 declareBinder :: (MonadReader LevelInfo m, MonadQuote m) => m a -> m a
  247 declareBinder act = do
  248   newU <- freshUnique
  249   local (\(LevelInfo current ls) -> LevelInfo current $ BM.insert newU current ls) act
  250 
  251 {- | Enter a scope, incrementing the current 'Level' by one
  252 Maintains invariant-A (that the current level is positive).
  253 -}
  254 withScope :: (MonadReader LevelInfo m) => m a -> m a
  255 withScope = local $ \(LevelInfo current ls) -> LevelInfo (current + 1) ls

  257 {- | We cannot do a correct translation to or from de Bruijn indices if the program is

  260 data FreeVariableError

  271 {- | Get the 'Index' corresponding to a given 'Unique'.
  272 Uses supplied handler for free names (uniques).
  273 -}
  274 getIndex :: (MonadReader LevelInfo m) => Unique -> (Unique -> m Index) -> m Index
  275 getIndex u h = do
  276   LevelInfo current ls <- ask
  277   case BM.lookup u ls of
  278     Just foundlvl -> pure $ levelToIx current foundlvl
  279     -- This call should return an index greater than the current level,
  280     -- otherwise it will map unbound variables to bound variables.
  281     Nothing       -> h u
  282   where
  283     -- Compute the relative 'Index' of a absolute 'Level' relative to the current 'Level'.
  284     levelToIx :: Level -> Level -> Index
  285     levelToIx (Level current) (Level foundLvl) =
  286       -- Thanks to invariant-A, we can be sure that 'level >= foundLvl ', since foundLvl
  287       -- is in the levelMapping and thus the computation 'current-foundLvl' is '>=0' and
  288       -- its conversion to Natural will not lead to arithmetic underflow.
  289       fromIntegral $ current - foundLvl
  290 
  291 {- | Get the 'Unique' corresponding to a given 'Index'.
  292 Uses supplied handler for free debruijn indices.
  293 -}
  294 getUnique :: (MonadReader LevelInfo m) => Index -> (Index -> m Unique) -> m Unique
  295 getUnique ix h = do
  296   LevelInfo current ls <- ask
  297   case BM.lookupR (ixToLevel current ix) ls of
  298     -- Because of invariant-B, the levelMapping contains only positive (absolute) levels.
  299     Just u -> pure u
  300     -- This call should return a free/unbound unique,
  301     -- otherwise it will map unbound variables to bound variables.
  302     Nothing ->
  303       -- the lookup failed, meaning the index corresponds to a strictly-negative
  304       -- (absolute) level.
  305       h ix

  359 {- | A different implementation of a handler,  where "free" debruijn indices do not throw an error
  360 but are instead gracefully converted to fresh uniques.
  361 These generated uniques remain free; i.e.  if the original term was open, it will remain open
  362 after applying this handler.
  363 These generated free uniques are consistent across the open term (by using a state cache).
  364 -}
  365 freeIndexAsConsistentLevel
  366   :: (MonadReader LevelInfo m, MonadState (M.Map Level Unique) m, MonadQuote m)
  367   => Index
  368   -> m Unique
  369 freeIndexAsConsistentLevel ix = do
  370   cache <- get
  371   LevelInfo current _ <- ask
  372   -- the absolute level is strictly-negative
  373   let absoluteLevel = ixToLevel current ix
  374   case M.lookup absoluteLevel cache of
  375     Nothing -> do
  376       u <- freshUnique
  377       -- the cache contains only strictly-negative levels
  378       put (M.insert absoluteLevel u cache)
  379       pure u
  380     Just u -> pure u
  381 
  382 -- Compute the absolute 'Level' of a relative 'Index' relative to the current 'Level'.
  383 -- The index `ixAST` may be malformed or point to a free variable because it comes straight
  384 -- from the AST; in such a case, this function may return a negative level.
  385 ixToLevel :: Level -> Index -> Level
  386 ixToLevel (Level current) ixAST = Level $ current - fromIntegral ixAST
