   18 -- | 'Either' as a PLC type.
   19 --
   20 -- > \(a b :: *) -> all (r :: *). (a -> r) -> (b -> r) -> r
   21 sum :: Type TyName uni ()
   22 sum = runQuote $ do
   23     a <- freshTyName "a"
   24     b <- freshTyName "b"
   25     r <- freshTyName "r"
   26     return
   27         . TyLam () a (Type ())
   28         . TyLam () b (Type ())
   29         . TyForall () r (Type ())
   30         . TyFun () (TyFun () (TyVar () a) $ TyVar () r)
   31         . TyFun () (TyFun () (TyVar () b) $ TyVar () r)
   32         $ TyVar () r
   33 
   34 -- | 'Left' as a PLC term.
   35 --
   36 -- > /\(a b :: *) -> \(x : a) -> /\(r :: *) -> \(f : a -> r) -> (g : b -> r) -> f x
   37 left :: TermLike term TyName Name uni fun => term ()
   38 left = runQuote $ do
   39     a <- freshTyName "a"
   40     b <- freshTyName "b"
   41     x <- freshName "x"
   42     r <- freshTyName "r"
   43     f <- freshName "f"
   44     g <- freshName "g"
   45     return
   46         . tyAbs  () a (Type ())
   47         . tyAbs  () b (Type ())
   48         . lamAbs () x (TyVar () a)
   49         . tyAbs  () r (Type ())
   50         . lamAbs () f (TyFun () (TyVar () a) $ TyVar () r)
   51         . lamAbs () g (TyFun () (TyVar () b) $ TyVar () r)
   52         . apply  () (var () f)
   53         $ var    () x
   54 
   55 -- | 'Right' as a PLC term.
   56 --
   57 -- > /\(a b :: *) -> \(y : b) -> /\(r :: *) -> \(f : a -> r) -> (g : b -> r) -> g y
   58 right :: TermLike term TyName Name uni fun => term ()
   59 right = runQuote $ do
   60     a <- freshTyName "a"
   61     b <- freshTyName "b"
   62     y <- freshName "y"
   63     r <- freshTyName "r"
   64     f <- freshName "f"
   65     g <- freshName "g"
   66     return
   67         . tyAbs  () a (Type ())
   68         . tyAbs  () b (Type ())
   69         . lamAbs () y (TyVar () b)
   70         . tyAbs  () r (Type ())
   71         . lamAbs () f (TyFun () (TyVar () a) $ TyVar () r)
   72         . lamAbs () g (TyFun () (TyVar () b) $ TyVar () r)
   73         . apply  () (var () g)
   74         $ var    () y
