   27 -- Note that we can't just use 'makeClassyPrisms' for 'EvaluationResult' as that would generate
   28 -- @_EvaluationSuccess@ as well as @_EvaluationFailure@, which would no longer be prismatic error
   29 -- handling: it'd constrain the monad, not the error in it.
   30 --
   31 -- We could define
   32 --
   33 --     data EvaluationFailure = EvaluationFailure
   34 --
   35 -- just for the purpose of using 'makeClassyPrisms' over it, but then it's clearer to write out
   36 -- the class definition manually than to hide it behind a TH function called over a redundant
   37 -- data type name-clashing with the useful 'EvaluationResult'.
   38 -- | A class for viewing errors as evaluation failures (in the sense of Plutus).
   39 class AsEvaluationFailure err where
   40     _EvaluationFailure :: Prism' err ()
   41 
   42 evaluationFailure :: AsEvaluationFailure err => err
   43 evaluationFailure = _EvaluationFailure # ()
   44 {-# INLINE evaluationFailure #-}
   45 
   46 -- | Construct a prism focusing on the @*EvaluationFailure@ part of @err@ by taking
   47 -- that @*EvaluationFailure@ and
   48 --
   49 -- 1. returning it for the setter part of the prism
   50 -- 2. checking the error for equality with @*EvaluationFailure@ for the opposite direction.
   51 _EvaluationFailureVia :: Eq err => err -> Prism' err ()
   52 _EvaluationFailureVia = only
   53 {-# INLINE _EvaluationFailureVia #-}

   55 -- | The parameterized type of results various evaluation engines return.
   56 -- On the PLC side this becomes (via @makeKnown@) either a call to 'Error' or
   57 -- a value of the PLC counterpart of type @a@.
   58 data EvaluationResult a
   59     = EvaluationSuccess !a
   60     | EvaluationFailure
   61     deriving stock (Show, Eq, Generic, Functor, Foldable, Traversable)
   62     deriving anyclass (NFData)
   63 
   64 -- >>> evaluationFailure :: EvaluationResult Bool
   65 -- EvaluationFailure
   66 --
   67 -- >>> import Control.Lens
   68 -- >>> matching _EvaluationFailure (EvaluationFailure :: EvaluationResult Bool)
   69 -- Right ()
   70 --
   71 -- >>> matching _EvaluationFailure $ EvaluationSuccess True
   72 -- Left (EvaluationSuccess True)
   73 instance AsEvaluationFailure (EvaluationResult a) where
   74     _EvaluationFailure = prism (const EvaluationFailure) $ \case
   75         a@EvaluationSuccess{} -> Left a
   76         EvaluationFailure     -> Right ()
   77     {-# INLINE _EvaluationFailure #-}
   78 
   79 -- This and the next one are two instances that allow us to write the following:
   80 --
   81 -- >>> import Control.Monad.Error.Lens
   82 -- >>> throwing_ _EvaluationFailure :: EvaluationResult Bool
   83 -- EvaluationFailure
   84 instance AsEvaluationFailure () where
   85     _EvaluationFailure = id
   86     {-# INLINE _EvaluationFailure #-}
   87 
   88 instance MonadError () EvaluationResult where
   89     throwError () = EvaluationFailure
   90     {-# INLINE throwError #-}
   91 
   92     catchError EvaluationFailure f = f ()
   93     catchError x                 _ = x
   94     {-# INLINE catchError #-}
   95 
   96 instance Applicative EvaluationResult where
   97     pure = EvaluationSuccess
   98     {-# INLINE pure #-}
   99 
  100     EvaluationSuccess f <*> a = fmap f a
  101     EvaluationFailure   <*> _ = EvaluationFailure
  102     {-# INLINE (<*>) #-}
  103 
  104     EvaluationSuccess _ *> b = b
  105     EvaluationFailure   *> _ = EvaluationFailure
  106     {-# INLINE (*>) #-}
  107 
  108 instance Monad EvaluationResult where
  109     EvaluationSuccess x >>= f = f x
  110     EvaluationFailure   >>= _ = EvaluationFailure
  111     {-# INLINE (>>=) #-}
  112 
  113     (>>) = (*>)
  114     {-# INLINE (>>) #-}
  115 
  116 instance Alternative EvaluationResult where
  117     empty = EvaluationFailure
  118     {-# INLINE empty #-}
  119 
  120     a@EvaluationSuccess{} <|> _ = a
  121     EvaluationFailure     <|> b = b
  122     {-# INLINE (<|>) #-}
  123 
  124 instance MonadFail EvaluationResult where
  125     fail _ = EvaluationFailure
  126     {-# INLINE fail #-}
  127 
  128 instance PrettyBy config a => PrettyBy config (EvaluationResult a) where
  129     prettyBy config (EvaluationSuccess x) = prettyBy config x
  130     prettyBy _      EvaluationFailure     = "Failure"
  131 
  132 instance PrettyClassic a => Pretty (EvaluationResult a) where
  133     pretty = prettyClassic

  135 -- | Check whether an 'EvaluationResult' is an 'EvaluationSuccess'.
  136 isEvaluationSuccess :: EvaluationResult a -> Bool
  137 isEvaluationSuccess = not . null
  138 
  139 -- | Check whether an 'EvaluationResult' is an 'EvaluationFailure'.
  140 isEvaluationFailure :: EvaluationResult a -> Bool
  141 isEvaluationFailure = null
