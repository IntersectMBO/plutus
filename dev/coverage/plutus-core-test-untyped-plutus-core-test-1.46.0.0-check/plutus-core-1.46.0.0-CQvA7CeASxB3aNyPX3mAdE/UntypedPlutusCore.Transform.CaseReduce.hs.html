   15 caseReduce
   16     :: Monad m
   17     => Term name uni fun a
   18     -> SimplifierT name uni fun a m (Term name uni fun a)
   19 caseReduce term = do
   20     let result = transformOf termSubterms processTerm term
   21     recordSimplification term CaseReduce result
   22     return result
   23 
   24 processTerm :: Term name uni fun a -> Term name uni fun a
   25 processTerm = \case
   26     Case ann (Constr _ i args) cs | Just c <- (V.!?) cs (fromIntegral i) ->
   27                                     mkIterApp c ((ann,) <$> args)
   28     t                                                     -> t
