   65 {-| Evaluate a term using the CEK machine with logging enabled and keep track of costing.
   66 A wrapper around the internal runCek to debruijn input and undebruijn output.
   67 *THIS FUNCTION IS PARTIAL if the input term contains free variables*
   68 -}
   69 runCek
   70     :: ThrowableBuiltins uni fun
   71     => MachineParameters CekMachineCosts fun (CekValue uni fun ann)
   72     -> ExBudgetMode cost uni fun
   73     -> EmitterMode uni fun
   74     -> Term Name uni fun ann
   75     -> (Either (CekEvaluationException Name uni fun) (Term Name uni fun ()), cost, [Text])
   76 runCek = Common.runCek runCekDeBruijn
