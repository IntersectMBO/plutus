    5 {- | A special case of case-of-case optimisation: transforms
    6 
    7 @
    8     case ((force ifThenElse) b (constr t) (constr f)) alts
    9 @
   10 
   11 into
   12 
   13 @
   14     force ifThenElse b (delay (case (constr t) alts)) (delay (case (constr f) alts))
   15 @
   16 
   17 This is always an improvement.
   18 -}
   19 module UntypedPlutusCore.Transform.CaseOfCase (caseOfCase) where
   20 
   21 import PlutusCore qualified as PLC
   22 import PlutusCore.MkPlc
   23 import UntypedPlutusCore.Core
   24 import UntypedPlutusCore.Transform.Simplifier (SimplifierStage (CaseOfCase), SimplifierT,
   25                                                recordSimplification)
   26 
   27 import Control.Lens
   28 
   29 caseOfCase
   30     :: fun ~ PLC.DefaultFun
   31     => Monad m
   32     => Term name uni fun a
   33     -> SimplifierT name uni fun a m (Term name uni fun a)
   34 caseOfCase term = do
   35   let result = transformOf termSubterms processTerm term
   36   recordSimplification term CaseOfCase result
   37   return result
   38 
   39 processTerm :: (fun ~ PLC.DefaultFun) => Term name uni fun a -> Term name uni fun a
   40 processTerm = \case
   41   Case ann scrut alts
   42     | ( ite@(Force a (Builtin _ PLC.IfThenElse))
   43         , [cond, (trueAnn, true@Constr{}), (falseAnn, false@Constr{})]
   44         ) <-
   45         splitApplication scrut ->
   46         Force a $
   47           mkIterApp
   48             ite
   49             [ cond
   50             , (trueAnn, Delay trueAnn (Case ann true alts))
   51             , (falseAnn, Delay falseAnn (Case ann false alts))
   52             ]
   53   other -> other
