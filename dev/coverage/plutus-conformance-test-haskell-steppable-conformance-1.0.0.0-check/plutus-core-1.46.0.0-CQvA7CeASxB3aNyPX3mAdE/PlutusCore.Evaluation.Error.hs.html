   30 {- | The type of errors that can occur during evaluation. There are two kinds of errors:
   31 
   32 1. Structural ones -- these are errors that are indicative of the _structure_ of the program being
   33    wrong. For example, a free variable was encountered during evaluation, a non-function was
   34    applied to an argument or 'tailList' was applied to a non-list.
   35 2. Operational ones -- these are errors that are indicative of the _logic_ of the program being
   36    wrong. For example, 'error' was executed, 'tailList' was applied to an empty list or evaluation
   37    ran out of gas.
   38 
   39 On the chain both of these are just regular failures and we don't distinguish between them there:
   40 if a script fails, it fails, it doesn't matter what the reason was. However in the tests it does
   41 matter why the failure occurred: a structural error may indicate that the test was written
   42 incorrectly while an operational error may be entirely expected.
   43 
   44 In other words, structural errors are \"runtime type errors\" and operational errors are regular
   45 runtime errors. Which means that evaluating an (erased) well-typed program should never produce a
   46 structural error, only an operational one. This creates a sort of \"runtime type system\" for UPLC
   47 and it would be great to stick to it and enforce in tests etc, but we currently don't.
   48 -}
   49 data EvaluationError structural operational
   50     = StructuralEvaluationError !structural
   51     | OperationalEvaluationError !operational
   52     deriving stock (Show, Eq, Functor, Generic)
   53     deriving anyclass (NFData)
   54 
   55 mtraverse makeClassyPrisms
   56     [ ''EvaluationError
   57     ]
   58 
   59 instance Bifunctor EvaluationError where
   60     bimap f _ (StructuralEvaluationError err)  = StructuralEvaluationError $ f err
   61     bimap _ g (OperationalEvaluationError err) = OperationalEvaluationError $ g err
   62     {-# INLINE bimap #-}
   63 
   64 instance Bifoldable EvaluationError where
   65     bifoldMap f _ (StructuralEvaluationError err)  = f err
   66     bifoldMap _ g (OperationalEvaluationError err) = g err
   67     {-# INLINE bifoldMap #-}
   68 
   69 instance Bitraversable EvaluationError where
   70     bitraverse f _ (StructuralEvaluationError err)  = StructuralEvaluationError <$> f err
   71     bitraverse _ g (OperationalEvaluationError err) = OperationalEvaluationError <$> g err
   72     {-# INLINE bitraverse #-}
   73 
   74 -- | A raw evaluation failure is always an operational error.
   75 instance AsEvaluationFailure operational =>
   76         AsEvaluationFailure (EvaluationError structural operational) where
   77     _EvaluationFailure = _OperationalEvaluationError . _EvaluationFailure
   78     {-# INLINE _EvaluationFailure #-}
   79 
   80 instance
   81         ( HasPrettyDefaults config ~ 'True
   82         , PrettyBy config structural, Pretty operational
   83         ) => PrettyBy config (EvaluationError structural operational) where
   84     prettyBy config (StructuralEvaluationError structural)   = prettyBy config structural
   85     prettyBy _      (OperationalEvaluationError operational) = pretty operational
   86 
   87 instance (Pretty structural, Pretty operational) =>
   88         Pretty (EvaluationError structural operational) where
   89     pretty (StructuralEvaluationError structural)   = pretty structural
   90     pretty (OperationalEvaluationError operational) = pretty operational
