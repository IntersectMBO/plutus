   14 mkFfiSimplifierTrace
   15   :: SimplifierTrace UPLC.Name UPLC.DefaultUni UPLC.DefaultFun a
   16   -> [(SimplifierStage, (FFI.UTerm, FFI.UTerm))]
   17 mkFfiSimplifierTrace (SimplifierTrace simplTrace) = reverse $ toFfiAst <$> simplTrace
   18   where
   19     toFfiAst Simplification {beforeAST, stage, afterAST} =
   20       case (UPLC.deBruijnTerm beforeAST, UPLC.deBruijnTerm afterAST) of
   21         (Right before', Right after')             ->
   22           (stage, (FFI.conv (void before'), FFI.conv (void after')))
   23         (Left (err :: UPLC.FreeVariableError), _) -> error $ show err
   24         (_, Left (err :: UPLC.FreeVariableError)) -> error $ show err
