  191 -- |Denotational equality for floating point numbers, checks bitwise equality.
  192 --
  193 --  NOTE: Denotational equality distinguishes NaNs, so its results may vary
  194 --        depending on the architecture and compilation flags. Unfortunately,
  195 --        this is a problem with floating-point numbers in general.
  196 --
  197 doubleDenotEq :: Double -> Double -> Bool
  198 doubleDenotEq = (==) `on` doubleToWord64
  199 
  200 -- |I guess "denotational orderings" are now a thing? The point is that we need
  201 --  an Ord instance which provides a total ordering, and is consistent with the
  202 --  denotational equality.
  203 --
  204 --  NOTE: The ordering induced via `doubleToWord64` is total, and is consistent
  205 --        with `doubleDenotEq`. However, it is *deeply* unintuitive. For one, it
  206 --        considers all negative numbers to be larger than positive numbers.
  207 --
  208 doubleDenotOrd :: Double -> Double -> Ordering
  209 doubleDenotOrd = compare `on` doubleToWord64

  218 -- |Decode a Double to an integer ratio.
  219 doubleToRatio :: Double -> (Integer, Integer)
  220 doubleToRatio x
  221   | isNaN      x = (0, 0)
  222   | isInfinite x = (signum (floor x), 0)
  223   | otherwise    = let r = toRational x in (numerator r, denominator r)

  225 -- |Encode an integer ratio as a double.
  226 ratioToDouble :: Integer -> Integer -> Double
  227 ratioToDouble n d
  228   | d == 0 = case compare n 0 of
  229       LT -> negativeInfinity
  230       EQ -> nan
  231       GT -> positiveInfinity
  232   | otherwise = fromRational (n % d)

  234 -- |Decode a Double to its mantissa and its exponent, normalised such that the
  235 --  mantissa is the smallest possible number without loss of accuracy.
  236 doubleDecode :: Double -> Maybe (Integer, Integer)
  237 doubleDecode x
  238   | isNaN      x = Nothing
  239   | isInfinite x = Nothing
  240   | otherwise    = Just (uncurry normalise (second toInteger (decodeFloat x)))
  241   where
  242     normalise :: Integer -> Integer -> (Integer, Integer)
  243     normalise mantissa exponent
  244       | even mantissa = normalise (mantissa `div` 2) (exponent + 1)
  245       | otherwise = (mantissa, exponent)

  261 -- |The smallest representable mantissa. Simultaneously, the smallest integer which can be
  262 --  represented as a Double without loss of precision.
  263 minMantissa :: Integer
  264 minMantissa = - maxMantissa
  265 
  266 -- |The largest representable mantissa. Simultaneously, the largest integer which can be
  267 --  represented as a Double without loss of precision.
  268 maxMantissa :: Integer
  269 maxMantissa = (doubleRadix ^ toInteger doubleDigits) - 1
  270 
  271 -- |The largest representable exponent.
  272 minExponent :: Integer
  273 minExponent = toInteger $ (fst doubleRange - doubleDigits) - 1
  274 
  275 -- |The smallest representable exponent.
  276 maxExponent :: Integer
  277 maxExponent = toInteger $ snd doubleRange - doubleDigits

  279 -- |Encode a mantissa and an exponent as a Double.
  280 doubleEncode :: Integer -> Integer -> Maybe Double
  281 doubleEncode mantissa exponent
  282   = if minMantissa <= mantissa && mantissa <= maxMantissa &&
  283        minExponent <= exponent && exponent <= maxExponent
  284     then Just (encodeFloat mantissa (fromInteger exponent))
  285     else Nothing
