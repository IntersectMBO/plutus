   23 usToHyphen :: String -> String
   24 usToHyphen = map (\c -> if c == '_' then '-' else c)
   25 
   26 -- | A class for types that can be unparsed to Agda code.
   27 class AgdaUnparse a where
   28   agdaUnparse :: a -> String
   29 
   30 instance AgdaUnparse AgdaFFI.UTerm where
   31   agdaUnparse =
   32     \case
   33       AgdaFFI.UVar n -> "(UVar " ++ agdaUnparse (fromInteger n :: Natural) ++ ")"
   34       AgdaFFI.ULambda term -> "(ULambda " ++ agdaUnparse term ++ ")"
   35       AgdaFFI.UApp t u -> "(UApp " ++ agdaUnparse t ++ " " ++ agdaUnparse u ++ ")"
   36       AgdaFFI.UCon someValue -> "(UCon " ++ agdaUnparse someValue ++ ")"
   37       AgdaFFI.UError -> "UError"
   38       AgdaFFI.UBuiltin fun -> "(UBuiltin " ++ agdaUnparse fun ++ ")"
   39       AgdaFFI.UDelay term -> "(UDelay " ++ agdaUnparse term ++ ")"
   40       AgdaFFI.UForce term -> "(UForce " ++ agdaUnparse term ++ ")"
   41       AgdaFFI.UConstr i terms -> "(UConstr " ++ agdaUnparse (fromInteger i :: Natural)
   42                                       ++ " " ++ agdaUnparse terms ++ ")"
   43       AgdaFFI.UCase term cases -> "(UCase " ++ agdaUnparse term ++ " " ++ agdaUnparse cases ++ ")"
   44 
   45 instance AgdaUnparse UPLC.DefaultFun where
   46   agdaUnparse = usToHyphen . lowerInitialChar . show
   47 
   48 instance AgdaUnparse SimplifierStage where
   49   agdaUnparse FloatDelay     = "floatDelayT"
   50   agdaUnparse ForceDelay     = "forceDelayT"
   51   agdaUnparse ForceCaseDelay = "forceCaseDelayT"
   52   agdaUnparse CaseOfCase     = "caseOfCaseT"
   53   agdaUnparse CaseReduce     = "caseReduceT"
   54   agdaUnparse Inline         = "inlineT"
   55   agdaUnparse CSE            = "cseT"
   56 
   57 instance AgdaUnparse Natural where
   58   agdaUnparse = show
   59 
   60 instance AgdaUnparse Integer where
   61   agdaUnparse x =
   62     case (x < 0) of
   63       True  -> "(ℤ.negsuc " ++ show (x - 1) ++ ")"
   64       False ->  "(ℤ.pos " ++ show x ++ ")"
   65 
   66 instance AgdaUnparse Bool where
   67   agdaUnparse True  = "true"
   68   agdaUnparse False = "false"
   69 
   70 instance AgdaUnparse Char where
   71   agdaUnparse c = show c
   72 instance AgdaUnparse Text where
   73   agdaUnparse = show . T.unpack
   74 
   75 instance AgdaUnparse ByteString where
   76   agdaUnparse bs = "(mkByteString " <> show bs <> ")"
   77 
   78 instance AgdaUnparse () where
   79   agdaUnparse _ = "tt"
   80 
   81 agdaUnfold :: (AgdaUnparse a , Foldable f) => f a -> String
   82 agdaUnfold l = "(" ++ foldr (\x xs -> agdaUnparse x ++ " ∷ " ++ xs) "[]" l ++ ")"
   83 
   84 instance AgdaUnparse a => AgdaUnparse [a] where
   85   agdaUnparse = agdaUnfold
   86 
   87 instance (AgdaUnparse a, AgdaUnparse b) => AgdaUnparse (a, b) where
   88   agdaUnparse (x, y) = "(" ++ agdaUnparse x ++ " , " ++ agdaUnparse y ++ ")"
   89 
   90 instance (AgdaUnparse a) => AgdaUnparse (Vector a) where
   91   agdaUnparse v = "(mkArray (" ++ agdaUnfold v ++ "))"
   92 
   93 instance AgdaUnparse Data where
   94   agdaUnparse (Data.Constr i args) =
   95     "(ConstrDATA " ++ agdaUnparse i ++ " " ++ agdaUnparse args ++ ")"
   96   agdaUnparse (Data.Map assocList) =
   97     "(MapDATA " ++ agdaUnparse assocList ++ ")"
   98   agdaUnparse (Data.List xs) =
   99     "(ListDATA " ++ agdaUnparse xs ++ ")"
  100   agdaUnparse (Data.I i) =
  101     "(iDATA " ++ agdaUnparse i ++ ")"
  102   agdaUnparse (Data.B b) =
  103     "(bDATA " ++ agdaUnparse b ++ ")"

  118 instance AgdaUnparse (UPLC.DefaultUni (PLC.Esc a)) where
  119   agdaUnparse PLC.DefaultUniInteger = "integer"
  120   agdaUnparse PLC.DefaultUniByteString = "bytestring"
  121   agdaUnparse PLC.DefaultUniString = "string"
  122   agdaUnparse PLC.DefaultUniBool = "bool"
  123   agdaUnparse PLC.DefaultUniUnit = "unit"
  124   agdaUnparse PLC.DefaultUniData = "pdata"
  125   agdaUnparse PLC.DefaultUniValue = "value"
  126   agdaUnparse (PLC.DefaultUniList t) =
  127     "(list " ++ agdaUnparse t ++ ")"
  128   agdaUnparse (PLC.DefaultUniPair t1 t2) =
  129     "(pair " ++ agdaUnparse t1 ++ " " ++ agdaUnparse t2 ++ ")"
  130   agdaUnparse PLC.DefaultUniBLS12_381_G1_Element = "bls12-381-g1-element"
  131   agdaUnparse PLC.DefaultUniBLS12_381_G2_Element = "bls12-381-g2-element"
  132   agdaUnparse PLC.DefaultUniBLS12_381_MlResult = "bls12-381-mlresult"
  133   agdaUnparse (PLC.DefaultUniArray t) =
  134     "(array " ++ agdaUnparse t ++ ")"
  135   agdaUnparse (PLC.DefaultUniApply _ _) = error "Application of an unknown type is not supported."
  136 
  137 instance AgdaUnparse (PLC.Some (PLC.ValueOf UPLC.DefaultUni)) where
  138   agdaUnparse (PLC.Some valOf) =
  139     "(tagCon " ++
  140       case valOf of
  141         PLC.ValueOf PLC.DefaultUniInteger val ->
  142           "integer " ++ agdaUnparse val
  143         PLC.ValueOf PLC.DefaultUniByteString val ->
  144           "bytestring " ++ agdaUnparse val
  145         PLC.ValueOf PLC.DefaultUniString val ->
  146           "string " ++ agdaUnparse val
  147         PLC.ValueOf PLC.DefaultUniBool val ->
  148           "bool " ++ agdaUnparse val
  149         PLC.ValueOf PLC.DefaultUniUnit _ ->
  150           "unit " ++ agdaUnparse ()
  151         PLC.ValueOf PLC.DefaultUniData val ->
  152           "pdata " ++ agdaUnparse val
  153         PLC.ValueOf PLC.DefaultUniValue val ->
  154           "value " ++ agdaUnparse val
  155         PLC.ValueOf univ@(PLC.DefaultUniList elemType) val ->
  156           agdaUnparse univ
  157           ++ " "
  158           ++
  159             ( PLC.bring (Proxy @AgdaUnparse) elemType
  160             $ agdaUnparse val
  161             )
  162         PLC.ValueOf univ@(PLC.DefaultUniPair type1 type2) val ->
  163           agdaUnparse univ
  164           ++ " "
  165           ++
  166             ( PLC.bring (Proxy @AgdaUnparse) type1
  167             $ PLC.bring (Proxy @AgdaUnparse) type2
  168             $ agdaUnparse val
  169             )
  170         PLC.ValueOf PLC.DefaultUniBLS12_381_G1_Element val ->
  171           "bls12-381-g1-element " ++  agdaUnparse val
  172         PLC.ValueOf PLC.DefaultUniBLS12_381_G2_Element val ->
  173           "bls12-381-g2-element " ++  agdaUnparse val
  174         PLC.ValueOf PLC.DefaultUniBLS12_381_MlResult val ->
  175           "bls12-381-mlresult " ++ agdaUnparse val
  176         PLC.ValueOf univ@(PLC.DefaultUniArray elemType)  val ->
  177           agdaUnparse univ
  178           ++ " "
  179           ++
  180             ( PLC.bring (Proxy @AgdaUnparse) elemType
  181             $ agdaUnparse val
  182             )
  183         PLC.ValueOf (PLC.DefaultUniApply _ _) _ ->
  184           error "Application of an unknown type is not supported."
  185     ++ ")"
