   73 -- #4999: Data.Text maps surrogate code points (\xD800 - \xDFFF) to the replacement character
   74 -- \xFFFD, so to keep strings isomorphic to list of characters we do the same for characters.
   75 natToChar :: Integer -> Char
   76 natToChar n | generalCategory c == Surrogate = '\xFFFD'
   77             | otherwise                      = c
   78   where c = toEnum $ fromIntegral $ mod n 0x110000
