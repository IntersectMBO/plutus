   13 -- Untyped (Raw) syntax
   14 
   15 data UTerm = UVar Integer
   16            | ULambda UTerm
   17            | UApp UTerm UTerm
   18            | UCon (Some (ValueOf DefaultUni))
   19            | UError
   20            | UBuiltin DefaultFun
   21            | UDelay UTerm
   22            | UForce UTerm
   23            | UConstr Integer [UTerm]
   24            | UCase UTerm [UTerm]
   25            deriving (Eq, Show)
   26 
   27 unIndex :: Index -> Integer
   28 unIndex (Index n) = toInteger n
   29 
   30 convP :: Program NamedDeBruijn DefaultUni DefaultFun a -> UTerm
   31 convP (Program _ _ t) = conv t
   32 
   33 conv :: Term NamedDeBruijn DefaultUni DefaultFun a -> UTerm
   34 conv (Var _ x)       = UVar (unIndex (ndbnIndex x) - 1)
   35 conv (LamAbs _ _ t)  = ULambda (conv t)
   36 conv (Apply _ t u)   = UApp (conv t) (conv u)
   37 conv (Builtin _ b)   = UBuiltin b
   38 conv (Constant _ c)  = UCon c
   39 conv (Error _)       = UError
   40 conv (Delay _ t)     = UDelay (conv t)
   41 conv (Force _ t)     = UForce (conv t)
   42 conv (Constr _ i es) = UConstr (toInteger i) (toList (fmap conv es))
   43 conv (Case _ arg cs) = UCase (conv arg) (toList (fmap conv cs))
   44 
   45 tmnames :: String
   46 tmnames = ['a' .. 'z']
   47 
   48 uconv ::  Int -> UTerm -> Term NamedDeBruijn DefaultUni DefaultFun ()
   49 uconv i (UVar x)     = Var
   50   ()
   51   (NamedDeBruijn (T.pack [tmnames !! (i - 1 - fromInteger x)])
   52                 -- PLC's debruijn starts counting from 1, while in the metatheory it starts from 0.
   53                  (Index (fromInteger x + 1)))
   54 uconv i (ULambda t)  = LamAbs
   55   ()
   56   (NamedDeBruijn (T.pack [tmnames !! i]) deBruijnInitIndex)
   57   (uconv (i+1) t)
   58 uconv i (UApp t u)     = Apply () (uconv i t) (uconv i u)
   59 uconv _ (UCon c)       = Constant () c
   60 uconv _ UError         = Error ()
   61 uconv _ (UBuiltin b)   = Builtin () b
   62 uconv i (UDelay t)     = Delay () (uconv i t)
   63 uconv i (UForce t)     = Force () (uconv i t)
   64 uconv i (UConstr j xs) = Constr () (fromInteger j) (fmap (uconv i) xs)
   65 uconv i (UCase t xs)   = Case () (uconv i t) (fromList (fmap (uconv i) xs))
