   41 prettyCertifierError :: CertifierError -> String
   42 prettyCertifierError (InvalidCertificate certDir) =
   43   "\n\nInvalid certificate: " <> certDir <>
   44   "\nThe compilation was not successfully certified. \
   45   \Please open a bug report at https://www.github.com/IntersectMBO/plutus \

   48   "\n\nInvalid compiler output: \
   49   \\nThe certifier was not able to process the trace produced by the compiler. \

   51   \containing a minimal program that when compiled reproduces the issue.\n"

   53   "\n\nInvalid certificate name: \

   68 -- | Run the Agda certifier on the simplification trace, if requested
   69 mkCertifier
   70   :: SimplifierTrace UPLC.Name UPLC.DefaultUni UPLC.DefaultFun a
   71   -- ^ The trace produced by the simplification process
   72   -> CertName
   73   -- ^ The name of the certificate to be produced
   74   -> Certifier CertifierSuccess
   75 mkCertifier simplTrace certName = do
   76   certName' <- validCertName certName
   77   let rawAgdaTrace = mkFfiSimplifierTrace simplTrace
   78   case runCertifierMain rawAgdaTrace of
   79     Just True -> do
   80       let cert = mkAgdaCertificateProject $ mkCertificate certName' rawAgdaTrace
   81       CertifierSuccess <$> writeCertificateProject cert
   82     Just False -> do
   83       let cert = mkAgdaCertificateProject $ mkCertificate certName' rawAgdaTrace
   84       certDir <- writeCertificateProject cert
   85       throwError $ InvalidCertificate certDir
   86     Nothing -> throwError InvalidCompilerOutput

   88 validCertName :: String -> Certifier String
   89 validCertName [] = throwError $ ValidationError []
   90 validCertName name@(fstC : rest) =
   91   if all isValidChar name
   92     then pure (toUpper fstC : rest)
   93     else throwError $ ValidationError name
   94   where
   95     isValidChar c = c `elem` ['a'..'z'] ++ ['A'..'Z'] ++ ['0'..'9'] ++ "_-"

  109 getTermId :: Ast -> Int

  112 data Certificate = Certificate
  113   { certName     :: String
  114   , certTrace    :: [(SimplifierStage, (Ast, Ast))]
  115   , certReprAsts :: [Ast]
  116   }
  117 
  118 mkCertificate :: String -> [(SimplifierStage, (UTerm, UTerm))] -> Certificate
  119 mkCertificate certName rawTrace =
  120   let traceWithIds = addIds rawTrace
  121       allTermWithIds = extractTermWithIds traceWithIds
  122       groupedAsts = findEquivClasses allTermWithIds
  123       allAsts = groupedAsts >>= NE.toList
  124       certTrace = mkAstTrace allAsts traceWithIds
  125       certReprAsts = getRepresentatives groupedAsts
  126    in Certificate
  127         { certName
  128         , certTrace
  129         , certReprAsts
  130         }
  131   where
  132     addIds
  133       :: [(SimplifierStage, (UTerm, UTerm))]
  134       -> [(SimplifierStage, (TermWithId, TermWithId))]
  135     addIds = go 0
  136       where
  137         go
  138           :: Int
  139           -> [(SimplifierStage, (UTerm, UTerm))]
  140           -> [(SimplifierStage, (TermWithId, TermWithId))]
  141         go _ [] = []
  142         go id' ((stage, (before, after)) : rest) =
  143           let beforeWithId = TermWithId id' before
  144               afterWithId = TermWithId (id' + 1) after
  145            in (stage, (beforeWithId, afterWithId)) : go (id' + 2) rest
  146 
  147     extractTermWithIds
  148       :: [(SimplifierStage, (TermWithId, TermWithId))]
  149       -> [TermWithId]
  150     extractTermWithIds = concatMap (\(_, (before, after)) -> [before, after])
  151 
  152     findEquivClasses :: [TermWithId] -> [NonEmpty Ast]
  153     findEquivClasses =
  154       go 0 . NE.groupBy (\x y -> term x == term y)
  155       where
  156         go :: EquivClass -> [NonEmpty TermWithId] -> [NonEmpty Ast]
  157         go _ [] = []
  158         go cl (ts : rest) =
  159           let asts = fmap (\t -> Ast {astTermWithId = t, equivClass = cl}) ts
  160            in asts : go (cl + 1) rest
  161 
  162     getRepresentatives :: [NonEmpty Ast] -> [Ast]
  163     getRepresentatives = fmap NE.head
  164 
  165     errorMessage :: String
  166     errorMessage =
  167       "Internal error: could not find AST.\
  168       \ This is an issue in the certifier, please open a bug report at\
  169       \ https://github.com/IntersectMBO/plutus/issues"
  170 
  171     mkAstTrace
  172       :: [Ast]
  173       -> [(SimplifierStage, (TermWithId, TermWithId))]
  174       -> [(SimplifierStage, (Ast, Ast))]
  175     mkAstTrace _ [] = []
  176     mkAstTrace allAsts ((stage, (rawBefore, rawAfter)) : rest) =
  177       let processedBefore =
  178             fromMaybe (error errorMessage)
  179             $ find (\ast -> getTermId ast == termId rawBefore) allAsts
  180           processedAfter =
  181             fromMaybe (error errorMessage)
  182             $ find (\ast -> getTermId ast == termId rawAfter) allAsts
  183        in (stage, (processedBefore, processedAfter)) : mkAstTrace allAsts rest

  185 mkAstModuleName :: Ast -> String
  186 mkAstModuleName Ast { equivClass } =
  187   "Ast" <> show equivClass

  189 mkAgdaAstFile :: Ast -> (FilePath, String)
  190 mkAgdaAstFile ast =
  191   let agdaAst = agdaUnparse (term . astTermWithId $ ast)
  192       agdaId = equivClass ast
  193       agdaModuleName = mkAstModuleName ast
  194       agdaIdStr = "ast" <> show agdaId
  195       agdaAstTy = agdaIdStr <> " : Untyped"
  196       agdaAstDef = agdaIdStr <> " = " <> agdaAst
  197       agdaAstFile = agdaModuleName <> ".agda"
  198    in (agdaAstFile, mkAstModule agdaModuleName agdaAstTy agdaAstDef)

  230 instance AgdaUnparse AgdaVar where

  233 mkCertificateFile :: Certificate -> (FilePath, String)
  234 mkCertificateFile Certificate { certName, certTrace, certReprAsts } =
  235   let imports = fmap (mkAgdaOpenImport . mkAstModuleName) certReprAsts
  236       agdaTrace =
  237         agdaUnparse
  238         $ (\(st, (ast1, ast2)) ->
  239             (st
  240             , (AgdaVar $ "ast" <> (show . equivClass) ast1
  241               , AgdaVar $ "ast" <> (show . equivClass) ast2
  242               )
  243             )
  244           )
  245         <$> certTrace
  246       certFile = certName <> ".agda"
  247    in (certFile, mkCertificateModule certName agdaTrace imports)

  278 data AgdaCertificateProject = AgdaCertificateProject

  285 mkAgdaLib :: String -> (FilePath, String)
  286 mkAgdaLib name =
  287   let contents =
  288         "name: " <> name <>
  289         "\ndepend:\
  290         \\n  standard-library-2.1.1\
  291         \\n  plutus-metatheory\
  292         \\ninclude: src"
  293    in (name <> ".agda-lib", contents)

  295 mkAgdaCertificateProject
  296   :: Certificate
  297   -> AgdaCertificateProject
  298 mkAgdaCertificateProject cert =
  299   let name = certName cert
  300       mainModule = mkCertificateFile cert
  301       astModules = fmap mkAgdaAstFile (certReprAsts cert)
  302       projectDir = name
  303       agdalib = mkAgdaLib name
  304    in AgdaCertificateProject { mainModule, astModules, projectDir, agdalib }
  305 
  306 writeCertificateProject
  307   :: AgdaCertificateProject
  308   -> Certifier CertDir
  309 writeCertificateProject
  310   AgdaCertificateProject
  311     { mainModule, astModules, projectDir, agdalib }
  312   = liftIO $ do
  313       let (mainModulePath, mainModuleContents) = mainModule
  314           (agdalibPath, agdalibContents) = agdalib
  315       time <- systemNanoseconds <$> getSystemTime
  316       let actualProjectDir = projectDir <> "-" <> show time
  317       createDirectory actualProjectDir
  318       createDirectory (actualProjectDir </> "src")
  319       writeFile (actualProjectDir </> "src" </> mainModulePath) mainModuleContents
  320       writeFile (actualProjectDir </> agdalibPath) agdalibContents
  321       mapM_ (\(path, contents) ->
  322         writeFile (actualProjectDir </> "src" </> path) contents) astModules
  323       pure actualProjectDir
