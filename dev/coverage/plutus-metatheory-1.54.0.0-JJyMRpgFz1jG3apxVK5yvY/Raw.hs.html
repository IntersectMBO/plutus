   20 data KIND = Star | Sharp | Arrow KIND KIND
   21            deriving Show

   23 data RType = RTyVar Integer
   24            | RTyFun RType RType
   25            | RTyPi KIND RType
   26            | RTyLambda KIND RType
   27            | RTyApp RType RType
   28            | RTyCon RTyCon
   29            | RTyMu RType RType
   30            | RTySOP [[RType]]
   31            deriving Show

   51 data RTerm = RVar Integer
   52            | RTLambda KIND RTerm
   53            | RTApp RTerm RType
   54            | RLambda RType RTerm
   55            | RApp RTerm RTerm
   56            | RCon (Some (ValueOf DefaultUni))
   57            | RError RType
   58            | RBuiltin DefaultFun
   59            | RWrap RType RType RTerm
   60            | RUnWrap RTerm
   61            | RConstr RType Integer [RTerm]
   62            | RCase RType RTerm [RTerm]
   63   deriving Show
   64 
   65 unIndex :: Index -> Integer
   66 unIndex (Index n) = toInteger n

   68 convP :: Program NamedTyDeBruijn NamedDeBruijn DefaultUni DefaultFun a -> RTerm
   69 convP (Program _ _ t) = conv t

   71 convK :: Kind a -> KIND
   72 convK (Type _)           = Star
   73 convK (KindArrow _ k k') = Arrow (convK k) (convK k')
   74 
   75 convT :: Type NamedTyDeBruijn DefaultUni a -> RType
   76 convT (TyVar _ (NamedTyDeBruijn x)) = RTyVar (unIndex (ndbnIndex x))
   77 convT (TyFun _ _A _B)               = RTyFun (convT _A) (convT _B)
   78 convT (TyForall _ _ _K _A)          = RTyPi (convK _K) (convT _A)
   79 convT (TyLam _ _ _K _A)             = RTyLambda (convK _K) (convT _A)
   80 convT (TyApp _ _A _B)               = RTyApp (convT _A) (convT _B)
   81 convT (TyBuiltin ann (SomeTypeIn (DefaultUniApply f x))) =
   82      RTyApp (convT (TyBuiltin ann (SomeTypeIn f)))
   83             (convT (TyBuiltin ann (SomeTypeIn x)))
   84 convT (TyBuiltin _ someUni)         = convTyCon someUni
   85 convT (TyIFix _ a b)                = RTyMu (convT a) (convT b)
   86 convT (TySOP _ xss)                 = RTySOP (map (map convT) xss)
   87 
   88 convTyCon :: SomeTypeIn DefaultUni -> RType
   89 convTyCon (SomeTypeIn DefaultUniInteger)              = RTyCon (RTyConAtom ATyConInt)
   90 convTyCon (SomeTypeIn DefaultUniByteString)           = RTyCon (RTyConAtom ATyConBS)
   91 convTyCon (SomeTypeIn DefaultUniString)               = RTyCon (RTyConAtom ATyConStr)
   92 convTyCon (SomeTypeIn DefaultUniBool)                 = RTyCon (RTyConAtom ATyConBool)
   93 convTyCon (SomeTypeIn DefaultUniUnit)                 = RTyCon (RTyConAtom ATyConUnit)
   94 convTyCon (SomeTypeIn DefaultUniData)                 = RTyCon (RTyConAtom ATyConData)
   95 convTyCon (SomeTypeIn DefaultUniBLS12_381_G1_Element) = RTyCon (RTyConAtom ATyConBLS12_381_G1_Element)
   96 convTyCon (SomeTypeIn DefaultUniBLS12_381_G2_Element) = RTyCon (RTyConAtom ATyConBLS12_381_G2_Element)
   97 convTyCon (SomeTypeIn DefaultUniBLS12_381_MlResult)   = RTyCon (RTyConAtom ATyConBLS12_381_MlResult)
   98 convTyCon (SomeTypeIn DefaultUniProtoList)            = RTyCon RTyConList
   99 convTyCon (SomeTypeIn DefaultUniProtoArray)           = RTyCon RTyConArray
  100 convTyCon (SomeTypeIn DefaultUniProtoPair)            = RTyCon RTyConPair
  101 convTyCon (SomeTypeIn (DefaultUniApply _ _))          = error "unsupported builtin type application"

  103 conv :: Term NamedTyDeBruijn NamedDeBruijn DefaultUni DefaultFun a -> RTerm
  104 conv (Var _ x)           = RVar (unIndex (ndbnIndex x))
  105 conv (TyAbs _ _ _K t)    = RTLambda (convK _K) (conv t)
  106 conv (TyInst _ t _A)     = RTApp (conv t) (convT _A)
  107 conv (LamAbs _ _ _A t)   = RLambda (convT _A) (conv t)
  108 conv (Apply _ t u)       = RApp (conv t) (conv u)
  109 conv (Builtin _ b)       = RBuiltin b
  110 conv (Constant _ c)      = RCon c
  111 conv (Unwrap _ t)        = RUnWrap (conv t)
  112 conv (IWrap _ ty1 ty2 t) = RWrap (convT ty1) (convT ty2) (conv t)
  113 conv (Error _ _A)        = RError (convT _A)
  114 conv (Constr _ _A i cs)  = RConstr (convT _A) (toInteger i) (fmap conv cs)
  115 conv (Case _ _A arg cs)  = RCase (convT _A) (conv arg) (fmap conv cs)

  117 varTm :: Int -> NamedDeBruijn
  118 varTm i = NamedDeBruijn (T.pack [tmnames !! i]) deBruijnInitIndex
  119 
  120 varTy :: Int -> NamedDeBruijn
  121 varTy i = NamedDeBruijn (T.pack [tynames !! i]) deBruijnInitIndex
  122 
  123 unconvK :: KIND -> Kind ()
  124 unconvK (Arrow k k') = KindArrow () (unconvK k) (unconvK k')
  125 unconvK _            = Type ()

  127 -- this should take a level and render levels as names
  128 unconvT :: Int -> RType -> Type NamedTyDeBruijn DefaultUni ()
  129 unconvT i (RTyVar x)        =
  130   TyVar () (NamedTyDeBruijn (NamedDeBruijn (T.pack [tynames !! (i - fromIntegral x)]) (Index (fromInteger x))))
  131 unconvT i (RTyFun t u)      = TyFun () (unconvT i t) (unconvT i u)
  132 unconvT i (RTyPi k t)       =
  133   TyForall () (NamedTyDeBruijn (varTy i)) (unconvK k) (unconvT (i+1) t)
  134 unconvT i (RTyLambda k t) = TyLam () (NamedTyDeBruijn (varTy i)) (unconvK k) (unconvT (i+1) t)
  135 
  136 unconvT i (RTyApp t u)      = TyApp () (unconvT i t) (unconvT i u)
  137 unconvT _ (RTyCon c)        = TyBuiltin () (unconvTyCon c)
  138 unconvT i (RTyMu t u)       = TyIFix () (unconvT i t) (unconvT i u)
  139 unconvT i (RTySOP xss)      = TySOP () (map (map (unconvT i)) xss)
  140 
  141 unconvTyCon :: RTyCon -> SomeTypeIn DefaultUni
  142 unconvTyCon (RTyConAtom ATyConInt)  = SomeTypeIn DefaultUniInteger
  143 unconvTyCon (RTyConAtom ATyConBS)   = SomeTypeIn DefaultUniByteString
  144 unconvTyCon (RTyConAtom ATyConStr)  = SomeTypeIn DefaultUniString
  145 unconvTyCon (RTyConAtom ATyConBool) = SomeTypeIn DefaultUniBool
  146 unconvTyCon (RTyConAtom ATyConUnit) = SomeTypeIn DefaultUniUnit
  147 unconvTyCon (RTyConAtom ATyConData) = SomeTypeIn DefaultUniData
  148 unconvTyCon (RTyConAtom ATyConBLS12_381_G1_Element)
  149                                     = SomeTypeIn DefaultUniBLS12_381_G1_Element
  150 unconvTyCon (RTyConAtom ATyConBLS12_381_G2_Element)
  151                                     = SomeTypeIn DefaultUniBLS12_381_G2_Element
  152 unconvTyCon (RTyConAtom ATyConBLS12_381_MlResult)
  153                                     = SomeTypeIn DefaultUniBLS12_381_MlResult
  154 unconvTyCon RTyConList              = SomeTypeIn DefaultUniProtoList
  155 unconvTyCon RTyConArray             = SomeTypeIn DefaultUniProtoArray
  156 unconvTyCon RTyConPair              = SomeTypeIn DefaultUniProtoPair

  164 unconv :: Int -> RTerm -> Term NamedTyDeBruijn NamedDeBruijn DefaultUni DefaultFun ()
  165 unconv i (RVar x)          =
  166   Var () (NamedDeBruijn (T.pack [tmnames !! (i - fromInteger x )]) (Index (fromInteger x)))
  167 unconv i (RTLambda k tm)   = TyAbs () (NamedTyDeBruijn (varTy i)) (unconvK k) (unconv (i+1) tm)
  168 unconv i (RTApp t ty)      = TyInst () (unconv i t) (unconvT i ty)
  169 unconv i (RLambda ty tm)   = LamAbs () (varTm i) (unconvT (i+1) ty) (unconv (i+1) tm)
  170 unconv i (RApp t u)        = Apply () (unconv i t) (unconv i u)
  171 unconv _ (RCon c)          = Constant () c
  172 unconv i (RError ty)       = Error () (unconvT i ty)
  173 unconv _ (RBuiltin b)      = Builtin () b
  174 unconv i (RWrap tyA tyB t) = IWrap () (unconvT i tyA) (unconvT i tyB) (unconv i t)
  175 unconv i (RUnWrap t)       = Unwrap () (unconv i t)
  176 unconv i (RConstr ty j cs) = Constr () (unconvT i ty) (fromInteger j) (fmap (unconv i) cs)
  177 unconv i (RCase ty arg cs) = Case () (unconvT i ty) (unconv i arg) (fmap (unconv i) cs)
