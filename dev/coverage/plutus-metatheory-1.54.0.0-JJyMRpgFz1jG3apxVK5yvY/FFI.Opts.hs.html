   18 -- the different budget modes of plc-agda
   19 data BudgetMode a = Silent
   20                  | Counting a
   21                  | Tallying a
   22       deriving Functor
   23 
   24 countingbudget :: Parser (BudgetMode ())
   25 countingbudget = flag' (Counting ())
   26                  (  long "counting"
   27                  <> short 'c'
   28                  <> help "Run machine in counting mode and report results" )
   29 
   30 tallyingbudget :: Parser (BudgetMode ())
   31 tallyingbudget = flag' (Tallying ())
   32                  (  long "tallying"
   33                  <> short 't'
   34                  <> help "Run machine in tallying mode and report results" )
   35 
   36 budgetmode :: Parser (BudgetMode ())
   37 budgetmode = asum
   38     [ countingbudget
   39     , tallyingbudget
   40     , pure Silent
   41     ]
   42 
   43 -- The different evaluation modes of plc-agda
   44 data EvalMode = U | TL | TCK | TCEK deriving stock (Show, Read)
   45 
   46 data EvalOptions a = EvalOpts Input Format EvalMode (BudgetMode a)
   47   deriving Functor
   48 
   49 evalMode :: Parser EvalMode
   50 evalMode = option auto
   51   (  long "mode"
   52   <> short 'm'
   53   <> metavar "MODE"
   54   <> value TL
   55   <> showDefault
   56   <> help "Evaluation mode (U , TL, TCK, TCEK)" )
   57 
   58 evalOpts :: Parser (EvalOptions ())
   59 evalOpts = EvalOpts <$> input <*> inputformat <*> evalMode <*> budgetmode
   60 
   61 data TypecheckOptions = TCOpts Input Format
   62 
   63 typecheckOpts :: Parser TypecheckOptions
   64 typecheckOpts = TCOpts <$> input <*> inputformat
   65 
   66 data Command a = Eval (EvalOptions a)
   67              | Typecheck TypecheckOptions
   68           deriving Functor
   69 
   70 commands :: Parser (Command ())
   71 commands = hsubparser (
   72          command "evaluate"
   73           (info (Eval <$> evalOpts)
   74           (fullDesc <> progDesc "run a Plutus Core program"))
   75       <> command "typecheck"
   76           (info (Typecheck <$> typecheckOpts)
   77           (fullDesc <> progDesc "typecheck a Plutus Core program")))

   79 -- A CostModel has all the information to run the Agda machine
   80 -- with cost reporting
   81 type CostModel = (CekMachineCosts , BuiltinCostMap)

   83 addJSONParameters :: Command a -> Command CostModel
   84 addJSONParameters = fmap (const (defaultCekMachineCostsForTesting, defaultSimpleBuiltinCostModel))

   86 execP :: IO (Command CostModel)
