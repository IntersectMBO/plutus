   19 {-| We need to account for the costs of evaluator steps and also built-in function
   20    evaluation.  The models for these have different structures and are used in
   21    different parts of the code, so inside the valuator we pass separate objects
   22    about most of the time .  It's convenient for clients of the evaluator to
   23    only have to worry about a single object, so the CostModel type bundles the
   24    two together.  We could conceivably have different evaluators with different
   25    internal costs, so we keep the machine costs abstract.  The model for Cek
   26    machine steps is in UntypedPlutusCore.Evaluation.Machine.Cek.CekMachineCosts.
   27 -}
   28 data CostModel machinecosts builtincosts =
   29     CostModel {
   30       _machineCostModel :: machinecosts
   31     , _builtinCostModel :: builtincosts
   32     } deriving stock (Eq, Show)
   33 makeLenses ''CostModel
   34 
   35 {-| At execution time we need a 'BuiltinsRuntime' object which includes both the
   36   cost model for builtins and their denotations.  This bundles one of those
   37   together with the cost model for evaluator steps.  The 'term' type will be
   38   CekValue when we're using this with the CEK machine. -}
   39 data MachineParameters machinecosts fun val =
   40     MachineParameters {
   41       machineCosts    :: machinecosts
   42     , builtinsRuntime :: BuiltinsRuntime fun val
   43     }
   44     deriving stock Generic
   45     deriving anyclass (NFData)
   46 
   47 -- For some reason the generic instance gives incorrect nothunk errors,
   48 -- see https://github.com/input-output-hk/nothunks/issues/24
   49 instance (NoThunks machinecosts, Bounded fun, Enum fun) => NoThunks (MachineParameters machinecosts fun val) where
   50   wNoThunks ctx (MachineParameters costs runtime) = allNoThunks [ noThunks ctx costs, noThunks ctx runtime ]
   51 
   52 {- Note [The CostingPart constraint in mkMachineParameters]
   53 Discharging the @CostingPart uni fun ~ builtincosts@ constraint in 'mkMachineParameters' causes GHC
   54 to fail to inline the function at its call site regardless of the @INLINE@ pragma and an explicit
   55 'inline' call.
   56 
   57 We think that this is because discharging the @CostingPart uni fun ~ builtincosts@ constraint in the
   58 type of 'mkMachineParameters' somehow causes it to be compiled into @expr `Cast@` co@ , where @co@'s
   59 type is
   60 
   61     (CostingPart DefaultUni DefaultFun) ... ~R# ... BuiltinCostModel ...
   62 
   63 and this fails to inline, because in order for @inline f@ to work, @f@ must be compiled into either
   64 a @Var@, or an @App@ whose head is a @Var@, according to https://gitlab.haskell.org/ghc/ghc/-/blob/1f02b7430b2fbab403d7ffdde9cfd006e884678e/compiler/prelude/PrelRules.hs#L1442-1449
   65 And if @f@ is compiled into a @Cast@ like 'mkMachineParameters' with the discharged constraint, then
   66 inlining won't work. We don't know why it's implemented this way in GHC.
   67 
   68 It seems fully applying @f@ helps, i.e.
   69 
   70     - inline mkMachineParameters unlMode <$> <...>
   71     + (\x -> inline (mkMachineParameters unlMode x)) <$> <...>
   72 
   73 which makes sense: if @f@ receives all its type and term args then there's less reason to throw a
   74 @Cast@ in there.
   75 -}
   76 
   77 -- See Note [Inlining meanings of builtins].
   78 {-| This just uses 'toBuiltinsRuntime' function to convert a BuiltinCostModel to a BuiltinsRuntime. -}
   79 mkMachineParameters ::
   80     ( -- WARNING: do not discharge the equality constraint as that causes GHC to fail to inline the
   81       -- function at its call site, see Note [The CostingPart constraint in mkMachineParameters].
   82       CostingPart uni fun ~ builtincosts
   83     , HasMeaningIn uni val
   84     , ToBuiltinMeaning uni fun
   85     )
   86     => BuiltinSemanticsVariant fun
   87     -> CostModel machinecosts builtincosts
   88     -> MachineParameters machinecosts fun val
   89 mkMachineParameters semvar (CostModel mchnCosts builtinCosts) =
   90     MachineParameters mchnCosts (inline toBuiltinsRuntime semvar builtinCosts)
   91 {-# INLINE mkMachineParameters #-}
