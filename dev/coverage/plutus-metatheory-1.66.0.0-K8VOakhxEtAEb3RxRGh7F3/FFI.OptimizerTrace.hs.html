   25 -- A certifier trace is a non-empty list of asts of type `a`, separated by the
   26 -- optimizer pass that ran and the hints that were emitted
   27 type Trace a = NonEmptySep (OptStage, Hints) a
   28 
   29 mkFfiOptimizerTrace
   30   :: OptimizerTrace UPLC.Name UPLC.DefaultUni UPLC.DefaultFun a
   31   -> Trace FFI.UTerm
   32 mkFfiOptimizerTrace (OptimizerTrace simplNonEmptySep) = go (reverse simplNonEmptySep)
   33   where
   34     go
   35       :: [Optimization UPLC.Name UPLC.DefaultUni UPLC.DefaultFun a]
   36       -> Trace FFI.UTerm
   37     go [] = error "Empty trace"
   38     go [Optimization before stage hints after] =
   39       case (UPLC.deBruijnTerm before, UPLC.deBruijnTerm after) of
   40         (Right before', Right after') ->
   41           Cons
   42             (FFI.conv (void before'))
   43             (stage, hints)
   44             (Singleton (FFI.conv (void after')))
   45         (Left (err :: UPLC.FreeVariableError), _) -> error $ show err
   46         (_, Left (err :: UPLC.FreeVariableError)) -> error $ show err
   47     -- ignore _after, it should be equal to subsequent before
   48     go (Optimization before stage hints _after : xs) =
   49       case UPLC.deBruijnTerm before of
   50         Right before' ->
   51           Cons (FFI.conv (void before')) (stage, hints) (go xs)
   52         Left (err :: UPLC.FreeVariableError) -> error $ show err

   54 toEvalResult
   55   :: Maybe (CekEvaluationException UPLC.NamedDeBruijn UPLC.DefaultUni UPLC.DefaultFun)
   56   -> ExBudget
   57   -> EvalResult
   58 toEvalResult res budget = case res of
   59   Just err -> EvalFailure (T.pack $ show err) cpu mem
   60   Nothing -> EvalSuccess cpu mem
   61   where
   62     cpu = fromSatInt $ coerce (exBudgetCPU budget)
   63     mem = fromSatInt $ coerce (exBudgetMemory budget)
