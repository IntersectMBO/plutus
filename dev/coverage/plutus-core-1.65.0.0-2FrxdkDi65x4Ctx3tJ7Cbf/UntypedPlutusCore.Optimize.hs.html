   45 optimizeProgram
   46   :: forall name uni fun m a
   47    . Compiling m uni fun name a
   48   => OptimizeOpts name a
   49   -> BuiltinSemanticsVariant fun
   50   -> Program name uni fun a
   51   -> m (Program name uni fun a)
   52 optimizeProgram opts builtinSemanticsVariant (Program a v t) =
   53   Program a v <$> optimizeTerm opts builtinSemanticsVariant t
   54 
   55 optimizeProgramWithTrace
   56   :: forall name uni fun m a
   57    . Compiling m uni fun name a
   58   => OptimizeOpts name a
   59   -> BuiltinSemanticsVariant fun
   60   -> Program name uni fun a
   61   -> m (Program name uni fun a, OptimizerTrace name uni fun a)
   62 optimizeProgramWithTrace opts builtinSemanticsVariant (Program a v t) = do
   63   (result, trace) <-
   64     runOptimizerT $
   65       termOptimizer opts builtinSemanticsVariant t
   66   pure (Program a v result, trace)

   68 optimizeTerm
   69   :: forall name uni fun m a
   70    . Compiling m uni fun name a
   71   => OptimizeOpts name a
   72   -> BuiltinSemanticsVariant fun
   73   -> Term name uni fun a
   74   -> m (Term name uni fun a)
   75 optimizeTerm opts builtinSemanticsVariant term =
   76   evalOptimizerT $ termOptimizer opts builtinSemanticsVariant term

   78 termOptimizer
   79   :: forall name uni fun m a
   80    . Compiling m uni fun name a
   81   => OptimizeOpts name a
   82   -> BuiltinSemanticsVariant fun
   83   -> Term name uni fun a
   84   -> OptimizerT name uni fun a m (Term name uni fun a)
   85 termOptimizer opts builtinSemanticsVariant =
   86   simplifyNTimes (_ooMaxSimplifierIterations opts)
   87     >=> runStage CseStage
   88     >=> runStage PolyBuiltinStage
   89     >=> runStage ApplyToCaseStage
   90   where
   91     -- Run the simplifier @n@ times
   92     simplifyNTimes
   93       :: Int
   94       -> Term name uni fun a
   95       -> OptimizerT name uni fun a m (Term name uni fun a)
   96     simplifyNTimes n = List.foldl' (>=>) pure $ map simplifyStep [1 .. n]
   97 
   98     -- Run CSE @n@ times, interleaved with the simplifier.
   99     -- See Note [CSE]
  100     cseNTimes
  101       :: Int
  102       -> Term name uni fun a
  103       -> OptimizerT name uni fun a m (Term name uni fun a)
  104     cseNTimes n = foldl' (>=>) pure $ concatMap (\i -> [cseStep i, simplifyStep i]) [1 .. n]
  105 
  106     -- generate simplification step
  107     simplifyStep
  108       :: Int
  109       -> Term name uni fun a
  110       -> OptimizerT name uni fun a m (Term name uni fun a)
  111     simplifyStep _ =
  112       runStage FloatDelayStage
  113         >=> runStage ForceCaseDelayStage
  114         >=> runStage LetFloatOutStage
  115         >=> runStage ForceDelayStage
  116         >=> runStage CaseOfCaseStage
  117         >=> runStage CaseReduceStage
  118         >=> runStage InlineStage
  119         >=> runStage ConstantFoldingStage
  120 
  121     certifiedOnly = _ooCertifiedOptsOnly opts
  122 
  123     runStage stage'
  124       | certifiedOnly && not certified = pure
  125       | otherwise = action
  126       where
  127         certified = isRight stage'
  128         action = case stage' of
  129           FloatDelayStage ->
  130             floatDelay builtinSemanticsVariant
  131           ForceCaseDelayStage ->
  132             forceCaseDelay
  133           ForceDelayStage ->
  134             case (eqT @uni @PLC.DefaultUni, eqT @fun @DefaultFun) of
  135               (Just Refl, Just Refl) -> forceDelay builtinSemanticsVariant
  136               _ -> pure
  137           CaseOfCaseStage ->
  138             caseOfCase'
  139           CaseReduceStage ->
  140             caseReduce
  141           InlineStage ->
  142             inline
  143               (_ooInlineUnconditionalGrowth opts)
  144               (_ooInlineCallsiteGrowth opts)
  145               (_ooInlineConstants opts)
  146               (_ooPreserveLogging opts)
  147               (_ooInlineHints opts)
  148               builtinSemanticsVariant
  149           CseStage ->
  150             cseNTimes cseTimes
  151           ApplyToCaseStage ->
  152             if _ooApplyToCase opts then applyToCase else pure
  153           LetFloatOutStage ->
  154             letFloatOut
  155           ConstantFoldingStage ->
  156             case (eqT @uni @PLC.DefaultUni, eqT @fun @DefaultFun) of
  157               (Just Refl, Just Refl) ->
  158                 evaluateBuiltinsPass
  159                   (_ooPreserveLogging opts)
  160                   ((def :: BuiltinsInfo PLC.DefaultUni DefaultFun) & biSemanticsVariant .~ builtinSemanticsVariant)
  161                   (def :: CostingPart PLC.DefaultUni DefaultFun)
  162               _ -> pure
  163           PolyBuiltinStage -> if _ooHoistPolyBuiltins opts then polyBuiltinStep else pure
  164 
  165     caseOfCase'
  166       :: Term name uni fun a
  167       -> OptimizerT name uni fun a m (Term name uni fun a)
  168     caseOfCase' = case eqT @fun @DefaultFun of
  169       Just Refl -> caseOfCase
  170       Nothing -> pure
  171 
  172     cseStep
  173       :: Int
  174       -> Term name uni fun a
  175       -> OptimizerT name uni fun a m (Term name uni fun a)
  176     cseStep _ =
  177       case (eqT @name @Name, eqT @uni @PLC.DefaultUni) of
  178         (Just Refl, Just Refl) -> cse (_ooCseWhichSubterms opts) builtinSemanticsVariant
  179         _ -> pure
  180 
  181     cseTimes = if _ooConservativeOpts opts then 0 else _ooMaxCseIterations opts
  182 
  183     polyBuiltinStep
  184       :: Term name uni fun a
  185       -> OptimizerT name uni fun a m (Term name uni fun a)
  186     polyBuiltinStep = case (eqT @name @Name, eqT @uni @PLC.DefaultUni) of
  187       (Just Refl, Just Refl) -> polyBuiltin builtinSemanticsVariant
  188       _ -> pure
