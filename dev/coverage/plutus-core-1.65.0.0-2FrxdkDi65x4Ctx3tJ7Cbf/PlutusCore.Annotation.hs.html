   39 -- | An annotation type used during the compilation.
   40 data Ann = Ann
   41   { annInline :: Inline
   42   , annCase :: Case
   43   , annSrcSpans :: SrcSpans
   44   , annIsAsDataMatcher :: Bool
   45   }
   46   deriving stock (Eq, Ord, Generic, Show)
   47   deriving anyclass (Hashable)
   48 
   49 instance Default Ann where
   50   def =
   51     Ann
   52       { annInline = MayInline
   53       , annCase = NotSafeToDrop
   54       , annSrcSpans = mempty
   55       , annIsAsDataMatcher = False
   56       }

   58 data Inline
   59   = {-| When calling @PlutusIR.Compiler.Definitions.defineTerm@ to add a new term definition,
   60     if we annotation the var on the LHS of the definition with `AlwaysInline`, the inliner will
   61     always inline that var.
   62 
   63     This is currently used to ensure builtin functions such as @trace@ (when the @remove-trace@
   64     flag is on and @trace@ is rewritten to @const@) are inlined, because the inliner would
   65     otherwise not inline them. To achieve that, we annotate the definition with `AlwaysInline`
   66     when defining @trace@, i.e., @trace <AlwaysInline> = \_ a -> a@. -}
   67     AlwaysInline
   68   | {-| Signaling to the compiler that a binding is safe to inline. This is useful for
   69     annotating strict bindings that aren't obviously safe to inline. -}
   70     SafeToInline
   71   | MayInline
   72   deriving stock (Eq, Ord, Generic, Show)
   73   deriving anyclass (Hashable)
   74 
   75 instance Pretty Ann where
   76   pretty = viaShow
   77 
   78 class AnnInline a where
   79   -- | An annotation instructing the inliner to always inline a binding.
   80   annAlwaysInline :: a
   81 
   82   {-| An annotation signaling to the inliner that a binding is safe to inline.
   83   The inlining decision is left to the inliner. This is useful for
   84   annotating strict bindings that aren't obviously safe to inline. -}
   85   annSafeToInline :: a
   86 
   87   -- | An annotation that leaves the inlining decision to the inliner.
   88   annMayInline :: a
   89 
   90 instance AnnInline () where
   91   annAlwaysInline = ()
   92   annSafeToInline = ()
   93   annMayInline = ()
   94 
   95 instance AnnInline Ann where
   96   annAlwaysInline = def {annInline = AlwaysInline}
   97   annSafeToInline = def {annInline = SafeToInline}
   98   annMayInline = def {annInline = MayInline}

  100 data Case
  101   = {-| Signaling to the compiler that a @case@ expression is safe to drop,
  102     so long as its binders are unused. -}
  103     SafeToDrop
  104   | NotSafeToDrop
  105   deriving stock (Eq, Ord, Generic, Show)
  106   deriving anyclass (Hashable)
  107 
  108 -- | Methods used to guide optimization of @case@ expressions.
  109 class AnnCase a where
  110   {-| An annotation signaling that the @case@ expression is safe to drop,
  111   so long as its binders are unused. -}
  112   annSafeToDrop :: a
  113 
  114   annNotSafeToDrop :: a
  115 
  116   annIsSafeToDrop :: a -> Bool
  117 
  118 instance AnnCase () where
  119   annSafeToDrop = ()
  120   annNotSafeToDrop = ()
  121   annIsSafeToDrop = const False
  122 
  123 instance AnnCase Ann where
  124   annSafeToDrop = def {annCase = SafeToDrop}
  125   annNotSafeToDrop = def {annCase = NotSafeToDrop}
  126   annIsSafeToDrop = (== SafeToDrop) . annCase

  128 {-| The span between two source locations.
  129 
  130 This corresponds roughly to the `SrcSpan` used by GHC,
  131 but we define our own version so we don't have to depend on `ghc` to use it.
  132 
  133 The line and column numbers are 1-based, and the unit is Unicode code point (or `Char`). -}
  134 data SrcSpan = SrcSpan
  135   { srcSpanFile :: FilePath
  136   , srcSpanSLine :: Int
  137   , srcSpanSCol :: Int
  138   , srcSpanELine :: Int
  139   , srcSpanECol :: Int
  140   {-^ Same as GHC's @SrcSpan@, @srcSpanECol@ is usually one more than the column of
  141   the last character of the thing this @SrcSpan@ is for (unless the last character
  142   is the line break). -}
  143   }
  144   deriving stock (Eq, Ord, Generic)
  145   deriving anyclass (Flat, Hashable, NFData)
  146 
  147 instance Show SrcSpan where
  148   showsPrec _ s =
  149     showString (srcSpanFile s)
  150       . showChar ':'
  151       . showsPrec 0 (srcSpanSLine s)
  152       . showChar ':'
  153       . showsPrec 0 (srcSpanSCol s)
  154       . showChar '-'
  155       . showsPrec 0 (srcSpanELine s)
  156       . showChar ':'
  157       . showsPrec 0 (if srcSpanECol s == 0 then 0 else srcSpanECol s - 1)
  158 
  159 instance Pretty SrcSpan where
  160   pretty = viaShow
  161 
  162 newtype SrcSpans = SrcSpans {unSrcSpans :: Set SrcSpan}
  163   deriving newtype (Eq, Ord, Hashable, Semigroup, Monoid, MonoFoldable, NFData)
  164   deriving stock (Generic)
  165   deriving anyclass (Flat)
  166 
  167 type instance Element SrcSpans = SrcSpan
  168 
  169 instance Show SrcSpans where
  170   showsPrec _ (SrcSpans xs) =
  171     showString "{ "
  172       . showString
  173         ( case Set.toList xs of
  174             [] -> "no-src-span"
  175             ys -> List.intercalate ", " (show <$> ys)
  176         )
  177       . showString " }"
  178 
  179 instance Pretty SrcSpans where
  180   pretty = viaShow
  181 
  182 -- | Add an extra SrcSpan to existing 'SrcSpans' of 'Ann'
  183 addSrcSpan :: SrcSpan -> Ann -> Ann
  184 addSrcSpan s (Ann i c (SrcSpans ss) b) = Ann i c (SrcSpans $ Set.insert s ss) b
  185 
  186 -- | Tells if a line (positive integer) falls inside a SrcSpan.
  187 lineInSrcSpan :: Pos -> SrcSpan -> Bool
  188 lineInSrcSpan pos spn =
  189   let i = Megaparsec.unPos pos
  190    in i >= srcSpanSLine spn && i <= srcSpanELine spn
