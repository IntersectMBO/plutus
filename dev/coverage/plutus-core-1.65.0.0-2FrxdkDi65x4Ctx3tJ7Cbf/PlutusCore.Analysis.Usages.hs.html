    3 -- | Functions for computing variable usage inside types.
    4 module PlutusCore.Analysis.Usages (typeUsages, Usages, getUsageCount, allUsed) where
    5 
    6 import PlutusCore.Core
    7 import PlutusCore.Name.Unique qualified as PLC
    8 import PlutusCore.Subst (tvTy)
    9 
   10 import Control.Lens
   11 
   12 import Data.MultiSet qualified as MSet
   13 import Data.MultiSet.Lens
   14 import Data.Set qualified as Set
   15 
   16 type Usages = MSet.MultiSet PLC.Unique
   17 
   18 -- | Get the usage count of @n@.
   19 getUsageCount :: PLC.HasUnique n unique => n -> Usages -> Int
   20 getUsageCount n = MSet.occur (n ^. PLC.unique . coerced)
   21 
   22 -- | Get a set of @n@s which are used at least once.
   23 allUsed :: Usages -> Set.Set PLC.Unique
   24 allUsed = MSet.toSet
   25 
   26 typeUsages
   27   :: PLC.HasUnique tyname PLC.TypeUnique
   28   => Type tyname uni a
   29   -> Usages
   30 typeUsages = multiSetOf (tvTy . PLC.theUnique)
