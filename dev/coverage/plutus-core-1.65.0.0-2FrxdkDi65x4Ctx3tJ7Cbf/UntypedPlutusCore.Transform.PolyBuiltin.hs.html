    7 {-| Hoist fully-forced polymorphic builtins.
    8 
    9 Before: @...force (force fstPair)...@
   10 After: @(\fstPairForced -> ...fstPairForced...) (force (force fstPair))@ -}

   35 polyBuiltin
   36   :: forall m uni fun a
   37    . ( MonadQuote m
   38      , Hashable fun
   39      , ToBuiltinMeaning uni fun
   40      )
   41   => BuiltinSemanticsVariant fun
   42   -> Term Name uni fun a
   43   -> OptimizerT Name uni fun a m (Term Name uni fun a)
   44 polyBuiltin semvar term = do
   45   let numForces :: fun -> Int
   46       numForces = length . takeWhile (== TypeParam) . builtinArity (Proxy @uni) semvar
   47 
   48       -- If it is a fully forced polymorphic builtin, return `Just (ann, fun, number of forces)`.
   49       checkFullyForced :: Term Name uni fun a -> Maybe (a, fun, Int)
   50       checkFullyForced t = case t' of
   51         Builtin _ fun
   52           | n >= 1
   53           , n == numForces fun ->
   54               Just (getAnn t, fun, n)
   55         _ -> Nothing
   56         where
   57           (n, t') = peelForces t
   58 
   59       candidates :: HashMap fun Int
   60       candidates =
   61         foldlOf' termSubtermsDeep alg Map.empty term
   62         where
   63           alg !acc t = case checkFullyForced t of
   64             Just (_, fun, n) -> Map.insert fun n acc
   65             Nothing -> acc
   66 
   67   result <-
   68     if Map.null candidates
   69       then pure term
   70       else do
   71         named <- for (Map.toList candidates) $ \(fun, n) -> do
   72           name <- freshName "forced"
   73           pure (fun, (name, n))
   74         let binders = Map.fromList named
   75             subst t = case checkFullyForced t of
   76               Just (ann, fun, _)
   77                 | Just (name, _) <- Map.lookup fun binders ->
   78                     Var ann name
   79               _ -> t
   80             rewritten = transformOf termSubterms subst term
   81         pure $ wrap (getAnn term) named rewritten
   82 
   83   recordOptimization term PolyBuiltinStage result
   84   pure result
   85 
   86 peelForces :: Term name uni fun a -> (Int, Term name uni fun a)
   87 peelForces = go 0
   88   where
   89     go !k (Force _ t) = go (k + 1) t
   90     go !k t = (k, t)
   91 
   92 wrap
   93   :: forall uni fun a
   94    . a
   95   -> [(fun, (Name, Int))]
   96   -> Term Name uni fun a
   97   -> Term Name uni fun a
   98 wrap ann entries body =
   99   let lams = foldr (\(_, (name, _)) -> LamAbs ann name) body entries
  100    in F.foldl'
  101         (\acc (fun, (_, n)) -> Apply ann acc (applyForces n (Builtin ann fun)))
  102         lams
  103         entries
  104   where
  105     applyForces :: Int -> Term Name uni fun a -> Term Name uni fun a
  106     applyForces 0 = id
  107     applyForces n = applyForces (n - 1) . Force ann
