   80 recordOptimizationWithHints
   81   :: Monad m
   82   => Hints.Hints
   83   -> Term name uni fun a
   84   -> Trace.OptStage
   85   -> Term name uni fun a
   86   -> OptimizerT name uni fun a m ()
   87 recordOptimizationWithHints hints before stage after =
   88   let optimization = Trace.Optimization before stage hints after
   89    in OptimizerT . State.modify' $ \st ->
   90         st {Trace.optimizerTrace = optimization : Trace.optimizerTrace st}
