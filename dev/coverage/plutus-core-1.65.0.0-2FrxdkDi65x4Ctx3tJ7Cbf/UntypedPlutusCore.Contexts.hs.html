   10 -- | A context for an iterated term/type application.
   11 data AppCtx name uni fun a
   12   = AppCtxTerm a (Term name uni fun a) (AppCtx name uni fun a)
   13   | AppCtxType a (AppCtx name uni fun a)
   14   | AppCtxEnd
   15 
   16 {-| Takes a term and views it as a head plus an 'AppCtx', e.g.
   17 @
   18   [{ f t } u v]
   19   -->
   20   (f, [{ _ t } u v])
   21   ==
   22   f (AppCtxType t (AppCtxTerm u (AppCtxTerm v AppCtxEnd)))
   23 @ -}
   24 splitAppCtx :: Term nam uni fun a -> (Term nam uni fun a, AppCtx nam uni fun a)
   25 splitAppCtx = go AppCtxEnd
   26   where
   27     go appCtx = \case
   28       Apply ann function arg -> go (AppCtxTerm ann arg appCtx) function
   29       Force ann forcedTerm -> go (AppCtxType ann appCtx) forcedTerm
   30       term -> (term, appCtx)
   31 
   32 -- | Fills in the hole in an 'AppCtx', the inverse of 'splitAppCtx'.
   33 fillAppCtx
   34   :: Term name uni fun ann
   35   -> AppCtx name uni fun ann
   36   -> Term name uni fun ann
   37 fillAppCtx term = \case
   38   AppCtxEnd -> term
   39   AppCtxTerm ann arg ctx -> fillAppCtx (Apply ann term arg) ctx
   40   AppCtxType ann ctx -> fillAppCtx (Force ann term) ctx
   41 
   42 data Saturation = Oversaturated | Undersaturated | Saturated
   43 
   44 -- | Do the given arguments saturate the given arity?
   45 saturates :: AppCtx name uni fun a -> Arity -> Maybe Saturation
   46 -- Exactly right
   47 saturates AppCtxEnd [] = Just Saturated
   48 -- Parameters left - undersaturated
   49 saturates AppCtxEnd _ = Just Undersaturated
   50 -- Match a term parameter to a term arg
   51 saturates (AppCtxTerm _ _ ctx) (TermParam : arities) = saturates ctx arities
   52 -- Match a type parameter to a type arg
   53 saturates (AppCtxType _ ctx) (TypeParam : arities) = saturates ctx arities
   54 -- Param/arg mismatch
   55 saturates (AppCtxTerm {}) (TypeParam : _) = Nothing
   56 saturates (AppCtxType {}) (TermParam : _) = Nothing
   57 -- Arguments left - undersaturated
   58 saturates (AppCtxTerm {}) [] = Just Oversaturated
   59 saturates (AppCtxType {}) [] = Just Oversaturated
