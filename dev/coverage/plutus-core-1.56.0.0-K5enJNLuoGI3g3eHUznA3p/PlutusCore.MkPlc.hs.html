   72 -- | A final encoding for Term, to allow PLC terms to be used transparently as PIR terms.
   73 class TermLike term tyname name uni fun | term -> tyname name uni fun where
   74   var :: ann -> name -> term ann
   75   tyAbs :: ann -> tyname -> Kind ann -> term ann -> term ann
   76   lamAbs :: ann -> name -> Type tyname uni ann -> term ann -> term ann
   77   apply :: ann -> term ann -> term ann -> term ann
   78   constant :: ann -> Some (ValueOf uni) -> term ann
   79   builtin :: ann -> fun -> term ann
   80   tyInst :: ann -> term ann -> Type tyname uni ann -> term ann
   81   unwrap :: ann -> term ann -> term ann
   82   iWrap :: ann -> Type tyname uni ann -> Type tyname uni ann -> term ann -> term ann
   83   error :: ann -> Type tyname uni ann -> term ann
   84   constr :: ann -> Type tyname uni ann -> Word64 -> [term ann] -> term ann
   85   kase :: ann -> Type tyname uni ann -> term ann -> [term ann] -> term ann
   86 
   87   termLet :: ann -> TermDef term tyname name uni ann -> term ann -> term ann
   88   typeLet :: ann -> TypeDef tyname uni ann -> term ann -> term ann
   89 
   90   termLet = mkImmediateLamAbs
   91   typeLet = mkImmediateTyAbs

   97 mkTyBuiltinOf ann = TyBuiltin ann . SomeTypeIn

   99 -- | Embed a Haskell value (given its explicit type tag) into a PLC term.
  100 mkConstantOf
  101   :: forall a uni fun term tyname name ann
  102    . TermLike term tyname name uni fun
  103   => ann -> uni (Esc a) -> a -> term ann
  104 mkConstantOf ann uni = constant ann . someValueOf uni
  105 
  106 -- | Embed a Haskell value (provided its type is in the universe) into a PLC term.
  107 mkConstant
  108   :: forall a uni fun term tyname name ann
  109    . (TermLike term tyname name uni fun, uni `HasTermLevel` a)
  110   => ann -> a -> term ann
  111 mkConstant ann = constant ann . someValue

  113 instance TermLike (Term tyname name uni fun) tyname name uni fun where
  114   var = Var
  115   tyAbs = TyAbs
  116   lamAbs = LamAbs
  117   apply = Apply
  118   constant = Constant
  119   builtin = Builtin
  120   tyInst = TyInst
  121   unwrap = Unwrap
  122   iWrap = IWrap
  123   error = Error
  124   constr = Constr
  125   kase = Case

  127 embedTerm :: TermLike term tyname name uni fun => Term tyname name uni fun ann -> term ann
  128 embedTerm = \case
  129   Var a n -> var a n
  130   TyAbs a tn k t -> tyAbs a tn k (embedTerm t)
  131   LamAbs a n ty t -> lamAbs a n ty (embedTerm t)
  132   Apply a t1 t2 -> apply a (embedTerm t1) (embedTerm t2)
  133   Constant a c -> constant a c
  134   Builtin a bi -> builtin a bi
  135   TyInst a t ty -> tyInst a (embedTerm t) ty
  136   Error a ty -> error a ty
  137   Unwrap a t -> unwrap a (embedTerm t)
  138   IWrap a ty1 ty2 t -> iWrap a ty1 ty2 (embedTerm t)
  139   Constr a ty i es -> constr a ty i (fmap embedTerm es)
  140   Case a ty arg cs -> kase a ty (embedTerm arg) (fmap embedTerm cs)

  142 {-| Make a 'Var' referencing the given 'VarDecl'.
  143 The @ann@ is propagated from the 'VarDecl' to the 'Var'. -}
  144 mkVar :: TermLike term tyname name uni fun => VarDecl tyname name uni ann -> term ann
  145 mkVar v = var (_varDeclAnn v) (_varDeclName v)

  153 -- | A definition. Pretty much just a pair with more descriptive names.
  154 data Def var val = Def
  155   { defVar :: var
  156   , defVal :: val
  157   }
  158   deriving stock (Show, Eq, Ord, Generic)

  160 -- | A term definition as a variable.
  161 type TermDef term tyname name uni ann = Def (VarDecl tyname name uni ann) (term ann)

  163 -- | A type definition as a type variable.
  164 type TypeDef tyname uni ann = Def (TyVarDecl tyname ann) (Type tyname uni ann)

  166 -- | The type of a PLC function.
  167 data FunctionType tyname uni ann = FunctionType
  168   { _functionTypeAnn :: ann
  169   -- ^ An annotation.
  170   , _functionTypeDom :: Type tyname uni ann
  171   -- ^ The domain of a function.
  172   , _functionTypeCod :: Type tyname uni ann
  173   -- ^ The codomain of the function.
  174   }
  175 
  176 -- Should we parameterize 'VarDecl' by @ty@ rather than @tyname@, so that we can define
  177 -- 'FunctionDef' as 'TermDef FunctionType tyname name uni fun ann'?
  178 -- Perhaps we even should define general 'Decl' and 'Def' that cover all of the cases?
  179 -- | A PLC function.
  180 data FunctionDef term tyname name uni fun ann = FunctionDef
  181   { _functionDefAnn :: ann
  182   -- ^ An annotation.
  183   , _functionDefName :: name
  184   -- ^ The name of a function.
  185   , _functionDefType :: FunctionType tyname uni ann
  186   -- ^ The type of the function.
  187   , _functionDefTerm :: term ann
  188   -- ^ The definition of the function.
  189   }
  190 
  191 -- | Convert a 'FunctionType' to the corresponding 'Type'.
  192 functionTypeToType :: FunctionType tyname uni ann -> Type tyname uni ann
  193 functionTypeToType (FunctionType ann dom cod) = TyFun ann dom cod
  194 
  195 -- | Get the type of a 'FunctionDef'.
  196 functionDefToType :: FunctionDef term tyname name uni fun ann -> Type tyname uni ann
  197 functionDefToType (FunctionDef _ _ funTy _) = functionTypeToType funTy
  198 
  199 -- | Convert a 'FunctionDef' to a 'VarDecl'. I.e. ignore the actual term.
  200 functionDefVarDecl :: FunctionDef term tyname name uni fun ann -> VarDecl tyname name uni ann
  201 functionDefVarDecl (FunctionDef ann name funTy _) = VarDecl ann name $ functionTypeToType funTy
  202 
  203 -- | Make a 'FunctionDef'. Return 'Nothing' if the provided type is not functional.
  204 mkFunctionDef
  205   :: ann
  206   -> name
  207   -> Type tyname uni ann
  208   -> term ann
  209   -> Maybe (FunctionDef term tyname name uni fun ann)
  210 mkFunctionDef annName name (TyFun annTy dom cod) term =
  211   Just $ FunctionDef annName name (FunctionType annTy dom cod) term
  212 mkFunctionDef _ _ _ _ = Nothing

  214 -- | Make a "let-binding" for a term as an immediately applied lambda abstraction.
  215 mkImmediateLamAbs
  216   :: TermLike term tyname name uni fun
  217   => ann
  218   -> TermDef term tyname name uni ann
  219   -> term ann
  220   -- ^ The body of the let, possibly referencing the name.
  221   -> term ann
  222 mkImmediateLamAbs ann1 (Def (VarDecl ann2 name ty) bind) body =
  223   apply ann1 (lamAbs ann2 name ty body) bind
  224 
  225 -- | Make a "let-binding" for a type as an immediately instantiated type abstraction. Note: the body must be a value.
  226 mkImmediateTyAbs
  227   :: TermLike term tyname name uni fun
  228   => ann
  229   -> TypeDef tyname uni ann
  230   -> term ann
  231   -- ^ The body of the let, possibly referencing the name.
  232   -> term ann
  233 mkImmediateTyAbs ann1 (Def (TyVarDecl ann2 name k) bind) body =
  234   tyInst ann1 (tyAbs ann2 name k body) bind

  236 {-| Make an iterated application. Each `apply` node uses the annotation associated with
  237 the corresponding argument. -}
  238 mkIterApp
  239   :: TermLike term tyname name uni fun
  240   => term ann
  241   -> [(ann, term ann)]
  242   -> term ann
  243 mkIterApp = foldl' $ \acc (ann, arg) -> apply ann acc arg
  244 
  245 -- | Make an iterated application with no annotation.
  246 mkIterAppNoAnn
  247   :: TermLike term tyname name uni fun
  248   => term ()
  249   -- ^ @f@
  250   -> [term ()]
  251   -- ^ @[ x0 ... xn ]@
  252   -> term ()
  253   -- ^ @[f x0 ... xn ]@
  254 mkIterAppNoAnn term = mkIterApp term . fmap ((),)
  255 
  256 -- | An infix synonym for `mkIterAppNoAnn`
  257 (@@)
  258   :: TermLike term tyname name uni fun
  259   => term ()
  260   -- ^ @f@
  261   -> [term ()]
  262   -- ^ @[ x0 ... xn ]@
  263   -> term ()
  264   -- ^ @[f x0 ... xn ]@
  265 (@@) = mkIterAppNoAnn

  267 {-| Make an iterated instantiation. Each `tyInst` node uses the annotation associated with
  268 the corresponding argument. -}
  269 mkIterInst
  270   :: TermLike term tyname name uni fun
  271   => term ann
  272   -- ^ @a@
  273   -> [(ann, Type tyname uni ann)]
  274   -- ^ @ [ x0 ... xn ] @
  275   -> term ann
  276   -- ^ @{ a x0 ... xn }@
  277 mkIterInst = foldl' $ \acc (ann, arg) -> tyInst ann acc arg

  284   -> [Type tyname uni ()]
  285   -- ^ @ [ x0 ... xn ] @

  287   -- ^ @{ a x0 ... xn }@

  290 -- | Lambda abstract a list of names.
  291 mkIterLamAbs
  292   :: TermLike term tyname name uni fun
  293   => [VarDecl tyname name uni ann]
  294   -> term ann
  295   -> term ann
  296 mkIterLamAbs args body =
  297   foldr (\(VarDecl ann name ty) acc -> lamAbs ann name ty acc) body args

  303   -> term ann
  304   -> term ann
  305 mkIterTyAbs args body =

  308 {-| Make an iterated type application. Each `TyApp` node uses the annotation associated with
  309 the corresponding argument. -}
  310 mkIterTyApp
  311   :: Type tyname uni ann
  312   -- ^ @f@
  313   -> [(ann, Type tyname uni ann)]

  317 mkIterTyApp = foldl' $ \acc (ann, arg) -> TyApp ann acc arg

  325   -> Type tyname uni ()
  326   -- ^ @[ f x0 ... xn ]@

  329 -- | Make an iterated function type.
  330 mkIterTyFun
  331   :: ann
  332   -> [Type tyname uni ann]
  333   -> Type tyname uni ann
  334   -> Type tyname uni ann
  335 mkIterTyFun ann tys target = foldr (\ty acc -> TyFun ann ty acc) target tys
  336 
  337 -- | Universally quantify a list of names.
  338 mkIterTyForall
  339   :: [TyVarDecl tyname ann]
  340   -> Type tyname uni ann
  341   -> Type tyname uni ann
  342 mkIterTyForall args body =
  343   foldr (\(TyVarDecl ann name kind) acc -> TyForall ann name kind acc) body args

  345 -- | Lambda abstract a list of names.
  346 mkIterTyLam
  347   :: [TyVarDecl tyname ann]
  348   -> Type tyname uni ann
  349   -> Type tyname uni ann
  350 mkIterTyLam args body =
  351   foldr (\(TyVarDecl ann name kind) acc -> TyLam ann name kind acc) body args

  353 -- | Make an iterated function kind.
  354 mkIterKindArrow

  356   -> [Kind ann]
  357   -> Kind ann

  361 {-| A helper to create a single, fresh strict binding; It returns the fresh bound `Var`iable and
  362 a function `Term -> Term`, expecting an "in-Term" to form a let-expression. -}
  363 mkFreshTermLet
  364   :: (MonadQuote m, TermLike t tyname Name uni fun, Monoid a)
  365   => Type tyname uni a
  366   -- ^ the type of binding
  367   -> t a
  368   -- ^ the term bound to the fresh variable
  369   -> m (t a, t a -> t a)
  370   -- ^ the fresh Var and a function that takes an "in" term to construct the Let
  371 mkFreshTermLet aT a = do
  372   -- I wish this was less constrained to Name
  373   genName <- freshName "generated"
  374   pure (var mempty genName, termLet mempty (Def (VarDecl mempty genName aT) a))

  376 -- | 'apply' the head of the application to the arguments iteratively.
  377 headSpineToTerm :: TermLike term tyname name uni fun => ann -> MonoHeadSpine err (term ann) -> Either err (term ann)
  378 headSpineToTerm _ (HeadError e) = Left e
  379 headSpineToTerm _ (HeadOnly t) = Right t
  380 headSpineToTerm ann (HeadSpine t ts) = Right $ foldl (apply ann) t ts
  381 
  382 -- | @headSpineToTerm@ but without annotation.
  383 headSpineToTermNoAnn :: TermLike term tyname name uni fun => MonoHeadSpine err (term ()) -> Either err (term ())
  384 headSpineToTermNoAnn (HeadError e) = Left e
  385 headSpineToTermNoAnn (HeadOnly t) = Right t
  386 headSpineToTermNoAnn (HeadSpine t ts) = Right $ foldl (apply ()) t ts
