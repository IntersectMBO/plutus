   19 -- | Convert an object to a hashmap. Compatibility shim for pre-/post-aeson-2.
   20 objToHm :: Object -> HM.HashMap Text.Text Value
   21 #if MIN_VERSION_aeson(2,0,0)
   22 objToHm = toHashMapText
   23 #else
   24 objToHm = id
   25 #endif

   27 -- | Convert a hashmap to an object. Compatibility shim for pre-/post-aeson-2.
   28 hmToObj :: HM.HashMap Text.Text Value -> Object
   29 #if MIN_VERSION_aeson(2,0,0)
   30 hmToObj = fromHashMapText
   31 #else
   32 hmToObj = id
   33 #endif

   35 {-| Turn a nested object into a "flat" object where the keys represent paths into the original
   36 object. The keys in the result will be the keys in the original path, separated by `sep`.
   37 The inverse of 'unflattenObject'. -}
   38 flattenObject :: Text.Text -> Object -> Object
   39 flattenObject sep o = hmToObj $ go Nothing (objToHm o)
   40   where
   41     go :: Maybe Text.Text -> HM.HashMap Text.Text Value -> HM.HashMap Text.Text Value
   42     go mprefix = HM.foldMapWithKey $ \k v ->
   43       let newName = case mprefix of
   44             Just prefix -> prefix <> sep <> k
   45             Nothing -> k
   46        in case v of
   47             Object o' -> go (Just newName) $ objToHm o'
   48             leaf -> HM.singleton newName leaf
   49 
   50 {-| Turn a "flat" object whose keys represent paths into an unflattened object.
   51 The keys in the result will be the resulting path, separated by `sep`.
   52 The inverse of 'flattenObject'. -}
   53 unflattenObject :: Text.Text -> Object -> Object
   54 unflattenObject sep o =
   55   HM.foldlWithKey (\acc k v -> mergeObject acc (mkPathObject k v)) mempty (objToHm o)
   56   where
   57     mkPathObject :: Text.Text -> Value -> Object
   58     mkPathObject k value =
   59       let path = Text.splitOn sep k
   60        in hmToObj $ go path value
   61       where
   62         go :: [Text.Text] -> Value -> HM.HashMap Text.Text Value
   63         go [] _ = error "empty path"
   64         go [n] v = HM.singleton n v
   65         go (n : n' : xs) v = HM.singleton n $ Object $ hmToObj $ go (n' : xs) v
   66 
   67 {-| Merge two objects, merging the values where both sides have an entry for a key rather than
   68 taking the first. -}
   69 mergeObject :: Object -> Object -> Object
   70 mergeObject o1 o2 = hmToObj $ HM.unionWith mergeValue (objToHm o1) (objToHm o2)
   71 
   72 -- | Merge two values, merging the objects using 'mergeObject'. Can't merge anything else.
   73 mergeValue :: Value -> Value -> Value
   74 mergeValue (Object o1) (Object o2) = Object $ mergeObject o1 o2
   75 mergeValue _ _ = error "can't merge"
