   18 {-| Verify an ECDSA signature made using the SECP256k1 curve.
   19 
   20 = Note
   21 
   22 There are additional well-formation requirements for the arguments beyond
   23 their length:
   24 
   25 * The first byte of the public key must correspond to the sign of the /y/
   26 coordinate: this is @0x02@ if /y/ is even, and @0x03@ otherwise.
   27 * The remaining bytes of the public key must correspond to the /x/
   28 coordinate, as a big-endian integer.
   29 * The first 32 bytes of the signature must correspond to the big-endian
   30 integer representation of _r_.
   31 * The last 32 bytes of the signature must correspond to the big-endian
   32 integer representation of _s_.
   33 
   34 While this primitive /accepts/ a hash, any caller should only pass it hashes
   35 that they computed themselves: specifically, they should receive the
   36 /message/ from a sender and hash it, rather than receiving the /hash/ from
   37 said sender. Failure to do so can be
   38 [dangerous](https://bitcoin.stackexchange.com/a/81116/35586). Other than
   39 length, we make no requirements of what hash gets used. -}
   40 verifyEcdsaSecp256k1Signature
   41   :: BS.ByteString
   42   -- ^ Public key   (33 bytes)
   43   -> BS.ByteString
   44   -- ^ Message hash (32 bytes)
   45   -> BS.ByteString
   46   -- ^ Signature    (64 bytes)
   47   -> BuiltinResult Bool
   48 verifyEcdsaSecp256k1Signature pk msg sig =
   49   case DSIGN.rawDeserialiseVerKeyDSIGN @EcdsaSecp256k1DSIGN pk of
   50     Nothing -> failWithMessage loc "Invalid verification key."
   51     Just pk' -> case DSIGN.rawDeserialiseSigDSIGN @EcdsaSecp256k1DSIGN sig of
   52       Nothing -> failWithMessage loc "Invalid signature."
   53       Just sig' -> case toMessageHash msg of
   54         Nothing -> failWithMessage loc "Invalid message hash."
   55         Just msg' -> pure $ case DSIGN.verifyDSIGN () pk' msg' sig' of
   56           Left _ -> False
   57           Right () -> True
   58   where
   59     loc :: Text
   60     loc = "ECDSA SECP256k1 signature verification"

   62 {-| Verify a Schnorr signature made using the SECP256k1 curve.
   63 
   64 = Note
   65 
   66 There are additional well-formation requirements for the arguments beyond
   67 their length. Throughout, we refer to co-ordinates of the point @R@.
   68 
   69 * The bytes of the public key must correspond to the /x/ coordinate, as a
   70 big-endian integer, as specified in BIP-340.
   71 * The first 32 bytes of the signature must correspond to the /x/ coordinate,
   72 as a big-endian integer, as specified in BIP-340.
   73 * The last 32 bytes of the signature must correspond to the bytes of /s/, as
   74 a big-endian integer, as specified in BIP-340.
   75 
   76 = See also
   77 
   78 * [BIP-340](https://github.com/bitcoin/bips/blob/master/bip-0340.mediawiki) -}
   79 verifySchnorrSecp256k1Signature
   80   :: BS.ByteString
   81   -- ^ Public key (32 bytes)
   82   -> BS.ByteString
   83   -- ^ Message    (arbitrary length)
   84   -> BS.ByteString
   85   -- ^ Signature  (64 bytes)
   86   -> BuiltinResult Bool
   87 verifySchnorrSecp256k1Signature pk msg sig =
   88   case DSIGN.rawDeserialiseVerKeyDSIGN @SchnorrSecp256k1DSIGN pk of
   89     Nothing -> failWithMessage loc "Invalid verification key."
   90     Just pk' -> case DSIGN.rawDeserialiseSigDSIGN @SchnorrSecp256k1DSIGN sig of
   91       Nothing -> failWithMessage loc "Invalid signature."
   92       Just sig' -> pure $ case DSIGN.verifyDSIGN () pk' msg sig' of
   93         Left _ -> False
   94         Right () -> True
   95   where
   96     loc :: Text
   97     loc = "Schnorr SECP256k1 signature verification"
