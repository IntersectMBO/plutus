    8 -- | A context for an iterated term/type application.
    9 data AppCtx name uni fun a
   10   = AppCtxTerm a (Term name uni fun a) (AppCtx name uni fun a)
   11   | AppCtxType a (AppCtx name uni fun a)
   12   | AppCtxEnd
   13 
   14 {-| Takes a term and views it as a head plus an 'AppCtx', e.g.
   15 @
   16   [{ f t } u v]
   17   -->
   18   (f, [{ _ t } u v])
   19   ==
   20   f (AppCtxType t (AppCtxTerm u (AppCtxTerm v AppCtxEnd)))
   21 @ -}
   22 splitAppCtx :: Term nam uni fun a -> (Term nam uni fun a, AppCtx nam uni fun a)
   23 splitAppCtx = go AppCtxEnd
   24   where
   25     go appCtx = \case
   26       Apply ann function argument -> go (AppCtxTerm ann argument appCtx) function
   27       Force ann forcedTerm -> go (AppCtxType ann appCtx) forcedTerm
   28       term -> (term, appCtx)
   29 
   30 -- | Fills in the hole in an 'AppCtx', the inverse of 'splitAppCtx'.
   31 fillAppCtx
   32   :: Term name uni fun ann
   33   -> AppCtx name uni fun ann
   34   -> Term name uni fun ann
   35 fillAppCtx term = \case
   36   AppCtxEnd -> term
   37   AppCtxTerm ann arg ctx -> fillAppCtx (Apply ann term arg) ctx
   38   AppCtxType ann ctx -> fillAppCtx (Force ann term) ctx
