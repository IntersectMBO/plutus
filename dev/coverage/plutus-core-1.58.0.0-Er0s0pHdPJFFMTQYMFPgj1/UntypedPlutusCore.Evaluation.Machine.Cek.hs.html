   75 {-| Evaluate a term using the CEK machine with logging enabled and keep track of costing.
   76 A wrapper around the internal runCek to debruijn input and undebruijn output.
   77 *THIS FUNCTION IS PARTIAL if the input term contains free variables* -}
   78 runCek
   79   :: ThrowableBuiltins uni fun
   80   => MachineParameters CekMachineCosts fun (CekValue uni fun ann)
   81   -> ExBudgetMode cost uni fun
   82   -> EmitterMode uni fun
   83   -> Term Name uni fun ann
   84   -> CekReport cost Name uni fun
   85 runCek = Common.runCek runCekDeBruijn
   86 
   87 {-| Evaluate a term using the CEK machine with logging disabled and keep track of costing.
   88 *THIS FUNCTION IS PARTIAL if the input term contains free variables* -}
   89 runCekNoEmit
   90   :: ThrowableBuiltins uni fun
   91   => MachineParameters CekMachineCosts fun (CekValue uni fun ann)
   92   -> ExBudgetMode cost uni fun
   93   -> Term Name uni fun ann
   94   -> (Either (CekEvaluationException Name uni fun) (Term Name uni fun ()), cost)
   95 runCekNoEmit = Common.runCekNoEmit runCekDeBruijn
