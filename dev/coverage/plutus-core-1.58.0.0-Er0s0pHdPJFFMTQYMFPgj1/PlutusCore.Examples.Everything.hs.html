    3 {-| This module exports all available examples via a data type which allows to test
    4 various procedures (pretty-printing, type checking, etc) over the entire set of examples
    5 in a convenient way: each time a function / data type is added to examples, none of the
    6 tests is required to be adapted, instead you just add the new definition to 'examples'
    7 defined below and all the tests see it automatically. -}
    8 module PlutusCore.Examples.Everything
    9   ( examples
   10   , builtins
   11   ) where

   32 -- | All examples exported as a single value.
   33 examples :: PlcFolderContents DefaultUni (Either DefaultFun ExtensionFun)
   34 examples =
   35   FolderContents
   36     [ treeFolderContents
   37         "Examples"
   38         [ treeFolderContents "Data" $
   39             plcTermFile "exampleData" exampleData
   40               : [ plcTermFile (name ++ show optMatch) $ f optMatch
   41                 | optMatch <- enumerate
   42                 , (name, f) <- [("ofoldrData", ofoldrData)]
   43                 ]
   44         , treeFolderContents
   45             "Function"
   46             [ plcTermFile "unsafeCoerce" unsafeCoerce
   47             ]
   48         , treeFolderContents
   49             "InterList"
   50             [ plcTypeFile "InterList" $ _recursiveType interListData
   51             , plcTermFile "InterNil" interNil
   52             , plcTermFile "InterCons" interCons
   53             , plcTermFile "FoldrInterList" foldrInterList
   54             ]
   55         , treeFolderContents
   56             "List"
   57             [ plcTermFile (name ++ show optMatch) $ f optMatch
   58             | optMatch <- enumerate
   59             , (name, f) <- [("omapList", omapList)]
   60             ]
   61         , treeFolderContents
   62             "Pair"
   63             [ plcTermFile "obothPair" obothPair
   64             ]
   65         , treeFolderContents
   66             "TreeForest"
   67             [ plcTypeFile "Tree" $ _recursiveType treeData
   68             , plcTypeFile "Forest" $ _recursiveType forestData
   69             , plcTermFile "TreeNode" treeNode
   70             , plcTermFile "ForestNil" forestNil
   71             , plcTermFile "ForestCons" forestCons
   72             ]
   73         , treeFolderContents
   74             "Vec"
   75             [ plcTypeFile "zeroT" zeroT
   76             , plcTypeFile "succT" succT
   77             , plcTypeFile "plusT" plusT
   78             , plcTypeFile "churchVec" churchVec
   79             , plcTermFile "churchNil" churchNil
   80             , plcTermFile "churchCons" churchCons
   81             , plcTermFile "churchConcat" churchConcat
   82             , plcTypeFile "scottVec" scottVec
   83             , plcTermFile "scottNil" scottNil
   84             , plcTermFile "scottCons" scottCons
   85             , plcTermFile "scottHead" scottHead
   86             , plcTermFile "scottSumHeadsOr0" $ mapFun Left scottSumHeadsOr0
   87             ]
   88         , treeFolderContents
   89             "Shad"
   90             [ plcTypeFile "shad" shad
   91             , plcTermFile "mkShad" mkShad
   92             ]
   93         , treeFolderContents
   94             "RecUnit"
   95             [ plcTypeFile "recUnit" recUnit
   96             , plcTermFile "runRecUnit" runRecUnit
   97             ]
   98         ]
   99     ]
