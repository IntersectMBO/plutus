   24 {-|
   25 The version of Plutus Core used by this program.
   26 
   27 The intention is to convey different levels of backwards compatibility for existing scripts:
   28 - Major version changes are backwards-incompatible
   29 - Minor version changes are backwards-compatible
   30 - Patch version changes should be entirely invisible (and we will likely not use this level)
   31 
   32 The version used should be changed only when the /language itself/ changes.
   33 For example, adding a new kind of term to the language would require a minor
   34 version bump; removing a kind of term would require a major version bump.
   35 
   36 Similarly, changing the semantics of the language will require a version bump,
   37 typically a major one. This is the main reason why the version is actually
   38 tracked in the AST: we can have two language versions with identical ASTs but
   39 different semantics, so we need to track the version explicitly.
   40 
   41 Compatibility is about compatibility for specific scripts, not about e.g. tools which consume
   42 scripts. Adding a new kind of term does not change how existing scripts behave, but does
   43 change what tools would need to do to process scripts. -}
   44 data Version
   45   = Version {_versionMajor :: Natural, _versionMinor :: Natural, _versionPatch :: Natural}
   46   deriving stock (Eq, Show, Generic)
   47   deriving anyclass (NFData, Hashable)
   48 
   49 makeLenses ''Version
   50 
   51 -- This is probably what the derived version would do, but better to be explicit since it's
   52 -- important
   53 instance Ord Version where
   54   compare (Version major1 minor1 patch1) (Version major2 minor2 patch2) =
   55     compare major1 major2 <> compare minor1 minor2 <> compare patch1 patch2
   56 
   57 -- | The first version of Plutus Core supported by this library.
   58 firstVersion :: Version

   73 {-| The set of versions that are "known", i.e. that have been released

   75 knownVersions :: Set.Set Version

   78 instance Pretty Version where
   79   pretty (Version i j k) = pretty i <> "." <> pretty j <> "." <> pretty k
