   15 -- In the three instances below the added variable is always the last field of a constructor.

   18 instance tyname ~ TyName => Reference TyName (Type tyname uni) where
   19   referenceVia reg tyname ty = TyApp NotAName ty $ TyVar (reg tyname) tyname
   20 
   21 instance tyname ~ TyName => Reference TyName (Term tyname name uni fun) where
   22   referenceVia reg tyname term = TyInst NotAName term $ TyVar (reg tyname) tyname
   23 
   24 instance name ~ Name => Reference Name (Term tyname name uni fun) where
   25   referenceVia reg name term = Apply NotAName term $ Var (reg name) name

   27 -- Kinds have no names, hence the simple instance.
   28 instance EstablishScoping Kind where
   29   establishScoping kind = pure $ NotAName <$ kind
   30 
   31 -- Kinds have no names, hence the simple instance.
   32 instance CollectScopeInfo Kind where
   33   collectScopeInfo _ = mempty

   35 instance tyname ~ TyName => EstablishScoping (Type tyname uni) where
   36   establishScoping (TyLam _ nameDup kind ty) = do
   37     name <- freshenTyName nameDup
   38     establishScopingBinder TyLam name kind ty
   39   establishScoping (TyForall _ nameDup kind ty) = do
   40     name <- freshenTyName nameDup
   41     establishScopingBinder TyForall name kind ty
   42   establishScoping (TyIFix _ pat arg) =
   43     TyIFix NotAName <$> establishScoping pat <*> establishScoping arg
   44   establishScoping (TyApp _ fun arg) =
   45     TyApp NotAName <$> establishScoping fun <*> establishScoping arg
   46   establishScoping (TyFun _ dom cod) =
   47     TyFun NotAName <$> establishScoping dom <*> establishScoping cod
   48   establishScoping (TyVar _ nameDup) = do
   49     name <- freshenTyName nameDup
   50     pure $ TyVar (registerFree name) name
   51   establishScoping (TyBuiltin _ someUni) = pure $ TyBuiltin NotAName someUni
   52   establishScoping (TySOP _ tyls) =
   53     TySOP NotAName <$> (traverse . traverse) establishScoping tyls
   54 
   55 firstBound :: Term tyname name uni fun ann -> [name]
   56 firstBound (Apply _ (LamAbs _ name _ body) _) = name : firstBound body
   57 firstBound _ = []

   59 instance (tyname ~ TyName, name ~ Name) => EstablishScoping (Term tyname name uni fun) where
   60   establishScoping (LamAbs _ nameDup ty body) = do
   61     name <- freshenName nameDup
   62     establishScopingBinder LamAbs name ty body
   63   establishScoping (TyAbs _ nameDup kind body) = do
   64     name <- freshenTyName nameDup
   65     establishScopingBinder TyAbs name kind body
   66   establishScoping (IWrap _ pat arg term) =
   67     IWrap NotAName <$> establishScoping pat <*> establishScoping arg <*> establishScoping term
   68   establishScoping (Apply _ fun arg) =
   69     Apply NotAName <$> establishScoping fun <*> establishScoping arg
   70   establishScoping (Unwrap _ term) = Unwrap NotAName <$> establishScoping term
   71   establishScoping (Error _ ty) = Error NotAName <$> establishScoping ty
   72   establishScoping (TyInst _ term ty) =
   73     TyInst NotAName <$> establishScoping term <*> establishScoping ty
   74   establishScoping (Var _ nameDup) = do
   75     name <- freshenName nameDup
   76     pure $ Var (registerFree name) name
   77   establishScoping (Constant _ con) = pure $ Constant NotAName con
   78   establishScoping (Builtin _ bi) = pure $ Builtin NotAName bi
   79   establishScoping (Constr _ ty i es) =
   80     Constr NotAName <$> establishScoping ty <*> pure i <*> traverse establishScoping es
   81   establishScoping (Case _ ty a es) = do
   82     esScoped <- traverse establishScoping es
   83     let esScopedPoked = addTheRest $ map (\e -> (e, firstBound e)) esScoped
   84         branchBounds = map (snd . fst) esScopedPoked
   85         referenceInBranch ((branch, _), others) = referenceOutOfScope (map snd others) branch
   86     tyScoped <- establishScoping ty
   87     aScoped <- establishScoping a
   88     -- For each of the branches reference (as out-of-scope) the variables bound in that branch
   89     -- in all the other ones, as well as outside of the whole case-expression. This is to check
   90     -- that none of the transformations leak variables outside of the branch they're bound in.
   91     pure . referenceOutOfScope branchBounds $
   92       Case NotAName tyScoped aScoped $
   93         map referenceInBranch esScopedPoked
   94 
   95 instance (tyname ~ TyName, name ~ Name) => EstablishScoping (Program tyname name uni fun) where
   96   establishScoping (Program _ ver term) =
   97     Program NotAName ver <$> establishScoping term

   99 instance tyname ~ TyName => CollectScopeInfo (Type tyname uni) where
  100   collectScopeInfo (TyLam ann name kind ty) =
  101     handleSname ann name <> collectScopeInfo kind <> collectScopeInfo ty
  102   collectScopeInfo (TyForall ann name kind ty) =
  103     handleSname ann name <> collectScopeInfo kind <> collectScopeInfo ty
  104   collectScopeInfo (TyIFix _ pat arg) = collectScopeInfo pat <> collectScopeInfo arg
  105   collectScopeInfo (TyApp _ fun arg) = collectScopeInfo fun <> collectScopeInfo arg
  106   collectScopeInfo (TyFun _ dom cod) = collectScopeInfo dom <> collectScopeInfo cod
  107   collectScopeInfo (TyVar ann name) = handleSname ann name
  108   collectScopeInfo (TyBuiltin _ _) = mempty
  109   collectScopeInfo (TySOP _ tyls) = (foldMap . foldMap) collectScopeInfo tyls
  110 
  111 instance (tyname ~ TyName, name ~ Name) => CollectScopeInfo (Term tyname name uni fun) where
  112   collectScopeInfo (LamAbs ann name ty body) =
  113     handleSname ann name <> collectScopeInfo ty <> collectScopeInfo body
  114   collectScopeInfo (TyAbs ann name kind body) =
  115     handleSname ann name <> collectScopeInfo kind <> collectScopeInfo body
  116   collectScopeInfo (IWrap _ pat arg term) =
  117     collectScopeInfo pat <> collectScopeInfo arg <> collectScopeInfo term
  118   collectScopeInfo (Apply _ fun arg) = collectScopeInfo fun <> collectScopeInfo arg
  119   collectScopeInfo (Unwrap _ term) = collectScopeInfo term
  120   collectScopeInfo (Error _ ty) = collectScopeInfo ty
  121   collectScopeInfo (TyInst _ term ty) = collectScopeInfo term <> collectScopeInfo ty
  122   collectScopeInfo (Var ann name) = handleSname ann name
  123   collectScopeInfo (Constant _ _) = mempty
  124   collectScopeInfo (Builtin _ _) = mempty
  125   collectScopeInfo (Constr _ ty _ es) =
  126     collectScopeInfo ty <> foldMap collectScopeInfo es
  127   collectScopeInfo (Case _ ty arg cs) =
  128     collectScopeInfo ty <> collectScopeInfo arg <> foldMap collectScopeInfo cs
  129 
  130 instance (tyname ~ TyName, name ~ Name) => CollectScopeInfo (Program tyname name uni fun) where
  131   collectScopeInfo (Program _ _ term) = collectScopeInfo term
