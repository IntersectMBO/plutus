   41 {- Note [Term constructor ordering and numbers]
   42 Ordering of constructors has a small but real effect on efficiency.
   43 It's slightly more efficient to hit the earlier constructors, so it's better to put the more
   44 common ones first.
   45 
   46 The current ordering is based on their *empirically observed* frequency. We should check this
   47 occasionally.
   48 
   49 Additionally, the first 7 (or 3 on 32-bit systems) constructors will get *pointer tags*, which allows
   50 more efficient access when casing on them. So we ideally want to keep the number of constructors
   51 at 7 or fewer.
   52 
   53 We've got 8 constructors, *but* the last one is Error, which is only going to be seen at most
   54 once per program, so it's not too big a deal if it doesn't get a tag.
   55 
   56 See the GHC Notes "Tagging big families" and "Double switching for big families" in
   57 GHC.StgToCmm.Expr for more details.
   58 -}

   60 {- Note [Supported case-expressions]
   61 Originally 'Case' was introduced to only work with 'Constr', which together create the
   62 sums-of-products approach to representing data types.
   63 
   64 However in https://github.com/IntersectMBO/plutus/issues/6602 we decided that the best way to
   65 speed up processing values of built-in types is to extend 'Case' such that it supports pattern
   66 matching on those.
   67 
   68 Currently, 'Case' only supports booleans and integers, but we plan to extend it to lists and data.
   69 
   70 See the @CaseBuiltin DefaultUni@ instance for how casing behaves for supported built-in types.
   71 -}

   73 {-| The type of Untyped Plutus Core terms. Mirrors the type of Typed Plutus Core terms except
   74 
   75 1. all types are removed
   76 2. 'IWrap' and 'Unwrap' are removed
   77 3. type abstractions are replaced with 'Delay'
   78 4. type instantiations are replaced with 'Force'
   79 
   80 The latter two are due to the fact that we don't have value restriction in Typed Plutus Core
   81 and hence a computation can be stuck expecting only a single type argument for the computation
   82 to become unstuck. Therefore we can't just silently remove type abstractions and instantiations and
   83 need to replace them with something else that also blocks evaluation (in order for the semantics
   84 of an erased program to match with the semantics of the original typed one). 'Delay' and 'Force'
   85 serve exactly this purpose. -}
   86 
   87 -- Making all the fields strict gives us a couple of percent in benchmarks
   88 -- See Note [Term constructor ordering and numbers]
   89 data Term name uni fun ann
   90   = Var !ann !name
   91   | LamAbs !ann !name !(Term name uni fun ann)
   92   | Apply !ann !(Term name uni fun ann) !(Term name uni fun ann)
   93   | Force !ann !(Term name uni fun ann)
   94   | Delay !ann !(Term name uni fun ann)
   95   | Constant !ann !(Some (ValueOf uni))
   96   | Builtin !ann !fun
   97   | -- This is the cutoff at which constructors won't get pointer tags
   98     -- See Note [Term constructor ordering and numbers]
   99     Error !ann
  100   | -- TODO: worry about overflow, maybe use an Integer
  101     -- See Note [Constr tag type]
  102     Constr !ann !Word64 ![Term name uni fun ann]
  103   | -- See Note [Supported case-expressions].
  104     Case !ann !(Term name uni fun ann) !(Vector (Term name uni fun ann))
  105   deriving stock (Functor, Generic)
  106 
  107 deriving stock instance
  108   (Show name, GShow uni, Everywhere uni Show, Show fun, Show ann, Closed uni)
  109   => Show (Term name uni fun ann)
  110 
  111 deriving anyclass instance
  112   (NFData name, NFData fun, NFData ann, Everywhere uni NFData, Closed uni)
  113   => NFData (Term name uni fun ann)

  135 instance TermLike (Term name uni fun) TPLC.TyName name uni fun where
  136   var = Var
  137   tyAbs = \ann _ _ -> Delay ann
  138   lamAbs = \ann name _ -> LamAbs ann name
  139   apply = Apply
  140   constant = Constant
  141   builtin = Builtin
  142   tyInst = \ann term _ -> Force ann term
  143   unwrap = const id
  144   iWrap = \_ _ _ -> id
  145   error = \ann _ -> Error ann
  146   constr = \ann _ i es -> Constr ann i es
  147   kase = \ann _ arg cs -> Case ann arg (fromList cs)

  167 -- | Return the outermost annotation of a 'Term'.
  168 termAnn :: Term name uni fun ann -> ann
  169 termAnn (Constant ann _) = ann
  170 termAnn (Builtin ann _) = ann
  171 termAnn (Var ann _) = ann
  172 termAnn (LamAbs ann _ _) = ann
  173 termAnn (Apply ann _ _) = ann
  174 termAnn (Delay ann _) = ann
  175 termAnn (Force ann _) = ann
  176 termAnn (Error ann) = ann
  177 termAnn (Constr ann _ _) = ann
  178 termAnn (Case ann _ _) = ann
  179 
  180 modifyTermAnn :: (ann -> ann) -> Term name uni fun ann -> Term name uni fun ann
  181 modifyTermAnn f = \case
  182   Constant ann c -> Constant (f ann) c
  183   Builtin ann b -> Builtin (f ann) b
  184   Var ann v -> Var (f ann) v
  185   LamAbs ann x body -> LamAbs (f ann) x body
  186   Apply ann fun arg -> Apply (f ann) fun arg
  187   Delay ann body -> Delay (f ann) body
  188   Force ann body -> Force (f ann) body
  189   Error ann -> Error (f ann)
  190   Constr ann i args -> Constr (f ann) i args
  191   Case ann scrut alts -> Case (f ann) scrut alts
  192 
  193 bindFunM
  194   :: Monad m
  195   => (ann -> fun -> m (Term name uni fun' ann))
  196   -> Term name uni fun ann
  197   -> m (Term name uni fun' ann)
  198 bindFunM f = go
  199   where
  200     go (Constant ann val) = pure $ Constant ann val
  201     go (Builtin ann fun) = f ann fun
  202     go (Var ann name) = pure $ Var ann name
  203     go (LamAbs ann name body) = LamAbs ann name <$> go body
  204     go (Apply ann fun arg) = Apply ann <$> go fun <*> go arg
  205     go (Delay ann term) = Delay ann <$> go term
  206     go (Force ann term) = Force ann <$> go term
  207     go (Error ann) = pure $ Error ann
  208     go (Constr ann i args) = Constr ann i <$> traverse go args
  209     go (Case ann arg cs) = Case ann <$> go arg <*> traverse go cs
  210 
  211 bindFun
  212   :: (ann -> fun -> Term name uni fun' ann)
  213   -> Term name uni fun ann
  214   -> Term name uni fun' ann
  215 bindFun f = runIdentity . bindFunM (coerce f)
  216 
  217 mapFun :: (ann -> fun -> fun') -> Term name uni fun ann -> Term name uni fun' ann
  218 mapFun f = bindFun $ \ann fun -> Builtin ann (f ann fun)
