   27 -- | Configuration for the classic pretty-printing.
   28 data PrettyConfigClassic configName = PrettyConfigClassic
   29   { _pccConfigName :: configName
   30   -- ^ How to pretty-print names.
   31   , _pccDisplayAnn :: Bool
   32   -- ^ Whether to display annotations.
   33   }
   34   deriving stock (Show)
   35 
   36 type instance HasPrettyDefaults (PrettyConfigClassic _) = 'True
   37 
   38 -- | The "classically pretty-printable" constraint.
   39 type PrettyClassicBy configName = PrettyBy (PrettyConfigClassic configName)
   40 
   41 type PrettyClassic = PrettyClassicBy PrettyConfigName
   42 
   43 instance configName ~ PrettyConfigName => HasPrettyConfigName (PrettyConfigClassic configName) where
   44   toPrettyConfigName = _pccConfigName
   45 
   46 isEmptyDoc :: Doc ann -> Bool
   47 isEmptyDoc Empty = True
   48 isEmptyDoc _ = False
   49 
   50 {-| Add a pretty-printed annotation to a list of 'Doc's if the given config enables pretty-printing
   51 of annotations. -}
   52 consAnnIf :: Pretty ann => PrettyConfigClassic configName -> ann -> [Doc dann] -> [Doc dann]
   53 consAnnIf config ann rest = filter (not . isEmptyDoc) [pretty ann | _pccDisplayAnn config] ++ rest
   54 
   55 prettyConfigClassic :: PrettyConfigClassic PrettyConfigName
   56 prettyConfigClassic = PrettyConfigClassic prettyConfigName False
   57 
   58 prettyConfigClassicSimple :: PrettyConfigClassic PrettyConfigName
   59 prettyConfigClassicSimple = PrettyConfigClassic prettyConfigNameSimple False
   60 
   61 -- | Pretty-print a value in the default mode using the classic view.
   62 prettyClassic :: PrettyClassic a => a -> Doc ann
   63 prettyClassic = prettyBy prettyConfigClassic
   64 
   65 -- | Pretty-print a value in the simple mode using the classic view.
   66 prettyClassicSimple :: PrettyClassic a => a -> Doc ann
   67 prettyClassicSimple = prettyBy prettyConfigClassicSimple
