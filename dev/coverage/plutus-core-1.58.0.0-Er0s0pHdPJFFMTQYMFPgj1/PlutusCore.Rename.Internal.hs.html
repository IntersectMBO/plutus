   23 {-| A wrapper for signifying that the value inside of it satisfies global uniqueness.
   24 
   25 It's safe to call 'unRenamed', it's not safe to call 'Renamed', hence the latter is only exported
   26 from this internal module and should not be exported from the main API.
   27 
   28 Don't provide any instances allowing the user to create a 'Renamed' (even out of an existing one
   29 like with 'Functor'). -}
   30 newtype Renamed a = Renamed
   31   { unRenamed :: a
   32   }
   33   deriving stock (Show, Eq)
   34 
   35 {-| @Dupable a@ is isomorphic to @a@, but the only way to extract the @a@ is via 'liftDupable'
   36 (defined in the main API module because of a constraint requirement) which renames the stored
   37 value along the way. This type is used whenever
   38 
   39 1. preserving global uniqueness is required
   40 2. some value may be used multiple times
   41 
   42 so we annotate such a value with 'Dupable' and call 'liftDupable' at each usage, which ensures
   43 global uniqueness is preserved.
   44 
   45 'unDupable' is not supposed to be exported. Don't provide any instances allowing the user to
   46 access the underlying value. -}
   47 newtype Dupable a = Dupable
   48   { unDupable :: a
   49   }
   50   deriving stock (Show, Eq)

   52 {-| Replace the unique in the name stored in a 'TyVarDecl' by a new unique, save the mapping
   53 from the old unique to the new one and supply the updated 'TyVarDecl' to a continuation. -}
   54 withFreshenedTyVarDecl
   55   :: (HasRenaming ren TypeUnique, HasUniques (Type tyname uni ann), MonadQuote m, MonadReader ren m)
   56   => TyVarDecl tyname ann
   57   -> (TyVarDecl tyname ann -> m c)
   58   -> m c
   59 withFreshenedTyVarDecl (TyVarDecl ann name kind) cont =
   60   withFreshenedName name $ \nameFr -> cont $ TyVarDecl ann nameFr kind
   61 
   62 {-| Replace the unique in the name stored in a 'VarDecl' by a new unique, save the mapping
   63 from the old unique to the new one and supply to a continuation the computation that
   64 renames the type stored in the updated 'VarDecl'.
   65 The reason the continuation receives a computation rather than a pure term is that we may want
   66 to bring several term and type variables in scope before renaming the types of term variables.
   67 This situation arises when we want to rename a bunch of mutually recursive bindings. -}
   68 withFreshenedVarDecl
   69   :: (HasUniques (Term tyname name uni fun ann), MonadQuote m, MonadReader ScopedRenaming m)
   70   => VarDecl tyname name uni ann
   71   -> (m (VarDecl tyname name uni ann) -> m c)
   72   -> m c
   73 withFreshenedVarDecl (VarDecl ann name ty) cont =
   74   withFreshenedName name $ \nameFr -> cont $ VarDecl ann nameFr <$> renameTypeM ty

   76 -- | Rename a 'Type' in the 'RenameM' monad.
   77 renameTypeM
   78   :: (HasRenaming ren TypeUnique, HasUniques (Type tyname uni ann), MonadQuote m, MonadReader ren m)
   79   => Type tyname uni ann -> m (Type tyname uni ann)
   80 renameTypeM (TyLam ann name kind ty) =
   81   withFreshenedName name $ \nameFr -> TyLam ann nameFr kind <$> renameTypeM ty
   82 renameTypeM (TyForall ann name kind ty) =
   83   withFreshenedName name $ \nameFr -> TyForall ann nameFr kind <$> renameTypeM ty
   84 renameTypeM (TyIFix ann pat arg) = TyIFix ann <$> renameTypeM pat <*> renameTypeM arg
   85 renameTypeM (TyApp ann fun arg) = TyApp ann <$> renameTypeM fun <*> renameTypeM arg
   86 renameTypeM (TyFun ann dom cod) = TyFun ann <$> renameTypeM dom <*> renameTypeM cod
   87 renameTypeM (TyVar ann name) = TyVar ann <$> renameNameM name
   88 renameTypeM (TySOP ann tyls) = TySOP ann <$> (traverse . traverse) renameTypeM tyls
   89 renameTypeM ty@TyBuiltin {} = pure ty
   90 
   91 -- | Rename a 'Term' in the 'RenameM' monad.
   92 renameTermM
   93   :: (HasUniques (Term tyname name uni fun ann), MonadQuote m, MonadReader ScopedRenaming m)
   94   => Term tyname name uni fun ann -> m (Term tyname name uni fun ann)
   95 renameTermM (LamAbs ann name ty body) =
   96   withFreshenedName name $ \nameFr -> LamAbs ann nameFr <$> renameTypeM ty <*> renameTermM body
   97 renameTermM (TyAbs ann name kind body) =
   98   withFreshenedName name $ \nameFr -> TyAbs ann nameFr kind <$> renameTermM body
   99 renameTermM (IWrap ann pat arg term) =
  100   IWrap ann <$> renameTypeM pat <*> renameTypeM arg <*> renameTermM term
  101 renameTermM (Apply ann fun arg) = Apply ann <$> renameTermM fun <*> renameTermM arg
  102 renameTermM (Unwrap ann term) = Unwrap ann <$> renameTermM term
  103 renameTermM (Error ann ty) = Error ann <$> renameTypeM ty
  104 renameTermM (TyInst ann term ty) = TyInst ann <$> renameTermM term <*> renameTypeM ty
  105 renameTermM (Var ann name) = Var ann <$> renameNameM name
  106 renameTermM (Constr ann ty i es) = Constr ann <$> renameTypeM ty <*> pure i <*> traverse renameTermM es
  107 renameTermM (Case ann ty arg cs) = Case ann <$> renameTypeM ty <*> renameTermM arg <*> traverse renameTermM cs
  108 renameTermM con@Constant {} = pure con
  109 renameTermM bi@Builtin {} = pure bi
  110 
  111 -- | Rename a 'Program' in the 'RenameM' monad.
  112 renameProgramM
  113   :: (HasUniques (Program tyname name uni fun ann), MonadQuote m, MonadReader ScopedRenaming m)
  114   => Program tyname name uni fun ann -> m (Program tyname name uni fun ann)
  115 renameProgramM (Program ann ver term) = Program ann ver <$> renameTermM term
