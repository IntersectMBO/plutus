   29 -- | A Plutus Core (Scott-encoded) tuple.
   30 data Tuple term uni ann
   31   = Tuple
   32   { _tupleElementTypes :: [Type TyName uni ann]
   33   -- ^ The types of elements of a tuple.
   34   , _tupleTerm :: term ann
   35   -- ^ A term representation of the tuple.
   36   }
   37 
   38 {-| Get the type of a 'Tuple'.
   39 
   40 > getTupleType _ (Tuple [a1, ... , an] _) = all r. (a1 -> ... -> an -> r) -> r -}
   41 getTupleType :: MonadQuote m => ann -> Tuple term uni ann -> m (Type TyName uni ann)
   42 getTupleType ann (Tuple elTys _) = liftQuote $ do
   43   r <- freshTyName "r"
   44   let caseTy = mkIterTyFun ann elTys $ TyVar ann r
   45   pure . TyForall ann r (Type ann) . TyFun ann caseTy $ TyVar ann r
   46 
   47 {-| Convert a Haskell spine of 'Term's to a PLC 'Tuple'.
   48 
   49 > getSpineToTuple _ [(a1, x1), ... , (an, xn)] =
   50 >     Tuple [a1, ... , an] (/\(r :: *) -> \(f :: a1 -> ... -> an -> r) -> f x1 ... xn) -}
   51 getSpineToTuple
   52   :: (TermLike term TyName Name uni fun, MonadQuote m)
   53   => ann -> [(Type TyName uni ann, term ann)] -> m (Tuple term uni ann)
   54 getSpineToTuple ann spine = liftQuote $ do
   55   r <- freshTyName "r"
   56   f <- freshName "f"
   57   let (as, xs) = unzip spine
   58       caseTy = mkIterTyFun ann as $ TyVar ann r
   59       y = mkIterApp (var ann f) ((ann,) <$> xs)
   60   pure . Tuple as . tyAbs ann r (Type ann) $ lamAbs ann f caseTy y

   62 {-| Get the type of the ith element of a 'Tuple' along with the element itself.
   63 
   64 > tupleTypeTermAt _ i (Tuple [a0, ... , an] term) =
   65 >     (ai, term {ai} (\(x0 : a0) ... (xn : an) -> xi)) -}
   66 tupleTypeTermAt
   67   :: (TermLike term TyName Name uni fun, MonadQuote m)
   68   => ann -> Int -> Tuple term uni ann -> m (Type TyName uni ann, term ann)
   69 tupleTypeTermAt ann ind (Tuple elTys term) = liftQuote $ do
   70   args <- ifor elTys $ \i ty -> do
   71     n <- freshName $ "arg_" <> showText i
   72     pure $ VarDecl ann n ty
   73   let selectedTy = elTys !! ind
   74       selectedArg = mkVar $ args !! ind
   75       selector = mkIterLamAbs args selectedArg
   76 
   77   pure
   78     ( selectedTy
   79     , apply ann (tyInst ann term selectedTy) selector
   80     )
   81 
   82 -- | Get the ith element of a 'Tuple'.
   83 tupleTermAt
   84   :: (TermLike term TyName Name uni fun, MonadQuote m)
   85   => ann -> Int -> Tuple term uni ann -> m (term ann)
   86 tupleTermAt ann ind tuple = snd <$> tupleTypeTermAt ann ind tuple
   87 
   88 -- | Get the ith element of a 'Tuple' as a 'TermDef'.
   89 tupleDefAt
   90   :: (TermLike term TyName Name uni fun, MonadQuote m)
   91   => ann
   92   -> Int
   93   -> Name
   94   -> Tuple term uni ann
   95   -> m (TermDef term TyName Name uni ann)
   96 tupleDefAt ann ind name tuple = uncurry (Def . VarDecl ann name) <$> tupleTypeTermAt ann ind tuple

   98 {-| Bind all elements of a 'Tuple' inside a 'Term'.
   99 
  100 > bindTuple _ [x_1, ... , x_n] (Tuple [a1, ... , an] term) body =
  101 >     (\(tup : all r. (a_1 -> ... -> a_n -> r) -> r) ->
  102 >       let x_1 = _1 tup
  103 >           ...
  104 >           x_n = _n tup
  105 >         in body
  106 >     ) term -}
  107 bindTuple
  108   :: (TermLike term TyName Name uni fun, MonadQuote m)
  109   => ann -> [Name] -> Tuple term uni ann -> term ann -> m (term ann)
  110 bindTuple ann names (Tuple elTys term) body = liftQuote $ do
  111   tup <- freshName "tup"
  112   let tupVar = Tuple elTys $ var ann tup
  113   tupTy <- getTupleType ann tupVar
  114   tupDefs <- itraverse (\i name -> tupleDefAt ann i name tupVar) names
  115   pure $ apply ann (lamAbs ann tup tupTy $ foldr (termLet ann) body tupDefs) term

  117 {-| Given an arity @n@, create the n-ary product type.
  118 
  119 @\(T_1 :: *) .. (T_n :: *) . all (R :: *) . (T_1 -> .. -> T_n -> R) -> R@ -}
  120 prodN :: Int -> Type TyName uni ()
  121 prodN arity = runQuote $ do
  122   tyVars <- for [0 .. (arity - 1)] $ \i -> do
  123     tn <- liftQuote $ freshTyName $ "t_" <> showText i
  124     pure $ TyVarDecl () tn $ Type ()
  125 
  126   resultType <- liftQuote $ freshTyName "r"
  127   let caseType = mkIterTyFun () (fmap (mkTyVar ()) tyVars) (TyVar () resultType)
  128   pure $
  129     -- \T_1 .. T_n
  130     mkIterTyLam tyVars $
  131       -- all R
  132       TyForall () resultType (Type ()) $
  133         -- (T_1 -> .. -> T_n -> r) -> r
  134         TyFun () caseType (TyVar () resultType)
  135 
  136 {-| Given an arity @n@, create the constructor for n-ary products.
  137 
  138 @
  139     /\(T_1 :: *) .. (T_n :: *) .
  140         \(arg_1 : T_1) .. (arg_n : T_n) .
  141             /\(R :: *).
  142                 \(case : T_1 -> .. -> T_n -> R) -> case arg_1 .. arg_n
  143 @ -}
  144 prodNConstructor :: TermLike term TyName Name uni fun => Int -> term ()
  145 prodNConstructor arity = runQuote $ do
  146   tyVars <- for [0 .. (arity - 1)] $ \i -> do
  147     tn <- liftQuote $ freshTyName $ "t_" <> showText i
  148     pure $ TyVarDecl () tn $ Type ()
  149 
  150   resultType <- liftQuote $ freshTyName "r"
  151 
  152   args <- for [0 .. (arity - 1)] $ \i -> do
  153     n <- liftQuote $ freshName $ "arg_" <> showText i
  154     pure $ VarDecl () n $ mkTyVar () $ tyVars !! i
  155 
  156   caseArg <- liftQuote $ freshName "case"
  157   let caseTy = mkIterTyFun () (fmap (mkTyVar ()) tyVars) (TyVar () resultType)
  158   pure $
  159     -- /\T_1 .. T_n
  160     mkIterTyAbs tyVars $
  161       -- \arg_1 .. arg_n
  162       mkIterLamAbs args $
  163         -- /\R
  164         tyAbs () resultType (Type ()) $
  165           -- \case
  166           lamAbs () caseArg caseTy $
  167             -- case arg_1 .. arg_n
  168             mkIterAppNoAnn (var () caseArg) $
  169               fmap mkVar args
  170 
  171 {-| Given an arity @n@ and an index @i@, create a function for accessing the i'th component of a n-tuple.
  172 
  173 @
  174     /\(T_1 :: *) .. (T_n :: *) .
  175         \(tuple : all (R :: *) . (T_1 -> .. -> T_n -> R) -> R)) .
  176             tuple {T_i} (\(arg_1 : T_1) .. (arg_n : T_n) . arg_i)
  177 @ -}
  178 prodNAccessor :: TermLike term TyName Name uni fun => Int -> Int -> term ()
  179 prodNAccessor arity index = runQuote $ do
  180   tyVars <- for [0 .. (arity - 1)] $ \i -> do
  181     tn <- liftQuote $ freshTyName $ "t_" <> showText i
  182     pure $ TyVarDecl () tn $ Type ()
  183 
  184   let tupleTy = mkIterTyAppNoAnn (prodN arity) (fmap (mkTyVar ()) tyVars)
  185       selectedTy = mkTyVar () $ tyVars !! index
  186 
  187   args <- for [0 .. (arity - 1)] $ \i -> do
  188     n <- liftQuote $ freshName $ "arg_" <> showText i
  189     pure $ VarDecl () n $ mkTyVar () $ tyVars !! i
  190   let selectedArg = mkVar $ args !! index
  191 
  192   tupleArg <- liftQuote $ freshName "tuple"
  193   pure $
  194     -- /\T_1 .. T_n
  195     mkIterTyAbs tyVars $
  196       -- \tuple :: (tupleN T_1 .. T_n)
  197       lamAbs () tupleArg tupleTy $
  198         -- tuple {T_i}
  199         apply () (tyInst () (var () tupleArg) selectedTy) $
  200           -- \arg_1 .. arg_n . arg_i
  201           mkIterLamAbs args selectedArg
