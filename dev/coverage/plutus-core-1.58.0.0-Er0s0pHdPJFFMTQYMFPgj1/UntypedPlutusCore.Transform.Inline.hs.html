  203 {-| Inline simple bindings. Relies on global uniqueness, and preserves it.
  204 See Note [Inlining and global uniqueness] -}
  205 inline
  206   :: forall name uni fun m a
  207    . ExternalConstraints name uni fun m
  208   => AstSize
  209   -- ^ inline threshold
  210   -> Bool
  211   -- ^ inline constants
  212   -> Bool
  213   -- ^ preserve logging
  214   -> InlineHints name a
  215   -> PLC.BuiltinSemanticsVariant fun
  216   -> Term name uni fun a
  217   -> SimplifierT name uni fun a m (Term name uni fun a)
  218 inline
  219   callsiteGrowth
  220   inlineConstants
  221   preserveLogging
  222   hints
  223   builtinSemanticsVariant
  224   t = do
  225     decoratedResult <-
  226       liftQuote $
  227         flip evalStateT mempty $
  228           runReaderT
  229             (processTerm (emptyDecoration <$> t))
  230             InlineInfo
  231               { _iiUsages = Usages.termUsages t
  232               , _iiHints = hints
  233               , _iiBuiltinSemanticsVariant = builtinSemanticsVariant
  234               , _iiInlineConstants = inlineConstants
  235               , _iiInlineCallsiteGrowth = callsiteGrowth
  236               , _iiPreserveLogging = preserveLogging
  237               }
  238     let result = snd <$> decoratedResult
  239     recordSimplificationWithHints (CertifierHints.Inline (mkHints decoratedResult)) t Inline result
  240     return result

  244 {-| Extract the list of applications from a term,
  245 a bit like a "multi-beta" reduction.
  246 
  247 Some examples will help (annotations are omitted):
  248 [(\x . t) a] -> Just ([(x, a)], t)
  249 
  250 [[[(\x . (\y . (\z . t))) a] b] c] -> Just ([(x, a), (y, b), (z, c)]) t)
  251 
  252 [[(\x . t) a] b] -> Nothing -}
  253 extractApps
  254   :: Term name uni fun a
  255   -> Maybe ([(UTermDef name uni fun a, a)], Term name uni fun a)
  256 extractApps = go []
  257   where
  258     go argStack (Apply aa f arg) = go ((arg, aa) : argStack) f
  259     go argStack t = matchArgs argStack [] t
  260     matchArgs ((arg, aa) : rest) acc (LamAbs a n body) =
  261       matchArgs rest ((Def (UVarDecl a n) arg, aa) : acc) body
  262     matchArgs [] acc t =
  263       if null acc then Nothing else Just (reverse acc, t)
  264     matchArgs (_ : _) _ _ = Nothing
  265 
  266 -- | The inverse of 'extractApps'.
  267 restoreApps
  268   :: [(Either (Ann a) (UTermDef name uni fun (Ann a)), Ann a)]
  269   -> Term name uni fun (Ann a)
  270   -> Term name uni fun (Ann a)
  271 restoreApps defs t = makeLams [] t (reverse defs)
  272   where
  273     -- `aa` is the annotation on the Apply node, and `al` is the annotation
  274     -- on the LamAbs node.
  275     makeLams args acc ((Left al, aa) : rest) =
  276       -- `Left` means the binding is dropped, so `acc` should be decorated with `al`.
  277       -- This corresponds to Note [Inliner's Certifier Hints], #3.
  278       makeLams ((Nothing, aa) : args) (decorateWith (al ^. decorations) acc) rest
  279     makeLams args acc ((Right (Def (UVarDecl al n) rhs), aa) : rest) =
  280       makeLams ((Just rhs, aa) : args) (LamAbs al n acc) rest
  281     makeLams args acc [] =
  282       makeApps args acc
  283 
  284     makeApps ((Nothing, aa) : args) acc =
  285       -- `Nothing` means the binding is dropped, so we decorate `acc with `aa` plus `DDrop`.
  286       -- This corresponds to Note [Inliner's Certifier Hints], #2.
  287       makeApps args (decorateWith (aa ^. decorations ++ [DDrop]) acc)
  288     makeApps ((Just arg, aa) : args) acc =
  289       makeApps args (Apply aa acc arg)
  290     makeApps [] acc = acc

  292 -- | Run the inliner on a `UntypedPlutusCore.Core.Type.Term`.
  293 processTerm
  294   :: forall name uni fun a
  295    . InliningConstraints name uni fun
  296   => Term name uni fun (Ann a)
  297   -> InlineM name uni fun a (Term name uni fun (Ann a))
  298 processTerm = handleTerm
  299   where
  300     handleTerm
  301       :: Term name uni fun (Ann a)
  302       -> InlineM name uni fun a (Term name uni fun (Ann a))
  303     handleTerm = \case
  304       v@(Var a n) ->
  305         maybe
  306           v
  307           -- See Note [Inliner's Certifier Hints], #1
  308           (decorateWith (a ^. decorations) . decorateWith [DExpand])
  309           <$> substName n
  310       -- See Note [Differences from PIR inliner] 3
  311       (extractApps -> Just (bs, t)) -> do
  312         ebs <- traverse (processSingleBinding t) bs
  313         t' <- processTerm t
  314         pure $ restoreApps ebs t'
  315       t -> inlineSaturatedApp =<< forMOf termSubterms t processTerm
  316 
  317     -- See Note [Renaming strategy]
  318     substName :: name -> InlineM name uni fun a (Maybe (Term name uni fun (Ann a)))
  319     substName name = gets (lookupTerm name) >>= traverse renameTerm
  320 
  321     -- See Note [Inlining approach and 'Secrets of the GHC Inliner']
  322     renameTerm
  323       :: InlineTerm name uni fun (Ann a)
  324       -> InlineM name uni fun a (Term name uni fun (Ann a))
  325     renameTerm = \case
  326       -- Already processed term, just rename and put it in, don't do any
  327       -- further optimization here.
  328       Done t -> liftDupable t
  329 
  330 processSingleBinding
  331   :: InliningConstraints name uni fun
  332   => Term name uni fun (Ann a)
  333   -> (UTermDef name uni fun (Ann a), Ann a)
  334   -> InlineM name uni fun a (Either (Ann a) (UTermDef name uni fun (Ann a)), Ann a)
  335 processSingleBinding body (Def vd@(UVarDecl a n) rhs0, aa) =
  336   (,aa) <$> do
  337     maybeAddSubst body (snd a) n rhs0 >>= \case
  338       Just rhs -> do
  339         let (binders, rhsBody) = UPLC.splitParams rhs
  340         modify' . extendVarInfo n $
  341           VarInfo
  342             { _varBinders = binders
  343             , _varRhs = rhs
  344             , _varRhsBody = Done (dupable rhsBody)
  345             }
  346         pure . Right $ Def vd rhs
  347       Nothing -> pure (Left a)

  349 {-| Check against the heuristics we have for inlining and either inline
  350 the term binding or not. The arguments to this function are the fields of the
  351 'TermBinding' being processed.
  352 Nothing means that we are inlining the term:
  353   * we have extended the substitution, and
  354   * we are removing the binding (hence we return Nothing). -}
  355 maybeAddSubst
  356   :: forall name uni fun a
  357    . InliningConstraints name uni fun
  358   => Term name uni fun (Ann a)
  359   -> a
  360   -> name
  361   -> Term name uni fun (Ann a)
  362   -> InlineM name uni fun a (Maybe (Term name uni fun (Ann a)))
  363 maybeAddSubst body a n rhs0 = do
  364   rhs <- processTerm rhs0
  365 
  366   -- Check whether we've been told specifically to inline this
  367   hints <- view iiHints
  368   case shouldInline hints a n of
  369     AlwaysInline ->
  370       -- if we've been told specifically, then do it right away
  371       extendAndDrop (Done $ dupable rhs)
  372     hint ->
  373       let safeToInline = hint == SafeToInline
  374        in ifM
  375             (shouldUnconditionallyInline safeToInline n rhs body)
  376             (extendAndDrop (Done $ dupable rhs))
  377             (pure $ Just rhs)
  378   where
  379     extendAndDrop
  380       :: forall b
  381        . InlineTerm name uni fun (Ann a)
  382       -> InlineM name uni fun a (Maybe b)
  383     extendAndDrop t = modify' (extendTerm n t) >> pure Nothing
  384 
  385 shouldUnconditionallyInline
  386   :: InliningConstraints name uni fun
  387   => Bool
  388   {-^ Whether we know that the binding is safe to inline.
  389   If so, bypass the purity check. -}
  390   -> name
  391   -> Term name uni fun a
  392   -> Term name uni fun b
  393   -> InlineM name uni fun c Bool
  394 shouldUnconditionallyInline safe n rhs body = do
  395   isTermPure <- checkPurity rhs
  396   inlineConstants <- view iiInlineConstants
  397   preUnconditional isTermPure ||^ postUnconditional inlineConstants isTermPure
  398   where
  399     -- similar to the paper, preUnconditional inlining checks that the binder
  400     -- is 'OnceSafe'.  I.e., it's used at most once AND it neither duplicate code
  401     -- or work. While we don't check for lambda etc like in the paper,
  402     -- 'effectSafe' ensures that it isn't doing any substantial work.
  403     -- We actually also inline 'Dead' binders (i.e., remove dead code) here.
  404     preUnconditional isTermPure =
  405       nameUsedAtMostOnce n &&^ (pure safe ||^ effectSafe body n isTermPure)
  406     -- See Note [Inlining approach and 'Secrets of the GHC Inliner'] and
  407     -- [Inlining and purity]. This is the case where we don't know that the number
  408     -- of occurrences is exactly one, so there's no point checking if the term is
  409     -- immediately evaluated.
  410     postUnconditional inlineConstants isTermPure =
  411       pure (safe || isTermPure) &&^ acceptable inlineConstants rhs
  412 
  413 -- | Check if term is pure. See Note [Inlining and purity]
  414 checkPurity
  415   :: PLC.ToBuiltinMeaning uni fun
  416   => Term name uni fun a
  417   -> InlineM name uni fun b Bool
  418 checkPurity t = do
  419   builtinSemanticsVariant <- view iiBuiltinSemanticsVariant
  420   pure $ isPure builtinSemanticsVariant t

  422 nameUsedAtMostOnce
  423   :: forall name uni fun a
  424    . InliningConstraints name uni fun
  425   => name
  426   -> InlineM name uni fun a Bool
  427 nameUsedAtMostOnce n = do
  428   usgs <- view iiUsages
  429   -- 'inlining' terms used 0 times is a cheap way to remove dead code
  430   -- while we're here
  431   pure $ Usages.getUsageCount n usgs <= 1

  433 isFirstVarBeforeEffects
  434   :: forall name uni fun ann
  435    . InliningConstraints name uni fun
  436   => BuiltinSemanticsVariant fun
  437   -> name
  438   -> Term name uni fun ann
  439   -> Bool
  440 isFirstVarBeforeEffects builtinSemanticsVariant n t =
  441   -- This can in the worst case traverse a lot of the term, which could lead to
  442   -- us doing ~quadratic work as we process the program. However in practice
  443   -- most terms have a relatively short evaluation order before we hit Unknown,
  444   -- so it's not too bad.
  445   go (unEvalOrder (termEvaluationOrder builtinSemanticsVariant t))
  446   where
  447     -- Found the variable we're looking for!
  448     go ((EvalTerm _ _ (Var _ n')) : _) | n == n' = True
  449     -- Found a pure term, ignore it and continue
  450     go ((EvalTerm Pure _ _) : rest) = go rest
  451     -- Found a possibly impure term, our variable is definitely not first
  452     go ((EvalTerm MaybeImpure _ _) : _) = False
  453     -- Don't know, be conservative
  454     go (Unknown : _) = False
  455     go [] = False

  457 {-| Check if the given name is strict in the given term.
  458   This means that at least one occurrence of the name is found outside of the following:
  459   * 'delay' term
  460   * lambda body
  461   * case branch -}
  462 isStrictIn
  463   :: forall name uni fun a
  464    . Eq name
  465   => name
  466   -> Term name uni fun a
  467   -> Bool
  468 isStrictIn name = go
  469   where
  470     go :: Term name uni fun a -> Bool
  471     go = \case
  472       Var _ann name' -> name == name'
  473       LamAbs _ann _paramName _body -> False
  474       Apply _ann t1 t2 -> go t1 || go t2
  475       Force _ann t -> go t
  476       Delay _ann _term -> False
  477       Constant {} -> False
  478       Builtin {} -> False
  479       Error {} -> False
  480       Constr _ann _idx terms -> any go terms
  481       Case _ann scrut _branches -> go scrut

  483 effectSafe
  484   :: forall name uni fun a b
  485    . InliningConstraints name uni fun
  486   => Term name uni fun a
  487   -> name
  488   -> Bool
  489   -- ^ is it pure? See Note [Inlining and purity]
  490   -> InlineM name uni fun b Bool
  491 effectSafe body n termIsPure = do
  492   preserveLogging <- view iiPreserveLogging
  493   builtinSemantics <- view iiBuiltinSemanticsVariant
  494   return $
  495     termIsPure
  496       || isFirstVarBeforeEffects builtinSemantics n body
  497       || (not preserveLogging && isStrictIn n body)

  499 {-| Should we inline? Should only inline things that won't duplicate work
  500 or code.  See Note [Inlining approach and 'Secrets of the GHC Inliner'] -}
  501 acceptable
  502   :: Bool
  503   -- ^ inline constants
  504   -> Term name uni fun a
  505   -> InlineM name uni fun b Bool
  506 acceptable inlineConstants t =
  507   -- See Note [Inlining criteria]
  508   pure $ costIsAcceptable t && sizeIsAcceptable inlineConstants t
  509 
  510 {-| Is the cost increase (in terms of evaluation work) of inlining a variable
  511 whose RHS is the given term acceptable? -}
  512 costIsAcceptable :: Term name uni fun a -> Bool
  513 costIsAcceptable = \case
  514   Builtin {} -> True
  515   Var {} -> True
  516   Constant {} -> True
  517   Error {} -> True
  518   -- This will mean that we create closures at each use site instead of
  519   -- once, but that's a very low cost which we're okay rounding to 0.
  520   LamAbs {} -> True
  521   Apply {} -> False
  522   -- Inlining constructors of size 1 or 0 seems okay, but does result in doing
  523   -- the work for the elements at each use site.
  524   Constr _ _ es ->
  525     case es of
  526       [] -> True
  527       [e] -> costIsAcceptable e
  528       _ -> False
  529   -- Inlining a case means redoing the match at each use site
  530   Case {} -> False
  531   Force {} -> False
  532   Delay {} -> True
  533 
  534 {-| Is the size increase (in the AST) of inlining a variable whose RHS is
  535 the given term acceptable? -}
  536 sizeIsAcceptable
  537   :: Bool
  538   -- ^ inline constants
  539   -> Term name uni fun a
  540   -> Bool
  541 sizeIsAcceptable inlineConstants = \case
  542   Builtin {} -> True
  543   Var {} -> True
  544   Error {} -> True
  545   -- See Note [Differences from PIR inliner] 4
  546   LamAbs {} -> False
  547   -- Inlining constructors of size 1 or 0 seems okay
  548   Constr _ _ es -> case es of
  549     [] -> True
  550     [e] -> sizeIsAcceptable inlineConstants e
  551     _ -> False
  552   -- Cases are pretty big, due to the case branches
  553   Case {} -> False
  554   -- Inlining constants is deemed acceptable if the 'inlineConstants'
  555   -- flag is turned on, see Note [Inlining constants].
  556   Constant {} -> inlineConstants
  557   Apply {} -> False
  558   Force _ t -> sizeIsAcceptable inlineConstants t
  559   Delay _ t -> sizeIsAcceptable inlineConstants t

  561 -- | Fully apply and beta reduce.
  562 fullyApplyAndBetaReduce
  563   :: forall name uni fun a
  564    . InliningConstraints name uni fun
  565   => Ann a
  566   -- ^ Annotation on the variable to be replaced
  567   -> VarInfo name uni fun (Ann a)
  568   -> [(Ann a, Term name uni fun (Ann a))]
  569   -> InlineM name uni fun a (Maybe (Term name uni fun (Ann a)))
  570 fullyApplyAndBetaReduce av info args0 = do
  571   rhsBody <-
  572     -- Since we would like to fully apply here, we add decorations to
  573     -- `rhsBody` in the following order:
  574     --
  575     --   * decorations on the outermost Apply node, plus DDrop
  576     --   * decorations on the next Apply node, plus DDrop
  577     --   * ...
  578     --   * decorations on the innermost Apply node, plus DDrop
  579     --   * `av` plus DExpand.
  580     --   * decorations on the first LamAbs
  581     --   * decorations on the second LamAbs
  582     --   * ...
  583     --   * decorations on the last LamAbs
  584     --
  585     -- See Note [Inliner's Certifier Hints].
  586     decorateWith
  587       ( concatMap (\(annApply, _arg) -> annApply ^. decorations ++ [DDrop]) (reverse args0)
  588           ++ av ^. decorations
  589           ++ [DExpand]
  590           ++ concatMap (\(_name, annLam) -> annLam ^. decorations) (info ^. varBinders)
  591       )
  592       <$> liftDupable (let Done rhsBody = info ^. varRhsBody in rhsBody)
  593   let go
  594         :: Term name uni fun (Ann a)
  595         -> [(name, Ann a)]
  596         -> [(Ann a, Term name uni fun (Ann a))]
  597         -> InlineM name uni fun a (Maybe (Term name uni fun (Ann a)))
  598       go acc bs args = case (bs, args) of
  599         ([], _) -> pure . Just $ mkIterApp acc args
  600         ((param, _annLam) : params, (_annArg, arg) : args') -> do
  601           safe <- safeToBetaReduce param arg
  602           if safe
  603             then do
  604               acc' <-
  605                 termSubstNamesM
  606                   ( \n a ->
  607                       if n == param
  608                         then
  609                           Just
  610                             -- See Note [Inliner's Certifier Hints], #1
  611                             . decorateWith (a ^. decorations)
  612                             . decorateWith [DExpand]
  613                             <$> PLC.rename arg
  614                         else pure Nothing
  615                   )
  616                   acc
  617               go acc' params args'
  618             else pure Nothing
  619         _ -> pure Nothing
  620 
  621       -- Is it safe to turn `(\a -> body) arg` into `body [a := arg]`?
  622       -- The criteria is the same as the criteria for unconditionally
  623       -- inlining `a`, since inlining is the same as beta reduction.
  624       safeToBetaReduce
  625         :: name
  626         -> Term name uni fun (Ann a)
  627         -> InlineM name uni fun a Bool
  628       safeToBetaReduce a arg = shouldUnconditionallyInline False a arg rhsBody
  629   go rhsBody (info ^. varBinders) args0
  630 
  631 {-| This works in the same way as
  632 'PlutusIR.Transform.Inline.CallSiteInline.inlineSaturatedApp'.
  633 See Note [Inlining and beta reduction of functions]. -}
  634 inlineSaturatedApp
  635   :: forall name uni fun a
  636    . InliningConstraints name uni fun
  637   => Term name uni fun (Ann a)
  638   -> InlineM name uni fun a (Term name uni fun (Ann a))
  639 inlineSaturatedApp t
  640   | (Var av name, args) <- UPLC.splitApplication t =
  641       gets (lookupVarInfo name) >>= \case
  642         Nothing -> pure t
  643         Just varInfo ->
  644           fullyApplyAndBetaReduce av varInfo args >>= \case
  645             Nothing -> pure t
  646             Just fullyApplied -> do
  647               thresh <- view iiInlineCallsiteGrowth
  648               let
  649                 -- Inline only if the size is no bigger than
  650                 -- not inlining plus threshold.
  651                 sizeIsOk =
  652                   termAstSize fullyApplied <= termAstSize t + max 0 thresh
  653                 rhs = varInfo ^. varRhs
  654                 -- Cost is always OK if the RHS is a LamAbs,
  655                 -- but may not be otherwise.
  656                 costIsOk = costIsAcceptable rhs
  657               -- The RHS is always pure if it is a LamAbs,
  658               -- but may not be otherwise.
  659               rhsPure <- checkPurity rhs
  660               pure $
  661                 if sizeIsOk && costIsOk && rhsPure
  662                   then fullyApplied
  663                   else t
  664   | otherwise = pure t

  671 -- | An enclosing Drop or Expand layer
  672 data Decoration = DDrop | DExpand
  673 
  674 type Ann a = ([Decoration], a)
  675 
  676 emptyDecoration :: a -> Ann a
  677 emptyDecoration = (mempty,)
  678 {-# INLINE emptyDecoration #-}
  679 
  680 decorations :: Lens' (Ann a) [Decoration]
  681 decorations = _1
  682 {-# INLINE decorations #-}
  683 
  684 -- | Prepend the given decorations
  685 decorateWith :: [Decoration] -> Term name uni fun (Ann a) -> Term name uni fun (Ann a)
  686 decorateWith ds = modifyTermAnn (first (ds ++))
  687 {-# INLINE decorateWith #-}

  689 -- | Fold a decorated term into certifier hints.
  690 mkHints :: Term name uni fun (Ann a) -> CertifierHints.Inline
  691 mkHints = go
  692   where
  693     go t = decorate hints ds
  694       where
  695         decorate :: CertifierHints.Inline -> [Decoration] -> CertifierHints.Inline
  696         decorate = foldr $ \d h -> case d of
  697           DDrop -> CertifierHints.InlDrop h
  698           DExpand -> CertifierHints.InlExpand h
  699 
  700         ds = termAnn t ^. decorations
  701 
  702         hints = case t of
  703           Var {} -> CertifierHints.InlVar
  704           LamAbs _ _ body -> CertifierHints.InlLam (go body)
  705           Apply _ fun arg -> CertifierHints.InlApply (go fun) (go arg)
  706           Force _ body -> CertifierHints.InlForce (go body)
  707           Delay _ body -> CertifierHints.InlDelay (go body)
  708           Constant {} -> CertifierHints.InlCon
  709           Builtin {} -> CertifierHints.InlBuiltin
  710           Error {} -> CertifierHints.InlError
  711           Constr _ _ args -> CertifierHints.InlConstr (go <$> args)
  712           Case _ scrut alts -> CertifierHints.InlCase (go scrut) (go <$> V.toList alts)

  714 {- Note [Inliner's Certifier Hints]
  715 
  716 To certify the inliner, the certifier requires the inliner to emit hints that record
  717 where inlining operations occur. These hints are represented by the `Inline` data type
  718 defined in `UntypedPlutusCore.Transform.Certify.Hints`.
  719 
  720 Given a pre-term and a post-term, the expected hints can be defined by viewing the
  721 inliner as a process that repeatedly performs one of the following two operations:
  722 
  723 1. Substituting a term for a variable
  724 2. Dropping a dead binding
  725 
  726 Certifier hints are then defined constructively according to the following procedure.
  727 In the end, the final hints mirror the post-term structure, except that each node may be
  728 decorated with an arbitrary number of `InlDrop` or `InlExpand` layers, reflecting the
  729 inlining operations that the inliner has performed.
  730 
  731 - Initially the hints mirror the pre-term structure, using constructors of `Inline`
  732   except `InlExpand` and `InlDrop`. Excluding these two constructors, there is a
  733   1-1 correspondence between hints constructors and AST constructors.
  734 
  735 - Suppose the inliner performs operation 1, substituting term `N` for variable `v`.
  736   Before the substitution, we have
  737 
  738     hints(v) = decorations(InlVar)
  739 
  740   where `decorations` denotes an arbitrary number of enclosing `InlDrop` or `InlExpand`
  741   layers. After the substitution, the hints at this location should become
  742 
  743     decorations(InlExpand hints(N))                         (#1)
  744 
  745   In other words, `InlVar` is replaced by `hints(N)` wrapped in an additional `InlExpand`.
  746 
  747 - Suppose the inliner performs operation 2, turning `(\x y. M) X Y` into `(\x. M) X`,
  748   where `x` and `X` denote sequences `x x ... x` and `X X ... X` with `n >= 0`.
  749   Before this, we have
  750 
  751     hints((\x y. M) X Y) = decorations(InlApply (hints((\x y. M) X)) (hints(Y)))
  752     hints(\y. M) = decorations(InlLam hints(M))
  753 
  754   Afterwards, the hints for `(\x y. M) X` should become
  755 
  756     decorations(InlDrop (hints((\x y. M) X)))           (#2)
  757 
  758   That is, it is decorated with `decorations` plus `InlDrop`.
  759   The hints for `M` should become
  760 
  761     decorations(hints(M))                                   (#3)
  762 
  763   That is, it is decorated with `decorations`. Note that when `n = 0`,
  764   `(\x y. M) X` becomes `M`, and therefore, `hints(M)` should be decorated with
  765   `decorations` plus `InlDrop` plus `decorations`.
  766 
  767 Refer to the golden tests for examples of hints (look for files ending with
  768 `.golden.certifier-hints`).
  769 
  770 Implementing this is quite fiddly, but is still easier than it sounds from the above
  771 description. Observe that at any point during inlining, the hints structure mirrors
  772 the structure of the current term, except for the decorations. Therefore all we need
  773 is to store the decorations of each node in its `ann`. When inlining is done, we can
  774 construct the hints structure using these decorations.
  775 -}
