   38 -- | Whether to pretty-print PLC errors in full or with some information omitted.
   39 data CondensedErrors
   40   = CondensedErrorsYes
   41   | CondensedErrorsNo
   42   deriving stock (Show, Eq)
   43 
   44 -- | Options for pretty-printing PLC entities.
   45 newtype PrettyConfigPlcOptions = PrettyConfigPlcOptions
   46   { _pcpoCondensedErrors :: CondensedErrors
   47   }
   48   deriving stock (Show)
   49 
   50 -- | Strategy for pretty-printing PLC entities.
   51 data PrettyConfigPlcStrategy
   52   = PrettyConfigPlcClassic (PrettyConfigClassic PrettyConfigName)
   53   | PrettyConfigPlcReadable (PrettyConfigReadable PrettyConfigName)
   54   deriving stock (Show)
   55 
   56 -- | Global configuration used for pretty-printing PLC entities.
   57 data PrettyConfigPlc = PrettyConfigPlc
   58   { _pcpOptions :: PrettyConfigPlcOptions
   59   , _pcpStrategy :: PrettyConfigPlcStrategy
   60   }
   61   deriving stock (Show)
   62 
   63 type instance HasPrettyDefaults PrettyConfigPlc = 'True
   64 
   65 -- | The "pretty-printable PLC entity" constraint.
   66 type PrettyPlc = PrettyBy PrettyConfigPlc
   67 
   68 -- | A constraint that allows to derive @PrettyBy PrettyConfigPlc@ instances, see below.
   69 type DefaultPrettyPlcStrategy a =
   70   ( PrettyClassic a
   71   , PrettyReadable a
   72   )
   73 
   74 instance HasPrettyConfigName PrettyConfigPlcStrategy where
   75   toPrettyConfigName (PrettyConfigPlcClassic configClassic) = toPrettyConfigName configClassic
   76   toPrettyConfigName (PrettyConfigPlcReadable configReadable) = toPrettyConfigName configReadable
   77 
   78 instance HasPrettyConfigName PrettyConfigPlc where
   79   toPrettyConfigName = toPrettyConfigName . _pcpStrategy
   80 
   81 instance DefaultPrettyPlcStrategy a => PrettyBy PrettyConfigPlcStrategy (PrettyAny a) where
   82   prettyBy (PrettyConfigPlcClassic configClassic) = prettyBy configClassic . unPrettyAny
   83   prettyBy (PrettyConfigPlcReadable configReadable) = prettyBy configReadable . unPrettyAny
   84 
   85 instance DefaultPrettyPlcStrategy a => PrettyBy PrettyConfigPlc (PrettyAny a) where
   86   prettyBy = prettyBy . _pcpStrategy
   87 
   88 {-| The 'PrettyConfigPlcOptions' used by default:
   89 print errors in full. -}
   90 prettyConfigPlcOptions :: PrettyConfigPlcOptions
   91 prettyConfigPlcOptions = PrettyConfigPlcOptions CondensedErrorsNo
   92 
   93 {-| The 'PrettyConfigPlc' used by default:
   94 use the classic view and print neither 'Unique's, nor name attachments. -}
   95 prettyConfigPlcClassic :: PrettyConfigPlcOptions -> PrettyConfigPlc
   96 prettyConfigPlcClassic opts =
   97   PrettyConfigPlc opts $ PrettyConfigPlcClassic prettyConfigClassic
   98 
   99 {-| The 'PrettyConfigPlc' used for debugging:
  100 use the classic view and print 'Unique's, but not name attachments. -}
  101 prettyConfigPlcClassicSimple :: PrettyConfigPlcOptions -> PrettyConfigPlc
  102 prettyConfigPlcClassicSimple opts =
  103   PrettyConfigPlc opts $ PrettyConfigPlcClassic prettyConfigClassicSimple
  104 
  105 {-| The 'PrettyConfigPlc' used by default and for readability:
  106 use the refined view and print 'Unique's but not name attachments. -}
  107 prettyConfigPlcReadable :: PrettyConfigPlcOptions -> PrettyConfigPlc
  108 prettyConfigPlcReadable opts =
  109   PrettyConfigPlc opts . PrettyConfigPlcReadable $
  110     botPrettyConfigReadable prettyConfigName def
  111 
  112 {-| The 'PrettyConfigPlc' used for debugging and readability:
  113 use the refined view and print neither 'Unique's nor name attachments. -}
  114 prettyConfigPlcReadableSimple :: PrettyConfigPlcOptions -> PrettyConfigPlc
  115 prettyConfigPlcReadableSimple opts =
  116   PrettyConfigPlc opts . PrettyConfigPlcReadable $
  117     botPrettyConfigReadable prettyConfigNameSimple def
  118 
  119 -- | Pretty-print a PLC value in the default mode using the classic view.
  120 prettyPlcClassic :: PrettyPlc a => a -> Doc ann
  121 prettyPlcClassic = prettyBy $ prettyConfigPlcClassic prettyConfigPlcOptions
  122 
  123 -- | Pretty-print a PLC value without unique indices using the classic view.
  124 prettyPlcClassicSimple :: PrettyPlc a => a -> Doc ann
  125 prettyPlcClassicSimple = prettyBy $ prettyConfigPlcClassicSimple prettyConfigPlcOptions
  126 
  127 -- | Pretty-print a PLC value in the default mode using the readable view.
  128 prettyPlcReadable :: PrettyPlc a => a -> Doc ann
  129 prettyPlcReadable = prettyBy $ prettyConfigPlcReadable prettyConfigPlcOptions
  130 
  131 -- | Pretty-print a PLC value without unique indices using the readable view.
  132 prettyPlcReadableSimple :: PrettyPlc a => a -> Doc ann
  133 prettyPlcReadableSimple = prettyBy $ prettyConfigPlcReadableSimple prettyConfigPlcOptions
  134 
  135 {-| Pretty-print a PLC value using the condensed way (see 'CondensedErrors')
  136 of pretty-printing PLC errors (in case there are any). -}
  137 prettyPlcCondensedErrorBy
  138   :: PrettyPlc a => (PrettyConfigPlcOptions -> PrettyConfigPlc) -> a -> Doc ann
  139 prettyPlcCondensedErrorBy toConfig = prettyBy . toConfig $ PrettyConfigPlcOptions CondensedErrorsYes
