   27 {-| @Nat@ as a PLC type.
   28 
   29 > fix \(nat :: *) -> all r. r -> (nat -> r) -> r -}
   30 natData :: RecursiveType uni fun ()
   31 natData = runQuote $ do
   32   nat <- freshTyName "nat"
   33   r <- freshTyName "r"
   34   makeRecursiveType () nat []
   35     . TyForall () r (Type ())
   36     . TyFun () (TyVar () r)
   37     . TyFun () (TyFun () (TyVar () nat) $ TyVar () r)
   38     $ TyVar () r
   39 
   40 natTy :: Type TyName uni ()
   41 natTy = _recursiveType natData

   43 {-|  '0' as a PLC term.
   44 
   45 > wrapNat [] /\(r :: *) -> \(z : r) (f : nat -> r) -> z -}
   46 zero :: TermLike term TyName Name uni fun => term ()
   47 zero = runQuote $ do
   48   let RecursiveType nat wrapNat = natData
   49   r <- freshTyName "r"
   50   z <- freshName "z"
   51   f <- freshName "f"
   52   return
   53     . wrapNat []
   54     . tyAbs () r (Type ())
   55     . lamAbs () z (TyVar () r)
   56     . lamAbs () f (TyFun () nat $ TyVar () r)
   57     $ var () z

   59 {-|  'succ' as a PLC term.
   60 
   61 > \(n : nat) -> wrapNat [] /\(r :: *) -> \(z : r) (f : nat -> r) -> f n -}
   62 succ :: TermLike term TyName Name uni fun => term ()
   63 succ = runQuote $ do
   64   let RecursiveType nat wrapNat = natData
   65   n <- freshName "n"
   66   r <- freshTyName "r"
   67   z <- freshName "z"
   68   f <- freshName "f"
   69   return
   70     . lamAbs () n nat
   71     . wrapNat []
   72     . tyAbs () r (Type ())
   73     . lamAbs () z (TyVar () r)
   74     . lamAbs () f (TyFun () nat $ TyVar () r)
   75     . apply () (var () f)
   76     $ var () n

   78 {-|  @foldrNat@ as a PLC term.
   79 
   80 > /\(r :: *) -> \(f : r -> r) (z : r) ->
   81 >     fix {nat} {r} \(rec : nat -> r) (n : nat) ->
   82 >         unwrap n {r} z \(n' : nat) -> f (rec n') -}
   83 foldrNat :: TermLike term TyName Name uni fun => term ()
   84 foldrNat = runQuote $ do
   85   let nat = _recursiveType natData
   86   r <- freshTyName "r"
   87   f <- freshName "f"
   88   z <- freshName "z"
   89   rec <- freshName "rec"
   90   n <- freshName "n"
   91   n' <- freshName "n'"
   92   return
   93     . tyAbs () r (Type ())
   94     . lamAbs () f (TyFun () (TyVar () r) (TyVar () r))
   95     . lamAbs () z (TyVar () r)
   96     . apply () (mkIterInstNoAnn fix [nat, TyVar () r])
   97     . lamAbs () rec (TyFun () nat $ TyVar () r)
   98     . lamAbs () n nat
   99     . apply () (apply () (tyInst () (unwrap () (var () n)) $ TyVar () r) $ var () z)
  100     . lamAbs () n' nat
  101     . apply () (var () f)
  102     . apply () (var () rec)
  103     $ var () n'

  105 {-|  @foldNat@ as a PLC term.
  106 
  107 > /\(r :: *) -> \(f : r -> r) ->
  108 >     fix {r} {nat -> r} \(rec : r -> nat -> r) (z : r) (n : nat) ->
  109 >         unwrap n {r} z (\(n' : nat) -> rec (f z) n') -}
  110 foldNat :: TermLike term TyName Name uni fun => term ()
  111 foldNat = runQuote $ do
  112   let nat = _recursiveType natData
  113   r <- freshTyName "r"
  114   f <- freshName "f"
  115   rec <- freshName "rec"
  116   z <- freshName "z"
  117   n <- freshName "n"
  118   n' <- freshName "n'"
  119   return
  120     . tyAbs () r (Type ())
  121     . lamAbs () f (TyFun () (TyVar () r) (TyVar () r))
  122     . apply () (mkIterInstNoAnn fix [TyVar () r, TyFun () nat $ TyVar () r])
  123     . lamAbs () rec (TyFun () (TyVar () r) . TyFun () nat $ TyVar () r)
  124     . lamAbs () z (TyVar () r)
  125     . lamAbs () n nat
  126     . apply () (apply () (tyInst () (unwrap () (var () n)) $ TyVar () r) $ var () z)
  127     . lamAbs () n' nat
  128     . mkIterAppNoAnn (var () rec)
  129     $ [ apply () (var () f) $ var () z
  130       , var () n'
  131       ]

  133 {-| Convert a @nat@ to an @integer@.
  134 
  135 > foldNat {integer} (addInteger 1) 1 -}
  136 natToInteger
  137   :: (TermLike term TyName Name uni DefaultFun, uni `HasTypeAndTermLevel` Integer) => term ()
  138 natToInteger = runQuote $ do
  139   return $
  140     mkIterAppNoAnn
  141       (tyInst () foldNat $ mkTyBuiltin @_ @Integer ())
  142       [ apply () (builtin () AddInteger) (mkConstant @Integer () 1)
  143       , mkConstant @Integer () 0
  144       ]
