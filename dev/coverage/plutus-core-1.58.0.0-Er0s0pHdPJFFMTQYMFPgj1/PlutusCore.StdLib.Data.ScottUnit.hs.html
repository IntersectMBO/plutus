   14 {-| '()' as a PLC type.
   15 
   16 > all (A :: *). A -> A -}
   17 unit :: Type TyName uni ()
   18 unit = runQuote $ do
   19   a <- freshTyName "a"
   20   return
   21     . TyForall () a (Type ())
   22     . TyFun () (TyVar () a)
   23     $ TyVar () a
   24 
   25 {-| '()' as a PLC term.
   26 
   27 > /\(A :: *) -> \(x : A) -> x -}
   28 unitval :: TermLike term TyName Name uni fun => term ()
   29 unitval = runQuote $ do
   30   a <- freshTyName "a"
   31   x <- freshName "x"
   32   return
   33     . tyAbs () a (Type ())
   34     . lamAbs () x (TyVar () a)
   35     $ var () x
