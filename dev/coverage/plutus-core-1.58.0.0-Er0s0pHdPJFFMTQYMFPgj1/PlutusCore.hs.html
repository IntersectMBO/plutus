  155 {-| Applies one program to another. Fails if the versions do not match
  156 and tries to merge annotations. -}
  157 applyProgram
  158   :: (MonadError ApplyProgramError m, Semigroup a)
  159   => Program tyname name uni fun a
  160   -> Program tyname name uni fun a
  161   -> m (Program tyname name uni fun a)
  162 applyProgram (Program a1 v1 t1) (Program a2 v2 t2)
  163   | v1 == v2 =
  164       pure $ Program (a1 <> a2) v1 (Apply (termAnn t1 <> termAnn t2) t1 t2)
  165 applyProgram (Program _a1 v1 _t1) (Program _a2 v2 _t2) =
  166   throwError $ MkApplyProgramError v1 v2
