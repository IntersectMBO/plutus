   37 {- Note [Prettyprinting built-in constants]
   38 When we're printing PLC code, the prettyprinter has to render built-in constants. Unfortunately the
   39 instance of "Data.Text.Pretyprint.Doc.Pretty" for 'Text' does the wrong thing if control characters
   40 are involved. For example, the 'Text' "abc\nx\tyz" renders as
   41 
   42 abc
   43 x    yz
   44 
   45 which the PLC parser can't deal with. However, 'show' renders the string as "abc\nx\tyz" (including
   46 the quotes).
   47 
   48 This module provides a 'prettyConst' method which should be used whenever it's necessary to render a
   49 built-in constant: see for example "PlutusCore.Core.Instance.Pretty.Classic". The constraint @uni
   50 `Everywhere` PrettyConst@ occurs in many places in the codebase to make sure that we know how to
   51 print a constant from any type appearing in a universe of built-in types.
   52 
   53 Setting up our own machinery for overloading pretty-printing behavior would be laborious,
   54 but fortunately the @prettyprinter-configurable@ library already provides us with all the tools
   55 for doing that and so we define a dummy config for pretty-printing constants, implement a bunch of
   56 instances and derive pretty-printing behavior for non-polymorphic types (including how lists of
   57 such types are pretty-printed) via 'Show'. However always pretty-printing the spine of, say, a list
   58 via 'Show' while pretty-printing its contents via 'PrettyConst' is not something that can be easily
   59 done with the present-day @prettyprinter-configurable@, so we opt for pretty-printing the spine of
   60 a value of a compound type (list of lists, list of tuples, tuple of lists etc) via 'Pretty'.
   61 In practice this means that we have some additional spaces printed after punctuation symbols
   62 that 'show' alone would have omitted, for example:
   63 
   64 >>> let whateverList = ("abc\nx\tyzâˆ€" :: Text, [((), False), ((), True)])
   65 >>> print $ prettyConst botRenderContext whateverList
   66 ("abc\nx\tyz\8704", [((), False), ((), True)])
   67 >>> putStrLn $ show whateverList
   68 ("abc\nx\tyz\8704",[((),False),((),True)])
   69 
   70 Not a big deal, since our parser isn't whitespace-sensitive.
   71 -}

   73 -- See Note [Prettyprinting built-in constants].
   74 {-| The type of configs used for pretty-printing constants. Has a 'RenderContext' inside, so that
   75 we don't add redundant parens to the output. -}
   76 newtype ConstConfig = ConstConfig
   77   { unConstConfig :: RenderContext
   78   }
   79 
   80 type instance HasPrettyDefaults ConstConfig = 'False
   81 
   82 instance HasRenderContext ConstConfig where
   83   renderContext = coerced
   84 
   85 type PrettyConst = PrettyBy ConstConfig

   87 -- | The set of constraints we need to be able to print built-in types and their values.
   88 type PrettyUni uni = (PrettyParens (SomeTypeIn uni), Closed uni, uni `Everywhere` PrettyConst)
   89 
   90 {-| The set of constraints we need to be able to throw exceptions with things with built-in types
   91 and functions in them. -}
   92 type ThrowableBuiltins uni fun = (PrettyUni uni, Pretty fun, Typeable uni, Typeable fun)
   93 
   94 -- These two can be generalized to any @config@, but that breaks some use cases of 'PrettyAny'
   95 -- then. Perhaps we should split the functionality and have two separate @newtype@ wrappers
   96 -- in @prettyprinter-configurable@ instead of a single 'PrettyAny'.
   97 -- For that we'll also need to ensure that it's alright when @HasPrettyDefaults config ~ 'True@.
   98 instance DefaultPrettyBy ConstConfig (PrettyAny a) => NonDefaultPrettyBy ConstConfig (PrettyAny a)
   99 instance DefaultPrettyBy ConstConfig (PrettyAny a) => PrettyBy ConstConfig (PrettyAny a) where
  100   prettyBy = defaultPrettyBy
  101   prettyListBy = defaultPrettyListBy
  102 
  103 instance Show a => DefaultPrettyBy ConstConfig (PrettyAny a) where
  104   defaultPrettyBy _ = pretty . show @a . coerce
  105   defaultPrettyListBy _ = pretty . show @[a] . coerce
  106 
  107 prettyConst :: PrettyConst a => RenderContext -> a -> Doc ann
  108 prettyConst = prettyBy . ConstConfig
  109 
  110 -- This instance for Text quotes control characters (which is what we want)
  111 -- but doesn't escape Unicode characters (\8704 and so on).
  112 instance NonDefaultPrettyBy ConstConfig T.Text where
  113   nonDefaultPrettyListBy conf = Prettyprinter.list . Prelude.map (nonDefaultPrettyBy conf)
  114   nonDefaultPrettyBy = inContextM $ \t -> pure $ pretty $ "\"" <> escape t <> "\""
  115     where
  116       escape = T.foldr' prettyChar ""
  117       prettyChar c acc
  118         | c == '"' = "\\\"" <> acc -- Not handled by 'showLitChar'
  119         | c == '\\' = "\\\\" <> acc -- Not handled by 'showLitChar'
  120         | Char.isPrint c = [c] <> acc
  121         | otherwise = Char.showLitChar c acc
  122 
  123 deriving via PrettyAny () instance NonDefaultPrettyBy ConstConfig ()
  124 deriving via PrettyAny Bool instance NonDefaultPrettyBy ConstConfig Bool
  125 deriving via PrettyAny Integer instance NonDefaultPrettyBy ConstConfig Integer

  127 -- | For rendering values without parens, i.e. in 'botRenderContext'.
  128 newtype NoParens a = NoParens
  129   { unNoParens :: a
  130   }
  131 
  132 instance PrettyConst a => PrettyBy ConstConfig (NoParens a) where
  133   prettyBy config = prettyBy @_ @a (config & renderContext .~ botRenderContext) . coerce
  134   prettyListBy config = prettyListBy @_ @a (config & renderContext .~ botRenderContext) . coerce
  135 
  136 instance PrettyConst a => NonDefaultPrettyBy ConstConfig [a] where
  137   nonDefaultPrettyBy config = defaultPrettyBy @_ @[NoParens a] config . coerce
  138 instance PrettyConst a => NonDefaultPrettyBy ConstConfig (Vector a) where
  139   nonDefaultPrettyBy config = defaultPrettyBy @_ @(Vector (NoParens a)) config . coerce
  140 instance (PrettyConst a, PrettyConst b) => NonDefaultPrettyBy ConstConfig (a, b) where
  141   nonDefaultPrettyBy config = defaultPrettyBy @_ @(NoParens a, NoParens b) config . coerce
  142 
  143 -- Special instance for bytestrings
  144 asBytes :: Word8 -> Doc ann
  145 asBytes x = Text 2 $ T.pack $ addLeadingZero $ showHex x mempty
  146   where
  147     addLeadingZero :: String -> String
  148     addLeadingZero
  149       | x < 16 = ('0' :)
  150       | otherwise = id
  151 
  152 toBytes :: BS.ByteString -> Doc ann
  153 toBytes = foldMap asBytes . BS.unpack
  154 
  155 instance PrettyBy ConstConfig Data where
  156   prettyBy = inContextM $ \d0 -> iterAppDocM $ \_ prettyArg -> case d0 of
  157     Constr i ds -> ("Constr" <+> prettyArg i) :| [prettyArg ds]
  158     Map ps -> "Map" :| [prettyArg ps]
  159     List ds -> "List" :| [prettyArg ds]
  160     I i -> ("I" <+> prettyArg i) :| []
  161     B b -> ("B" <+> prettyArg b) :| []
  162 
  163 instance PrettyBy ConstConfig Value.K where
  164   prettyBy config = prettyBy config . Value.unK
  165 
  166 instance PrettyBy ConstConfig Value.Quantity where
  167   prettyBy config = prettyBy config . Value.unQuantity
  168 
  169 instance PrettyBy ConstConfig Value where
  170   prettyBy config = prettyBy config . Value.toList
  171 
  172 instance PrettyBy ConstConfig BS.ByteString where
  173   prettyBy _ b = "#" <> toBytes b
  174 
  175 instance Pretty (SomeTypeIn uni) => Pretty (SomeTypeIn (Kinded uni)) where
  176   pretty (SomeTypeIn (Kinded uni)) = pretty (SomeTypeIn uni)

  178 -- See Note [Prettyprinting built-in constants].
  179 instance (Closed uni, uni `Everywhere` PrettyConst) => PrettyBy ConstConfig (ValueOf uni a) where
  180   prettyBy config (ValueOf uni x) = bring (Proxy @PrettyConst) uni $ prettyBy config x
  181 
  182 -- See Note [Prettyprinting built-in constants].
  183 instance
  184   (Closed uni, uni `Everywhere` PrettyConst)
  185   => PrettyBy ConstConfig (Some (ValueOf uni))
  186   where
  187   prettyBy config (Some s) = prettyBy config s
