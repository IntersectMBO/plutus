   20 type MonadRename m = (MonadQuote m, MonadReader (Renaming TermUnique) m)
   21 
   22 -- | Rename a 'Term' in the 'RenameM' monad.
   23 renameTermM
   24   :: (MonadRename m, HasUniques (Term name uni fun ann))
   25   => Term name uni fun ann -> m (Term name uni fun ann)
   26 renameTermM (LamAbs ann name body) =
   27   withFreshenedName name $ \nameFr -> LamAbs ann nameFr <$> renameTermM body
   28 renameTermM (Apply ann fun arg) = Apply ann <$> renameTermM fun <*> renameTermM arg
   29 renameTermM err@Error {} = pure err
   30 renameTermM (Var ann name) = Var ann <$> renameNameM name
   31 renameTermM (Delay ann term) = Delay ann <$> renameTermM term
   32 renameTermM (Force ann term) = Force ann <$> renameTermM term
   33 renameTermM (Constr ann i es) = Constr ann i <$> traverse renameTermM es
   34 renameTermM (Case ann arg cs) = Case ann <$> renameTermM arg <*> traverse renameTermM cs
   35 renameTermM con@Constant {} = pure con
   36 renameTermM bi@Builtin {} = pure bi
   37 
   38 -- | Rename a 'Program' in the 'RenameM' monad.
   39 renameProgramM
   40   :: (MonadRename m, HasUniques (Program name uni fun ann))
   41   => Program name uni fun ann -> m (Program name uni fun ann)
   42 renameProgramM (Program ann ver term) = Program ann ver <$> renameTermM term
