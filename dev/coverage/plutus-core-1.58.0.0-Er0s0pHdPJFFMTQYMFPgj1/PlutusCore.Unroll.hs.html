   13 {-| Static loop unrolling utilities using type-level Peano numbers.
   14 
   15 This module provides utilities for statically unrolling loops at compile time.
   16 GHC resolves the type class instances at compile time, resulting in unrolled
   17 code that avoids runtime overhead.
   18 
   19 See Note [Static loop unrolling] for details on the technique. -}
   20 module PlutusCore.Unroll
   21   ( -- * Peano numbers
   22     Peano (..)
   23   , NatToPeano
   24 
   25     -- * Statically unrolled list drop
   26   , Drop (..)
   27   , dropN
   28 
   29     -- * Statically unrolled effectful iteration
   30   , UpwardsM (..)
   31   ) where
   32 
   33 import Data.Kind (Constraint, Type)
   34 import GHC.TypeNats (Nat, type (-))
   35 
   36 {- Note [Static loop unrolling]
   37 To avoid runtime loop overhead, we use type-level Peano numbers to guide instance
   38 resolution. GHC resolves the instances at compile time and unrolls the loop into
   39 straight-line code.
   40 
   41 For example, 'dropN @3' becomes pattern matching equivalent to:
   42   case xs of
   43     (_:_:_:rest) -> Just rest
   44     _ -> Nothing
   45 
   46 And 'upwardsM @_ @(NatToPeano 3) 0 k' becomes:
   47   k 0 *> k 1 *> k 2
   48 
   49 Benefits:
   50 1. No runtime recursion overhead
   51 2. No allocations for intermediate structures
   52 3. GHC can optimize the unrolled code very well
   53 4. Instance resolution happens entirely at compile time
   54 
   55 The technique was originally used in the CEK machine step counter:
   56 https://github.com/IntersectMBO/plutus/pull/5265
   57 -}
   58 
   59 --------------------------------------------------------------------------------
   60 -- Peano numbers ---------------------------------------------------------------
   61 
   62 -- | Peano representation of natural numbers for type-level recursion.
   63 data Peano = Z | S Peano
   64 
   65 -- | Convert a type-level 'Nat' to 'Peano' form for instance resolution.
   66 type family NatToPeano (n :: Nat) :: Peano where
   67   NatToPeano 0 = 'Z
   68   NatToPeano n = 'S (NatToPeano (n - 1))
   69 
   70 --------------------------------------------------------------------------------
   71 -- Statically unrolled list drop -----------------------------------------------
   72 
   73 {-| Statically unrolled drop: drops exactly @n@ elements if possible.
   74 Returns 'Nothing' if the list has fewer than @n@ elements.
   75 
   76 The @n@ parameter is a 'Peano' number that guides instance resolution.
   77 Use 'dropN' for a more convenient interface with 'Nat' literals. -}
   78 class Drop (n :: Peano) where
   79   drop' :: [a] -> Maybe [a]
   80 
   81 instance Drop 'Z where
   82   drop' = Just
   83   {-# INLINE drop' #-}
   84 
   85 instance Drop n => Drop ('S n) where
   86   drop' [] = Nothing
   87   drop' (_ : xs) = drop' @n xs
   88   {-# INLINE drop' #-}
   89 
   90 {-| Drop exactly @n@ elements from a list, returning 'Nothing' if too short.
   91 
   92 The drop count is statically unrolled at compile time via instance resolution.
   93 
   94 @
   95 dropN \@3 [1,2,3,4,5] == Just [4,5]
   96 dropN \@3 [1,2]       == Nothing
   97 @ -}
   98 dropN :: forall n a. Drop (NatToPeano n) => [a] -> Maybe [a]
   99 dropN = drop' @(NatToPeano n)
  100 {-# INLINE dropN #-}
  101 
  102 --------------------------------------------------------------------------------
  103 -- Statically unrolled effectful iteration -------------------------------------
  104 
  105 {-| Statically unrolled effectful iteration over a range.
  106 
  107 @upwardsM i k@ means @k i *> k (i + 1) *> ... *> k (i + n - 1)@.
  108 
  109 We use this to statically unroll loops through instance resolution.
  110 This makes validation benchmarks a couple of percent faster. -}
  111 type UpwardsM :: (Type -> Type) -> Peano -> Constraint
  112 class Applicative f => UpwardsM f n where
  113   upwardsM :: Int -> (Int -> f ()) -> f ()
  114 
  115 instance Applicative f => UpwardsM f 'Z where
  116   upwardsM _ _ = pure ()
  117   {-# INLINE upwardsM #-}
  118 
  119 instance UpwardsM f n => UpwardsM f ('S n) where
  120   upwardsM !i k = k i *> upwardsM @f @n (i + 1) k
  121   {-# INLINE upwardsM #-}
