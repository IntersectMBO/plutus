   86 {- Note [PLC types and universes]
   87 We encode built-in types in PLC as tags for Haskell types (the latter are also called meta-types),
   88 see Note [Universes]. A built-in type in PLC is an inhabitant of
   89 
   90     Some (TypeIn uni)
   91 
   92 where @uni@ is some universe, i.e. a collection of tags that have meta-types associated with them.
   93 
   94 A value of a built-in type is a regular Haskell value stored in
   95 
   96     Some (ValueOf uni)
   97 
   98 (together with the tag associated with its type) and such a value is also called a meta-constant.
   99 
  100 The default universe has the following constructor (pattern synonym actually):
  101 
  102     DefaultUniList :: !(DefaultUni a) -> DefaultUni [a]
  103 
  104 But note that this doesn't directly lead to interop between Plutus Core and Haskell, i.e. you can't
  105 have a meta-list whose elements are of a PLC type. You can only have a meta-list constant with
  106 elements of a meta-type (i.e. a type from the universe).
  107 
  108 However it is possible to apply a built-in type to an arbitrary PLC/PIR type, since we can embed
  109 a type of an arbitrary kind into 'Type' and then apply it via 'TyApp'. But we only use it to
  110 get polymorphic built-in functions over polymorphic built-in types, since it's not possible
  111 to juggle values of polymorphic built-in types instantiated with non-built-in types at runtime
  112 (it's not even possible to represent such a value in the AST, even though it's possible to represent
  113 such a 'Type').
  114 
  115 Finally, it is not necessary to allow embedding PLC terms into meta-constants.
  116 We already allow built-in functions with polymorphic types. There might be a way to utilize this
  117 feature and have meta-constructors as built-in functions.
  118 -}

  198 instance GEq DefaultUni where
  199   -- We define 'geq' manually instead of using 'deriveGEq', because the latter creates a single
  200   -- recursive definition and we want two instead. The reason why we want two is because this
  201   -- allows GHC to inline the initial step that appears non-recursive to GHC, because recursion
  202   -- is hidden in the other function that is marked as @OPAQUE@ and is chosen by GHC as a
  203   -- loop-breaker, see https://wiki.haskell.org/Inlining_and_Specialisation#What_is_a_loop-breaker
  204   -- (we're not really sure if this is a reliable solution, but if it stops working, we won't miss
  205   -- very much and we've failed to settle on any other approach).
  206   --
  207   -- On the critical path this definition should only be used for builtins that perform equality
  208   -- checking of statically unknown runtime type tags ('MkCons' is one such builtin for
  209   -- example). All other builtins should use 'geqL' (the latter is internal to 'readKnownConstant'
  210   -- and is therefore hidden from the person adding a new builtin).
  211   --
  212   -- We use @NOINLINE@ instead of @OPAQUE@, because we don't actually care about the recursive
  213   -- definition not being inlined, we just want it to be chosen as the loop breaker.
  214   geq = goStep
  215     where
  216       goStep, goRec :: DefaultUni a1 -> DefaultUni a2 -> Maybe (a1 :~: a2)
  217       goStep DefaultUniInteger a2 = do
  218         DefaultUniInteger <- pure a2
  219         pure Refl
  220       goStep DefaultUniByteString a2 = do
  221         DefaultUniByteString <- pure a2
  222         pure Refl
  223       goStep DefaultUniString a2 = do
  224         DefaultUniString <- pure a2
  225         pure Refl
  226       goStep DefaultUniUnit a2 = do
  227         DefaultUniUnit <- pure a2
  228         pure Refl
  229       goStep DefaultUniBool a2 = do
  230         DefaultUniBool <- pure a2
  231         pure Refl
  232       goStep DefaultUniProtoList a2 = do
  233         DefaultUniProtoList <- pure a2
  234         pure Refl
  235       goStep DefaultUniProtoArray a2 = do
  236         DefaultUniProtoArray <- pure a2
  237         pure Refl
  238       goStep DefaultUniProtoPair a2 = do
  239         DefaultUniProtoPair <- pure a2
  240         pure Refl
  241       goStep (DefaultUniApply f1 x1) a2 = do
  242         DefaultUniApply f2 x2 <- pure a2
  243         Refl <- goRec f1 f2
  244         Refl <- goRec x1 x2
  245         pure Refl
  246       goStep DefaultUniData a2 = do
  247         DefaultUniData <- pure a2
  248         pure Refl
  249       goStep DefaultUniBLS12_381_G1_Element a2 = do
  250         DefaultUniBLS12_381_G1_Element <- pure a2
  251         pure Refl
  252       goStep DefaultUniBLS12_381_G2_Element a2 = do
  253         DefaultUniBLS12_381_G2_Element <- pure a2
  254         pure Refl
  255       goStep DefaultUniBLS12_381_MlResult a2 = do
  256         DefaultUniBLS12_381_MlResult <- pure a2
  257         pure Refl
  258       goStep DefaultUniValue a2 = do
  259         DefaultUniValue <- pure a2
  260         pure Refl
  261       {-# INLINE goStep #-}
  262 
  263       goRec = goStep
  264       {-# NOINLINE goRec #-}

  484 {- Note [Integral types as Integer]
  485 Technically our universe only contains 'Integer', but many of the builtin functions that we would
  486 like to use work over 'Int', 'Word8' etc.
  487 
  488 This is inconvenient and also error-prone: dealing with a function that takes an 'Int' or 'Word8'
  489 means carefully downcasting the 'Integer', running the function, potentially upcasting at the end.
  490 And it's easy to get wrong by e.g. blindly using 'fromInteger' or 'fromIntegral'.
  491 
  492 Moreover, there is a latent risk here: if we *were* to build on a 32-bit architecture, then programs
  493 which use arguments between @maxBound :: Int32@ and @maxBound :: Int64@ would behave differently!
  494 
  495 So, what to do? We adopt the following strategy:
  496 - We allow lifting/unlifting bounded integral types such as 'Word8' or 'Int64' via 'Integer',
  497   including a safe upcast in 'makeKnown' (which we could have on any architecture, but we only add
  498   it for 64-bit for uniformity) and a safe checked downcast in 'readKnown'.
  499 - We allow lifting/unlifting 'Int' via 'Int64' and 'Word' via 'Word64', constraining the conversion
  500   between them to 64-bit architectures where this conversion is safe.
  501 
  502 Doing this effectively bans builds on all non-64-bit systems, which is acceptable since 64-bit
  503 systems are required for node deployment, and this way any attempts to build on them will fail fast.
  504 
  505 We used to use newtype- and via-deriving for implementations of relevant instances, but at some
  506 point GHC stopped attaching @INLINE@ annotations to those causing the GHC Core for builtins to have
  507 multiple 'readKnown' and 'makeKnown' calls, so now we don't rely on deriving and implement instances
  508 manually.
  509 -}

  783 outOfBoundsErr :: Pretty a => a -> Vector.Vector term -> Text
  784 outOfBoundsErr x branches =
  785   fold
  786     [ "'case "
  787     , display x
  788     , "' is out of bounds for the given number of branches: "
  789     , display $ Vector.length branches
  790     ]

  792 instance AnnotateCaseBuiltin DefaultUni where
  793   annotateCaseBuiltin ty branches = case ty of
  794     TyBuiltin _ (SomeTypeIn DefaultUniUnit) ->
  795       case branches of
  796         [x] -> Right [(x, [])]
  797         _ -> Left "Casing on unit only allows exactly one branch"
  798     TyBuiltin _ (SomeTypeIn DefaultUniBool) ->
  799       case branches of
  800         [f] -> Right [(f, [])]
  801         [f, t] -> Right [(f, []), (t, [])]
  802         _ -> Left "Casing on bool requires exactly one branch or two branches"
  803     TyBuiltin _ (SomeTypeIn DefaultUniInteger) ->
  804       Right $ map (,[]) branches
  805     listTy@(TyApp _ (TyBuiltin _ (SomeTypeIn DefaultUniProtoList)) argTy) ->
  806       case branches of
  807         [cons] -> Right [(cons, [argTy, listTy])]
  808         [cons, nil] -> Right [(cons, [argTy, listTy]), (nil, [])]
  809         _ -> Left "Casing on list requires exactly one branch or two branches"
  810     (TyApp _ (TyApp _ (TyBuiltin _ (SomeTypeIn DefaultUniProtoPair)) lTyArg) rTyArg) ->
  811       case branches of
  812         [f] -> Right [(f, [lTyArg, rTyArg])]
  813         _ -> Left "Casing on pair requires exactly one branch"
  814     _ -> Left $ display (void ty) <> " isn't supported in 'case'"

  816 instance CaseBuiltin DefaultUni where
  817   caseBuiltin someVal@(Some (ValueOf uni x)) branches = case uni of
  818     DefaultUniUnit
  819       | 1 == len -> HeadOnly $ branches Vector.! 0
  820       | otherwise -> HeadError $ outOfBoundsErr someVal branches
  821     DefaultUniBool -> case x of
  822       -- We allow there to be only one branch as long as the scrutinee is 'False'.
  823       -- This is strictly to save size by not having the 'True' branch if it was gonna be
  824       -- 'Error' anyway.
  825       False | len == 1 || len == 2 -> HeadOnly $ branches Vector.! 0
  826       True | len == 2 -> HeadOnly $ branches Vector.! 1
  827       _ -> HeadError $ outOfBoundsErr someVal branches
  828     DefaultUniInteger
  829       | 0 <= x && x < toInteger len -> HeadOnly $ branches Vector.! fromInteger x
  830       | otherwise -> HeadError $ outOfBoundsErr someVal branches
  831     DefaultUniList ty
  832       | len == 1 ->
  833           case x of
  834             [] -> HeadError "Expected non-empty list, got empty list for casing list"
  835             (y : ys) -> headSpine (branches Vector.! 0) [someValueOf ty y, someValueOf uni ys]
  836       | len == 2 ->
  837           case x of
  838             [] -> HeadOnly $ branches Vector.! 1
  839             (y : ys) -> headSpine (branches Vector.! 0) [someValueOf ty y, someValueOf uni ys]
  840       | otherwise -> HeadError $ outOfBoundsErr someVal branches
  841     DefaultUniPair tyL tyR
  842       | len == 1 ->
  843           case x of
  844             (l, r) -> headSpine (branches Vector.! 0) [someValueOf tyL l, someValueOf tyR r]
  845       | otherwise -> HeadError $ outOfBoundsErr someVal branches
  846     _ -> HeadError $ display uni <> " isn't supported in 'case'"
  847     where
  848       !len = Vector.length branches
  849   {-# INLINE caseBuiltin #-}
