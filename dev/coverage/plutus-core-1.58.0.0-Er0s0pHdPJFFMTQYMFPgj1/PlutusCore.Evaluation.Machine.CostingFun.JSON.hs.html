   19 type ModelJSON prefix = CustomJSON '[FieldLabelModifier (StripPrefix prefix, CamelToSnake)]
   20 
   21 type ModelArgumentJSON prefix =
   22   CustomJSON
   23     -- Without TagSingleConstructors the format can change unexpectedly if
   24     -- you add/remove constructors because you don't get the tags if there's
   25     -- only one constructor but you do if there's more than one.
   26     '[ TagSingleConstructors
   27      , SumTaggedObject "type" "arguments"
   28      , ConstructorTagModifier (StripPrefix prefix, CamelToSnake)
   29      ]
