   10 import Cardano.Crypto.DSIGN.Class qualified as DSIGN
   11 import Cardano.Crypto.DSIGN.Ed25519 (Ed25519DSIGN)
   12 import Data.ByteString qualified as BS
   13 import Data.Text (Text)

   15 {-| Ed25519 signature verification
   16 This will fail if the key or the signature are not of the expected length.
   17 This version uses the cardano-crypto-class implementation of the verification
   18 function (using libsodium). -}
   19 verifyEd25519Signature
   20   :: BS.ByteString
   21   -- ^ Public Key (32 bytes)
   22   -> BS.ByteString
   23   -- ^ Message    (arbitrary length)
   24   -> BS.ByteString
   25   -- ^ Signature  (64 bytes)
   26   -> BuiltinResult Bool
   27 verifyEd25519Signature pk msg sig =
   28   case DSIGN.rawDeserialiseVerKeyDSIGN @Ed25519DSIGN pk of
   29     Nothing -> failWithMessage loc "Invalid verification key."
   30     Just pk' -> case DSIGN.rawDeserialiseSigDSIGN @Ed25519DSIGN sig of
   31       Nothing -> failWithMessage loc "Invalid signature."
   32       Just sig' ->
   33         pure $
   34           case DSIGN.verifyDSIGN () pk' msg sig' of
   35             Left _ -> False
   36             Right () -> True
   37   where
   38     loc :: Text
   39     loc = "Ed25519 signature verification"
