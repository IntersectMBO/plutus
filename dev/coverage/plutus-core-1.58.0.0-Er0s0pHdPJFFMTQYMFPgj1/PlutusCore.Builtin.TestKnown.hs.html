    9 {-| A module providing a tool for making sure that each type from a particular universe implements
   10 'KnownTypeAst', 'ReadKnown' and 'MakeKnown'. Kept separate from the modules defining those
   11 classes in order not to introduce dependencies between them and because this extra safety is
   12 scary looking and it's best to keep it separate from what is necessary. -}
   13 module PlutusCore.Builtin.TestKnown
   14   ( TestTypesFromTheUniverseAreAllKnown
   15   ) where
   16 
   17 import PlutusCore.Builtin.KnownType
   18 import PlutusCore.Builtin.KnownTypeAst
   19 
   20 import Universe
   21 
   22 {-| For providing a 'KnownTypeAst' instance for a built-in type it's enough for that type to
   23 satisfy 'KnownBuiltinTypeAst'. -}
   24 class
   25   (forall tyname. KnownBuiltinTypeAst tyname uni a => KnownTypeAst tyname uni a) =>
   26   ImplementedKnownTypeAst uni a
   27 
   28 instance
   29   (forall tyname. KnownBuiltinTypeAst tyname uni a => KnownTypeAst tyname uni a)
   30   => ImplementedKnownTypeAst uni a
   31 
   32 {-| For providing a 'ReadKnownIn' instance for a built-in type it's enough for that type to
   33 satisfy 'KnownBuiltinTypeIn'. -}
   34 class
   35   (forall val. KnownBuiltinTypeIn uni val a => ReadKnownIn uni val a) =>
   36   ImplementedReadKnownIn uni a
   37 
   38 instance
   39   (forall val. KnownBuiltinTypeIn uni val a => ReadKnownIn uni val a)
   40   => ImplementedReadKnownIn uni a
   41 
   42 {-| For providing a 'MakeKnownIn' instance for a built-in type it's enough for that type to
   43 satisfy 'KnownBuiltinTypeIn'. -}
   44 class
   45   (forall val. KnownBuiltinTypeIn uni val a => MakeKnownIn uni val a) =>
   46   ImplementedMakeKnownIn uni a
   47 
   48 instance
   49   (forall val. KnownBuiltinTypeIn uni val a => MakeKnownIn uni val a)
   50   => ImplementedMakeKnownIn uni a
   51 
   52 {-| An instance of this class not having any constraints ensures that every type (according to
   53 'Everywhere') from the universe has 'KnownTypeAst, 'ReadKnownIn' and 'MakeKnownIn' instances. -}
   54 class
   55   ( uni `Everywhere` ImplementedKnownTypeAst uni
   56   , uni `Everywhere` ImplementedReadKnownIn uni
   57   , uni `Everywhere` ImplementedMakeKnownIn uni
   58   ) =>
   59   TestTypesFromTheUniverseAreAllKnown uni
