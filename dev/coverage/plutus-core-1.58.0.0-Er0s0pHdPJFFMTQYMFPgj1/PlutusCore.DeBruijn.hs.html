   50 {- Note [DeBruijn indices of Binders]
   51 In a debruijnijfied Term AST, the Binders have a debruijn index
   52 at their the specific AST location.
   53 During *undebruijnification* we ignore such binders' indices because they are meaningless,
   54 and instead use the convention that such introduced binders have
   55 a fixed debruijn index '0' at their introduction.
   56 -}

  116 {- Note [De Bruijn conversion and recursion schemes]
  117 These are quite repetitive, but we can't use a catamorphism for this, because
  118 we are not only altering the recursive type, but also the name type parameters.
  119 These are normally constant in a catamorphic application.
  120 -}
  121 
  122 deBruijnTyWithM
  123   :: forall m uni ann
  124    . MonadReader LevelInfo m
  125   => (Unique -> m Index)
  126   -> Type TyName uni ann
  127   -> m (Type NamedTyDeBruijn uni ann)
  128 deBruijnTyWithM h = go
  129   where
  130     go :: Type TyName uni ann -> m (Type NamedTyDeBruijn uni ann)
  131     go = \case
  132       -- variable case
  133       TyVar ann n -> TyVar ann <$> tyNameToDeBruijn h n
  134       -- binder cases
  135       TyForall ann tn k ty -> declareUnique tn $ do
  136         tn' <- tyNameToDeBruijn h tn
  137         withScope $ TyForall ann tn' k <$> go ty
  138       TyLam ann tn k ty -> declareUnique tn $ do
  139         tn' <- tyNameToDeBruijn h tn
  140         withScope $ TyLam ann tn' k <$> go ty
  141       -- boring recursive cases
  142       TyFun ann i o -> TyFun ann <$> go i <*> go o
  143       TyApp ann fun arg -> TyApp ann <$> go fun <*> go arg
  144       TyIFix ann pat arg -> TyIFix ann <$> go pat <*> go arg
  145       TySOP ann tyls -> TySOP ann <$> (traverse . traverse) go tyls
  146       -- boring non-recursive cases
  147       TyBuiltin ann someUni -> pure $ TyBuiltin ann someUni
  148 
  149 deBruijnTermWithM
  150   :: forall m uni fun ann
  151    . MonadReader LevelInfo m
  152   => (Unique -> m Index)
  153   -> Term TyName Name uni fun ann
  154   -> m (Term NamedTyDeBruijn NamedDeBruijn uni fun ann)
  155 deBruijnTermWithM h = go
  156   where
  157     goT :: Type TyName uni ann -> m (Type NamedTyDeBruijn uni ann)
  158     goT = deBruijnTyWithM h
  159 
  160     go :: Term TyName Name uni fun ann -> m (Term NamedTyDeBruijn NamedDeBruijn uni fun ann)
  161     go = \case
  162       -- variable case
  163       Var ann n -> Var ann <$> nameToDeBruijn h n
  164       -- binder cases
  165       TyAbs ann tn k t -> declareUnique tn $ do
  166         tn' <- tyNameToDeBruijn h tn
  167         withScope $ TyAbs ann tn' k <$> go t
  168       LamAbs ann n ty t -> declareUnique n $ do
  169         n' <- nameToDeBruijn h n
  170         withScope $ LamAbs ann n' <$> goT ty <*> go t
  171       -- boring recursive cases
  172       Apply ann t1 t2 -> Apply ann <$> go t1 <*> go t2
  173       TyInst ann t ty -> TyInst ann <$> go t <*> goT ty
  174       Unwrap ann t -> Unwrap ann <$> go t
  175       IWrap ann pat arg t -> IWrap ann <$> goT pat <*> goT arg <*> go t
  176       Error ann ty -> Error ann <$> goT ty
  177       Constr ann ty i es -> Constr ann <$> goT ty <*> pure i <*> traverse go es
  178       Case ann ty arg cs -> Case ann <$> goT ty <*> go arg <*> traverse go cs
  179       -- boring non-recursive cases
  180       Constant ann con -> pure $ Constant ann con
  181       Builtin ann bn -> pure $ Builtin ann bn
  182 
  183 -- | Takes a "handler" function to execute when encountering free variables.
  184 unDeBruijnTyWithM
  185   :: forall m uni ann
  186    . (MonadReader LevelInfo m, MonadQuote m)
  187   => (Index -> m Unique)
  188   -> Type NamedTyDeBruijn uni ann
  189   -> m (Type TyName uni ann)
  190 unDeBruijnTyWithM h = go
  191   where
  192     go :: Type NamedTyDeBruijn uni ann -> m (Type TyName uni ann)
  193     go = \case
  194       -- variable case
  195       TyVar ann n -> TyVar ann <$> deBruijnToTyName h n
  196       -- binder cases
  197       TyForall ann tn k ty ->
  198         -- See Note [DeBruijn indices of Binders]
  199         declareBinder $ do
  200           tn' <- deBruijnToTyName h $ set index deBruijnInitIndex tn
  201           withScope $ TyForall ann tn' k <$> go ty
  202       TyLam ann tn k ty ->
  203         -- See Note [DeBruijn indices of Binders]
  204         declareBinder $ do
  205           tn' <- deBruijnToTyName h $ set index deBruijnInitIndex tn
  206           withScope $ TyLam ann tn' k <$> go ty
  207       -- boring recursive cases
  208       TyFun ann i o -> TyFun ann <$> go i <*> go o
  209       TyApp ann fun arg -> TyApp ann <$> go fun <*> go arg
  210       TyIFix ann pat arg -> TyIFix ann <$> go pat <*> go arg
  211       TySOP ann tyls -> TySOP ann <$> (traverse . traverse) go tyls
  212       -- boring non-recursive cases
  213       TyBuiltin ann someUni -> pure $ TyBuiltin ann someUni

  243       Apply ann t1 t2 -> Apply ann <$> go t1 <*> go t2
  244       TyInst ann t ty -> TyInst ann <$> go t <*> goT ty
  245       Unwrap ann t -> Unwrap ann <$> go t

  247       Error ann ty -> Error ann <$> goT ty
