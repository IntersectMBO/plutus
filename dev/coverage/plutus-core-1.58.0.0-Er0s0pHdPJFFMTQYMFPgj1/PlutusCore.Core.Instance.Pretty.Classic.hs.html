   10 -- | A "classic" (i.e. as seen in the specification) way to pretty-print PLC entities.

   23 instance Pretty ann => PrettyBy (PrettyConfigClassic configName) (Kind ann) where
   24   prettyBy config = \case
   25     Type ann ->
   26       parens
   27         ( sep
   28             ( consAnnIf
   29                 config
   30                 ann
   31                 ["type"]
   32             )
   33         )
   34     KindArrow ann k k' ->
   35       sexp
   36         "fun"
   37         ( consAnnIf
   38             config
   39             ann
   40             [prettyBy config k, prettyBy config k']
   41         )
   42 
   43 instance
   44   (PrettyClassicBy configName tyname, PrettyParens (SomeTypeIn uni), Pretty ann)
   45   => PrettyBy (PrettyConfigClassic configName) (Type tyname uni ann)
   46   where
   47   prettyBy config = \case
   48     TyApp ann t t' ->
   49       brackets'
   50         ( sep
   51             ( consAnnIf
   52                 config
   53                 ann
   54                 [prettyBy config t, prettyBy config t']
   55             )
   56         )
   57     TyVar ann n ->
   58       sep
   59         ( consAnnIf
   60             config
   61             ann
   62             [prettyBy config n]
   63         )
   64     TyFun ann t t' ->
   65       sexp
   66         "fun"
   67         ( consAnnIf
   68             config
   69             ann
   70             [prettyBy config t, prettyBy config t']
   71         )
   72     TyIFix ann pat arg ->
   73       sexp
   74         "ifix"
   75         ( consAnnIf
   76             config
   77             ann
   78             [prettyBy config pat, prettyBy config arg]
   79         )
   80     TyForall ann n k t ->
   81       sexp
   82         "all"
   83         ( consAnnIf
   84             config
   85             ann
   86             [prettyBy config n, prettyBy config k, prettyBy config t]
   87         )
   88     TyBuiltin ann someUni ->
   89       sexp "con" (consAnnIf config ann [prettyBy juxtRenderContext someUni])
   90     TyLam ann n k t ->
   91       sexp
   92         "lam"
   93         ( consAnnIf
   94             config
   95             ann
   96             [prettyBy config n, prettyBy config k, prettyBy config t]
   97         )
   98     TySOP ann tyls ->
   99       sexp "sop" (consAnnIf config ann (fmap prettyTyl tyls))
  100       where
  101         prettyTyl tyl = brackets (sep (fmap (prettyBy config) tyl))
  102 
  103 instance
  104   ( PrettyClassicBy configName tyname
  105   , PrettyClassicBy configName name
  106   , PrettyUni uni
  107   , Pretty fun
  108   , Pretty ann
  109   )
  110   => PrettyBy (PrettyConfigClassic configName) (Term tyname name uni fun ann)
  111   where
  112   prettyBy config = \case
  113     Var ann n ->
  114       sep (consAnnIf config ann [prettyBy config n])
  115     TyAbs ann tn k t ->
  116       sexp
  117         "abs"
  118         ( consAnnIf
  119             config
  120             ann
  121             [prettyBy config tn, prettyBy config k, prettyBy config t]
  122         )
  123     LamAbs ann n ty t ->
  124       sexp
  125         "lam"
  126         ( consAnnIf
  127             config
  128             ann
  129             [prettyBy config n, prettyBy config ty, prettyBy config t]
  130         )
  131     Apply ann t1 t2 ->
  132       brackets'
  133         ( sep
  134             ( consAnnIf
  135                 config
  136                 ann
  137                 [prettyBy config t1, prettyBy config t2]
  138             )
  139         )
  140     Constant ann c ->
  141       sexp "con" (consAnnIf config ann [prettyTypeOf c, pretty c])
  142     Builtin ann bi ->
  143       sexp "builtin" (consAnnIf config ann [pretty bi])
  144     TyInst ann t ty ->
  145       braces'
  146         ( sep
  147             ( consAnnIf
  148                 config
  149                 ann
  150                 [prettyBy config t, prettyBy config ty]
  151             )
  152         )
  153     Error ann ty ->
  154       sexp "error" (consAnnIf config ann [prettyBy config ty])
  155     IWrap ann ty1 ty2 t ->
  156       sexp
  157         "iwrap"
  158         ( consAnnIf
  159             config
  160             ann
  161             [prettyBy config ty1, prettyBy config ty2, prettyBy config t]
  162         )
  163     Unwrap ann t ->
  164       sexp "unwrap" (consAnnIf config ann [prettyBy config t])
  165     Constr ann ty i es ->
  166       sexp
  167         "constr"
  168         ( consAnnIf
  169             config
  170             ann
  171             ( [prettyBy config ty, pretty i]
  172                 ++ (fmap (prettyBy config) es)
  173             )
  174         )
  175     Case ann ty arg cs ->
  176       sexp
  177         "case"
  178         ( consAnnIf
  179             config
  180             ann
  181             ( [prettyBy config ty, prettyBy config arg]
  182                 ++ (fmap (prettyBy config) cs)
  183             )
  184         )
  185     where
  186       prettyTypeOf :: Some (ValueOf uni) -> Doc dann
  187       prettyTypeOf (Some (ValueOf uni _)) = prettyBy juxtRenderContext $ SomeTypeIn uni
  188 
  189 instance
  190   (PrettyClassicBy configName (Term tyname name uni fun ann), Pretty ann)
  191   => PrettyBy (PrettyConfigClassic configName) (Program tyname name uni fun ann)
  192   where
  193   prettyBy config (Program ann version term) =
  194     sexp "program" (consAnnIf config ann [pretty version, prettyBy config term])
