   34 {- Note [Unique usage errors]
   35 The definitions analysis can find a number of problems with usage of uniques, however
   36 clients may not always want to throw an error on all of them, hence we simply return
   37 them all and allow the client to chose if they want to throw some of them.
   38 -}

   40 -- | Information about a unique, a pair of a definition if we have one and a set of uses.
   41 type UniqueInfo ann = (Maybe (ScopedLoc ann), Set.Set (ScopedLoc ann))
   42 
   43 type UniqueInfos ann = UniqueMap Unique (UniqueInfo ann)
   44 
   45 data ScopedLoc ann = ScopedLoc ScopeType ann deriving stock (Eq, Ord)
   46 
   47 {-| Tag for distinguishing between whether we are talking about the term scope
   48 for variables or the type scope for variables. -}
   49 data ScopeType = TermScope | TypeScope deriving stock (Eq, Ord)
   50 
   51 lookupDef
   52   :: ( Ord ann
   53      , HasUnique name unique
   54      , MonadState (UniqueInfos ann) m
   55      )
   56   => name
   57   -> m (UniqueInfo ann)
   58 lookupDef n = do
   59   previousDef <- gets $ lookupNameIndex n
   60   case previousDef of
   61     Just d -> pure d
   62     Nothing -> do
   63       let empty = (Nothing, mempty)
   64       modify $ insertByNameIndex n empty
   65       pure empty
   66 
   67 addDef
   68   :: ( Ord ann
   69      , HasUnique n unique
   70      , MonadState (UniqueInfos ann) m
   71      , MonadWriter [UniqueError ann] m
   72      )
   73   => n
   74   -- ^ The variable
   75   -> ann
   76   -- ^ The annotation of the variable
   77   -> ScopeType
   78   -- ^ The scope type
   79   -> m ()
   80 addDef n newDef tpe = do
   81   let def = ScopedLoc tpe newDef
   82 
   83   d@(_, uses) <- lookupDef n
   84   checkUndefined n def d
   85   modify $ insertByNameIndex n (Just def, uses)
   86 
   87 -- | Check that a variable is currently undefined.
   88 checkUndefined
   89   :: ( HasUnique n u
   90      , MonadState (UniqueInfos ann) m
   91      , MonadWriter [UniqueError ann] m
   92      )
   93   => n
   94   -- ^ The variable
   95   -> ScopedLoc ann
   96   -- ^ The new definition
   97   -> UniqueInfo ann
   98   -- ^ The existing info
   99   -> m ()
  100 checkUndefined n (ScopedLoc _ newDef) info = case info of
  101   (Just (ScopedLoc _ prevDef), _) -> tell [MultiplyDefined (n ^. theUnique) prevDef newDef]
  102   _ -> pure ()
  103 
  104 addUsage
  105   :: ( Ord ann
  106      , HasUnique n unique
  107      , MonadState (UniqueInfos ann) m
  108      , MonadWriter [UniqueError ann] m
  109      )
  110   => n
  111   -- ^ The variable
  112   -> ann
  113   -- ^ The annotation of the variable
  114   -> ScopeType
  115   -- ^ The scope type
  116   -> m ()
  117 addUsage n newUse tpe = do
  118   let use = ScopedLoc tpe newUse
  119 
  120   d@(def, uses) <- lookupDef n
  121   checkCoherency n use d
  122   checkDefined n use d
  123   modify $ insertByNameIndex n (def, Set.insert use uses)
  124 
  125 checkDefined
  126   :: ( HasUnique n u
  127      , MonadWriter [UniqueError ann] m
  128      )
  129   => n
  130   -- ^ The variable
  131   -> ScopedLoc ann
  132   -- ^ The new definition
  133   -> UniqueInfo ann
  134   -- ^ The existing info
  135   -> m ()
  136 checkDefined n (ScopedLoc _ loc) (def, _) = case def of
  137   Nothing -> tell [FreeVariable (n ^. theUnique) loc]
  138   Just _ -> pure ()
  139 
  140 checkCoherency
  141   :: ( HasUnique n u
  142      , MonadWriter [UniqueError ann] m
  143      )
  144   => n
  145   -- ^ The variable
  146   -> ScopedLoc ann
  147   -- ^ The new definition
  148   -> UniqueInfo ann
  149   -- ^ The existing info
  150   -> m ()
  151 checkCoherency n (ScopedLoc tpe loc) (def, uses) = do
  152   for_ def checkLoc
  153   for_ (Set.toList uses) checkLoc
  154   where
  155     checkLoc (ScopedLoc tpe' loc') =
  156       when (tpe' /= tpe) $
  157         tell [IncoherentUsage (n ^. theUnique) loc' loc]

  159 {-| Given a PLC term, add all of its term and type definitions and usages, including its subterms
  160 and subtypes, to a global map. -}
  161 termDefs
  162   :: ( Ord ann
  163      , HasUnique name TermUnique
  164      , HasUnique tyname TypeUnique
  165      , MonadState (UniqueInfos ann) m
  166      , MonadWriter [UniqueError ann] m
  167      )
  168   => Term tyname name uni fun ann
  169   -> m ()
  170 termDefs tm = do
  171   forMOf_ termSubtermsDeep tm handleTerm
  172   forMOf_ termSubtypesDeep tm handleType
  173 
  174 handleTerm
  175   :: ( Ord ann
  176      , HasUnique name TermUnique
  177      , HasUnique tyname TypeUnique
  178      , MonadState (UniqueInfos ann) m
  179      , MonadWriter [UniqueError ann] m
  180      )
  181   => Term tyname name uni fun ann
  182   -> m ()
  183 handleTerm = \case
  184   Var ann n ->
  185     addUsage n ann TermScope
  186   LamAbs ann n _ _ -> do
  187     addDef n ann TermScope
  188   TyAbs ann tn _ _ -> do
  189     addDef tn ann TypeScope
  190   _ -> pure ()
  191 
  192 -- | Given a type, add its type definition/usage, including its subtypes, to a global map.
  193 handleType
  194   :: ( Ord ann
  195      , HasUnique tyname TypeUnique
  196      , MonadState (UniqueInfos ann) m
  197      , MonadWriter [UniqueError ann] m
  198      )
  199   => Type tyname uni ann
  200   -> m ()
  201 handleType = \case
  202   TyVar ann n ->
  203     addUsage n ann TypeScope
  204   TyForall ann tn _ _ ->
  205     addDef tn ann TypeScope
  206   TyLam ann tn _ _ ->
  207     addDef tn ann TypeScope
  208   _ -> pure ()
  209 
  210 runTermDefs
  211   :: ( Ord ann
  212      , HasUnique name TermUnique
  213      , HasUnique tyname TypeUnique
  214      , Monad m
  215      )
  216   => Term tyname name uni fun ann
  217   -> m (UniqueInfos ann, [UniqueError ann])
  218 runTermDefs = runWriterT . flip execStateT mempty . termDefs
