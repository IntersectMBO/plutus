   52 -- | The type of errors that 'readKnown' and 'makeKnown' can return.
   53 data BuiltinError
   54   = BuiltinUnliftingEvaluationError UnliftingEvaluationError
   55   | BuiltinEvaluationFailure
   56   deriving stock (Show, Eq)
   57 
   58 {-| The monad that 'makeKnown' runs in.
   59 Equivalent to @ExceptT BuiltinError (Writer (DList Text))@, except optimized in two ways:
   60 
   61 1. everything is strict
   62 2. has the 'BuiltinSuccess' constructor that is used for returning a value with no logs
   63    attached, which is the most common case for us, so it helps a lot not to construct and
   64    deconstruct a redundant tuple
   65 
   66 Moving from @ExceptT BuiltinError (Writer (DList Text))@ to this data type gave us a speedup of
   67 8% of total evaluation time.
   68 
   69 Logs are represented as a 'DList', because we don't particularly care about the efficiency of
   70 logging, since there's no logging on the chain and builtins don't emit much anyway. Otherwise
   71 we'd have to use @text-builder@ or @text-builder-linear@ or something of this sort. -}
   72 data BuiltinResult a
   73   = -- 'BuiltinSuccess' is the first constructor to make it a bit more likely for GHC to
   74     -- branch-predict it (which is something that we want, because most builtins return this
   75     -- constructor). It is however not guaranteed that GHC will predict it, because even though
   76     -- it's likely going to be a recursive case (it certainly is in the CEK machine) and thus the
   77     -- constructor has precedence over 'BuiltinFailure', it doesn't have precedence over
   78     -- 'BuiltinSuccessWithLogs', since that case is equally likely to be recursive.
   79     --
   80     -- Unfortunately, GHC doesn't offer any explicit control over branch-prediction (see this
   81     -- ticket: https://gitlab.haskell.org/ghc/ghc/-/issues/849), so relying on hope is the best we
   82     -- can do here.
   83     BuiltinSuccess a
   84   | BuiltinSuccessWithLogs (DList Text) a
   85   | BuiltinFailure (DList Text) BuiltinError
   86   deriving stock (Show, Foldable)
   87 
   88 instance MonadFail BuiltinResult where
   89   fail err = BuiltinFailure (pure $ Text.pack err) BuiltinEvaluationFailure
   90   {-# INLINE fail #-}

  106 {- Note [INLINE and OPAQUE on error-related definitions]
  107 We mark error-related definitions such as prisms like 'structuralUnliftingError' and regular
  108 functions like 'notAConstant' with @INLINE@, because this produces significantly less cluttered
  109 GHC Core. Not doing so results in 20+% larger Core for builtins.
  110 
  111 However in a few specific cases we use @OPAQUE@ instead to get tighter Core. @OPAQUE@ is the
  112 same as @NOINLINE@ except the former _actually_ prevents GHC from inlining the definition unlike the
  113 former. See this for details: https://github.com/ghc-proposals/ghc-proposals/blob/5577fd008924de8d89cfa9855fa454512e7dcc75/proposals/0415-opaque-pragma.rst
  114 
  115 It's hard to predict where @OPAQUE@ instead of @INLINE@ will help to make GHC Core tidier, so it's
  116 mostly just looking into the Core and seeing where there's obvious duplication that can be removed.
  117 Such cases tend to be functions returning a value of a concrete error type (as opposed to a type
  118 variable).
  119 -}

  121 -- See Note [Ignoring context in OperationalError].
  122 {-| Construct a prism focusing on the @*EvaluationFailure@ part of @err@ by taking
  123 that @*EvaluationFailure@ and
  124 
  125 1. pretty-printing and embedding it into an 'UnliftingError' for the setter part of the prism
  126 2. returning it directly for the opposite direction (there's no other way to convert an
  127    'UnliftingError' to an evaluation failure, since the latter doesn't carry any content)
  128 
  129 This is useful for providing 'AsUnliftingError' instances for types such as 'CkUserError' and
  130 'CekUserError'. -}
  131 operationalUnliftingError :: Text -> BuiltinError
  132 operationalUnliftingError =
  133   BuiltinUnliftingEvaluationError . MkUnliftingEvaluationError . OperationalError . MkUnliftingError
  134 {-# INLINE operationalUnliftingError #-}
  135 
  136 structuralUnliftingError :: Text -> BuiltinError
  137 structuralUnliftingError =
  138   BuiltinUnliftingEvaluationError . MkUnliftingEvaluationError . StructuralError . MkUnliftingError
  139 {-# INLINE structuralUnliftingError #-}

  149 -- | Add a log line to the logs.
  150 emit :: Text -> BuiltinResult ()
  151 emit txt = BuiltinSuccessWithLogs (pure txt) ()
  152 {-# INLINE emit #-}
  153 
  154 -- | Prepend logs to a 'BuiltinResult' computation.
  155 withLogs :: DList Text -> BuiltinResult a -> BuiltinResult a
  156 withLogs logs1 = \case
  157   BuiltinSuccess x -> BuiltinSuccessWithLogs logs1 x
  158   BuiltinSuccessWithLogs logs2 x -> BuiltinSuccessWithLogs (logs1 <> logs2) x
  159   BuiltinFailure logs2 err -> BuiltinFailure (logs1 <> logs2) err
  160 {-# INLINE withLogs #-}
  161 
  162 instance Functor BuiltinResult where
  163   fmap f (BuiltinSuccess x) = BuiltinSuccess (f x)
  164   fmap f (BuiltinSuccessWithLogs logs x) = BuiltinSuccessWithLogs logs (f x)
  165   fmap _ (BuiltinFailure logs err) = BuiltinFailure logs err
  166   {-# INLINE fmap #-}
  167 
  168   -- Written out explicitly just in case.
  169   x <$ BuiltinSuccess _ = BuiltinSuccess x
  170   x <$ BuiltinSuccessWithLogs logs _ = BuiltinSuccessWithLogs logs x
  171   _ <$ BuiltinFailure logs err = BuiltinFailure logs err
  172   {-# INLINE (<$) #-}
  173 
  174 instance Applicative BuiltinResult where
  175   pure = BuiltinSuccess
  176   {-# INLINE pure #-}
  177 
  178   BuiltinSuccess f <*> a = fmap f a
  179   BuiltinSuccessWithLogs logs f <*> a = withLogs logs $ fmap f a
  180   BuiltinFailure logs err <*> _ = BuiltinFailure logs err
  181   {-# INLINE (<*>) #-}
  182 
  183   -- Better than the default implementation, because the value in the 'BuiltinSuccess' case
  184   -- doesn't need to be retained.
  185   BuiltinSuccess _ *> b = b
  186   BuiltinSuccessWithLogs logs _ *> b = withLogs logs b
  187   BuiltinFailure logs err *> _ = BuiltinFailure logs err
  188   {-# INLINE (*>) #-}
  189 
  190 instance Monad BuiltinResult where
  191   BuiltinSuccess x >>= f = f x
  192   BuiltinSuccessWithLogs logs x >>= f = withLogs logs $ f x
  193   BuiltinFailure logs err >>= _ = BuiltinFailure logs err
  194   {-# INLINE (>>=) #-}
  195 
  196   (>>) = (*>)
  197   {-# INLINE (>>) #-}

  199 {-| 'throwError' puts every operational unlifting error into the 'BuiltinFailure' logs. This is to
  200 compensate for the historical lack of error message content in operational errors (structural
  201 ones don't have this problem) in our evaluators (the CK and CEK machines). It would be better to
  202 fix the underlying issue and allow operational evaluation errors to carry some form of content,
  203 but for now we just fix the symptom in order for the end user to see the error message that they
  204 are supposed to see. The fix even makes some sense: what we do here is we emulate logging when
  205 the thrown unlifting error is an operational one, i.e. this is similar to what some builtins do
  206 manually (like when a crypto builtin fails and puts info about the failure into the logs). -}
  207 instance MonadError BuiltinError BuiltinResult where
  208   throwError builtinErr = BuiltinFailure operationalLogs builtinErr
  209     where
  210       operationalLogs = case builtinErr of
  211         BuiltinUnliftingEvaluationError
  212           ( MkUnliftingEvaluationError
  213               ( OperationalError
  214                   (MkUnliftingError operationalErr)
  215                 )
  216             ) -> pure operationalErr
  217         _ -> mempty
  218   {-# INLINE throwError #-}
  219 
  220   -- Throwing logs out is lame, but embedding them into the error would be weird, since that
  221   -- would change the error. Not that any of that matters, we only implement this because it's a
  222   -- method of 'MonadError' and we can't not implement it.
  223   --
  224   -- We could make it @MonadError (DList Text, BuiltinError)@, but logs are arbitrary and are not
  225   -- necessarily an inherent part of an error, so preserving them is as questionable as not doing
  226   -- so.
  227   BuiltinFailure _ err `catchError` f = f err
  228   res `catchError` _ = res
  229   {-# INLINE catchError #-}
  230 
  231 builtinResultFailure :: BuiltinResult a
  232 builtinResultFailure = BuiltinFailure mempty BuiltinEvaluationFailure
  233 {-# INLINE builtinResultFailure #-}
