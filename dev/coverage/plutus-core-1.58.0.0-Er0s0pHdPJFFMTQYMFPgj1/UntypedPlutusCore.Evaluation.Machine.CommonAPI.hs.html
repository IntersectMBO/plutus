   75 -- The type of the machine (runner function).
   76 type MachineRunner cost uni fun ann =
   77   MachineParameters CekMachineCosts fun (CekValue uni fun ann)
   78   -> ExBudgetMode cost uni fun
   79   -> EmitterMode uni fun
   80   -> NTerm uni fun ann
   81   -> CekReport cost NamedDeBruijn uni fun
   82 
   83 {- Note [CEK runners naming convention]
   84 A function whose name ends in @NoEmit@ does not perform logging and so does not return any logs.
   85 A function whose name starts with @unsafe@ throws exceptions instead of returning them purely.
   86 A function from the @runCek@ family takes an 'ExBudgetMode' parameter and returns the final
   87 'CekExBudgetState' (and possibly logs). Note that 'runCek' is defined in @...Cek.Internal@ for
   88 reasons explained in Note [Compilation peculiarities].
   89 A function from the @evaluateCek@ family does not return the final 'ExBudgetMode', nor does it
   90 allow one to specify an 'ExBudgetMode'. I.e. such functions are only for fully evaluating programs
   91 (and possibly returning logs). See also haddocks of 'enormousBudget'.
   92 -}

   94 {-| Evaluate a term using a machine with logging enabled and keep track of costing.
   95 A wrapper around the internal runCek to debruijn input and undebruijn output.
   96 *THIS FUNCTION IS PARTIAL if the input term contains free variables* -}
   97 runCek
   98   :: MachineRunner cost uni fun ann
   99   -> MachineParameters CekMachineCosts fun (CekValue uni fun ann)
  100   -> ExBudgetMode cost uni fun
  101   -> EmitterMode uni fun
  102   -> Term Name uni fun ann
  103   -> CekReport cost Name uni fun
  104 runCek runner params mode emitMode term =
  105   -- translating input
  106   case runExcept @FreeVariableError $ deBruijnTerm term of
  107     Left fvError -> throw fvError
  108     Right dbt -> do
  109       -- Don't use 'let': https://github.com/IntersectMBO/plutus/issues/3876
  110       case runner params mode emitMode dbt of
  111         -- translating back the output
  112         CekReport res cost' logs ->
  113           CekReport (mapTermCekResult gracefulUnDeBruijn res) cost' logs
  114   where
  115     -- \*GRACEFULLY* undebruijnifies: a) the error-cause-term (if it exists) or b) the success
  116     -- \*value-term.
  117     -- 'Graceful' means that the (a) && (b) undebruijnifications do not throw an error upon a free
  118     -- variable encounter: free debruijn indices will be turned to free, consistent uniques
  119     gracefulUnDeBruijn :: Term NamedDeBruijn uni fun () -> Term Name uni fun ()
  120     gracefulUnDeBruijn t =
  121       runQuote
  122         . flip evalStateT mempty
  123         $ unDeBruijnTermWith freeIndexAsConsistentLevel t

  125 {-| Evaluate a term using a machine with logging disabled and keep track of costing.
  126 *THIS FUNCTION IS PARTIAL if the input term contains free variables* -}
  127 runCekNoEmit
  128   :: MachineRunner cost uni fun ann
  129   -> MachineParameters CekMachineCosts fun (CekValue uni fun ann)
  130   -> ExBudgetMode cost uni fun
  131   -> Term Name uni fun ann
  132   -> (Either (CekEvaluationException Name uni fun) (Term Name uni fun ()), cost)
  133 runCekNoEmit runner params mode =
  134   -- throw away the logs
  135   (\(CekReport res cost _logs) -> (cekResultToEither res, cost))
  136     . runCek runner params mode noEmitter
  137 
  138 {-| Evaluate a term using a machine with logging enabled.
  139 *THIS FUNCTION IS PARTIAL if the input term contains free variables* -}
  140 evaluateCek
  141   :: ThrowableBuiltins uni fun
  142   => MachineRunner RestrictingSt uni fun ann
  143   -> EmitterMode uni fun
  144   -> MachineParameters CekMachineCosts fun (CekValue uni fun ann)
  145   -> Term Name uni fun ann
  146   -> (Either (CekEvaluationException Name uni fun) (Term Name uni fun ()), [Text])
  147 evaluateCek runner emitMode params =
  148   -- throw away the cost
  149   (\(CekReport res _cost logs) -> (cekResultToEither res, logs))
  150     . runCek runner params restrictingEnormous emitMode
  151 
  152 {-| Evaluate a term using a machine with logging disabled.
  153 *THIS FUNCTION IS PARTIAL if the input term contains free variables* -}
  154 evaluateCekNoEmit
  155   :: ThrowableBuiltins uni fun
  156   => MachineRunner RestrictingSt uni fun ann
  157   -> MachineParameters CekMachineCosts fun (CekValue uni fun ann)
  158   -> Term Name uni fun ann
  159   -> Either (CekEvaluationException Name uni fun) (Term Name uni fun ())
  160 evaluateCekNoEmit runner params = fst . runCekNoEmit runner params restrictingEnormous
