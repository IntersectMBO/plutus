   15 {-| Church-encoded @Nat@ as a PLC type.
   16 
   17 > all (r :: *). r -> (r -> r) -> r -}
   18 churchNat :: Type TyName uni ()
   19 churchNat = runQuote $ do
   20   r <- freshTyName "r"
   21   return
   22     . TyForall () r (Type ())
   23     . TyFun () (TyVar () r)
   24     . TyFun () (TyFun () (TyVar () r) $ TyVar () r)
   25     $ TyVar () r

   27 {-| Church-encoded '0' as a PLC term.
   28 
   29 > /\(r :: *) -> \(z : r) (f : r -> r) -> z -}
   30 churchZero :: TermLike term TyName Name uni fun => term ()
   31 churchZero = runQuote $ do
   32   r <- freshTyName "r"
   33   z <- freshName "z"
   34   f <- freshName "f"
   35   return
   36     . tyAbs () r (Type ())
   37     . lamAbs () z (TyVar () r)
   38     . lamAbs () f (TyFun () (TyVar () r) $ TyVar () r)
   39     $ var () z

   41 {-| Church-encoded 'succ' as a PLC term.
   42 
   43 > \(n : nat) -> /\(r :: *) -> \(z : r) (f : r -> r) -> f (n {r} z f) -}
   44 churchSucc :: TermLike term TyName Name uni fun => term ()
   45 churchSucc = runQuote $ do
   46   n <- freshName "n"
   47   r <- freshTyName "r"
   48   z <- freshName "z"

   50   return
   51     . lamAbs () n churchNat
   52     . tyAbs () r (Type ())
   53     . lamAbs () z (TyVar () r)
   54     . lamAbs () f (TyFun () (TyVar () r) $ TyVar () r)
   55     . apply () (var () f)
   56     $ mkIterAppNoAnn
   57       (tyInst () (var () n) $ TyVar () r)
   58       [ var () z
   59       , var () f
