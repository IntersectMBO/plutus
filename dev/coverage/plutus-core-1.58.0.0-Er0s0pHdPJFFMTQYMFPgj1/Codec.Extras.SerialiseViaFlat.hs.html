   20 {-| Newtype to provide 'Serialise' instances for types with a 'Flat' instance
   21   that just encodes the flat-serialized value as a CBOR bytestring -}
   22 newtype SerialiseViaFlat a = SerialiseViaFlat {unSerialiseViaFlat :: a}
   23 
   24 instance Flat.Flat a => Serialise (SerialiseViaFlat a) where
   25   encode = encode . Flat.flat . unSerialiseViaFlat
   26   decode = SerialiseViaFlat <$> decodeViaFlatWith Flat.decode
   27 
   28 decodeViaFlatWith :: Flat.Get a -> CBOR.Decoder s a
   29 decodeViaFlatWith decoder = do
   30   bs <- CBOR.decodeBytes
   31   -- lift any flat's failures to be cborg failures (MonadFail)
   32   fromRightM (fail . show) $ Flat.unflatWith decoder bs

   34 {-| The errors returned by `cborg` are plain strings (untyped). With this
   35 function we try to map onto datatypes, those cborg error messages that we are
   36 interested in.
   37 
   38 Currently we are only interested in error messages returned by the
   39 `CBOR.decodeBytes` decoder;
   40 see `PlutusLedgerApi.Common.SerialisedScript.scriptCBORDecoder`. -}
   41 readDeserialiseFailureInfo :: CBOR.DeserialiseFailure -> DeserialiseFailureInfo
   42 readDeserialiseFailureInfo (CBOR.DeserialiseFailure byteOffset reason) =
   43   DeserialiseFailureInfo byteOffset $ interpretReason reason
   44   where
   45     -- Note that this is subject to change if `cborg` dependency changes.
   46     -- Currently: cborg-0.2.10.0
   47     interpretReason :: String -> DeserialiseFailureReason
   48     interpretReason = \case
   49       -- Relevant Sources:
   50       -- <https://github.com/well-typed/cborg/blob/cborg-0.2.10.0/cborg/src/Codec/CBOR/Read.hs#L226>
   51       -- <https://github.com/well-typed/cborg/blob/cborg-0.2.10.0/cborg/src/Codec/CBOR/Read.hs#L1424>
   52       -- <https://github.com/well-typed/cborg/blob/cborg-0.2.10.0/cborg/src/Codec/CBOR/Read.hs#L1441>
   53       "end of input" -> EndOfInput
   54       -- Relevant Sources:
   55       -- <https://github.com/well-typed/cborg/blob/cborg-0.2.10.0/cborg/src/Codec/CBOR/Read.hs#L1051>
   56       "expected bytes" -> ExpectedBytes
   57       msg -> OtherReason msg
   58 
   59 {-| Similar to `CBOR.DeserialiseFailure`, with the difference that plain
   60 string reason messages are turned into the datatype: `DeserialiseFailureReason`. -}
   61 data DeserialiseFailureInfo = DeserialiseFailureInfo
   62   { dfOffset :: CBOR.ByteOffset
   63   , dfReason :: DeserialiseFailureReason
   64   }
   65   deriving stock (Eq, Show)
   66 
   67 instance Pretty DeserialiseFailureInfo where
   68   pretty (DeserialiseFailureInfo offset reason) =
   69     "CBOR deserialisation failed at the offset"
   70       <+> pretty offset
   71       <+> "for the following reason:"
   72       <+> pretty reason
   73 
   74 -- | The reason of the cbor failure as a datatype, not as a plain string.
   75 data DeserialiseFailureReason
   76   = -- | Not enough input provided
   77     EndOfInput
   78   | -- | The bytes inside the input are malformed.
   79     ExpectedBytes
   80   | {-| This is either a cbor failure that we (plutus) are not aware of,
   81     or an underlying flat failure. We use whatever message `cborg` or flat returns. -}
   82     OtherReason String
   83   deriving stock (Eq, Show)
   84 
   85 instance Pretty DeserialiseFailureReason where
   86   pretty = \case
   87     EndOfInput -> "reached the end of input while more data was expected."
   88     ExpectedBytes -> "the bytes inside the input are malformed."
   89     OtherReason msg -> pretty msg
