   40 {-| Turn a list of Haskell types @args@ into a functional type ending in @res@.
   41 
   42 >>> :set -XDataKinds
   43 >>> :kind! FoldArgs [(), Bool] Integer
   44 FoldArgs [(), Bool] Integer :: *
   45 = () -> Bool -> Integer -}
   46 type family FoldArgs args res where
   47   FoldArgs '[] res = res
   48   FoldArgs (arg ': args) res = arg -> FoldArgs args res

   50 {-| The meaning of a built-in function consists of its type represented as a 'TypeScheme',
   51 its Haskell denotation and its uninstantiated runtime denotation.
   52 
   53 The 'TypeScheme' of a built-in function is used for example for
   54 
   55 1. computing the PLC type of the function to be used during type checking
   56 2. getting arity information
   57 3. generating arbitrary values to apply the function to in tests
   58 
   59 The denotation is lazy, so that we don't need to worry about a builtin being bottom
   60 (happens in tests). The production path is not affected by that, since only runtime denotations
   61 are used for evaluation. -}
   62 data BuiltinMeaning val cost
   63   = forall args res. BuiltinMeaning
   64       (TypeScheme val args res)
   65       ~(FoldArgs args res)
   66       (cost -> BuiltinRuntime val)
   67 
   68 -- | Constraints available when defining a built-in function.
   69 type HasMeaningIn uni val = (Typeable val, ExMemoryUsage val, HasConstantIn uni val)

  117 {- Note [Builtin semantics variants]
  118 The purpose of the "builtin semantics variant" feature is to provide multiple,
  119 different denotations (implementations) for the same builtin(s).  An example use
  120 of this feature is for "fixing" the behaviour of `ConsByteString` builtin to
  121 throw an error instead of overflowing its first argument.
  122 
  123 One denotation from each builtin is grouped into a 'BuiltinSemanticsVariant'.
  124 Each Plutus Language version is linked to a specific 'BuiltinSemanticsVariant'
  125 (done by plutus-ledger-api); e.g. plutus-v1 and plutus-v2 are linked to
  126 'DefaultFunSemanticsVariantA' and 'DefaultFunSemanticsVariantB', whereas
  127 plutus-v3 changes the set of denotations to 'DefaultFunSemanticsVariantC' (thus
  128 fixing 'ConsByteString').
  129 
  130 Each 'BuiltinSemanticsVariant' (grouping) can change the denotation of one or
  131 more builtins --- or none, but what's the point in that?
  132 
  133 This 'BuiltinSemanticsVariant' is modelled as a datatype *associated* to the
  134 @fun@. This associated datatype is required to provide an instance of 'Default'
  135 for quality-of-life purpose; the 'def'ault builtin semantics variant is expected
  136 to point to the builtin semantics variant that the plutus-tx/plutus-ir compiler
  137 is currently targeting.
  138 
  139 Note that (old) denotations of a 'BuiltinSemanticsVariant' cannot be removed or
  140 deprecated once published to the chain.
  141 
  142 The way that this feature is implemented buys us more than we currently need:
  143 - allows also a versioned change to a builtin's *type signature*, i.e. type of arguments/result as
  144   well as number of arguments.
  145 - allows also a versioned change to a builtin's cost model parameters
  146 
  147 Besides having no need for this currently, it complicates the codebase since the
  148 typechecker now pointlessly wants to know the builtin semantics before
  149 typechecking. To alleviate this, we use the 'Default.def' builtin semantics
  150 variant during typechecking / lifting. @effectfully: the solution to the problem
  151 would be to establish what kind of backwards compatibility we're willing to
  152 maintain and pull all of that into a separate data type and make it a part of
  153 BuiltinMeaning.
  154 -}

  156 {- Note [Automatic derivation of type schemes]
  157 We use two type classes for automatic derivation of type/runtime schemes and runtime denotations:
  158 'KnownPolytype' and 'KnownMonotype'.
  159 The terminology is due to
  160 https://en.wikipedia.org/wiki/Hindley%E2%80%93Milner_type_system#The_Hindley%E2%80%93Milner_
  161 type_system
  162 
  163 Check out the source of "PlutusCore.Builtin.Runtime" for an explanation of what a runtime
  164 denotation is.
  165 
  166 'KnownPolytype' and 'KnownMonotype' are responsible for deriving polymorphic and monomorphic types,
  167 respectively.
  168 
  169 'KnownPolytype' turns every bound variable into a 'TypeSchemeAll'/'BuiltinExpectForce'. We extract
  170 variables from the type of the Haskell denotation using the 'ToBinds' associated type
  171 family. Variables are collected in the order that they appear in (i.e. just like in Haskell). For
  172 example, processing a type like
  173 
  174       :: ( a ~ Opaque val (TyVarRep ('TyNameRep "a" 0))
  175          , b ~ Opaque val (TyVarRep ('TyNameRep "b" 1))
  176          )
  177       => b -> a -> b
  178 
  179 with 'ToBinds' results in the following list of bindings:
  180 
  181     '[ 'Some ('TyNameRep "b" 1), 'Some ('TyNameRep "a" 0) ]
  182 
  183 'KnownMonotype' turns every argument that the Haskell denotation of a builtin receives into a
  184 'TypeSchemeArrow'/'BuiltinExpectArgument'. We extract the arguments from the type of the Haskell
  185 denotation using the 'GetArgs' type family.
  186 
  187 Higher-kinded type variables are fully supported.
  188 
  189 At this point in the pipeline the type of the denotation of a builtin is assumed to be fully
  190 elaborated (i.e. monomorphic).
  191 -}
  192 
  193 -- | Chop a function type to get a list of its argument types.
  194 type GetArgs :: GHC.Type -> [GHC.Type]
  195 type family GetArgs a where
  196   GetArgs (a -> b) = a ': GetArgs b
  197   GetArgs _ = '[]

  199 {- Note [Merging the denotation and the costing function]
  200 The runtime denotation of a builtin computes both the builtin application and its cost
  201 (see the docs of 'BuiltinRuntime' for details). Doing both at the same time has a number of benefits
  202 (see Note [runCostingFun* API]), however in the user-facing API we want to separate the concepts
  203 of the denotation and the costing function. This is because:
  204 
  205 1. the two are fundamentally distinct and we have loads of documentation for each of them
  206    separately, so conflating them in the actual API would be unnecessary coupling
  207 2. right now it's clear which bits of the definition of a builtin constitute evaluation and which
  208    ones constitute costing as the two are different arguments to 'makeBuiltinMeaning'. If evaluation
  209    and costing were intertwined, it would be much harder to review the definition of a builtin
  210 3. ... and it would also be more boilerplate and less clear type signatures
  211 
  212 Hence we want 'makeBuiltinMeaning' to take evaluation and costing bits separately and intertwine
  213 them behind the scenes. Which is straightforward: we only need to pass the two together in the
  214 methods of the 'KnownMonotype' and 'KnownPolytype' classes and zip them argument by argument
  215 into a single 'BuiltinRuntime'.
  216 -}

  218 {-| A class that allows us to derive a monotype for a builtin.
  219 We could've computed the runtime denotation from the
  220 'TypeScheme' and the denotation of the builtin, but not statically (due to unfolding not working
  221 for recursive functions and 'TypeScheme' being recursive, i.e. requiring the conversion function
  222 to be recursive), and so it would cause us to retain a lot of evaluation-irrelevant stuff in the
  223 constructors of 'BuiltinRuntime', which has to make evaluation slower (we didn't check) and
  224 certainly makes the generated Core much harder to read. Technically speaking, we could get
  225 a 'RuntimeScheme' from the 'TypeScheme' and the denotation statically if we changed the
  226 definition of 'TypeScheme' and made it a singleton, but then the conversion function would have
  227 to become a class anyway and we'd just replicate what we have here, except in a much more
  228 complicated way. -}
  229 class KnownMonotype val args res where
  230   knownMonotype :: TypeScheme val args res
  231 
  232   {-| Convert the denotation of a builtin to its runtime counterpart .
  233   The argument is in 'ReadKnownM', because that's what deferred unlifting amounts to:
  234   passing the action returning the builtin application around until full saturation, which is
  235   when the action actually gets run. -}
  236   toMonoF
  237     :: ReadKnownM (FoldArgs args res, FoldArgs args ExBudgetStream)
  238     -> BuiltinRuntime val

  240 {-| Once we've run out of term-level arguments, we return a
  241 'TypeSchemeResult'/'RuntimeSchemeResult'. -}
  242 instance
  243   (Typeable res, KnownTypeAst TyName (UniOf val) res, MakeKnown val res)
  244   => KnownMonotype val '[] res
  245   where
  246   knownMonotype = TypeSchemeResult
  247 
  248   toMonoF =
  249     either
  250       -- Unlifting has failed and we don't care about costing at this point, since we're about
  251       -- to terminate evaluation anyway, hence we put 'mempty' as the cost of the operation.
  252       --
  253       -- Note that putting the cost inside of 'BuiltinResult' is not an option, since forcing
  254       -- the 'BuiltinResult' computation is exactly forcing the builtin application, which we
  255       -- can't do before accounting for the cost of the application, i.e. the cost must be
  256       -- outside of 'BuiltinResult'.
  257       --
  258       -- We could introduce a level of indirection and say that a 'BuiltinCostedResult' is
  259       -- either a budgeting failure or a budgeting success with a cost and a 'BuiltinResult'
  260       -- computation inside, but that would slow things down a bit and the current strategy is
  261       -- reasonable enough.
  262       builtinRuntimeFailure
  263       (\(x, cost) -> BuiltinCostedResult cost $ makeKnown x)
  264   {-# INLINE toMonoF #-}

  266 {- Note [One-shotting runtime denotations]
  267 In @KnownMonotype val (arg ': args) res@ we 'oneShot' the runtime denotations. Otherwise GHC creates
  268 let-bindings and lifts them out of some of the lambdas in the runtime denotation, which would speed
  269 up partial applications if they were getting reused, but at some point it was verified that we
  270 didn't have any reusage of partial applications: https://github.com/IntersectMBO/plutus/pull/4629
  271 
  272 One-shotting the runtime denotations alone made certain game contracts slower by ~9%. A lot of time
  273 was spent on the investigation, but we still don't know why that was happening. Plus, basically any
  274 other change to the builtins machinery would cause the same kind of slowdown, so we just admitted
  275 defeat and decided it wasn't worth investigating the issue further.
  276 Relevant thread: https://github.com/IntersectMBO/plutus/pull/4620
  277 
  278 The speedup that adding a call to 'oneShot' gives us, if any, is smaller than our noise threshold,
  279 however it also makes those confusing allocations disappear from the generated Core, which is enough
  280 of a reason to add the call.
  281 -}

  283 {- Note [Strict application in runtime denotations]
  284 Runtime denotations contain strict let-bindings. Those are important: without them GHC thinks that
  285 the argument may not be needed in the end and so creates a thunk for it, which is not only
  286 unnecessary allocation, but also prevents things from being unboxed. The argument may indeed not be
  287 needed in the end, but in that case we've got an evaluation failure and we're about to terminate
  288 evaluation anyway, hence we care much more about optimizing the happy path.
  289 -}

  291 -- | Every term-level argument becomes a 'TypeSchemeArrow'/'RuntimeSchemeArrow'.
  292 instance
  293   ( Typeable arg
  294   , KnownTypeAst TyName (UniOf val) arg
  295   , MakeKnown val arg
  296   , ReadKnown val arg
  297   , KnownMonotype val args res
  298   )
  299   => KnownMonotype val (arg ': args) res
  300   where
  301   knownMonotype = TypeSchemeArrow knownMonotype
  302 
  303   -- See Note [One-shotting runtime denotations].
  304   -- Grow the builtin application within the received action and recurse on the result.
  305   toMonoF getBoth = BuiltinExpectArgument . oneShot $ \arg ->
  306     -- The lazy application of 'toMonoF' ensures that unlifting of the argument will happen
  307     -- upon full saturation and not before that. This is known as "operationally deferred
  308     -- unlifting" (as opposed to "operationally immediate unlifting") or "call-by-name
  309     -- unlifting" (as opposed to "call-by-value unlifting"). We do it this way to guarantee that
  310     -- the cost of unlifting will be accounted for before unlifting is performed. If we did
  311     -- unlifting eagerly here, this would make the node do work that is not accounted for until
  312     -- full saturation is reached, which may never happen if the partial application is thrown
  313     -- away.
  314     --
  315     -- The disadvantage of this approach is that @addInteger 42@ will always unlift @42@ upon
  316     -- full saturation even if this partial application is saved to a variable. But the way
  317     -- costing calibration benchmarks are set up, we always evaluate a single application, so
  318     -- the cost of unlifting is included in the cost of the builtin regardless of whether
  319     -- there's caching of unlifting or not. Hence the user pays for unlifting anyway and we can
  320     -- prioritize safety over performance here.
  321     --
  322     -- 'oneShot' ensures that GHC doesn't attempt to pull stuff out of the builtin
  323     -- implementation to create thunks. This would give us a "call-by-need" behavior, which may
  324     -- sound enticing as it would give us both caching and operationally deferred unlifting, but
  325     -- this comes at a cost of creating unnecessary thunks in the most common case where there's
  326     -- no benefit from having caching as the builtin application is going to be computed only
  327     -- once. So we choose the "call-by-name" behavior and 'oneShot' is what enables that.
  328     oneShot (toMonoF @val @args @res) $ do
  329       (f, exF) <- getBoth
  330       -- Force the argument that gets passed to the denotation. This seems to help performance
  331       -- a bit (possibly due to its impact on strictness analysis), plus this way we ensure
  332       -- that if computing the argument throws an exception (isn't supposed to happen), we'll
  333       -- catch it in tests.
  334       !x <- readKnown arg
  335       -- See Note [Strict application in runtime denotations].
  336       let !exY = exF x
  337       pure (f x, exY)
  338   {-# INLINE toMonoF #-}

  359 -- Here we unpack an existentially packed @kind@ and constrain it afterwards!
  360 -- So promoted existentials are true sigmas! If we were at the term level, we'd have to pack
  361 -- @kind@ along with the @KnownKind kind@ constraint, otherwise when we unpack the existential,
  362 -- all information is lost and we can't do anything with @kind@.
  363 -- | Every type-level argument becomes a 'TypeSchemeAll'.
  364 instance
  365   (KnownSymbol name, KnownNat uniq, KnownKind kind, KnownPolytype binds val args res)
  366   => KnownPolytype ('Some ('TyNameRep @kind name uniq) ': binds) val args res
  367   where
  368   knownPolytype = TypeSchemeAll @name @uniq @kind Proxy $ knownPolytype @binds
  369 
  370   toPolyF = BuiltinExpectForce . toPolyF @binds @val @args @res

  373 -- | Ensure a built-in function is not nullary and throw a nice error otherwise.
  374 type ThrowOnBothEmpty :: [Some TyNameRep] -> [GHC.Type] -> Bool -> GHC.Type -> GHC.Constraint
  375 type family ThrowOnBothEmpty binds args isBuiltin a where
  376   ThrowOnBothEmpty '[] '[] 'True a =
  377     TypeError
  378       ( 'Text "A built-in function must take at least one type or term argument"
  379           ':$$: 'Text "‘"
  380             ':<>: 'ShowType a
  381             ':<>: 'Text "’ is a built-in type"
  382             ':<>: 'Text " so you can embed any of its values as a constant"
  383           ':$$: 'Text "If you still want a built-in function, add a dummy ‘()’ argument"
  384       )
  385   ThrowOnBothEmpty '[] '[] 'False a =
  386     TypeError
  387       ( 'Text "A built-in function must take at least one type or term argument"
  388           ':$$: 'Text "To fix this error add a dummy ‘()’ argument"
  389       )
  390   ThrowOnBothEmpty _ _ _ _ = ()

  392 {-| A function turned into a type class with exactly one fully general instance.
  393 We can't package up the constraints of 'makeBuiltinMeaning' (see the instance) into a type or
  394 class synonym, because they contain a bunch of variables defined by @~@ or determined via
  395 functional dependencies and neither class nor type definitions can handle that
  396 (see https://gitlab.haskell.org/ghc/ghc/-/issues/7100). Inlining three lines of constraints
  397 whenever we need to call 'makeBuiltinMeaning' over a non-concrete type is a bad option and this
  398 abstraction is free anyway, hence its existence.
  399 
  400 The @a@ type variable goes first, because @makeBuiltinMeaning \@A@ is a common pattern. -}
  401 class MakeBuiltinMeaning a val where
  402   -- See Note [Automatic derivation of type schemes]
  403   {-| Construct the meaning for a built-in function by automatically deriving its
  404   'TypeScheme', given
  405 
  406   1. the denotation of the builtin
  407   2. an uninstantiated costing function -}
  408   makeBuiltinMeaning
  409     :: a
  410     -> (cost -> FoldArgs (GetArgs a) ExBudgetStream)
  411     -> BuiltinMeaning val cost
  412 
  413 instance
  414   ( uni ~ UniOf val
  415   , binds ~ ToBinds uni '[] a
  416   , args ~ GetArgs a
  417   , a ~ FoldArgs args res
  418   , ThrowOnBothEmpty binds args (IsBuiltin uni a) a
  419   , ElaborateFromTo uni 0 j val a
  420   , KnownPolytype binds val args res
  421   )
  422   => MakeBuiltinMeaning a val
  423   where
  424   makeBuiltinMeaning f toExF =
  425     BuiltinMeaning (knownPolytype @binds @val @args @res) f $ \cost ->
  426       -- In order to make the 'BuiltinRuntime' of a builtin cacheable we need to tell GHC to
  427       -- create a thunk for it, which we achieve by applying 'lazy' to the 'BuiltinRuntime'
  428       -- here.
  429       --
  430       -- Those thunks however require a lot of care to be properly shared rather than
  431       -- recreated every time a builtin application is evaluated, see 'toBuiltinsRuntime' for
  432       -- how we sort it out.
  433       lazy $ case toExF cost of
  434         -- See Note [Optimizations of runCostingFun*] for why we use strict @case@.
  435         !exF -> toPolyF @binds @val @args @res $ pure (f, exF)
  436   {-# INLINE makeBuiltinMeaning #-}

  439 toBuiltinRuntime :: cost -> BuiltinMeaning val cost -> BuiltinRuntime val
  440 toBuiltinRuntime cost (BuiltinMeaning _ _ denot) = denot cost

  443 -- See Note [Inlining meanings of builtins].
  444 {-| Calculate runtime info for all built-in functions given meanings of builtins (as a constraint),
  445 the semantics variant of the set of builtins and a cost model. -}
  446 toBuiltinsRuntime
  447   :: (cost ~ CostingPart uni fun, ToBuiltinMeaning uni fun, HasMeaningIn uni val)
  448   => BuiltinSemanticsVariant fun
  449   -> cost
  450   -> BuiltinsRuntime fun val
  451 toBuiltinsRuntime semvar cost =
  452   -- A call to 'lazy' is to make sure that the returned 'BuiltinsRuntime' is properly cached in a
  453   -- 'let'-binding. This makes it easier for GHC to optimize the internals of builtins, because
  454   -- without a 'let'-binding GHC would sometimes refuse to cooperate and push 'toBuiltinRuntime'
  455   -- to the inside of the inlined 'toBuiltinMeaning' call, creating lots of 'BuiltinMeaning's
  456   -- instead of 'BuiltinRuntime's with the former hiding the costing optimizations behind a lambda
  457   -- binding the @cost@ variable, which makes the optimizations useless.
  458   -- By using 'lazy' we tell GHC to create a separate thunk, which it can properly optimize,
  459   -- because the other bazillion things don't get in the way. We used to use an explicit
  460   -- 'let'-binding marked with @OPAQUE@, but that turned out to be unreliable, because GHC
  461   -- feels free to turn it into a join point instead of a proper thunk.
  462   lazy . BuiltinsRuntime $ toBuiltinRuntime cost . inline toBuiltinMeaning semvar
  463 {-# INLINE toBuiltinsRuntime #-}
