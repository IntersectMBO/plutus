   15 {- Note [Single-element streams]
   16 Both 'CostStream' and 'ExBudgetStream' are semantically equivalent to 'NonEmpty' (modulo strictness)
   17 except instead of making the first element of each of these stream types a special one, we make
   18 special the last one. The reason for this is that we want to maximally optimize the case of a
   19 single-element stream, because it's the most common one and with a 'NonEmpty'-style data type we'd
   20 have to pattern match twice in order to extract the value from the head of the stream and make sure
   21 it's a single one, while with our approach it's only one pattern match. Plus we don't need to touch
   22 any recursive parts at all when we're operating with single-element streams. It comes at a cost
   23 however: recursive functions over streams often need to have the single-element case hardcoded even
   24 when the general recursion would suffice, because GHC can't inline recursive functions and we rely
   25 on inlining heavily. So we often manually unwrap one step of recursion just to make the wrapper
   26 inlineable allowing for optimized handling of single-element streams.
   27 -}
   28 
   29 -- See Note [Single-element streams]
   30 {-| A lazy stream of 'CostingInteger's. Basically @NonEmpty CostingInteger@, except the elements
   31 are stored strictly.
   32 
   33 The semantics of a stream are those of the sum of its elements. I.e. a stream that is a reordered
   34 version of another stream is considered equal to that stream.
   35 
   36 All costs are assumed not to be negative and functions handling 'CostStream's may rely on this
   37 assumption. Negative costs (a.k.a. allowing the user to forge execution units at runtime)
   38 wouldn't make sense. -}
   39 data CostStream
   40   = CostLast {-# UNPACK #-} !CostingInteger
   41   | CostCons {-# UNPACK #-} !CostingInteger CostStream
   42   deriving stock (Show)
   43 
   44 -- TODO: (# CostingInteger, (# (# #) | CostStream #) #)?
   45 -- | Uncons an element from a 'CostStream' and return the rest of the stream, if not empty.
   46 unconsCost :: CostStream -> (CostingInteger, Maybe CostStream)
   47 unconsCost (CostLast cost) = (cost, Nothing)
   48 unconsCost (CostCons cost costs) = (cost, Just costs)
   49 {-# INLINE unconsCost #-}
   50 
   51 {-| Cons an element to a 'CostStream', if given any. Otherwise create a new 'CostStream' using
   52 'CostLast'. -}
   53 reconsCost :: CostingInteger -> Maybe CostStream -> CostStream
   54 reconsCost cost = maybe (CostLast cost) (CostCons cost)
   55 {-# INLINE reconsCost #-}

   57 {- Note [Global local functions]
   58 Normally when defining a helper function one would put it into a @where@ or a @let@ block.
   59 However if the enclosing function gets inlined, then the definition of the helper one gets inlined
   60 too, which when happens in multiple places can create serious GHC Core bloat, making it really hard
   61 to analyze the generated code. Hence in some cases we optimize for lower amounts of produced GHC
   62 Core by turning some helper functions into global ones.
   63 
   64 This doesn't work as well when the helper function captures a variable bound by the enclosing one,
   65 so we leave such helper functions local. We could probably create a global helper and a local
   66 function within it instead, but so far it doesn't appear as those capturing helpers actually get
   67 duplicated in the generated Core.
   68 -}
   69 
   70 -- See Note [Global local functions].
   71 sumCostStreamGo :: CostingInteger -> CostStream -> CostingInteger
   72 sumCostStreamGo !acc (CostLast cost) = acc + cost
   73 sumCostStreamGo !acc (CostCons cost costs) = sumCostStreamGo (acc + cost) costs
   74 
   75 -- | Add up all the costs in a 'CostStream'.
   76 sumCostStream :: CostStream -> CostingInteger
   77 sumCostStream (CostLast cost0) = cost0
   78 sumCostStream (CostCons cost0 costs0) = sumCostStreamGo cost0 costs0
   79 {-# INLINE sumCostStream #-}
   80 
   81 -- See Note [Global local functions].
   82 -- | Map a function over a 'CostStream'.
   83 mapCostStream :: (CostingInteger -> CostingInteger) -> CostStream -> CostStream
   84 -- See Note [Single-element streams]
   85 mapCostStream f (CostLast cost0) = CostLast (f cost0)
   86 mapCostStream f (CostCons cost0 costs0) = CostCons (f cost0) $ go costs0
   87   where
   88     go :: CostStream -> CostStream
   89     go (CostLast cost) = CostLast (f cost)
   90     go (CostCons cost costs) = CostCons (f cost) $ go costs
   91 {-# INLINE mapCostStream #-}
   92 
   93 -- See Note [Global local functions].
   94 addCostStreamGo :: CostStream -> CostStream -> CostStream
   95 addCostStreamGo (CostLast costL) costsR = CostCons costL costsR
   96 addCostStreamGo (CostCons costL costsL) costsR = CostCons costL $ addCostStreamGo costsR costsL
   97 
   98 {-| Add two streams by interleaving their elements (as opposed to draining out one of the streams
   99 before starting to take elements from the other one). No particular reason to prefer
  100 interleaving over draining out one of the streams first. -}
  101 addCostStream :: CostStream -> CostStream -> CostStream
  102 addCostStream costsL0 costsR0 = case (costsL0, costsR0) of
  103   -- See Note [Single-element streams].
  104   (CostLast costL, CostLast costR) -> CostLast $ costL + costR
  105   _ -> addCostStreamGo costsL0 costsR0
  106 {-# INLINE addCostStream #-}
  107 
  108 -- See Note [Global local functions].
  109 -- Didn't attempt to optimize it.
  110 minCostStreamGo :: CostStream -> CostStream -> CostStream
  111 minCostStreamGo costsL costsR =
  112   -- Peel off a cost from each of the streams, if there's any, compare the two costs, emit
  113   -- the minimum cost to the outside and recurse. If the two elements aren't equal, then we put
  114   -- the difference between them back to the stream that had the greatest cost (thus subtracting
  115   -- the minimum cost from the stream -- since we just accounted for it by lazily emitting it to
  116   -- the outside). Proceed until one of the streams is drained out.
  117   let (!costL, !mayCostsL') = unconsCost costsL
  118       (!costR, !mayCostsR') = unconsCost costsR
  119       (!costMin, !mayCostsL'', !mayCostsR'') = case costL `compare` costR of
  120         LT -> (costL, mayCostsL', pure $ reconsCost (costR - costL) mayCostsR')
  121         EQ -> (costL, mayCostsL', mayCostsR')
  122         GT -> (costR, pure $ reconsCost (costL - costR) mayCostsL', mayCostsR')
  123    in reconsCost costMin $ minCostStreamGo <$> mayCostsL'' <*> mayCostsR''
  124 
  125 {-| Calculate the minimum of two 'CostStream's. May return a stream that is longer than either of
  126 the two (but not more than twice). -}
  127 minCostStream :: CostStream -> CostStream -> CostStream
  128 minCostStream costsL0 costsR0 = case (costsL0, costsR0) of
  129   -- See Note [Single-element streams].
  130   (CostLast costL, CostLast costR) -> CostLast $ min costL costR
  131   _ -> minCostStreamGo costsL0 costsR0
  132 {-# INLINE minCostStream #-}
