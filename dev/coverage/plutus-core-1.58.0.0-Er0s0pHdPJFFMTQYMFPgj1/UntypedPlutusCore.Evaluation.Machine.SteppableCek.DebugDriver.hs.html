   34 {- Note [Stepping the driver]
   35 
   36 The client instructs the driver to do a single step of the Cek (`Step` Cmd) or multiple steps until
   37 a condition is met (`Continue`,`Next`,`Finish`). The driver upon receiving a Command (via `inputF`)
   38 fires a series of `stepF` actions to fulfill the command. A `stepF` action is supposed to move the
   39 underlying CEK machine a *single* step forward. Since the driver is written as a free monad, the
   40 interpretation of the `stepF` action is left open.
   41 
   42 When the driver calls `stepF state`, it yields its coroutine with the current driver/cek state
   43 (before the stepping interpretation). The interpreter of `stepF` receives the currentState and
   44 calculates a newstate from it (e.g. **actually stepping the cek machine**). The interpreter then
   45 resumes the driver coroutine with the new state.
   46 
   47 The sensible interpretation to this is the CEK's state transition function (`cekTrans oldState
   48 ===> newState`), but other exotic options may be: logging the before/after states, or even
   49 "mutating" the running program's variables in the environment before/after the state transition
   50 (similar to C debuggers). The interpreter can even supply a different state transition function
   51 (e.g. `id`), but we see little benefit in it at the moment. Currently all interpreters
   52 (brick,repl,testing) just call`cekTrans`.
   53 -}

   55 {-| Leave abstract the types of annotation and breakpoints.
   56 The only thing the driver requires is an inclusion relation of breakpoints into the Annotation -}
   57 class Breakpointable ann bps | ann -> bps where
   58   -- MAYBE: we cannot know which breakpoint fired, return instead `Maybe Breakpoint`?
   59   hasBreakpoints :: ann -> bps -> Bool
   60 
   61 -- | The commands that the driver may receive from the client (tui,cli,test,etc)
   62 data Cmd bps
   63   = {-| Instruct the driver to a *SINGLE* step.
   64     Note: No need to pass breakpoints here because the stepping granularity is *minimal*. -}
   65     Step
   66   | -- | Instruct to multi-step until end-of-program or until breakpoint reached
   67     Continue bps
   68   | -- | Instruct to multi-step over the function call at point or until breakpoint reached
   69     Next bps
   70   | -- | Instruct to multi-step to end of current function or until breakpoint reached
   71     Finish bps
   72   deriving stock (Show, Read)
   73 
   74 -- | The drivers's suspension functor
   75 data DebugF uni fun ann bps a
   76   = -- | Await for the client (e.g. TUI) to tell what to do next (Cmd).
   77     InputF (Cmd bps -> a)
   78   | -- | The debug driver wants to log something
   79     DriverLogF Text a
   80   | {-| An enumeratee of Driver State (generator+iteratee):
   81     Yield a state before doing a step, then await for a state to resume after the step.
   82     See Note [Stepping the driver]. -}
   83     StepF
   84       (CekState uni fun ann)
   85       -- ^ yield with the current driver's state before running a step
   86       (CekState uni fun ann -> a)
   87       {-^ resume back with a state after the step interpretation
   88       | A generator of CekState to yield to client (e.g. TUI) -}
   89   | UpdateClientF (CekState uni fun ann) a
   90   deriving stock (Functor)
   91 
   92 -- | The monad that the driver operates in
   93 type Driving m uni fun ann bps =
   94   ( MonadReader (CekState uni fun ann) m -- the state of the debugger
   95   , MonadFree (DebugF uni fun ann bps) m -- the effects of the driver
   96   , Breakpointable ann bps
   97   )
   98 
   99 -- | Entrypoint of the driver
  100 runDriverT
  101   :: forall uni fun ann bps m
  102    . (Breakpointable ann bps, MonadFree (DebugF uni fun ann bps) m)
  103   => NTerm uni fun ann -> m ()
  104 runDriverT = void . runReaderT driver . initState
  105   where
  106     initState :: NTerm uni fun ann -> CekState uni fun ann
  107     initState = Starting
  108 
  109 -- | The driver action. The driver repeatedly:
  110 
  111 ---
  112 -- 1) waits for a `Cmd`
  113 -- 2) runs one or more CEK steps
  114 -- 3) informs the client for CEK updates&logs
  115 ---
  116 -- The driver computation exits when it reaches a CEK `Terminating` state.
  117 driver :: Driving m uni fun ann bps => m ()
  118 driver =
  119   inputF >>= \case
  120     -- Condition immediately satisfied
  121     Step -> multiStepUntil $ const True
  122     Continue bs -> multiStepUntil $ maybe False (`hasBreakpoints` bs) . cekStateAnn
  123     Next bs -> do
  124       startState <- ask
  125       multiStepUntil $ \curState ->
  126         maybe False (`hasBreakpoints` bs) (cekStateAnn curState)
  127           ||
  128           -- has activation record length been restored?
  129           let leCtx = onCtxLen (<)
  130            in curState `leCtx` startState
  131     Finish bs -> do
  132       startState <- ask
  133       multiStepUntil $ \curState ->
  134         maybe False (`hasBreakpoints` bs) (cekStateAnn curState)
  135           ||
  136           -- has activation record length become smaller?
  137           let ltCtx = onCtxLen (<=)
  138            in curState `ltCtx` startState
  139   where
  140     -- \| Comparison on states' contexts
  141     onCtxLen
  142       :: ( state ~ CekState uni fun ann
  143          , ctxLen ~ Maybe Word -- `Maybe` because ctx can be missing if terminating
  144          )
  145       => (ctxLen -> ctxLen -> a)
  146       -> state
  147       -> state
  148       -> a
  149     onCtxLen = (`on` preview (cekStateContext . to lenContext))
  150 
  151 -- | Do one or more cek steps until Terminating state is reached or condition on 'CekState' is met.
  152 multiStepUntil
  153   :: Driving m uni fun ann bp
  154   => (CekState uni fun ann -> Bool) -> m ()
  155 multiStepUntil cond = do
  156   driverLogF "Driver is going to do a single step"
  157   newState <- stepF =<< ask
  158   case newState of
  159     Terminating {} ->
  160       -- don't recurse to driver, but EXIT the driver
  161       updateClientF newState
  162     _ ->
  163       -- update state
  164       local (const newState) $
  165         if cond newState
  166           then do
  167             updateClientF newState
  168             driver -- tail recurse
  169           else
  170             multiStepUntil cond -- tail recurse
  171 
  172 -- * boilerplate "suspension actions"
  173 
  174 -- Being in 'Driving' monad here is too constraining, but it does not matter.
  175 inputF :: Driving m uni fun ann bps => m (Cmd bps)
  176 inputF = liftF $ InputF id
  177 driverLogF :: Driving m uni fun ann bps => Text -> m ()
  178 driverLogF text = liftF $ DriverLogF text ()
  179 updateClientF :: Driving m uni fun ann bps => CekState uni fun ann -> m ()
  180 updateClientF dState = liftF $ UpdateClientF dState ()
  181 stepF :: Driving m uni fun ann bps => CekState uni fun ann -> m (CekState uni fun ann)
  182 stepF yieldState = liftF $ StepF yieldState id
