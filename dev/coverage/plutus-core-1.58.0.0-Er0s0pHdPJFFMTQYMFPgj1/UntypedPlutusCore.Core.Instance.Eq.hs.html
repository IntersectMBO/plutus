   79 -- | Check equality of two 'Term's.
   80 eqTermM
   81   :: (GEq uni, Closed uni, uni `Everywhere` Eq, Eq fun, Eq ann, HasUnique name TermUnique)
   82   => Term name uni fun ann -> Term name uni fun ann -> EqRename (Renaming TermUnique)
   83 eqTermM (Constant ann1 con1) (Constant ann2 con2) = do
   84   eqM ann1 ann2
   85   eqM con1 con2
   86 eqTermM (Builtin ann1 bi1) (Builtin ann2 bi2) = do
   87   eqM ann1 ann2
   88   eqM bi1 bi2
   89 eqTermM (Var ann1 name1) (Var ann2 name2) = do
   90   eqM ann1 ann2
   91   eqNameM name1 name2
   92 eqTermM (LamAbs ann1 name1 body1) (LamAbs ann2 name2 body2) = do
   93   eqM ann1 ann2
   94   withTwinBindings name1 name2 $ eqTermM body1 body2
   95 eqTermM (Apply ann1 fun1 arg1) (Apply ann2 fun2 arg2) = do
   96   eqM ann1 ann2
   97   eqTermM fun1 fun2
   98   eqTermM arg1 arg2
   99 eqTermM (Delay ann1 term1) (Delay ann2 term2) = do
  100   eqM ann1 ann2
  101   eqTermM term1 term2
  102 eqTermM (Force ann1 term1) (Force ann2 term2) = do
  103   eqM ann1 ann2
  104   eqTermM term1 term2
  105 eqTermM (Error ann1) (Error ann2) = eqM ann1 ann2
  106 eqTermM (Constr ann1 i1 args1) (Constr ann2 i2 args2) = do
  107   eqM ann1 ann2
  108   eqM i1 i2
  109   case zipExact args1 args2 of
  110     Just ps -> for_ ps $ \(t1, t2) -> eqTermM t1 t2
  111     Nothing -> empty
  112 eqTermM (Case ann1 a1 cs1) (Case ann2 a2 cs2) = do
  113   eqM ann1 ann2
  114   eqTermM a1 a2
  115   case zipExact (toList cs1) (toList cs2) of
  116     Just ps -> for_ ps $ \(t1, t2) -> eqTermM t1 t2
  117     Nothing -> empty
  118 eqTermM Constant {} _ = empty
  119 eqTermM Builtin {} _ = empty
  120 eqTermM Var {} _ = empty
  121 eqTermM LamAbs {} _ = empty
  122 eqTermM Apply {} _ = empty
  123 eqTermM Delay {} _ = empty
  124 eqTermM Force {} _ = empty
  125 eqTermM Error {} _ = empty
  126 eqTermM Constr {} _ = empty
  127 eqTermM Case {} _ = empty
