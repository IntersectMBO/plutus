    9 import UntypedPlutusCore.Core.Type as UPLC

   12 import Control.Monad (unless)
   13 import Control.Monad.Except (MonadError, throwError)

   15 {-| A pass to check that the input term:
   16 1) does not contain free variables and
   17 2) that all binders are set to debruijn index 0.
   18 
   19 Feeding the result of the debruijnification to this function is expected to pass.
   20 
   21 On the other hand, because of (2), this pass is
   22 stricter than the undebruijnification's (indirect)
   23 scope-checking, see Note [DeBruijn indices of Binders].
   24 
   25 Inlining this function makes a big difference,
   26 since it will usually be called in a context where all the type variables are known.
   27 That then means that GHC can optimize go locally in a completely monomorphic setting, which helps a lot. -}
   28 checkScope
   29   :: forall m name uni fun a
   30    . (HasIndex name, MonadError FreeVariableError m)
   31   => UPLC.Term name uni fun a
   32   -> m ()
   33 checkScope = go 0
   34   where
   35     -- the current level as a reader value
   36     go :: Word -> UPLC.Term name uni fun a -> m ()
   37     go !lvl = \case
   38       Var _ n -> do
   39         let i = n ^. index
   40         -- var index must be larger than 0
   41         -- var index must be LEQ to the current level
   42         unless (i > 0 && fromIntegral i <= lvl) $
   43           throwError $
   44             FreeIndex i
   45       LamAbs _ binder t -> do
   46         let bIx = binder ^. index
   47         -- binder index must be equal to 0
   48         unless (bIx == 0) $
   49           throwError $
   50             FreeIndex bIx
   51         go (lvl + 1) t
   52       Apply _ t1 t2 -> go lvl t1 >> go lvl t2
   53       Force _ t -> go lvl t
   54       Delay _ t -> go lvl t
   55       _ -> pure ()
   56 {-# INLINE checkScope #-}
