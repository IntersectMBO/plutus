   17 {-|
   18 
   19 > getShadF a = \(rec :: * -> *) (i :: *) -> a -> all (a :: * -> *). a i -> integer -}
   20 getShadF :: uni `HasTypeLevel` Integer => TyName -> Quote (Type TyName uni ())
   21 getShadF a = do
   22   rec <- freshTyName "rec"
   23   i <- freshTyName "i"
   24   return
   25     . TyLam () rec (KindArrow () (Type ()) $ Type ())
   26     . TyLam () i (Type ())
   27     . TyFun () (TyVar () a)
   28     . TyForall () a (KindArrow () (Type ()) $ Type ())
   29     . TyFun () (TyApp () (TyVar () a) $ TyVar () i)
   30     $ mkTyBuiltin @_ @Integer ()
   31 
   32 {-|
   33 
   34 > \(a :: *) -> ifix (getShadF a) a -}
   35 shad :: uni `HasTypeLevel` Integer => Type TyName uni ()
   36 shad = runQuote $ do
   37   a <- freshTyName "a"
   38   shadF <- getShadF a
   39   return
   40     . TyLam () a (Type ())
   41     . TyIFix () shadF
   42     $ TyVar () a

   44 {-| Test that shadowing does not result in variable capture. The definition is as follows:
   45 
   46 > /\(a :: *) -> wrap (getShadF a) a (\(x : a) -> /\(f :: * -> *) -> \(y : f i) -> 0)
   47 
   48 Type checking this term we eventually reach
   49 
   50 > NORM (vPat (\(a :: k) -> ifix vPat a) arg)
   51 
   52 (where in our case @vPat@ is @shadF@ and @arg@ is @a@), which, if we were naive,
   53 would unfold into
   54 
   55 > a -> all (a :: * -> *). a a -> integer
   56 
   57 i.e. we substituted the outer @a@ for @i@, but due to variable capture via @all@ that outer @a@
   58 now became an inner one, which would be a bug.
   59 
   60 But that problem is already solved before type checking starts as we rename the program and that
   61 makes all binders uniques, so no variable capture is possible due to the outer and inner bindings
   62 being distinct. -}
   63 mkShad :: uni `HasTypeAndTermLevel` Integer => Term TyName Name uni fun ()
   64 mkShad = runQuote $ do
   65   a <- freshTyName "a"
   66   f <- freshTyName "f"
   67   shadF <- getShadF a
   68   x <- freshName "x"
   69   y <- freshName "y"
   70   return
   71     . TyAbs () a (Type ())
   72     . IWrap () shadF (TyVar () a)
   73     . LamAbs () x (TyVar () a)
   74     . TyAbs () f (KindArrow () (Type ()) $ Type ())
   75     . LamAbs () y (TyApp () (TyVar () f) $ TyVar () a)
   76     $ mkConstant @Integer () 0
   77 
   78 {-|
   79 
   80 > recUnitF = \(rec :: * -> *) (i :: *) -> all (r :: *). rec i -> r -> r -}
   81 recUnitF :: Type TyName uni ()
   82 recUnitF = runQuote $ do
   83   rec <- freshTyName "rec"
   84   i <- freshTyName "i"
   85   r <- freshTyName "r"
   86   return
   87     . TyLam () rec (KindArrow () (Type ()) $ Type ())
   88     . TyLam () i (Type ())
   89     . TyForall () r (Type ())
   90     . TyFun () (TyApp () (TyVar () rec) $ TyVar () i)
   91     . TyFun () (TyVar () r)
   92     $ TyVar () r
   93 
   94 {-|
   95 
   96 > ifix recUnitF () -}
   97 recUnit :: uni `HasTypeLevel` () => Type TyName uni ()
   98 recUnit = TyIFix () recUnitF $ mkTyBuiltin @_ @() ()
   99 
  100 {-|  Test that a binder in a pattern functor does not get duplicated. The definition is as follows:
  101 
  102 > /\(a :: *) -> \(ru : recUnit) -> unwrap ru {a} ru
  103 
  104 Type checking this term we eventually reach
  105 
  106 > NORM (vPat (\(a :: k) -> ifix vPat a) arg)
  107 
  108 (where in our case @vPat@ is @recUnitF@ and @arg@ is @()@), which, if we were naive,
  109 would unfold into
  110 
  111 > all (r :: *). ifix (\(rec :: * -> *) (i :: *) -> all (r :: *). rec i -> r -> r) () -> r -> r
  112 
  113 and break global uniqueness as the @all (r :: *)@ binder appears twice.
  114 
  115 But this doesn't happen in the actual code, since when a variable gets looked up during type
  116 normalization, its value gets renamed, which means that a fresh variable will be generated for
  117 the inner binder and there will be no shadowing. -}
  118 runRecUnit :: uni `HasTypeAndTermLevel` () => Term TyName Name uni fun ()
  119 runRecUnit = runQuote $ do
  120   a <- freshTyName "a"
  121   ru <- freshName "ru"
  122   return
  123     . TyAbs () a (Type ())
  124     . LamAbs () ru recUnit
  125     . Apply () (TyInst () (Unwrap () $ Var () ru) $ TyVar () a)
  126     $ Var () ru
