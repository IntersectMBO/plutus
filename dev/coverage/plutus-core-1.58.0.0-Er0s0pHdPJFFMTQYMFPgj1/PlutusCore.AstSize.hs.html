   31 {-| Count the number of AST nodes in a kind.
   32 
   33 >>> kindAstSize $ Type ()
   34 AstSize {unAstSize = 1}
   35 >>> kindAstSize $ KindArrow () (KindArrow () (Type ()) (Type ())) (Type ())
   36 AstSize {unAstSize = 5} -}
   37 kindAstSize :: Kind a -> AstSize
   38 kindAstSize kind =
   39   fold
   40     [ AstSize 1
   41     , kind ^. kindSubkinds . to kindAstSize
   42     ]
   43 
   44 -- | Count the number of AST nodes in a type.
   45 typeAstSize :: Type tyname uni ann -> AstSize
   46 typeAstSize ty =
   47   fold
   48     [ AstSize 1
   49     , ty ^. typeSubkinds . to kindAstSize
   50     , ty ^. typeSubtypes . to typeAstSize
   51     ]
   52 
   53 tyVarDeclAstSize :: TyVarDecl tyname ann -> AstSize
   54 tyVarDeclAstSize tyVarDecl =
   55   fold
   56     [ AstSize 1
   57     , tyVarDecl ^. tyVarDeclSubkinds . to kindAstSize
   58     ]
   59 
   60 -- | Count the number of AST nodes in a term.
   61 termAstSize :: Term tyname name uni fun ann -> AstSize
   62 termAstSize term =
   63   fold
   64     [ AstSize 1
   65     , term ^. termSubkinds . to kindAstSize
   66     , term ^. termSubtypes . to typeAstSize
   67     , term ^. termSubterms . to termAstSize
   68     ]
   69 
   70 varDeclAstSize :: VarDecl tyname name uni ann -> AstSize
   71 varDeclAstSize varDecl =
   72   fold
   73     [ AstSize 1
   74     , varDecl ^. varDeclSubtypes . to typeAstSize
   75     ]
   76 
   77 -- | Count the number of AST nodes in a program.
   78 programAstSize :: Program tyname name uni fun ann -> AstSize
   79 programAstSize (Program _ _ t) = termAstSize t
   80 
   81 -- | Compute the size of the serializabled form of a value.
   82 serialisedAstSize :: Flat a => a -> Integer
   83 serialisedAstSize = fromIntegral . BS.length . flat
