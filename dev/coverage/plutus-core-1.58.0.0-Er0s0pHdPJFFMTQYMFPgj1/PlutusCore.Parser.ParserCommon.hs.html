   60 {-| Run an action conditionally based on a predicate on the version.
   61 If we don't know the version then the predicate is assumed to be
   62 false, i.e. we act if we _know_ the predicate is satisfied. -}
   63 whenVersion :: (Version -> Bool) -> Parser () -> Parser ()
   64 whenVersion p act = do
   65   mv <- getVersion
   66   case mv of
   67     Nothing -> pure ()
   68     Just v -> when (p v) act

   70 parse
   71   :: (MonadError ParserErrorBundle m, MonadQuote m)
   72   => Parser a
   73   -> String
   74   -> Text
   75   -> m a
   76 parse p file str = do
   77   let res = fmap toErrorB (runReaderT (evalStateT (runParserT p file str) initial) Nothing)
   78   liftEither =<< liftQuote res

   98 -- This is samething from @Text.Megaparsec.Stream@.
   99 reachOffsetNoLine'
  100   :: forall s
  101    . Stream s
  102   => (Int -> s -> (Tokens s, s))
  103   -- ^ How to split input stream at given offset
  104   -> (forall b. (b -> Token s -> b) -> b -> Tokens s -> b)
  105   -- ^ How to fold over input stream
  106   -> (Token s, Token s)
  107   -- ^ Newline token and tab token
  108   -> (Token s -> Pos)
  109   {-^ Offset to reach
  110   | Increment in column position for a token -}
  111   -> Int
  112   -> PosState s
  113   -- ^ Initial 'PosState' to use
  114   -> PosState s
  115   -- ^ Updated 'PosState'
  116 reachOffsetNoLine'
  117   splitAt'
  118   foldl''
  119   (newlineTok, tabTok)
  120   columnIncrement
  121   o
  122   PosState {..} =
  123     ( PosState
  124         { pstateInput = post
  125         , pstateOffset = max pstateOffset o
  126         , pstateSourcePos = spos
  127         , pstateTabWidth = pstateTabWidth
  128         , pstateLinePrefix = pstateLinePrefix
  129         }
  130     )
  131     where
  132       spos = foldl'' go pstateSourcePos pre
  133       (pre, post) = splitAt' (o - pstateOffset) pstateInput
  134       go (SourcePos n l c) ch =
  135         let c' = unPos c
  136             w = unPos pstateTabWidth
  137          in if
  138               | ch == newlineTok ->
  139                   SourcePos n (l <> pos1) pos1
  140               | ch == tabTok ->
  141                   SourcePos n l (mkPos $ c' + w - ((c' - 1) `rem` w))
  142               | otherwise ->
  143                   SourcePos n l (c <> columnIncrement ch)
  144 {-# INLINE reachOffsetNoLine' #-}
  145 
  146 getSourcePos' :: MonadParsec e Text m => m SourcePos
  147 getSourcePos' = do
  148   st <- getParserState
  149   let
  150     pst =
  151       reachOffsetNoLine'
  152         Text.splitAt
  153         Text.foldl'
  154         ('\n', '\t')
  155         (const pos1)
  156         (stateOffset st)
  157         (statePosState st)
  158   setParserState st {statePosState = pst}
  159   return (pstateSourcePos pst)

  161 {-| Returns a parser for @a@ by calling the supplied function on the starting
  162 and ending positions of @a@.
  163 
  164 The supplied function should usually return a parser that does /not/ consume trailing
  165 whitespaces. Otherwise, the end position will be the first character after the
  166 trailing whitespaces. -}
  167 withSpan' :: (SrcSpan -> Parser a) -> Parser a
  168 withSpan' f = mdo
  169   start <- getSourcePos'
  170   res <- f sp
  171   end <- getSourcePos'
  172   let sp = toSrcSpan start end
  173   pure res
  174 
  175 {-| Like `withSpan'`, but the result parser consumes whitespaces.
  176 
  177 @withSpan = (<* whitespace) . withSpan'@ -}
  178 withSpan :: (SrcSpan -> Parser a) -> Parser a
  179 withSpan = (<* whitespace) . withSpan'

  196 toSrcSpan :: SourcePos -> SourcePos -> SrcSpan
  197 toSrcSpan start end =
  198   SrcSpan
  199     { srcSpanFile = sourceName start
  200     , srcSpanSLine = unPos (sourceLine start)
  201     , srcSpanSCol = unPos (sourceColumn start)
  202     , srcSpanELine = unPos (sourceLine end)
  203     , srcSpanECol = unPos (sourceColumn end)
  204     }

  214 -- | Parses a `Name`. Does not consume leading or trailing whitespaces.
  215 name :: Parser Name
  216 name = try $ parseUnquoted <|> parseQuoted
  217   where
  218     parseUnquoted :: Parser Name
  219     parseUnquoted = do
  220       _ <- lookAhead (satisfy isIdentifierStartingChar)
  221       str <- takeWhileP (Just "identifier-unquoted") isIdentifierChar
  222       Name str <$> uniqueSuffix str
  223 
  224     parseQuoted :: Parser Name
  225     parseQuoted = do
  226       _ <- char '`'
  227       _ <- lookAhead (satisfy isQuotedIdentifierChar)
  228       str <- takeWhileP (Just "identifier-quoted") isQuotedIdentifierChar
  229       _ <- char '`'
  230       Name str <$> uniqueSuffix str
  231 
  232     -- Tries to parse a `Unique` value.
  233     -- If it fails then looks up the `Unique` value for the given name.
  234     -- If lookup fails too then generates a fresh `Unique` value.
  235     uniqueSuffix :: Text -> Parser Unique
  236     uniqueSuffix nameStr = try (Unique <$> (char '-' *> Lex.decimal)) <|> uniqueForName nameStr
  237 
  238     -- Return the unique identifier of a name.
  239     -- If it's not in the current parser state, map the name to a fresh id and add it to the state.
  240     uniqueForName :: (MonadState ParserState m, MonadQuote m) => Text -> m Unique
  241     uniqueForName nameStr = do
  242       parserState <- get
  243       case M.lookup nameStr (identifiers parserState) of
  244         Just u -> pure u
  245         Nothing -> do
  246           fresh <- freshUnique
  247           put $ ParserState $ M.insert nameStr fresh $ identifiers parserState
  248           pure fresh
