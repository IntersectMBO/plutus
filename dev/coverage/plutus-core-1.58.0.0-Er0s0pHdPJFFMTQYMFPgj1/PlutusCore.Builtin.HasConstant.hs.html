   23 {- Note [Existence of HasConstant]
   24 We don't really need 'HasConstant' and could get away with only having 'HasConstantIn', however
   25 defining the latter directly as a @class@ instead of a type synonym in terms of the former is
   26 detrimental to performance, see the comments in https://github.com/IntersectMBO/plutus/pull/4417
   27 
   28 This is likely due to the same reason as in 'mkMachineParameters',
   29 see Note [The CostingPart constraint in mkMachineParameters].
   30 -}

   32 -- See Note [Existence of HasConstant].
   33 -- We name it @term@ rather than @val@, because various @Term@ (UPLC/TPLC/PIR) data types have
   34 -- 'Constant'-like constructors too and we lift to / unlift from those in tests.
   35 -- | Ensures that @term@ has a 'Constant'-like constructor to lift values to and unlift values from.
   36 class HasConstant term where
   37   -- Switching from 'MonadError' to 'Either' here gave us a speedup of 2-4%.
   38   {-| Unwrap from a 'Constant'-like constructor throwing an 'UnliftingError' if the provided
   39   @term@ is not a wrapped Haskell value. -}
   40   asConstant :: term -> Either BuiltinError (Some (ValueOf (UniOf term)))
   41 
   42   -- | Wrap a Haskell value as a @term@.
   43   fromConstant :: Some (ValueOf (UniOf term)) -> term
   44 
   45 {-| Ensures that @term@ has a 'Constant'-like constructor to lift values to and unlift values from
   46 and connects @term@ and its @uni@. -}
   47 type HasConstantIn uni term = (UniOf term ~ uni, HasConstant term)
   48 
   49 -- | Wrap a Haskell value (given its explicit type tag) as a @term@.
   50 fromValueOf :: forall a term. HasConstant term => UniOf term (Esc a) -> a -> term
   51 fromValueOf uni = fromConstant . someValueOf uni
   52 {-# INLINE fromValueOf #-}
   53 
   54 -- | Wrap a Haskell value (provided its type is in the universe) as a @term@.
   55 fromValue :: forall a term. (HasConstant term, UniOf term `HasTermLevel` a) => a -> term
   56 fromValue = fromValueOf knownUni
   57 {-# INLINE fromValue #-}
   58 
   59 instance HasConstant (Term TyName Name uni fun ()) where
   60   asConstant (Constant _ val) = pure val
   61   asConstant _ = throwError notAConstant
   62 
   63   fromConstant = Constant ()
