    2  This approach is preferred when it is more efficient to compare the associated
    3  'Unique's instead of the underlying type. -}

   29 {-| A set containing 'Unique's. Since 'Unique' is equivalent to 'Int'
   30  (see "PlutusCore.Name.Unique"), we can use an 'IntSet' representation for this type. -}
   31 newtype UniqueSet unique = UniqueSet
   32   { unUniqueSet :: IS.IntSet
   33   }
   34   deriving stock (Show, Eq)
   35   deriving newtype (Semigroup, Monoid)
   36 
   37 -- | Insert a @unique@.
   38 insertByUnique
   39   :: Coercible unique Unique
   40   => unique
   41   -> UniqueSet unique
   42   -> UniqueSet unique
   43 insertByUnique = coerce . IS.insert . coerce
   44 
   45 -- | Insert the @unique@ associated to the @name@.
   46 insertByName :: HasUnique name unique => name -> UniqueSet unique -> UniqueSet unique
   47 insertByName = insertByUnique . view unique
   48 
   49 -- | Create the singleton set of the @unique@ associated to the @name@.
   50 singletonName :: HasUnique name unique => name -> UniqueSet unique
   51 singletonName n = insertByName n mempty
   52 
   53 -- | Convert a 'Foldable' into a 'UniqueSet' using the given insertion function.
   54 fromFoldable
   55   :: Foldable f
   56   => (i -> UniqueSet unique -> UniqueSet unique)
   57   -> f i
   58   -> UniqueSet unique
   59 fromFoldable ins = List.foldl' (flip ins) mempty
   60 
   61 -- | Convert a 'Foldable' with uniques into a 'UniqueSet'.
   62 fromUniques :: Foldable f => Coercible Unique unique => f unique -> UniqueSet unique
   63 fromUniques = fromFoldable insertByUnique
   64 
   65 -- | Convert a 'Foldable' with names into a 'UniqueSet'.
   66 fromNames :: Foldable f => HasUnique name unique => f name -> UniqueSet unique
   67 fromNames = fromFoldable insertByName
   68 
   69 -- | Is the @unique@ a member of the set?
   70 memberByUnique :: Coercible unique Unique => unique -> UniqueSet unique -> Bool
   71 memberByUnique uniq = IS.member (coerce uniq) . unUniqueSet
   72 
   73 -- | Is the @name@ associated to the @unique@ a member of the set?
   74 memberByName :: HasUnique name unique => name -> UniqueSet unique -> Bool
   75 memberByName = memberByUnique . view unique
   76 
   77 {-| The negation of 'memberByName', useful for converting to operator form,
   78  e.g. @name `notMemberByName` set@. -}
   79 notMemberByName :: HasUnique name unique => name -> UniqueSet unique -> Bool
   80 notMemberByName n = not . memberByName n
   81 
   82 -- | The difference of two 'UniqueSet's.
   83 (\\) :: UniqueSet unique -> UniqueSet unique -> UniqueSet unique
   84 (\\) (UniqueSet s1) (UniqueSet s2) = UniqueSet $ s1 IS.\\ s2
   85 
   86 -- | The union of two 'UniqueSet's.
   87 union :: UniqueSet unique -> UniqueSet unique -> UniqueSet unique
   88 union (UniqueSet s1) (UniqueSet s2) = UniqueSet $ s1 `IS.union` s2
   89 
   90 -- | Build a set of @unique@s from the 'Getting'.
   91 setOfByUnique
   92   :: Coercible unique Unique
   93   => Getting (UniqueSet unique) s unique
   94   -> s
   95   -> UniqueSet unique
   96 setOfByUnique g = UniqueSet <$> IS.setOf (coerce g)
   97 
   98 -- | Build a set of @unique@s associated to the names in the 'Getting'.
   99 setOfByName
  100   :: HasUnique name unique
  101   => Getting (UniqueSet unique) s name
  102   -> s
  103   -> UniqueSet unique
  104 setOfByName l = views l singletonName
