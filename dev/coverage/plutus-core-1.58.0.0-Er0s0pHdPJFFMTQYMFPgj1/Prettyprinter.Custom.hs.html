   14 {-| An area bracketed by two delimiters. When on multiple lines the delimiters are not
   15 indented but the content is. -}
   16 section' :: Doc a -> Doc a -> Doc a -> Doc a
   17 -- The subtlety here is that the nest call surrounds the first delimiter and the content, but not
   18 -- the final one. This is because of how nest behaves, where it doesn't indent until it hits
   19 -- the first line break. So we need to include the first delimiter so that the main content gets
   20 -- indented, but not the final delimiter.
   21 section' c1 c2 d = group $ nest 2 (c1 <> d) <> c2
   22 
   23 {-| An area bracketed by two delimiters. When on one line, there are spaces between the delimiters
   24 and the content, when on multiple lines the delimiters are not indented but the content is. -}
   25 section :: Doc a -> Doc a -> Doc a -> Doc a
   26 section c1 c2 = section' (c1 <> line) (line <> c2)
   27 
   28 {-| This prints a document enclosed by brackets, possibly indenting the output on
   29 a new line if it does not fit. -}
   30 brackets' :: Doc a -> Doc a
   31 brackets' = section "[" "]"
   32 
   33 {-| This prints a document enclosed by braces, possibly indenting the output on
   34 a new line if it does not fit. -}
   35 braces' :: Doc a -> Doc a
   36 braces' = section "{" "}"
   37 
   38 {-| This prints a document enclosed by parentheses, aligning the opening and
   39 closing parentheses. -}
   40 parens' :: Doc a -> Doc a
   41 parens' = section "(" ")"

   43 -- | Print a "sexp", i.e. something like "(keyword arg1 ... argN)".
   44 sexp :: Doc a -> [Doc a] -> Doc a
   45 sexp a es =
   46   -- This is a bit funny, because we want the keyword to "stick" to the opening parenthesis
   47   -- when it's split over multiple lines. So we include it in the "initial" segment. But then
   48   -- we also have to have a space after that rather than no space. So we start with "(keyword"
   49   -- and a line-or-space, but end with a line-or-nothing and ")".
   50   -- However if @es@ is empty, then we don't want to insert a space right before @)@, so in that
   51   -- case we use line-or-nothing as well.
   52   section' ("(" <> a <> if null es then line' else line) (line' <> ")") (sep es)

   54 {-| Lay out a sequence of documents vertically with forced lines between documents. Useful
   55 for prettyprinting layout-sensitive things like let-bindings. -}
   56 vcatHard :: [Doc ann] -> Doc ann
   57 vcatHard = concatWith (\x y -> x <> hardline <> y)
   58 
   59 {-| Separate two documents `p` and `q` and increase indentation if `q` has to be put on a new
   60 line. Useful to e.g. pretty-print function application like `fun <?> sep arguments`. -}
   61 (<?>) :: Doc ann -> Doc ann -> Doc ann
   62 p <?> q = align . nest 2 $ sep [p, q]
