   38 -- | The monad the renamer runs in.
   39 newtype RenameT ren m a = RenameT
   40   { unRenameT :: ReaderT ren m a
   41   }
   42   deriving newtype
   43     ( Functor
   44     , Applicative
   45     , Alternative
   46     , Monad
   47     , MonadReader ren
   48     , MonadQuote
   49     )
   50 
   51 -- | A renaming is a mapping from old uniques to new ones.
   52 newtype Renaming unique = Renaming
   53   { unRenaming :: UniqueMap unique unique
   54   }
   55   deriving newtype (Semigroup, Monoid)
   56 
   57 {-| A type-level renaming.
   58 Needed for instantiating functions running over types in generic @RenameT ren m@ to
   59 a particular type of renaming. -}
   60 type TypeRenaming = Renaming TypeUnique
   61 
   62 {-| A class that specifies which 'Renaming' a @ren@ has inside.
   63 A @ren@ can contain several 'Renaming's (like 'Scoped', for example). -}
   64 class Coercible unique Unique => HasRenaming ren unique where
   65   renaming :: Lens' ren (Renaming unique)
   66 
   67 -- | Scoping-aware mapping from locally unique uniques to globally unique uniques.
   68 data ScopedRenaming = ScopedRenaming
   69   { _scopedRenamingTypes :: Renaming TypeUnique
   70   , _scopedRenamingTerms :: Renaming TermUnique
   71   }
   72 
   73 makeLenses ''ScopedRenaming
   74 
   75 type ScopedRenameT = RenameT ScopedRenaming
   76 
   77 instance Semigroup ScopedRenaming where
   78   ScopedRenaming types1 terms1 <> ScopedRenaming types2 terms2 =
   79     ScopedRenaming (types1 <> types2) (terms1 <> terms2)
   80 
   81 instance Monoid ScopedRenaming where
   82   mempty = ScopedRenaming mempty mempty
   83 
   84 instance
   85   (Coercible unique1 Unique, unique1 ~ unique2)
   86   => HasRenaming (Renaming unique1) unique2
   87   where
   88   renaming = id
   89 
   90 instance HasRenaming ScopedRenaming TypeUnique where
   91   renaming = scopedRenamingTypes . renaming
   92 
   93 instance HasRenaming ScopedRenaming TermUnique where
   94   renaming = scopedRenamingTerms . renaming
   95 
   96 -- | Run a 'RenameT' computation with an empty renaming.
   97 runRenameT :: Monoid ren => RenameT ren m a -> m a
   98 runRenameT (RenameT a) = runReaderT a mempty
   99 
  100 -- | Map the underlying representation of 'Renaming'.
  101 mapRenaming
  102   :: (UniqueMap unique unique -> UniqueMap unique unique)
  103   -> Renaming unique
  104   -> Renaming unique
  105 mapRenaming = coerce
  106 
  107 -- | Save the mapping from the @unique@ of a name to a new @unique@.
  108 insertByNameM
  109   :: (HasUnique name unique, HasRenaming ren unique)
  110   => name -> unique -> ren -> ren
  111 insertByNameM name = over renaming . mapRenaming . insertByName name
  112 
  113 -- | Look up the new unique a name got mapped to.
  114 lookupNameM
  115   :: (HasUnique name unique, HasRenaming ren unique, MonadReader ren m)
  116   => name -> m (Maybe unique)
  117 lookupNameM name = asks $ lookupName name . unRenaming . view renaming
  118 
  119 -- | Rename a name that has a unique inside.
  120 renameNameM
  121   :: (HasRenaming ren unique, HasUnique name unique, MonadReader ren m)
  122   => name -> m name
  123 renameNameM name = do
  124   mayUniqNew <- lookupNameM name
  125   pure $ case mayUniqNew of
  126     Nothing -> name
  127     Just uniqNew -> name & unique .~ uniqNew
  128 
  129 {-| Replace the unique in a name by a new unique, save the mapping
  130 from the old unique to the new one and supply the updated value to a continuation. -}
  131 withFreshenedName
  132   :: (HasRenaming ren unique, HasUnique name unique, MonadQuote m, MonadReader ren m)
  133   => name -> (name -> m c) -> m c
  134 withFreshenedName nameOld k = do
  135   uniqNew <- coerce <$> freshUnique
  136   local (insertByNameM nameOld uniqNew) $ k (nameOld & unique .~ uniqNew)
  137 
  138 {-| Run a 'RenameT' computation in the environment extended by the mapping from an old name
  139 to a new one. -}
  140 withRenamedName
  141   :: (HasRenaming ren unique, HasUnique name unique, MonadReader ren m)
  142   => name -> name -> m c -> m c
  143 withRenamedName old new = local $ insertByNameM old (new ^. unique)
