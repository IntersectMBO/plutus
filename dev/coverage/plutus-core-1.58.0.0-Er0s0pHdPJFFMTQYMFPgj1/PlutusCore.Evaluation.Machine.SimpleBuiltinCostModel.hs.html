    6 {-| A program to parse a JSON representation of costing functions for Plutus Core
    7    builtins and and produce a simple cost model which can be used from Agda and other
    8    executables -}
    9 module PlutusCore.Evaluation.Machine.SimpleBuiltinCostModel
   10   ( BuiltinCostMap
   11   , BuiltinCostKeyMap
   12   , toSimpleBuiltinCostModel
   13   , defaultSimpleBuiltinCostModel
   14   ) where
   15 
   16 import Data.Aeson.Key as Key (toText)
   17 import Data.Aeson.KeyMap qualified as KeyMap
   18 import Data.Aeson.THReader (readJSONFromFile)
   19 import Data.Bifunctor
   20 import Data.Text (Text, replace)
   21 import PlutusCore.DataFilePaths qualified as DFP
   22 import PlutusCore.Evaluation.Machine.CostingFun.SimpleJSON
   23 
   24 type BuiltinCostMap = [(Text, CpuAndMemoryModel)]
   25 type BuiltinCostKeyMap = KeyMap.KeyMap CpuAndMemoryModel
   26 
   27 {-| The default builtin cost map.
   28 TODO: maybe we should take account of the semantic variant here. -}
   29 defaultBuiltinCostKeyMap :: BuiltinCostKeyMap
   30 defaultBuiltinCostKeyMap =
   31   $$(readJSONFromFile DFP.latestBuiltinCostModelFile)
   32 
   33 -- replace underscores _ by dashes -
   34 builtinName :: Text -> Text
   35 builtinName = replace "_" "-"
   36 
   37 toSimpleBuiltinCostModel :: BuiltinCostKeyMap -> BuiltinCostMap
   38 toSimpleBuiltinCostModel m = map (first (builtinName . toText)) (KeyMap.toList m)
   39 
   40 defaultSimpleBuiltinCostModel :: BuiltinCostMap
   41 defaultSimpleBuiltinCostModel = toSimpleBuiltinCostModel defaultBuiltinCostKeyMap
