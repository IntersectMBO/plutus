   18 {-| Monomorphic @map@ over built-in lists.
   19 
   20 /\(a :: *) -> \(f : a -> a) ->
   21     fix {list a} {list a} \(rec : list a -> list a) (xs : list a) ->
   22         matchList {a} xs {list a} xs \(x : a) (xs' : list a) -> cons {a} (f x) (rec xs') -}
   23 omapList :: MatchOption -> Term TyName Name DefaultUni (Either DefaultFun ExtensionFun) ()
   24 omapList optMatch = runQuote $ do
   25   a <- freshTyName "a"
   26   f <- freshName "f"
   27   rec <- freshName "rec"
   28   xs <- freshName "xs"
   29   x <- freshName "x"
   30   xs' <- freshName "xs'"
   31   let listA = TyApp () list $ TyVar () a
   32       unwrap' ann = apply ann . tyInst () (mapFun Left (matchList optMatch)) $ TyVar () a
   33   return
   34     . tyAbs () a (Type ())
   35     . lamAbs () f (TyFun () (TyVar () a) $ TyVar () a)
   36     . apply () (mkIterInstNoAnn fix [listA, listA])
   37     . lamAbs () rec (TyFun () listA listA)
   38     . lamAbs () xs listA
   39     . apply () (apply () (tyInst () (unwrap' () (var () xs)) listA) $ var () xs)
   40     . lamAbs () x (TyVar () a)
   41     . lamAbs () xs' listA
   42     $ mkIterAppNoAnn
   43       (tyInst () (builtin () $ Left MkCons) $ TyVar () a)
   44       [ apply () (var () f) $ var () x
   45       , apply () (var () rec) $ var () xs'
