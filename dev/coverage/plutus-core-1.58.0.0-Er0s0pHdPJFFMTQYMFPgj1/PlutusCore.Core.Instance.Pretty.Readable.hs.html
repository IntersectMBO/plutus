   21 -- | Split an iterated 'KindArrow' (if any) into the list of argument types and the resulting type.
   22 viewKindArrow :: Kind ann -> Maybe ([Kind ann], Kind ann)
   23 viewKindArrow kind0@KindArrow {} = Just $ go kind0
   24   where
   25     go (KindArrow _ dom cod) = first (dom :) $ go cod
   26     go kind = ([], kind)
   27 viewKindArrow _ = Nothing
   28

   72 -- | Split an iterated 'TyApp' (if any) into the head of the application and the spine.
   73 viewTyApp
   74   :: Type tyname uni ann -> Maybe (Type tyname uni ann, [Type tyname uni ann])
   75 viewTyApp ty0 = go ty0 []
   76   where
   77     go (TyApp _ fun arg) args = go fun $ arg : args
   78     go _ [] = Nothing
   79     go fun args = Just (fun, args)
   80 
   81 -- | Split an iterated 'Apply'/'TyInst' (if any) into the head of the application and the spine.
   82 viewApp
   83   :: Term tyname name uni fun ann
   84   -> Maybe
   85        ( Term tyname name uni fun ann
   86        , [Either (Type tyname uni ann) (Term tyname name uni fun ann)]
   87        )
   88 viewApp term0 = go term0 []
   89   where
   90     go (Apply _ fun argTerm) args = go fun $ Right argTerm : args
   91     go (TyInst _ fun argTy) args = go fun $ Left argTy : args
   92     go _ [] = Nothing
   93     go fun args = Just (fun, args)

   95 instance
   96   PrettyReadableBy configName tyname
   97   => PrettyBy (PrettyConfigReadable configName) (TyVarDecl tyname ann)
   98   where
   99   prettyBy = inContextM $ \case
  100     TyVarDecl _ x k -> do
  101       showKinds <- view $ prettyConfig . pcrShowKinds
  102       withPrettyAt ToTheRight botFixity $ \prettyBot -> do
  103         case showKinds of
  104           ShowKindsYes -> encloseM binderFixity $ (prettyBot x <+> "::") <?> prettyBot k
  105           ShowKindsNonType -> case k of
  106             Type {} -> pure $ prettyBot x
  107             _ -> encloseM binderFixity $ (prettyBot x <+> "::") <?> prettyBot k
  108           ShowKindsNo -> pure $ prettyBot x

  110 instance
  111   ( PrettyReadableBy configName tyname
  112   , PrettyReadableBy configName name
  113   , PrettyUni uni
  114   )
  115   => PrettyBy (PrettyConfigReadable configName) (VarDecl tyname name uni ann)
  116   where
  117   prettyBy = inContextM $ \case
  118     VarDecl _ x t -> do
  119       withPrettyAt ToTheRight botFixity $ \prettyBot -> do
  120         encloseM binderFixity $ (prettyBot x <+> ":") <?> prettyBot t

  122 instance PrettyBy (PrettyConfigReadable configName) (Kind a) where
  123   prettyBy = inContextM $ \case
  124     (viewKindArrow -> Just (args, res)) -> iterArrowPrettyM args res
  125     KindArrow {} -> error "Panic: 'KindArrow' is not covered by 'viewKindArrow'"
  126     Type {} -> "*"
  127 
  128 instance
  129   (PrettyReadableBy configName tyname, PrettyParens (SomeTypeIn uni))
  130   => PrettyBy (PrettyConfigReadable configName) (Type tyname uni a)
  131   where
  132   prettyBy = inContextM $ \case
  133     (viewTyApp -> Just (fun, args)) -> iterAppPrettyM fun args
  134     TyApp {} -> error "Panic: 'TyApp' is not covered by 'viewTyApp'"
  135     TyVar _ name -> prettyM name
  136     (viewTyFun -> Just (args, res)) -> iterArrowPrettyM args res
  137     TyFun {} -> error "Panic: 'TyFun' is not covered by 'viewTyFun'"
  138     TyIFix _ pat arg -> iterAppDocM $ \_ prettyArg -> "ifix" :| map prettyArg [pat, arg]
  139     (viewTyForall -> Just (args, body)) -> iterTyForallPrettyM args body
  140     TyForall {} -> error "Panic: 'TyForall' is not covered by 'viewTyForall'"
  141     TyBuiltin _ someUni -> lmap _pcrRenderContext $ prettyM someUni
  142     (viewTyLam -> Just (args, body)) -> iterLamAbsPrettyM args body
  143     TyLam {} -> error "Panic: 'TyLam' is not covered by 'viewTyLam'"
  144     TySOP _ tls -> iterAppDocM $ \_ prettyArg -> "sop" :| fmap prettyArg tls
  145 
  146 instance
  147   ( PrettyReadableBy configName tyname
  148   , PrettyReadableBy configName name
  149   , PrettyUni uni
  150   , Pretty fun
  151   )
  152   => PrettyBy (PrettyConfigReadable configName) (Term tyname name uni fun a)
  153   where
  154   prettyBy = inContextM $ \case
  155     Constant _ con -> lmap (ConstConfig . _pcrRenderContext) $ prettyM con
  156     Builtin _ bi -> unitDocM $ pretty bi
  157     (viewApp -> Just (fun, args)) -> iterInterAppPrettyM fun args
  158     Apply {} -> error "Panic: 'Apply' is not covered by 'viewApp'"
  159     TyInst {} -> error "Panic: 'TyInst' is not covered by 'viewApp'"
  160     Var _ name -> prettyM name
  161     (viewTyAbs -> Just (args, body)) -> iterTyAbsPrettyM args body
  162     TyAbs {} -> error "Panic: 'TyAbs' is not covered by 'viewTyAbs'"
  163     (viewLamAbs -> Just (args, body)) -> iterLamAbsPrettyM args body
  164     LamAbs {} -> error "Panic: 'LamAbs' is not covered by 'viewLamAbs'"
  165     Unwrap _ term -> iterAppDocM $ \_ prettyArg -> "unwrap" :| [prettyArg term]
  166     IWrap _ pat arg term ->
  167       iterAppDocM $ \_ prettyArg ->
  168         "iwrap" :| [prettyArg pat, prettyArg arg, prettyArg term]
  169     Error _ ty -> iterAppDocM $ \_ prettyArg -> "error" :| [prettyArg $ inBraces ty]
  170     Constr _ ty i es ->
  171       iterAppDocM $ \_ prettyArg -> "constr" :| [prettyArg ty, prettyArg i, prettyArg es]
  172     Case _ ty arg cs ->
  173       iterAppDocM $ \_ prettyArg -> "case" :| [prettyArg ty, prettyArg arg, prettyArg cs]
  174 
  175 instance
  176   PrettyReadableBy configName (Term tyname name uni fun a)
  177   => PrettyBy (PrettyConfigReadable configName) (Program tyname name uni fun a)
  178   where
  179   prettyBy = inContextM $ \(Program _ version term) ->
  180     iterAppDocM $ \_ prettyArg -> "program" :| [pretty version, prettyArg term]
