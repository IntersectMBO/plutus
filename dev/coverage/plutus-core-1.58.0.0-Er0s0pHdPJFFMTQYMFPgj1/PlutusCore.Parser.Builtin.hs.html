   32 cachedBuiltin :: Map.Map T.Text DefaultFun
   33 cachedBuiltin = Map.fromList [(display fn, fn) | fn <- [minBound .. maxBound]]

   35 -- | Parser for builtin functions. Atm the parser can only parse `DefaultFun`.
   36 builtinFunction :: Parser DefaultFun
   37 builtinFunction = lexeme $ do
   38   txt <- takeWhileP (Just "builtin function identifier") isIdentifierChar
   39   case Map.lookup txt cachedBuiltin of
   40     Nothing -> do
   41       let lBuiltin = fmap fst $ Map.toList cachedBuiltin
   42       pos <- getSourcePos
   43       customFailure $ UnknownBuiltinFunction txt pos lBuiltin
   44     Just builtin -> pure builtin

   46 -- | Parser for integer constants.
   47 conInteger :: Parser Integer
   48 conInteger = Lex.signed whitespace (lexeme Lex.decimal)
   49 
   50 -- | Parser for a pair of hex digits to a Word8.
   51 hexByte :: Parser Word8
   52 hexByte = do
   53   high <- hexDigitChar
   54   low <- hexDigitChar
   55   pure $ fromIntegral (hexDigitToInt high * 16 + hexDigitToInt low)
   56 
   57 -- | Parser for bytestring constants. They start with "#".
   58 conBS :: Parser ByteString
   59 conBS = lexeme . fmap pack $ char '#' *> many hexByte
   60 
   61 {-| Parser for string constants (wrapped in double quotes).  Note that
   62  Data.Text.pack "performs replacement on invalid scalar values", which means
   63  that Unicode surrogate code points (corresponding to integers in the range
   64  0xD800-0xDFFF) are converted to the Unicode replacement character U+FFFD
   65  (decimal 65533).  Thus `(con string "X\xD800Z")` parses to a `Text` object
   66  whose second character is U+FFFD. -}
   67 conText :: Parser T.Text
   68 conText = lexeme . fmap T.pack $ char '\"' *> manyTill Lex.charLiteral (char '\"')
   69 
   70 -- | Parser for unit.
   71 conUnit :: Parser ()
   72 conUnit = void (symbol "(" *> symbol ")")
   73 
   74 -- | Parser for bool.
   75 conBool :: Parser Bool
   76 conBool =
   77   choice
   78     [ True <$ symbol "True"
   79     , False <$ symbol "False"
   80     ]

   82 -- | Parser for lists.
   83 conList :: DefaultUni (Esc a) -> Parser [a]
   84 conList uniA =
   85   trailingWhitespace . inBrackets $
   86     constantOf ExpectParensNo uniA `sepBy` symbol ","
   87 
   88 -- | Parser for arrays.
   89 conArray :: DefaultUni (Esc a) -> Parser (Vector a)
   90 conArray uniA = Vector.fromList <$> conList uniA
   91 
   92 -- | Parser for values.
   93 conValue :: Parser PLC.Value
   94 conValue = do
   95   keys <- traverse validateKeys =<< conList knownUni
   96   case Value.fromList keys of
   97     PlutusCore.Builtin.Result.BuiltinSuccess v -> pure v
   98     PlutusCore.Builtin.Result.BuiltinSuccessWithLogs _logs v -> pure v
   99     PlutusCore.Builtin.Result.BuiltinFailure logs _trace ->
  100       fail $ "Failed to construct Value: " <> show logs
  101   where
  102     validateToken (token, amt) = do
  103       tk <-
  104         maybe
  105           (fail $ "Token name exceeds maximum length of 32 bytes: " <> show (unpack token))
  106           pure
  107           (Value.k token)
  108       qty <-
  109         maybe
  110           (fail $ "Token quantity out of signed 128-bit integer bounds: " <> show amt)
  111           pure
  112           (Value.quantity amt)
  113       pure (tk, qty)
  114     validateKeys (currency, tokens) = do
  115       ck <-
  116         maybe
  117           (fail $ "Currency symbol exceeds maximum length of 32 bytes: " <> show (unpack currency))
  118           pure
  119           (Value.k currency)
  120       tks <- traverse validateToken tokens
  121       pure (ck, tks)

  123 -- | Parser for pairs.
  124 conPair :: DefaultUni (Esc a) -> DefaultUni (Esc b) -> Parser (a, b)
  125 conPair uniA uniB = trailingWhitespace . inParens $ do
  126   a <- constantOf ExpectParensNo uniA
  127   _ <- symbol ","
  128   b <- constantOf ExpectParensNo uniB
  129   pure (a, b)

  131 conDataNoParens :: Parser Data
  132 conDataNoParens =
  133   choice
  134     [ symbol "Constr" *> (Constr <$> conInteger <*> conList knownUni)
  135     , symbol "Map" *> (Map <$> conList knownUni)
  136     , symbol "List" *> (List <$> conList knownUni)
  137     , symbol "I" *> (I <$> conInteger)
  138     , symbol "B" *> (B <$> conBS)
  139     ]
  140 
  141 conData :: ExpectParens -> Parser Data
  142 conData ExpectParensYes = trailingWhitespace $ inParens conDataNoParens
  143 conData ExpectParensNo = conDataNoParens

  145 -- Serialised BLS12_381 elements are "0x" followed by a hex string of even

  147 con0xBS :: Parser ByteString
  148 con0xBS = lexeme . fmap pack $ string "0x" *> many hexByte

  150 conBLS12_381_G1_Element :: Parser BLS12_381.G1.Element
  151 conBLS12_381_G1_Element = do
  152   s <- con0xBS
  153   case BLS12_381.G1.uncompress s of
  154     Left err -> fail $ "Failed to decode value of type bls12_381_G1_element: " ++ show err
  155     Right e -> pure e

  164 -- | Parser for constants of the given type.
  165 constantOf :: ExpectParens -> DefaultUni (Esc a) -> Parser a
  166 constantOf expectParens uni =
  167   case uni of
  168     DefaultUniInteger -> conInteger
  169     DefaultUniByteString -> conBS
  170     DefaultUniString -> conText
  171     DefaultUniUnit -> conUnit
  172     DefaultUniBool -> conBool
  173     DefaultUniValue -> conValue
  174     DefaultUniProtoList `DefaultUniApply` uniA -> conList uniA
  175     DefaultUniProtoArray `DefaultUniApply` uniA -> conArray uniA
  176     DefaultUniProtoPair `DefaultUniApply` uniA `DefaultUniApply` uniB -> conPair uniA uniB
  177     f `DefaultUniApply` _ `DefaultUniApply` _ `DefaultUniApply` _ -> noMoreTypeFunctions f
  178     DefaultUniData -> conData expectParens
  179     DefaultUniBLS12_381_G1_Element -> conBLS12_381_G1_Element
  180     DefaultUniBLS12_381_G2_Element -> conBLS12_381_G2_Element
  181     DefaultUniBLS12_381_MlResult ->
  182       fail "Constants of type bls12_381_mlresult are not supported"
  183 
  184 -- | Parser of constants whose type is in 'DefaultUni'.
  185 constant :: Parser (Some (ValueOf DefaultUni))
  186 constant = do
  187   -- Parse the type tag.
  188   SomeTypeIn (Kinded uni) <- defaultUni
  189   -- Check it's of kind @*@, because a constant that we're about to parse can only be of type of
  190   -- kind @*@.
  191   Refl <- reoption $ checkStar uni
  192   -- Parse the constant of the type represented by the type tag.
  193   someValueOf uni <$> constantOf ExpectParensYes uni
  194 
  195 data ExpectParens
  196   = ExpectParensYes
  197   | ExpectParensNo
