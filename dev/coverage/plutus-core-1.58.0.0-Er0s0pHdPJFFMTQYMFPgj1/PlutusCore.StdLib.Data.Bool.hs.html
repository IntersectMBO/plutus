   33 {-| @if_then_else_@ as a PLC term.
   34 
   35 > /\(A :: *) -> \(b : Bool) (x y : () -> A) -> IfThenElse {() -> A} b x y () -}
   36 ifThenElse
   37   :: forall term uni
   38    . ( TermLike term TyName Name uni DefaultFun
   39      , uni `HasTypeAndTermLevel` Bool
   40      , uni `HasTypeAndTermLevel` ()
   41      )
   42   => term ()
   43 ifThenElse = runQuote $ do
   44   a <- freshTyName "a"
   45   b <- freshName "b"
   46   x <- freshName "x"
   47   y <- freshName "y"
   48   let unitFunA = TyFun () unit (TyVar () a)
   49   return
   50     . tyAbs () a (Type ())
   51     $ mkIterLamAbs
   52       [ VarDecl () b bool
   53       , VarDecl () x unitFunA
   54       , VarDecl () y unitFunA
   55       ]
   56     $ mkIterAppNoAnn
   57       (tyInst () (builtin () IfThenElse) unitFunA)
   58       [var () b, var () x, var () y, unitval]
