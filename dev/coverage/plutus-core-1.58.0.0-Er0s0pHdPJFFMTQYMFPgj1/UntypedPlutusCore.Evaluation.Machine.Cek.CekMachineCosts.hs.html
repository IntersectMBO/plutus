   33 -- | Costs for evaluating AST nodes.  Times should be specified in picoseconds, memory sizes in bytes.
   34 type CekMachineCosts = CekMachineCostsBase Identity

   36 data CekMachineCostsBase f

   39   , cekVarCost :: f ExBudget
   40   , cekConstCost :: f ExBudget

   43   , cekForceCost :: f ExBudget
   44   , cekApplyCost :: f ExBudget
   45   , cekBuiltinCost :: f ExBudget
   46   {-^ Just the cost of evaluating a Builtin node, not the builtin itself.
   47   There's no entry for Error since we'll be exiting anyway; also, what would

   49   , cekConstrCost :: f ExBudget

   51   }

   83 -- Charge a unit CPU cost for AST nodes: this allows us to count the number of
   84 -- times each node type is evaluated.  For actual prediction/costing we use
   85 -- a different version of CekMachineCosts: see ExBudgetingDefaults.defaultCekMachineCosts.
   86 unitCekMachineCosts :: CekMachineCosts
   87 unitCekMachineCosts =
   88   CekMachineCostsBase
   89     { cekStartupCost = zeroCost
   90     , cekVarCost = unitCost
   91     , cekConstCost = unitCost
   92     , cekLamCost = unitCost
   93     , cekDelayCost = unitCost
   94     , cekForceCost = unitCost
   95     , cekApplyCost = unitCost
   96     , cekBuiltinCost = unitCost
   97     , cekConstrCost = unitCost
   98     , cekCaseCost = unitCost
   99     }
  100   where
  101     zeroCost = Identity $ ExBudget 0 0
  102     unitCost = Identity $ ExBudget 1 0
