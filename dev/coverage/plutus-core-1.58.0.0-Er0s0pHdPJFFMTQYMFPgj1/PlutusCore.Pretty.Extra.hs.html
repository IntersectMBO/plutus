   29 instance Profunctor InContextM where
   30   lmap
   31     :: forall config' config a
   32      . (config' -> config)
   33     -> InContextM config a
   34     -> InContextM config' a
   35   lmap = coerce (withReader :: (config' -> config) -> Reader config a -> Reader config' a)
   36   {-# INLINE lmap #-}
   37 
   38   rmap
   39     :: forall config a b
   40      . (a -> b)
   41     -> InContextM config a
   42     -> InContextM config b
   43   rmap = coerce (mapReader :: (a -> b) -> Reader config a -> Reader config b)
   44   {-# INLINE rmap #-}

   46 -- | For pretty-printing a value with a minimum amount of parens.
   47 type PrettyParens = PrettyBy RenderContext
   48 
   49 {-| An initial 'RenderContext'.
   50 An expression printed in this context gets enclosed in parens unless its outermost operator (if
   51 any) binds even stronger than function application. -}
   52 juxtRenderContext :: RenderContext
   53 juxtRenderContext = RenderContext ToTheRight juxtFixity
