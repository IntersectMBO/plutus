   71 -- | A final encoding for Term, to allow PLC terms to be used transparently as PIR terms.
   72 class TermLike term tyname name uni fun | term -> tyname name uni fun where
   73   var :: ann -> name -> term ann
   74   tyAbs :: ann -> tyname -> Kind ann -> term ann -> term ann
   75   lamAbs :: ann -> name -> Type tyname uni ann -> term ann -> term ann
   76   apply :: ann -> term ann -> term ann -> term ann
   77   constant :: ann -> Some (ValueOf uni) -> term ann
   78   builtin :: ann -> fun -> term ann
   79   tyInst :: ann -> term ann -> Type tyname uni ann -> term ann
   80   unwrap :: ann -> term ann -> term ann
   81   iWrap :: ann -> Type tyname uni ann -> Type tyname uni ann -> term ann -> term ann
   82   error :: ann -> Type tyname uni ann -> term ann
   83   constr :: ann -> Type tyname uni ann -> Word64 -> [term ann] -> term ann
   84   kase :: ann -> Type tyname uni ann -> term ann -> [term ann] -> term ann
   85 
   86   termLet :: ann -> TermDef term tyname name uni ann -> term ann -> term ann
   87   typeLet :: ann -> TypeDef tyname uni ann -> term ann -> term ann
   88 
   89   termLet = mkImmediateLamAbs
   90   typeLet = mkImmediateTyAbs

   95 mkTyBuiltinOf :: forall k (a :: k) uni tyname ann. ann -> uni (Esc a) -> Type tyname uni ann
   96 mkTyBuiltinOf ann = TyBuiltin ann . SomeTypeIn

   98 -- | Embed a Haskell value (given its explicit type tag) into a PLC term.
   99 mkConstantOf
  100   :: forall a uni fun term tyname name ann
  101    . TermLike term tyname name uni fun
  102   => ann -> uni (Esc a) -> a -> term ann
  103 mkConstantOf ann uni = constant ann . someValueOf uni
  104 
  105 -- | Embed a Haskell value (provided its type is in the universe) into a PLC term.
  106 mkConstant
  107   :: forall a uni fun term tyname name ann
  108    . (TermLike term tyname name uni fun, uni `HasTermLevel` a)
  109   => ann -> a -> term ann
  110 mkConstant ann = constant ann . someValue

  112 instance TermLike (Term tyname name uni fun) tyname name uni fun where
  113   var = Var
  114   tyAbs = TyAbs
  115   lamAbs = LamAbs
  116   apply = Apply
  117   constant = Constant
  118   builtin = Builtin
  119   tyInst = TyInst
  120   unwrap = Unwrap
  121   iWrap = IWrap
  122   error = Error
  123   constr = Constr
  124   kase = Case

  126 embedTerm :: TermLike term tyname name uni fun => Term tyname name uni fun ann -> term ann
  127 embedTerm = \case
  128   Var a n -> var a n
  129   TyAbs a tn k t -> tyAbs a tn k (embedTerm t)
  130   LamAbs a n ty t -> lamAbs a n ty (embedTerm t)
  131   Apply a t1 t2 -> apply a (embedTerm t1) (embedTerm t2)
  132   Constant a c -> constant a c
  133   Builtin a bi -> builtin a bi
  134   TyInst a t ty -> tyInst a (embedTerm t) ty
  135   Error a ty -> error a ty
  136   Unwrap a t -> unwrap a (embedTerm t)
  137   IWrap a ty1 ty2 t -> iWrap a ty1 ty2 (embedTerm t)
  138   Constr a ty i es -> constr a ty i (fmap embedTerm es)
  139   Case a ty arg cs -> kase a ty (embedTerm arg) (fmap embedTerm cs)

  141 {-| Make a 'Var' referencing the given 'VarDecl'.
  142 The @ann@ is propagated from the 'VarDecl' to the 'Var'. -}
  143 mkVar :: TermLike term tyname name uni fun => VarDecl tyname name uni ann -> term ann
  144 mkVar v = var (_varDeclAnn v) (_varDeclName v)

  152 -- | A definition. Pretty much just a pair with more descriptive names.
  153 data Def var val = Def
  154   { defVar :: var
  155   , defVal :: val
  156   }
  157   deriving stock (Show, Eq, Ord, Generic)

  159 -- | A term definition as a variable.
  160 type TermDef term tyname name uni ann = Def (VarDecl tyname name uni ann) (term ann)

  162 -- | A type definition as a type variable.
  163 type TypeDef tyname uni ann = Def (TyVarDecl tyname ann) (Type tyname uni ann)

  165 -- | The type of a PLC function.
  166 data FunctionType tyname uni ann = FunctionType
  167   { _functionTypeAnn :: ann
  168   -- ^ An annotation.
  169   , _functionTypeDom :: Type tyname uni ann
  170   -- ^ The domain of a function.
  171   , _functionTypeCod :: Type tyname uni ann
  172   -- ^ The codomain of the function.
  173   }
  174 
  175 -- Should we parameterize 'VarDecl' by @ty@ rather than @tyname@, so that we can define
  176 -- 'FunctionDef' as 'TermDef FunctionType tyname name uni fun ann'?
  177 -- Perhaps we even should define general 'Decl' and 'Def' that cover all of the cases?
  178 -- | A PLC function.
  179 data FunctionDef term tyname name uni fun ann = FunctionDef
  180   { _functionDefAnn :: ann
  181   -- ^ An annotation.
  182   , _functionDefName :: name
  183   -- ^ The name of a function.
  184   , _functionDefType :: FunctionType tyname uni ann
  185   -- ^ The type of the function.
  186   , _functionDefTerm :: term ann
  187   -- ^ The definition of the function.
  188   }
  189 
  190 -- | Convert a 'FunctionType' to the corresponding 'Type'.
  191 functionTypeToType :: FunctionType tyname uni ann -> Type tyname uni ann
  192 functionTypeToType (FunctionType ann dom cod) = TyFun ann dom cod
  193 
  194 -- | Get the type of a 'FunctionDef'.
  195 functionDefToType :: FunctionDef term tyname name uni fun ann -> Type tyname uni ann
  196 functionDefToType (FunctionDef _ _ funTy _) = functionTypeToType funTy
  197 
  198 -- | Convert a 'FunctionDef' to a 'VarDecl'. I.e. ignore the actual term.
  199 functionDefVarDecl :: FunctionDef term tyname name uni fun ann -> VarDecl tyname name uni ann
  200 functionDefVarDecl (FunctionDef ann name funTy _) = VarDecl ann name $ functionTypeToType funTy
  201 
  202 -- | Make a 'FunctionDef'. Return 'Nothing' if the provided type is not functional.
  203 mkFunctionDef
  204   :: ann
  205   -> name
  206   -> Type tyname uni ann
  207   -> term ann
  208   -> Maybe (FunctionDef term tyname name uni fun ann)
  209 mkFunctionDef annName name (TyFun annTy dom cod) term =
  210   Just $ FunctionDef annName name (FunctionType annTy dom cod) term
  211 mkFunctionDef _ _ _ _ = Nothing

  213 -- | Make a "let-binding" for a term as an immediately applied lambda abstraction.
  214 mkImmediateLamAbs
  215   :: TermLike term tyname name uni fun
  216   => ann
  217   -> TermDef term tyname name uni ann
  218   -> term ann
  219   -- ^ The body of the let, possibly referencing the name.
  220   -> term ann
  221 mkImmediateLamAbs ann1 (Def (VarDecl ann2 name ty) bind) body =
  222   apply ann1 (lamAbs ann2 name ty body) bind
  223 
  224 -- | Make a "let-binding" for a type as an immediately instantiated type abstraction. Note: the body must be a value.
  225 mkImmediateTyAbs
  226   :: TermLike term tyname name uni fun
  227   => ann
  228   -> TypeDef tyname uni ann
  229   -> term ann
  230   -- ^ The body of the let, possibly referencing the name.
  231   -> term ann
  232 mkImmediateTyAbs ann1 (Def (TyVarDecl ann2 name k) bind) body =
  233   tyInst ann1 (tyAbs ann2 name k body) bind

  235 {-| Make an iterated application. Each `apply` node uses the annotation associated with
  236 the corresponding argument. -}
  237 mkIterApp
  238   :: TermLike term tyname name uni fun
  239   => term ann
  240   -> [(ann, term ann)]
  241   -> term ann
  242 mkIterApp = foldl' $ \acc (ann, arg) -> apply ann acc arg
  243 
  244 -- | Make an iterated application with no annotation.
  245 mkIterAppNoAnn
  246   :: TermLike term tyname name uni fun
  247   => term ()
  248   -- ^ @f@
  249   -> [term ()]
  250   -- ^ @[ x0 ... xn ]@
  251   -> term ()
  252   -- ^ @[f x0 ... xn ]@
  253 mkIterAppNoAnn term = mkIterApp term . fmap ((),)
  254 
  255 -- | An infix synonym for `mkIterAppNoAnn`
  256 (@@)
  257   :: TermLike term tyname name uni fun
  258   => term ()
  259   -- ^ @f@
  260   -> [term ()]
  261   -- ^ @[ x0 ... xn ]@
  262   -> term ()
  263   -- ^ @[f x0 ... xn ]@
  264 (@@) = mkIterAppNoAnn

  266 {-| Make an iterated instantiation. Each `tyInst` node uses the annotation associated with
  267 the corresponding argument. -}
  268 mkIterInst
  269   :: TermLike term tyname name uni fun
  270   => term ann
  271   -- ^ @a@
  272   -> [(ann, Type tyname uni ann)]
  273   -- ^ @ [ x0 ... xn ] @
  274   -> term ann
  275   -- ^ @{ a x0 ... xn }@
  276 mkIterInst = foldl' $ \acc (ann, arg) -> tyInst ann acc arg

  283   -> [Type tyname uni ()]
  284   -- ^ @ [ x0 ... xn ] @

  286   -- ^ @{ a x0 ... xn }@

  289 -- | Lambda abstract a list of names.
  290 mkIterLamAbs
  291   :: TermLike term tyname name uni fun
  292   => [VarDecl tyname name uni ann]
  293   -> term ann
  294   -> term ann
  295 mkIterLamAbs args body =
  296   foldr (\(VarDecl ann name ty) acc -> lamAbs ann name ty acc) body args

  302   -> term ann

  304 mkIterTyAbs args body =

  307 {-| Make an iterated type application. Each `TyApp` node uses the annotation associated with

  328 -- | Make an iterated function type.
  329 mkIterTyFun
  330   :: ann
  331   -> [Type tyname uni ann]
  332   -> Type tyname uni ann
  333   -> Type tyname uni ann
  334 mkIterTyFun ann tys target = foldr (\ty acc -> TyFun ann ty acc) target tys

  336 -- | Universally quantify a list of names.
  337 mkIterTyForall
  338   :: [TyVarDecl tyname ann]
  339   -> Type tyname uni ann
  340   -> Type tyname uni ann
  341 mkIterTyForall args body =
  342   foldr (\(TyVarDecl ann name kind) acc -> TyForall ann name kind acc) body args

  344 -- | Lambda abstract a list of names.
  345 mkIterTyLam
  346   :: [TyVarDecl tyname ann]
  347   -> Type tyname uni ann
  348   -> Type tyname uni ann
  349 mkIterTyLam args body =

  360 {-| A helper to create a single, fresh strict binding; It returns the fresh bound `Var`iable and
  361 a function `Term -> Term`, expecting an "in-Term" to form a let-expression. -}
  362 mkFreshTermLet
  363   :: (MonadQuote m, TermLike t tyname Name uni fun, Monoid a)
  364   => Type tyname uni a
  365   -- ^ the type of binding
  366   -> t a
  367   -- ^ the term bound to the fresh variable
  368   -> m (t a, t a -> t a)
  369   -- ^ the fresh Var and a function that takes an "in" term to construct the Let
  370 mkFreshTermLet aT a = do
  371   -- I wish this was less constrained to Name
  372   genName <- freshName "generated"
  373   pure (var mempty genName, termLet mempty (Def (VarDecl mempty genName aT) a))

  375 -- | 'apply' the head of the application to the arguments iteratively.
  376 headSpineToTerm :: TermLike term tyname name uni fun => ann -> MonoHeadSpine err (term ann) -> Either err (term ann)
  377 headSpineToTerm _ (HeadError e) = Left e
  378 headSpineToTerm _ (HeadOnly t) = Right t
  379 headSpineToTerm ann (HeadSpine t ts) = Right $ foldl (apply ann) t ts
  380 
  381 -- | @headSpineToTerm@ but without annotation.
  382 headSpineToTermNoAnn :: TermLike term tyname name uni fun => MonoHeadSpine err (term ()) -> Either err (term ())
  383 headSpineToTermNoAnn (HeadError e) = Left e
  384 headSpineToTermNoAnn (HeadOnly t) = Right t
  385 headSpineToTermNoAnn (HeadSpine t ts) = Right $ foldl (apply ()) t ts
