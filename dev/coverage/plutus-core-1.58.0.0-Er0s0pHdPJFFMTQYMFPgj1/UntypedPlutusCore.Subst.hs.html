   27 -- | Applicatively replace a variable using the given function.
   28 substVarA
   29   :: Applicative f
   30   => (name -> ann -> f (Maybe (Term name uni fun ann)))
   31   -> Term name uni fun ann
   32   -> f (Term name uni fun ann)
   33 substVarA nameF t@(Var a name) = fromMaybe t <$> nameF name a
   34 substVarA _ t = pure t

   36 -- | Replace a variable using the given function.
   37 substVar
   38   :: (name -> ann -> Maybe (Term name uni fun ann))
   39   -> Term name uni fun ann
   40   -> Term name uni fun ann
   41 substVar = purely (substVarA . curry) . uncurry

   58 -- | Completely replace the names with a new name type.
   59 termMapNames
   60   :: forall name name' uni fun ann
   61    . (name -> name')
   62   -> Term name uni fun ann
   63   -> Term name' uni fun ann
   64 termMapNames f = go
   65   where
   66     -- This is all a bit clunky because of the type-changing, I'm not sure of a nicer way to do it
   67     go :: Term name uni fun ann -> Term name' uni fun ann
   68     go = \case
   69       LamAbs ann name body -> LamAbs ann (f name) (go body)
   70       Var ann name -> Var ann (f name)
   71       Apply ann t1 t2 -> Apply ann (go t1) (go t2)
   72       Delay ann t -> Delay ann (go t)
   73       Force ann t -> Force ann (go t)
   74       Constr ann i es -> Constr ann i (fmap go es)
   75       Case ann arg cs -> Case ann (go arg) (fmap go cs)
   76       Constant ann c -> Constant ann c
   77       Builtin ann b -> Builtin ann b
   78       Error ann -> Error ann
   79 
   80 programMapNames
   81   :: forall name name' uni fun ann
   82    . (name -> name')
   83   -> Program name uni fun ann
   84   -> Program name' uni fun ann
   85 programMapNames f (Program a v term) = Program a v (termMapNames f term)
