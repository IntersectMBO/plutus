   31 > getEta n = \(f :: * -> *) (z :: *) -> n f z -}
   32 getEta :: Type TyName uni () -> Quote (Type TyName uni ())
   33 getEta n = do
   34   f <- freshTyName "f"
   35   z <- freshTyName "z"

  102 > stepFun a r1 r2 = all (p :: natK). a -> r1 p -> r2 (succT p) -}
  103 getStepFun :: TyName -> Type TyName uni () -> TyName -> Quote (Type TyName uni ())
  104 getStepFun a r1 r2 = do
  105   p <- freshTyName "p"
  106   return
  107     . TyForall () p natK
  108     . TyFun () (TyVar () a)
  109     . TyFun () (TyApp () r1 $ TyVar () p)
  110     . TyApp () (TyVar () r2)
  111     . TyApp () succT
  112     $ TyVar () p
