   35 -- | A parsable PLC term.
   36 type PTerm = Term TyName Name DefaultUni DefaultFun SrcSpan
   37 
   38 varTerm :: Parser PTerm
   39 varTerm = withSpan $ \sp ->
   40   Var sp <$> name
   41 
   42 tyAbsTerm :: Parser PTerm
   43 tyAbsTerm = withSpan $ \sp ->
   44   inParens $ TyAbs sp <$> (symbol "abs" *> trailingWhitespace tyName) <*> kind <*> term
   45 
   46 lamTerm :: Parser PTerm
   47 lamTerm = withSpan $ \sp ->
   48   inParens $ LamAbs sp <$> (symbol "lam" *> trailingWhitespace name) <*> pType <*> term
   49 
   50 appTerm :: Parser PTerm
   51 appTerm = withSpan $ \sp ->
   52   -- TODO: should not use the same `sp` for all arguments.
   53   inBrackets $ mkIterApp <$> term <*> (fmap (sp,) <$> some term)
   54 
   55 conTerm :: Parser PTerm
   56 conTerm = withSpan $ \sp ->
   57   inParens $ Constant sp <$> (symbol "con" *> constant)
   58 
   59 builtinTerm :: Parser PTerm
   60 builtinTerm = withSpan $ \sp ->
   61   inParens $ Builtin sp <$> (symbol "builtin" *> builtinFunction)
   62 
   63 tyInstTerm :: Parser PTerm
   64 tyInstTerm = withSpan $ \sp ->
   65   -- TODO: should not use the same `sp` for all arguments.
   66   inBraces $ mkIterInst <$> term <*> (fmap (sp,) <$> many pType)
   67 
   68 unwrapTerm :: Parser PTerm
   69 unwrapTerm = withSpan $ \sp ->
   70   inParens $ Unwrap sp <$> (symbol "unwrap" *> term)
   71 
   72 iwrapTerm :: Parser PTerm
   73 iwrapTerm = withSpan $ \sp ->
   74   inParens $ IWrap sp <$> (symbol "iwrap" *> pType) <*> pType <*> term
   75 
   76 errorTerm :: Parser PTerm
   77 errorTerm = withSpan $ \sp ->
   78   inParens $ Error sp <$> (symbol "error" *> pType)
   79 
   80 constrTerm :: Parser PTerm
   81 constrTerm = withSpan $ \sp ->
   82   inParens $ do
   83     let maxTag = fromIntegral (maxBound :: Word64)
   84     ty <- symbol "constr" *> pType
   85     tag :: Integer <- lexeme Lex.decimal
   86     args <- many term
   87     whenVersion (\v -> v < plcVersion110) $ fail "'constr' is not allowed before version 1.1.0"
   88     when (tag > maxTag) $ fail "constr tag too large: must be a legal Word64 value"
   89     pure $ Constr sp ty (fromIntegral tag) args
   90 
   91 caseTerm :: Parser PTerm
   92 caseTerm = withSpan $ \sp ->
   93   inParens $ do
   94     res <- Case sp <$> (symbol "case" *> pType) <*> term <*> many term
   95     whenVersion (\v -> v < plcVersion110) $ fail "'case' is not allowed before version 1.1.0"
   96     pure res
   97 
   98 -- | Parser for all PLC terms.
   99 term :: Parser PTerm
  100 term = leadingWhitespace go
  101   where
  102     go =
  103       choice $
  104         map
  105           try
  106           [ tyAbsTerm
  107           , lamTerm
  108           , appTerm
  109           , conTerm
  110           , builtinTerm
  111           , tyInstTerm
  112           , unwrapTerm
  113           , iwrapTerm
  114           , errorTerm
  115           , varTerm
  116           , constrTerm
  117           , caseTerm
  118           ]

  120 {-| Parse a PLC program. The resulting program will have fresh names. The
  121 underlying monad must be capable of handling any parse errors.  This passes
  122 "test" to the parser as the name of the input stream; to supply a name
  123 explicity, use `parse program <name> <input>`. -}
  124 parseProgram
  125   :: (MonadError ParserErrorBundle m, MonadQuote m)
  126   => Text
  127   -> m (Program TyName Name DefaultUni DefaultFun SrcSpan)
  128 parseProgram = parseGen program
  129 
  130 -- | Parser for PLC programs.
  131 program :: Parser (Program TyName Name DefaultUni DefaultFun SrcSpan)
  132 program = leadingWhitespace go
  133   where
  134     go = do
  135       prog <- withSpan $ \sp -> inParens $ do
  136         v <- symbol "program" *> version
  137         withVersion v $ Program sp v <$> term
  138       notFollowedBy anySingle
  139       pure prog
  140 
  141 {-| Parse a PLC term. The resulting program will have fresh names. The underlying monad
  142 must be capable of handling any parse errors. -}
  143 parseTerm
  144   :: (MonadError ParserErrorBundle m, MonadQuote m)
  145   => Text -> m (Term TyName Name DefaultUni DefaultFun SrcSpan)
  146 parseTerm = parseGen term

  148 {-| Parse a PLC type. The resulting program will have fresh names. The underlying monad
  149 must be capable of handling any parse errors. -}
  150 parseType
  151   :: (MonadError ParserErrorBundle m, MonadQuote m)
  152   => Text -> m (Type TyName DefaultUni SrcSpan)
  153 parseType = parseGen pType
