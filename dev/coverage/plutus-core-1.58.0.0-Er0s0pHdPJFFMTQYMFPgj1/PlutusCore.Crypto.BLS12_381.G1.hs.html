   47 {- Note [Wrapping the BLS12-381 types in Plutus Core].  In the Haskell bindings
   48 to the `blst` library in cardano-crypto-class, points in G1 and G2 are
   49 represented as ForeignPtrs pointing to C objects, with a phantom type
   50 determining which group is involved. We have to wrap these in a newtype here
   51 because otherwise the builtin machinery spots that they're applications and
   52 can't find the relevant type parameters.  In theory I think we could add a
   53 couple of phantom types to the default universe, but it seemed simpler and safer
   54 to use monomorphic types instead, even though it requires a bit of code
   55 duplication between G1 and G2.  Much of the code here has apparently identical
   56 versions in PlutusCore.Crypto.BLS12_381.G2, but the signatures of the functions
   57 in the two versions are invisibly instantiating the cardano-crypto-class
   58 functions at the appropriate phantom types.
   59 
   60 See also Note [Wrapping the BLS12-381 types in PlutusTx].
   61 -}
   62 newtype Element = Element {unElement :: BlstBindings.Point1}
   63   deriving newtype (Eq)
   64 instance Show Element where
   65   show = byteStringAsHex . compress
   66 instance Pretty Element where
   67   pretty = pretty . show
   68 instance PrettyBy ConstConfig Element

   70 {-| We don't support direct flat encoding of G2 elements because of the expense
   71    of on-chain uncompression.  Users should convert between G2 elements and
   72    bytestrings using `compress` and `uncompress`: the bytestrings can be
   73    flat-encoded in the usual way. -}
   74 instance Flat Element where
   75   -- This might happen on the chain, so `fail` rather than `error`.
   76   decode = fail "Flat decoding is not supported for objects of type bls12_381_G1_element: use bls12_381_G1_uncompress on a bytestring instead."
   77 
   78   -- This will be a Haskell runtime error, but encoding doesn't happen on chain,
   79   -- so it's not too bad.
   80   encode = error "Flat encoding is not supported for objects of type bls12_381_G1_element: use bls12_381_G1_compress to obtain a bytestring instead."
   81   size _ = id
   82 
   83 instance NFData Element where
   84   rnf (Element x) = rwhnf x -- Just to be on the safe side.
   85 
   86 instance Hashable Element where
   87   hashWithSalt salt = hashWithSalt salt . compress

   99 {-| Multiplication of group elements by scalars. In the blst library the
  100 arguments are the other way round, but scalars acting on the left is more
  101 consistent with standard mathematical practice. -}
  102 scalarMul :: Integer -> Element -> Element
  103 scalarMul = coerce $ flip (BlstBindings.blsMult @BlstBindings.Curve1)
  104 {-# INLINE scalarMul #-}

  106 {-| Compress a G1 element to a bytestring. This serialises a curve point to its
  107  x coordinate only.  The compressed bytestring is 48 bytes long, with three
  108  spare bits used to convey extra information about the point, including
  109  determining which of two possible y coordinates the point has and whether the
  110  point is the point at infinity. See
  111  https://github.com/supranational/blst#serialization-format -}
  112 compress :: Element -> ByteString
  113 compress = coerce (BlstBindings.blsCompress @BlstBindings.Curve1)
  114 {-# INLINE compress #-}
  115 
  116 {-| Uncompress a bytestring to get a G1 point.  This will fail if any of the
  117    following are true.
  118      * The bytestring is not exactly 48 bytes long.
  119      * The most significant three bits are used incorrectly.
  120      * The bytestring encodes a field element which is not the
  121        x coordinate of a point on the E1 curve.
  122      * The bytestring does represent a point on the E1 curve, but the
  123        point is not in the G1 subgroup. -}
  124 uncompress :: ByteString -> Either BlstBindings.BLSTError Element
  125 uncompress = coerce (BlstBindings.blsUncompress @BlstBindings.Curve1)
  126 {-# INLINE uncompress #-}

  128 {-  Note [Hashing and Domain Separation Tags].  The hashToGroup functions take a
  129    bytestring and hash it to obtain an element in the relevant group, as
  130    described in
  131 
  132    https://datatracker.ietf.org/doc/html/draft-irtf-cfrg-hash-to-curve.
  133 
  134    In addition to the input bytestring (the "message"), the hashing function
  135    also takes another (possibly empty) bytestring argument, a "Domain Separation
  136    Tag" (DST), which is incorporated in the hashing process; the intention is
  137    that for security purposes different protocols should use different DSTs to
  138    ensure that hashes are unique to the protocol in use: see Section 2.2.5 of
  139    the above specification.  In principle, arbitrary-length DSTs can be used,
  140    but we only allow DSTs of up to 255 bytes, failing if a larger DST is
  141    supplied.  If a larger DST is required, it should be hashed beforehand to
  142    obtain a hash of accpetable size, as described in Section 5.3.3 of the
  143    specification.
  144 
  145    The hashing functions in the blst library allow a third argument as well (an
  146    "augmentation string").  We don't support this functionality directly because
  147    precisely the same effect can be achieved by prepending the augmentation
  148    string to the message.
  149 -}
  150 
  151 {-| Take an arbitrary bytestring and a Domain Separation Tag (DST) and hash
  152 them to a get point in G1. -}
  153 hashToGroup :: ByteString -> ByteString -> Either BLS12_381_Error Element
  154 hashToGroup msg dst =
  155   if Data.ByteString.length dst > 255
  156     then Left HashToCurveDstTooBig
  157     else Right . Element $ BlstBindings.blsHash @BlstBindings.Curve1 msg (Just dst) Nothing

  185 {-| Multi-scalar multiplication of G1 points.  We limit the allowable size of
  186 scalars to simplify costing. -}
  187 multiScalarMul :: [Integer] -> [Element] -> BuiltinResult Element
  188 multiScalarMul ss p
  189   | any msmScalarOutOfBounds ss = fail "Scalar exceeds 512-byte bound for G1.multiScalarMul"
  190   | otherwise = pure . coerce $ BlstBindings.blsMSM @BlstBindings.Curve1 (zip ss (coerce p))
