   58 data CkValue uni fun
   59   = VCon (Some (ValueOf uni))
   60   | VTyAbs TyName (Kind ()) (Term TyName Name uni fun ())
   61   | VLamAbs Name (Type TyName uni ()) (Term TyName Name uni fun ())
   62   | VIWrap (Type TyName uni ()) (Type TyName uni ()) (CkValue uni fun)
   63   | VBuiltin (Term TyName Name uni fun ()) (BuiltinRuntime (CkValue uni fun))
   64   | VConstr (Type TyName uni ()) Word64 [CkValue uni fun]
   65 
   66 deriving stock instance
   67   (GShow uni, Everywhere uni Show, Show fun, Closed uni)
   68   => Show (CkValue uni fun)
   69 
   70 ckValueToTerm :: CkValue uni fun -> Term TyName Name uni fun ()
   71 ckValueToTerm = \case
   72   VCon val -> Constant () val
   73   VTyAbs tn k body -> TyAbs () tn k body
   74   VLamAbs name ty body -> LamAbs () name ty body
   75   VIWrap ty1 ty2 val -> IWrap () ty1 ty2 $ ckValueToTerm val
   76   VBuiltin term _ -> term
   77   VConstr ty i es -> Constr () ty i (fmap ckValueToTerm es)
   78 
   79 data CkEnv uni fun s = CkEnv
   80   { ckEnvRuntime :: BuiltinsRuntime fun (CkValue uni fun)

   83     -- as we store logs as "latest go last".
   84     ckEnvMayEmitRef :: Maybe (STRef s (DList Text))
   85   }

  124 -- The 'DList' is just be consistent with the CEK machine (see Note [DList-based emitting]).
  125 emitCkM :: DList Text -> CkM uni fun s ()
  126 emitCkM logs = do
  127   mayLogsRef <- asks ckEnvMayEmitRef
  128   case mayLogsRef of
  129     Nothing -> pure ()
  130     Just logsRef -> lift . lift $ modifySTRef logsRef (`DList.append` logs)

  140 data Frame uni fun
  141   = -- | @[V _]@
  142     FrameAwaitArg (CkValue uni fun)
  143   | -- | @[_ N]@
  144     FrameAwaitFunTerm (Term TyName Name uni fun ())
  145   | -- | @[_ V]@
  146     FrameAwaitFunValue (CkValue uni fun)
  147   | -- | @{_ A}@
  148     FrameTyInstArg (Type TyName uni ())
  149   | -- | @(unwrap _)@
  150     FrameUnwrap
  151   | -- | @(iwrap A B _)@
  152     FrameIWrap (Type TyName uni ()) (Type TyName uni ())
  153   | FrameConstr (Type TyName uni ()) Word64 [Term TyName Name uni fun ()] [CkValue uni fun]
  154   | FrameCase [Term TyName Name uni fun ()]
  155 
  156 deriving stock instance
  157   (GShow uni, Closed uni, uni `Everywhere` Show, Show fun)
  158   => Show (Frame uni fun)
  159

  166 runCkM
  167   :: BuiltinsRuntime fun (CkValue uni fun)
  168   -> CaserBuiltin uni
  169   -> Bool
  170   -> (forall s. CkM uni fun s a)
  171   -> (Either (CkEvaluationException uni fun) a, [Text])
  172 runCkM runtime caser emitting a = runST $ do
  173   mayLogsRef <- if emitting then Just <$> newSTRef DList.empty else pure Nothing
  174   errOrRes <- runExceptT . runReaderT a $ CkEnv runtime caser mayLogsRef
  175   logs <- case mayLogsRef of
  176     Nothing -> pure []
  177     Just logsRef -> DList.toList <$> readSTRef logsRef
  178   pure (errOrRes, logs)

  180 -- FIXME: make sure that the specification is up to date and that this matches.
  181 -- Tracked by https://github.com/IntersectMBO/plutus-private/issues/1552.
  182 {-| The computing part of the CK machine. Rules are as follows:
  183 
  184 > s ▷ {M A}      ↦ s , {_ A}        ▷ M
  185 > s ▷ [M N]      ↦ s , [_ N]        ▷ M
  186 > s ▷ wrap α A M ↦ s , (wrap α S _) ▷ M
  187 > s ▷ unwrap M   ↦ s , (unwrap _)   ▷ M
  188 > s ▷ abs α K M  ↦ s ◁ abs α K M
  189 > s ▷ lam x A M  ↦ s ◁ lam x A M
  190 > s ▷ builtin bn ↦ s ◁ builtin (Builtin () bn) (runtimeOf bn)
  191 > s ▷ con cn     ↦ s ◁ con cn
  192 > s ▻ constr I T0 .. Tn ↦ s , (constr I _ T1 Tn) ▻ T0
  193 > s ▻ case S C0 ... Cn ↦ s , (case _ C0 ... Cn) ▻ S
  194 > s ▷ error A    ↦ ◆ -}
  195 (|>)
  196   :: Context uni fun -> Term TyName Name uni fun () -> CkM uni fun s (Term TyName Name uni fun ())
  197 stack |> TyInst _ fun ty = FrameTyInstArg ty : stack |> fun
  198 stack |> Apply _ fun arg = FrameAwaitFunTerm arg : stack |> fun
  199 stack |> IWrap _ pat arg term = FrameIWrap pat arg : stack |> term
  200 stack |> Unwrap _ term = FrameUnwrap : stack |> term
  201 stack |> TyAbs _ tn k term = stack <| VTyAbs tn k term
  202 stack |> LamAbs _ name ty body = stack <| VLamAbs name ty body
  203 stack |> Builtin _ bn = do
  204   runtime <- lookupBuiltin bn . ckEnvRuntime <$> ask
  205   stack <| VBuiltin (Builtin () bn) runtime
  206 stack |> Constant _ val = stack <| VCon val
  207 stack |> Constr _ ty i es = case es of
  208   [] -> stack <| VConstr ty i []
  209   t : ts -> FrameConstr ty i ts [] : stack |> t
  210 stack |> Case _ _ arg cs = FrameCase cs : stack |> arg
  211 _ |> err@Error {} =
  212   throwErrorWithCause (OperationalError CkEvaluationFailure) $ void err
  213 _ |> var@Var {} =
  214   throwErrorWithCause (StructuralError OpenTermEvaluatedMachineError) var

  216 -- FIXME: make sure that the specification is up to date and that this matches.
  217 -- Tracked by https://github.com/IntersectMBO/plutus-private/issues/1552.
  218 {-| The returning part of the CK machine. Rules are as follows:
  219 
  220 > s , {_ A}           ◁ abs α K M  ↦ s         ▷ {A/α}M
  221 > s , [_ N]           ◁ V          ↦ s , [V _] ▷ N
  222 > s , [_ V1 ... Vm]   ◁ (lam x A M) ↦ s , [_ V2 ... Vm] ▷ [V1/x]M
  223 > s , [(lam x A M) _] ◁ V          ↦ s         ▷ [V/x]M
  224 > s , {_ A}           ◁ F          ↦ s ◁ {F A}  -- Partially instantiated builtin application.
  225 > s , [F _]           ◁ V          ↦ s ◁ [F V]  -- Partially saturated builtin application.
  226 > s , [F _]           ◁ V          ↦ s ◁ W      -- Fully saturated builtin application, [F V] ~> W.
  227 > s , (wrap α S _)    ◁ V          ↦ s ◁ wrap α S V
  228 > s , (unwrap _)      ◁ wrap α A V ↦ s ◁ V
  229 > s , (constr I V0 ... Vj-1 _ Tj+1 ... Tn) ◅ Vj ↦ s , (constr i V0 ... Vj _ Tj+2... Tn) ▻ Tj+1
  230 > s , (case _ C0 ... CN) ◅ (constr i V1 .. Vm) ↦ s , [_ V1 ... Vm] ▻ Ci -}
  231 (<|)
  232   :: Context uni fun -> CkValue uni fun -> CkM uni fun s (Term TyName Name uni fun ())
  233 [] <| val = pure $ ckValueToTerm val
  234 FrameTyInstArg ty : stack <| fun = instantiateEvaluate stack ty fun
  235 FrameAwaitFunTerm arg : stack <| fun = FrameAwaitArg fun : stack |> arg
  236 FrameAwaitArg fun : stack <| arg = applyEvaluate stack fun arg
  237 FrameAwaitFunValue arg : stack <| fun = applyEvaluate stack fun arg
  238 FrameIWrap pat arg : stack <| value = stack <| VIWrap pat arg value
  239 FrameUnwrap : stack <| wrapped = case wrapped of
  240   VIWrap _ _ term -> stack <| term
  241   _ ->
  242     throwErrorWithCause (StructuralError NonWrapUnwrappedMachineError) $ ckValueToTerm wrapped
  243 FrameConstr ty i todo done : stack <| e =
  244   let done' = e : done
  245    in case todo of
  246         t : ts -> FrameConstr ty i ts done' : stack |> t
  247         [] -> stack <| VConstr ty i (reverse done')
  248 FrameCase cs : stack <| e = case e of
  249   VConstr _ i args -> case cs ^? wix i of
  250     Just t -> go (reverse args) stack |> t
  251       where
  252         go [] s = s
  253         go (arg : rest) s = go rest (FrameAwaitFunValue arg : s)
  254     Nothing ->
  255       throwErrorWithCause (StructuralError $ MissingCaseBranchMachineError i) $ ckValueToTerm e
  256   VCon val -> do
  257     caser <- asks ckCaserBuiltin
  258     case unCaserBuiltin caser val $ Vector.fromList cs of
  259       HeadError err ->
  260         throwErrorWithCause (OperationalError $ CkCaseBuiltinError err) $ ckValueToTerm e
  261       HeadOnly fX -> stack |> fX
  262       HeadSpine f xs -> transferConstantSpine xs stack |> f
  263   _ -> throwErrorWithCause (StructuralError NonConstrScrutinizedMachineError) $ ckValueToTerm e
  264 
  265 transferConstantSpine :: Spine (Some (ValueOf uni)) -> Context uni fun -> Context uni fun
  266 transferConstantSpine args ctx = foldr ((:) . FrameAwaitFunValue . VCon) ctx args

  268 {-| Take a possibly partial builtin application and
  269 
  270 - either create a 'CkValue' by evaluating the application if it's saturated (emitting logs, if
  271    any, along the way), potentially failing evaluation
  272 - or create a partial builtin application otherwise
  273 
  274 and proceed with the returning phase of the CK machine. -}
  275 evalBuiltinApp
  276   :: Context uni fun
  277   -> Term TyName Name uni fun ()
  278   -> BuiltinRuntime (CkValue uni fun)
  279   -> CkM uni fun s (Term TyName Name uni fun ())
  280 evalBuiltinApp stack term runtime = case runtime of
  281   BuiltinCostedResult _ getFXs -> case getFXs of
  282     BuiltinSuccess y -> stack <| y
  283     BuiltinSuccessWithLogs logs y -> emitCkM logs *> (stack <| y)
  284     BuiltinFailure logs err -> emitCkM logs *> throwBuiltinErrorWithCause term err
  285   _ -> stack <| VBuiltin term runtime

  287 {-| Instantiate a term with a type and proceed.
  288 In case of 'TyAbs' just ignore the type. Otherwise check if the term is builtin application
  289 expecting a type argument, in which case either calculate the builtin application or stick a
  290 'TyInst' on top of its 'Term' representation depending on whether the application is saturated or
  291 not. In any other case, fail. -}
  292 instantiateEvaluate
  293   :: Context uni fun
  294   -> Type TyName uni ()
  295   -> CkValue uni fun
  296   -> CkM uni fun s (Term TyName Name uni fun ())
  297 instantiateEvaluate stack ty (VTyAbs tn _k body) =
  298   -- No kind check - too expensive at run time.
  299   stack |> termSubstClosedType tn ty body
  300 instantiateEvaluate stack ty (VBuiltin term runtime) = do
  301   let term' = TyInst () term ty
  302   case runtime of
  303     -- We allow a type argument to appear last in the type of a built-in function,
  304     -- otherwise we could just assemble a 'VBuiltin' without trying to evaluate the
  305     -- application.
  306     BuiltinExpectForce runtime' -> evalBuiltinApp stack term' runtime'
  307     _ -> throwErrorWithCause (StructuralError BuiltinTermArgumentExpectedMachineError) term'
  308 instantiateEvaluate _ _ val =
  309   throwErrorWithCause (StructuralError NonPolymorphicInstantiationMachineError) $ ckValueToTerm val
  310 
  311 {-| Apply a function to an argument and proceed.
  312 If the function is a lambda, then perform substitution and proceed.
  313 If the function is a builtin application then check that it's expecting a term argument,
  314 and either calculate the builtin application or stick a 'Apply' on top of its 'Term'
  315 representation depending on whether the application is saturated or not. -}
  316 applyEvaluate
  317   :: Context uni fun
  318   -> CkValue uni fun
  319   -> CkValue uni fun
  320   -> CkM uni fun s (Term TyName Name uni fun ())
  321 applyEvaluate stack (VLamAbs name _ body) arg =
  322   stack |> termSubstClosedTerm name (ckValueToTerm arg) body
  323 applyEvaluate stack (VBuiltin term runtime) arg = do
  324   let argTerm = ckValueToTerm arg
  325       term' = Apply () term argTerm
  326   case runtime of
  327     -- It's only possible to apply a builtin application if the builtin expects a term
  328     -- argument next.
  329     BuiltinExpectArgument f -> do
  330       evalBuiltinApp stack term' $ f arg
  331     _ ->
  332       throwErrorWithCause (StructuralError UnexpectedBuiltinTermArgumentMachineError) term'
  333 applyEvaluate _ val _ =
  334   throwErrorWithCause (StructuralError NonFunctionalApplicationMachineError) $ ckValueToTerm val

  349   -> (Either (CkEvaluationException uni fun) (Term TyName Name uni fun ()), [Text])

  352 -- | Evaluate a term using the CK machine with logging disabled.
  353 evaluateCkNoEmit
