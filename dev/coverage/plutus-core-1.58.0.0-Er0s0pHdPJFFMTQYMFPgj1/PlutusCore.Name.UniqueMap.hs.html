    1 {-| A type for maps (key-value associations), where the key type can be
    2  identified by 'Unique's. In practice, these types are usually names.
    3  This approach is preferred when it is more efficient to compare the associated
    4  'Unique's instead of the underlying type. -}

   30 {-| A mapping from 'Unique's to arbitrary values of type @a@.
   31  Since 'Unique' is equivalent to 'Int' (see "PlutusCore.Name.Unique"),
   32  we can use an 'IntMap' representation for this type. -}
   33 newtype UniqueMap unique a = UniqueMap
   34   { unUniqueMap :: IM.IntMap a
   35   }
   36   deriving stock (Show, Eq)
   37   deriving newtype (Semigroup, Monoid, Functor, Foldable)
   38 
   39 -- | Insert a value @a@ by a @unique@.
   40 insertByUnique
   41   :: Coercible unique Unique
   42   => unique
   43   -> a
   44   -> UniqueMap unique a
   45   -> UniqueMap unique a
   46 insertByUnique uniq = coerce . IM.insert (coerce uniq)
   47 
   48 -- | Insert a value @a@ by the @unique@ of a @name@.
   49 insertByName :: HasUnique name unique => name -> a -> UniqueMap unique a -> UniqueMap unique a
   50 insertByName = insertByUnique . view unique
   51 
   52 -- | Create the singleton map of the @unique@ of a @name@ and a value @a@.
   53 singletonByName :: HasUnique name unique => name -> a -> UniqueMap unique a
   54 singletonByName n a = insertByName n a mempty
   55 
   56 -- | Insert a named value @a@ by the index of the @unique@ of the @name@.
   57 insertNamed
   58   :: (HasText name, HasUnique name unique)
   59   => name
   60   -> a
   61   -> UniqueMap unique (Named a)
   62   -> UniqueMap unique (Named a)
   63 insertNamed name = insertByName name . Named (name ^. theText)
   64 
   65 {-| Insert a value by the index of the unique of a name.
   66 Unlike 'insertByUnique' and 'insertByName', this function does not provide any static guarantees,
   67 so you can for example insert by a type-level name in a map from term-level uniques. -}
   68 insertByNameIndex
   69   :: (HasUnique name unique1, Coercible unique2 Unique)
   70   => name
   71   -> a
   72   -> UniqueMap unique2 a
   73   -> UniqueMap unique2 a
   74 insertByNameIndex = insertByUnique . coerce . view unique
   75 
   76 -- | Convert a 'Foldable' into a 'UniqueMap' using the given insertion function.
   77 fromFoldable
   78   :: Foldable f
   79   => (i -> a -> UniqueMap unique a -> UniqueMap unique a)
   80   -> f (i, a)
   81   -> UniqueMap unique a
   82 fromFoldable ins = List.foldl' (flip $ uncurry ins) mempty
   83 
   84 -- | Convert a 'Foldable' with uniques into a 'UniqueMap'.
   85 fromUniques :: Foldable f => Coercible Unique unique => f (unique, a) -> UniqueMap unique a
   86 fromUniques = fromFoldable insertByUnique
   87 
   88 -- | Convert a 'Foldable' with names into a 'UniqueMap'.
   89 fromNames :: Foldable f => HasUnique name unique => f (name, a) -> UniqueMap unique a
   90 fromNames = fromFoldable insertByName
   91 
   92 -- | Look up a value by a unique.
   93 lookupUnique :: Coercible unique Unique => unique -> UniqueMap unique a -> Maybe a
   94 lookupUnique uniq = IM.lookup (coerce uniq) . unUniqueMap
   95 
   96 -- | Look up a value by the unique of a name.
   97 lookupName :: HasUnique name unique => name -> UniqueMap unique a -> Maybe a
   98 lookupName = lookupUnique . view unique
   99 
  100 restrictKeys :: UniqueMap unique v -> UniqueSet unique -> UniqueMap unique v
  101 restrictKeys (UniqueMap m) (UniqueSet s) =
  102   UniqueMap $ IM.restrictKeys m s
  103
