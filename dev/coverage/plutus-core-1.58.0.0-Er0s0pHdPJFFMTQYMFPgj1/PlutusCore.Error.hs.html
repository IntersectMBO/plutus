   44 -- | Lifts an 'Either' into an error context where we can embed the 'Left' value into the error.
   45 throwingEither :: MonadError e m => AReview e t -> Either t a -> m a
   46 throwingEither r e = case e of
   47   Left t -> throwing r t
   48   Right v -> pure v

   94 {-| This is needed for nice kind/type checking error messages. In some cases the type checker knows
   95 the exact type that an expression has to have for type checking to succeed (see any of
   96 'checkTypeM' functions and its usages), which is what 'ExpectedExact' is suitable for. In other
   97 cases the type checker only cares about the shape of the inferred type, e.g. the type checker
   98 knows that the type of a function has to be @dom -> cod@ for type checking to succeed, but it
   99 doesn't yet care what @dom@ and @cod@ exactly are. Which is what 'ExpectedShape' is useful for as
  100 it allows one to specify the shape of an expected type with some existential variables in it when
  101 it's impossible to provide an exact type. -}
  102 data ExpectedShapeOr a
  103   = ExpectedShape
  104       !T.Text
  105       -- ^ The expected shape potentially referencing existential variables.
  106       ![T.Text]
  107       -- ^ The list of existential variables.
  108   | ExpectedExact !a
  109   deriving stock (Show, Eq, Generic, Functor)
  110   deriving anyclass (NFData)
  111 
  112 data TypeError term uni fun ann
  113   = KindMismatch
  114       !ann
  115       !(Type TyName uni ())
  116       !(ExpectedShapeOr (Kind ()))
  117       -- ^ The expected type or the shape of a kind.
  118       !(Kind ())
  119       -- ^ The actual kind.
  120   | TypeMismatch
  121       !ann
  122       !term
  123       !(ExpectedShapeOr (Type TyName uni ()))
  124       -- ^ The expected type or the shape of a type.
  125       !(Normalized (Type TyName uni ()))
  126       -- ^ The actual type.
  127   | TyNameMismatch !ann !TyName !TyName
  128   | NameMismatch !ann !Name !Name
  129   | FreeTypeVariableE !ann !TyName
  130   | FreeVariableE !ann !Name
  131   | UnknownBuiltinFunctionE !ann !fun
  132   | UnsupportedCaseBuiltin !ann !T.Text
  133   deriving stock (Show, Eq, Generic, Functor)
  134   deriving anyclass (NFData)

  241 {-| Align a list of existential variables in a pretty way.
  242 
  243 >>> :set -XOverloadedStrings
  244 >>> existentialVars []
  245 >>> existentialVars ["'a'"]
  246  for some 'a'
  247 >>> existentialVars ["'a'", "'b'"]
  248  for some 'a' and 'b'
  249 >>> existentialVars ["'a'", "'b'", "'c'"]
  250  for some 'a', 'b' and 'c' -}
  251 existentialVars :: [Doc ann] -> Doc ann
  252 existentialVars [] = ""
  253 existentialVars (x0 : xs0) = " for some " <> go x0 xs0
  254   where
  255     go x [] = x
  256     go x [y] = x <> " and " <> y
  257     go x (y : xs) = x <> ", " <> go y xs
  258 
  259 instance PrettyBy PrettyConfigPlc a => PrettyBy PrettyConfigPlc (ExpectedShapeOr a) where
  260   prettyBy _ (ExpectedShape shape vars) =
  261     squotes (sexp (pretty shape) []) <> existentialVars (map (squotes . pretty) vars)
  262   prettyBy config (ExpectedExact thing) = squotes (prettyBy config thing)

  264 instance
  265   (Pretty term, PrettyUni uni, Pretty fun, Pretty ann)
  266   => PrettyBy PrettyConfigPlc (TypeError term uni fun ann)
  267   where
  268   prettyBy config (KindMismatch ann ty shapeOrK k') =
  269     "Kind mismatch at"
  270       <+> pretty ann
  271       <> hardline
  272       <> "Expected a type of kind"
  273       <> hardline
  274       <> indent 2 (prettyBy config shapeOrK)
  275       <> hardline
  276       <> "But found one of kind"
  277       <> hardline
  278       <> indent 2 (squotes (prettyBy config k'))
  279       <> hardline
  280       <> "Namely,"
  281       <> hardline
  282       <> indent 2 (squotes (prettyBy config ty))
  283   prettyBy config (TypeMismatch ann t shapeOrTy ty') =
  284     "Type mismatch at"
  285       <+> pretty ann
  286       <> hardline
  287       <> "Expected a term of type"
  288       <> hardline
  289       <> indent 2 (prettyBy config shapeOrTy)
  290       <> hardline
  291       <> "But found one of type"
  292       <> hardline
  293       <> indent 2 (squotes (prettyBy config ty'))
  294       <> ( if _pcpoCondensedErrors (_pcpOptions config) == CondensedErrorsYes
  295              then mempty
  296              -- TODO: we should use prettyBy here but the problem is
  297              -- that `instance PrettyClassic PIR.Term` whereas `instance PrettyPLC PLC.Term`
  298              else hardline <> "Namely," <> hardline <> indent 2 (squotes (pretty t))
  299          )
  300   prettyBy config (FreeTypeVariableE ann name) =
  301     "Free type variable at " <+> pretty ann <+> ": " <+> prettyBy config name
  302   prettyBy config (FreeVariableE ann name) =
  303     "Free variable at " <+> pretty ann <+> ": " <+> prettyBy config name
  304   prettyBy _ (UnknownBuiltinFunctionE ann fun) =
  305     "An unknown built-in function at" <+> pretty ann <> ":" <+> pretty fun
  306   prettyBy _ (TyNameMismatch ann name1 name2) =
  307     hsep
  308       [ "Type-level name mismatch at"
  309       , pretty ann <> ":"
  310       , pretty $ name1 ^. theText
  311       , "is in scope, but"
  312       , pretty $ name2 ^. theText
  313       , "having the same Unique"
  314       , pretty $ name1 ^. theUnique
  315       , "is attempted to be referenced"
  316       ]
  317   prettyBy _ (NameMismatch ann name1 name2) =
  318     hsep
  319       [ "Term-level name mismatch at"
  320       , pretty ann <> ":"
  321       , pretty $ name1 ^. theText
  322       , "is in scope, but"
  323       , pretty $ name2 ^. theText
  324       , "having the same Unique"
  325       , pretty $ name1 ^. theUnique
  326       , "is attempted to be referenced"
  327       ]
  328   prettyBy _ (UnsupportedCaseBuiltin ann err) =
  329     hsep
  330       [ "Unsupported 'case' of a value of a built-in type at"
  331       , pretty ann <> ":"
  332       , hardline
  333       , pretty err
  334       ]
