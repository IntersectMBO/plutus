   48 {-| Construct an 'ExBudgetMode' out of a function returning a value of the budgeting state type.
   49 The value then gets added to the current state via @(<>)@. -}
   50 monoidalBudgeting
   51   :: Monoid cost => (ExBudgetCategory fun -> ExBudget -> cost) -> ExBudgetMode cost uni fun
   52 monoidalBudgeting toCost = ExBudgetMode $ do
   53   costRef <- newSTRef mempty
   54   budgetRef <- newSTRef mempty
   55   let spend key budgetToSpend = CekM $ do
   56         modifySTRef' costRef (<> toCost key budgetToSpend)
   57         modifySTRef' budgetRef (<> budgetToSpend)
   58       spender = CekBudgetSpender spend
   59       cumulative = readSTRef budgetRef
   60       final = readSTRef costRef
   61   pure $ ExBudgetInfo spender final cumulative

   63 -- | For calculating the cost of execution by counting up using the 'Monoid' instance of 'ExBudget'.
   64 newtype CountingSt = CountingSt ExBudget
   65   deriving stock (Eq, Show)
   66   deriving newtype (Semigroup, Monoid, PrettyBy config, NFData)
   67 
   68 instance Pretty CountingSt where
   69   pretty (CountingSt budget) = parens $ "required budget:" <+> pretty budget <> line

   71 -- | For calculating the cost of execution.
   72 counting :: ExBudgetMode CountingSt uni fun
   73 counting = monoidalBudgeting $ const CountingSt

   75 {-| For a detailed report on what costs how much + the same overall budget that 'Counting' gives.
   76 The (derived) 'Monoid' instance of 'CekExTally' is the main piece of the machinery. -}
   77 newtype CekExTally fun = CekExTally (MonoidalHashMap (ExBudgetCategory fun) ExBudget)
   78   deriving stock (Eq, Generic, Show)
   79   deriving (Semigroup, Monoid) via (GenericSemigroupMonoid (CekExTally fun))
   80   deriving anyclass (NFData)
   81   deriving (PrettyBy config) via (IgnorePrettyConfig (CekExTally fun))
   82 
   83 instance (Show fun, Ord fun) => Pretty (CekExTally fun) where
   84   pretty (CekExTally m) =
   85     let om = Map.fromList $ HashMap.toList m
   86      in parens $
   87           encloseSep "{" "}" "| " $
   88             fmap group $
   89               Map.elems $
   90                 imap (\k v -> (pretty k <+> "causes" <+> group (pretty v))) om
   91 
   92 data TallyingSt fun = TallyingSt (CekExTally fun) ExBudget
   93   deriving stock (Eq, Show, Generic)
   94   deriving (Semigroup, Monoid) via (GenericSemigroupMonoid (TallyingSt fun))
   95   deriving anyclass (NFData)
   96   deriving (PrettyBy config) via (IgnorePrettyConfig (TallyingSt fun))
   97 
   98 instance (Show fun, Ord fun) => Pretty (TallyingSt fun) where
   99   pretty (TallyingSt tally budget) =
  100     parens $
  101       fold
  102         [ "{ tally: "
  103         , pretty tally
  104         , line
  105         , "| budget: "
  106         , pretty budget
  107         , line
  108         , "}"
  109         ]

  111 -- | For a detailed report on what costs how much + the same overall budget that 'Counting' gives.
  112 tallying :: Hashable fun => ExBudgetMode (TallyingSt fun) uni fun
  113 tallying =
  114   monoidalBudgeting $ \key budgetToSpend ->
  115     TallyingSt (CekExTally $ singleton key budgetToSpend) budgetToSpend

  125 -- | For execution, to avoid overruns.
  126 restricting
  127   :: ThrowableBuiltins uni fun
  128   => ExRestrictingBudget -> ExBudgetMode RestrictingSt uni fun
  129 restricting (ExRestrictingBudget initB@(ExBudget cpuInit memInit)) = ExBudgetMode $ do
  130   -- We keep the counters in a PrimArray. This is better than an STRef since it stores its contents unboxed.
  131   --
  132   -- If we don't specify the element type then GHC has difficulty inferring it, but it's
  133   -- annoying to specify the monad, since it refers to the 's' which is not in scope.
  134   ref <- newPrimArray @_ @SatInt 2
  135   let
  136     cpuIx = 0
  137     memIx = 1
  138     readCpu = coerce @_ @ExCPU <$> readPrimArray ref cpuIx
  139     writeCpu cpu = writePrimArray ref cpuIx $ coerce cpu
  140     readMem = coerce @_ @ExMemory <$> readPrimArray ref memIx
  141     writeMem mem = writePrimArray ref memIx $ coerce mem
  142 
  143   writeCpu cpuInit
  144   writeMem memInit
  145   let
  146     spend _ (ExBudget cpuToSpend memToSpend) = do
  147       cpuLeft <- CekM readCpu
  148       memLeft <- CekM readMem
  149       let cpuLeft' = cpuLeft - cpuToSpend
  150       let memLeft' = memLeft - memToSpend
  151       -- Note that even if we throw an out-of-budget error, we still need to record
  152       -- what the final state was.
  153       CekM $ writeCpu cpuLeft'
  154       CekM $ writeMem memLeft'
  155       when (cpuLeft' < 0 || memLeft' < 0) $ do
  156         let
  157           -- You'd think whether the budget is computed strictly or not before throwing
  158           -- an error isn't important, but GHC refuses to unbox the second argument of
  159           -- @spend@ without this bang. Bangs on @cpuLeft'@ and @memLeft'@ don't help
  160           -- either as those are forced by 'writeCpu' and 'writeMem' anyway. Go figure.
  161           !budgetLeft = ExBudget cpuLeft' memLeft'
  162         throwError $
  163           ErrorWithCause
  164             (OperationalError (CekOutOfExError $ ExRestrictingBudget budgetLeft))
  165             Nothing
  166     spender = CekBudgetSpender spend
  167     remaining = ExBudget <$> readCpu <*> readMem
  168     cumulative = do
  169       r <- remaining
  170       pure $ initB `minusExBudget` r
  171     final = RestrictingSt . ExRestrictingBudget <$> remaining
  172   pure $ ExBudgetInfo spender final cumulative
