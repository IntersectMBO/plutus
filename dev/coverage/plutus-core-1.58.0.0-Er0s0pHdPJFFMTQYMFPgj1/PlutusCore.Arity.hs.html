   11 -- | Is the next argument a term or a type?
   12 data Param
   13   = TermParam
   14   | TypeParam
   15   deriving stock (Show, Eq)
   16 
   17 instance Pretty Param where
   18   pretty = viaShow
   19 
   20 {-|
   21 The (syntactic) arity of a term. That is, a record of the arguments that the
   22 term expects before it may do some work. Since we have both type and lambda
   23 abstractions, this is not a simple argument count, but rather a list of values
   24 indicating whether the next argument should be a term or a type.
   25 
   26 Note that this is the syntactic arity, i.e. it just corresponds to the number of
   27 syntactic lambda and type abstractions on the outside of the term. It is thus
   28 an under-approximation of how many arguments the term may need.
   29 e.g. consider the term @let id = \x -> x in id@: the variable @id@ has syntactic
   30 arity @[]@, but does in fact need an argument before it does any work. -}
   31 type Arity = [Param]
   32 
   33 -- | Get the 'Arity' from a 'TypeScheme'.
   34 typeSchemeArity :: TypeScheme val args res -> Arity
   35 typeSchemeArity TypeSchemeResult {} = []
   36 typeSchemeArity (TypeSchemeArrow sch) = TermParam : typeSchemeArity sch
   37 typeSchemeArity (TypeSchemeAll _ sch) = TypeParam : typeSchemeArity sch
   38 
   39 -- | Get the arity of a builtin function from the 'PLC.BuiltinSemanticsVariant'.
   40 builtinArity
   41   :: forall uni fun
   42    . ToBuiltinMeaning uni fun
   43   => Proxy uni
   44   -> BuiltinSemanticsVariant fun
   45   -> fun
   46   -> Arity
   47 builtinArity _ semvar fun =
   48   case toBuiltinMeaning @uni @fun @(Term TyName Name uni fun ()) semvar fun of
   49     BuiltinMeaning sch _ _ -> typeSchemeArity sch
