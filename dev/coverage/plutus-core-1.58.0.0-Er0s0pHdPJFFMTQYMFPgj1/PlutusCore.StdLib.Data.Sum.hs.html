   17 {-| 'Either' as a PLC type.
   18 
   19 > \(a b :: *) -> all (r :: *). (a -> r) -> (b -> r) -> r -}
   20 sum :: Type TyName uni ()
   21 sum = runQuote $ do
   22   a <- freshTyName "a"
   23   b <- freshTyName "b"
   24   r <- freshTyName "r"
   25   return
   26     . TyLam () a (Type ())
   27     . TyLam () b (Type ())
   28     . TyForall () r (Type ())
   29     . TyFun () (TyFun () (TyVar () a) $ TyVar () r)
   30     . TyFun () (TyFun () (TyVar () b) $ TyVar () r)
   31     $ TyVar () r

   33 {-| 'Left' as a PLC term.
   34 
   35 > /\(a b :: *) -> \(x : a) -> /\(r :: *) -> \(f : a -> r) -> (g : b -> r) -> f x -}
   36 left :: TermLike term TyName Name uni fun => term ()
   37 left = runQuote $ do
   38   a <- freshTyName "a"
   39   b <- freshTyName "b"
   40   x <- freshName "x"
   41   r <- freshTyName "r"
   42   f <- freshName "f"
   43   g <- freshName "g"
   44   return
   45     . tyAbs () a (Type ())
   46     . tyAbs () b (Type ())
   47     . lamAbs () x (TyVar () a)
   48     . tyAbs () r (Type ())
   49     . lamAbs () f (TyFun () (TyVar () a) $ TyVar () r)
   50     . lamAbs () g (TyFun () (TyVar () b) $ TyVar () r)
   51     . apply () (var () f)
   52     $ var () x
