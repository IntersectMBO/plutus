   33 {-| Pattern matching on built-in lists. @matchList {a} xs@ on built-in lists is
   34 equivalent to @unwrap xs@ on lists defined in PLC itself (hence why we bind @r@ after @xs@).
   35 
   36 Either
   37 
   38 > /\(a :: *) -> \(xs : list a) -> /\(r :: *) -> (z : r) (f : a -> list a -> r) ->
   39 >     matchList
   40 >         {a}
   41 >         {r}
   42 >         z
   43 >         f
   44 >         xs
   45 
   46 or
   47 
   48 > /\(a :: *) -> \(xs : list a) -> /\(r :: *) -> (z : r) (f : a -> list a -> r) ->
   49 >     chooseList
   50 >         {a}
   51 >         {() -> r}
   52 >         xs
   53 >         (\(u : ()) -> z)
   54 >         (\(u : ()) -> f (head {a} xs) (tail {a} xs))
   55 >         ()
   56 
   57 depending on the 'MatchOption' argument. -}
   58 matchList :: TermLike term TyName Name DefaultUni DefaultFun => MatchOption -> term ()
   59 matchList optMatch = runQuote $ do
   60   a <- freshTyName "a"
   61   r <- freshTyName "r"
   62   xs <- freshName "xs"
   63   z <- freshName "z"
   64   f <- freshName "f"
   65   u <- freshName "u"
   66   let listA = TyApp () list $ TyVar () a
   67       funAtXs fun = apply () (tyInst () (builtin () fun) $ TyVar () a) $ var () xs
   68   return
   69     . tyAbs () a (Type ())
   70     . lamAbs () xs listA
   71     . tyAbs () r (Type ())
   72     . lamAbs () z (TyVar () r)
   73     . lamAbs () f (TyFun () (TyVar () a) . TyFun () listA $ TyVar () r)
   74     $ case optMatch of
   75       UseChoose ->
   76         mkIterAppNoAnn
   77           ( mkIterInstNoAnn
   78               (builtin () ChooseList)
   79               [ TyVar () a
   80               , TyFun () unit $ TyVar () r
   81               ]
   82           )
   83           [ var () xs
   84           , lamAbs () u unit $ var () z
   85           , lamAbs () u unit $
   86               mkIterAppNoAnn
   87                 (var () f)
   88                 [funAtXs HeadList, funAtXs TailList]
   89           , unitval
   90           ]

   92 {-|  @foldr@ over built-in lists.
   93 
   94 > /\(a :: *) (r :: *) -> \(f : a -> r -> r) (z : r) ->
   95 >     fix {list a} {r} \(rec : list a -> r) (xs : list a) ->
   96 >         matchList {a} xs {r} z \(x : a) (xs' : list a) -> f x (rec xs') -}
   97 foldrList :: TermLike term TyName Name DefaultUni DefaultFun => MatchOption -> term ()
   98 foldrList optMatch = runQuote $ do
   99   a <- freshTyName "a"
  100   r <- freshTyName "r"
  101   f <- freshName "f"
  102   z <- freshName "z"
  103   rec <- freshName "rec"
  104   xs <- freshName "xs"
  105   x <- freshName "x"
  106   xs' <- freshName "xs'"
  107   let listA = TyApp () list $ TyVar () a
  108       unwrap' ann = apply ann . tyInst () (matchList optMatch) $ TyVar () a
  109   -- Copypasted verbatim from @foldrList@ over Scott-encoded lists.
  110   return
  111     . tyAbs () a (Type ())
  112     . tyAbs () r (Type ())
  113     . lamAbs () f (TyFun () (TyVar () a) . TyFun () (TyVar () r) $ TyVar () r)
  114     . lamAbs () z (TyVar () r)
  115     . apply () (mkIterInstNoAnn fix [listA, TyVar () r])
  116     . lamAbs () rec (TyFun () listA $ TyVar () r)
  117     . lamAbs () xs listA
  118     . apply () (apply () (tyInst () (unwrap' () (var () xs)) $ TyVar () r) $ var () z)
  119     . lamAbs () x (TyVar () a)
  120     . lamAbs () xs' listA
  121     $ mkIterAppNoAnn
  122       (var () f)
  123       [ var () x
  124       , apply () (var () rec) $ var () xs'
  125       ]

  127 {-|  'foldl\'' as a PLC term.
  128 
  129 > /\(a :: *) (r :: *) -> \(f : r -> a -> r) ->
  130 >     fix {r} {list a -> r} \(rec : r -> list a -> r) (z : r) (xs : list a) ->
  131 >         matchList {a} xs {r} z \(x : a) (xs' : list a) -> rec (f z x) xs' -}
  132 foldList :: TermLike term TyName Name DefaultUni DefaultFun => MatchOption -> term ()
  133 foldList optMatch = runQuote $ do
  134   a <- freshTyName "a"
  135   r <- freshTyName "r"
  136   f <- freshName "f"
  137   rec <- freshName "rec"
  138   z <- freshName "z"
  139   xs <- freshName "xs"
  140   x <- freshName "x"
  141   xs' <- freshName "xs'"
  142   let listA = TyApp () list $ TyVar () a
  143       unwrap' ann = apply ann . tyInst () (matchList optMatch) $ TyVar () a
  144   return
  145     . tyAbs () a (Type ())
  146     . tyAbs () r (Type ())
  147     . lamAbs () f (TyFun () (TyVar () r) . TyFun () (TyVar () a) $ TyVar () r)
  148     . apply () (mkIterInstNoAnn fix [TyVar () r, TyFun () listA $ TyVar () r])
  149     . lamAbs () rec (TyFun () (TyVar () r) . TyFun () listA $ TyVar () r)
  150     . lamAbs () z (TyVar () r)
  151     . lamAbs () xs listA
  152     . apply () (apply () (tyInst () (unwrap' () (var () xs)) $ TyVar () r) $ var () z)
  153     . lamAbs () x (TyVar () a)
  154     . lamAbs () xs' listA
  155     . mkIterAppNoAnn (var () rec)
  156     $ [ mkIterAppNoAnn (var () f) [var () z, var () x]
  157       , var () xs'
  158       ]

  160 -- > foldList {integer} {integer} addInteger 0
  161 sum :: TermLike term TyName Name DefaultUni DefaultFun => MatchOption -> term ()
  162 sum optMatch = runQuote $ do
  163   let int = mkTyBuiltin @_ @Integer ()
  164       add = builtin () AddInteger
  165   return
  166     . mkIterAppNoAnn (mkIterInstNoAnn (foldList optMatch) [int, int])
  167     $ [add, mkConstant @Integer () 0]
