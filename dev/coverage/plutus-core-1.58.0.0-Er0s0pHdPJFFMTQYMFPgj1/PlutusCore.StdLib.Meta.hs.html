   18 -- | Convert an 'Integer' to a @nat@. TODO: convert PLC's @integer@ to @nat@ instead.
   19 metaIntegerToNat :: TermLike term TyName Name uni fun => Integer -> term ()
   20 metaIntegerToNat n
   21   | n < 0 = Prelude.error $ "getBuiltinIntegerToNat: negative argument: " ++ show n
   22   | otherwise = go n
   23   where
   24     go 0 = zero
   25     go m = apply () Plc.succ $ go (m - 1)
   26 
   27 -- | Convert a Haskell 'Either' to a PLC @sum@.
   28 metaEitherToSum
   29   :: TermLike term TyName Name uni fun
   30   => Type TyName uni ()
   31   -> Type TyName uni ()
   32   -> Either (term ()) (term ())
   33   -> term ()
   34 metaEitherToSum a b (Left x) = apply () (mkIterInstNoAnn left [a, b]) x
   35 metaEitherToSum a b (Right y) = apply () (mkIterInstNoAnn right [a, b]) y
   36 
   37 -- | Convert a Haskell list of 'Term's to a PLC @list@.
   38 metaListToScottList
   39   :: TermLike term TyName Name uni fun => Type TyName uni () -> [term ()] -> term ()
   40 metaListToScottList ty =
   41   foldr
   42     (\x xs -> mkIterAppNoAnn (tyInst () cons ty) [x, xs])
   43     (tyInst () nil ty)
