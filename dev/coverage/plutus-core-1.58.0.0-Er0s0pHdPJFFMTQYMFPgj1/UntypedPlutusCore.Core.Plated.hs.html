   22 -- | Get all the direct constants of the given 'Term' from 'Constant's.
   23 termConstants :: Traversal' (Term name uni fun ann) (Some (ValueOf uni))
   24 termConstants f term0 = case term0 of
   25   Constant ann val -> Constant ann <$> f val
   26   Var {} -> pure term0
   27   LamAbs {} -> pure term0
   28   Error {} -> pure term0
   29   Apply {} -> pure term0
   30   Force {} -> pure term0
   31   Delay {} -> pure term0
   32   Builtin {} -> pure term0
   33   Constr {} -> pure term0
   34   Case {} -> pure term0
   35 
   36 -- | Get all the direct child 'name a's of the given 'Term' from 'LamAbs'es.
   37 termBinds :: Traversal' (Term name uni fun ann) name
   38 termBinds f = \case
   39   LamAbs ann n t -> f n <&> \n' -> LamAbs ann n' t
   40   x -> pure x
   41 
   42 -- | Get all the direct child 'name a's of the given 'Term' from 'Var's.
   43 termVars :: Traversal' (Term name uni fun ann) name
   44 termVars f = \case
   45   Var ann n -> Var ann <$> f n
   46   x -> pure x
   47 
   48 -- | Get all the direct child 'Unique's of the given 'Term'.
   49 termUniques :: HasUniques (Term name uni fun ann) => Traversal' (Term name uni fun ann) Unique
   50 termUniques f = \case
   51   LamAbs ann n t -> theUnique f n <&> \n' -> LamAbs ann n' t
   52   Var ann n -> theUnique f n <&> Var ann
   53   x -> pure x
   54 
   55 -- | Get all the direct child 'Term's of the given 'Term'.
   56 termSubterms :: Traversal' (Term name uni fun ann) (Term name uni fun ann)
   57 termSubterms f = \case
   58   LamAbs ann n t -> LamAbs ann n <$> f t
   59   Apply ann t1 t2 -> Apply ann <$> f t1 <*> f t2
   60   Delay ann t -> Delay ann <$> f t
   61   Force ann t -> Force ann <$> f t
   62   Constr ann i args -> Constr ann i <$> traverse f args
   63   Case ann arg cs -> Case ann <$> f arg <*> traverse f cs
   64   e@Error {} -> pure e
   65   v@Var {} -> pure v
   66   c@Constant {} -> pure c
   67   b@Builtin {} -> pure b
   68 {-# INLINE termSubterms #-}
   69 
   70 -- | Get all the transitive child 'Constant's of the given 'Term'.
   71 termConstantsDeep :: Fold (Term name uni fun ann) (Some (ValueOf uni))
   72 termConstantsDeep = termSubtermsDeep . termConstants

   74 -- | Get all the transitive child 'Term's of the given 'Term'.
   75 termSubtermsDeep :: Fold (Term name uni fun ann) (Term name uni fun ann)
   76 termSubtermsDeep = cosmosOf termSubterms

   78 -- | Get all the transitive child 'Unique's of the given 'Term'.

   80 termUniquesDeep = termSubtermsDeep . termUniques
