   33 {- Note [MakeKnown in TypeSchemeArrow]
   34 There's a @MakeKnown val arg@ constrained packed in the 'TypeSchemeArrow' constructor. It's not
   35 supposed to be there, but unfortunately, in the @Generators@ tests we use 'TypeScheme' for
   36 generation of arbitrary arguments of builtins and that requires 'makeKnown', which makes us have
   37 the 'MakeKnown' in 'TypeSchemeArrow'.
   38 
   39 The solution is to fix the @Generators@ tests. Explicitly constraining @args@ outside of
   40 'TypeScheme' sounds like a promising strategy. Maybe we could just delete those tests altogether.
   41 
   42 However it's also worth considering untangling 'RuntimeScheme' from 'TypeScheme' and generating the
   43 two in parallel, so that we only need to optimize the former. Then we will be able to afford having
   44 any kind of nonsense in 'TypeScheme'. Another reason for that would be the fact that Core output
   45 has 'typeSchemeToRuntimeScheme' all over the place as it can't be inlined due to being a recursive
   46 function, which we can't turn into an inlinable class method, because the indices of 'TypeScheme'
   47 don't reflect its structure due to the 'TypeSchemeAll' constructor not being reflected at the type
   48 level in any way. It's unlikely that having those 'typeSchemeToRuntimeScheme' has any impact on
   49 performance, because they're only evaluated once during initialization, but it certainly has impact
   50 on readability of the Core output.
   51 -}

   53 -- We have these 'Typeable' constraints here just for the generators tests. It's fine since
   54 -- 'TypeScheme' is not used for evaluation and so we can shove into 'TypeScheme' whatever we want.
   55 {-| The type of type schemes of built-in functions.
   56 @args@ is a list of types of arguments, @res@ is the resulting type.
   57 E.g. @Text -> Bool -> Integer@ is encoded as @TypeScheme val [Text, Bool] Integer@. -}
   58 data TypeScheme val (args :: [GHC.Type]) res where
   59   TypeSchemeResult
   60     :: (Typeable res, KnownTypeAst TyName (UniOf val) res, MakeKnown val res)
   61     => TypeScheme val '[] res
   62   TypeSchemeArrow
   63     :: (Typeable arg, KnownTypeAst TyName (UniOf val) arg, MakeKnown val arg, ReadKnown val arg)
   64     => TypeScheme val args res -> TypeScheme val (arg ': args) res
   65   TypeSchemeAll
   66     :: (KnownSymbol text, KnownNat uniq, KnownKind kind)
   67     => Proxy '(text, uniq, kind)
   68     -> TypeScheme val args res
   69     -> TypeScheme val args res
   70 
   71 argProxy :: TypeScheme val (arg ': args) res -> Proxy arg
   72 argProxy _ = Proxy
   73 
   74 -- | Convert a 'TypeScheme' to the corresponding 'Type'.
   75 typeSchemeToType :: TypeScheme val args res -> Type TyName (UniOf val) ()
   76 typeSchemeToType sch@TypeSchemeResult = toTypeAst sch
   77 typeSchemeToType sch@(TypeSchemeArrow schB) =
   78   TyFun () (toTypeAst $ argProxy sch) $ typeSchemeToType schB
   79 typeSchemeToType (TypeSchemeAll (_ :: Proxy '(text, uniq, kind)) schB) =
   80   let text = Text.pack $ symbolVal @text Proxy
   81       uniq = fromIntegral $ natVal @uniq Proxy
   82       a = TyName $ Name text $ Unique uniq
   83    in TyForall () a (demoteKind $ knownKind @kind) $ typeSchemeToType schB
   84 
   85 -- The precedence of @->@ is @-1@, which is why this number appears in the implementation of the
   86 -- instance.
   87 instance Show (TypeScheme val args res) where
   88   showsPrec p sch@TypeSchemeResult =
   89     showParen (p > 0) $
   90       showsPrec (-1) (typeRep sch)
   91   showsPrec p sch@(TypeSchemeArrow schB) =
   92     showParen (p > 0) $ -- @0@ is to account for associativity, see https://stackoverflow.com/a/43639618
   93       showsPrec 0 (typeRep $ argProxy sch)
   94         . showString " -> "
   95         . showsPrec (-1) schB
   96   showsPrec p (TypeSchemeAll (_ :: Proxy '(text, uniq, kind)) schB) =
   97     showParen (p > 0) $
   98       showString "forall "
   99         . showString (symbolVal @text Proxy)
  100         . showString ". "
  101         . showsPrec 0 schB
