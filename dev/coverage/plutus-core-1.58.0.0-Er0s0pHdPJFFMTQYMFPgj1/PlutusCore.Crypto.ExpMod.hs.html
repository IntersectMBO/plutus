    8 import PlutusCore.Builtin

   10 import GHC.Natural
   11 import GHC.Num.Integer
   12 
   13 {-| Modular exponentiation.  This uses GHC.Num.integerPowMod#, which gives the
   14 wrong answer in some cases.  TODO: we'll be able to remove some of the guards
   15 when/if integerPowMod# gets fixed. -}
   16 expMod :: Integer -> Integer -> Natural -> BuiltinResult Natural
   17 expMod b e m
   18   | m <= 0 = fail "expMod: invalid modulus"
   19   -- \^ We can't have m<0 when m is a Natural, but we may as well be paranoid.
   20   | m == 1 = pure 0
   21   -- \^ Just in case: GHC.Num.Integer.integerRecip# gets this wrong.  Note that 0
   22   -- is invertible modulo 1, with inverse 0.
   23   | b == 0 && e < 0 = failNonInvertible 0 m
   24   -- \^ integerPowMod# incorrectly returns 0 in this case.
   25   | otherwise =
   26       case integerPowMod# b e m of
   27         (# n | #) -> pure n
   28         (# | () #) -> failNonInvertible b m
   29   where
   30     failNonInvertible :: Integer -> Natural -> BuiltinResult Natural
   31     failNonInvertible b1 m1 =
   32       fail ("expMod: " ++ (show b1) ++ " is not invertible modulo " ++ (show m1))
   33 {-# INLINE expMod #-}
