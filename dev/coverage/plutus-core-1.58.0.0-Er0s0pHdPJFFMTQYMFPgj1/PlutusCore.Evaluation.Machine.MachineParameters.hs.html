   20 {-| We need to account for the costs of evaluator steps and also built-in function
   21    evaluation.  The models for these have different structures and are used in
   22    different parts of the code, so inside the valuator we pass separate objects
   23    about most of the time .  It's convenient for clients of the evaluator to
   24    only have to worry about a single object, so the CostModel type bundles the
   25    two together.  We could conceivably have different evaluators with different
   26    internal costs, so we keep the machine costs abstract.  The model for Cek
   27    machine steps is in UntypedPlutusCore.Evaluation.Machine.Cek.CekMachineCosts. -}
   28 data CostModel machinecosts builtincosts
   29   = CostModel
   30   { _machineCostModel :: machinecosts
   31   , _builtinCostModel :: builtincosts
   32   }
   33   deriving stock (Eq, Show)
   34 
   35 makeLenses ''CostModel
   36 
   37 {-| The part of 'MachineParameters' that is individual for each semantics variant of 'DefaultFun'.
   38 
   39 'CaserBuiltin' isn't included, because it only explicitly depends on the protocol version and not
   40 the language version (even though there's an implicit dependency on the language version: older
   41 languages don't support 'Case' in general, but it's safe to ignore that, because support for
   42 'Case' is controlled by the AST version, which is a separate check during deserialisation). -}
   43 data MachineVariantParameters machineCosts fun val
   44   = MachineVariantParameters
   45   { machineCosts :: machineCosts
   46   , builtinsRuntime :: BuiltinsRuntime fun val
   47   }
   48   deriving stock (Generic)
   49   deriving anyclass (NFData)
   50 
   51 {-| At execution time we need a 'BuiltinsRuntime' object which includes both the cost model for
   52 builtins and their denotations. This bundles one of those together with the cost model for evaluator
   53 steps and a 'CaserBuiltin' specifying how casing on values of built-in types works.
   54 The @val@ type will be 'CekValue' when we're using this with the CEK machine. -}
   55 data MachineParameters machineCosts fun val
   56   = MachineParameters
   57   { machineCaserBuiltin :: CaserBuiltin (UniOf val)
   58   , machineVariantParameters :: MachineVariantParameters machineCosts fun val
   59   }
   60   deriving stock (Generic)
   61   deriving anyclass (NFData)
   62 
   63 -- For some reason the generic instance gives incorrect nothunk errors,
   64 -- see https://github.com/input-output-hk/nothunks/issues/24
   65 instance (NoThunks machinecosts, Bounded fun, Enum fun) => NoThunks (MachineVariantParameters machinecosts fun val) where
   66   wNoThunks ctx (MachineVariantParameters costs runtime) =
   67     allNoThunks [noThunks ctx costs, noThunks ctx runtime]

   69 instance (NoThunks machinecosts, Bounded fun, Enum fun) => NoThunks (MachineParameters machinecosts fun val) where
   70   wNoThunks ctx (MachineParameters caser varPars) =
   71     allNoThunks [noThunks ctx caser, noThunks ctx varPars]

   73 {- Note [The CostingPart constraint in mkMachineVariantParameters]
   74 Discharging the @CostingPart uni fun ~ builtincosts@ constraint in 'mkMachineParameters' causes GHC
   75 to fail to inline the function at its call site regardless of the @INLINE@ pragma and an explicit
   76 'inline' call.
   77 
   78 We think that this is because discharging the @CostingPart uni fun ~ builtincosts@ constraint in the
   79 type of 'mkMachineParameters' somehow causes it to be compiled into @expr `Cast@` co@ , where @co@'s
   80 type is
   81 
   82     (CostingPart DefaultUni DefaultFun) ... ~R# ... BuiltinCostModel ...
   83 
   84 and this fails to inline, because in order for @inline f@ to work, @f@ must be compiled into either
   85 a @Var@, or an @App@ whose head is a @Var@, according to https://gitlab.haskell.org/ghc/ghc/-/blob/1f02b7430b2fbab403d7ffdde9cfd006e884678e/compiler/prelude/PrelRules.hs#L1442-1449
   86 And if @f@ is compiled into a @Cast@ like 'mkMachineParameters' with the discharged constraint, then
   87 inlining won't work. We don't know why it's implemented this way in GHC.
   88 
   89 It seems fully applying @f@ helps, i.e.
   90 
   91     - inline mkMachineParameters unlMode <$> <...>
   92     + (\x -> inline (mkMachineParameters unlMode x)) <$> <...>
   93 
   94 which makes sense: if @f@ receives all its type and term args then there's less reason to throw a
   95 @Cast@ in there.
   96 -}

   98 -- See Note [Inlining meanings of builtins].
   99 -- | This just uses 'toBuiltinsRuntime' function to convert a BuiltinCostModel to a BuiltinsRuntime.
  100 mkMachineVariantParameters
  101   :: ( -- WARNING: do not discharge the equality constraint as that causes GHC to fail to inline the
  102        -- function at its call site, see Note [The CostingPart constraint in mkMachineParameters].
  103        CostingPart uni fun ~ builtincosts
  104      , HasMeaningIn uni val
  105      , ToBuiltinMeaning uni fun
  106      )
  107   => BuiltinSemanticsVariant fun
  108   -> CostModel machineCosts builtincosts
  109   -> MachineVariantParameters machineCosts fun val
  110 mkMachineVariantParameters semvar (CostModel mchnCosts builtinCosts) =
  111   MachineVariantParameters mchnCosts $ inline toBuiltinsRuntime semvar builtinCosts
  112 {-# INLINE mkMachineVariantParameters #-}
