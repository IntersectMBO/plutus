    9 -- | This module defines a common type various evaluation machine use to return their results.
   10 module PlutusCore.Evaluation.Result
   11   ( EvaluationResult (..)
   12   , isEvaluationSuccess
   13   , isEvaluationFailure
   14   ) where
   15 
   16 import PlutusPrelude
   17 
   18 import PlutusCore.Pretty
   19 
   20 import Control.Monad.Except (MonadError, catchError, throwError)
   21 
   22 -- | The parameterized type of results various evaluation engines return.
   23 data EvaluationResult a
   24   = EvaluationSuccess !a
   25   | EvaluationFailure
   26   deriving stock (Show, Eq, Generic, Functor, Foldable, Traversable)
   27   deriving anyclass (NFData)
   28 
   29 instance MonadError () EvaluationResult where
   30   throwError () = EvaluationFailure
   31   {-# INLINE throwError #-}
   32 
   33   catchError EvaluationFailure f = f ()
   34   catchError x _ = x
   35   {-# INLINE catchError #-}
   36 
   37 instance Applicative EvaluationResult where
   38   pure = EvaluationSuccess
   39   {-# INLINE pure #-}
   40 
   41   EvaluationSuccess f <*> a = fmap f a
   42   EvaluationFailure <*> _ = EvaluationFailure
   43   {-# INLINE (<*>) #-}
   44 
   45   EvaluationSuccess _ *> b = b
   46   EvaluationFailure *> _ = EvaluationFailure
   47   {-# INLINE (*>) #-}
   48 
   49 instance Monad EvaluationResult where
   50   EvaluationSuccess x >>= f = f x
   51   EvaluationFailure >>= _ = EvaluationFailure
   52   {-# INLINE (>>=) #-}
   53 
   54   (>>) = (*>)
   55   {-# INLINE (>>) #-}
   56 
   57 instance Alternative EvaluationResult where
   58   empty = EvaluationFailure
   59   {-# INLINE empty #-}
   60 
   61   a@EvaluationSuccess {} <|> _ = a
   62   EvaluationFailure <|> b = b
   63   {-# INLINE (<|>) #-}
   64 
   65 instance MonadFail EvaluationResult where
   66   fail _ = EvaluationFailure
   67   {-# INLINE fail #-}
   68 
   69 instance PrettyBy config a => PrettyBy config (EvaluationResult a) where
   70   prettyBy config (EvaluationSuccess x) = prettyBy config x
   71   prettyBy _ EvaluationFailure = "Failure"
   72 
   73 instance PrettyClassic a => Pretty (EvaluationResult a) where
   74   pretty = prettyClassic
   75 
   76 -- | Check whether an 'EvaluationResult' is an 'EvaluationSuccess'.
   77 isEvaluationSuccess :: EvaluationResult a -> Bool
   78 isEvaluationSuccess = not . null
   79 
   80 -- | Check whether an 'EvaluationResult' is an 'EvaluationFailure'.
   81 isEvaluationFailure :: EvaluationResult a -> Bool
   82 isEvaluationFailure = null
