    5 -- | @unit@ and related functions.

   17 -- | '()' as a PLC type.
   18 unit :: uni `HasTypeLevel` () => Type tyname uni ()
   19 unit = mkTyBuiltin @_ @() ()
   20 
   21 -- | '()' as a PLC term.
   22 unitval :: (TermLike term tyname name uni fun, uni `HasTermLevel` ()) => term ()
   23 unitval = mkConstant () ()
   24 
   25 -- | 'seq' specified to '()' as a PLC term.
   26 sequ :: (TermLike term tyname Name uni fun, uni `HasTypeAndTermLevel` ()) => term ()
   27 sequ = runQuote $ do
   28   x <- freshName "x"
   29   y <- freshName "y"
   30   return
   31     . lamAbs () x unit
   32     . lamAbs () y unit
   33     $ unitval
