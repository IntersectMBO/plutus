   42 {-| The constraint for built-in types\/functions are kind\/type-checkable.
   43 
   44 We keep this separate from 'MonadKindCheck'\/'MonadTypeCheck', because those mainly constrain the
   45 monad and 'Typecheckable' constraints only the builtins. In particular useful when the monad gets
   46 instantiated and builtins don't. Another reason is that 'Typecheckable' is not required during
   47 type checking, since it's only needed for computing 'BuiltinTypes', which is passed as a regular
   48 argument to the worker of the type checker. -}
   49 type Typecheckable uni fun =
   50   (ToKind uni, HasUniApply uni, ToBuiltinMeaning uni fun, AnnotateCaseBuiltin uni)
   51 
   52 -- | The default kind checking config.
   53 defKindCheckConfig :: KindCheckConfig
   54 defKindCheckConfig = KindCheckConfig DetectNameMismatches

   56 {-| Extract the 'TypeScheme' from a 'BuiltinMeaning' and convert it to the
   57 corresponding 'Type' for each built-in function. -}
   58 builtinMeaningsToTypes
   59   :: (MonadKindCheck (TypeError term uni fun ann) term uni fun ann m, Typecheckable uni fun)
   60   => BuiltinSemanticsVariant fun
   61   -> ann
   62   -> m (BuiltinTypes uni fun)
   63 builtinMeaningsToTypes semvar ann =
   64   runQuoteT . fmap BuiltinTypes . sequence . tabulateArray $ \fun -> do
   65     let ty = typeOfBuiltinFunction semvar fun
   66     _ <- inferKind defKindCheckConfig $ ann <$ ty
   67     dupable <$> normalizeType ty
   68 
   69 -- | Get the default type checking config.
   70 getDefTypeCheckConfig
   71   :: (MonadKindCheck (TypeError term uni fun ann) term uni fun ann m, Typecheckable uni fun)
   72   => ann -> m (TypeCheckConfig uni fun)
   73 getDefTypeCheckConfig ann =
   74   TypeCheckConfig defKindCheckConfig <$> builtinMeaningsToTypes def ann

   76 -- | Infer the kind of a type.
   77 inferKind
   78   :: MonadKindCheck (TypeError term uni fun ann) term uni fun ann m
   79   => KindCheckConfig -> Type TyName uni ann -> m (Kind ())
   80 inferKind config = runTypeCheckM config . inferKindM
   81 
   82 {-| Check a type against a kind.
   83 Infers the kind of the type and checks that it's equal to the given kind
   84 throwing a 'TypeError' (annotated with the value of the @ann@ argument) otherwise. -}
   85 checkKind
   86   :: MonadKindCheck (TypeError term uni fun ann) term uni fun ann m
   87   => KindCheckConfig -> ann -> Type TyName uni ann -> Kind () -> m ()
   88 checkKind config ann ty = runTypeCheckM config . checkKindM ann ty
   89 
   90 -- | Infer the type of a term.
   91 inferType
   92   :: MonadTypeCheckPlc uni fun ann m
   93   => TypeCheckConfig uni fun
   94   -> Term TyName Name uni fun ann
   95   -> m (Normalized (Type TyName uni ()))
   96 inferType config = rename >=> runTypeCheckM config . inferTypeM
   97 
   98 {-| Check a term against a type.
   99 Infers the type of the term and checks that it's equal to the given type
  100 throwing a 'TypeError' (annotated with the value of the @ann@ argument) otherwise. -}
  101 checkType
  102   :: MonadTypeCheckPlc uni fun ann m
  103   => TypeCheckConfig uni fun
  104   -> ann
  105   -> Term TyName Name uni fun ann
  106   -> Normalized (Type TyName uni ())
  107   -> m ()
  108 checkType config ann term ty = do
  109   termRen <- rename term
  110   runTypeCheckM config $ checkTypeM ann termRen ty
  111 
  112 -- | Infer the type of a program.
  113 inferTypeOfProgram
  114   :: MonadTypeCheckPlc uni fun ann m
  115   => TypeCheckConfig uni fun
  116   -> Program TyName Name uni fun ann
  117   -> m (Normalized (Type TyName uni ()))
  118 inferTypeOfProgram config (Program _ _ term) = inferType config term
  119 
  120 {-| Check a program against a type.
  121 Infers the type of the program and checks that it's equal to the given type
  122 throwing a 'TypeError' (annotated with the value of the @ann@ argument) otherwise. -}
  123 checkTypeOfProgram
  124   :: MonadTypeCheckPlc uni fun ann m
  125   => TypeCheckConfig uni fun
  126   -> ann
  127   -> Program TyName Name uni fun ann
  128   -> Normalized (Type TyName uni ())
  129   -> m ()
  130 checkTypeOfProgram config ann (Program _ _ term) = checkType config ann term
