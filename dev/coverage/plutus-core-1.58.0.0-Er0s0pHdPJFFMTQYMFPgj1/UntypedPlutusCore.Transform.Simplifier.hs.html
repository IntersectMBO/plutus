   67 recordSimplificationWithHints
   68   :: Monad m
   69   => Hints.Hints
   70   -> Term name uni fun a
   71   -> Trace.SimplifierStage
   72   -> Term name uni fun a
   73   -> SimplifierT name uni fun a m ()
   74 recordSimplificationWithHints hints before stage after =
   75   let simplification = Trace.Simplification before stage hints after
   76    in SimplifierT . State.modify' $ \st ->
   77         st {Trace.simplifierTrace = simplification : Trace.simplifierTrace st}
