   20 {- Note [Example: InterList]
   21 We encode the following in this module:
   22 
   23     open import Function
   24 
   25     data InterList (A B : Set) : Set where
   26       InterNil  : InterList A B
   27       InterCons : A -> B -> InterList B A -> InterList A B
   28 
   29     foldrInterList : (A B R : Set) -> (A -> B -> R -> R) -> R -> InterList A B -> R
   30     foldrInterList A0 B0 R f0 z = go A0 B0 f0 where
   31       go : âˆ€ A B -> (A -> B -> R -> R) -> InterList A B -> R
   32       go A B f  InterNil          = z
   33       go A B f (InterCons x y xs) = f x y $ go B A (flip f) xs
   34 -}

   36 -- This definition is used as an example in Note [Spiney API] in "PlutusCore.StdLib.Type",

   38 {-| @InterList@ as a PLC type.
   39 
   40 > fix \(interlist :: * -> * -> *) (a :: *) (b :: *) ->
   41 >     all (r :: *). r -> (a -> b -> interlist b a -> r) -> r -}
   42 interListData :: RecursiveType uni fun ()
   43 interListData = runQuote $ do
   44   a <- freshTyName "a"
   45   b <- freshTyName "b"

   50     . TyForall () r (Type ())
   51     . TyFun () (TyVar () r)
   52     . TyFun () (mkIterTyFun () [TyVar () a, TyVar () b, interlistBA] $ TyVar () r)
   53     $ TyVar () r

  102 foldrInterList :: uni `HasTypeAndTermLevel` () => Term TyName Name uni fun ()
  103 foldrInterList = runQuote $ do
  104   let interlist = _recursiveType interListData
  105   a0 <- freshTyName "a0"
  106   b0 <- freshTyName "b0"
  107   r <- freshTyName "r"
  108   f <- freshName "f"
  109   z <- freshName "z"
  110   rec <- freshName "rec"
  111   u <- freshName "u"
  112   a <- freshTyName "a"
  113   b <- freshTyName "b"
  114   f' <- freshName "f'"
  115   xs <- freshName "xs"
  116   x <- freshName "x"
  117   y <- freshName "y"
  118   xs' <- freshName "xs'"
  119   x' <- freshName "x'"
  120   y' <- freshName "y'"
  121   let interlistOf a' b' = mkIterTyAppNoAnn interlist [TyVar () a', TyVar () b']
  122       fTy a' b' = mkIterTyFun () [TyVar () a', TyVar () b', TyVar () r] $ TyVar () r
  123       fixTyArg2 =
  124         TyForall () a (Type ())
  125           . TyForall () b (Type ())
  126           . TyFun () (fTy a b)
  127           . TyFun () (interlistOf a b)
  128           $ TyVar () r
  129       instedFix = mkIterInstNoAnn fix [unit, fixTyArg2]
  130       unwrappedXs = TyInst () (Unwrap () (Var () xs)) $ TyVar () r
  131       instedRec = mkIterInstNoAnn (Apply () (Var () rec) unitval) [TyVar () b, TyVar () a]
  132   return
  133     . TyAbs () a0 (Type ())
  134     . TyAbs () b0 (Type ())
  135     . TyAbs () r (Type ())
  136     . LamAbs () f (fTy a0 b0)
  137     . LamAbs () z (TyVar () r)
  138     $ mkIterInstNoAnn
  139       ( mkIterAppNoAnn
  140           instedFix
  141           [ LamAbs () rec (TyFun () unit fixTyArg2)
  142               . LamAbs () u unit
  143               . TyAbs () a (Type ())
  144               . TyAbs () b (Type ())
  145               . LamAbs () f' (fTy a b)
  146               . LamAbs () xs (interlistOf a b)
  147               $ mkIterAppNoAnn
  148                 unwrappedXs
  149                 [ Var () z
  150                 , LamAbs () x (TyVar () a)
  151                     . LamAbs () y (TyVar () b)
  152                     . LamAbs () xs' (interlistOf b a)
  153                     $ mkIterAppNoAnn
  154                       (Var () f')

  156                       , Var () y
  157                       , mkIterAppNoAnn
  158                           instedRec
  159                           [ LamAbs () y' (TyVar () b)
  160                               . LamAbs () x' (TyVar () a)
  161                               $ mkIterAppNoAnn
  162                                 (Var () f')
  163                                 [ Var () x'
  164                                 , Var () y'
  165                                 ]

  168                       ]
