   27 {- Note [Modulo alpha]
   28 The implemented algorithm of checking equality modulo alpha works as follows
   29 (taking types as an example):
   30 
   31 - traverse both the types simultaneously
   32 - if the outermost constructors differ, return 'False'
   33 - otherwise if the constructors are binders, then record that the two possibly distinct bound
   34     variables map to each other
   35 - otherwise if the constructors are variables, look them up in the current scope
   36     * if both are in scope, then those are bound variables, so check that they map to each other
   37         (i.e. are introduced by twin binders)
   38     * if both are not in scope, then those are free variables, so check if their uniques are equal
   39     * if one is in scope and the other one is not, then return 'False'
   40 - otherwise check equality of non-recursive constituents and recurse for recursive occurrences
   41 -}
   42 
   43 {- Note [Scope tracking]
   44 Scopes (term level vs type level) are resolved automatically in a type-driven way much like we do
   45 in the renamer. This allows not to think about scopes when implementing the actual equality checks
   46 and makes it impossible to confuse the scopes and e.g. insert a type-level name into a term-level
   47 scope. Also allows to define a function that records bindings and a function that checks equality
   48 of two names in a general manner for both the scopes.
   49 -}
   50 
   51 {- Note [Side tracking]
   52 The simplest way to track that twin variables map to each other is to have two contexts:
   53 
   54 - left-hand side variables and what they map to
   55 - right-hand side variables and what they map to
   56 
   57 (we refer to a first argument as being on the left-hand side and to a second argument as being on
   58 the right-hand side)
   59 
   60 E.g. when checking equality of these two types:
   61 
   62     all (x_3 :: *) (x_3 :: *). x_3 -> x_3
   63     all (y_4 :: *) (z_5 :: *). y_4 -> z_5
   64 
   65 we first record that @x_3@ maps to @y_4@ and vice versa, then record that @x_3@ maps to @z_5@ and
   66 vice versa. This way when we later check equality of @x_3@ and @y_4@ we know that it doesn't hold,
   67 because even though @y_4@ maps to @x_3@, @x_3@ doesn't map to @y_4@, because that mapping was
   68 overwritten by the @x_3@-to-@z_5@ one.
   69 
   70 For storing the left-to-right and right-to-left mappings separately we use the 'Bilateral' data
   71 type. Given that we track not only sides, but also scopes, we instantiate 'Bilateral' with either
   72 type-level-only renamings (for checking equality of types) or scoped ones (for checking equality
   73 of terms and programs). This amounts to the following generic monad:
   74 
   75     RenameT (Bilateral ren) m
   76 
   77 i.e. regardless of what the underlying renaming is, it has to be bilateral.
   78 
   79 We zoom into the sides of a bilateral renaming using the 'LR' and 'RL' newtype wrappers using the
   80 same 'HasRenaming' machinery that we use for zooming into the scopes of a scoped renaming:
   81 
   82 - the 'LR' wrapper allows to retrieve a lens focusing on the left  renaming
   83 - the 'RL' wrapper allows to retrieve a lens focusing on the right renaming
   84 
   85 I.e. you wrap a name into either 'LR' and 'RL' and depending on that you get focused on either
   86 left-to-right or right-to-left part of a 'Bilateral' renaming.
   87 
   88 So e.g.
   89 
   90     withRenamedName (LR name1) (LR name2)
   91 
   92 reads as "record that @name1@ maps to @name2@ from left to right"
   93 
   94 and
   95 
   96     lookupNameM $ RL name2
   97 
   98 reads as "look the right-to-left mapping of @name2@ up".
   99 
  100 I.e. we first resolve sides using explicit wrappers and then scope resolution happens automatically
  101 on the basis of existing type information (e.g. 'TyName' is a type-level name, hence we need the
  102 type-level renaming).
  103 -}

  105 -- See Note [Side tracking].
  106 -- | From left to right.
  107 newtype LR a = LR
  108   { unLR :: a
  109   }
  110   deriving stock (Generic)
  111 
  112 -- See Note [Side tracking].
  113 -- | From right to left.
  114 newtype RL a = RL
  115   { unRL :: a
  116   }
  117   deriving stock (Generic)
  118 
  119 -- See Note [Side tracking].
  120 -- | A left @a@ and a right @a@.
  121 data Bilateral a = Bilateral
  122   { _bilateralL :: a
  123   , _bilateralR :: a
  124   }
  125 
  126 makeLenses ''Bilateral
  127 
  128 instance Wrapped (LR a)
  129 instance Wrapped (RL a)
  130 instance HasUnique name unique => HasUnique (LR name) (LR unique)
  131 instance HasUnique name unique => HasUnique (RL name) (RL unique)
  132 
  133 instance Semigroup a => Semigroup (Bilateral a) where
  134   Bilateral l1 r1 <> Bilateral l2 r2 = Bilateral (l1 <> l2) (r1 <> r2)
  135 
  136 instance Monoid a => Monoid (Bilateral a) where
  137   mempty = Bilateral mempty mempty
  138 
  139 -- To rename from left to right is to update the left renaming.
  140 instance HasRenaming ren unique => HasRenaming (Bilateral ren) (LR unique) where
  141   renaming = bilateralL . renaming . coerced @(Renaming unique)
  142 
  143 -- To rename from right to left is to update the right renaming.
  144 instance HasRenaming ren unique => HasRenaming (Bilateral ren) (RL unique) where
  145   renaming = bilateralR . renaming . coerced @(Renaming unique)

  147 {-| The type of a runnable equality check. @Maybe ()@ is isomorphic to 'Bool' and we use it
  148 instead of 'Bool', because this unlocks the convenient and readable do-notation and allows for
  149 automatic short-circuiting, which would be tedious with @Rename (Bilateral ren) Bool@. -}
  150 type EqRename ren = RenameT (Bilateral ren) Maybe ()
  151 
  152 type ScopedEqRename = EqRename ScopedRenaming
  153 
  154 -- | Run an 'EqRename' computation.
  155 runEqRename :: Monoid ren => EqRename ren -> Bool
  156 runEqRename = isJust . runRenameT

  158 -- See Note [Modulo alpha].
  159 -- | Record that two names map to each other.
  160 withTwinBindings
  161   :: (HasRenaming ren unique, HasUnique name unique, Monad m)
  162   => name -> name -> RenameT (Bilateral ren) m c -> RenameT (Bilateral ren) m c
  163 withTwinBindings name1 name2 k =
  164   withRenamedName (LR name1) (LR name2) $
  165     withRenamedName (RL name2) (RL name1) k
  166 
  167 -- See Note [Modulo alpha].
  168 -- | Check equality of two names.
  169 eqNameM
  170   :: (HasRenaming ren unique, HasUnique name unique, Eq unique)
  171   => name -> name -> EqRename ren
  172 eqNameM name1 name2 = do
  173   mayUniq2' <- lookupNameM $ LR name1
  174   mayUniq1' <- lookupNameM $ RL name2
  175   let uniq1 = name1 ^. unique
  176       uniq2 = name2 ^. unique
  177   guard $ case (mayUniq1', mayUniq2') of
  178     (Nothing, Nothing) -> uniq1 == uniq2
  179     (Just (RL uniq1'), Just (LR uniq2')) -> uniq1 == uniq1' && uniq2 == uniq2'
  180     (_, _) -> False
  181 
  182 -- | Check equality of things having an 'Eq' instance.
  183 eqM :: Eq a => a -> a -> EqRename ren
  184 eqM x1 x2 = guard $ x1 == x2
