    4 -- | Functions for computing variable usage inside terms.
    5 module UntypedPlutusCore.Analysis.Usages (termUsages, Usages, getUsageCount, allUsed) where
    6 
    7 import UntypedPlutusCore.Core.Type
    8 import UntypedPlutusCore.Subst
    9 
   10 import PlutusCore qualified as PLC
   11 import PlutusCore.Name.Unique qualified as PLC
   12 
   13 import Control.Lens
   14 
   15 import Data.MultiSet qualified as MSet
   16 import Data.MultiSet.Lens
   17 import Data.Set qualified as Set
   18 
   19 type Usages = MSet.MultiSet PLC.Unique
   20 
   21 -- | Get the usage count of @n@.
   22 getUsageCount :: PLC.HasUnique n unique => n -> Usages -> Int
   23 getUsageCount n = MSet.occur (n ^. PLC.unique . coerced)
   24 
   25 -- | Get a set of @n@s which are used at least once.
   26 allUsed :: Usages -> Set.Set PLC.Unique
   27 allUsed = MSet.toSet
   28 
   29 termUsages
   30   :: PLC.HasUnique name PLC.TermUnique
   31   => Term name uni fun a
   32   -> Usages
   33 termUsages = multiSetOf (vTerm . PLC.theUnique)
