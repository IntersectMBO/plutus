   39 {-| Mapping from variables to what they stand for (each row represents a substitution).
   40 Needed for efficiency reasons, otherwise we could just use substitutions. -}
   41 type TypeVarEnv tyname uni ann = UniqueMap TypeUnique (Dupable (Normalized (Type tyname uni ann)))

   43 -- | The environments that type normalization runs in.
   44 newtype NormalizeTypeEnv tyname uni ann = NormalizeTypeEnv
   45   { _normalizeTypeEnvTypeVarEnv :: TypeVarEnv tyname uni ann
   46   }

   50 {- Note [NormalizeTypeT]
   51 Type normalization requires 'Quote' (because we need to be able to generate fresh names), but we
   52 do not put 'Quote' into 'NormalizeTypeT'. The reason for this is that it makes type signatures of
   53 various runners much nicer and also more generic. For example, we have
   54 
   55     runNormalizeTypeT :: MonadQuote m => NormalizeTypeT m tyname uni ann a -> m a
   56 
   57 If 'NormalizeTypeT' contained 'Quote', it would be
   58 
   59     runNormalizeTypeT :: NormalizeTypeT m tyname uni ann a -> QuoteT m a
   60 
   61 which hardcodes 'QuoteT' to be the outermost transformer.
   62 
   63 Type normalization can run in any @m@ (as long as it's a 'MonadQuote') as witnessed by
   64 the following type signature:
   65 
   66     normalizeTypeM
   67         :: (HasUnique (tyname ann) TypeUnique, MonadQuote m)
   68         => Type tyname uni ann -> NormalizeTypeT m tyname uni ann (Normalized (Type tyname uni ann))
   69 
   70 so it's natural to have runners that do not break this genericity.
   71 -}
   72 
   73 {- Note [Normalization API]
   74 Normalization is split in two parts:
   75 
   76 1. functions returning computations that perform reductions and run in defined in this module
   77    monad transformers (e.g. 'NormalizeTypeT')
   78 2. runners of those computations
   79 
   80 The reason for splitting the API is that this way the type-theoretic notion of normalization is
   81 separated from implementation-specific details. (This used to be more important when we had
   82 to deal with gas, and could maybe be changed now.)
   83 -}

   85 -- See Note [NormalizeTypeT].
   86 -- | The monad transformer that type normalization runs in.
   87 newtype NormalizeTypeT m tyname uni ann a = NormalizeTypeT
   88   { unNormalizeTypeT :: ReaderT (NormalizeTypeEnv tyname uni ann) m a
   89   }
   90   deriving newtype
   91     ( Functor
   92     , Applicative
   93     , Alternative
   94     , Monad
   95     , MonadPlus
   96     , MonadReader (NormalizeTypeEnv tyname uni ann)
   97     , MonadState s
   98     , MonadQuote
   99     )

  101 -- | The constraints that type normalization requires.
  102 type MonadNormalizeType uni m =
  103   ( MonadQuote m -- Type normalization must preserve global uniqueness.
  104   , HasUniApply uni -- See Note [Normalization of built-in types].
  105   )
  106 
  107 -- | Run a 'NormalizeTypeT' computation.
  108 runNormalizeTypeT :: NormalizeTypeT m tyname uni ann a -> m a
  109 runNormalizeTypeT = flip runReaderT (NormalizeTypeEnv mempty) . unNormalizeTypeT

  111 -- | Locally extend a 'TypeVarEnv' in a 'NormalizeTypeT' computation.
  112 withExtendedTypeVarEnv
  113   :: (HasUnique tyname TypeUnique, Monad m)
  114   => tyname
  115   -> Normalized (Type tyname uni ann)
  116   -> NormalizeTypeT m tyname uni ann a
  117   -> NormalizeTypeT m tyname uni ann a
  118 withExtendedTypeVarEnv name =
  119   local . over normalizeTypeEnvTypeVarEnv . insertByName name . dupable

  121 -- | Look up a @tyname@ in a 'TypeVarEnv'.
  122 lookupTyNameM
  123   :: (HasUnique tyname TypeUnique, Monad m)
  124   => tyname
  125   -> NormalizeTypeT m tyname uni ann (Maybe (Dupable (Normalized (Type tyname uni ann))))
  126 lookupTyNameM name = asks $ lookupName name . _normalizeTypeEnvTypeVarEnv

  128 {- Note [Normalization]
  129 Normalization works under the assumption that variables are globally unique.
  130 We use environments instead of substitutions as they're more efficient.
  131 
  132 Since all names are unique and there is no need to track scopes, type normalization has only two
  133 interesting cases: function application and a variable usage. In the function application case we
  134 normalize a function and its argument, add the normalized argument to the environment and continue
  135 normalization. In the variable case we look up the variable in the current environment: if it's not
  136 found, we leave the variable untouched. If the variable is found, then what this variable stands for
  137 was previously added to an environment (while handling the function application case), so we pick
  138 this value and rename all bound variables in it to preserve the global uniqueness condition. It is
  139 safe to do so, because picked values cannot contain uninstantiated variables as only normalized
  140 types are added to environments and normalization instantiates all variables presented in an
  141 environment.
  142 
  143 See also Note [Normalization of built-in types].
  144 -}
  145 
  146 {- Note [Normalization of built-in types]
  147 Instantiating a polymorphic built-in type amounts to applying it to some arguments. However,
  148 the notion of "applying" is ambiguous, it can mean one of these two things:
  149 
  150 1. lifting the built-in type to 'Type' and applying that via 'TyApp'
  151 2. applying the built-in type right inside the universe to get a monomorphized type tag
  152    (e.g. the default universe has 'DefaultUniApply' for that purpose)
  153 
  154 We need both of these things. The former allows us to assign types to polymorphic built-in functions
  155 (otherwise applying a built-in type to a type variable would be unrepresentable), the latter is
  156 used at runtime to juggle type tags so that we can avoid @unsafeCoerce@-ing, bring instances in
  157 scope via 'bring' etc -- for all of that we have to have fully monomorphized type tags at runtime.
  158 
  159 So in order for type checking to work we need to normalize polymorphic built-in types. For that
  160 we simply turn intra-universe applications into regular type applications during type normalization.
  161 
  162 We could go the other way around and "reduce" regular type applications into intra-universe ones,
  163 however that would be harder to implement, because collapsing a general 'Type' into a 'SomeTypeIn'
  164 is harder than expanding a 'SomeTypeIn' into a 'Type'. And it would also be impossible to do in the
  165 general case, 'cause you can't collapse an application of a built-in type to, say, a type variable
  166 into an intra-universe application as there are no type variables there. I guess we could "reduce"
  167 application of built-in types in some cases and not reduce them in others and still make the whole
  168 thing work, but that requires substantially more logic and is also a lot harder to get right.
  169 Hence we do the opposite, which is straightforward.
  170 -}

  174 {-| Normalize a built-in type by replacing each application inside the universe with regular
  175 type application. -}
  176 normalizeUni :: forall k (a :: k) uni tyname. HasUniApply uni => uni (Esc a) -> Type tyname uni ()
  177 normalizeUni uni =
  178   matchUniApply
  179     uni
  180     -- If @uni@ is not an intra-universe application, then we're done.
  181     (mkTyBuiltinOf () uni)
  182     -- If it is, then we turn that application into normal type application and recurse
  183     -- into both the function and its argument.
  184     (\uniF uniA -> TyApp () (normalizeUni uniF) $ normalizeUni uniA)

  188 -- | Normalize a 'Type' in the 'NormalizeTypeT' monad.
  189 normalizeTypeM
  190   :: (HasUnique tyname TypeUnique, MonadNormalizeType uni m)
  191   => Type tyname uni ann
  192   -> NormalizeTypeT m tyname uni ann (Normalized (Type tyname uni ann))
  193 normalizeTypeM (TyForall ann name kind body) =
  194   TyForall ann name kind <<$>> normalizeTypeM body
  195 normalizeTypeM (TyIFix ann pat arg) =
  196   TyIFix ann <<$>> normalizeTypeM pat <<*>> normalizeTypeM arg
  197 normalizeTypeM (TyFun ann dom cod) =
  198   TyFun ann <<$>> normalizeTypeM dom <<*>> normalizeTypeM cod
  199 normalizeTypeM (TyLam ann name kind body) =
  200   TyLam ann name kind <<$>> normalizeTypeM body
  201 normalizeTypeM (TyApp ann fun arg) = do
  202   vFun <- normalizeTypeM fun
  203   vArg <- normalizeTypeM arg
  204   case unNormalized vFun of
  205     TyLam _ nArg _ body -> substNormalizeTypeM vArg nArg body
  206     _ -> pure $ TyApp ann <$> vFun <*> vArg
  207 normalizeTypeM var@(TyVar _ name) = do
  208   mayTy <- lookupTyNameM name
  209   case mayTy of
  210     -- A variable is always normalized.
  211     Nothing -> pure $ Normalized var
  212     Just ty -> liftDupable ty
  213 normalizeTypeM (TyBuiltin ann (SomeTypeIn uni)) =
  214   pure . Normalized $ ann <$ normalizeUni uni
  215 normalizeTypeM (TySOP ann tyls) = do
  216   tyls' <- (traverse . traverse) (fmap unNormalized . normalizeTypeM) tyls
  217   pure $ Normalized $ TySOP ann tyls'

  226 -- See Note [Normalizing substitution].
  227 -- | Substitute a type for a variable in a type and normalize in the 'NormalizeTypeT' monad.
  228 substNormalizeTypeM
  229   :: (HasUnique tyname TypeUnique, MonadNormalizeType uni m)
  230   => Normalized (Type tyname uni ann)
  231   -- ^ @ty@
  232   -> tyname
  233   -- ^ @name@
  234   -> Type tyname uni ann
  235   -- ^ @body@
  236   -> NormalizeTypeT m tyname uni ann (Normalized (Type tyname uni ann))
  237   -- ^ @NORM ([ty / name] body)@
  238 substNormalizeTypeM ty name = withExtendedTypeVarEnv name ty . normalizeTypeM

  240 -- | Normalize every 'Type' in a 'Term'.
  241 normalizeTypesInM
  242   :: (HasUnique tyname TypeUnique, MonadNormalizeType uni m)
  243   => Term tyname name uni fun ann
  244   -> NormalizeTypeT m tyname uni ann (Term tyname name uni fun ann)
  245 normalizeTypesInM = transformMOf termSubterms normalizeChildTypes
  246   where
  247     normalizeChildTypes = termSubtypes (fmap unNormalized . normalizeTypeM)
