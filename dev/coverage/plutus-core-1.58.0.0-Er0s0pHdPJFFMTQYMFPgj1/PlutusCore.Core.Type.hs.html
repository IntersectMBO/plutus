   74 data Kind ann
   75   = Type ann
   76   | KindArrow ann (Kind ann) (Kind ann)
   77   deriving stock (Eq, Show, Functor, Generic, Lift)
   78   deriving anyclass (NFData, Hashable)

   80 {-| The kind of a pattern functor (the first 'Type' argument of 'TyIFix') at a given kind (of the
   81 second 'Type' argument of 'TyIFix'):
   82 
   83 > toPatFuncKind k = (k -> *) -> k -> * -}
   84 toPatFuncKind :: Kind () -> Kind ()
   85 toPatFuncKind k = KindArrow () (KindArrow () k (Type ())) (KindArrow () k (Type ()))
   86 
   87 fromPatFuncKind :: Kind () -> Maybe (Kind ())
   88 fromPatFuncKind (KindArrow () (KindArrow () k1 (Type ())) (KindArrow () k2 (Type ())))
   89   | k1 == k2 = Just k1
   90 fromPatFuncKind _ = Nothing

   92 -- | Extract all @a_i@ from @a_0 -> a_1 -> ... -> r@.
   93 argsFunKind :: Kind ann -> [Kind ann]
   94 argsFunKind Type {} = []
   95 argsFunKind (KindArrow _ k l) = k : argsFunKind l

   97 -- | A 'Type' assigned to expressions.
   98 type Type :: GHC.Type -> (GHC.Type -> GHC.Type) -> GHC.Type -> GHC.Type
   99 data Type tyname uni ann
  100   = -- | Type variable
  101     TyVar ann tyname
  102   | -- | Function type
  103     TyFun ann (Type tyname uni ann) (Type tyname uni ann)
  104   | -- | Fix-point type, for constructing self-recursive types
  105     TyIFix ann (Type tyname uni ann) (Type tyname uni ann)
  106   | -- | Polymorphic type
  107     TyForall ann tyname (Kind ann) (Type tyname uni ann)
  108   | -- | Builtin type
  109     TyBuiltin ann (SomeTypeIn uni)
  110   | -- | Type lambda
  111     TyLam ann tyname (Kind ann) (Type tyname uni ann)
  112   | -- | Type application
  113     TyApp ann (Type tyname uni ann) (Type tyname uni ann)
  114   | -- | Sum-of-products type
  115     TySOP ann [[Type tyname uni ann]]
  116   deriving stock (Show, Functor, Generic)
  117   deriving anyclass (NFData)
  118 
  119 {-| Get recursively all the domains and codomains of a type.
  120 @splitFunTyParts (A->B->C) = [A, B, C]@
  121 @splitFunTyParts (X) = [X]@ -}
  122 splitFunTyParts :: Type tyname uni a -> NE.NonEmpty (Type tyname uni a)
  123 splitFunTyParts = \case
  124   TyFun _ t1 t2 -> t1 NE.<| splitFunTyParts t2
  125   t -> pure t
  126 
  127 {-| Get the argument types of a function type.
  128 @funTyArgs (A->B->C) = [A, B]@ -}
  129 funTyArgs :: Type tyname uni a -> [Type tyname uni a]
  130 funTyArgs = NE.init . splitFunTyParts
  131 
  132 {-| Get the result type of a function.
  133 If not a function, then is the same as `id`
  134 @funResultType (A->B->C) = C@
  135 @funResultType (X) = X@ -}
  136 funTyResultType :: Type tyname uni a -> Type tyname uni a
  137 funTyResultType = NE.last . splitFunTyParts

  139 data Term tyname name uni fun ann
  140   = -- | a named variable
  141     Var ann name
  142   | -- | lambda abstraction
  143     LamAbs ann name (Type tyname uni ann) (Term tyname name uni fun ann)
  144   | -- | application
  145     Apply ann (Term tyname name uni fun ann) (Term tyname name uni fun ann)
  146   | -- | type abstraction
  147     TyAbs ann tyname (Kind ann) (Term tyname name uni fun ann)
  148   | -- | instantiation
  149     TyInst ann (Term tyname name uni fun ann) (Type tyname uni ann)
  150   | -- | wrapping
  151     IWrap ann (Type tyname uni ann) (Type tyname uni ann) (Term tyname name uni fun ann)
  152   | {-| unwrapping
  153     See Note [Constr tag type] -}
  154     Unwrap ann (Term tyname name uni fun ann)
  155   | -- | constructor
  156     Constr ann (Type tyname uni ann) Word64 [Term tyname name uni fun ann]
  157   | -- | case
  158     Case ann (Type tyname uni ann) (Term tyname name uni fun ann) [Term tyname name uni fun ann]
  159   | -- | constants
  160     Constant ann (Some (ValueOf uni))
  161   | -- | builtin functions
  162     Builtin ann fun
  163   | -- | fail with error
  164     Error ann (Type tyname uni ann)
  165   deriving stock (Functor, Generic)
  166 
  167 deriving stock instance
  168   (Show tyname, Show name, GShow uni, Everywhere uni Show, Show fun, Show ann, Closed uni)
  169   => Show (Term tyname name uni fun ann)
  170 
  171 deriving anyclass instance
  172   (NFData tyname, NFData name, NFData fun, NFData ann, Everywhere uni NFData, Closed uni)
  173   => NFData (Term tyname name uni fun ann)

  276 termAnn :: Term tyname name uni fun ann -> ann
  277 termAnn (Var ann _) = ann
  278 termAnn (TyAbs ann _ _ _) = ann
  279 termAnn (Apply ann _ _) = ann
  280 termAnn (Constant ann _) = ann
  281 termAnn (Builtin ann _) = ann
  282 termAnn (TyInst ann _ _) = ann
  283 termAnn (Unwrap ann _) = ann
  284 termAnn (IWrap ann _ _ _) = ann
  285 termAnn (Error ann _) = ann
  286 termAnn (LamAbs ann _ _ _) = ann
  287 termAnn (Constr ann _ _ _) = ann
  288 termAnn (Case ann _ _ _) = ann

  290 -- | Map a function over the set of built-in functions.
  291 mapFun :: (fun -> fun') -> Term tyname name uni fun ann -> Term tyname name uni fun' ann
  292 mapFun f = go
  293   where
  294     go (LamAbs ann name ty body) = LamAbs ann name ty (go body)
  295     go (TyAbs ann name kind body) = TyAbs ann name kind (go body)
  296     go (IWrap ann pat arg term) = IWrap ann pat arg (go term)
  297     go (Apply ann fun arg) = Apply ann (go fun) (go arg)
  298     go (Unwrap ann term) = Unwrap ann (go term)
  299     go (Error ann ty) = Error ann ty
  300     go (TyInst ann term ty) = TyInst ann (go term) ty
  301     go (Var ann name) = Var ann name
  302     go (Constant ann con) = Constant ann con
  303     go (Builtin ann fun) = Builtin ann (f fun)
  304     go (Constr ann ty i args) = Constr ann ty i (map go args)
  305     go (Case ann ty arg cs) = Case ann ty (go arg) (map go cs)

  307 {-| This is a wrapper to mark the place where the binder is introduced (i.e. LamAbs/TyAbs)
  308 and not where it is actually used (TyVar/Var..).
  309 This marking allows us to skip the (de)serialization of binders at LamAbs/TyAbs positions
  310 iff 'name' is DeBruijn-encoded (level or index). See for example the instance of  'UntypedPlutusCore.Core.Instance.Flat' -}

  316 {- Note [Constr tag type]
  317 Constructor tags are not dynamically created, they can only come from the program itself.
  318 
  319 So we do not have to worry about e.g. overflow.
  320 
  321 The main constraints that we have are:
  322 1. The use of the tags in the evaluator must be fast.
  323 2. We don't want to allow nonsensical things (e.g. negative tags)
  324 3. We don't want the format to change unexpectedly
  325 
  326 These are in tension. The structure we use for storing case branches is a list, which is
  327 traditionally indexed with an 'Int' (and most of the other alternatives are the same). That
  328 suggests using 'Int' as the tag, but 'Int' allows negative tags. A 'Word' would be more
  329 appropriate (of fixed size to satisfy 3), but then we need to be able to index lists by
  330 'Word's.
  331 
  332 Well, except there's nothing that _stops_ us indexing lists by 'Word's, we just need to
  333 write our own indexing functions. So that's what we have done.
  334 
  335 We use 'Word64' since it's the actual machine word, so no benefit from using a smaller
  336 one (smaller values will be serialized by flat in fewer bytes anyway).
  337 -}
