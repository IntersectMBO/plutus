   47 -- | See Note [Wrapping the BLS12-381 types in Plutus Core].
   48 newtype Element = Element {unElement :: BlstBindings.Point2}
   49   deriving newtype (Eq)
   50 
   51 instance Show Element where
   52   show = byteStringAsHex . compress
   53 instance Pretty Element where
   54   pretty = pretty . show
   55 instance PrettyBy ConstConfig Element
   56 
   57 {-| We don't support direct flat encoding of G1 elements because of the expense
   58    of on-chain uncompression.  Users should convert between G1 elements and
   59    bytestrings using `compress` and `uncompress`: the bytestrings can be
   60    flat-encoded in the usual way. -}
   61 instance Flat Element where
   62   -- This might happen on the chain, so `fail` rather than `error`.
   63   decode = fail "Flat decoding is not supported for objects of type bls12_381_G2_element: use bls12_381_G2_uncompress on a bytestring instead."
   64 
   65   -- This will be a Haskell runtime error, but encoding doesn't happen on chain,
   66   -- so it's not too bad.
   67   encode = error "Flat encoding is not supported for objects of type bls12_381_G2_element: use bls12_381_G2_compress to obtain a bytestring instead."
   68   size _ = id
   69 
   70 instance NFData Element where
   71   rnf (Element x) = rwhnf x -- Just to be on the safe side.
   72 
   73 instance Hashable Element where
   74   hashWithSalt salt = hashWithSalt salt . compress

   76 -- | Add two G2 group elements
   77 add :: Element -> Element -> Element
   78 add = coerce (BlstBindings.blsAddOrDouble @BlstBindings.Curve2)
   79 {-# INLINE add #-}

   81 -- | Negate a G2 group element
   82 neg :: Element -> Element
   83 neg = coerce (BlstBindings.blsNeg @BlstBindings.Curve2)

   86 scalarMul :: Integer -> Element -> Element -- Other way round from library function
   87 scalarMul = coerce $ flip (BlstBindings.blsMult @BlstBindings.Curve2)
   88 {-# INLINE scalarMul #-}

   90 {-| Compress a G2 element to a bytestring. This serialises a curve point to its x
   91  coordinate only, using an extra bit to determine which of two possible y
   92  coordinates the point has. The compressed bytestring is 96 bytes long. See
   93  https://github.com/supranational/blst#serialization-format -}
   94 compress :: Element -> ByteString
   95 compress = coerce (BlstBindings.blsCompress @BlstBindings.Curve2)
   96 {-# INLINE compress #-}
   97 
   98 {-| Uncompress a bytestring to get a G2 point.  This will fail if any of the
   99    following are true:
  100      * The bytestring is not exactly 96 bytes long
  101      * The most significant three bits are used incorrectly
  102      * The bytestring encodes a field element which is not the
  103        x coordinate of a point on the E2 curve
  104      * The bytestring does represent a point on the E2 curve, but the
  105        point is not in the G2 subgroup -}
  106 uncompress :: ByteString -> Either BlstBindings.BLSTError Element
  107 uncompress = coerce (BlstBindings.blsUncompress @BlstBindings.Curve2)
  108 {-# INLINE uncompress #-}

  110 -- Take an arbitrary bytestring and a Domain Separation Tag and hash them to a
  111 -- get point in G2.  See Note [Hashing and Domain Separation Tags].
  112 hashToGroup :: ByteString -> ByteString -> Either BLS12_381_Error Element
  113 hashToGroup msg dst =
  114   if Data.ByteString.length dst > 255
  115     then Left HashToCurveDstTooBig
  116     else Right . Element $ BlstBindings.blsHash @BlstBindings.Curve2 msg (Just dst) Nothing

  143 {-| Multi-scalar multiplication of G2 points.  We limit the allowable size of
  144 scalars to simplify costing. -}
  145 multiScalarMul :: [Integer] -> [Element] -> BuiltinResult Element
  146 multiScalarMul ss p
  147   | any msmScalarOutOfBounds ss = fail "Scalar exceeds 512-byte bound for G2.multiScalarMul"
  148   | otherwise = pure . coerce $ BlstBindings.blsMSM @BlstBindings.Curve2 (zip ss (coerce p))
