   10 module PlutusCore.Evaluation.ErrorWithCause
   11   ( ErrorWithCause (..)
   12   , throwErrorWithCause
   13   ) where
   14 
   15 import PlutusPrelude
   16 
   17 import PlutusCore.Pretty
   18 
   19 import Control.Monad.Except
   20 import Data.Bifunctor
   21 import Prettyprinter
   22 
   23 -- | An error and (optionally) what caused it.
   24 data ErrorWithCause err cause = ErrorWithCause
   25   { _ewcError :: !err
   26   , _ewcCause :: !(Maybe cause)
   27   }
   28   deriving stock (Eq, Functor, Foldable, Traversable, Generic)
   29   deriving anyclass (NFData)
   30 
   31 instance Bifunctor ErrorWithCause where
   32   bimap f g (ErrorWithCause err cause) = ErrorWithCause (f err) (g <$> cause)
   33   {-# INLINE bimap #-}
   34 
   35 instance (Pretty err, Pretty cause) => Pretty (ErrorWithCause err cause) where
   36   pretty (ErrorWithCause e c) = pretty e <+> "caused by:" <+> pretty c
   37 
   38 instance
   39   (PrettyBy config cause, PrettyBy config err)
   40   => PrettyBy config (ErrorWithCause err cause)
   41   where
   42   prettyBy config (ErrorWithCause err mayCause) =
   43     fold
   44       [ "An error has occurred:"
   45       , hardline
   46       , prettyBy config err
   47       , case mayCause of
   48           Nothing -> mempty
   49           Just cause -> hardline <> "Caused by:" <+> prettyBy config cause
   50       ]
   51 
   52 instance
   53   (PrettyPlc cause, PrettyPlc err)
   54   => Show (ErrorWithCause err cause)
   55   where
   56   show = render . prettyPlcReadable
   57 
   58 deriving anyclass instance
   59   (PrettyPlc cause, PrettyPlc err, Typeable cause, Typeable err)
   60   => Exception (ErrorWithCause err cause)
   61 
   62 throwErrorWithCause
   63   :: MonadError (ErrorWithCause e cause) m
   64   => e
   65   -> cause
   66   -> m x
   67 throwErrorWithCause e = throwError . ErrorWithCause e . Just
   68 {-# INLINE throwErrorWithCause #-}
