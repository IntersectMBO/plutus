  168 {-| A newtype wrapper around @a@ whose point is to provide a 'Show' instance
  169 for anything that has a 'Pretty' instance. -}
  170 newtype ShowPretty a = ShowPretty
  171   { unShowPretty :: a
  172   }
  173   deriving stock (Eq)
  174 
  175 instance Pretty a => Show (ShowPretty a) where
  176   show = display . unShowPretty

  178 instance (Pretty a, Pretty b) => Pretty (Either a b) where
  179   pretty (Left x) = parens ("Left" <+> pretty x)
  180   pretty (Right y) = parens ("Right" <+> pretty y)

  192 {-| Coerce the second argument to the result type of the first one. The motivation for this
  193 function is that it's often more annoying to explicitly specify a target type for 'coerce' than
  194 to construct an explicit coercion function, so this combinator can be used in cases like that.
  195 Plus the code reads better, as it becomes clear what and where gets wrapped/unwrapped. -}
  196 coerceVia :: Coercible a b => (a -> b) -> a -> b
  197 coerceVia _ = coerce
  198 {-# INLINE coerceVia #-}

  200 -- | Same as @\f -> f . coerce@, but does not create any closures and so is completely free.
  201 coerceArg :: Coercible a b => (a -> s) -> b -> s
  202 coerceArg = coerce
  203 {-# INLINE coerceArg #-}

  205 -- | Same as @\f -> coerce . f@, but does not create any closures and so is completely free.
  206 coerceRes :: Coercible s t => (a -> s) -> a -> t

  216 (<<$>>) :: (Functor f1, Functor f2) => (a -> b) -> f1 (f2 a) -> f1 (f2 b)
  217 (<<$>>) = fmap . fmap

  219 (<<*>>) :: (Applicative f1, Applicative f2) => f1 (f2 (a -> b)) -> f1 (f2 a) -> f1 (f2 b)
  220 (<<*>>) = liftA2 (<*>)

  222 -- | Makes an effectful function ignore its result value and return its input value.
  223 through :: Functor f => (a -> f b) -> (a -> f a)
  224 through f x = f x $> x

  226 forJoin :: (Monad m, Traversable m, Applicative f) => m a -> (a -> f (m b)) -> f (m b)
  227 forJoin a f = join <$> for a f

  229 -- | Fold a monadic function over a 'Foldable'. The monadic version of 'foldMap'.
  230 foldMapM :: (Foldable f, Monad m, Monoid b) => (a -> m b) -> f a -> m b
  231 foldMapM f xs = foldr step return xs mempty
  232   where
  233     step x r z = f x >>= \y -> r $! z `mappend` y

  235 {-| This function generalizes 'eitherToMaybe', 'eitherToList',
  236 'listToMaybe' and other such functions. -}
  237 reoption :: (Foldable f, Alternative g) => f a -> g a
  238 reoption = foldr (const . pure) empty

  240 {-| Basically a @Data.Functor.Representable@ instance for 'Array'.
  241 We can't provide an actual instance because of the @Distributive@ superclass: @Array i@ is not
  242 @Distributive@ unless we assume that indices in an array range over the entirety of @i@. -}
  243 tabulateArray :: (Bounded i, Enum i, Ix i) => (i -> a) -> Array i a
  244 tabulateArray f = listArray (minBound, maxBound) $ map f enumerate

  253 -- | @b ? x@ is equal to @pure x@ whenever @b@ holds and is 'empty' otherwise.
  254 (?) :: Alternative f => Bool -> a -> f a
  255 (?) b x = x <$ guard b

  257 -- | @ensure p x@ is equal to @pure x@ whenever @p x@ holds and is 'empty' otherwise.
  258 ensure :: Alternative f => (a -> Bool) -> a -> f a
  259 ensure p x = p x ? x

  277 -- | Compose two folds to make them run in parallel. The results are concatenated.
  278 (<^>) :: Fold s a -> Fold s a -> Fold s a
  279 (f1 <^> f2) g s = f1 g s *> f2 g s

  281 {-| Zips two lists of the same length together, returning 'Nothing' if they are not
  282 the same length. -}
  283 zipExact :: [a] -> [b] -> Maybe [(a, b)]
  284 zipExact [] [] = Just []
  285 zipExact (a : as) (b : bs) = (:) (a, b) <$> zipExact as bs
  286 zipExact _ _ = Nothing

  288 {-| Similar to Maybe's `fromJust`. Returns the `Right` and errors out with the show instance
  289 of the `Left`. -}
  290 unsafeFromRight :: Show e => Either e a -> a
  291 unsafeFromRight (Right a) = a
  292 unsafeFromRight (Left e) = error $ show e

  294 -- | function recursively applied N times
  295 timesA :: Natural -> (a -> a) -> a -> a

  298 {-| Pair each element of the given list with all the other elements.
  299 
  300 >>> addTheRest "abcd"
  301 [('a',"bcd"),('b',"acd"),('c',"abd"),('d',"abc")] -}
  302 addTheRest :: [a] -> [(a, [a])]
  303 addTheRest [] = []
  304 addTheRest (x : xs) = (x, xs) : map (fmap (x :)) (addTheRest xs)

  306 allSame :: Eq a => [a] -> Bool
  307 allSame [] = True
  308 allSame (x : xs) = all (x ==) xs
  309 
  310 distinct :: Eq a => [a] -> Bool
  311 distinct = not . allSame
