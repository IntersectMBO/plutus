  128 {-| Substitute the given closed 'Type' for the given type variable in the given 'Type'. Does not
  129 descend under binders that bind the same variable as the one we're substituting for (since from
  130 there that variable is no longer free). The resulting 'Term' may and likely will not satisfy
  131 global uniqueness. -}
  132 typeSubstClosedType
  133   :: Eq tyname => tyname -> Type tyname uni a -> Type tyname uni a -> Type tyname uni a
  134 typeSubstClosedType tn0 ty0 = go
  135   where
  136     go = \case
  137       TyVar a tn -> if tn == tn0 then ty0 else TyVar a tn
  138       TyForall a tn k ty -> TyForall a tn k (goUnder tn ty)
  139       TyLam a tn k ty -> TyLam a tn k (goUnder tn ty)
  140       ty -> ty & over typeSubtypes go
  141     goUnder tn ty = if tn == tn0 then ty else go ty
  142 
  143 {-| Substitute the given closed 'Type' for the given type variable in the given 'Term'. Does not
  144 descend under binders that bind the same variable as the one we're substituting for (since from
  145 there that variable is no longer free). The resulting 'Term' may and likely will not satisfy
  146 global uniqueness. -}
  147 termSubstClosedType
  148   :: Eq tyname
  149   => tyname -> Type tyname uni a -> Term tyname name uni fun a -> Term tyname name uni fun a
  150 termSubstClosedType tn0 ty0 = go
  151   where
  152     go = \case
  153       TyAbs a tn k body -> TyAbs a tn k (goUnder tn body)
  154       t -> t & over termSubtypes goTy & over termSubterms go
  155     goUnder tn term = if tn == tn0 then term else go term
  156     goTy = typeSubstClosedType tn0 ty0
  157 
  158 {-| Substitute the given closed 'Term' for the given term variable in the given 'Term'. Does not
  159 descend under binders that bind the same variable as the one we're substituting for (since from
  160 there that variable is no longer free). The resulting 'Term' may and likely will not satisfy
  161 global uniqueness. -}
  162 termSubstClosedTerm
  163   :: Eq name
  164   => name
  165   -> Term tyname name uni fun a
  166   -> Term tyname name uni fun a
  167   -> Term tyname name uni fun a
  168 termSubstClosedTerm varFor new = go
  169   where
  170     go = \case
  171       Var a var -> if var == varFor then new else Var a var
  172       LamAbs a var ty body -> LamAbs a var ty (goUnder var body)
  173       t -> t & over termSubterms go
  174     goUnder var term = if var == varFor then term else go term

  176 -- Mapping name-modification functions over types and terms.

  178 typeMapNames
  179   :: forall tyname tyname' uni ann
  180    . (tyname -> tyname')
  181   -> Type tyname uni ann
  182   -> Type tyname' uni ann
  183 typeMapNames f = go
  184   where
  185     go :: Type tyname uni ann -> Type tyname' uni ann
  186     go = \case
  187       TyVar ann tn -> TyVar ann (f tn)
  188       TyFun ann ty1 ty2 -> TyFun ann (go ty1) (go ty2)
  189       TyIFix ann ty1 ty2 -> TyIFix ann (go ty1) (go ty2)
  190       TyForall ann tn k ty -> TyForall ann (f tn) k (go ty)
  191       TyBuiltin ann s -> TyBuiltin ann s
  192       TyLam ann tn k ty -> TyLam ann (f tn) k (go ty)
  193       TyApp ann ty1 ty2 -> TyApp ann (go ty1) (go ty2)
  194       TySOP ann tyls -> TySOP ann ((fmap . fmap) go tyls)
  195 
  196 -- termMapNames requires two function arguments: one (called f) to modify type names
  197 -- and another (called g) to modify variable names.
  198 termMapNames
  199   :: forall tyname tyname' name name' uni fun ann
  200    . (tyname -> tyname')
  201   -> (name -> name')
  202   -> Term tyname name uni fun ann
  203   -> Term tyname' name' uni fun ann
  204 termMapNames f g = go
  205   where
  206     go :: Term tyname name uni fun ann -> Term tyname' name' uni fun ann
  207     go = \case
  208       LamAbs ann name ty body -> LamAbs ann (g name) (typeMapNames f ty) (go body)
  209       TyAbs ann tyname k body -> TyAbs ann (f tyname) k (go body)
  210       Var ann name -> Var ann (g name)
  211       Apply ann t1 t2 -> Apply ann (go t1) (go t2)
  212       Constant ann c -> Constant ann c
  213       Builtin ann b -> Builtin ann b
  214       TyInst ann body ty -> TyInst ann (go body) (typeMapNames f ty)
  215       Unwrap ann body -> Unwrap ann (go body)
  216       IWrap ann ty1 ty2 body -> IWrap ann (typeMapNames f ty1) (typeMapNames f ty2) (go body)
  217       Constr ann ty i es -> Constr ann (typeMapNames f ty) i (fmap go es)
  218       Case ann ty arg cs -> Case ann (typeMapNames f ty) (go arg) (fmap go cs)
  219       Error ann ty -> Error ann (typeMapNames f ty)
  220 
  221 programMapNames
  222   :: forall tyname tyname' name name' uni fun ann
  223    . (tyname -> tyname')
  224   -> (name -> name')
  225   -> Program tyname name uni fun ann
  226   -> Program tyname' name' uni fun ann
  227 programMapNames f g (Program a v term) = Program a v (termMapNames f g term)

  231 -- | Get all the free term variables in a term.
  232 fvTerm :: HasUnique name unique => Traversal' (Term tyname name uni fun ann) name
  233 fvTerm = fvTermCtx mempty
  234 
  235 fvTermCtx
  236   :: HasUnique name unique
  237   => UniqueSet unique -> Traversal' (Term tyname name uni fun ann) name
  238 fvTermCtx bound f = \case
  239   Var a n -> Var a <$> (if USet.memberByName n bound then pure n else f n)
  240   LamAbs a n ty t -> LamAbs a n ty <$> fvTermCtx (USet.insertByName n bound) f t
  241   t -> (termSubterms . fvTermCtx bound) f t
  242 
  243 -- | Get all the free type variables in a term.
  244 ftvTerm :: HasUnique tyname unique => Traversal' (Term tyname name uni fun ann) tyname
  245 ftvTerm = ftvTermCtx mempty
  246 
  247 ftvTermCtx
  248   :: HasUnique tyname unique
  249   => UniqueSet unique
  250   -> Traversal' (Term tyname name uni fun ann) tyname
  251 ftvTermCtx bound f = \case
  252   TyAbs a ty k t -> TyAbs a ty k <$> ftvTermCtx (USet.insertByName ty bound) f t
  253   -- sound because the subterms and subtypes are disjoint
  254   t ->
  255     ((termSubterms . ftvTermCtx bound) `Unsound.adjoin` (termSubtypes . ftvTyCtx bound)) f t
  256 
  257 -- | Get all the free type variables in a type.
  258 ftvTy
  259   :: HasUnique tyname unique
  260   => Traversal' (Type tyname uni ann) tyname
  261 ftvTy = ftvTyCtx mempty
  262 
  263 ftvTyCtx
  264   :: HasUnique tyname unique
  265   => UniqueSet unique
  266   -> Traversal' (Type tyname uni ann) tyname
  267 ftvTyCtx bound f = \case
  268   TyVar a ty -> TyVar a <$> (if USet.memberByName ty bound then pure ty else f ty)
  269   TyForall a bnd k ty -> TyForall a bnd k <$> ftvTyCtx (USet.insertByName bnd bound) f ty
  270   TyLam a bnd k ty -> TyLam a bnd k <$> ftvTyCtx (USet.insertByName bnd bound) f ty
  271   t -> (typeSubtypes . ftvTyCtx bound) f t
