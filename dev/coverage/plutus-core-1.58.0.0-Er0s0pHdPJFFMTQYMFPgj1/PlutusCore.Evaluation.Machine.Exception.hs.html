   42 -- | Errors which can occur during a run of an abstract machine.
   43 data MachineError fun
   44   = -- | An attempt to reduce a not immediately reducible type instantiation.
   45     NonPolymorphicInstantiationMachineError
   46   | -- | An attempt to unwrap a not wrapped term.
   47     NonWrapUnwrappedMachineError
   48   | -- | An attempt to reduce a not immediately reducible application.
   49     NonFunctionalApplicationMachineError
   50   | -- | An attempt to evaluate an open term.
   51     OpenTermEvaluatedMachineError
   52   | -- | An attempt to compute a constant application resulted in 'UnliftingError'.
   53     UnliftingMachineError UnliftingError
   54   | -- | A builtin expected a term argument, but something else was received.
   55     BuiltinTermArgumentExpectedMachineError
   56   | -- | A builtin received a term argument when something else was expected
   57     UnexpectedBuiltinTermArgumentMachineError
   58   | -- | An attempt to scrutinize a non-constr.
   59     NonConstrScrutinizedMachineError
   60   | -- | An attempt to go into a non-existent case branch.
   61     MissingCaseBranchMachineError Word64
   62   | -- | A GHC exception was thrown.
   63     PanicMachineError String
   64   deriving stock (Show, Eq, Functor, Generic)
   65   deriving anyclass (NFData)

   67 type EvaluationException structural operational =

   70 {- Note [Ignoring context in OperationalError]
   71 The 'OperationalError' error has a term argument, but 'splitStructuralOperational' just
   72 discards this and returns 'EvaluationFailure'. This means that, for example, if we use the @plc@
   73 command to execute a program containing a division by zero, @plc@ exits silently without reporting
   74 that anything has gone wrong (but returning a non-zero exit code to the shell via 'exitFailure').
   75 This is because 'OperationalError' is used in cases when a PLC program itself goes wrong
   76 (see the Haddock of 'EvaluationError'). This is used to signal unsuccessful validation and so is
   77 not regarded as a real error; in contrast structural errors are genuine errors and we report their
   78 context if available.
   79 -}
   80 
   81 -- See the Haddock of 'EvaluationError' for what structural and operational errors are.
   82 -- See Note [Ignoring context in OperationalError].
   83 {-| Preserve the contents of an 'StructuralError' as a 'Left' and turn an
   84 'OperationalError' into a @Right EvaluationFailure@ (thus erasing the content of the
   85 error in the latter case). -}
   86 splitStructuralOperational
   87   :: Either (EvaluationException structural operational term) a
   88   -> Either (ErrorWithCause structural term) (EvaluationResult a)
   89 splitStructuralOperational (Right term) = Right $ EvaluationSuccess term
   90 splitStructuralOperational (Left (ErrorWithCause evalErr cause)) = case evalErr of
   91   StructuralError err -> Left $ ErrorWithCause err cause
   92   OperationalError _ -> Right EvaluationFailure
   93 
   94 {-| Throw on a 'StructuralError' and turn an 'OperationalError' into an
   95 'EvaluationFailure' (thus erasing the content of the error in the latter case). -}
   96 unsafeSplitStructuralOperational
   97   :: (PrettyPlc structural, PrettyPlc term, Typeable structural, Typeable term)
   98   => Either (EvaluationException structural operational term) a
   99   -> EvaluationResult a
  100 unsafeSplitStructuralOperational = unsafeFromEither . splitStructuralOperational

  102 instance
  103   (HasPrettyDefaults config ~ 'True, Pretty fun)
  104   => PrettyBy config (MachineError fun)
  105   where
  106   prettyBy _ NonPolymorphicInstantiationMachineError =
  107     "Attempted to instantiate a non-polymorphic term."
  108   prettyBy _ NonWrapUnwrappedMachineError =
  109     "Cannot unwrap a not wrapped term."
  110   prettyBy _ NonFunctionalApplicationMachineError =
  111     "Attempted to apply a non-function."
  112   prettyBy _ OpenTermEvaluatedMachineError =
  113     "Cannot evaluate an open term"
  114   prettyBy _ BuiltinTermArgumentExpectedMachineError =
  115     "A builtin expected a term argument, but something else was received"
  116   prettyBy _ UnexpectedBuiltinTermArgumentMachineError =
  117     "A builtin received a term argument when something else was expected"
  118   prettyBy _ (UnliftingMachineError unliftingError) =
  119     pretty unliftingError
  120   prettyBy _ NonConstrScrutinizedMachineError =
  121     "A non-constructor/non-builtin value was scrutinized in a case expression"
  122   prettyBy _ (MissingCaseBranchMachineError i) =
  123     "Case expression missing the branch required by the scrutinee tag:" <+> pretty i
  124   prettyBy _ (PanicMachineError err) =
  125     vcat
  126       [ "Panic: a GHC exception was thrown, please report this as a bug."
  127       , "The error: " <+> pretty err
  128       ]

  130 class BuiltinErrorToEvaluationError structural operational where
  131   builtinErrorToEvaluationError :: BuiltinError -> EvaluationError structural operational
  132 
  133 {-| Attach a @cause@ to a 'BuiltinError' and throw that.
  134 Note that an evaluator might require the cause to be computed lazily for best performance on the
  135 happy path, hence this function must not force its first argument.
  136 TODO: wrap @cause@ in 'Lazy' once we have it. -}
  137 throwBuiltinErrorWithCause
  138   :: ( MonadError (EvaluationException structural operational cause) m
  139      , BuiltinErrorToEvaluationError structural operational
  140      )
  141   => cause -> BuiltinError -> m void
  142 throwBuiltinErrorWithCause cause e = throwErrorWithCause (builtinErrorToEvaluationError e) cause
  143 {-# INLINE throwBuiltinErrorWithCause #-}
