    9 {-| A 'Name' is a datatype used to identify a variable inside the Plutus Core languages.
   10  Name comparisons are a fundamental part of the domain logic, and comparing 'Text' directly
   11  is inefficient. As a solution to this problem, we provide the 'Unique' type which is an
   12  integer associated to the 'Name', unique to each instantiation of the type. We can,
   13  therefore, compare the integers instead, which is obviously much more cost-effective.
   14 
   15  We distinguish between the names of term variables and type variables by defining the
   16  'TyName' wrapper over 'Name'. Since the code we usually write is polymorphic in the
   17  name type, we want to be able to define a class of names which have an associated 'Unique'.
   18  This class is 'HasUnique', see the definition below. -}

   64 -- | A 'Name' represents variables/names in Plutus Core.
   65 data Name = Name
   66   { _nameText :: T.Text
   67   -- ^ The identifier name, for use in error messages.
   68   , _nameUnique :: Unique
   69   -- ^ A 'Unique' assigned to the name, allowing for cheap comparisons in the compiler.
   70   }
   71   deriving stock (Show, Generic, Lift)
   72   deriving anyclass (NFData)
   73 
   74 -- | Allowed characters in the starting position of a non-quoted identifier.
   75 isIdentifierStartingChar :: Char -> Bool
   76 isIdentifierStartingChar c = isAscii c && isAlpha c || c == '_'
   77 
   78 -- | Allowed characters in a non-starting position of a non-quoted identifier.
   79 isIdentifierChar :: Char -> Bool
   80 isIdentifierChar c = isIdentifierStartingChar c || isDigit c || c == '\''
   81 
   82 -- | Allowed characters in a quoted identifier.
   83 isQuotedIdentifierChar :: Char -> Bool
   84 isQuotedIdentifierChar c =
   85   (isAlpha c || isDigit c || isPunctuation c || isSymbol c)
   86     && isAscii c
   87     && c /= '`'
   88 
   89 isValidUnquotedName :: Text -> Bool
   90 isValidUnquotedName n = case T.uncons n of
   91   Just (hd, tl) -> isIdentifierStartingChar hd && T.all isIdentifierChar tl
   92   Nothing -> False
   93 
   94 {-| Quote the name with backticks if it is not a valid unquoted name.
   95 It does not check whether the given name is a valid quoted name. -}
   96 toPrintedName :: Text -> Text
   97 toPrintedName txt = if isValidUnquotedName txt then txt else "`" <> txt <> "`"

  113 instance HasPrettyConfigName config => PrettyBy config Name where
  114   prettyBy config (Name txt (Unique uniq))
  115     | showsUnique = pretty $ toPrintedName txt <> "-" <> render (pretty uniq)
  116     | otherwise = pretty $ toPrintedName txt
  117     where
  118       PrettyConfigName showsUnique = toPrettyConfigName config

  120 instance Eq Name where
  121   (==) = (==) `on` _nameUnique
  122 
  123 instance Ord Name where
  124   (<=) = (<=) `on` _nameUnique
  125 
  126 -- Hashable follows Eq and Ord in only depending on the unique
  127 instance Hashable Name where
  128   hashWithSalt s = hashWithSalt s . _nameUnique

  130 {-| A unique identifier
  131 This is normally a positive integral number, except
  132 in `LetFloatOut.topUnique` where we make use of a negative unique to signify top-level. -}
  133 newtype Unique = Unique {unUnique :: Int}
  134   deriving stock (Eq, Show, Ord, Lift)
  135   deriving newtype (Enum, NFData, Pretty, Hashable)

  161 -- | Types which have a textual name attached to them.
  162 class HasText a where
  163   theText :: Lens' a Text
  164 
  165 instance HasText Name where
  166   theText = nameText
  167 
  168 instance HasText TyName where
  169   theText = coerced . theText @Name

  171 -- | Types which have a 'Unique' attached to them, mostly names.
  172 class Coercible unique Unique => HasUnique a unique | a -> unique where
  173   unique :: Lens' a unique
  174 
  175   -- | The default implementation of 'HasUnique' for newtypes.
  176   default unique
  177     :: (Wrapped a, HasUnique (Unwrapped a) unique', Coercible unique' unique)
  178     => Lens' a unique
  179   unique = _Wrapped' . unique . coerced
  180 
  181 instance HasUnique Unique Unique where
  182   unique = id
  183 
  184 instance HasUnique Name TermUnique where
  185   unique = nameUnique . coerced
  186 
  187 instance HasUnique TyName TypeUnique

  190 theUnique :: HasUnique name unique => Lens' name Unique
