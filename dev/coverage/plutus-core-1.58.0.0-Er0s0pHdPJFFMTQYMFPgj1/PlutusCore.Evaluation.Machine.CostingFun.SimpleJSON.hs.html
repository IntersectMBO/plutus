   98 {-| This type reflects what is actually in the JSON.  The stuff in
   99    CostingFun.Core and CostingFun.JSON is much more rigid, allowing parsing only
  100    for the model types applicable to the various ModelNArguments types; it also
  101    requires entries for everything in DefaultFun. Using the type defined here
  102    allows us to be more flexible and parse stuff that's not exactly what's
  103    expected in builtinCostModel.json. -}
  104 data Model
  105   = ConstantCost Integer
  106   | AddedSizes LinearFunction
  107   | MultipliedSizes LinearFunction
  108   | MinSize LinearFunction
  109   | MaxSize LinearFunction
  110   | LinearInX LinearFunction
  111   | LinearInY LinearFunction
  112   | LinearInZ LinearFunction
  113   | LinearInU LinearFunction
  114   | LiteralInYOrLinearInZ LinearFunction
  115   | LinearInMaxYZ LinearFunction
  116   | LinearInXAndY TwoVariableLinearFunction
  117   | LinearInYAndZ TwoVariableLinearFunction
  118   | QuadraticInX OneVariableQuadraticFunction
  119   | QuadraticInY OneVariableQuadraticFunction
  120   | QuadraticInZ OneVariableQuadraticFunction
  121   | QuadraticInXAndY TwoVariableQuadraticFunction
  122   | WithInteractionInXAndY TwoVariableWithInteractionFunction
  123   | -- | Linear model in x-y plus minimum value for the case x-y < 0.
  124     SubtractedSizes LinearFunction Integer
  125   | ConstAboveDiagonal Integer Model
  126   | ConstBelowDiagonal Integer Model
  127   | ConstOffDiagonal Integer Model
  128   | ExpModCost ExpModCostingFunction
  129   deriving stock (Show, Lift)
  130 
  131 {- The JSON representation consists of a list of pairs of (type, arguments)
  132    values.  The "type" field corresponds to the possible constructors above, the
  133    "arguments" field contains the arguments for that particular constructor.
  134 
  135    The JSON format is a bit inconsistent, which adds some complexity.  For
  136    example, the "arguments" field is sometimes a constant, sometimes an object
  137    representing a linear function, and sometimes an object which contains the
  138    coefficients of a linear function together with some extra data. It would be
  139    nice to rationalise this a bit, but it may be too late to do that.
  140 -}
  141 
  142 instance FromJSON Model where
  143   parseJSON =
  144     withObject "Model" $
  145       \obj -> do
  146         ty :: Text <- obj .: "type"
  147         args :: Value <- obj .: "arguments"
  148         {- We always have an "arguments" field which is a Value.  Usually it's
  149            actually an Object (ie, a map) representing a linear function, but
  150            sometimes it contains other data, and in those cases we need to
  151            coerce it to an Object (with objOf) to extract the relevant data.
  152            We could do that once here and rely on laziness to save us in the
  153            cases when we don't have an Object, but that looks a bit misleading. -}
  154         case ty of
  155           "constant_cost" -> ConstantCost <$> parseJSON args
  156           "added_sizes" -> AddedSizes <$> parseJSON args
  157           "min_size" -> MinSize <$> parseJSON args
  158           "max_size" -> MaxSize <$> parseJSON args
  159           "multiplied_sizes" -> MultipliedSizes <$> parseJSON args
  160           "linear_in_x" -> LinearInX <$> parseJSON args
  161           "linear_in_y" -> LinearInY <$> parseJSON args
  162           "linear_in_z" -> LinearInZ <$> parseJSON args
  163           "linear_in_u" -> LinearInU <$> parseJSON args
  164           "quadratic_in_x" -> QuadraticInX <$> parseJSON args
  165           "quadratic_in_y" -> QuadraticInY <$> parseJSON args
  166           "quadratic_in_z" -> QuadraticInZ <$> parseJSON args
  167           "quadratic_in_x_and_y" -> QuadraticInXAndY <$> parseJSON args
  168           "exp_mod_cost" -> ExpModCost <$> parseJSON args
  169           "literal_in_y_or_linear_in_z" -> LiteralInYOrLinearInZ <$> parseJSON args
  170           "linear_in_max_yz" -> LinearInMaxYZ <$> parseJSON args
  171           "linear_in_x_and_y" -> LinearInXAndY <$> parseJSON args
  172           "linear_in_y_and_z" -> LinearInYAndZ <$> parseJSON args
  173           "with_interaction_in_x_and_y" -> WithInteractionInXAndY <$> parseJSON args
  174           "subtracted_sizes" -> SubtractedSizes <$> parseJSON args <*> objOf args .: "minimum"
  175           "const_above_diagonal" -> modelWithConstant ConstAboveDiagonal args
  176           "const_below_diagonal" -> modelWithConstant ConstBelowDiagonal args
  177           "const_off_diagonal" -> modelWithConstant ConstOffDiagonal args
  178           {- An adaptor to deal with the old "linear_on_diagonal" tag.  See Note [Backward
  179              compatibility for costing functions].  We never want to convert back to JSON
  180              here, so it's OK to forget that we originally got something tagged with
  181             "linear_on_diagonal". -}
  182           "linear_on_diagonal" ->
  183             let o = objOf args
  184              in do
  185                   constant <- o .: "constant"
  186                   intercept <- o .: "intercept"
  187                   slope <- o .: "slope"
  188                   pure $ ConstOffDiagonal constant (LinearInX $ LinearFunction intercept slope)
  189           _ -> errorWithoutStackTrace $ "Unknown model type " ++ show ty
  190     where
  191       objOf (Object o) = o
  192       objOf _ =
  193         errorWithoutStackTrace "Failed to get Object while parsing \"arguments\""
  194 
  195       modelWithConstant constr x = constr <$> o .: "constant" <*> o .: "model"
  196         where
  197           o = objOf x
