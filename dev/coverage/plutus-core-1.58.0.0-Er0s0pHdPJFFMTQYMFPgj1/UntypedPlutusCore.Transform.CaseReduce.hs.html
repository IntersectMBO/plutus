   21 caseReduce
   22   :: (Monad m, CaseBuiltin uni)
   23   => Term name uni fun a
   24   -> SimplifierT name uni fun a m (Term name uni fun a)
   25 caseReduce term = do
   26   let result = transformOf termSubterms processTerm term
   27   recordSimplification term CaseReduce result
   28   return result
   29 
   30 processTerm :: CaseBuiltin uni => Term name uni fun a -> Term name uni fun a
   31 processTerm = \case
   32   -- We could've rewritten those patterns as 'Error' in the 'Nothing' cases, but that would turn a
   33   -- structural error into an operational one, which would be unfortunate, so instead we decided
   34   -- not to fully optimize such scripts, since they aren't valid anyway.
   35   Case ann (Constr _ i args) cs
   36     | Just c <- (V.!?) cs (fromIntegral i) ->
   37         mkIterApp c ((ann,) <$> args)
   38   Case ann (Constant _ con) cs
   39     | Right t <- headSpineToTerm ann (second (Constant ann) (caseBuiltin con cs)) -> t
   40   t -> t
