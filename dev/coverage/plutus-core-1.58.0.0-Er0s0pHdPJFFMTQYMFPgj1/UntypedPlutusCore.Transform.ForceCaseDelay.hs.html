    3 {- Note [Applying force to delays in case branches]
    4 
    5 Often, the following pattern occurs in UPLC terms:
    6 
    7 > force (case scrut [\x1... -> delay term_1, ..., \x1... -> delay term_m])
    8 
    9 It's sound to remove the 'force' and the 'delay's, as long as the original term is "well-formed".
   10 Note that the lambda abstraction may be missing, and we consider that case as well.
   11 
   12 Intuitively, what we mean by "well-formed" is that the term does not evaluate to bottom unless
   13 the user intended it to (i.e. by introducing an 'error' subterm).
   14 
   15 In the the context of the Plinth compiler pipeline, UPLC is always generated from TPLC, by
   16 erasing the types of a TPLC term. So at the beginning of the UPLC phase of the compiler,
   17 the UPLC can be considered "well-formed", since it is guaranteed by the types of the original
   18 TPLC term.
   19 
   20 The other UPLC transformations we have are guaranteed to preserve this property, so we can assume
   21 that any UPLC term coming from the UPLC phase of the compiler is "well-formed".
   22 
   23 What about preserving the correct evaluation strategy? Since we are removing 'delay's, we need to
   24 ensure that the terms under the delays are not evaluated ahead of time. However, this is not a problem
   25 because the 'case' construct is lazy in its branches, so the terms under the 'delay's will not be
   26 evaluated unless the corresponding branch is taken.
   27 
   28 We should, however, formally define what "well-formed" means, and this is left as future work:
   29 FIXME(https://github.com/IntersectMBO/plutus-private/issues/1644).
   30 
   31 -}
   32 module UntypedPlutusCore.Transform.ForceCaseDelay
   33   ( forceCaseDelay
   34   )
   35 where
   36 
   37 import UntypedPlutusCore.Core
   38 import UntypedPlutusCore.Transform.Simplifier
   39   ( SimplifierStage (ForceCaseDelay)
   40   , SimplifierT
   41   , recordSimplification
   42   )
   43 
   44 import Control.Lens
   45 
   46 forceCaseDelay
   47   :: Monad m
   48   => Term name uni fun a
   49   -> SimplifierT name uni fun a m (Term name uni fun a)
   50 forceCaseDelay term = do
   51   let result = transformOf termSubterms processTerm term
   52   recordSimplification term ForceCaseDelay result
   53   return result
   54 
   55 processTerm :: Term name uni fun a -> Term name uni fun a
   56 processTerm = \case
   57   original@(Force _ (Case cAnn scrut branches)) ->
   58     let mNewBranches = traverse findDelayUnderLambdas branches
   59      in case mNewBranches of
   60           Just newBranches ->
   61             Case cAnn scrut newBranches
   62           Nothing -> original
   63   other -> other
   64   where
   65     findDelayUnderLambdas :: Term name uni fun a -> Maybe (Term name uni fun a)
   66     findDelayUnderLambdas = \case
   67       LamAbs ann var body -> LamAbs ann var <$> findDelayUnderLambdas body
   68       Delay _ term -> Just term
   69       _ -> Nothing
