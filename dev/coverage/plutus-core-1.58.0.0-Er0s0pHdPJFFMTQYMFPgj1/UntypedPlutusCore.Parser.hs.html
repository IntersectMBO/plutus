   39 -- Parsers for UPLC terms
   40 
   41 -- | A parsable UPLC term.
   42 type PTerm = UPLC.Term PLC.Name PLC.DefaultUni PLC.DefaultFun SrcSpan
   43 
   44 conTerm :: SrcSpan -> Parser PTerm
   45 conTerm sp =
   46   UPLC.Constant sp <$> constant
   47 
   48 builtinTerm :: SrcSpan -> Parser PTerm
   49 builtinTerm sp =
   50   UPLC.Builtin sp <$> builtinFunction
   51 
   52 varTerm :: Parser PTerm
   53 varTerm =
   54   withSpan $ \sp -> UPLC.Var sp <$> name
   55 
   56 lamTerm :: SrcSpan -> Parser PTerm
   57 lamTerm sp =
   58   UPLC.LamAbs sp <$> (trailingWhitespace name) <*> term
   59 
   60 appTerm :: SrcSpan -> Parser PTerm
   61 appTerm sp =
   62   -- TODO: should not use the same `sp` for all arguments.
   63   mkIterApp <$> term <*> (fmap (sp,) <$> some term)
   64 
   65 delayTerm :: SrcSpan -> Parser PTerm
   66 delayTerm sp =
   67   UPLC.Delay sp <$> term
   68 
   69 forceTerm :: SrcSpan -> Parser PTerm
   70 forceTerm sp =
   71   UPLC.Force sp <$> term
   72 
   73 errorTerm :: SrcSpan -> Parser PTerm
   74 errorTerm sp =
   75   return (UPLC.Error sp)
   76 
   77 constrTerm :: SrcSpan -> Parser PTerm
   78 constrTerm sp = do
   79   let maxTag = fromIntegral (maxBound :: Word64)
   80   tag :: Integer <- lexeme Lex.decimal
   81   args <- many term
   82   whenVersion (\v -> v < plcVersion110) $ fail "'constr' is not allowed before version 1.1.0"
   83   when (tag > maxTag) $ fail "constr tag too large: must be a legal Word64 value"
   84   pure $ UPLC.Constr sp (fromIntegral tag) args
   85 
   86 caseTerm :: SrcSpan -> Parser PTerm
   87 caseTerm sp = do
   88   res <- UPLC.Case sp <$> term <*> (V.fromList <$> many term)
   89   whenVersion (\v -> v < plcVersion110) $ fail "'case' is not allowed before version 1.1.0"
   90   pure res
   91 
   92 -- | Parser for all UPLC terms.
   93 term :: Parser PTerm
   94 term =
   95   leadingWhitespace $ do
   96     choice
   97       [ tryAppTerm
   98       , tryTermInParens
   99       , varTerm
  100       ]
  101   where
  102     tryAppTerm :: Parser PTerm
  103     tryAppTerm =
  104       withSpan $ \sp ->
  105         between
  106           (symbol "[" <?> "opening bracket '['")
  107           (char ']' <?> "closing bracket ']'")
  108           (appTerm sp)
  109 
  110     tryTermInParens :: Parser PTerm
  111     tryTermInParens =
  112       withSpan $ \sp ->
  113         between
  114           (symbol "(" <?> "opening parenthesis '('")
  115           (char ')' <?> "closing parenthesis ')'")
  116           ( choice
  117               [ symbol "builtin" *> builtinTerm sp
  118               , symbol "lam" *> lamTerm sp
  119               , symbol "constr" *> constrTerm sp -- "constr" must come before "con"
  120               , symbol "con" *> conTerm sp
  121               , symbol "delay" *> delayTerm sp
  122               , symbol "force" *> forceTerm sp
  123               , symbol "error" *> errorTerm sp
  124               , symbol "case" *> caseTerm sp
  125               ]
  126               <?> "term keyword (builtin, lam, constr, con, delay, force, error, case)"
  127           )
  128 
  129 -- | Parser for UPLC programs.
  130 program :: Parser (UPLC.Program PLC.Name PLC.DefaultUni PLC.DefaultFun SrcSpan)
  131 program = leadingWhitespace go
  132   where
  133     go = do
  134       prog <- withSpan $ \sp -> inParens $ do
  135         v <- symbol "program" *> version
  136         withVersion v $ UPLC.Program sp v <$> term
  137       notFollowedBy anySingle
  138       pure prog
  139 
  140 {-| Parse a UPLC term. The resulting program will have fresh names. The underlying monad must be capable
  141 of handling any parse errors. -}
  142 parseTerm :: (MonadError PLC.ParserErrorBundle m, PLC.MonadQuote m) => Text -> m PTerm
  143 parseTerm = parseGen term
  144 
  145 {-| Parse a UPLC program. The resulting program will have fresh names. The
  146 underlying monad must be capable of handling any parse errors.  This passes
  147 "test" to the parser as the name of the input stream; to supply a name
  148 explicity, use `parse program <name> <input>`.` -}
  149 parseProgram
  150   :: (MonadError PLC.ParserErrorBundle m, PLC.MonadQuote m)
  151   => Text
  152   -> m (UPLC.Program PLC.Name PLC.DefaultUni PLC.DefaultFun SrcSpan)
  153 parseProgram = parseGen program
  154 
  155 {-| Parse and rewrite so that names are globally unique, not just unique within
  156 their scope. -}
  157 parseScoped
  158   :: (MonadError (PLC.Error uni fun SrcSpan) m, PLC.MonadQuote m)
  159   => Text
  160   -> m (UPLC.Program PLC.Name PLC.DefaultUni PLC.DefaultFun SrcSpan)
  161 -- don't require there to be no free variables at this point, we might be parsing an open term
  162 parseScoped =
  163   through (modifyError PLC.UniqueCoherencyErrorE . checkProgram (const True))
  164     <=< rename
  165     <=< modifyError PLC.ParseErrorE . parseProgram
