    3 {-| Definition analysis for untyped Plutus Core.
    4 This just adapts term-related code from PlutusCore.Analysis.Definitions;
    5 we just re-use the typed machinery to do the hard work here. -}
    6 module UntypedPlutusCore.Analysis.Definitions
    7   ( termDefs
    8   , runTermDefs
    9   ) where
   10 
   11 import UntypedPlutusCore.Core
   12 
   13 import PlutusCore.Analysis.Definitions (ScopeType (TermScope), UniqueInfos, addDef, addUsage)
   14 import PlutusCore.Error (UniqueError)
   15 import PlutusCore.Name.Unique (HasUnique, TermUnique (TermUnique), Unique (Unique))
   16 
   17 import Control.Lens (forMOf_)
   18 import Control.Monad.State (MonadState, execStateT)
   19 import Control.Monad.Writer (MonadWriter, WriterT (runWriterT))
   20 
   21 {-| Given a UPLC term, add all of its term definitions and usages, including its subterms,
   22 to a global map. -}
   23 termDefs
   24   :: ( Ord ann
   25      , HasUnique name TermUnique
   26      , MonadState (UniqueInfos ann) m
   27      , MonadWriter [UniqueError ann] m
   28      )
   29   => Term name uni fun ann
   30   -> m ()
   31 termDefs tm = do
   32   forMOf_ termSubtermsDeep tm handleTerm
   33 
   34 handleTerm
   35   :: ( Ord ann
   36      , HasUnique name TermUnique
   37      , MonadState (UniqueInfos ann) m
   38      , MonadWriter [UniqueError ann] m
   39      )
   40   => Term name uni fun ann
   41   -> m ()
   42 handleTerm = \case
   43   Var ann n ->
   44     addUsage n ann TermScope
   45   LamAbs ann n _ ->
   46     addDef n ann TermScope
   47   _ -> pure ()
   48 
   49 runTermDefs
   50   :: ( Ord ann
   51      , HasUnique name TermUnique
   52      , Monad m
   53      )
   54   => Term name uni fun ann
   55   -> m (UniqueInfos ann, [UniqueError ann])
   56 runTermDefs = runWriterT . flip execStateT mempty . termDefs
