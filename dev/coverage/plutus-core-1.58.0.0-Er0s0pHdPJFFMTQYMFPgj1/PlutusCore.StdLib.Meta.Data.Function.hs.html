   14 {-| 'const' as a PLC term.
   15 
   16 > constPartial t = /\(A :: *) -> \(x : A) -> t -}
   17 constPartial :: TermLike term TyName Name uni fun => term () -> term ()
   18 constPartial t = runQuote $ do
   19   a <- freshTyName "a"
   20   x <- freshName "x"
   21   return
   22     . tyAbs () a (Type ())
   23     . lamAbs () x (TyVar () a)
   24     $ t

   26 {-| Eta-expand a function at a given type. Note that this has to be a \"meta\" function
   27 for it not force the function it receives and instead directly hide it under a lambda.
   28 
   29 > etaExpand ty fun = \(x : ty) -> fun x -}
   30 etaExpand :: TermLike term tyname Name uni fun => Type tyname uni () -> term () -> term ()
   31 etaExpand ty fun = runQuote $ do
   32   x <- freshName "x"
   33   return
   34     . lamAbs () x ty
   35     . apply () fun
   36     $ var () x
