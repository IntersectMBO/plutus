   27 -- | @Data@ as a built-in PLC type.
   28 dataTy :: uni `HasTypeLevel` Data => Type tyname uni ()
   29 dataTy = mkTyBuiltin @_ @Data ()
   30 
   31 {-| Pattern matching over 'Data' inside PLC.
   32 
   33 > \(d : data) ->
   34 >     /\(r :: *) ->
   35 >      \(fConstr : integer -> list data -> r)
   36 >       (fMap : list (pair data data) -> r)
   37 >       (fList : list data -> r)
   38 >       (fI : integer -> r)
   39 >       (fB : bytestring -> r) ->
   40 >           chooseData
   41 >               d
   42 >               {unit -> r}
   43 >               (\(u : unit) -> uncurry {integer} {list data} {r} fConstr (unConstrB d))
   44 >               (\(u : unit) -> fMap (unMapB d))
   45 >               (\(u : unit) -> fList (unListB d))
   46 >               (\(u : unit) -> fI (unIB d))
   47 >               (\(u : unit) -> fB (unBB d))
   48 >               unitval -}
   49 matchData :: TermLike term TyName Name DefaultUni DefaultFun => term ()
   50 matchData = runQuote $ do
   51   r <- freshTyName "r"
   52   fConstr <- freshName "fConstr"
   53   fMap <- freshName "fMap"
   54   fList <- freshName "fList"
   55   fI <- freshName "fI"
   56   fB <- freshName "fB"
   57   d <- freshName "d"
   58   u <- freshName "u"
   59   let listData = mkTyBuiltin @_ @[Data] ()
   60       listPairData = mkTyBuiltin @_ @[(Data, Data)] ()
   61       bytestring = mkTyBuiltin @_ @ByteString ()
   62   return
   63     . lamAbs () d dataTy
   64     . tyAbs () r (Type ())
   65     . lamAbs () fConstr (TyFun () integer . TyFun () listData $ TyVar () r)
   66     . lamAbs () fMap (TyFun () listPairData $ TyVar () r)
   67     . lamAbs () fList (TyFun () listData $ TyVar () r)
   68     . lamAbs () fI (TyFun () integer $ TyVar () r)
   69     . lamAbs () fB (TyFun () bytestring $ TyVar () r)
   70     $ mkIterAppNoAnn
   71       (tyInst () (builtin () ChooseData) . TyFun () unit $ TyVar () r)
   72       [ var () d
   73       , lamAbs () u unit $
   74           mkIterAppNoAnn
   75             (mkIterInstNoAnn uncurry [integer, listData, TyVar () r])
   76             [ var () fConstr
   77             , apply () (builtin () UnConstrData) $ var () d
   78             ]
   79       , lamAbs () u unit
   80           . apply () (var () fMap)
   81           . apply () (builtin () UnMapData)
   82           $ var () d
   83       , lamAbs () u unit
   84           . apply () (var () fList)
   85           . apply () (builtin () UnListData)
   86           $ var () d
   87       , lamAbs () u unit
   88           . apply () (var () fI)
   89           . apply () (builtin () UnIData)
   90           $ var () d
   91       , lamAbs () u unit
   92           . apply () (var () fB)
   93           . apply () (builtin () UnBData)
   94           $ var () d
   95       , unitval
   96       ]
