    3 {-| This module exports everything from the stdlib via a data type which allows to test
    4 various procedures (pretty-printing, type checking, etc) over the entire stdlib in a
    5 convenient way: each time a function / data type is added to the stdlib, none of the
    6 tests is required to be adapted, instead you just add the new definition to 'stdLib'
    7 defined below and all the tests see it automatically. -}
    8 module PlutusCore.StdLib.Everything
    9   ( stdLib
   10   ) where
