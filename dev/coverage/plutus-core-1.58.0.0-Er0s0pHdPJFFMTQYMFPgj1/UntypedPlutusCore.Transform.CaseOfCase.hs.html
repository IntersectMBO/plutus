    5 {-|
    6 Perform the case-of-case transformation. This pushes
    7 case expressions into the case branches of other case
    8 expressions, which can often yield optimization opportunities.
    9 
   10 Example:
   11 @
   12     case (case s of { C1 a -> x; C2 b -> y; }) of
   13       D1 -> w
   14       D2 -> z
   15 
   16     -->
   17 
   18     case s of
   19       C1 a -> case x of { D1 -> w; D2 -> z; }
   20       C2 b -> case y of { D1 -> w; D2 -> z; }
   21 @
   22 
   23 We also transform
   24 
   25 @
   26     case ((force ifThenElse) b (constr t) (constr f)) alts
   27 @
   28 
   29 into
   30 
   31 @
   32     force (force ifThenElse b (delay (case (constr t) alts)) (delay (case (constr f) alts)))
   33 @
   34 
   35 This is always an improvement. -}
   36 module UntypedPlutusCore.Transform.CaseOfCase (caseOfCase) where
   37 
   38 import PlutusPrelude
   39 
   40 import PlutusCore qualified as PLC
   41 import PlutusCore.Builtin (CaseBuiltin (..))
   42 import PlutusCore.MkPlc (mkIterApp)
   43 import UntypedPlutusCore.Core
   44 import UntypedPlutusCore.Transform.CaseReduce qualified as CaseReduce
   45 import UntypedPlutusCore.Transform.Simplifier
   46   ( SimplifierStage (CaseOfCase)
   47   , SimplifierT
   48   , recordSimplification
   49   )
   50 
   51 import Control.Lens
   52 import Data.List (nub)
   53 
   54 caseOfCase
   55   :: ( fun ~ PLC.DefaultFun
   56      , Monad m
   57      , CaseBuiltin uni
   58      , PLC.GEq uni
   59      , PLC.Closed uni
   60      , uni `PLC.Everywhere` Eq
   61      )
   62   => Term name uni fun a
   63   -> SimplifierT name uni fun a m (Term name uni fun a)
   64 caseOfCase term = do
   65   let result = transformOf termSubterms processTerm term
   66   recordSimplification term CaseOfCase result
   67   return result
   68 
   69 processTerm
   70   :: ( fun ~ PLC.DefaultFun
   71      , CaseBuiltin uni
   72      , PLC.GEq uni
   73      , PLC.Closed uni
   74      , uni `PLC.Everywhere` Eq
   75      )
   76   => Term name uni fun a -> Term name uni fun a
   77 processTerm = \case
   78   Case ann scrut alts
   79     | ( ite@(Force a (Builtin _ PLC.IfThenElse))
   80         , [cond, (trueAnn, true@Constr {}), (falseAnn, false@Constr {})]
   81         ) <-
   82         splitApplication scrut ->
   83         Force a $
   84           mkIterApp
   85             ite
   86             [ cond
   87             , -- Here we call a single step of case-reduce in order to immediately clean up the
   88               -- duplication of @alts@. Otherwise optimizing case-of-case-of-case-of... would create
   89               -- exponential blowup of the case branches, which would eventually get deduplicated
   90               -- with case-reduce, but only after that exponential blowup has already slowed the
   91               -- optimizer down unnecessarily.
   92               (trueAnn, Delay trueAnn . CaseReduce.processTerm $ Case ann true alts)
   93             , (falseAnn, Delay falseAnn . CaseReduce.processTerm $ Case ann false alts)
   94             ]
   95   original@(Case annOuter (Case annInner scrut altsInner) altsOuter) ->
   96     maybe
   97       original
   98       (Case annInner scrut)
   99       ( do
  100           constrs <- for altsInner $ \case
  101             c@(Constr _ i _) -> Just (Left i, c)
  102             c@(Constant _ val) -> Just (Right val, c)
  103             _ -> Nothing
  104           -- See Note [Case-of-case and duplicating code].
  105           guard $ length (nub . toList $ fmap fst constrs) == length constrs
  106           pure $ constrs <&> \(_, c) -> CaseReduce.processTerm $ Case annOuter c altsOuter
  107       )
  108   other -> other
