  105 -- | A signed 128-bit integer quantity.
  106 newtype Quantity = UnsafeQuantity {unQuantity :: Integer}
  107   deriving newtype (Eq, Ord, Show, NFData, Hashable)
  108   deriving stock (Generic)
  109 
  110 instance CBOR.Serialise Quantity where
  111   encode (UnsafeQuantity i) = CBOR.encode i
  112   {-# INLINE encode #-}
  113   decode = do
  114     i <- CBOR.decode
  115     case quantity i of
  116       Just q -> pure q
  117       Nothing -> fail $ "Quantity out of signed 128-bit integer bounds: " <> show i
  118   {-# INLINEABLE decode #-}

  120 instance Flat.Flat Quantity where
  121   encode (UnsafeQuantity i) = Flat.encode i
  122   {-# INLINE encode #-}
  123   decode = do
  124     i <- Flat.decode
  125     case quantity i of
  126       Just q -> pure q
  127       Nothing -> fail $ "Quantity out of signed 128-bit integer bounds: " <> show i
  128   {-# INLINEABLE decode #-}

  130 instance Pretty Quantity where

  133 instance Bounded Quantity where
  134   minBound = UnsafeQuantity (-(2 ^ (127 :: Integer)))
  135   {-# INLINE minBound #-}
  136   maxBound = UnsafeQuantity (2 ^ (127 :: Integer) - 1)

  139 -- | Smart constructor for Quantity that validates bounds.
  140 quantity :: Integer -> Maybe Quantity
  141 quantity i
  142   | i >= unQuantity minBound && i <= unQuantity maxBound = Just (UnsafeQuantity i)
  143   | otherwise = Nothing

  153 addQuantity (UnsafeQuantity x) (UnsafeQuantity y) = quantity (x + y)

  168   = Value

  205 {-| Unpack a `Value` into a map from (currency symbol, token name) to quantity.
  206 
  207 The map is guaranteed to not contain empty inner map or zero quantity. -}
  208 unpack :: Value -> NestedMap
  209 unpack (Value v _ _ _) = v
  210 {-# INLINE unpack #-}

  212 {-| Pack a map from (currency symbol, token name) to quantity into a `Value`.
  213 
  214 The map will be filtered so that it does not contain empty inner map or zero quantity. -}
  215 pack :: NestedMap -> Value
  216 pack = pack' . normalize
  217 {-# INLINE pack #-}
  218 
  219 -- | Like `pack` but does not normalize.
  220 pack' :: NestedMap -> Value
  221 pack' v = Value v sizes total neg
  222   where
  223     (sizes, total, neg) = Map.foldl' alg (mempty, 0, 0) v
  224     alg (ss, t, n) inner =
  225       ( IntMap.alter (maybe (Just 1) (Just . (+ 1))) (Map.size inner) ss
  226       , t + Map.size inner
  227       , n + Map.size (Map.filter (< zeroQuantity) inner)
  228       )
  229 {-# INLINEABLE pack' #-}
  230

  258 fromList :: [(K, [(K, Quantity)])] -> BuiltinResult Value
  259 fromList xs = do
  260   -- Use unchecked addition during construction
  261   let outerMap =
  262         Map.fromListWith
  263           (Map.unionWith unsafeAddQuantity) -- combine inner maps with unchecked addition
  264           (second (Map.fromListWith unsafeAddQuantity) <$> xs)
  265   -- Validate all quantities are within bounds
  266   pack <$> validateQuantities outerMap
  267 {-# INLINEABLE fromList #-}
  268 
  269 -- | Unsafe addition of quantities without bounds checking.
  270 unsafeAddQuantity :: Quantity -> Quantity -> Quantity
  271 unsafeAddQuantity (UnsafeQuantity x) (UnsafeQuantity y) = UnsafeQuantity (x + y)
  272 {-# INLINE unsafeAddQuantity #-}

  274 -- | Validate all quantities in a nested map are within bounds.
  275 validateQuantities :: HasCallStack => NestedMap -> BuiltinResult NestedMap
  276 validateQuantities nestedMap =
  277   case find isOutOfBounds allQuantities of
  278     Just (UnsafeQuantity i) -> fail $ context <> ": quantity out of bounds: " <> show i
  279     Nothing -> pure nestedMap
  280   where
  281     allQuantities = concatMap Map.elems $ Map.elems nestedMap
  282     isOutOfBounds (UnsafeQuantity i) =
  283       i < unQuantity minBound || i > unQuantity maxBound
  284     context = case getCallStack callStack of
  285       (fnName, _) : _ -> fnName
  286       [] -> "<unknown>"
  287 {-# INLINEABLE validateQuantities #-}
  288 
  289 normalize :: NestedMap -> NestedMap
  290 normalize = Map.filter (not . Map.null) . Map.map (Map.filter (/= zeroQuantity))
  291 {-# INLINEABLE normalize #-}

  298 {-| \(O(\log \max(m, k))\), where \(m\) is the size of the outer map, and \(k\) is
  299 the size of the largest inner map. -}
  300 insertCoin :: ByteString -> ByteString -> Integer -> Value -> BuiltinResult Value
  301 insertCoin unsafeCurrency unsafeToken unsafeAmount v@(Value outer sizes total neg)
  302   | unsafeAmount == 0 = pure $ deleteCoin unsafeCurrency unsafeToken v
  303   | otherwise = case (k unsafeCurrency, k unsafeToken, quantity unsafeAmount) of
  304       (Nothing, _, _) -> fail $ "insertCoin: invalid currency: " <> show (B.unpack unsafeCurrency)
  305       (_, Nothing, _) -> fail $ "insertCoin: invalid token: " <> show (B.unpack unsafeToken)
  306       (_, _, Nothing) -> fail $ "insertCoin: quantity out of bounds: " <> show unsafeAmount
  307       (Just currency, Just token, Just qty) ->
  308         let f
  309               :: Maybe (Map K Quantity)
  310               -> ( -- Left (old size of inner map) if the total size grows by 1,
  311                    -- otherwise, Right (old quantity)
  312                    Either Int Quantity
  313                  , Maybe (Map K Quantity)
  314                  )
  315             f = \case
  316               Nothing -> (Left 0, Just (Map.singleton token qty))
  317               Just inner ->
  318                 let (mOldQuantity, inner') =
  319                       Map.insertLookupWithKey (\_ _ _ -> qty) token qty inner
  320                  in (maybe (Left (Map.size inner)) Right mOldQuantity, Just inner')
  321             (res, outer') = Map.alterF f currency outer
  322             (sizes', total', neg') = case res of
  323               Left oldSize ->
  324                 ( updateSizes oldSize (oldSize + 1) sizes
  325                 , total + 1
  326                 , if qty < zeroQuantity then neg + 1 else neg
  327                 )
  328               Right oldQuantity ->
  329                 ( sizes
  330                 , total
  331                 , if oldQuantity < zeroQuantity && qty > zeroQuantity
  332                     then neg - 1
  333                     else
  334                       if oldQuantity > zeroQuantity && qty < zeroQuantity
  335                         then neg + 1
  336                         else neg
  337                 )
  338          in pure $ Value outer' sizes' total' neg'
  339 {-# INLINEABLE insertCoin #-}
  340 
  341 -- | \(O(\log \max(m, k))\)
  342 deleteCoin :: ByteString -> ByteString -> Value -> Value
  343 deleteCoin (UnsafeK -> currency) (UnsafeK -> token) (Value outer sizes total neg) =
  344   Value outer' sizes' total' neg'
  345   where
  346     (mold, outer') = Map.alterF f currency outer
  347     (sizes', total', neg') = case mold of
  348       Just (oldSize, oldQuantity) ->
  349         ( updateSizes oldSize (oldSize - 1) sizes
  350         , total - 1
  351         , if oldQuantity < zeroQuantity then neg - 1 else neg
  352         )
  353       Nothing -> (sizes, total, neg)
  354     f
  355       :: Maybe (Map K Quantity)
  356       -> ( -- Just (old size of inner map, old quantity) if the total size shrinks by 1,
  357            -- otherwise Nothing
  358            Maybe (Int, Quantity)
  359          , Maybe (Map K Quantity)
  360          )
  361     f = \case
  362       Nothing -> (Nothing, Nothing)
  363       Just inner ->
  364         let (qty, inner') = Map.updateLookupWithKey (\_ _ -> Nothing) token inner
  365          in ((Map.size inner,) <$> qty, if Map.null inner' then Nothing else Just inner')
  366 
  367 -- | \(O(\log \max(m, k))\)
  368 lookupCoin :: ByteString -> ByteString -> Value -> Integer
  369 lookupCoin (UnsafeK -> currency) (UnsafeK -> token) (unpack -> outer) =
  370   case Map.lookup currency outer of
  371     Nothing -> 0
  372     Just inner -> unQuantity $ Map.findWithDefault zeroQuantity token inner

  374 {-| \(O(n_{2}\log \max(m_{1}, k_{1}))\), where \(n_{2}\) is the total size of the second
  375 `Value`, \(m_{1}\) is the size of the outer map in the first `Value` and \(k_{1}\) is
  376 the size of the largest inner map in the first `Value`.
  377 
  378 @a@ contains @b@ if for each @(currency, token, quantity)@ in @b@,
  379 @lookup currency token a >= quantity@.
  380 
  381 Both values must not contain negative amounts. -}
  382 valueContains :: Value -> Value -> BuiltinResult Bool
  383 valueContains v1 v2
  384   | negativeAmounts v1 > 0 = fail "valueContains: first value contains negative amounts"
  385   | negativeAmounts v2 > 0 = fail "valueContains: second value contains negative amounts"
  386   | totalSize v1 < totalSize v2 = BuiltinSuccess False
  387   -- \^ v2 is too big to be contained in v1: `isSubmapOfBy` has a similar check,
  388   -- but that only applies to the top-level structure of the map, so v1 can
  389   -- have a larger outer map than v2 but a smaller number of total entries, and
  390   -- without this check the outer call of `isSubmapOfBy` would carry on and
  391   -- compare the inner maps even though v2 can't fit inside v1.
  392   | otherwise = BuiltinSuccess $ Map.isSubmapOfBy (Map.isSubmapOfBy (<=)) (unpack v2) (unpack v1)
  393 {-# INLINEABLE valueContains #-}
  394 
  395 {-| \(O(n_{1}) + O(n_{2})\), where \(n_{1}\) and \(n_{2}\) are the total sizes
  396 (i.e., sum of inner map sizes) of the two maps.
  397 
  398 Shortcircuits if either value is empty.
  399 
  400 Since 'unionValue' is commutative, we switch the arguments whenever the second
  401 value is larger in total size than the first one. We have found through experimentation
  402 that this results in better performance in practice. -}
  403 unionValue :: Value -> Value -> BuiltinResult Value
  404 unionValue vA vB
  405   | totalSize vA == 0 = BuiltinSuccess vB
  406   | totalSize vB == 0 = BuiltinSuccess vA
  407   | otherwise =
  408       pack'
  409         <$> M.mergeA
  410           M.preserveMissing
  411           M.preserveMissing
  412           ( M.zipWithMaybeAMatched \_ innerA innerB ->
  413               fmap (\inner -> if Map.null inner then Nothing else Just inner) $
  414                 M.mergeA
  415                   M.preserveMissing
  416                   M.preserveMissing
  417                   ( M.zipWithMaybeAMatched \_ x y ->
  418                       case addQuantity x y of
  419                         Just z -> pure if z == zeroQuantity then Nothing else Just z
  420                         Nothing ->
  421                           fail "unionValue: quantity is out of the signed 128-bit integer bounds"
  422                   )
  423                   innerA
  424                   innerB
  425           )
  426           v1
  427           v2
  428   where
  429     (v1, v2) =
  430       if totalSize vB > totalSize vA
  431         then (unpack vB, unpack vA)
  432         else (unpack vA, unpack vB)
  433 {-# INLINEABLE unionValue #-}

  438 {-| \(O(n)\). Encodes `Value` as `Data`, in the same way as non-builtin @Value@.
  439 This is the denotation of @ValueData@ in Plutus V1, V2 and V3. -}
  440 valueData :: Value -> BuiltinResult Data
  441 valueData v =
  442   if totalSize v <= valueDataMaxSize
  443     then pure $ Map . fmap (bimap (B . unK) tokensData) . Map.toList . unpack $ v
  444     else fail $ "valueData: maximum input size (" ++ show valueDataMaxSize ++ ") exceeded"
  445   where
  446     tokensData :: Map K Quantity -> Data
  447     tokensData = Map . fmap (bimap (B . unK) (I . unQuantity)) . Map.toList
  448 {-# INLINEABLE valueData #-}

  450 {-| \(O(n)\). Decodes `Data` into `Value`.
  451 This is the denotation of @UnValueData@ in Plutus V1, V2 and V3. -}
  452 unValueData :: Data -> BuiltinResult Value
  453 unValueData = \case
  454   Map cs -> do
  455     (outerList, sizes, total, neg) <- goCurrencies cs
  456     pure $ Value (Map.fromDistinctDescList outerList) sizes total neg
  457   _ -> fail "unValueData: non-Map constructor"
  458   where
  459     unB :: Data -> BuiltinResult K
  460     unB = \case
  461       B b -> maybe (fail $ "unValueData: invalid key: " <> show (B.unpack b)) pure (k b)
  462       _ -> fail "unValueData: non-B constructor"
  463     {-# INLINEABLE unB #-}
  464 
  465     unQ :: Data -> BuiltinResult Quantity
  466     unQ = \case
  467       I i
  468         | i == 0 || i < unQuantity minBound || i > unQuantity maxBound ->
  469             fail "unValueData: invalid quantity"
  470         | otherwise -> pure (UnsafeQuantity i)
  471       _ -> fail "unValueData: non-I constructor"
  472     {-# INLINEABLE unQ #-}
  473 
  474     -- Returns the inner map and the number of negative quantities in it.
  475     unTokens :: Data -> BuiltinResult (Map K Quantity, Int)
  476     unTokens = \case
  477       Map ts -> do
  478         when (null ts) $ fail "unValueData: empty inner map"
  479         (innerList, neg) <- goTokens ts
  480         pure (Map.fromDistinctDescList innerList, neg)
  481       _ -> fail "unValueData: non-Map constructor"
  482     {-# INLINEABLE unTokens #-}
  483 
  484     -- Returns outer map's list, plus stats (sizes, total, neg).
  485     goCurrencies
  486       :: [(Data, Data)]
  487       -> BuiltinResult ([(K, Map K Quantity)], IntMap Int, Int, Int)
  488     goCurrencies = go Nothing mempty mempty 0 0
  489       where
  490         go !prev !acc !sizes !total !neg = \case
  491           [] -> pure (acc, sizes, total, neg)
  492           (cData, tsData) : rest -> do
  493             !c <- unB cData
  494             -- Verify that currencies are strictly ascending
  495             whenJust
  496               prev
  497               ( \p ->
  498                   unless
  499                     (p < c)
  500                     (fail "unValueData: currency symbols not strictly ascending")
  501               )
  502             (!inner, !innerNeg) <- unTokens tsData
  503             let sizes' = IntMap.alter (maybe (Just 1) (Just . (+ 1))) (Map.size inner) sizes
  504                 total' = total + Map.size inner
  505                 neg' = neg + innerNeg
  506                 acc' = (c, inner) : acc
  507             go (Just c) acc' sizes' total' neg' rest
  508 
  509     -- Returns inner map's list, plus the number of negative quantities in the inner map.
  510     goTokens :: [(Data, Data)] -> BuiltinResult ([(K, Quantity)], Int)
  511     goTokens = go Nothing mempty 0
  512       where
  513         go !prev !acc !neg = \case
  514           [] -> pure (acc, neg)
  515           (tData, qData) : rest -> do
  516             !t <- unB tData
  517             -- Verify that token names within an inner map are strictly ascending
  518             whenJust
  519               prev
  520               ( \p ->
  521                   unless
  522                     (p < t)
  523                     (fail "unValueData: token names not strictly ascending")
  524               )
  525             !q <- unQ qData
  526             let neg' = if q < zeroQuantity then neg + 1 else neg
  527                 acc' = (t, q) : acc
  528             go (Just t) acc' neg' rest
  529 {-# INLINEABLE unValueData #-}

  531 -- | Decrement bucket @old@, and increment bucket @new@.
  532 updateSizes :: Int -> Int -> IntMap Int -> IntMap Int
  533 updateSizes old new = dec . inc
  534   where
  535     inc =
  536       if new == 0
  537         then id
  538         else IntMap.alter (maybe (Just 1) (Just . (+ 1))) new
  539     dec =
  540       if old == 0
  541         then id
  542         else IntMap.update (\n -> if n <= 1 then Nothing else Just (n - 1)) old
  543 {-# INLINEABLE updateSizes #-}

  545 -- | \(O(n)\). Scale each token by the given constant factor.
  546 scaleValue :: Integer -> Value -> BuiltinResult Value
  547 scaleValue c (Value outer sizes size neg)
  548   -- When scaling by positive factor, no need to change sizes and number of negative amounts.
  549   | c > 0 = do
  550       outer' <- go outer
  551       BuiltinSuccess $ Value outer' sizes size neg
  552   -- When scaling by negative factor, only need to "flip" negative amounts.
  553   | c < 0 = do
  554       outer' <- go outer
  555       BuiltinSuccess $ Value outer' sizes size (size - neg)
  556   -- Scaling by 0 is always empty value
  557   | otherwise = BuiltinSuccess empty
  558   where
  559     go :: NestedMap -> BuiltinResult NestedMap
  560     go x = traverse (traverse goScale) x
  561     goScale :: Quantity -> BuiltinResult Quantity
  562     goScale x =
  563       case scaleQuantity c x of
  564         Nothing ->
  565           fail $
  566             "scaleValue: quantity out of bounds: "
  567               <> show c
  568               <> " * "
  569               <> show (unQuantity x)
  570         Just q -> pure q
