    9 import Control.Monad.Primitive

   11 import Data.Primitive qualified as P

   16 -- See Note [Step counter data structure]
   17 -- You might think that since we can store whatever we like in here we might as well
   18 -- use machine words (i.e. 'Word64'), but that is actually slower.
   19 {-| A set of 'Word8' counters that is used in the CEK machine
   20 to count steps. -}
   21 newtype StepCounter (n :: Nat) s = StepCounter (P.MutablePrimArray s Word8)
   22 
   23 -- | Make a new 'StepCounter' with the given number of counters.
   24 newCounter :: (KnownNat n, PrimMonad m) => Proxy n -> m (StepCounter n (PrimState m))
   25 newCounter p = do
   26   let sz = fromIntegral $ natVal p
   27   c <- StepCounter <$> P.newPrimArray sz
   28   resetCounter c
   29   pure c
   30 {-# INLINE newCounter #-}
   31 
   32 -- | Reset all the counters in the given 'StepCounter' to zero.
   33 resetCounter :: forall n m. (KnownNat n, PrimMonad m) => StepCounter n (PrimState m) -> m ()
   34 resetCounter (StepCounter arr) =
   35   let sz = fromIntegral $ natVal (Proxy @n)
   36    in P.setPrimArray arr 0 sz 0
   37 {-# INLINE resetCounter #-}
   38 
   39 -- | Read the value of a counter.
   40 readCounter :: forall m n. PrimMonad m => StepCounter n (PrimState m) -> Int -> m Word8
   41 readCounter =
   42   coerce
   43     @(P.MutablePrimArray (PrimState m) Word8 -> Int -> m Word8)
   44     @(StepCounter n (PrimState m) -> Int -> m Word8)
   45     P.readPrimArray
   46 {-# INLINE readCounter #-}
   47 
   48 -- | Write to a counter.
   49 writeCounter
   50   :: forall m n
   51    . PrimMonad m
   52   => StepCounter n (PrimState m)
   53   -> Int
   54   -> Word8
   55   -> m ()
   56 writeCounter =
   57   coerce
   58     @(P.MutablePrimArray (PrimState m) Word8 -> Int -> Word8 -> m ())
   59     @(StepCounter n (PrimState m) -> Int -> Word8 -> m ())
   60     P.writePrimArray
   61 {-# INLINE writeCounter #-}
   62 
   63 -- | Modify the value of a counter. Returns the modified value.
   64 modifyCounter
   65   :: PrimMonad m
   66   => Int
   67   -> (Word8 -> Word8)
   68   -> StepCounter n (PrimState m)
   69   -> m Word8
   70 modifyCounter i f c = do
   71   v <- readCounter c i
   72   let modified = f v
   73   writeCounter c i modified
   74   pure modified
   75 {-# INLINE modifyCounter #-}
   76 
   77 -- | Traverse the counters with an effectful function.
   78 itraverseCounter_
   79   :: forall n m
   80    . (UpwardsM m (NatToPeano n), PrimMonad m)
   81   => (Int -> Word8 -> m ())
   82   -> StepCounter n (PrimState m)
   83   -> m ()
   84 itraverseCounter_ f (StepCounter arr) = do
   85   -- The safety of this operation is a little subtle. The frozen array is only
   86   -- safe to use if the underlying mutable array is not mutated 'afterwards'.
   87   -- In our case it likely _will_ be mutated afterwards... but not until we
   88   -- are done with the frozen version. That ordering is enforced by the fact that
   89   -- the whole thing runs in 'm': future accesses to the mutable array can't
   90   -- happen until this whole function is finished.
   91   arr' <- P.unsafeFreezePrimArray arr
   92   upwardsM @_ @(NatToPeano n) 0 $ \i -> f i $ P.indexPrimArray arr' i
   93 {-# INLINE itraverseCounter_ #-}
   94 
   95 -- | Traverse the counters with an effectful function.
   96 iforCounter_
   97   :: (UpwardsM m (NatToPeano n), PrimMonad m)
   98   => StepCounter n (PrimState m)
   99   -> (Int -> Word8 -> m ())
  100   -> m ()
  101 iforCounter_ = flip itraverseCounter_
  102 {-# INLINE iforCounter_ #-}
  103 
  104 {- Note [Step counter data structure]
  105 The step counter data structure has had several iterations.
  106 
  107 Previously we used a "word array", which was a single 'Word64' into which we
  108 packed 8 'Word8's. This worked pretty well: it was pure, and everything reduced
  109 to a bunch of primitive integer operations.
  110 
  111 However, it has a key limitation which is that it can only hold 8 counters.
  112 The obvious attempt to expand it to use a 'Word128' performed badly.
  113 
  114 The 'PrimArray' approach on the other hand was fairly competitive with the
  115 original 'WordArray', and scales fine to more than 8 counters, so we switched
  116 to using that instead.
  117 -}
