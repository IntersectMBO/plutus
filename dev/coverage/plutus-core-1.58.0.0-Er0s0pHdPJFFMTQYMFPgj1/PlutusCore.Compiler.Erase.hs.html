    3 import Data.Vector (fromList)

    8 {-| Erase a Typed Plutus Core term to its untyped counterpart.
    9 
   10 Restricted to Plc terms with `Name`s, because erasing a (Named-)Debruijn term will
   11 mess up its debruijn indexing and thus break scope-checking.
   12 -- FIXME: Lift this restriction of `eraseTerm` for (Named-)DeBruijn terms. -}
   13 eraseTerm
   14   :: HasUnique name TermUnique
   15   => Term tyname name uni fun ann
   16   -> UPLC.Term name uni fun ann
   17 eraseTerm (Var ann name) = UPLC.Var ann name
   18 eraseTerm (TyAbs ann _ _ body) = UPLC.Delay ann (eraseTerm body)
   19 eraseTerm (LamAbs ann name _ body) = UPLC.LamAbs ann name (eraseTerm body)
   20 eraseTerm (Apply ann fun arg) = UPLC.Apply ann (eraseTerm fun) (eraseTerm arg)
   21 eraseTerm (Constant ann con) = UPLC.Constant ann con
   22 eraseTerm (Builtin ann bn) = UPLC.Builtin ann bn
   23 eraseTerm (TyInst ann term _) = UPLC.Force ann (eraseTerm term)
   24 eraseTerm (Unwrap _ term) = eraseTerm term
   25 eraseTerm (IWrap _ _ _ term) = eraseTerm term
   26 eraseTerm (Error ann _) = UPLC.Error ann
   27 eraseTerm (Constr ann _ i args) = UPLC.Constr ann i (fmap eraseTerm args)
   28 eraseTerm (Case ann _ arg cs) = UPLC.Case ann (eraseTerm arg) (fromList $ fmap eraseTerm cs)
   29 
   30 eraseProgram
   31   :: HasUnique name TermUnique
   32   => Program tyname name uni fun ann
   33   -> UPLC.Program name uni fun ann
   34 eraseProgram (Program a v t) = UPLC.Program a v $ eraseTerm t
