   35 {-| @List@ as a PLC type.
   36 
   37 > fix \(list :: * -> *) (a :: *) -> all (r :: *). r -> (a -> list a -> r) -> r -}
   38 listData :: RecursiveType uni fun ()
   39 listData = runQuote $ do
   40   a <- freshTyName "a"
   41   list <- freshTyName "list"
   42   r <- freshTyName "r"
   43   let listA = TyApp () (TyVar () list) (TyVar () a)
   44   makeRecursiveType () list [TyVarDecl () a $ Type ()]
   45     . TyForall () r (Type ())
   46     . TyFun () (TyVar () r)
   47     . TyFun () (TyFun () (TyVar () a) . TyFun () listA $ TyVar () r)
   48     $ TyVar () r
   49 
   50 listTy :: Type TyName uni ()
   51 listTy = _recursiveType listData
   52 
   53 {-|  '[]' as a PLC term.
   54 
   55 >  /\(a :: *) -> wrapList [a] /\(r :: *) -> \(z : r) (f : a -> list a -> r) -> z) -}
   56 nil :: TermLike term TyName Name uni fun => term ()
   57 nil = runQuote $ do
   58   let RecursiveType list wrapList = listData
   59   a <- freshTyName "a"
   60   r <- freshTyName "r"
   61   z <- freshName "z"
   62   f <- freshName "f"
   63   let listA = TyApp () list (TyVar () a)
   64   return
   65     . tyAbs () a (Type ())
   66     . wrapList [TyVar () a]
   67     . tyAbs () r (Type ())
   68     . lamAbs () z (TyVar () r)
   69     . lamAbs () f (TyFun () (TyVar () a) . TyFun () listA $ TyVar () r)
   70     $ var () z

   72 {-|  '(:)' as a PLC term.
   73 
   74 > /\(a :: *) -> \(x : a) (xs : list a) ->
   75 >     wrapList [a] /\(r :: *) -> \(z : r) (f : a -> list a -> r) -> f x xs -}
   76 cons :: TermLike term TyName Name uni fun => term ()
   77 cons = runQuote $ do
   78   let RecursiveType list wrapList = listData
   79   a <- freshTyName "a"
   80   x <- freshName "x"
   81   xs <- freshName "xs"
   82   r <- freshTyName "r"
   83   z <- freshName "z"
   84   f <- freshName "f"
   85   let listA = TyApp () list (TyVar () a)
   86   return
   87     . tyAbs () a (Type ())
   88     . lamAbs () x (TyVar () a)
   89     . lamAbs () xs listA
   90     . wrapList [TyVar () a]
   91     . tyAbs () r (Type ())
   92     . lamAbs () z (TyVar () r)
   93     . lamAbs () f (TyFun () (TyVar () a) . TyFun () listA $ TyVar () r)
   94     $ mkIterAppNoAnn
   95       (var () f)
   96       [ var () x
   97       , var () xs
   98       ]

  100 {-|  @foldrList@ as a PLC term.
  101 
  102 > /\(a :: *) (r :: *) -> \(f : a -> r -> r) (z : r) ->
  103 >     fix {list a} {r} \(rec : list a -> r) (xs : list a) ->
  104 >         unwrap xs {r} z \(x : a) (xs' : list a) -> f x (rec xs') -}
  105 foldrList :: TermLike term TyName Name uni fun => term ()
  106 foldrList = runQuote $ do
  107   a <- freshTyName "a"
  108   r <- freshTyName "r"
  109   f <- freshName "f"
  110   z <- freshName "z"
  111   rec <- freshName "rec"
  112   xs <- freshName "xs"
  113   x <- freshName "x"
  114   xs' <- freshName "xs'"
  115   let listA = TyApp () listTy (TyVar () a)
  116   return
  117     . tyAbs () a (Type ())
  118     . tyAbs () r (Type ())
  119     . lamAbs () f (TyFun () (TyVar () a) . TyFun () (TyVar () r) $ TyVar () r)
  120     . lamAbs () z (TyVar () r)
  121     . apply () (mkIterInstNoAnn fix [listA, TyVar () r])
  122     . lamAbs () rec (TyFun () listA $ TyVar () r)
  123     . lamAbs () xs listA
  124     . apply () (apply () (tyInst () (unwrap () (var () xs)) $ TyVar () r) $ var () z)
  125     . lamAbs () x (TyVar () a)
  126     . lamAbs () xs' listA
  127     $ mkIterAppNoAnn
  128       (var () f)
  129       [ var () x
  130       , apply () (var () rec) $ var () xs'
  131       ]

  133 {-|  @map@ as a PLC term.
  134 
  135 > /\(a :: *) (b :: *) -> \(f : a -> b) ->
  136 >     foldrList {a} {list b} (\(x : a) -> cons {b} (f x)) (nil {b}) -}
  137 map :: TermLike term TyName Name uni fun => term ()
  138 map = runQuote $ do
  139   a <- freshTyName "a"
  140   b <- freshTyName "b"
  141   f <- freshName "f"
  142   x <- freshName "x"
  143   return
  144     . tyAbs () a (Type ())
  145     . tyAbs () b (Type ())
  146     . lamAbs () f (TyFun () (TyVar () a) $ TyVar () b)
  147     . mkIterAppNoAnn (mkIterInstNoAnn foldrList [TyVar () a, TyApp () listTy $ TyVar () b])
  148     $ [ lamAbs () x (TyVar () a)
  149           . apply () (tyInst () cons (TyVar () b))
  150           . apply () (var () f)
  151           $ var () x
  152       , tyInst () nil $ TyVar () b
  153       ]

  155 {-|  'foldl\'' as a PLC term.
  156 
  157 > /\(a :: *) (r :: *) -> \(f : r -> a -> r) ->
  158 >     fix {r} {list a -> r} \(rec : r -> list a -> r) (z : r) (xs : list a) ->
  159 >         unwrap xs {r} z \(x : a) (xs' : list a) -> rec (f z x) xs' -}
  160 foldList :: TermLike term TyName Name uni fun => term ()
  161 foldList = runQuote $ do
  162   a <- freshTyName "a"
  163   r <- freshTyName "r"
  164   f <- freshName "f"
  165   rec <- freshName "rec"
  166   z <- freshName "z"
  167   xs <- freshName "xs"
  168   x <- freshName "x"
  169   xs' <- freshName "xs'"
  170   let listA = TyApp () listTy (TyVar () a)
  171   return
  172     . tyAbs () a (Type ())
  173     . tyAbs () r (Type ())
  174     . lamAbs () f (TyFun () (TyVar () r) . TyFun () (TyVar () a) $ TyVar () r)
  175     . apply () (mkIterInstNoAnn fix [TyVar () r, TyFun () listA $ TyVar () r])
  176     . lamAbs () rec (TyFun () (TyVar () r) . TyFun () listA $ TyVar () r)
  177     . lamAbs () z (TyVar () r)
  178     . lamAbs () xs listA
  179     . apply () (apply () (tyInst () (unwrap () (var () xs)) $ TyVar () r) $ var () z)
  180     . lamAbs () x (TyVar () a)
  181     . lamAbs () xs' listA
  182     . mkIterAppNoAnn (var () rec)
  183     $ [ mkIterAppNoAnn (var () f) [var () z, var () x]
  184       , var () xs'
  185       ]

  187 {-|  'reverse' as a PLC term.
  188 
  189 > /\(a :: *) -> \(xs : list a) ->
  190 >     foldList {a} {list a} (\(r : list a) (x : a) -> cons {a} x r) (nil {a}) -}
  191 reverse :: TermLike term TyName Name uni fun => term ()
  192 reverse = runQuote $ do
  193   a <- freshTyName "a"
  194   xs <- freshName "xs"
  195   x <- freshName "x"
  196   r <- freshName "r"
  197   let vA = TyVar () a
  198       listA = TyApp () listTy vA
  199   return
  200     . tyAbs () a (Type ())
  201     . lamAbs () xs listA
  202     $ mkIterAppNoAnn
  203       (mkIterInstNoAnn foldList [vA, listA])
  204       [ lamAbs () r listA . lamAbs () x vA $
  205           mkIterAppNoAnn (tyInst () cons vA) [var () x, var () r]
  206       , tyInst () nil vA
  207       , var () xs
  208       ]

  210 {-| 'enumFromTo' as a PLC term
  211 
  212 > \(n m : integer) ->
  213 >     fix {integer} {list (integer)}
  214 >         (\(rec : integer -> list (integer)) (n' : integer) ->
  215 >             ifThenElse {list (integer)}
  216 >                 (lessThanEqualsInteger n' m)
  217 >                 (cons {integer} n' (rec (succInteger n')))
  218 >                 (nil {integer}))
  219 >         n -}
  220 enumFromTo
  221   :: ( TermLike term TyName Name uni DefaultFun
  222      , uni `HasTypeAndTermLevel` Integer
  223      , uni `HasTypeAndTermLevel` ()
  224      , uni `HasTypeAndTermLevel` Bool
  225      )
  226   => term ()
  227 enumFromTo = runQuote $ do
  228   n <- freshName "n"
  229   m <- freshName "m"
  230   rec <- freshName "rec"
  231   n' <- freshName "n'"
  232   u <- freshName "u"
  233   let leqInteger = builtin () LessThanEqualsInteger
  234       int = mkTyBuiltin @_ @Integer ()
  235       listInt = TyApp () listTy int
  236   return
  237     . lamAbs () n int
  238     . lamAbs () m int
  239     . mkIterAppNoAnn (mkIterInstNoAnn fix [int, listInt])
  240     $ [ lamAbs () rec (TyFun () int listInt)
  241           . lamAbs () n' int
  242           . mkIterAppNoAnn (tyInst () ifThenElse listInt)
  243           $ [ mkIterAppNoAnn leqInteger [var () n', var () m]
  244             , lamAbs () u unit $
  245                 mkIterAppNoAnn
  246                   (tyInst () cons int)
  247                   [ var () n'
  248                   , apply () (var () rec)
  249                       . apply () succInteger
  250                       $ var () n'
  251                   ]
  252             , lamAbs () u unit $ tyInst () nil int
  253             ]
  254       , var () n
  255       ]
