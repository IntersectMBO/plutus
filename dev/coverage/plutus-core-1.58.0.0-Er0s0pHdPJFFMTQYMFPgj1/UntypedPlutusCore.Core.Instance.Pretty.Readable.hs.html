   19 -- | Split an iterated 'LamAbs' (if any) into a list of variables that it binds and its body.
   20 viewLamAbs :: Term name uni fun ann -> Maybe ([name], Term name uni fun ann)
   21 viewLamAbs term0@LamAbs {} = Just $ go term0
   22   where
   23     go (LamAbs _ name body) = first (name :) $ go body
   24     go term = ([], term)
   25 viewLamAbs _ = Nothing
   26 
   27 -- | Split an iterated 'Apply' (if any) into the head of the application and the spine.
   28 viewApp
   29   :: Term name uni fun ann
   30   -> Maybe (Term name uni fun ann, [Term name uni fun ann])
   31 viewApp term0 = go term0 []
   32   where
   33     go (Apply _ fun arg) args = go fun $ arg : args
   34     go _ [] = Nothing
   35     go fun args = Just (fun, args)
   36 
   37 instance
   38   (PrettyReadableBy configName name, PrettyUni uni, Pretty fun, Show configName)
   39   => PrettyBy (PrettyConfigReadable configName) (Term name uni fun a)
   40   where
   41   prettyBy = inContextM $ \case
   42     Constant _ val -> lmap (ConstConfig . _pcrRenderContext) $ prettyM val
   43     Builtin _ bi -> unitDocM $ pretty bi
   44     Var _ name -> prettyM name
   45     (viewLamAbs -> Just (args, body)) -> iterLamAbsPrettyM args body
   46     LamAbs {} -> error "Panic: 'LamAbs' is not covered by 'viewLamAbs'"
   47     (viewApp -> Just (fun, args)) -> iterAppPrettyM fun args
   48     Apply {} -> error "Panic: 'Apply' is not covered by 'viewApp'"
   49     Delay _ term -> iterAppDocM $ \_ prettyArg -> "delay" :| [prettyArg term]
   50     Force _ term -> iterAppDocM $ \_ prettyArg -> "force" :| [prettyArg term]
   51     Error _ -> unitDocM "error"
   52     -- Always rendering the tag on the same line for more compact output, it's just a tiny integer
   53     -- anyway.
   54     Constr _ i es -> iterAppDocM $ \_ prettyArg ->
   55       ("constr" <+> prettyArg i) :| [prettyArg es]
   56     Case _ arg cs -> iterAppDocM $ \_ prettyArg -> "case" :| [prettyArg arg, prettyArg (toList cs)]
   57 
   58 instance
   59   PrettyReadableBy configName (Term name uni fun a)
   60   => PrettyBy (PrettyConfigReadable configName) (Program name uni fun a)
   61   where
   62   prettyBy = inContextM $ \(Program _ version term) ->
   63     iterAppDocM $ \_ prettyArg -> "program" :| [pretty version, prettyArg term]
