   40 -- | Configuration for the readable pretty-printing.
   41 data PrettyConfigReadable configName = PrettyConfigReadable
   42   { _pcrConfigName :: configName
   43   , _pcrRenderContext :: RenderContext
   44   , _pcrShowKinds :: ShowKinds

   73 {-| For rendering things in a readable manner regardless of the pretty-printing function chosen.
   74 I.e. all of 'show', 'pretty', 'prettyClassic' will use 'PrettyReadable' instead of doing what
   75 they normally do. @prettyBy config (AsReadable x)@ requires @config@ to have a 'PrettyConfigName'
   76 and respects it.
   77 
   78 This wrapper can be particularly useful if you want to apply a function having a 'Show' or
   79 'Pretty' or 'PrettyClassic' or 'PrettyPlc' or whatever constraint, but want to get the argument
   80 rendered in a readable manner instead. -}
   81 newtype AsReadable a = AsReadable
   82   { unAsReadable :: a
   83   }
   84 
   85 instance
   86   (HasPrettyConfigName config, PrettyReadable a)
   87   => DefaultPrettyBy config (AsReadable a)
   88   where
   89   defaultPrettyBy config (AsReadable x) =
   90     prettyBy (botPrettyConfigReadable (toPrettyConfigName config) def) x

   92 instance PrettyReadable a => Show (AsReadable a) where

  103 -- | A value of type @a@ to render in parens using the readable pretty-printer.
  104 data Parened a = Parened
  105   { parenOpening :: String
  106   , parenClosing :: String
  107   , parenedValue :: a
  108   }
  109 
  110 instance
  111   PrettyReadableBy configName a
  112   => PrettyBy (PrettyConfigReadable configName) (Parened a)
  113   where
  114   prettyBy config (Parened opening closing x) =
  115     fold
  116       [ pretty opening
  117       , prettyBy (config & renderContext .~ botRenderContext) x
  118       , pretty closing
  119       ]
  120 
  121 {-| Enclose the given value, so that it's rendered inside of braces with no additional parens
  122 regardless of the 'RenderContext'. -}
  123 inBraces :: a -> Parened a
  124 inBraces = Parened "{" "}"

  142 {-| Lay out an iterated binder. For example, this function lays out iterated lambdas either as
  143 
  144 > \(x : a) (y : b) (z : c) -> body
  145 
  146 or as
  147 
  148 > \(x : a)
  149 >  (y : b)
  150 >  (z : c) ->
  151 >   body -}
  152 iterBinderPrettyM
  153   :: ( MonadPrettyReadable configName env m
  154      , PrettyReadableBy configName arg
  155      , PrettyReadableBy configName body
  156      )
  157   => (Doc ann -> Doc ann)
  158   -> [arg]
  159   -> body
  160   -> m (Doc ann)
  161 iterBinderPrettyM enframe args body =
  162   infixDocM binderFixity $ \prettyBind prettyBody ->
  163     let prettyBinds = align . sep $ map prettyBind args
  164      in enframe prettyBinds <?> prettyBody body
  165 
  166 -- | Lay out an iterated 'TyForall' via 'iterBinderPrettyM'.
  167 iterTyForallPrettyM
  168   :: ( MonadPrettyReadable configName env m
  169      , PrettyReadableBy configName arg
  170      , PrettyReadableBy configName body
  171      )
  172   => [arg]
  173   -> body
  174   -> m (Doc ann)
  175 iterTyForallPrettyM = iterBinderPrettyM $ \binds -> "all" <+> binds <> "."
  176 
  177 -- | Lay out an iterated 'LamAbs' via 'iterBinderPrettyM'.
  178 iterLamAbsPrettyM
  179   :: ( MonadPrettyReadable configName env m
  180      , PrettyReadableBy configName arg
  181      , PrettyReadableBy configName body
  182      )
  183   => [arg]
  184   -> body
  185   -> m (Doc ann)
  186 iterLamAbsPrettyM = iterBinderPrettyM $ \binds -> "\\" <> binds <+> "->"
  187 
  188 -- | Lay out an iterated 'TyAbs' via 'iterBinderPrettyM'.
  189 iterTyAbsPrettyM
  190   :: ( MonadPrettyReadable configName env m
  191      , PrettyReadableBy configName arg
  192      , PrettyReadableBy configName body
  193      )
  194   => [arg]
  195   -> body
  196   -> m (Doc ann)
  197 iterTyAbsPrettyM = iterBinderPrettyM $ \binds -> "/\\" <> binds <+> "->"
  198 
  199 -- | Lay out an iterated @->@.
  200 iterArrowPrettyM
  201   :: (MonadPrettyReadable configName env m, PrettyReadableBy configName a)
  202   => [a]
  203   -> a
  204   -> m (Doc ann)
  205 iterArrowPrettyM args res =
  206   infixDocM arrowFixity $ \prettyArg prettyRes ->
  207     align . sep $ map (\x -> prettyArg x <+> "->") args ++ [prettyRes res]

  209 -- | The type of a 'PrettyConfigReadable'-based pretty-printer, similar to 'AnyToDoc'.
  210 type ReadableToDoc configName ann = forall a. PrettyReadableBy configName a => a -> Doc ann

  212 {-| Lay out an iteration application, providing to the caller a function to render the head of the
  213 application and a function to render each of the arguments. -}
  214 iterAppDocM
  215   :: MonadPrettyContext config env m
  216   => (AnyToDoc config ann -> AnyToDoc config ann -> NonEmpty (Doc ann))
  217   -> m (Doc ann)
  218 iterAppDocM k =
  219   infixDocM juxtFixity $ \prettyFun prettyArg ->
  220     let fun :| args = k prettyFun prettyArg
  221      in if null args
  222           then fun
  223           else fun <?> vsep args
  224 
  225 {-| Lay out iterated function applications either as
  226 
  227 > foo x y z
  228 
  229 or as
  230 
  231 > foo
  232 >   x
  233 >   y
  234 >   z -}
  235 iterAppPrettyM
  236   :: ( MonadPrettyContext config env m
  237      , PrettyBy config fun
  238      , PrettyBy config term
  239      )
  240   => fun
  241   -> [term]
  242   -> m (Doc ann)
  243 iterAppPrettyM fun args =
  244   iterAppDocM $ \prettyFun prettyArg ->
  245     prettyFun fun :| map prettyArg args
  246 
  247 {-| Lay out interleaved function applications either as
  248 
  249 > foo {a} x {b} y z
  250 
  251 or as
  252 
  253 > foo
  254 >   {a}
  255 >   x
  256 >   {b}
  257 >   y
  258 >   z
  259 
  260 'Left's are laid out in braces, 'Right's are laid out without braces. -}
  261 iterInterAppPrettyM
  262   :: ( MonadPrettyReadable configName env m
  263      , PrettyReadableBy configName fun
  264      , PrettyReadableBy configName ty
  265      , PrettyReadableBy configName term
  266      )
  267   => fun
  268   -> [Either ty term]
  269   -> m (Doc ann)
  270 iterInterAppPrettyM fun args =
  271   iterAppDocM $ \prettyFun prettyArg ->
  272     let ppArg (Left ty) = prettyArg $ inBraces ty
  273         ppArg (Right term) = prettyArg term
  274      in prettyFun fun :| map ppArg args
