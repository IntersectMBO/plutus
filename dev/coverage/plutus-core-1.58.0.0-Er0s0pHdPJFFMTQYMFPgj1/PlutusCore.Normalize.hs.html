   15 -- See Note [Normalization].
   16 -- | Normalize a 'Type'.
   17 normalizeType
   18   :: (HasUnique tyname TypeUnique, MonadNormalizeType uni m)
   19   => Type tyname uni ann -> m (Normalized (Type tyname uni ann))
   20 normalizeType = rename >=> runNormalizeTypeT . normalizeTypeM
   21 
   22 -- | Normalize every 'Type' in a 'Term'.
   23 normalizeTypesIn
   24   :: (HasUnique tyname TypeUnique, HasUnique name TermUnique, MonadNormalizeType uni m)
   25   => Term tyname name uni fun ann -> m (Term tyname name uni fun ann)
   26 normalizeTypesIn = rename >=> runNormalizeTypeT . normalizeTypesInM
   27 
   28 -- | Normalize every 'Type' in a 'Program'.
   29 normalizeTypesInProgram
   30   :: (HasUnique tyname TypeUnique, HasUnique name TermUnique, MonadNormalizeType uni m)
   31   => Program tyname name uni fun ann -> m (Program tyname name uni fun ann)
   32 normalizeTypesInProgram (Program x v t) = Program x v <$> normalizeTypesIn t
