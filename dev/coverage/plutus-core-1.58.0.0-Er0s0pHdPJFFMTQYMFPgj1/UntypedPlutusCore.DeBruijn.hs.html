   44 {-| Convert a 'Term' with 'Name's into a 'Term' with 'DeBruijn's.
   45 Will throw an error if a free variable is encountered. -}
   46 deBruijnTerm
   47   :: MonadError FreeVariableError m
   48   => Term Name uni fun ann -> m (Term NamedDeBruijn uni fun ann)
   49 deBruijnTerm = deBruijnTermWith freeUniqueThrow

   51 {-| Convert a 'Term' with 'DeBruijn's into a 'Term' with 'Name's.
   52 Will throw an error if a free variable is encountered. -}
   53 unDeBruijnTerm
   54   :: (MonadQuote m, MonadError FreeVariableError m)
   55   => Term NamedDeBruijn uni fun ann -> m (Term Name uni fun ann)
   56 unDeBruijnTerm = unDeBruijnTermWith freeIndexThrow

   58 -- | Takes a "handler" function to execute when encountering free variables.
   59 deBruijnTermWith
   60   :: Monad m
   61   => (Unique -> ReaderT LevelInfo m Index)
   62   -> Term Name uni fun ann
   63   -> m (Term NamedDeBruijn uni fun ann)
   64 deBruijnTermWith = (runDeBruijnT .) . deBruijnTermWithM

   66 -- | Takes a "handler" function to execute when encountering free variables.
   67 unDeBruijnTermWith
   68   :: MonadQuote m
   69   => (Index -> ReaderT LevelInfo m Unique)
   70   -> Term NamedDeBruijn uni fun ann
   71   -> m (Term Name uni fun ann)
   72 unDeBruijnTermWith = (runDeBruijnT .) . unDeBruijnTermWithM

   74 deBruijnTermWithM
   75   :: MonadReader LevelInfo m
   76   => (Unique -> m Index)
   77   -> Term Name uni fun ann
   78   -> m (Term NamedDeBruijn uni fun ann)
   79 deBruijnTermWithM h = go
   80   where
   81     go = \case
   82       -- variable case
   83       Var ann n -> Var ann <$> nameToDeBruijn h n
   84       -- binder cases
   85       LamAbs ann n t -> declareUnique n $ do
   86         n' <- nameToDeBruijn h n
   87         withScope $ LamAbs ann n' <$> go t
   88       -- boring recursive cases
   89       Apply ann t1 t2 -> Apply ann <$> go t1 <*> go t2
   90       Delay ann t -> Delay ann <$> go t
   91       Force ann t -> Force ann <$> go t
   92       Constr ann i es -> Constr ann i <$> traverse go es
   93       Case ann arg cs -> Case ann <$> go arg <*> traverse go cs
   94       -- boring non-recursive cases
   95       Constant ann con -> pure $ Constant ann con
   96       Builtin ann bn -> pure $ Builtin ann bn
   97       Error ann -> pure $ Error ann
   98 
   99 -- | Takes a "handler" function to execute when encountering free variables.
  100 unDeBruijnTermWithM
  101   :: (MonadReader LevelInfo m, MonadQuote m)
  102   => (Index -> m Unique)
  103   -> Term NamedDeBruijn uni fun ann
  104   -> m (Term Name uni fun ann)
  105 unDeBruijnTermWithM h = go
  106   where
  107     go = \case
  108       -- variable case
  109       Var ann n -> Var ann <$> deBruijnToName h n
  110       -- binder cases
  111       LamAbs ann n t ->
  112         -- See Note [DeBruijn indices of Binders]
  113         declareBinder $ do
  114           n' <- deBruijnToName h $ set index deBruijnInitIndex n
  115           withScope $ LamAbs ann n' <$> go t
  116       -- boring recursive cases
  117       Apply ann t1 t2 -> Apply ann <$> go t1 <*> go t2
  118       Delay ann t -> Delay ann <$> go t
  119       Force ann t -> Force ann <$> go t
  120       Constr ann i es -> Constr ann i <$> traverse go es
  121       Case ann arg cs -> Case ann <$> go arg <*> traverse go cs
  122       -- boring non-recursive cases
  123       Constant ann con -> pure $ Constant ann con
  124       Builtin ann bn -> pure $ Builtin ann bn
  125       Error ann -> pure $ Error ann
