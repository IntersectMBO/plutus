    1 {-| Machinery defined in this module allows to export mulptiple Plutus Core definitions
    2 (types and terms) as a single value which enables convenient testing of various procedures
    3 (pretty-printing, type checking, etc): each time a function / data type is added to that value,
    4 none of the tests is required to be adapted, instead all the tests see the new definition
    5 automatically. -}
    6 module PlutusCore.FsTree
    7   ( FsTree (..)
    8   , FolderContents (..)
    9   , PlcEntity (..)
   10   , PlcFsTree
   11   , PlcFolderContents
   12   , treeFolderContents
   13   , plcTypeFile
   14   , plcTermFile
   15   , foldFsTree
   16   , foldPlcFsTree
   17   , foldPlcFolderContents
   18   ) where
   19 
   20 import PlutusCore.Core
   21 import PlutusCore.Name.Unique
   22 
   23 -- We use 'String's for names, because 'FilePath's are 'String's.
   24 -- | An 'FsTree' is either a file or a folder with a list of 'FsTree's inside.
   25 data FsTree a
   26   = FsFolder String (FolderContents a)
   27   | FsFile String a
   28 
   29 {-| The contents of a folder. A wrapper around @[FsTree a]@.
   30 Exists because of its 'Semigroup' instance which allows to concatenate two 'FolderContents's
   31 without placing them into the same folder immediately, so we can have various PLC "modules"
   32 (@stdlib@, @examples@, etc), define compound modules (e.g. @stdlib <> examples@) and run various
   33 tests (pretty-printing, type synthesis, etc) against simple and compound modules uniformly. -}
   34 newtype FolderContents a = FolderContents
   35   { unFolderContents :: [FsTree a]
   36   }
   37   deriving newtype (Semigroup, Monoid)
   38 
   39 -- | A 'PlcEntity' is either a 'Type' or a 'Term'.
   40 data PlcEntity uni fun
   41   = PlcType (Type TyName uni ())
   42   | PlcTerm (Term TyName Name uni fun ())
   43 
   44 type PlcFsTree uni fun = FsTree (PlcEntity uni fun)
   45 type PlcFolderContents uni fun = FolderContents (PlcEntity uni fun)
   46 
   47 -- | Construct an 'FsTree' out of the name of a folder and a list of 'FsTree's.
   48 treeFolderContents :: String -> [FsTree a] -> FsTree a
   49 treeFolderContents name = FsFolder name . FolderContents
   50 
   51 -- | Construct a single-file 'PlcFsTree' out of a type.
   52 plcTypeFile :: String -> Type TyName uni () -> PlcFsTree uni fun
   53 plcTypeFile name = FsFile name . PlcType
   54 
   55 -- | Construct a single-file 'PlcFsTree' out of a term.
   56 plcTermFile :: String -> Term TyName Name uni fun () -> PlcFsTree uni fun
   57 plcTermFile name = FsFile name . PlcTerm
   58 
   59 -- | Fold a 'FsTree'.
   60 foldFsTree
   61   :: (String -> [b] -> b)
   62   -- ^ What to do on a folder.
   63   -> (String -> a -> b)
   64   -- ^ What to do on a single file in a folder.
   65   -> FsTree a
   66   -> b
   67 foldFsTree onFolder onFile = go
   68   where
   69     go (FsFolder name (FolderContents trees)) = onFolder name $ map go trees
   70     go (FsFile name x) = onFile name x
   71 
   72 -- | Fold a 'PlcFsTree'.
   73 foldPlcFsTree
   74   :: (String -> [b] -> b)
   75   -- ^ What to do on a folder.
   76   -> (String -> Type TyName uni () -> b)
   77   -- ^ What to do on a type.
   78   -> (String -> Term TyName Name uni fun () -> b)
   79   -- ^ What to do on a term.
   80   -> PlcFsTree uni fun
   81   -> b
   82 foldPlcFsTree onFolder onType onTerm = foldFsTree onFolder onFile
   83   where
   84     onFile name (PlcType getTy) = onType name getTy
   85     onFile name (PlcTerm getTerm) = onTerm name getTerm
   86 
   87 -- | Fold the contents of a PLC folder.
   88 foldPlcFolderContents
   89   :: (String -> [b] -> b)
   90   -- ^ What to do on a folder.
   91   -> (String -> Type TyName uni () -> b)
   92   -- ^ What to do on a type.
   93   -> (String -> Term TyName Name uni fun () -> b)
   94   -- ^ What to do on a term.
   95   -> PlcFolderContents uni fun
   96   -> [b]
   97 foldPlcFolderContents onFolder onType onTerm (FolderContents trees) =
   98   map (foldPlcFsTree onFolder onType onTerm) trees
