   16 -- | A 'BuiltinRuntime' represents a possibly partial builtin application, including an empty
   17 -- builtin application (i.e. just the builtin with no arguments).
   18 --
   19 -- Applying or type-instantiating a builtin peels off the corresponding constructor from its
   20 -- 'BuiltinRuntime'.
   21 --
   22 -- 'BuiltinCostedResult' contains the cost (an 'ExBudgetStream') and the result (a
   23 -- @BuiltinResult (HeadSpine val)@) of the builtin application. The cost is stored strictly, since
   24 -- the evaluator is going to look at it and the result is stored lazily, since it's not supposed to
   25 -- be forced before accounting for the cost of the application. If the cost exceeds the available
   26 -- budget, the evaluator discards the result of the builtin application without ever forcing it and
   27 -- terminates with evaluation failure. Allowing the user to compute something that they don't have
   28 -- the budget for would be a major bug.
   29 --
   30 -- Evaluators that ignore the entire concept of costing (e.g. the CK machine) may of course force
   31 -- the result of the builtin application unconditionally.
   32 data BuiltinRuntime val
   33     = BuiltinCostedResult ExBudgetStream ~(BuiltinResult val)
   34     | BuiltinExpectArgument (val -> BuiltinRuntime val)
   35     | BuiltinExpectForce (BuiltinRuntime val)
   36 
   37 instance NoThunks (BuiltinRuntime val) where
   38     wNoThunks ctx = \case
   39         -- Unreachable, because we don't allow nullary builtins and the 'BuiltinArrow' case only
   40         -- checks for WHNF without recursing. Hence we can throw if we reach this clause somehow.
   41         -- TODO: remove the CPP when rest of IOE moves to nothunks>=0.2
   42 #if MIN_VERSION_nothunks(0,3,0)
   43         BuiltinCostedResult _ _    -> pure . Just $ ThunkInfo ctx Nothing
   44 #elif MIN_VERSION_nothunks(0,2,0)
   45         BuiltinCostedResult _ _    -> pure . Just . ThunkInfo $ Left ctx
   46 #else
   47         -- Plutus has moved to nothunks>=0.2, but some other IOE repos are using nothunks<0.2.
   48         -- As a consequence, cardano-constitution:create-json-envelope cannot be build.
   49         -- This is a workaround to make it build (default is buildable:False). See `cabal.project`
   50         BuiltinCostedResult _ _    -> pure . Just $ ThunkInfo ctx
   51 #endif
   52         -- This one doesn't do much. It only checks that the function stored in the 'BuiltinArrow'
   53         -- is in WHNF. The function may contain thunks inside of it. Not sure if it's possible to do
   54         -- better, since the final 'BuiltinCostedResult' contains a thunk for the result of the
   55         -- builtin application anyway.
   56         BuiltinExpectArgument f    -> noThunks ctx f
   57         BuiltinExpectForce runtime -> noThunks ctx runtime
   58 
   59     showTypeOf = const "PlutusCore.Builtin.Runtime.BuiltinRuntime"
   60 
   61 instance NFData (BuiltinRuntime val) where
   62     -- 'BuiltinRuntime' is strict (verified by the 'NoThunks' tests), hence we only need to force
   63     -- this to WHNF to get it forced to NF.
   64     rnf = rwhnf

   66 {- Note [DO NOT newtype-wrap functions]
   67 If you put a functional type into a @newtype@ rather than strict @data@, the GHC optimizer will be
   68 able to see through it and will feel compelled to rearrange lambdas and case-expressions as it sees
   69 fit, which it often does incorrectly due to @-fpedantic-bottoms@ not being enabled by default.
   70 
   71 We've had a very confusing case of GHC turning
   72 
   73     case y of
   74         y' -> \x -> z
   75 
   76 into
   77 
   78     \x -> case y of
   79         y' -> z
   80 
   81 and ruining carefully set up optimizations
   82 (see https://github.com/IntersectMBO/plutus/pull/4914#issuecomment-1396306606).
   83 
   84 Even strict @data@ is slower than @newtype@, but in practice GHC is going to
   85 worker-wrapper-transform such code making it as efficient as the @newtype@ version.
   86 
   87 So you should never newtype-wrap functions, use strict @data@ instead.
   88 -}

   90 -- See Note [DO NOT newtype-wrap functions].
   91 -- | A @data@ wrapper around a function returning the 'BuiltinRuntime' of a built-in function.
   92 --
   93 -- In order for lookups to be efficient the 'BuiltinRuntime's need to be cached, i.e. pulled out
   94 -- of the function statically. See 'makeBuiltinMeaning' for how we achieve that.
   95 data BuiltinsRuntime fun val = BuiltinsRuntime
   96     { unBuiltinsRuntime :: fun -> BuiltinRuntime val
   97     }
   98 
   99 instance (Bounded fun, Enum fun) => NFData (BuiltinsRuntime fun val) where
  100     -- Force every 'BuiltinRuntime' stored in the environment.
  101     rnf (BuiltinsRuntime env) = foldr (\fun res -> env fun `seq` res) () enumerate
  102 
  103 instance (Bounded fun, Enum fun) => NoThunks (BuiltinsRuntime fun val) where
  104     -- Ensure that every 'BuiltinRuntime' doesn't contain thunks after forcing it initially
  105     -- (we can't avoid the initial forcing, because we can't lookup the 'BuiltinRuntime' without
  106     -- forcing it, see https://stackoverflow.com/q/63441862).
  107     wNoThunks ctx (BuiltinsRuntime env) = allNoThunks $ map (wNoThunks ctx . env) enumerate
  108     showTypeOf = const "PlutusCore.Builtin.Runtime.BuiltinsRuntime"
  109 
  110 builtinRuntimeFailure :: BuiltinError -> BuiltinRuntime val
  111 builtinRuntimeFailure = BuiltinCostedResult (ExBudgetLast mempty) . throwError
  112 -- See Note [INLINE and OPAQUE on error-related definitions].
  113 {-# OPAQUE builtinRuntimeFailure #-}

  115 -- | Look up the runtime info of a built-in function during evaluation.
  116 lookupBuiltin :: fun -> BuiltinsRuntime fun val -> BuiltinRuntime val
  117 lookupBuiltin fun (BuiltinsRuntime env) = env fun
