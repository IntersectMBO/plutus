   24 {- Note [Tree]
   25 Here we encode the following:
   26 
   27     mutual
   28       data Tree (A : Set) : Set where
   29         node : A -> Forest A -> Tree A
   30 
   31       data Forest (A : Set) : Set where
   32         nil  : Forest A
   33         cons : Tree A -> Forest A -> Forest A
   34 
   35     mutual
   36       mapTree : ∀ A B -> (A -> B) -> Tree A -> Tree B
   37       mapTree A B f (node x forest) = node (f x) (mapForest A B f forest)
   38 
   39       mapForest : ∀ A B -> (A -> B) -> Forest A -> Forest B
   40       mapForest A B f  nil               = nil
   41       mapForest A B f (cons rose forest) = cons (mapTree A B f rose) (mapForest A B f forest)
   42 
   43 using this representation:
   44 
   45     {-# OPTIONS_GHC --type-in-type #-}
   46 
   47     {-# NO_POSITIVITY_CHECK #-}
   48     data IFix {A : Set} (F : (A -> Set) -> A -> Set) (x : A) : Set where
   49       wrap : F (IFix F) x -> IFix F x
   50 
   51     Fix₂ : ∀ {A B} -> ((A -> B -> Set) -> A -> B -> Set) -> A -> B -> Set
   52     Fix₂ F x y = IFix (λ B P -> P (F λ x y -> B λ R -> R x y)) (λ R -> R x y)
   53 
   54     treeTag : Set -> Set -> Set
   55     treeTag T F = T
   56 
   57     forestTag : Set -> Set -> Set
   58     forestTag T F = F
   59 
   60     AsTree : (Set -> (Set -> Set -> Set) -> Set) -> Set -> Set
   61     AsTree D A = D A treeTag
   62 
   63     AsForest : (Set -> (Set -> Set -> Set) -> Set) -> Set -> Set
   64     AsForest D A = D A forestTag
   65 
   66     TreeForest : Set -> (Set -> Set -> Set) -> Set
   67     TreeForest =
   68       Fix₂ λ Rec A tag ->
   69       ∀ R ->
   70         tag ((A -> AsForest Rec A -> R) -> R) (R -> (AsTree Rec A -> AsForest Rec A -> R) -> R)
   71 
   72     Tree : Set -> Set
   73     Tree = AsTree TreeForest
   74 
   75     Forest : Set -> Set
   76     Forest = AsForest TreeForest
   77 -}

   94 treeTag = runQuote $ do
   95   t <- freshTyName "t"
   96   f <- freshTyName "f"

  111 asTree :: Type TyName uni ()
  112 asTree = runQuote $ do
  113   d <- freshTyName "d"

  117     . TyLam () a star

  137 treeForestData :: RecursiveType uni fun ()
  138 treeForestData = runQuote $ do
  139   treeForest <- freshTyName "treeForest"
  140   a <- freshTyName "a"
  141   r <- freshTyName "r"
  142   tag <- freshTyName "tag"
  143   let vA = TyVar () a
  144       vR = TyVar () r
  145       recSpine = [TyVar () treeForest, vA]
  146   let tree = mkIterTyAppNoAnn asTree recSpine
  147       forest = mkIterTyAppNoAnn asForest recSpine
  148       body =
  149         TyForall () r (Type ()) $
  150           mkIterTyAppNoAnn
  151             (TyVar () tag)
  152             [ (vA ~~> forest ~~> vR) ~~> vR
  153             , vR ~~> (tree ~~> forest ~~> vR) ~~> vR
  154             ]
  155   makeRecursiveType
  156     ()
  157     treeForest
  158     [TyVarDecl () a star, TyVarDecl () tag $ star ~~> star ~~> star]
  159     body

  161 treeData :: RecursiveType uni fun ()
  162 treeData = runQuote $ do
  163   let RecursiveType treeForest wrapTreeForest = treeForestData
  164       tree = TyApp () asTree treeForest
  165   return $ RecursiveType tree (\[a] -> wrapTreeForest [a, treeTag])
  166 
  167 forestData :: RecursiveType uni fun ()
  168 forestData = runQuote $ do
  169   let RecursiveType treeForest wrapTreeForest = treeForestData
  170       forest = TyApp () asForest treeForest
  171   return $ RecursiveType forest (\[a] -> wrapTreeForest [a, forestTag])

  173 {-|
  174 
  175 > /\(a :: *) -> \(x : a) (fr : forest a) ->
  176 >     wrapTree [a] /\(r :: *) -> \(f : a -> forest a -> r) -> f x fr -}
  177 treeNode :: HasUniApply uni => Term TyName Name uni fun ()
  178 treeNode =
  179   runQuote $
  180     normalizeTypesIn =<< do
  181       let RecursiveType _ wrapTree = treeData
  182           RecursiveType forest _ = forestData
  183       a <- freshTyName "a"
  184       r <- freshTyName "r"
  185       x <- freshName "x"
  186       fr <- freshName "fr"
  187       f <- freshName "f"
  188       let vA = TyVar () a
  189           vR = TyVar () r
  190       Normalized forestA <- normalizeType $ TyApp () forest vA
  191       return
  192         . TyAbs () a (Type ())
  193         . LamAbs () x vA
  194         . LamAbs () fr forestA
  195         . wrapTree [vA]
  196         . TyAbs () r (Type ())
  197         . LamAbs () f (mkIterTyFun () [vA, forestA] vR)
  198         $ mkIterAppNoAnn
  199           (Var () f)
  200           [ Var () x
  201           , Var () fr
  202           ]
