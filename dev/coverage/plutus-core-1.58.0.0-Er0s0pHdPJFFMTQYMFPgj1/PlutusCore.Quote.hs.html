   48 -- | The state contains the "next" 'Unique' that should be used for a name

   54 {-| The "quotation" monad transformer. Within this monad you can do safe construction of PLC
   55 terms using quasiquotation, fresh-name generation, and parsing. -}
   56 newtype QuoteT m a = QuoteT {unQuoteT :: StateT FreshState m a}
   57   deriving newtype
   58     ( Functor
   59     , Applicative
   60     , Monad

   63     , MM.MFunctor
   64     , MonadError e
   65     , MonadReader r
   66     , MonadIO
   67     , MonadWriter w

   71 instance MonadState s m => MonadState s (QuoteT m) where
   72   get = lift get

   76 -- | A monad that allows lifting of quoted expressions.
   77 class Monad m => MonadQuote m where
   78   liftQuote :: Quote a -> m a
   79   -- This means we don't have to implement it when we're writing an instance for a MonadTrans
   80   -- monad. We can't just add an instance declaration for that, because it overlaps with the
   81   -- base instance.
   82   default liftQuote :: (MonadQuote n, MonadTrans t, t n ~ m) => Quote a -> m a
   83   liftQuote = lift . liftQuote

   95 {-| Run a quote from an empty identifier state. Note that the resulting term cannot necessarily
   96 be safely combined with other terms - that should happen inside 'QuoteT'. -}
   97 runQuoteT :: Monad m => QuoteT m a -> m a
   98 runQuoteT q = evalStateT (unQuoteT q) emptyFreshState

  100 -- | A non-transformer version of 'QuoteT'.
  101 type Quote = QuoteT Identity

  103 -- | See 'runQuoteT'.
  104 runQuote :: Quote a -> a
  105 runQuote = runIdentity . runQuoteT

  107 -- | Get a fresh 'Unique'.
  108 freshUnique :: MonadQuote m => m Unique
  109 freshUnique = liftQuote $ do
  110   nextU <- QuoteT get
  111   QuoteT $ put $ Unique (unUnique nextU + 1)
  112   pure nextU
  113 
  114 -- | Get a fresh 'Name', given the annotation and the 'Text.Text' name.
  115 freshName :: MonadQuote m => Text.Text -> m Name
  116 freshName str = Name str <$> freshUnique
  117 
  118 -- | Make a copy of the given 'Name' that is distinct from the old one.
  119 freshenName :: MonadQuote m => Name -> m Name
  120 freshenName (Name str _) = Name str <$> freshUnique
  121 
  122 -- | Get a fresh 'TyName', given the annotation and the 'Text.Text' name.
  123 freshTyName :: MonadQuote m => Text.Text -> m TyName
  124 freshTyName = fmap TyName <$> freshName
  125 
  126 -- | Make a copy of the given 'TyName' that is distinct from the old one.
  127 freshenTyName :: MonadQuote m => TyName -> m TyName
  128 freshenTyName (TyName name) = TyName <$> freshenName name

  130 {-| Mark all 'Unique's less than the given 'Unique' as used, so they will not be generated
  131 in future. -}
  132 markNonFreshBelow :: MonadQuote m => Unique -> m ()
  133 markNonFreshBelow = liftQuote . QuoteT . modify . max
  134 
  135 {-| Mark a given 'Unique' (and implicitly all 'Unique's less than it) as used, so they will
  136 not be generated in future. -}
  137 markNonFresh :: MonadQuote m => Unique -> m ()
  138 markNonFresh = markNonFreshBelow . succ
  139 
  140 {-| Mark the maximal 'Unique' from a set of 'Unique's (and implicitly all 'Unique's less than it)
  141 as used, so they will not be generated in future. -}
  142 markNonFreshMax :: MonadQuote m => Set Unique -> m ()
  143 markNonFreshMax = markNonFresh . fromMaybe (Unique 0) . Set.lookupMax
