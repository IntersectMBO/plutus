   28 {-| This type represents the result of computing a pairing using the Miller
   29    loop.  Values of this type are ephemeral, only created during script
   30    execution.  We do not provide any means of serialising, deserialising,
   31    printing, or parsing MlResult values. -}
   32 newtype MlResult = MlResult {unMlResult :: BlstBindings.PT}
   33   deriving newtype (Eq)
   34 
   35 instance Show MlResult where
   36   show _ = "<opaque>"
   37 instance Pretty MlResult where
   38   pretty = pretty . show
   39 instance PrettyBy ConstConfig MlResult where
   40   prettyBy _ = pretty
   41 
   42 -- We need a Flat instance to get everything to build properly; however we'll
   43 -- never want MlResult values in serialised scripts, so the decoding and
   44 -- encoding functions just raise errors.
   45 instance Flat MlResult where
   46   -- This might happen on the chain, so `fail` rather than `error`.
   47   decode = fail "Flat decoding is not supported for objects of type bls12_381_mlresult"
   48 
   49   -- This will be a Haskell runtime error, but encoding doesn't happen on chain,
   50   -- so it's not too bad.
   51   encode = error "Flat encoding is not supported for objects of type bls12_381_mlresult"
   52   size _ = id
   53 instance NFData MlResult where
   54   rnf _ = ()
   55 
   56 instance Hashable MlResult where
   57   hashWithSalt salt _MlResult = salt
   58 
   59 millerLoop :: G1.Element -> G2.Element -> MlResult
   60 millerLoop = coerce BlstBindings.millerLoop
   61 
   62 mulMlResult :: MlResult -> MlResult -> MlResult
   63 mulMlResult = coerce BlstBindings.ptMult
   64 
   65 finalVerify :: MlResult -> MlResult -> Bool
   66 finalVerify = coerce BlstBindings.ptFinalVerify
   67 
   68 -- Not exposed as builtins
   69 
   70 -- | Memory usage of an MlResult point (576 bytes)
   71 mlResultMemSizeBytes :: Int
   72 mlResultMemSizeBytes = BlstBindings.Internal.sizePT
   73 
   74 {-| For some of the tests we need a small element of the MlResult type.  We can
   75 get the identity element by pairing the zero elements of G1 and G2. -}
   76 identityMlResult :: MlResult
   77 identityMlResult = millerLoop G1.offchain_zero G2.offchain_zero
