   37 -- Simple Structural Equality of a `Term NamedDeBruijn`. This implies three things:
   38 -- b) We do not do equality "modulo starting index". E.g. `LamAbs 0 (Var 0) /= LamAbs 1 (Var 1)`.
   39 -- c) We do not do equality ""modulo annotations".

   41 -- If a user wants to ignore annotations he must prior do `void <$> term`, to throw away any annotations.

   68 type EqRenameOf ren a = HasUniques a => a -> a -> EqRename ren

   70 {- Note [No catch-all]
   71 Catch-all clauses like @f _x = <...>@ have the disadvantage that if somebody adds a new constructor
   72 to the definition of the type of @_x@, then GHC will not warn us about @f@ potentially no longer
   73 being defined correctly. This is especially pronounced for equality checking functions, having
   74 
   75     _ == _ = False
   76 
   77 as the last clause of '(==)' makes it much trickier to spot that a new clause needs to be added if
   78 the type of the arguments get extended with a new constructor. For this reason in equality checking
   79 functions we always match on the first argument exhaustively even if it means having a wall of
   80 
   81     C1 == _ = False
   82     C2 == _ = False
   83     <...>
   84     Cn == _ = False
   85 
   86 This way we get a warning from GHC about non-exhaustive pattern matching if the type of the
   87 arguments gets extended with additional constructors.
   88 -}

   94 -- | Check equality of two 'Type's.
   95 eqTypeM :: (HasRenaming ren TypeUnique, GEq uni, Eq ann) => EqRenameOf ren (Type tyname uni ann)
   96 eqTypeM (TyVar ann1 name1) (TyVar ann2 name2) = do
   97   eqM ann1 ann2
   98   eqNameM name1 name2
   99 eqTypeM (TyLam ann1 name1 kind1 ty1) (TyLam ann2 name2 kind2 ty2) = do
  100   eqM ann1 ann2
  101   eqM kind1 kind2
  102   withTwinBindings name1 name2 $ eqTypeM ty1 ty2
  103 eqTypeM (TyForall ann1 name1 kind1 ty1) (TyForall ann2 name2 kind2 ty2) = do
  104   eqM ann1 ann2
  105   eqM kind1 kind2
  106   withTwinBindings name1 name2 $ eqTypeM ty1 ty2
  107 eqTypeM (TyIFix ann1 pat1 arg1) (TyIFix ann2 pat2 arg2) = do
  108   eqM ann1 ann2
  109   eqTypeM pat1 pat2
  110   eqTypeM arg1 arg2
  111 eqTypeM (TyApp ann1 fun1 arg1) (TyApp ann2 fun2 arg2) = do
  112   eqM ann1 ann2
  113   eqTypeM fun1 fun2
  114   eqTypeM arg1 arg2
  115 eqTypeM (TyFun ann1 dom1 cod1) (TyFun ann2 dom2 cod2) = do
  116   eqM ann1 ann2
  117   eqTypeM dom1 dom2
  118   eqTypeM cod1 cod2
  119 eqTypeM (TyBuiltin ann1 someUni1) (TyBuiltin ann2 someUni2) = do
  120   eqM ann1 ann2
  121   eqM someUni1 someUni2
  122 eqTypeM (TySOP ann1 tyls1) (TySOP ann2 tyls2) = do
  123   eqM ann1 ann2
  124   case zipExact tyls1 tyls2 of
  125     Just ps -> for_ ps $ \(ptys1, ptys2) -> case zipExact ptys1 ptys2 of
  126       Just tys -> for_ tys $ \(ty1, ty2) -> eqTypeM ty1 ty2
  127       Nothing -> empty
  128     Nothing -> empty
  129 eqTypeM TyVar {} _ = empty
  130 eqTypeM TyLam {} _ = empty
  131 eqTypeM TyForall {} _ = empty
  132 eqTypeM TyIFix {} _ = empty
  133 eqTypeM TyApp {} _ = empty
  134 eqTypeM TyFun {} _ = empty
  135 eqTypeM TyBuiltin {} _ = empty
  136 eqTypeM TySOP {} _ = empty

  142 -- | Check equality of two 'Term's.
  143 eqTermM
  144   :: (GEq uni, Closed uni, uni `Everywhere` Eq, Eq fun, Eq ann)
  145   => EqRenameOf ScopedRenaming (Term tyname name uni fun ann)
  146 eqTermM (LamAbs ann1 name1 ty1 body1) (LamAbs ann2 name2 ty2 body2) = do
  147   eqM ann1 ann2
  148   eqTypeM ty1 ty2
  149   withTwinBindings name1 name2 $ eqTermM body1 body2
  150 eqTermM (TyAbs ann1 name1 kind1 body1) (TyAbs ann2 name2 kind2 body2) = do
  151   eqM ann1 ann2
  152   eqM kind1 kind2
  153   withTwinBindings name1 name2 $ eqTermM body1 body2
  154 eqTermM (IWrap ann1 pat1 arg1 term1) (IWrap ann2 pat2 arg2 term2) = do
  155   eqM ann1 ann2
  156   eqTypeM pat1 pat2
  157   eqTypeM arg1 arg2
  158   eqTermM term1 term2
  159 eqTermM (Apply ann1 fun1 arg1) (Apply ann2 fun2 arg2) = do
  160   eqM ann1 ann2
  161   eqTermM fun1 fun2
  162   eqTermM arg1 arg2
  163 eqTermM (Unwrap ann1 term1) (Unwrap ann2 term2) = do
  164   eqM ann1 ann2
  165   eqTermM term1 term2
  166 eqTermM (Error ann1 ty1) (Error ann2 ty2) = do
  167   eqM ann1 ann2
  168   eqTypeM ty1 ty2
  169 eqTermM (TyInst ann1 term1 ty1) (TyInst ann2 term2 ty2) = do
  170   eqM ann1 ann2
  171   eqTermM term1 term2
  172   eqTypeM ty1 ty2
  173 eqTermM (Var ann1 name1) (Var ann2 name2) = do
  174   eqM ann1 ann2
  175   eqNameM name1 name2
  176 eqTermM (Constant ann1 con1) (Constant ann2 con2) = do
  177   eqM ann1 ann2
  178   eqM con1 con2
  179 eqTermM (Builtin ann1 bi1) (Builtin ann2 bi2) = do
  180   eqM ann1 ann2
  181   eqM bi1 bi2
  182 eqTermM (Constr ann1 ty1 i1 args1) (Constr ann2 ty2 i2 args2) = do
  183   eqM ann1 ann2
  184   eqTypeM ty1 ty2
  185   eqM i1 i2
  186   case zipExact args1 args2 of
  187     Just ps -> for_ ps $ \(t1, t2) -> eqTermM t1 t2
  188     Nothing -> empty
  189 eqTermM (Case ann1 ty1 a1 cs1) (Case ann2 ty2 a2 cs2) = do
  190   eqM ann1 ann2
  191   eqTypeM ty1 ty2
  192   eqTermM a1 a2
  193   case zipExact cs1 cs2 of
  194     Just ps -> for_ ps $ \(t1, t2) -> eqTermM t1 t2
  195     Nothing -> empty
  196 eqTermM LamAbs {} _ = empty
  197 eqTermM TyAbs {} _ = empty
  198 eqTermM IWrap {} _ = empty
  199 eqTermM Apply {} _ = empty
  200 eqTermM Unwrap {} _ = empty
  201 eqTermM Error {} _ = empty
  202 eqTermM TyInst {} _ = empty
  203 eqTermM Var {} _ = empty
  204 eqTermM Constant {} _ = empty
  205 eqTermM Builtin {} _ = empty
  206 eqTermM Constr {} _ = empty
  207 eqTermM Case {} _ = empty
