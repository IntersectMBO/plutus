    5 -- | Built-in @pair@ and related functions.
    6 module PlutusCore.StdLib.Data.Pair

    8   , fstPair

   10   , uncurry
   11   ) where
   12 
   13 import Prelude hiding (fst, snd, uncurry)
   14

   21 -- | @(,)@ as a built-in PLC type.
   22 pair :: uni `HasTypeLevel` (,) => Type tyname uni ()
   23 pair = mkTyBuiltin @_ @(,) ()
   24 
   25 {-| @fst@ as a PLC term.
   26 
   27 > /\(a :: *) (b :: *) -> \(p : pair a b) -> fst {a} {b} p -}
   28 fstPair :: TermLike term tyname name DefaultUni DefaultFun => term ()
   29 fstPair = builtin () FstPair
   30 
   31 {-| @snd@ as a PLC term.
   32 
   33 > /\(a :: *) (b :: *) -> \(p : pair a b) -> snd {a} {b} p -}
   34 sndPair :: TermLike term tyname name DefaultUni DefaultFun => term ()
   35 sndPair = builtin () SndPair
   36 
   37 {-| @uncurry@ as a PLC term.
   38 
   39 > /\(a :: *) (b :: *) (c :: *) -> \(f : a -> b -> c) (p : pair a b) ->
   40 >     f (fst {a} {b} p) (snd {a} {b} p) -}
   41 uncurry :: TermLike term TyName Name DefaultUni DefaultFun => term ()
   42 uncurry = runQuote $ do
   43   a <- freshTyName "a"
   44   b <- freshTyName "b"
   45   c <- freshTyName "c"
   46   f <- freshName "f"
   47   p <- freshName "p"
   48   return
   49     . tyAbs () a (Type ())
   50     . tyAbs () b (Type ())
   51     . tyAbs () c (Type ())
   52     . lamAbs () f (TyFun () (TyVar () a) . TyFun () (TyVar () b) $ TyVar () c)
   53     . lamAbs () p (mkIterTyAppNoAnn pair [TyVar () a, TyVar () b])
   54     $ mkIterAppNoAnn
   55       (var () f)
   56       [ apply () (mkIterInstNoAnn fstPair [TyVar () a, TyVar () b]) $ var () p
   57       , apply () (mkIterInstNoAnn sndPair [TyVar () a, TyVar () b]) $ var () p
   58       ]
