    3 {-| The Float Delay optimization floats `Delay` from arguments into function bodies,
    4 if possible. It turns @(\n -> ...Force n...Force n...) (Delay arg)@ into
    5 @(\n -> ...Force (Delay n)...Force (Delay n)...) arg@.
    6 
    7 The above transformation is performed if:
    8 
    9     * All occurrences of @arg@ are under @Force@.
   10 
   11     * @arg@ is essentially work-free.
   12 
   13 This achieves a similar effect to Plutonomy's "Split Delay" optimization. The difference
   14 is that Split Delay simply splits the @Delay@ argument into multiple arguments, turning the
   15 above example into @(\m -> (\n -> ...Force n...Force n...) (Delay m)) arg@, and then relies
   16 on other optimizations to simplify it further. Specifically, once the inliner inlines
   17 @Delay m@, it will be identical to the result of Float Delay.
   18 
   19 The advantages of Float Delay are:
   20 
   21     * It doesn't rely on the inliner. In this example, Split Delay relies on the inliner to
   22       inline @Delay m@, but there's no guarantee that the inliner will do so, because inlining
   23       it may increase the program size.
   24 
   25       We can potentially modify the inliner such that it is aware of Float Delay and
   26       Force-Delay Cancel, and makes inlining decisions with these other optimizations in mind.
   27       The problem is that, not only does this makes the inlining heuristics much more
   28       complex, but it could easily lead to code duplication. Other optimizations often
   29       need to do some calculation in order to make certain optimization decisions (e.g., in
   30       this case, we want to check whether all occurrences of @arg@ are under @Force@), and
   31       if we rely on the inliner to inline the @Delay@, then the same check would need to be
   32       performed by the inliner.
   33 
   34     * Because Force Delay requires that all occurrences of @arg@ are under @Force@, it
   35       guarantees to not increase the size or the cost of the program. This is not the case
   36       with Split Delay: in this example, if the occurrences of @n@ are not under @Force@,
   37       then Split Delay may increase the size of the program, regardless of whether or not
   38       @Delay m@ is inlined. If @Delay m@ is not inlined, then it will also increase the
   39       cost of the program, due to the additional application.
   40 
   41 The alternative approach that always floats the @Delay@ regardless of whether or not all
   42 occurences of @arg@ are under @Force@ was implemented and tested, and it is strictly worse than
   43 Float Delay on our current test suite (specifically, Split Delay causes one test case
   44 to have a slightly bigger program, and everything else is equal).
   45 
   46 Why is this optimization performed on UPLC, not PIR?
   47 
   48     1. Not only are the types and let-bindings in PIR not useful for this optimization,
   49        they can also get in the way. For example, we cannot transform
   50        @let f = /\a. ...a... in ...{f t1}...{f t2}...@ into
   51        @ket f = ...a... in ...f...f...@.
   52 
   53     2. This optimization mainly interacts with ForceDelayCancel and the inliner, and
   54        both are part of the UPLC simplifier. -}
   55 module UntypedPlutusCore.Transform.FloatDelay (floatDelay) where

   69 import Control.Lens (forOf, forOf_, transformOf)
   70 import Control.Monad.Trans.Writer.CPS (Writer, execWriter, runWriter, tell)
   71 
   72 floatDelay
   73   :: ( PLC.MonadQuote m
   74      , PLC.Rename (Term name uni fun a)
   75      , PLC.HasUnique name PLC.TermUnique
   76      )
   77   => Term name uni fun a
   78   -> SimplifierT name uni fun a m (Term name uni fun a)
   79 floatDelay term = do
   80   result <-
   81     PLC.rename term >>= \t ->
   82       pure . uncurry (flip simplifyBodies) $ simplifyArgs (unforcedVars t) t
   83   recordSimplification term FloatDelay result
   84   return result
   85 
   86 {-| First pass. Returns the names of all variables, at least one occurrence
   87 of which is not under `Force`. -}
   88 unforcedVars
   89   :: forall name uni fun a
   90    . PLC.HasUnique name PLC.TermUnique
   91   => Term name uni fun a
   92   -> PLC.UniqueSet PLC.TermUnique
   93 unforcedVars = execWriter . go
   94   where
   95     go :: Term name uni fun a -> Writer (PLC.UniqueSet PLC.TermUnique) ()
   96     go = \case
   97       Var _ n -> tell (USet.singletonName n)
   98       Force _ Var {} -> pure ()
   99       t -> forOf_ termSubterms t go
  100 
  101 {-| Second pass. Removes `Delay` from eligible arguments, and returns
  102 the names of variables whose corresponding arguments are modified. -}
  103 simplifyArgs
  104   :: forall name uni fun a
  105    . PLC.HasUnique name PLC.TermUnique
  106   => PLC.UniqueSet PLC.TermUnique
  107   -- ^ The set of variables returned by `unforcedVars`.
  108   -> Term name uni fun a
  109   -> (Term name uni fun a, PLC.UniqueMap PLC.TermUnique a)
  110 simplifyArgs blacklist = runWriter . go
  111   where
  112     go :: Term name uni fun ann -> Writer (PLC.UniqueMap PLC.TermUnique ann) (Term name uni fun ann)
  113     go = \case
  114       Apply appAnn (LamAbs lamAnn n lamBody) (Delay delayAnn arg)
  115         | isEssentiallyWorkFree arg
  116         , n `USet.notMemberByName` blacklist -> do
  117             tell (UMap.singletonByName n delayAnn)
  118             (Apply appAnn . LamAbs lamAnn n <$> go lamBody) <*> go arg
  119       t -> forOf termSubterms t go
  120 
  121 -- | Third pass. Turns @Force n@ into @Force (Delay n)@ for all eligibile @n@.
  122 simplifyBodies
  123   :: PLC.HasUnique name PLC.TermUnique
  124   => PLC.UniqueMap PLC.TermUnique a
  125   -> Term name uni fun a
  126   -> Term name uni fun a
  127 simplifyBodies whitelist = transformOf termSubterms $ \case
  128   var@(Var _ n)
  129     | Just ann <- UMap.lookupName n whitelist -> Delay ann var
  130   t -> t

  132 {-| Whether evaluating the given `Term` is pure and essentially work-free
  133 (barring the CEK machine overhead). -}
  134 
  135 --- This should be the erased version of 'PlutusIR.Transform.LetFloat.isEssentiallyWorkFree'.
  136 isEssentiallyWorkFree :: Term name uni fun a -> Bool
  137 isEssentiallyWorkFree = \case
  138   LamAbs {} -> True
  139   Constant {} -> True
  140   Delay {} -> True
  141   Constr {} -> True
  142   Builtin {} -> True
  143   Var {} -> False
  144   Force {} -> False
  145   -- Unsaturated builtin applications should also be essentially work-free,
  146   -- but this is currently not implemented for UPLC.
  147   -- `UntypedPlutusCore.Transform.Inline.isPure` has the same problem.
  148   Apply {} -> False
  149   Case {} -> False
  150   Error {} -> False
