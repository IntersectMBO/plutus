    8 -- | A variant of 'ix' that takes a 'Word64' instead of an 'Int'.
    9 wix :: Word64 -> Traversal' [a] a
   10 wix k f xs0 = go xs0 k
   11   where
   12     go [] _ = pure []
   13     go (a : as) 0 = f a <&> (: as)
   14     go (a : as) i = (a :) <$> (go as $! i - 1)
