   24 {- Note [Marking]
   25 We use functions from the @markNonFresh*@ family in order to ensure that bound variables never get
   26 renamed to free ones. This means types/terms/etc are processed twice: once by a @markNonFresh*@
   27 function and once for the actual renaming. We may consider later to do some kind of meta-circular
   28 programming trick in order to perform renaming in a single pass.
   29 -}
   30 
   31 {-| The class of things that can be renamed.
   32 I.e. things that are capable of satisfying the global uniqueness condition. -}
   33 class Rename a where
   34   {-| Rename 'Unique's so that they're globally unique.
   35   In case there are any free variables, they must be left untouched and bound variables
   36   must not get renamed to free ones.
   37   Must always assign new names to bound variables,
   38   so that @rename@ can be used for alpha-renaming as well. -}
   39   rename :: MonadQuote m => a -> m a
   40 
   41 {-| 'rename' a value and wrap the result in 'Renamed', so that it can be passed around and it's
   42 visible in the types that the thing inside satisfies global uniqueness. -}
   43 getRenamed :: (Rename a, MonadQuote m) => a -> m (Renamed a)
   44 getRenamed = fmap Renamed . rename
   45 
   46 -- | Wrap a value in 'Dupable'.
   47 dupable :: a -> Dupable a
   48 dupable = Dupable
   49 
   50 -- | Extract the value stored in a @Dupable a@ and rename it.
   51 liftDupable :: (MonadQuote m, Rename a) => Dupable a -> m a
   52 liftDupable = liftQuote . rename . unDupable

   54 instance HasUniques (Type tyname uni ann) => Rename (Type tyname uni ann) where
   55   -- See Note [Marking].
   56   rename = through markNonFreshType >=> runRenameT @TypeRenaming . renameTypeM
   57 
   58 instance HasUniques (Term tyname name uni fun ann) => Rename (Term tyname name uni fun ann) where
   59   -- See Note [Marking].
   60   rename = through markNonFreshTerm >=> runRenameT . renameTermM
   61 
   62 instance
   63   HasUniques (Program tyname name uni fun ann)
   64   => Rename (Program tyname name uni fun ann)
   65   where
   66   -- See Note [Marking].
   67   rename = through markNonFreshProgram >=> runRenameT . renameProgramM
   68 
   69 instance Rename a => Rename (Normalized a) where
   70   rename = traverse rename
