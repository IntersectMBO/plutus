   29 {-| A PLC @Type@ to be parsed. ATM the parser only works
   30 for types in the @DefaultUni@ with @DefaultFun@. -}
   31 type PType = Type TyName DefaultUni SrcSpan
   32 
   33 varType :: Parser PType
   34 varType = withSpan $ \sp ->
   35   TyVar sp <$> tyName
   36 
   37 funType :: Parser PType
   38 funType = withSpan $ \sp ->
   39   inParens $ TyFun sp <$> (symbol "fun" *> pType) <*> pType
   40 
   41 allType :: Parser PType
   42 allType = withSpan $ \sp ->
   43   inParens $ TyForall sp <$> (symbol "all" *> trailingWhitespace tyName) <*> kind <*> pType
   44 
   45 lamType :: Parser PType
   46 lamType = withSpan $ \sp ->
   47   inParens $ TyLam sp <$> (symbol "lam" *> trailingWhitespace tyName) <*> kind <*> pType
   48 
   49 ifixType :: Parser PType
   50 ifixType = withSpan $ \sp ->
   51   inParens $ TyIFix sp <$> (symbol "ifix" *> pType) <*> pType
   52 
   53 builtinType :: Parser PType
   54 builtinType = withSpan $ \sp -> inParens $ do
   55   SomeTypeIn (Kinded uni) <- symbol "con" *> defaultUni
   56   pure $ TyBuiltin sp (SomeTypeIn uni)
   57 
   58 sopType :: Parser PType
   59 sopType = withSpan $ \sp -> inParens $ TySOP sp <$> (symbol "sop" *> many tyList)
   60   where
   61     tyList :: Parser [PType]
   62     tyList = (inBrackets $ many pType) <* whitespace
   63 
   64 appType :: Parser PType
   65 appType = withSpan $ \sp -> inBrackets $ do
   66   fn <- pType
   67   args <- some pType
   68   -- TODO: should not use the same `sp` for all arguments.
   69   pure $ mkIterTyApp fn ((sp,) <$> args)
   70 
   71 kind :: Parser (Kind SrcSpan)
   72 kind = withSpan $ \sp ->
   73   let typeKind = Type sp <$ symbol "type"
   74       funKind = KindArrow sp <$> (symbol "fun" *> kind) <*> kind
   75    in inParens (typeKind <|> funKind)
   76 
   77 -- | Parser for @PType@.
   78 pType :: Parser PType
   79 pType =
   80   choice $
   81     map
   82       try
   83       [ funType
   84       , ifixType
   85       , allType
   86       , builtinType
   87       , lamType
   88       , appType
   89       , varType
   90       , sopType
   91       ]
   92 
   93 {-| Parser for built-in type applications.  The textual names here should match
   94 the ones in the PrettyBy instance for DefaultUni in PlutusCore.Default.Universe. -}
   95 defaultUniApplication :: Parser (SomeTypeIn (Kinded DefaultUni))
   96 defaultUniApplication = do
   97   -- Parse the head of the application.
   98   f <- defaultUni
   99   -- Parse the arguments.
  100   as <- many defaultUni
  101   -- Iteratively apply the head to the arguments checking that the kinds match and
  102   -- failing otherwise.
  103   foldM tryUniApply f as
  104 
  105 {-| Parser for built-in types (the ones from 'DefaultUni' specifically).
  106 
  107 'Kinded' is needed for checking that a type function can be applied to its argument.
  108 I.e. we do Plutus kind checking of builtin type applications during parsing, which is
  109 unfortunate, but there's no way we could construct a 'DefaultUni' otherwise.
  110 
  111 In case of kind error no sensible message is shown, only an overly general one:
  112 
  113 >>> :set -XTypeApplications
  114 >>> :set -XOverloadedStrings
  115 >>> import PlutusCore.Error
  116 >>> import PlutusCore.Quote
  117 >>> let runP = putStrLn . either display display . runQuoteT . parseGen @ParserErrorBundle defaultUni
  118 >>> runP "(list integer)"
  119 (list integer)
  120 >>> runP "(bool integer)"
  121 test:1:14:
  122   |
  123 1 | (bool integer)
  124   |              ^
  125 expecting "bool", "bytestring", "data", "integer", "list", "pair", "string", "unit", or '('
  126 
  127 This is to be fixed.
  128 
  129 One thing we could do to avoid doing kind checking during parsing is to parse into
  130 
  131     data TextualUni a where
  132         TextualUni :: TextualUni (Esc (Tree Text))
  133 
  134 i.e. parse into @Tree Text@ and do the kind checking afterwards, but given that we'll still need
  135 to do the kind checking of builtins regardless (even for UPLC), we don't win much by deferring
  136 doing it. -}
  137 defaultUni :: Parser (SomeTypeIn (Kinded DefaultUni))
  138 defaultUni =
  139   ( choice $
  140       map
  141         try
  142         [ trailingWhitespace (inParens defaultUniApplication)
  143         , someType @_ @Integer <$ symbol "integer"
  144         , someType @_ @ByteString <$ symbol "bytestring"
  145         , someType @_ @Text <$ symbol "string"
  146         , someType @_ @() <$ symbol "unit"
  147         , someType @_ @Bool <$ symbol "bool"
  148         , someType @_ @[] <$ symbol "list"
  149         , someType @_ @Strict.Vector <$ symbol "array"
  150         , someType @_ @(,) <$ symbol "pair"
  151         , someType @_ @Data <$ symbol "data"
  152         , someType @_ @BLS12_381.G1.Element <$ symbol "bls12_381_G1_element"
  153         , someType @_ @BLS12_381.G2.Element <$ symbol "bls12_381_G2_element"
  154         , someType @_ @BLS12_381.Pairing.MlResult <$ symbol "bls12_381_mlresult"
  155         , someType @_ @Value <$ symbol "value"
  156         ]
  157   )
  158     <?> "type name (integer, bytestring, string, unit, bool, list, array, pair,\
  159         \ data, value, bls12_381_G1_element, bls12_381_G2_element,\
  160         \ bls12_381_mlresult, or type application)"
  161 
  162 tyName :: Parser TyName
  163 tyName = TyName <$> name
