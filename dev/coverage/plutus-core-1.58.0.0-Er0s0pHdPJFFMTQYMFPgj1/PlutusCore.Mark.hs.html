   12 {-| Marks all the 'Unique's in a type as used, so they will not be generated in future. Useful if
   13 you have a type which was not generated in 'Quote'. -}
   14 markNonFreshType
   15   :: (HasUniques (Type tyname uni ann), MonadQuote m)
   16   => Type tyname uni ann -> m ()
   17 markNonFreshType = markNonFreshMax . setOf typeUniquesDeep

   19 {-| Marks all the 'Unique's in a term as used, so they will not be generated in future. Useful if

   21 markNonFreshTerm
   22   :: (HasUniques (Term tyname name uni fun ann), MonadQuote m)
