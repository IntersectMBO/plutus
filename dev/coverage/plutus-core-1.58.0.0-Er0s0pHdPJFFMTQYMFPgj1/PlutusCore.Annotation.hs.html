   37 -- | An annotation type used during the compilation.
   38 data Ann = Ann
   39   { annInline :: Inline
   40   , annSrcSpans :: SrcSpans
   41   , annIsAsDataMatcher :: Bool
   42   }
   43   deriving stock (Eq, Ord, Generic, Show)
   44   deriving anyclass (Hashable)
   45 
   46 instance Default Ann where
   47   def =
   48     Ann
   49       { annInline = MayInline
   50       , annSrcSpans = mempty
   51       , annIsAsDataMatcher = False
   52       }

   54 data Inline
   55   = {-| When calling @PlutusIR.Compiler.Definitions.defineTerm@ to add a new term definition,
   56     if we annotation the var on the LHS of the definition with `AlwaysInline`, the inliner will
   57     always inline that var.
   58 
   59     This is currently used to ensure builtin functions such as @trace@ (when the @remove-trace@
   60     flag is on and @trace@ is rewritten to @const@) are inlined, because the inliner would
   61     otherwise not inline them. To achieve that, we annotate the definition with `AlwaysInline`
   62     when defining @trace@, i.e., @trace <AlwaysInline> = \_ a -> a@. -}
   63     AlwaysInline
   64   | {-| Signaling to the compiler that a binding is safe to inline. This is useful for
   65     annotating strict bindings that aren't obviously safe to inline. -}
   66     SafeToInline
   67   | MayInline
   68   deriving stock (Eq, Ord, Generic, Show)
   69   deriving anyclass (Hashable)
   70 
   71 instance Pretty Ann where
   72   pretty = viaShow
   73 
   74 class AnnInline a where
   75   -- | An annotation instructing the inliner to always inline a binding.
   76   annAlwaysInline :: a
   77 
   78   {-| An annotation signaling to the inliner that a binding is safe to inline.
   79   The inlining decision is left to the inliner. This is useful for
   80   annotating strict bindings that aren't obviously safe to inline. -}
   81   annSafeToInline :: a
   82 
   83   -- | An annotation that leaves the inlining decision to the inliner.
   84   annMayInline :: a
   85 
   86 instance AnnInline () where
   87   annAlwaysInline = ()
   88   annSafeToInline = ()
   89   annMayInline = ()
   90 
   91 instance AnnInline Ann where
   92   annAlwaysInline = def {annInline = AlwaysInline}
   93   annSafeToInline = def {annInline = SafeToInline}
   94   annMayInline = def {annInline = MayInline}

   96 {-| The span between two source locations.
   97 
   98 This corresponds roughly to the `SrcSpan` used by GHC,
   99 but we define our own version so we don't have to depend on `ghc` to use it.
  100 
  101 The line and column numbers are 1-based, and the unit is Unicode code point (or `Char`). -}
  102 data SrcSpan = SrcSpan
  103   { srcSpanFile :: FilePath
  104   , srcSpanSLine :: Int
  105   , srcSpanSCol :: Int
  106   , srcSpanELine :: Int
  107   , srcSpanECol :: Int
  108   {-^ Same as GHC's @SrcSpan@, @srcSpanECol@ is usually one more than the column of
  109   the last character of the thing this @SrcSpan@ is for (unless the last character
  110   is the line break). -}
  111   }
  112   deriving stock (Eq, Ord, Generic)
  113   deriving anyclass (Flat, Hashable, NFData)
  114 
  115 instance Show SrcSpan where
  116   showsPrec _ s =
  117     showString (srcSpanFile s)
  118       . showChar ':'
  119       . showsPrec 0 (srcSpanSLine s)
  120       . showChar ':'
  121       . showsPrec 0 (srcSpanSCol s)
  122       . showChar '-'
  123       . showsPrec 0 (srcSpanELine s)
  124       . showChar ':'
  125       . showsPrec 0 (if srcSpanECol s == 0 then 0 else srcSpanECol s - 1)
  126 
  127 instance Pretty SrcSpan where
  128   pretty = viaShow
  129 
  130 newtype SrcSpans = SrcSpans {unSrcSpans :: Set SrcSpan}
  131   deriving newtype (Eq, Ord, Hashable, Semigroup, Monoid, MonoFoldable, NFData)
  132   deriving stock (Generic)
  133   deriving anyclass (Flat)
  134 
  135 type instance Element SrcSpans = SrcSpan
  136 
  137 instance Show SrcSpans where
  138   showsPrec _ (SrcSpans xs) =
  139     showString "{ "
  140       . showString
  141         ( case Set.toList xs of
  142             [] -> "no-src-span"
  143             ys -> List.intercalate ", " (show <$> ys)
  144         )
  145       . showString " }"
  146 
  147 instance Pretty SrcSpans where
  148   pretty = viaShow
  149 
  150 -- | Add an extra SrcSpan to existing 'SrcSpans' of 'Ann'
  151 addSrcSpan :: SrcSpan -> Ann -> Ann
  152 addSrcSpan s (Ann i (SrcSpans ss) b) = Ann i (SrcSpans $ Set.insert s ss) b
  153 
  154 -- | Tells if a line (positive integer) falls inside a SrcSpan.
  155 lineInSrcSpan :: Pos -> SrcSpan -> Bool
  156 lineInSrcSpan pos spn =
  157   let i = Megaparsec.unPos pos
  158    in i >= srcSpanSLine spn && i <= srcSpanELine spn
