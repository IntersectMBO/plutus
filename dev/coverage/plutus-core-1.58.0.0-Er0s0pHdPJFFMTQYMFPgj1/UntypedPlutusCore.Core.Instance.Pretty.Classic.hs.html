    8 -- | A "classic" (i.e. as seen in the specification) way to pretty-print Untyped Plutus Core terms.
    9 module UntypedPlutusCore.Core.Instance.Pretty.Classic () where
   10 
   11 import PlutusPrelude
   12 
   13 import UntypedPlutusCore.Core.Type
   14 
   15 import PlutusCore.Pretty.Classic
   16 import PlutusCore.Pretty.PrettyConst
   17 
   18 import Prettyprinter
   19 import Prettyprinter.Custom
   20 import Universe (Some (..), SomeTypeIn (SomeTypeIn), ValueOf (..))
   21 
   22 instance
   23   (PrettyClassicBy configName name, PrettyUni uni, Pretty fun, Pretty ann)
   24   => PrettyBy (PrettyConfigClassic configName) (Term name uni fun ann)
   25   where
   26   prettyBy config = \case
   27     Var ann n ->
   28       sep (consAnnIf config ann [prettyBy config n])
   29     LamAbs ann n t ->
   30       sexp
   31         "lam"
   32         ( consAnnIf
   33             config
   34             ann
   35             [prettyBy config n, prettyBy config t]
   36         )
   37     Apply ann t1 t2 ->
   38       brackets'
   39         ( sep
   40             ( consAnnIf
   41                 config
   42                 ann
   43                 [prettyBy config t1, prettyBy config t2]
   44             )
   45         )
   46     Constant ann c ->
   47       sexp "con" (consAnnIf config ann [prettyTypeOf c, pretty c])
   48     Builtin ann bi ->
   49       sexp
   50         "builtin"
   51         ( consAnnIf
   52             config
   53             ann
   54             [pretty bi]
   55         )
   56     Error ann ->
   57       sexp "error" (consAnnIf config ann [])
   58     Delay ann term ->
   59       sexp
   60         "delay"
   61         ( consAnnIf
   62             config
   63             ann
   64             [prettyBy config term]
   65         )
   66     Force ann term ->
   67       sexp
   68         "force"
   69         ( consAnnIf
   70             config
   71             ann
   72             [prettyBy config term]
   73         )
   74     Constr ann i es ->
   75       sexp "constr" (consAnnIf config ann (pretty i : fmap (prettyBy config) es))
   76     Case ann arg cs ->
   77       sexp
   78         "case"
   79         ( consAnnIf
   80             config
   81             ann
   82             (prettyBy config arg : fmap (prettyBy config) (toList cs))
   83         )
   84     where
   85       prettyTypeOf :: Some (ValueOf uni) -> Doc dann
   86       prettyTypeOf (Some (ValueOf uni _)) = prettyBy juxtRenderContext $ SomeTypeIn uni
   87 
   88 instance
   89   (PrettyClassicBy configName (Term name uni fun ann), Pretty ann)
   90   => PrettyBy (PrettyConfigClassic configName) (Program name uni fun ann)
   91   where
   92   prettyBy config (Program ann version term) =
   93     sexp "program" (consAnnIf config ann [pretty version, prettyBy config term])
