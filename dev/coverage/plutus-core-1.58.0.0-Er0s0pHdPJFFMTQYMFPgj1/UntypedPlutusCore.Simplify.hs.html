   36 simplifyProgram
   37   :: forall name uni fun m a
   38    . Compiling m uni fun name a
   39   => SimplifyOpts name a
   40   -> BuiltinSemanticsVariant fun
   41   -> Program name uni fun a
   42   -> m (Program name uni fun a)
   43 simplifyProgram opts builtinSemanticsVariant (Program a v t) =
   44   Program a v <$> simplifyTerm opts builtinSemanticsVariant t
   45 
   46 simplifyProgramWithTrace
   47   :: forall name uni fun m a
   48    . Compiling m uni fun name a
   49   => SimplifyOpts name a
   50   -> BuiltinSemanticsVariant fun
   51   -> Program name uni fun a
   52   -> m (Program name uni fun a, SimplifierTrace name uni fun a)
   53 simplifyProgramWithTrace opts builtinSemanticsVariant (Program a v t) = do
   54   (result, trace) <-
   55     runSimplifierT $
   56       termSimplifier opts builtinSemanticsVariant t
   57   pure (Program a v result, trace)
   58 
   59 simplifyTerm
   60   :: forall name uni fun m a
   61    . Compiling m uni fun name a
   62   => SimplifyOpts name a
   63   -> BuiltinSemanticsVariant fun
   64   -> Term name uni fun a
   65   -> m (Term name uni fun a)
   66 simplifyTerm opts builtinSemanticsVariant term =
   67   evalSimplifierT $ termSimplifier opts builtinSemanticsVariant term
   68 
   69 termSimplifier
   70   :: forall name uni fun m a
   71    . Compiling m uni fun name a
   72   => SimplifyOpts name a
   73   -> BuiltinSemanticsVariant fun
   74   -> Term name uni fun a
   75   -> SimplifierT name uni fun a m (Term name uni fun a)
   76 termSimplifier opts builtinSemanticsVariant =
   77   simplifyNTimes (_soMaxSimplifierIterations opts) >=> cseNTimes cseTimes
   78   where
   79     -- Run the simplifier @n@ times
   80     simplifyNTimes
   81       :: Int
   82       -> Term name uni fun a
   83       -> SimplifierT name uni fun a m (Term name uni fun a)
   84     simplifyNTimes n = List.foldl' (>=>) pure $ map simplifyStep [1 .. n]
   85 
   86     -- Run CSE @n@ times, interleaved with the simplifier.
   87     -- See Note [CSE]
   88     cseNTimes
   89       :: Int
   90       -> Term name uni fun a
   91       -> SimplifierT name uni fun a m (Term name uni fun a)
   92     cseNTimes n = foldl' (>=>) pure $ concatMap (\i -> [cseStep i, simplifyStep i]) [1 .. n]
   93 
   94     -- generate simplification step
   95     simplifyStep
   96       :: Int
   97       -> Term name uni fun a
   98       -> SimplifierT name uni fun a m (Term name uni fun a)
   99     simplifyStep _ =
  100       floatDelay
  101         >=> forceCaseDelay
  102         >=> case (eqT @uni @PLC.DefaultUni, eqT @fun @DefaultFun) of
  103           (Just Refl, Just Refl) -> forceDelay builtinSemanticsVariant
  104           _ -> pure
  105         >=> caseOfCase'
  106         >=> caseReduce
  107         >=> inline
  108           (_soInlineCallsiteGrowth opts)
  109           (_soInlineConstants opts)
  110           (_soPreserveLogging opts)
  111           (_soInlineHints opts)
  112           builtinSemanticsVariant
  113 
  114     caseOfCase'
  115       :: Term name uni fun a
  116       -> SimplifierT name uni fun a m (Term name uni fun a)
  117     caseOfCase' = case eqT @fun @DefaultFun of
  118       Just Refl -> caseOfCase
  119       Nothing -> pure
  120 
  121     cseStep
  122       :: Int
  123       -> Term name uni fun a
  124       -> SimplifierT name uni fun a m (Term name uni fun a)
  125     cseStep _ =
  126       case (eqT @name @Name, eqT @uni @PLC.DefaultUni) of
  127         (Just Refl, Just Refl) -> cse (_soCseWhichSubterms opts) builtinSemanticsVariant
  128         _ -> pure
  129 
  130     cseTimes = if _soConservativeOpts opts then 0 else _soMaxCseIterations opts
