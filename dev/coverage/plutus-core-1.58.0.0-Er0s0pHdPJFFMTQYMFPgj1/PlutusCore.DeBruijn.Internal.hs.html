   68 {- Note [Why newtype FakeNamedDeBruijn]
   69 We use a newtype wrapper to optimize away the expensive re-traversing of the deserialized Term
   70 for adding fake names everywhere --- the CEK works on names, but the scripts on ledger
   71 don't have names for size reduction.
   72 
   73 Specifically the expensive pipeline:
   74 
   75 ```
   76 decode @(Term DeBruijn)
   77 >>> term-Map-Names(FakeNamedDeBruijn)
   78 >>> cekExecute
   79 ```
   80 
   81 is optimized to to the faster:
   82 
   83 ```
   84 decode @(Term FakeNamedDeBruijn)
   85 >>> coerce @(Term FakeNamedDeBruijn) @(Term NamedDeBruijn)
   86 >>> cekExecute
   87 ```
   88 
   89 To achieve this we make sure:
   90 - to use `coerce` since its 0-cost
   91 - not to GeneralizeNewtypeDeriving the`Flat FakeNamedDeBruijn` instance, but "derive via"
   92 the optimized `Flat DeBruijn` instance. This is ok, because `FND<->D` are
   93 isomorphic.
   94 -}

  199 {- Note [Levels and indices]
  200 The indices ('Index') that we actually store as our de Bruijn indices in the program
  201 are *relative* - that is, they say how many levels above the *current* level to look for
  202 the binder.
  203 
  204 However, when doing conversions it is easier to record the  *absolute* level of a variable,
  205 in our state, since that way we don't have to adjust our mapping when we go under a binder
  206 (whereas for relative indices we would need to increment them all by one, as the current
  207 level has increased).
  208 
  209 However, this means that we *do* need to do an adjustment when we store an index as a level
  210 or extract a level to use it as an index. The adjustment is fairly straightforward:
  211 - An index `i` points to a binder `i` levels above (smaller than) the current level, so the level
  212   of `i` is `current - i`.
  213 - A level `l` which is `i` levels above (smaller than) the current level has an index of `i`, so it
  214   is also calculated as `current - l`.
  215 
  216 We use a newtype to keep these separate, since getting it wrong will lead to annoying bugs.
  217 -}

  219 -- | An absolute level in the program.
  220 newtype Level = Level Integer deriving newtype (Eq, Ord, Num, Real, Enum, Integral)

  222 {-| During visiting the AST we hold a reader "state" of current level and a current
  223 scoping (levelMapping).
  224 Invariant-A: the current level is positive and greater than all levels in the levelMapping.
  225 Invariant-B: only positive levels are stored in the levelMapping. -}
  226 data LevelInfo = LevelInfo
  227   { currentLevel :: Level
  228   , levelMapping :: BM.Bimap Unique Level
  229   }

  231 -- | Declare a name with a unique, recording the mapping to a 'Level'.
  232 declareUnique :: (MonadReader LevelInfo m, HasUnique name unique) => name -> m a -> m a
  233 declareUnique n =
  234   local $ \(LevelInfo current ls) -> LevelInfo current $ BM.insert (n ^. theUnique) current ls
  235 
  236 {-| Declares a new binder by assigning a fresh unique to the *current level*.
  237 Maintains invariant-B of 'LevelInfo' (that only positive levels are stored),
  238 since current level is always positive (invariant-A).
  239 See Note [DeBruijn indices of Binders] -}
  240 declareBinder :: (MonadReader LevelInfo m, MonadQuote m) => m a -> m a
  241 declareBinder act = do
  242   newU <- freshUnique
  243   local (\(LevelInfo current ls) -> LevelInfo current $ BM.insert newU current ls) act
  244 
  245 {-| Enter a scope, incrementing the current 'Level' by one
  246 Maintains invariant-A (that the current level is positive). -}
  247 withScope :: MonadReader LevelInfo m => m a -> m a
  248 withScope = local $ \(LevelInfo current ls) -> LevelInfo (current + 1) ls

  250 {-| We cannot do a correct translation to or from de Bruijn indices if the program is
  251 not well-scoped. So we throw an error in such a case. -}
  252 data FreeVariableError
  253   = FreeUnique !Unique
  254   | FreeIndex !Index
  255   deriving stock (Show, Eq, Ord, Generic)
  256   deriving anyclass (Exception, NFData)

  258 instance Pretty FreeVariableError where

  260   pretty (FreeIndex i) = "Free index:" <+> pretty i
  261 makeClassyPrisms ''FreeVariableError

  263 {-| Get the 'Index' corresponding to a given 'Unique'.
  264 Uses supplied handler for free names (uniques). -}
  265 getIndex :: MonadReader LevelInfo m => Unique -> (Unique -> m Index) -> m Index
  266 getIndex u h = do
  267   LevelInfo current ls <- ask
  268   case BM.lookup u ls of
  269     Just foundlvl -> pure $ levelToIx current foundlvl
  270     -- This call should return an index greater than the current level,
  271     -- otherwise it will map unbound variables to bound variables.
  272     Nothing -> h u
  273   where
  274     -- Compute the relative 'Index' of a absolute 'Level' relative to the current 'Level'.
  275     levelToIx :: Level -> Level -> Index
  276     levelToIx (Level current) (Level foundLvl) =
  277       -- Thanks to invariant-A, we can be sure that 'level >= foundLvl ', since foundLvl
  278       -- is in the levelMapping and thus the computation 'current-foundLvl' is '>=0' and
  279       -- its conversion to Natural will not lead to arithmetic underflow.
  280       fromIntegral $ current - foundLvl
  281 
  282 {-| Get the 'Unique' corresponding to a given 'Index'.
  283 Uses supplied handler for free debruijn indices. -}
  284 getUnique :: MonadReader LevelInfo m => Index -> (Index -> m Unique) -> m Unique
  285 getUnique ix h = do
  286   LevelInfo current ls <- ask
  287   case BM.lookupR (ixToLevel current ix) ls of
  288     -- Because of invariant-B, the levelMapping contains only positive (absolute) levels.
  289     Just u -> pure u
  290     -- This call should return a free/unbound unique,
  291     -- otherwise it will map unbound variables to bound variables.
  292     Nothing ->
  293       -- the lookup failed, meaning the index corresponds to a strictly-negative
  294       -- (absolute) level.
  295       h ix

  347 {-| A different implementation of a handler,  where "free" debruijn indices do not throw an error
  348 but are instead gracefully converted to fresh uniques.
  349 These generated uniques remain free; i.e.  if the original term was open, it will remain open
  350 after applying this handler.
  351 These generated free uniques are consistent across the open term (by using a state cache). -}
  352 freeIndexAsConsistentLevel
  353   :: (MonadReader LevelInfo m, MonadState (M.Map Level Unique) m, MonadQuote m)
  354   => Index
  355   -> m Unique
  356 freeIndexAsConsistentLevel ix = do
  357   cache <- get
  358   LevelInfo current _ <- ask
  359   -- the absolute level is strictly-negative
  360   let absoluteLevel = ixToLevel current ix
  361   case M.lookup absoluteLevel cache of
  362     Nothing -> do
  363       u <- freshUnique
  364       -- the cache contains only strictly-negative levels
  365       put (M.insert absoluteLevel u cache)
  366       pure u
  367     Just u -> pure u
  368 
  369 -- Compute the absolute 'Level' of a relative 'Index' relative to the current 'Level'.
  370 -- The index `ixAST` may be malformed or point to a free variable because it comes straight
  371 -- from the AST; in such a case, this function may return a negative level.
  372 ixToLevel :: Level -> Index -> Level
  373 ixToLevel (Level current) ixAST = Level $ current - fromIntegral ixAST
