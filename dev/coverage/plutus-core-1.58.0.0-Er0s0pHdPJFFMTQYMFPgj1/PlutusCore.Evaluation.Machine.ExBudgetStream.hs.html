   15 {-| A lazy stream of 'ExBudget's. Basically @NonEmpty ExBudget@, except the elements are
   16 stored strictly.
   17 
   18 The semantics of a stream are those of the 'fold' of its elements. I.e. a stream that is a
   19 reordered version of another stream is considered equal to that stream.
   20 
   21 An 'ExBudgetStream' is what one gets by zipping two 'CostStream's (one for CPU, one for memory),
   22 which is why the two data types are so similar. The only reason why we don't express both the
   23 concepts in terms of a single data type is efficiency, in particular unboxing is crucial for
   24 'CostStream' and we don't care about it in 'ExBudgetStream', because we can't get the spender
   25 in the CEK machine to get inlined and so unboxing 'ExBudget' here would only result in boxing it
   26 back once it's about to be spent. -}
   27 data ExBudgetStream
   28   = ExBudgetLast !ExBudget
   29   | ExBudgetCons !ExBudget ExBudgetStream
   30   deriving stock (Show)
   31 
   32 -- See Note [Global local functions].
   33 sumExBudgetStreamGo :: ExBudget -> ExBudgetStream -> ExBudget
   34 sumExBudgetStreamGo !acc (ExBudgetLast budget) = acc <> budget
   35 sumExBudgetStreamGo !acc (ExBudgetCons budget budgets) = sumExBudgetStreamGo (acc <> budget) budgets
   36 
   37 -- | Add up all the budgets in a 'ExBudgetStream'.
   38 sumExBudgetStream :: ExBudgetStream -> ExBudget
   39 sumExBudgetStream (ExBudgetLast budget0) = budget0
   40 sumExBudgetStream (ExBudgetCons budget0 budgets0) = sumExBudgetStreamGo budget0 budgets0
   41 {-# INLINE sumExBudgetStream #-}
   42 
   43 -- | Convert a 'CostStream' to an 'ExBudgetStream' by applying a function to each element.
   44 costToExBudgetStream :: (CostingInteger -> ExBudget) -> CostStream -> ExBudgetStream
   45 costToExBudgetStream f = go
   46   where
   47     go (CostLast cost) = ExBudgetLast (f cost)
   48     go (CostCons cost costs) = ExBudgetCons (f cost) $ go costs
   49 {-# INLINE costToExBudgetStream #-}
   50 
   51 {-| Convert a 'CostingInteger' representing a CPU cost and a 'CostingInteger' representing a memory
   52 cost to an 'ExBudget'. -}
   53 toExBudget :: CostingInteger -> CostingInteger -> ExBudget
   54 toExBudget = coerce ExBudget
   55 {-# INLINE toExBudget #-}
   56 
   57 -- See Note [Global local functions].
   58 zipCostStreamGo :: CostStream -> CostStream -> ExBudgetStream
   59 zipCostStreamGo (CostLast cpu) (CostLast mem) =
   60   ExBudgetLast $ toExBudget cpu mem
   61 zipCostStreamGo (CostLast cpu) (CostCons mem mems) =
   62   ExBudgetCons (toExBudget cpu mem) $ costToExBudgetStream (\mem' -> toExBudget 0 mem') mems
   63 zipCostStreamGo (CostCons cpu cpus) (CostLast mem) =
   64   ExBudgetCons (toExBudget cpu mem) $ costToExBudgetStream (\cpu' -> toExBudget cpu' 0) cpus
   65 zipCostStreamGo (CostCons cpu cpus) (CostCons mem mems) =
   66   ExBudgetCons (toExBudget cpu mem) $ zipCostStreamGo cpus mems
   67 
   68 {-| Zip two 'CostStream' together (one with CPU costs and the other one with memory costs,
   69 respectively) to get an 'ExBudgetStream'. If one is longer than the other, then it's assumed to
   70 contain the required amount of zeros for two streams to have the same length (all those zeros
   71 \"appear\" in the tail of the stream). -}
   72 zipCostStream :: CostStream -> CostStream -> ExBudgetStream
   73 zipCostStream cpus0 mems0 = case (cpus0, mems0) of
   74   -- See Note [Single-element streams].
   75   (CostLast cpu, CostLast mem) -> ExBudgetLast $ toExBudget cpu mem
   76   _ -> zipCostStreamGo cpus0 mems0
   77 {-# INLINE zipCostStream #-}
