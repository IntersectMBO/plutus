   34 -- | Is this pure? Either yes, or maybe not.
   35 data Purity = MaybeImpure | Pure
   36   deriving stock (Eq, Show)
   37 
   38 instance Pretty Purity where
   39   pretty MaybeImpure = "impure?"
   40   pretty Pure = "pure"
   41 
   42 -- | Is this term essentially work-free? Either yes, or maybe not.
   43 data WorkFreedom = MaybeWork | WorkFree
   44   deriving stock (Eq, Show)
   45 
   46 instance Pretty WorkFreedom where
   47   pretty MaybeWork = "maybe work?"
   48   pretty WorkFree = "work-free"

   50 {-| Either the "next" term to be evaluated, along with its 'Purity' and 'WorkFreedom',
   51 or we don't know what comes next. -}
   52 data EvalTerm name uni fun a
   53   = Unknown
   54   | EvalTerm Purity WorkFreedom (Term name uni fun a)
   55 
   56 instance
   57   ( Show name
   58   , Everywhere uni Show
   59   , Show fun
   60   , Show a
   61   , GShow uni
   62   , Closed uni
   63   )
   64   => Show (EvalTerm name uni fun a)
   65   where
   66   show = \case
   67     Unknown -> "<unknown>"
   68     EvalTerm purity work t ->
   69       "EvalTerm " <> show purity <> " " <> show work <> " " <> show t
   70 
   71 instance
   72   PrettyBy config (Term name uni fun a)
   73   => PrettyBy config (EvalTerm name uni fun a)
   74   where
   75   prettyBy _ Unknown = "<unknown>"
   76   prettyBy config (EvalTerm eff work t) =
   77     pretty eff <+> pretty work <> ":" <+> prettyBy config t
   78 
   79 instance Eq (Term name uni fun a) => Eq (EvalTerm name uni fun a) where
   80   Unknown == Unknown = True
   81   (EvalTerm p1 w1 t1) == (EvalTerm p2 w2 t2) = p1 == p2 && w1 == w2 && t1 == t2
   82   _ == _ = False

   86 -- | The order in which terms get evaluated, along with their purities.
   87 newtype EvalOrder name uni fun a = EvalOrder (DList.DList (EvalTerm name uni fun a))
   88   deriving newtype (Semigroup, Monoid)
   89 
   90 {-| Get the evaluation order as a list of 'EvalTerm's. Either terminates in a single
   91 'Unknown', which means that we got to a point where evaluation continues but we don't
   92 know where; or terminates normally, in which case we actually got to the end of the
   93 evaluation order for the term. -}
   94 unEvalOrder :: EvalOrder name uni fun a -> [EvalTerm name uni fun a]
   95 unEvalOrder (EvalOrder ts) =
   96   -- This is where we avoid traversing the whole program beyond the first Unknown,
   97   -- since DList is lazy and we convert to a lazy list and then filter it.
   98   takeWhileInclusive (\case Unknown -> False; _ -> True) (DList.toList ts)
   99   where
  100     takeWhileInclusive :: (a -> Bool) -> [a] -> [a]
  101     takeWhileInclusive p = foldr (\x ys -> if p x then x : ys else [x]) []
  102 
  103 evalThis :: EvalTerm name uni fun a -> EvalOrder name uni fun a
  104 evalThis = EvalOrder . DList.singleton
  105 
  106 instance
  107   PrettyBy config (Term name uni fun a)
  108   => PrettyBy config (EvalOrder name uni fun a)
  109   where
  110   prettyBy config eo = vsep $ fmap (prettyBy config) (unEvalOrder eo)

  112 {-| Given a term, return the order in which it and its sub-terms will be evaluated.
  113 
  114 This aims to be a sound under-approximation: if we don't know, we just say 'Unknown'.
  115 Typically there will be a sequence of terms that we do know, which will terminate
  116 in 'Unknown' once we do something like call a function.
  117 
  118 This makes some assumptions about the evaluator, in particular about the order in
  119 which we evaluate sub-terms, but these match the current evaluator and we are not
  120 planning on changing it. -}
  121 termEvaluationOrder
  122   :: forall name uni fun a
  123    . ToBuiltinMeaning uni fun
  124   => BuiltinSemanticsVariant fun
  125   -> Term name uni fun a
  126   -> EvalOrder name uni fun a
  127 termEvaluationOrder builtinSemanticsVariant = goTerm
  128   where
  129     goTerm :: Term name uni fun a -> EvalOrder name uni fun a
  130     goTerm = \case
  131       (splitAppCtx -> (builtin@(Builtin _ann fun), appCtx)) ->
  132         appCtxEvalOrder appCtx <> go arity appCtx
  133         where
  134           arity = builtinArity @uni @fun (Proxy @uni) builtinSemanticsVariant fun
  135 
  136           appCtxEvalOrder :: AppCtx name uni fun a -> EvalOrder name uni fun a
  137           appCtxEvalOrder = \case
  138             AppCtxEnd -> mempty
  139             AppCtxTerm _ t rest -> goTerm t <> appCtxEvalOrder rest
  140             AppCtxType _ rest -> appCtxEvalOrder rest
  141 
  142           go :: [Param] -> AppCtx name uni fun a -> EvalOrder name uni fun a
  143           go parameters appContext =
  144             case parameters of
  145               -- All builtin parameters have been applied,
  146               -- (such term is considered impure).
  147               [] -> maybeImpureWork
  148               -- A term parameter is waiting to be applied
  149               TermParam : otherParams ->
  150                 case appContext of
  151                   AppCtxEnd ->
  152                     -- Builtin is not fully saturated with term arguments, thus pure.
  153                     pureWorkFree
  154                   AppCtxType _ann _remainingAppCtx ->
  155                     -- Term parameter expected, type argument applied.
  156                     -- Error is impure.
  157                     maybeImpureWork
  158                   AppCtxTerm _ann _argTerm remainingAppCtx ->
  159                     go otherParams remainingAppCtx
  160               -- A type parameter is waiting to be forced
  161               TypeParam : otherParams ->
  162                 case appContext of
  163                   AppCtxEnd ->
  164                     -- Builtin is not fully saturated with type arguments, thus pure.
  165                     pureWorkFree
  166                   AppCtxTerm _ann _term _remainingAppCtx ->
  167                     -- Type parameter expected, term argument applied.
  168                     -- Error is impure.
  169                     maybeImpureWork
  170                   AppCtxType _ann remainingAppCtx ->
  171                     go otherParams remainingAppCtx
  172             where
  173               maybeImpureWork :: EvalOrder name uni fun a
  174               maybeImpureWork = evalThis (EvalTerm MaybeImpure MaybeWork reconstructed)
  175 
  176               pureWorkFree :: EvalOrder name uni fun a
  177               pureWorkFree = evalThis (EvalTerm Pure WorkFree reconstructed)
  178 
  179               reconstructed :: Term name uni fun a
  180               reconstructed = fillAppCtx builtin appCtx
  181       t@(Apply _ fun arg) ->
  182         -- first the function
  183         goTerm fun
  184           -- then the arg
  185           <> goTerm arg
  186           -- then the whole term, which means environment manipulation, so work
  187           <> evalThis (EvalTerm Pure MaybeWork t)
  188           <> case fun of
  189             -- known function body
  190             LamAbs _ _ body -> goTerm body
  191             -- unknown function body
  192             _ -> evalThis Unknown
  193       t@(Force _ dterm) ->
  194         -- first delayed term
  195         goTerm dterm
  196           -- then the whole term, which will mean forcing, so work
  197           <> evalThis (EvalTerm Pure MaybeWork t)
  198           <> case dterm of
  199             -- known delayed term
  200             Delay _ body -> goTerm body
  201             -- unknown delayed term
  202             _ -> evalThis Unknown
  203       t@(Constr _ _ ts) ->
  204         -- first the arguments, in left-to-right order
  205         foldMap goTerm ts
  206           -- then the whole term, which means constructing the value, so work, unless there's no
  207           -- arguments, in which case it's not more work than evaluating, say, a @Delay@ node
  208           <> evalThis (EvalTerm Pure (if null ts then WorkFree else MaybeWork) t)
  209       t@(Case _ scrut _) ->
  210         -- first the scrutinee
  211         goTerm scrut
  212           -- then the whole term, which means finding the case so work
  213           <> evalThis (EvalTerm Pure MaybeWork t)
  214           -- then we go to an unknown scrutinee
  215           <> evalThis Unknown
  216       -- Leaf terms
  217       t@Var {} ->
  218         evalThis (EvalTerm Pure WorkFree t)
  219       t@Error {} ->
  220         -- definitely effectful! but not relevant from a work perspective
  221         evalThis (EvalTerm MaybeImpure WorkFree t)
  222           -- program terminates
  223           <> evalThis Unknown
  224       t@Builtin {} ->
  225         evalThis (EvalTerm Pure WorkFree t)
  226       t@Delay {} ->
  227         evalThis (EvalTerm Pure WorkFree t)
  228       t@LamAbs {} ->
  229         evalThis (EvalTerm Pure WorkFree t)
  230       t@Constant {} ->
  231         evalThis (EvalTerm Pure WorkFree t)

  233 {-| Will evaluating this term have side effects (looping or error)?
  234 This is slightly wider than the definition of a value, as
  235 it includes applications that are known to be pure, as well as
  236 things that can't be returned from the machine (as they'd be ill-scoped). -}
  237 isPure
  238   :: ToBuiltinMeaning uni fun
  239   => BuiltinSemanticsVariant fun
  240   -> Term name uni fun a
  241   -> Bool
  242 isPure builtinSemanticsVariant term =
  243   -- to work out if the term is pure, we see if we can look through
  244   -- the whole evaluation order without hitting something that might be
  245   -- effectful
  246   go (unEvalOrder (termEvaluationOrder builtinSemanticsVariant term))
  247   where
  248     go :: [EvalTerm name uni fun a] -> Bool
  249     go [] = True
  250     go (et : rest) = case et of
  251       -- Might be an effect here!
  252       EvalTerm MaybeImpure _ _ -> False
  253       -- This term is fine, what about the rest?
  254       EvalTerm Pure _ _ -> go rest
  255       -- We don't know what will happen, so be conservative
  256       Unknown -> False
  257 
  258 {-| Is the given term 'work-free'?
  259 
  260 Note: The definition of 'work-free' is a little unclear, but the idea is that
  261 evaluating this term should do very a trivial amount of work. -}
  262 isWorkFree
  263   :: ToBuiltinMeaning uni fun
  264   => BuiltinSemanticsVariant fun
  265   -> Term name uni fun a
  266   -> Bool
  267 isWorkFree builtinSemanticsVariant term =
  268   -- to work out if the term is pure, we see if we can look through
  269   -- the whole evaluation order without hitting something that might be
  270   -- effectful
  271   go (unEvalOrder (termEvaluationOrder builtinSemanticsVariant term))
  272   where
  273     go :: [EvalTerm name uni fun a] -> Bool
  274     go [] = True
  275     go (et : rest) = case et of
  276       -- Might be an effect here!
  277       EvalTerm _ MaybeWork _ -> False
  278       -- This term is fine, what about the rest?
  279       EvalTerm _ WorkFree _ -> go rest
  280       -- We don't know what will happen, so be conservative
  281       Unknown -> False
