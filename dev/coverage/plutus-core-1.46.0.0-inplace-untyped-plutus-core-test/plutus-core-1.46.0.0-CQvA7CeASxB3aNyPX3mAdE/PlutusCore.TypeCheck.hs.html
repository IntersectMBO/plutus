   39 -- | The constraint for built-in types\/functions are kind\/type-checkable.
   40 --
   41 -- We keep this separate from 'MonadKindCheck'\/'MonadTypeCheck', because those mainly constrain the
   42 -- monad and 'Typecheckable' constraints only the builtins. In particular useful when the monad gets
   43 -- instantiated and builtins don't. Another reason is that 'Typecheckable' is not required during
   44 -- type checking, since it's only needed for computing 'BuiltinTypes', which is passed as a regular
   45 -- argument to the worker of the type checker.
   46 type Typecheckable uni fun = (ToKind uni, HasUniApply uni, ToBuiltinMeaning uni fun)
   47 
   48 -- | The default kind checking config.
   49 defKindCheckConfig :: KindCheckConfig
   50 defKindCheckConfig = KindCheckConfig DetectNameMismatches
   51 
   52 -- | Extract the 'TypeScheme' from a 'BuiltinMeaning' and convert it to the
   53 -- corresponding 'Type' for each built-in function.
   54 builtinMeaningsToTypes
   55     :: (MonadKindCheck err term uni fun ann m, Typecheckable uni fun)
   56     => BuiltinSemanticsVariant fun
   57     -> ann
   58     -> m (BuiltinTypes uni fun)
   59 builtinMeaningsToTypes semvar ann =
   60     runQuoteT . fmap BuiltinTypes . sequence . tabulateArray $ \fun -> do
   61         let ty = typeOfBuiltinFunction semvar fun
   62         _ <- inferKind defKindCheckConfig $ ann <$ ty
   63         dupable <$> normalizeType ty
   64 
   65 -- | Get the default type checking config.
   66 getDefTypeCheckConfig
   67     :: (MonadKindCheck err term uni fun ann m, Typecheckable uni fun)
   68     => ann -> m (TypeCheckConfig uni fun)
   69 getDefTypeCheckConfig ann =
   70     TypeCheckConfig defKindCheckConfig <$> builtinMeaningsToTypes def ann
   71 
   72 -- | Infer the kind of a type.
   73 inferKind
   74     :: MonadKindCheck err term uni fun ann m
   75     => KindCheckConfig -> Type TyName uni ann -> m (Kind ())
   76 inferKind config = runTypeCheckM config . inferKindM
   77 
   78 -- | Check a type against a kind.
   79 -- Infers the kind of the type and checks that it's equal to the given kind
   80 -- throwing a 'TypeError' (annotated with the value of the @ann@ argument) otherwise.
   81 checkKind
   82     :: MonadKindCheck err term uni fun ann m
   83     => KindCheckConfig -> ann -> Type TyName uni ann -> Kind () -> m ()
   84 checkKind config ann ty = runTypeCheckM config . checkKindM ann ty
   85 
   86 -- | Infer the type of a term.
   87 inferType
   88     :: MonadTypeCheckPlc err uni fun ann m
   89     => TypeCheckConfig uni fun
   90     -> Term TyName Name uni fun ann
   91     -> m (Normalized (Type TyName uni ()))
   92 inferType config = rename >=> runTypeCheckM config . inferTypeM
   93 
   94 -- | Check a term against a type.
   95 -- Infers the type of the term and checks that it's equal to the given type
   96 -- throwing a 'TypeError' (annotated with the value of the @ann@ argument) otherwise.
   97 checkType
   98     :: MonadTypeCheckPlc err uni fun ann m
   99     => TypeCheckConfig uni fun
  100     -> ann
  101     -> Term TyName Name uni fun ann
  102     -> Normalized (Type TyName uni ())
  103     -> m ()
  104 checkType config ann term ty = do
  105     termRen <- rename term
  106     runTypeCheckM config $ checkTypeM ann termRen ty
  107 
  108 -- | Infer the type of a program.
  109 inferTypeOfProgram
  110     :: MonadTypeCheckPlc err uni fun ann m
  111     => TypeCheckConfig uni fun
  112     -> Program TyName Name uni fun ann
  113     -> m (Normalized (Type TyName uni ()))
  114 inferTypeOfProgram config (Program _ _ term) = inferType config term
  115 
  116 -- | Check a program against a type.
  117 -- Infers the type of the program and checks that it's equal to the given type
  118 -- throwing a 'TypeError' (annotated with the value of the @ann@ argument) otherwise.
  119 checkTypeOfProgram
  120     :: MonadTypeCheckPlc err uni fun ann m
  121     => TypeCheckConfig uni fun
  122     -> ann
  123     -> Program TyName Name uni fun ann
  124     -> Normalized (Type TyName uni ())
  125     -> m ()
  126 checkTypeOfProgram config ann (Program _ _ term) = checkType config ann term
