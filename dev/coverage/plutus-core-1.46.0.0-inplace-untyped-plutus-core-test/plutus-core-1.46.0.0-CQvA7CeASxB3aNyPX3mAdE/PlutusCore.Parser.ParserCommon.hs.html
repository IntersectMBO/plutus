   26 {- Note [Whitespace invariant]
   27 Every top-level 'Parser' must consume all whitespace after the thing that it parses, hence make
   28 sure to enclose every 'Parser' that doesn't consume trailing whitespce (e.g. 'takeWhileP',
   29 'manyTill', 'Lex.decimal' etc) in a call to 'lexeme'.
   30 -}
   31 
   32 newtype ParserState = ParserState {identifiers :: M.Map Text Unique}
   33   deriving stock (Show)

   35 type Parser =
   36   ParsecT ParserError Text (StateT ParserState (ReaderT (Maybe Version) Quote))
   37 
   38 instance (Stream s, MonadQuote m) => MonadQuote (ParsecT e s m)

   40 initial :: ParserState
   41 initial = ParserState M.empty

   43 -- | Get the version of the program being parsed, if we know it.
   44 getVersion :: Parser (Maybe Version)
   45 getVersion = ask
   46 
   47 -- | Set the version of the program being parsed.
   48 withVersion :: Version -> Parser a -> Parser a
   49 withVersion v = local (const $ Just v)
   50 
   51 {- | Run an action conditionally based on a predicate on the version.
   52 If we don't know the version then the predicate is assumed to be
   53 false, i.e. we act if we _know_ the predicate is satisfied.
   54 -}
   55 whenVersion :: (Version -> Bool) -> Parser () -> Parser ()
   56 whenVersion p act = do
   57   mv <- getVersion
   58   case mv of
   59     Nothing -> pure ()
   60     Just v  -> when (p v) act

   62 parse
   63   :: (AsParserErrorBundle e, MonadError e m, MonadQuote m)
   64   => Parser a
   65   -> String
   66   -> Text
   67   -> m a
   68 parse p file str = do
   69   let res = fmap toErrorB (runReaderT (evalStateT (runParserT p file str) initial) Nothing)
   70   throwingEither _ParserErrorBundle =<< liftQuote res
   71 
   72 toErrorB :: Either (ParseErrorBundle Text ParserError) a -> Either ParserErrorBundle a
   73 toErrorB (Left err) = Left $ ParseErrorB err
   74 toErrorB (Right a)  = Right a
   75 
   76 -- | Generic parser function in which the file path is just "test".
   77 parseGen :: (AsParserErrorBundle e, MonadError e m, MonadQuote m) => Parser a -> Text -> m a
   78 parseGen stuff = parse stuff "test"

   80 -- | Space consumer.
   81 whitespace :: Parser ()
   82 whitespace = Lex.space space1 (Lex.skipLineComment "--") (Lex.skipBlockCommentNested "{-" "-}")
   83 
   84 leadingWhitespace :: Parser a -> Parser a
   85 leadingWhitespace = (whitespace *>)
   86 
   87 trailingWhitespace :: Parser a -> Parser a
   88 trailingWhitespace = (<* whitespace)
   89 
   90 {- | Returns a parser for @a@ by calling the supplied function on the starting
   91 and ending positions of @a@.
   92 
   93 The supplied function should usually return a parser that does /not/ consume trailing
   94 whitespaces. Otherwise, the end position will be the first character after the
   95 trailing whitespaces.
   96 -}
   97 withSpan' :: (SrcSpan -> Parser a) -> Parser a
   98 withSpan' f = mdo
   99   start <- getSourcePos
  100   res <- f sp
  101   end <- getSourcePos
  102   let sp = toSrcSpan start end
  103   pure res
  104 
  105 {- | Like `withSpan'`, but the result parser consumes whitespaces.
  106 
  107 @withSpan = (<* whitespace) . withSpan'
  108 -}
  109 withSpan :: (SrcSpan -> Parser a) -> Parser a
  110 withSpan = (<* whitespace) . withSpan'
  111 
  112 lexeme :: Parser a -> Parser a
  113 lexeme = Lex.lexeme whitespace

  115 symbol :: Text -> Parser Text
  116 symbol = Lex.symbol whitespace
  117 
  118 inParens :: Parser a -> Parser a
  119 inParens = between (symbol "(") (char ')')
  120 
  121 inBrackets :: Parser a -> Parser a
  122 inBrackets = between (symbol "[") (char ']')
  123 
  124 inBraces :: Parser a -> Parser a
  125 inBraces = between (symbol "{") (char '}')

  127 toSrcSpan :: SourcePos -> SourcePos -> SrcSpan
  128 toSrcSpan start end =
  129   SrcSpan
  130     { srcSpanFile = sourceName start
  131     , srcSpanSLine = unPos (sourceLine start)
  132     , srcSpanSCol = unPos (sourceColumn start)
  133     , srcSpanELine = unPos (sourceLine end)
  134     , srcSpanECol = unPos (sourceColumn end)
  135     }

  137 version :: Parser Version
  138 version = trailingWhitespace $ do
  139   x <- Lex.decimal
  140   _ <- char '.'
  141   y <- Lex.decimal
  142   _ <- char '.'
  143   Version x y <$> Lex.decimal

  145 -- | Parses a `Name`. Does not consume leading or trailing whitespaces.
  146 name :: Parser Name
  147 name = try $ parseUnquoted <|> parseQuoted
  148   where
  149     parseUnquoted :: Parser Name
  150     parseUnquoted = do
  151       _ <- lookAhead (satisfy isIdentifierStartingChar)
  152       str <- takeWhileP (Just "identifier-unquoted") isIdentifierChar
  153       Name str <$> uniqueSuffix str
  154 
  155     parseQuoted :: Parser Name
  156     parseQuoted = do
  157       _ <- char '`'
  158       _ <- lookAhead (satisfy isQuotedIdentifierChar)
  159       str <- takeWhileP (Just "identifier-quoted") isQuotedIdentifierChar
  160       _ <- char '`'
  161       Name str <$> uniqueSuffix str
  162 
  163     -- Tries to parse a `Unique` value.
  164     -- If it fails then looks up the `Unique` value for the given name.
  165     -- If lookup fails too then generates a fresh `Unique` value.
  166     uniqueSuffix :: Text -> Parser Unique
  167     uniqueSuffix nameStr = try (Unique <$> (char '-' *> Lex.decimal)) <|> uniqueForName nameStr
  168 
  169     -- Return the unique identifier of a name.
  170     -- If it's not in the current parser state, map the name to a fresh id and add it to the state.
  171     uniqueForName :: (MonadState ParserState m, MonadQuote m) => Text -> m Unique
  172     uniqueForName nameStr = do
  173       parserState <- get
  174       case M.lookup nameStr (identifiers parserState) of
  175         Just u -> pure u
  176         Nothing -> do
  177           fresh <- freshUnique
  178           put $ ParserState $ M.insert nameStr fresh $ identifiers parserState
  179           pure fresh
