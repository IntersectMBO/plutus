   57 data CkValue uni fun =
   58     VCon (Some (ValueOf uni))
   59   | VTyAbs TyName (Kind ()) (Term TyName Name uni fun ())
   60   | VLamAbs Name (Type TyName uni ()) (Term TyName Name uni fun ())
   61   | VIWrap (Type TyName uni ()) (Type TyName uni ()) (CkValue uni fun)
   62   | VBuiltin (Term TyName Name uni fun ()) (BuiltinRuntime (CkValue uni fun))
   63   | VConstr (Type TyName uni ()) Word64 [CkValue uni fun]
   64 
   65 deriving stock instance (GShow uni, Everywhere uni Show, Show fun, Closed uni)
   66     => Show (CkValue uni fun)
   67 
   68 ckValueToTerm :: CkValue uni fun -> Term TyName Name uni fun ()
   69 ckValueToTerm = \case
   70     VCon val             -> Constant () val
   71     VTyAbs  tn k body    -> TyAbs  () tn k body
   72     VLamAbs name ty body -> LamAbs () name ty body
   73     VIWrap  ty1 ty2 val  -> IWrap  () ty1 ty2 $ ckValueToTerm val
   74     VBuiltin term _      -> term
   75     VConstr ty i es      -> Constr () ty i (fmap ckValueToTerm es)
   76 
   77 data CkEnv uni fun s = CkEnv

   80     -- as we store logs as "latest go last".

   96 type CkM uni fun s =

  110 -- The 'DList' is just be consistent with the CEK machine (see Note [DList-based emitting]).
  111 emitCkM :: DList Text -> CkM uni fun s ()
  112 emitCkM logs = do
  113     mayLogsRef <- asks ckEnvMayEmitRef
  114     case mayLogsRef of
  115         Nothing      -> pure ()
  116         Just logsRef -> lift . lift $ modifySTRef logsRef (`DList.append` logs)

  126 data Frame uni fun
  127     = FrameAwaitArg (CkValue uni fun)                       -- ^ @[V _]@
  128     | FrameAwaitFunTerm (Term TyName Name uni fun ())       -- ^ @[_ N]@
  129     | FrameAwaitFunValue (CkValue uni fun)                  -- ^ @[_ V]@
  130     | FrameTyInstArg (Type TyName uni ())                   -- ^ @{_ A}@
  131     | FrameUnwrap                                           -- ^ @(unwrap _)@
  132     | FrameIWrap (Type TyName uni ()) (Type TyName uni ())  -- ^ @(iwrap A B _)@
  133     | FrameConstr (Type TyName uni ()) Word64 [Term TyName Name uni fun ()] [CkValue uni fun]
  134     | FrameCase [Term TyName Name uni fun ()]
  135 
  136 deriving stock instance (GShow uni, Closed uni, uni `Everywhere` Show, Show fun) =>
  137     Show (Frame uni fun)

  145 runCkM
  146     :: BuiltinsRuntime fun (CkValue uni fun)
  147     -> Bool
  148     -> (forall s. CkM uni fun s a)
  149     -> (Either (CkEvaluationException uni fun) a, [Text])
  150 runCkM runtime emitting a = runST $ do
  151     mayLogsRef <- if emitting then Just <$> newSTRef DList.empty else pure Nothing
  152     errOrRes <- runExceptT . runReaderT a $ CkEnv runtime mayLogsRef
  153     logs <- case mayLogsRef of
  154         Nothing      -> pure []
  155         Just logsRef -> DList.toList <$> readSTRef logsRef
  156     pure (errOrRes, logs)

  158 -- FIXME: make sure that the specification is up to date and that this matches.
  159 -- Tracked by https://github.com/IntersectMBO/plutus-private/issues/1552.
  160 -- | The computing part of the CK machine. Rules are as follows:
  161 --
  162 -- > s ▷ {M A}      ↦ s , {_ A}        ▷ M
  163 -- > s ▷ [M N]      ↦ s , [_ N]        ▷ M
  164 -- > s ▷ wrap α A M ↦ s , (wrap α S _) ▷ M
  165 -- > s ▷ unwrap M   ↦ s , (unwrap _)   ▷ M
  166 -- > s ▷ abs α K M  ↦ s ◁ abs α K M
  167 -- > s ▷ lam x A M  ↦ s ◁ lam x A M
  168 -- > s ▷ builtin bn ↦ s ◁ builtin (Builtin () bn) (runtimeOf bn)
  169 -- > s ▷ con cn     ↦ s ◁ con cn
  170 -- > s ▻ constr I T0 .. Tn ↦ s , (constr I _ T1 Tn) ▻ T0
  171 -- > s ▻ case S C0 ... Cn ↦ s , (case _ C0 ... Cn) ▻ S
  172 -- > s ▷ error A    ↦ ◆
  173 (|>)
  174     :: Context uni fun -> Term TyName Name uni fun () -> CkM uni fun s (Term TyName Name uni fun ())
  175 stack |> TyInst  _ fun ty        = FrameTyInstArg ty  : stack |> fun
  176 stack |> Apply   _ fun arg       = FrameAwaitFunTerm arg  : stack |> fun
  177 stack |> IWrap   _ pat arg term  = FrameIWrap pat arg : stack |> term
  178 stack |> Unwrap  _ term          = FrameUnwrap        : stack |> term
  179 stack |> TyAbs   _ tn k term     = stack <| VTyAbs tn k term
  180 stack |> LamAbs  _ name ty body  = stack <| VLamAbs name ty body
  181 stack |> Builtin _ bn            = do
  182     runtime <- lookupBuiltin bn . ckEnvRuntime <$> ask
  183     stack <| VBuiltin (Builtin () bn) runtime
  184 stack |> Constant _ val          = stack <| VCon val
  185 stack |> Constr _ ty i es               = case es of
  186     []     -> stack <| VConstr ty i []
  187     t : ts -> FrameConstr ty i ts [] : stack |> t
  188 stack |> Case _ _ arg cs         = FrameCase cs : stack |> arg
  189 _     |> err@Error{}             =
  190     throwingWithCause _OperationalError CkEvaluationFailure $ void err
  191 _     |> var@Var{}               =
  192     throwingWithCause _StructuralError OpenTermEvaluatedMachineError var

  194 -- FIXME: make sure that the specification is up to date and that this matches.
  195 -- Tracked by https://github.com/IntersectMBO/plutus-private/issues/1552.
  196 -- | The returning part of the CK machine. Rules are as follows:
  197 --
  198 -- > s , {_ A}           ◁ abs α K M  ↦ s         ▷ {A/α}M
  199 -- > s , [_ N]           ◁ V          ↦ s , [V _] ▷ N
  200 -- > s , [_ V1 ... Vm]   ◁ (lam x A M) ↦ s , [_ V2 ... Vm] ▷ [V1/x]M
  201 -- > s , [(lam x A M) _] ◁ V          ↦ s         ▷ [V/x]M
  202 -- > s , {_ A}           ◁ F          ↦ s ◁ {F A}  -- Partially instantiated builtin application.
  203 -- > s , [F _]           ◁ V          ↦ s ◁ [F V]  -- Partially saturated builtin application.
  204 -- > s , [F _]           ◁ V          ↦ s ◁ W      -- Fully saturated builtin application, [F V] ~> W.
  205 -- > s , (wrap α S _)    ◁ V          ↦ s ◁ wrap α S V
  206 -- > s , (unwrap _)      ◁ wrap α A V ↦ s ◁ V
  207 -- > s , (constr I V0 ... Vj-1 _ Tj+1 ... Tn) ◅ Vj ↦ s , (constr i V0 ... Vj _ Tj+2... Tn) ▻ Tj+1
  208 -- > s , (case _ C0 ... CN) ◅ (constr i V1 .. Vm) ↦ s , [_ V1 ... Vm] ▻ Ci
  209 (<|)
  210     :: Context uni fun -> CkValue uni fun -> CkM uni fun s (Term TyName Name uni fun ())
  211 []                         <| val     = pure $ ckValueToTerm val
  212 FrameTyInstArg ty  : stack <| fun     = instantiateEvaluate stack ty fun
  213 FrameAwaitFunTerm arg  : stack <| fun = FrameAwaitArg fun : stack |> arg
  214 FrameAwaitArg fun  : stack <| arg     = applyEvaluate stack fun arg
  215 FrameAwaitFunValue arg : stack <| fun = applyEvaluate stack fun arg
  216 FrameIWrap pat arg : stack <| value   = stack <| VIWrap pat arg value
  217 FrameUnwrap        : stack <| wrapped = case wrapped of
  218     VIWrap _ _ term -> stack <| term
  219     _               ->
  220         throwingWithCause _StructuralError NonWrapUnwrappedMachineError $ ckValueToTerm wrapped
  221 FrameConstr ty i todo done : stack <| e =
  222     let done' = e:done
  223     in case todo of
  224         t : ts -> FrameConstr ty i ts done' : stack |> t
  225         []     -> stack <| VConstr ty i (reverse done')
  226 FrameCase cs : stack <| e = case e of
  227     VConstr _ i args -> case cs ^? wix i of
  228         Just t  -> go (reverse args) stack |> t
  229           where
  230             go [] s         = s
  231             go (arg:rest) s = go rest (FrameAwaitFunValue arg : s)
  232         Nothing ->
  233             throwingWithCause _StructuralError (MissingCaseBranchMachineError i) $ ckValueToTerm e
  234     _ -> throwingWithCause _StructuralError NonConstrScrutinizedMachineError $ ckValueToTerm e

  236 -- | Transfers a 'Spine' onto the stack. The first argument will be at the top of the stack.
  237 --
  238 -- >>> import PlutusCore.Default
  239 -- >>> import PlutusCore.Builtin
  240 -- >>> transferSpine (SpineCons (fromValue (1 :: Integer)) (SpineLast (fromValue (2 :: Integer)))) [FrameUnwrap :: Frame DefaultUni DefaultFun]
  241 -- [FrameAwaitFunValue (VCon (Some (ValueOf DefaultUniInteger 1))),FrameAwaitFunValue (VCon (Some (ValueOf DefaultUniInteger 2))),FrameUnwrap]
  242 transferSpine :: Spine (CkValue uni fun) -> Context uni fun -> Context uni fun
  243 transferSpine args ctx = foldr ((:) . FrameAwaitFunValue) ctx args

  245 -- | Evaluate a 'HeadSpine' by pushing the arguments (if any) onto the stack and proceeding with
  246 -- the returning phase of the CK machine, i.e. @<|@.
  247 returnCkHeadSpine
  248     :: Context uni fun
  249     -> HeadSpine (CkValue uni fun)
  250     -> CkM uni fun s (Term TyName Name uni fun ())
  251 returnCkHeadSpine stack (HeadOnly  x)    = stack <| x
  252 returnCkHeadSpine stack (HeadSpine f xs) = transferSpine xs stack <| f

  254 -- | Take a possibly partial builtin application and
  255 --
  256 -- - either create a 'CkValue' by evaluating the application if it's saturated (emitting logs, if
  257 --    any, along the way), potentially failing evaluation
  258 -- - or create a partial builtin application otherwise
  259 --
  260 -- and proceed with the returning phase of the CK machine.
  261 evalBuiltinApp
  262     :: Context uni fun
  263     -> Term TyName Name uni fun ()
  264     -> BuiltinRuntime (CkValue uni fun)
  265     -> CkM uni fun s (Term TyName Name uni fun ())
  266 evalBuiltinApp stack term runtime = case runtime of
  267     BuiltinCostedResult _ getFXs -> case getFXs of
  268         BuiltinSuccess fXs              -> returnCkHeadSpine stack fXs
  269         BuiltinSuccessWithLogs logs fXs -> emitCkM logs *> returnCkHeadSpine stack fXs
  270         BuiltinFailure logs err         -> emitCkM logs *> throwBuiltinErrorWithCause term err
  271     _ -> stack <| VBuiltin term runtime

  273 -- | Instantiate a term with a type and proceed.
  274 -- In case of 'TyAbs' just ignore the type. Otherwise check if the term is builtin application
  275 -- expecting a type argument, in which case either calculate the builtin application or stick a
  276 -- 'TyInst' on top of its 'Term' representation depending on whether the application is saturated or
  277 -- not. In any other case, fail.
  278 instantiateEvaluate
  279     :: Context uni fun
  280     -> Type TyName uni ()
  281     -> CkValue uni fun
  282     -> CkM uni fun s (Term TyName Name uni fun ())
  283 instantiateEvaluate stack ty (VTyAbs tn _k body) =
  284      -- No kind check - too expensive at run time.
  285     stack |> termSubstClosedType tn ty body
  286 instantiateEvaluate stack ty (VBuiltin term runtime) = do
  287     let term' = TyInst () term ty
  288     case runtime of
  289         -- We allow a type argument to appear last in the type of a built-in function,
  290         -- otherwise we could just assemble a 'VBuiltin' without trying to evaluate the
  291         -- application.
  292         BuiltinExpectForce runtime' -> evalBuiltinApp stack term' runtime'
  293         _ -> throwingWithCause _StructuralError BuiltinTermArgumentExpectedMachineError term'
  294 instantiateEvaluate _ _ val =
  295     throwingWithCause _StructuralError NonPolymorphicInstantiationMachineError $ ckValueToTerm val
  296 
  297 -- | Apply a function to an argument and proceed.
  298 -- If the function is a lambda, then perform substitution and proceed.
  299 -- If the function is a builtin application then check that it's expecting a term argument,
  300 -- and either calculate the builtin application or stick a 'Apply' on top of its 'Term'
  301 -- representation depending on whether the application is saturated or not.
  302 applyEvaluate
  303     :: Context uni fun
  304     -> CkValue uni fun
  305     -> CkValue uni fun
  306     -> CkM uni fun s (Term TyName Name uni fun ())
  307 applyEvaluate stack (VLamAbs name _ body) arg =
  308     stack |> termSubstClosedTerm name (ckValueToTerm arg) body
  309 applyEvaluate stack (VBuiltin term runtime) arg = do
  310     let argTerm = ckValueToTerm arg
  311         term' = Apply () term argTerm
  312     case runtime of
  313         -- It's only possible to apply a builtin application if the builtin expects a term
  314         -- argument next.
  315         BuiltinExpectArgument f -> do
  316             evalBuiltinApp stack term' $ f arg
  317         _ ->
  318             throwingWithCause _StructuralError UnexpectedBuiltinTermArgumentMachineError term'
  319 applyEvaluate _ val _ =
  320     throwingWithCause _StructuralError NonFunctionalApplicationMachineError $ ckValueToTerm val

  322 runCk
  323     :: BuiltinsRuntime fun (CkValue uni fun)
  324     -> Bool
  325     -> Term TyName Name uni fun ()
  326     -> (Either (CkEvaluationException uni fun) (Term TyName Name uni fun ()), [Text])
  327 runCk runtime emitting term = runCkM runtime emitting $ [] |> term

  329 -- | Evaluate a term using the CK machine with logging enabled.

  332     -> Term TyName Name uni fun ()
  333     -> (Either (CkEvaluationException uni fun) (Term TyName Name uni fun ()), [Text])
  334 evaluateCk runtime = runCk runtime True

  336 -- | Evaluate a term using the CK machine with logging disabled.
